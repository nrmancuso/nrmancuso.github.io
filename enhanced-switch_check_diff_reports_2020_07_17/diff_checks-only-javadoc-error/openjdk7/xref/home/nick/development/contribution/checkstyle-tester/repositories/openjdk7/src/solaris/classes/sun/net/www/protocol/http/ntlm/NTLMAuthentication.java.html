48  /*
49   * NTLM authentication is nominally based on the framework defined in RFC2617,
50   * but differs from the standard (Basic & Digest) schemes as follows:
51   *
52   * 1. A complete authentication requires three request/response transactions
53   *    as shown below:
54   *            REQ ------------------------------->
55   *            <---- 401 (signalling NTLM) --------
56   *
57   *            REQ (with type1 NTLM msg) --------->
58   *            <---- 401 (with type 2 NTLM msg) ---
59   *
60   *            REQ (with type3 NTLM msg) --------->
61   *            <---- OK ---------------------------
62   *
63   * 2. The scope of the authentication is the TCP connection (which must be kept-alive)
64   *    after the type2 response is received. This means that NTLM does not work end-to-end
65   *    through a proxy, rather between client and proxy, or between client and server (with no proxy)
66   */
67  
68  public class NTLMAuthentication extends AuthenticationInfo {
69      private static final long serialVersionUID = 170L;
70  
71      private static final NTLMAuthenticationCallback NTLMAuthCallback =
72          NTLMAuthenticationCallback.getNTLMAuthenticationCallback();
73  
74      private String hostname;
75      private static String defaultDomain; /* Domain to use if not specified by user */
76  
77      static {
78          defaultDomain = java.security.AccessController.doPrivileged(
79              new sun.security.action.GetPropertyAction("http.auth.ntlm.domain",
80                                                        "domain"));
81      };
82  
83      public static boolean supportsTransparentAuth () {
84          return false;
85      }
86  
87      /**
88       * Returns true if the given site is trusted, i.e. we can try
89       * transparent Authentication.
90       */
91      public static boolean isTrustedSite(URL url) {
92          return NTLMAuthCallback.isTrustedSite(url);
93      }
94  
95      private void init0() {
96  
97          hostname = java.security.AccessController.doPrivileged(
98              new java.security.PrivilegedAction<String>() {
99              public String run() {
100                 String localhost;
101                 try {
102                     localhost = InetAddress.getLocalHost().getHostName().toUpperCase();
103                 } catch (UnknownHostException e) {
104                      localhost = "localhost";
105                 }
106                 return localhost;
107             }
108         });
109         int x = hostname.indexOf ('.');
110         if (x != -1) {
111             hostname = hostname.substring (0, x);
112         }
113     };
114 
115     PasswordAuthentication pw;
116 
117     Client client;
118     /**
119      * Create a NTLMAuthentication:
120      * Username may be specified as domain<BACKSLASH>username in the application Authenticator.
121      * If this notation is not used, then the domain will be taken
122      * from a system property: "http.auth.ntlm.domain".
123      */
124     public NTLMAuthentication(boolean isProxy, URL url, PasswordAuthentication pw) {
125         super(isProxy ? PROXY_AUTHENTICATION : SERVER_AUTHENTICATION,
126                 AuthScheme.NTLM,
127                 url,
128                 "");
129         init (pw);
130     }
131 
132     private void init (PasswordAuthentication pw) {
133         String username;
134         String ntdomain;
135         char[] password;
136         this.pw = pw;
137         String s = pw.getUserName();
138         int i = s.indexOf ('\\');
139         if (i == -1) {
140             username = s;
141             ntdomain = defaultDomain;
142         } else {
143             ntdomain = s.substring (0, i).toUpperCase();
144             username = s.substring (i+1);
145         }
146         password = pw.getPassword();
147         init0();
148         try {
149             client = new Client(System.getProperty("ntlm.version"), hostname,
150                     username, ntdomain, password);
151         } catch (NTLMException ne) {
152             try {
153                 client = new Client(null, hostname, username, ntdomain, password);
154             } catch (NTLMException ne2) {
155                 // Will never happen
156                 throw new AssertionError("Really?");
157             }
158         }
159     }
160 
161    /**
162     * Constructor used for proxy entries
163     */
164     public NTLMAuthentication(boolean isProxy, String host, int port,
165                                 PasswordAuthentication pw) {
166         super(isProxy ? PROXY_AUTHENTICATION : SERVER_AUTHENTICATION,
167                 AuthScheme.NTLM,
168                 host,
169                 port,
170                 "");
171         init (pw);
172     }
173 
174     /**
175      * @return true if this authentication supports preemptive authorization
176      */
177     @Override
178     public boolean supportsPreemptiveAuthorization() {
179         return false;
180     }
181 
182     /**
183      * Not supported. Must use the setHeaders() method
184      */
185     @Override
186     public String getHeaderValue(URL url, String method) {
187         throw new RuntimeException ("getHeaderValue not supported");
188     }
189 
190     /**
191      * Check if the header indicates that the current auth. parameters are stale.
192      * If so, then replace the relevant field with the new value
193      * and return true. Otherwise return false.
194      * returning true means the request can be retried with the same userid/password
195      * returning false means we have to go back to the user to ask for a new
196      * username password.
197      */
198     @Override
199     public boolean isAuthorizationStale (String header) {
200         return false; /* should not be called for ntlm */
201     }
202 
203     /**
204      * Set header(s) on the given connection.
205      * @param conn The connection to apply the header(s) to
206      * @param p A source of header values for this connection, not used because
207      *          HeaderParser converts the fields to lower case, use raw instead
208      * @param raw The raw header field.
209      * @return true if all goes well, false if no headers were set.
210      */
211     @Override
212     public synchronized boolean setHeaders(HttpURLConnection conn, HeaderParser p, String raw) {
213 
214         try {
215             String response;
216             if (raw.length() < 6) { /* NTLM<sp> */
217                 response = buildType1Msg ();
218             } else {
219                 String msg = raw.substring (5); /* skip NTLM<sp> */
220                 response = buildType3Msg (msg);
221             }
222             conn.setAuthenticationProperty(getHeaderName(), response);
223             return true;
224         } catch (IOException e) {
225             return false;
226         } catch (GeneralSecurityException e) {
227             return false;
228         }
229     }
230 
231     private String buildType1Msg () {
232         byte[] msg = client.type1();
233         String result = "NTLM " + (new B64Encoder()).encode (msg);
234         return result;
235     }
236 
237     private String buildType3Msg (String challenge) throws GeneralSecurityException,
238                                                            IOException  {
239         /* First decode the type2 message to get the server nonce */
240         /* nonce is located at type2[24] for 8 bytes */
241 
242         byte[] type2 = (new sun.misc.BASE64Decoder()).decodeBuffer (challenge);
243         byte[] nonce = new byte[8];
244         new java.util.Random().nextBytes(nonce);
245         byte[] msg = client.type3(type2, nonce);
246         String result = "NTLM " + (new B64Encoder()).encode (msg);
247         return result;
248     }
249 }
250 
251 class B64Encoder extends sun.misc.BASE64Encoder {
252     /* to force it to to the entire encoding in one line */
253     protected int bytesPerLine () {
254         return 1024;
255     }
