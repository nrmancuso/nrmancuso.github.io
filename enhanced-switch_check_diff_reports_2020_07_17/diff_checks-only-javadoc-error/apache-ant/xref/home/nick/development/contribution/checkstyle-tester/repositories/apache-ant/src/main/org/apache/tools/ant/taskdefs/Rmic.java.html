39  /**
40   * <p>Runs the rmic compiler against classes.</p>
41   * 
42   * <p>Rmic can be run on a single class (as specified with the classname
43   * attribute) or a number of classes at once (all classes below base that
44   * are neither _Stub nor _Skel classes).  If you want to rmic a single
45   * class and this class is a class nested into another class, you have to
46   * specify the classname in the form <code>Outer$$Inner</code> instead of
47   * <code>Outer.Inner</code>.</p>
48   * 
49   * <p>It is possible to refine the set of files that are being rmiced. This can
50   * be done with the <i>includes</i>, <i>includesfile</i>, <i>excludes</i>,
51   * <i>excludesfile</i> and <i>defaultexcludes</i>
52   * attributes. With the <i>includes</i> or <i>includesfile</i> attribute you
53   * specify the files you want to have included by using patterns. The
54   * <i>exclude</i> or <i>excludesfile</i> attribute is used to specify
55   * the files you want to have excluded. This is also done with patterns. And
56   * finally with the <i>defaultexcludes</i> attribute, you can specify whether
57   * you want to use default exclusions or not. See the section on
58   * directory based tasks, on how the
59   * inclusion/exclusion of files works, and how to write patterns.</p>
60   * 
61   * <p>This task forms an implicit FileSet and
62   * supports all attributes of <code>&lt;fileset&gt;</code>
63   * (<code>dir</code> becomes <code>base</code>) as well as the nested
64   * <code>&lt;include&gt;</code>, <code>&lt;exclude&gt;</code> and
65   * <code>&lt;patternset&gt;</code> elements.</p>
66   * 
67   * <p>It is possible to use different compilers. This can be selected
68   * with the &quot;build.rmic&quot; property or the <code>compiler</code>
69   * attribute. <a name="compilervalues">There are three choices</a>:</p>
70   * 
71   * <ul>
72   *   <li>sun (the standard compiler of the JDK)</li>
73   *   <li>kaffe (the standard compiler of
74   *       {@link <a href="http://www.kaffe.org">Kaffe</a>})</li>
75   *   <li>weblogic</li>
76   * </ul>
77   *
78   * <p> The <a href="http://dione.zcu.cz/~toman40/miniRMI/">miniRMI</a>
79   * project contains a compiler implementation for this task as well,
80   * please consult miniRMI's documentation to learn how to use it.</p>
81   *
82   * @since Ant 1.1
83   *
84   * @ant.task category="java"
85   */

583     /**
584      * execute by creating an instance of an implementation
585      * class and getting to do the work
586      * @throws org.apache.tools.ant.BuildException
587      * if there's a problem with baseDir or RMIC
588      */
589     public void execute() throws BuildException {
590         try {
591             compileList.clear();
592 
593             File outputDir = getOutputDir();
594             if (outputDir == null) {
595                 throw new BuildException(ERROR_BASE_NOT_SET, getLocation());
596             }
597             if (!outputDir.exists()) {
598                 throw new BuildException(ERROR_NO_BASE_EXISTS + outputDir,
599                                          getLocation());
600             }
601             if (!outputDir.isDirectory()) {
602                 throw new BuildException(ERROR_NOT_A_DIR + outputDir, getLocation());
603             }
604             if (verify) {
605                 log("Verify has been turned on.", Project.MSG_VERBOSE);
606             }
607             RmicAdapter adapter =
608                 nestedAdapter != null ? nestedAdapter :
609                 RmicAdapterFactory.getRmic(getCompiler(), this,
610                                            createCompilerClasspath());
611 
612             // now we need to populate the compiler adapter
613             adapter.setRmic(this);
614 
615             Path classpath = adapter.getClasspath();
616             loader = getProject().createClassLoader(classpath);
617 
618             // scan base dirs to build up compile lists only if a
619             // specific classname is not given
620             if (classname == null) {
621                 DirectoryScanner ds = this.getDirectoryScanner(baseDir);
622                 String[] files = ds.getIncludedFiles();
623                 scanDir(baseDir, files, adapter.getMapper());
624             } else {
625                 // otherwise perform a timestamp comparison - at least
626                 String path = classname.replace('.', File.separatorChar)
627                     + ".class";
628                 File f = new File(baseDir, path);
629                 if (f.isFile()) {
630                     scanDir(baseDir, new String[] {path}, adapter.getMapper());
631                 } else {
632                     // Does not exist, so checking whether it is up to
633                     // date makes no sense.  Compilation will fail
634                     // later anyway, but tests expect a certain
635                     // output.
636                     compileList.add(classname);
637                 }
638             }
639             int fileCount = compileList.size();
640             if (fileCount > 0) {
641                 log("RMI Compiling " + fileCount + " class"
642                     + (fileCount > 1 ? "es" : "") + " to "
643                     + outputDir, Project.MSG_INFO);
644 
645                 if (listFiles) {
646                     for (int i = 0; i < fileCount; i++) {
647                         log(compileList.get(i).toString());
648                     }
649                 }
650 
651                 // finally, lets execute the compiler!!
652                 if (!adapter.execute()) {
653                     throw new BuildException(ERROR_RMIC_FAILED, getLocation());
654                 }
655             }
656             /*
657              * Move the generated source file to the base directory.  If
658              * base directory and sourcebase are the same, the generated
659              * sources are already in place.
660              */
661             if (null != sourceBase && !outputDir.equals(sourceBase)
662                 && fileCount > 0) {
663                 if (idl) {
664                     log("Cannot determine sourcefiles in idl mode, ",
665                         Project.MSG_WARN);
666                     log("sourcebase attribute will be ignored.",
667                         Project.MSG_WARN);
668                 } else {
669                     for (int j = 0; j < fileCount; j++) {
670                         moveGeneratedFile(outputDir, sourceBase,
671                                           (String) compileList.elementAt(j),
672                                           adapter);
673                     }
674                 }
675             }
676         } finally {
677             cleanup();
678         }
679     }

693     /**
694      * Move the generated source file(s) to the base directory
695      *
696      * @throws org.apache.tools.ant.BuildException When error
697      * copying/removing files.
698      */
699     private void moveGeneratedFile(File baseDir, File sourceBaseFile, String classname,
700                                    RmicAdapter adapter) throws BuildException {
701         String classFileName = classname.replace('.', File.separatorChar)
702             + ".class";
703         String[] generatedFiles = adapter.getMapper().mapFileName(classFileName);
704 
705         for (int i = 0; i < generatedFiles.length; i++) {
706             final String generatedFile = generatedFiles[i];
707             if (!generatedFile.endsWith(".class")) {
708                 // don't know how to handle that - a IDL file doesn't
709                 // have a corresponding Java source for example.
710                 continue;
711             }
712             String sourceFileName = StringUtils.removeSuffix(generatedFile,
713                                                              ".class")
714                 + ".java";
715 
716             File oldFile = new File(baseDir, sourceFileName);
717             if (!oldFile.exists()) {
718                 // no source file generated, nothing to move
719                 continue;
720             }
721 
722             File newFile = new File(sourceBaseFile, sourceFileName);
723             try {
724                 if (filtering) {
725                     FILE_UTILS.copyFile(oldFile, newFile,
726                                         new FilterSetCollection(getProject()
727                                                                 .getGlobalFilterSet()));
728                 } else {
729                     FILE_UTILS.copyFile(oldFile, newFile);
730                 }
731                 oldFile.delete();
732             } catch (IOException ioe) {
733                 String msg = "Failed to copy " + oldFile + " to " + newFile
734                     + " due to "
735                     + ioe.getMessage();
736                 throw new BuildException(msg, ioe, getLocation());
737             }
738         }
739     }

741     /**
742      * Scans the directory looking for class files to be compiled.
743      * The result is returned in the class variable compileList.
744      * @param baseDir the base direction
745      * @param files   the list of files to scan
746      * @param mapper  the mapper of files to target files
747      */
748     protected void scanDir(File baseDir, String[] files, FileNameMapper mapper) {
749         String[] newFiles = files;
750         if (idl) {
751             log("will leave uptodate test to rmic implementation in idl mode.",
752                 Project.MSG_VERBOSE);
753         } else if (iiop && iiopOpts != null && iiopOpts.indexOf("-always") > -1) {
754             log("no uptodate test as -always option has been specified",
755                 Project.MSG_VERBOSE);
756         } else {
757             SourceFileScanner sfs = new SourceFileScanner(this);
758             newFiles = sfs.restrict(files, baseDir, getOutputDir(), mapper);
759         }
760         for (int i = 0; i < newFiles.length; i++) {
761             String name = newFiles[i].replace(File.separatorChar, '.');
762             name = name.substring(0, name.lastIndexOf(".class"));
763             compileList.addElement(name);
764         }
765     }

768      * Load named class and test whether it can be rmic'ed
769      * @param classname the name of the class to be tested
770      * @return true if the class can be rmic'ed
771      */
772     public boolean isValidRmiRemote(String classname) {
773         try {
774             Class testClass = loader.loadClass(classname);
775             // One cannot RMIC an interface for "classic" RMI (JRMP)
776             if (testClass.isInterface() && !iiop && !idl) {
777                 return false;
778             }
779             return isValidRmiRemote(testClass);
780         } catch (ClassNotFoundException e) {
781             log(ERROR_UNABLE_TO_VERIFY_CLASS + classname + ERROR_NOT_FOUND,
782                 Project.MSG_WARN);
783         } catch (NoClassDefFoundError e) {
784             log(ERROR_UNABLE_TO_VERIFY_CLASS + classname + ERROR_NOT_DEFINED,
785                 Project.MSG_WARN);
786         } catch (Throwable t) {
787             log(ERROR_UNABLE_TO_VERIFY_CLASS + classname
788                 + ERROR_LOADING_CAUSED_EXCEPTION + t.getMessage(),
789                 Project.MSG_WARN);
790         }
791         // we only get here if an exception has been thrown
792         return false;
793     }

795     /**
796      * Returns the topmost interface that extends Remote for a given
797      * class - if one exists.
798      * @param testClass the class to be tested
799      * @return the topmost interface that extends Remote, or null if there
800      *         is none.
801      */
802     public Class getRemoteInterface(Class testClass) {
803         if (Remote.class.isAssignableFrom(testClass)) {
804             Class [] interfaces = testClass.getInterfaces();
805             if (interfaces != null) {
806                 for (int i = 0; i < interfaces.length; i++) {
807                     if (Remote.class.isAssignableFrom(interfaces[i])) {
808                         return interfaces[i];
809                     }
810                 }
811             }
812         }
813         return null;
814     }

816     /**
817      * Check to see if the class or (super)interfaces implement
818      * java.rmi.Remote.
819      */
820     private boolean isValidRmiRemote (Class testClass) {
821         return getRemoteInterface(testClass) != null;
822     }
