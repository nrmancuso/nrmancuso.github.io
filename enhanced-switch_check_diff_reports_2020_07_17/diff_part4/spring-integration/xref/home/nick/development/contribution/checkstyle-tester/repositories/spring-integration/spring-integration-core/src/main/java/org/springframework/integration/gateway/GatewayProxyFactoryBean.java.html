439 	@Override
440 	protected void onInit() {
441 		synchronized (this.initializationMonitor) {
442 			if (this.initialized) {
443 				return;
444 			}
445 			BeanFactory beanFactory = getBeanFactory();
446 			if (this.channelResolver == null && beanFactory != null) {
447 				this.channelResolver = ChannelResolverUtils.getChannelResolver(beanFactory);
448 			}
449 
450 			populateMethodInvocationGateways();
451 
452 			ProxyFactory gatewayProxyFactory =
453 					new ProxyFactory(this.serviceInterface, this);
454 			gatewayProxyFactory.addAdvice(new DefaultMethodInvokingMethodInterceptor());
455 			this.serviceProxy = gatewayProxyFactory.getProxy(this.beanClassLoader);
456 			if (this.asyncExecutor != null) {
457 				Callable<String> task = () -> null;
458 				Future<String> submitType = this.asyncExecutor.submit(task);
459 				this.asyncSubmitType = submitType.getClass();
460 				if (this.asyncExecutor instanceof AsyncListenableTaskExecutor) {
461 					submitType = ((AsyncListenableTaskExecutor) this.asyncExecutor).submitListenable(task);
462 					this.asyncSubmitListenableType = submitType.getClass();
463 				}
464 			}
465 			this.evaluationContext = ExpressionUtils.createStandardEvaluationContext(beanFactory);
466 			this.initialized = true;
467 		}
468 	}

470 	private void populateMethodInvocationGateways() {
471 		Method[] methods = ReflectionUtils.getUniqueDeclaredMethods(this.serviceInterface);
472 		for (Method method : methods) {
473 			if (Modifier.isAbstract(method.getModifiers())
474 					|| method.getAnnotation(Gateway.class) != null
475 					|| (method.isDefault() && this.proxyDefaultMethods)) {
476 
477 				MethodInvocationGateway gateway = createGatewayForMethod(method);
478 				this.gatewayMap.put(method, gateway);
479 			}
480 		}
481 	}

497 	@Override
498 	@Nullable
499 	public Object invoke(final MethodInvocation invocation) throws Throwable { // NOSONAR
500 		Class<?> returnType = invocation.getMethod().getReturnType();
501 		MethodInvocationGateway gateway = this.gatewayMap.get(invocation.getMethod());
502 		if (gateway != null) {
503 			returnType = gateway.returnType;
504 		}
505 		if (this.asyncExecutor != null && !Object.class.equals(returnType)) {
506 			Invoker invoker = new Invoker(invocation);
507 			if (returnType.isAssignableFrom(this.asyncSubmitType)) {
508 				return this.asyncExecutor.submit(invoker::get);
509 			}
510 			else if (returnType.isAssignableFrom(this.asyncSubmitListenableType)) {
511 				return ((AsyncListenableTaskExecutor) this.asyncExecutor).submitListenable(invoker::get);
512 			}
513 			else if (CompletableFuture.class.equals(returnType)) { // exact
514 				return CompletableFuture.supplyAsync(invoker, this.asyncExecutor);
515 			}
516 			else if (Future.class.isAssignableFrom(returnType) && logger.isDebugEnabled()) {
517 				logger.debug("AsyncTaskExecutor submit*() return types are incompatible with the method return " +
518 						"type; running on calling thread; the downstream flow must return the required Future: "
519 						+ returnType.getSimpleName());
520 			}
521 		}
522 		if (Mono.class.isAssignableFrom(returnType)) {
523 			return doInvoke(invocation, false);
524 		}
525 		else {
526 			return doInvoke(invocation, true);
527 		}
528 	}

530 	@Nullable
531 	protected Object doInvoke(MethodInvocation invocation, boolean runningOnCallerThread) throws Throwable { // NOSONAR
532 		Method method = invocation.getMethod();
533 		if (AopUtils.isToStringMethod(method)) {
534 			return "gateway proxy for service interface [" + this.serviceInterface + "]";
535 		}
536 		try {
537 			return invokeGatewayMethod(invocation, runningOnCallerThread);
538 		}
539 		catch (Throwable e) { //NOSONAR - ok to catch, rethrown below
540 			rethrowExceptionCauseIfPossible(e, invocation.getMethod());
541 			return null; // preceding call should always throw something
542 		}
543 	}
544 
545 	@Nullable
546 	private Object invokeGatewayMethod(MethodInvocation invocation, boolean runningOnCallerThread) {
547 		if (!this.initialized) {
548 			afterPropertiesSet();
549 		}
550 		Method method = invocation.getMethod();
551 		MethodInvocationGateway gateway = this.gatewayMap.get(method);
552 		if (gateway == null) {
553 			try {
554 				return invocation.proceed();
555 			}
556 			catch (Throwable throwable) { // NOSONAR
557 				throw new IllegalStateException(throwable);
558 			}
559 		}
560 		boolean shouldReturnMessage =
561 				Message.class.isAssignableFrom(gateway.returnType) || (!runningOnCallerThread && gateway.expectMessage);
562 		boolean shouldReply = gateway.returnType != void.class;
563 		int paramCount = method.getParameterTypes().length;
564 		Object response;
565 		boolean hasPayloadExpression = findPayloadExpression(method);
566 		if (paramCount == 0 && !hasPayloadExpression) {
567 			response = receive(gateway, method, shouldReply, shouldReturnMessage);
568 		}
569 		else {
570 			response = sendOrSendAndReceive(invocation, gateway, shouldReturnMessage, shouldReply);
571 		}
572 		return response(gateway.returnType, shouldReturnMessage, response);
573 	}
574 
575 	@Nullable
576 	private Object response(Class<?> returnType, boolean shouldReturnMessage, @Nullable Object response) {
577 		if (shouldReturnMessage) {
578 			return response;
579 		}
580 		else {
581 			return response != null ? convert(response, returnType) : null;
582 		}
583 	}

585 	private boolean findPayloadExpression(Method method) {
586 		boolean hasPayloadExpression = method.isAnnotationPresent(Payload.class);
587 		if (!hasPayloadExpression) {
588 			// check for the method metadata next
589 			if (this.methodMetadataMap != null) {
590 				GatewayMethodMetadata metadata = this.methodMetadataMap.get(method.getName());
591 				hasPayloadExpression = (metadata != null) && metadata.getPayloadExpression() != null;
592 			}
593 			else if (this.globalMethodMetadata != null) {
594 				hasPayloadExpression = this.globalMethodMetadata.getPayloadExpression() != null;
595 			}
596 		}
597 		return hasPayloadExpression;
598 	}

600 	@Nullable
601 	private Object receive(MethodInvocationGateway gateway, Method method, boolean shouldReply,
602 			boolean shouldReturnMessage) {
603 
604 		Long receiveTimeout = null;
605 		Expression receiveTimeoutExpression = gateway.getReceiveTimeoutExpression();
606 		if (receiveTimeoutExpression != null) {
607 			receiveTimeout = receiveTimeoutExpression.getValue(this.evaluationContext, Long.class);
608 		}
609 		if (shouldReply) {
610 			if (shouldReturnMessage) {
611 				if (receiveTimeout != null) {
612 					return gateway.receiveMessage(receiveTimeout);
613 				}
614 				else {
615 					return gateway.receiveMessage();
616 				}
617 			}
618 			if (receiveTimeout != null) {
619 				return gateway.receive(receiveTimeout);
620 			}
621 			else {
622 				return gateway.receive();
623 			}
624 		}
625 		throw new IllegalArgumentException("The 'void' method without arguments '" + method + "' is not eligible for" +
626 				" gateway invocation. Consider to use different signature or 'payloadExpression'.");
627 	}

629 	@Nullable
630 	private Object sendOrSendAndReceive(MethodInvocation invocation, MethodInvocationGateway gateway,
631 			boolean shouldReturnMessage, boolean shouldReply) {
632 
633 		Object[] args = invocation.getArguments();
634 		if (shouldReply) {
635 			if (gateway.isMonoReturn) {
636 				Mono<Message<?>> messageMono = gateway.sendAndReceiveMessageReactive(args);
637 				if (!shouldReturnMessage) {
638 					return messageMono.map(Message::getPayload);
639 				}
640 				else {
641 					return messageMono;
642 				}
643 			}
644 			else {
645 				return shouldReturnMessage ? gateway.sendAndReceiveMessage(args) : gateway.sendAndReceive(args);
646 			}
647 		}
648 		else {
649 			gateway.send(args);
650 		}
651 		return null;
652 	}

654 	private void rethrowExceptionCauseIfPossible(Throwable originalException, Method method)
655 			throws Throwable { // NOSONAR
656 		Class<?>[] exceptionTypes = method.getExceptionTypes();
657 		Throwable t = originalException;
658 		while (t != null) {
659 			for (Class<?> exceptionType : exceptionTypes) {
660 				if (exceptionType.isAssignableFrom(t.getClass())) {
661 					throw t;
662 				}
663 			}
664 			if (t instanceof RuntimeException // NOSONAR boolean complexity
665 					&& !(t instanceof MessagingException)
666 					&& !(t instanceof UndeclaredThrowableException)
667 					&& !(t instanceof IllegalStateException && "Unexpected exception thrown".equals(t.getMessage()))) {
668 				throw t;
669 			}
670 			t = t.getCause();
671 		}
672 		throw originalException;
673 	}

675 	private MethodInvocationGateway createGatewayForMethod(Method method) {
676 		Gateway gatewayAnnotation = method.getAnnotation(Gateway.class);
677 		GatewayMethodMetadata methodMetadata = null;
678 		if (!CollectionUtils.isEmpty(this.methodMetadataMap)) {
679 			methodMetadata = this.methodMetadataMap.get(method.getName());
680 		}
681 		Expression payloadExpression =
682 				extractPayloadExpressionFromAnnotationOrMetadata(gatewayAnnotation, methodMetadata);
683 		String requestChannelName = extractRequestChannelFromAnnotationOrMetadata(gatewayAnnotation, methodMetadata);
684 		String replyChannelName = extractReplyChannelFromAnnotationOrMetadata(gatewayAnnotation, methodMetadata);
685 		Expression requestTimeout = extractRequestTimeoutFromAnnotationOrMetadata(gatewayAnnotation, methodMetadata);
686 		Expression replyTimeout = extractReplyTimeoutFromAnnotationOrMetadata(gatewayAnnotation, methodMetadata);
687 
688 		Map<String, Expression> headerExpressions = new HashMap<>();
689 		if (gatewayAnnotation != null) {
690 			annotationHeaders(gatewayAnnotation, headerExpressions);
691 		}
692 		else if (methodMetadata != null && !CollectionUtils.isEmpty(methodMetadata.getHeaderExpressions())) {
693 			headerExpressions.putAll(methodMetadata.getHeaderExpressions());
694 		}
695 
696 		return doCreateMethodInvocationGateway(method, payloadExpression, headerExpressions,
697 				requestChannelName, replyChannelName, requestTimeout, replyTimeout);
698 	}

700 	@Nullable
701 	private Expression extractPayloadExpressionFromAnnotationOrMetadata(@Nullable Gateway gatewayAnnotation,
702 			@Nullable GatewayMethodMetadata methodMetadata) {
703 
704 		Expression payloadExpression =
705 				this.globalMethodMetadata != null
706 						? this.globalMethodMetadata.getPayloadExpression()
707 						: null;
708 
709 		if (gatewayAnnotation != null) {
710 			/*
711 			 * INT-2636 Unspecified annotation attributes should not
712 			 * override the default values supplied by explicit configuration.
713 			 * There is a small risk that someone has used Long.MIN_VALUE explicitly
714 			 * to indicate an indefinite timeout on a gateway method and that will
715 			 * no longer work as expected; they will need to use, say, -1 instead.
716 			 */
717 			if (payloadExpression == null && StringUtils.hasText(gatewayAnnotation.payloadExpression())) {
718 				payloadExpression = EXPRESSION_PARSER.parseExpression(gatewayAnnotation.payloadExpression());
719 			}
720 		}
721 		else if (methodMetadata != null && methodMetadata.getPayloadExpression() != null) {
722 			payloadExpression = methodMetadata.getPayloadExpression();
723 		}
724 
725 		return payloadExpression;
726 	}

755 	private Expression extractRequestTimeoutFromAnnotationOrMetadata(@Nullable Gateway gatewayAnnotation,
756 			@Nullable GatewayMethodMetadata methodMetadata) {
757 
758 		Expression requestTimeout = this.defaultRequestTimeout;
759 
760 		if (gatewayAnnotation != null) {
761 			/*
762 			 * INT-2636 Unspecified annotation attributes should not
763 			 * override the default values supplied by explicit configuration.
764 			 * There is a small risk that someone has used Long.MIN_VALUE explicitly
765 			 * to indicate an indefinite timeout on a gateway method and that will
766 			 * no longer work as expected; they will need to use, say, -1 instead.
767 			 */
768 			if (requestTimeout == null || gatewayAnnotation.requestTimeout() != Long.MIN_VALUE) {
769 				requestTimeout = new ValueExpression<>(gatewayAnnotation.requestTimeout());
770 			}
771 			if (StringUtils.hasText(gatewayAnnotation.requestTimeoutExpression())) {
772 				requestTimeout = ExpressionUtils.longExpression(gatewayAnnotation.requestTimeoutExpression());
773 			}
774 
775 		}
776 		else if (methodMetadata != null) {
777 			String reqTimeout = methodMetadata.getRequestTimeout();
778 			if (StringUtils.hasText(reqTimeout)) {
779 				requestTimeout = ExpressionUtils.longExpression(reqTimeout);
780 			}
781 		}
782 		return requestTimeout;
783 	}

816 	private void annotationHeaders(Gateway gatewayAnnotation, Map<String, Expression> headerExpressions) {
817 		if (!ObjectUtils.isEmpty(gatewayAnnotation.headers())) {
818 			for (GatewayHeader gatewayHeader : gatewayAnnotation.headers()) {
819 				String value = gatewayHeader.value();
820 				String expression = gatewayHeader.expression();
821 				String name = gatewayHeader.name();
822 				boolean hasValue = StringUtils.hasText(value);
823 
824 				if (hasValue == StringUtils.hasText(expression)) {
825 					throw new BeanDefinitionStoreException("exactly one of 'value' or 'expression' " +
826 							"is required on a gateway's header.");
827 				}
828 				headerExpressions.put(name, hasValue
829 						? new LiteralExpression(value)
830 						: EXPRESSION_PARSER.parseExpression(expression));
831 			}
832 		}
833 	}

835 	private MethodInvocationGateway doCreateMethodInvocationGateway(Method method,
836 			@Nullable Expression payloadExpression, Map<String, Expression> headerExpressions,
837 			@Nullable String requestChannelName, @Nullable String replyChannelName,
838 			@Nullable Expression requestTimeout, @Nullable Expression replyTimeout) {
839 
840 		GatewayMethodInboundMessageMapper messageMapper = createGatewayMessageMapper(method, headerExpressions);
841 		MethodInvocationGateway gateway = new MethodInvocationGateway(messageMapper);
842 		gateway.setupReturnType(this.serviceInterface, method);
843 
844 		if (method.getParameterTypes().length == 0 && !findPayloadExpression(method)) {
845 			gateway.setPollable();
846 		}
847 
848 		JavaUtils.INSTANCE
849 				.acceptIfNotNull(payloadExpression, messageMapper::setPayloadExpression)
850 				.acceptIfNotNull(getTaskScheduler(), gateway::setTaskScheduler);
851 
852 		channels(requestChannelName, replyChannelName, gateway);
853 
854 		timeouts(requestTimeout, replyTimeout, messageMapper, gateway);
855 
856 		gateway.setBeanName(getComponentName());
857 		gateway.setBeanFactory(getBeanFactory());
858 		gateway.setShouldTrack(this.shouldTrack);
859 		gateway.afterPropertiesSet();
860 
861 		return gateway;
862 	}

865 			Map<String, Expression> headerExpressions) {
866 
867 		Map<String, Object> headers = headers(method, headerExpressions);
868 
869 		return new GatewayMethodInboundMessageMapper(method,
870 				headerExpressions,
871 				this.globalMethodMetadata != null ? this.globalMethodMetadata.getHeaderExpressions() : null,
872 				headers, this.argsMapper, getMessageBuilderFactory());

875 	@Nullable // NOSONAR - complexitty
876 	private Map<String, Object> headers(Method method, Map<String, Expression> headerExpressions) {
877 		Map<String, Object> headers = null;
878 		// We don't want to eagerly resolve the error channel here
879 		Object errorChannelForVoidReturn = this.errorChannel == null ? this.errorChannelName : this.errorChannel;
880 		boolean isVoidReturnType = method.getReturnType().equals(void.class);
881 		if (errorChannelForVoidReturn != null && isVoidReturnType) {
882 			headers = new HashMap<>();
883 			headers.put(MessageHeaders.ERROR_CHANNEL, errorChannelForVoidReturn);
884 		}
885 
886 		if (isVoidReturnType &&
887 				(!headerExpressions.containsKey(MessageHeaders.REPLY_CHANNEL) ||
888 						(this.globalMethodMetadata != null &&
889 								!this.globalMethodMetadata.getHeaderExpressions()
890 										.containsKey(MessageHeaders.REPLY_CHANNEL)))) {
891 
892 			if (headers == null) {
893 				headers = new HashMap<>();
894 			}
895 			headers.put(MessageHeaders.REPLY_CHANNEL, IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME);
896 		}
897 
898 		Set<String> headerNames = new HashSet<>(headerExpressions.keySet());
899 
900 		if (this.globalMethodMetadata != null) {
901 			headerNames.addAll(this.globalMethodMetadata.getHeaderExpressions().keySet());
902 		}
903 
904 		List<MethodParameter> methodParameters = GatewayMethodInboundMessageMapper.getMethodParameterList(method);
905 
906 		for (MethodParameter methodParameter : methodParameters) {
907 			Header header = methodParameter.getParameterAnnotation(Header.class);
908 			if (header != null) {
909 				String headerName = GatewayMethodInboundMessageMapper.determineHeaderName(header, methodParameter);
910 				headerNames.add(headerName);
911 			}
912 		}
913 
914 		validateHeaders(headerNames);
915 		return headers;
916 	}
917 
918 	private void validateHeaders(Set<String> headerNames) {
919 		for (String header : headerNames) {
920 			if ((MessageHeaders.ID.equals(header) || MessageHeaders.TIMESTAMP.equals(header))) {
921 				throw new BeanInitializationException(
922 						"Messaging Gateway cannot override 'id' and 'timestamp' read-only headers.\n" +
923 								"Wrong headers configuration for " + getComponentName());
924 			}
925 		}
926 	}

938 	private void timeouts(@Nullable Expression requestTimeout, @Nullable Expression replyTimeout,
939 			GatewayMethodInboundMessageMapper messageMapper, MethodInvocationGateway gateway) {
940 		if (requestTimeout == null) {
941 			gateway.setRequestTimeout(-1);
942 		}
943 		else if (requestTimeout instanceof ValueExpression) {
944 			Long timeout = requestTimeout.getValue(Long.class);
945 			if (timeout != null) {
946 				gateway.setRequestTimeout(timeout);
947 			}
948 		}
949 		else {
950 			messageMapper.setSendTimeoutExpression(requestTimeout);
951 		}
952 		if (replyTimeout == null) {
953 			gateway.setReplyTimeout(-1);
954 		}
955 		else if (replyTimeout instanceof ValueExpression) {
956 			Long timeout = replyTimeout.getValue(Long.class);
957 			if (timeout != null) {
958 				gateway.setReplyTimeout(timeout);
959 			}
960 		}
961 		else {
962 			messageMapper.setReplyTimeoutExpression(replyTimeout);
963 		}
964 		if (replyTimeout != null) {
965 			gateway.setReceiveTimeoutExpression(replyTimeout);
966 		}
967 	}

1043 			return this.pollable ? IntegrationPatternType.outbound_channel_adapter
1044 					: this.isVoidReturn
1045 							? IntegrationPatternType.inbound_channel_adapter
1046 							: IntegrationPatternType.inbound_gateway;

1058 		void setupReturnType(Class<?> serviceInterface, Method method) {
1059 			ResolvableType resolvableType;
1060 			if (Function.class.isAssignableFrom(serviceInterface) && "apply".equals(method.getName())) {
1061 				resolvableType = ResolvableType.forClass(Function.class, serviceInterface).getGeneric(1);
1062 			}
1063 			else {
1064 				resolvableType = ResolvableType.forMethodReturnType(method);
1065 			}
1066 			this.returnType = resolvableType.getRawClass();
1067 			if (this.returnType == null) {
1068 				this.returnType = Object.class;
1069 			}
1070 			else {
1071 				this.isMonoReturn = Mono.class.isAssignableFrom(this.returnType);
1072 				this.expectMessage = hasReturnParameterizedWithMessage(resolvableType);
1073 			}
1074 			this.isVoidReturn = isVoidReturnType(resolvableType);
1075 		}

1077 		private boolean hasReturnParameterizedWithMessage(ResolvableType resolvableType) {
1078 			return (Future.class.isAssignableFrom(this.returnType) || Mono.class.isAssignableFrom(this.returnType))
1079 					&& Message.class.isAssignableFrom(resolvableType.getGeneric(0).resolve(Object.class));
1080 		}
1081 
1082 		private boolean isVoidReturnType(ResolvableType resolvableType) {
1083 			Class<?> returnTypeToCheck = this.returnType;
1084 			if (Future.class.isAssignableFrom(this.returnType) || Mono.class.isAssignableFrom(this.returnType)) {
1085 				returnTypeToCheck = resolvableType.getGeneric(0).resolve(Object.class);
1086 			}
1087 			return Void.class.isAssignableFrom(returnTypeToCheck);
1088 		}

1091 			this.pollable = true;
1092 		}
