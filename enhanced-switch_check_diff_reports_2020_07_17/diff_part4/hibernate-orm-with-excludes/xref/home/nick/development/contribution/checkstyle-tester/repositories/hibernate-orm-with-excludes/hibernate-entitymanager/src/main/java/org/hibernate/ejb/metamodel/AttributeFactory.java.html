78  	/**
79  	 * Build a normal attribute.
80  	 *
81  	 * @param ownerType The descriptor of the attribute owner (aka declarer).
82  	 * @param property The Hibernate property descriptor for the attribute
83  	 * @param <X> The type of the owner
84  	 * @param <Y> The attribute type
85  	 * @return The built attribute descriptor or null if the attribute is not part of the JPA 2 model (eg backrefs)
86  	 */
87  	@SuppressWarnings({ "unchecked" })
88  	public <X, Y> AttributeImplementor<X, Y> buildAttribute(AbstractManagedType<X> ownerType, Property property) {
89  		if ( property.isSynthetic() ) {
90  			// hide synthetic/virtual properties (fabricated by Hibernate) from the JPA metamodel.
91              LOG.tracef(
92  					"Skipping synthetic property %s(%s)",
93                      ownerType.getTypeName(),
94  					property.getName()
95  			);
96  			return null;
97  		}
98          LOG.trace("Building attribute [" + ownerType.getTypeName() + "." + property.getName() + "]");
99  		final AttributeContext<X> attributeContext = wrap( ownerType, property );
100 		final AttributeMetadata<X,Y> attributeMetadata =
101 				determineAttributeMetadata( attributeContext, NORMAL_MEMBER_RESOLVER );
102         if (attributeMetadata == null) {
103 			return null;
104 		}
105         if (attributeMetadata.isPlural()) {
106 			return buildPluralAttribute((PluralAttributeMetadata)attributeMetadata);
107 		}
108         final SingularAttributeMetadata<X, Y> singularAttributeMetadata = (SingularAttributeMetadata<X, Y>)attributeMetadata;
109         final Type<Y> metaModelType = getMetaModelType(singularAttributeMetadata.getValueContext());
110         return new SingularAttributeImpl<X, Y>(
111 				attributeMetadata.getName(),
112 				attributeMetadata.getJavaType(),
113 				ownerType,
114 				attributeMetadata.getMember(),
115 				false,
116 				false,
117 				property.isOptional(),
118 				metaModelType,
119 				attributeMetadata.getPersistentAttributeType()
120 		);
121 	}

127 			}

131 			}
132 		};

187 	@SuppressWarnings( "unchecked" )
188 	private <X, Y, E, K> AttributeImplementor<X, Y> buildPluralAttribute(PluralAttributeMetadata<X,Y,E> attributeMetadata) {
189 		final Type<E> elementType = getMetaModelType( attributeMetadata.getElementValueContext() );
190 		if ( java.util.Map.class.isAssignableFrom( attributeMetadata.getJavaType() ) ) {
191 			final Type<K> keyType = getMetaModelType( attributeMetadata.getMapKeyValueContext() );
192 			return PluralAttributeImpl.create( attributeMetadata.getOwnerType(), elementType, attributeMetadata.getJavaType(), keyType )
193 					.member( attributeMetadata.getMember() )
194 					.property( attributeMetadata.getPropertyMapping() )
195 					.persistentAttributeType( attributeMetadata.getPersistentAttributeType() )
196 					.build();
197 		}
198         return PluralAttributeImpl.create(attributeMetadata.getOwnerType(), elementType, attributeMetadata.getJavaType(), null).member(attributeMetadata.getMember()).property(attributeMetadata.getPropertyMapping()).persistentAttributeType(attributeMetadata.getPersistentAttributeType()).build();
199 	}

201 	@SuppressWarnings( "unchecked" )
202 	private <Y> Type<Y> getMetaModelType(ValueContext typeContext) {
203 		switch ( typeContext.getValueClassification() ) {
204 			case BASIC: {
205 				return new BasicTypeImpl<Y>(
206 						typeContext.getBindableType(),
207 						Type.PersistenceType.BASIC
208 				);
209 			}
210 			case ENTITY: {
211 				final org.hibernate.type.EntityType type = (EntityType) typeContext.getValue().getType();
212 				return (Type<Y>) context.locateEntityType( type.getAssociatedEntityName() );
213 			}
214 			case EMBEDDABLE: {
215 				final Component component = (Component) typeContext.getValue();
216 				final EmbeddableTypeImpl<Y> embeddableType = new EmbeddableTypeImpl<Y>(
217 						typeContext.getBindableType(),
218 						typeContext.getAttributeMetadata().getOwnerType(),
219 						(ComponentType) typeContext.getValue().getType()
220 				);
221 				context.registerEmbeddedableType( embeddableType );
222 				final Iterator<Property> subProperties = component.getPropertyIterator();
223 				while ( subProperties.hasNext() ) {
224 					final Property property = subProperties.next();
225 					final AttributeImplementor<Y, Object> attribute = buildAttribute( embeddableType, property );
226 					if ( attribute != null ) {
227 						embeddableType.getBuilder().addAttribute( attribute );
228 					}
229 				}
230 				embeddableType.lock();
231 				return embeddableType;
232 			}
233 			default: {
234 				throw new AssertionFailure( "Unknown type : " + typeContext.getValueClassification() );
235 			}
236 		}
237 	}

239 	private EntityMetamodel getDeclarerEntityMetamodel(AbstractIdentifiableType<?> ownerType) {
240 		final Type.PersistenceType persistenceType = ownerType.getPersistenceType();
241 		if ( persistenceType == Type.PersistenceType.ENTITY) {
242 			return context.getSessionFactory()
243 					.getEntityPersister( ownerType.getTypeName() )
244 					.getEntityMetamodel();
245 		}
246 		else if ( persistenceType == Type.PersistenceType.MAPPED_SUPERCLASS) {
247 			PersistentClass persistentClass =
248 					context.getPersistentClassHostingProperties( (MappedSuperclassTypeImpl<?>) ownerType );
249 			return context.getSessionFactory()
250 				.getEntityPersister( persistentClass.getClassName() )
251 				.getEntityMetamodel();
252 		}
253 		else {
254 			throw new AssertionFailure( "Cannot get the metamodel for PersistenceType: " + persistenceType );
255 		}

428 	/**
429 	 * Here is most of the nuts and bolts of this factory, where we interpret the known JPA metadata
430 	 * against the known Hibernate metadata and build a descriptor for the attribute.
431 	 *
432 	 * @param attributeContext The attribute to be described
433 	 * @param memberResolver Strategy for how to resolve the member defining the attribute.
434 	 * @param <X> The owner type
435 	 * @param <Y> The attribute type
436 	 *
437 	 * @return The attribute description
438 	 */
439 	@SuppressWarnings({ "unchecked" })
440 	private <X,Y> AttributeMetadata<X,Y> determineAttributeMetadata(
441 			AttributeContext<X> attributeContext,
442 			MemberResolver memberResolver) {
443         LOG.trace("Starting attribute metadata determination [" + attributeContext.getPropertyMapping().getName() + "]");
444 		final Member member = memberResolver.resolveMember( attributeContext );
445         LOG.trace("    Determined member [" + member + "]");
446 
447 		final Value value = attributeContext.getPropertyMapping().getValue();
448 		final org.hibernate.type.Type type = value.getType();
449         LOG.trace("    Determined type [name=" + type.getName() + ", class=" + type.getClass().getName() + "]");
450 
451 		if ( type.isAnyType() ) {
452 			// ANY mappings are currently not supported in the JPA metamodel; see HHH-6589
453             if ( context.isIgnoreUnsupported() ) {
454                 return null;
455             }
456 			else {
457                 throw new UnsupportedOperationException( "ANY not supported" );
458             }
459 		}
460 		else if ( type.isAssociationType() ) {
461 			// collection or entity
462 			if ( type.isEntityType() ) {
463 				// entity
464 				return new SingularAttributeMetadataImpl<X,Y>(
465 						attributeContext.getPropertyMapping(),
466 						attributeContext.getOwnerType(),
467 						member,
468 						determineSingularAssociationAttributeType( member )
469 				);
470 			}
471             // collection
472             if (value instanceof Collection) {
473                 final Collection collValue = (Collection)value;
474                 final Value elementValue = collValue.getElement();
475                 final org.hibernate.type.Type elementType = elementValue.getType();
476 
477                 // First, determine the type of the elements and use that to help determine the
478                 // collection type)
479                 final Attribute.PersistentAttributeType elementPersistentAttributeType;
480                 final Attribute.PersistentAttributeType persistentAttributeType;
481                 if (elementType.isAnyType()) {
482                 	if ( context.isIgnoreUnsupported() ) {
483                         return null;
484                     }
485         			else {
486         				throw new UnsupportedOperationException("collection of any not supported yet");
487         			}
488                 }
489                 final boolean isManyToMany = isManyToMany(member);
490                 if (elementValue instanceof Component) {
491                     elementPersistentAttributeType = Attribute.PersistentAttributeType.EMBEDDED;
492                     persistentAttributeType = Attribute.PersistentAttributeType.ELEMENT_COLLECTION;
493                 } else if (elementType.isAssociationType()) {
494                     elementPersistentAttributeType = isManyToMany ? Attribute.PersistentAttributeType.MANY_TO_MANY : Attribute.PersistentAttributeType.ONE_TO_MANY;
495                     persistentAttributeType = elementPersistentAttributeType;
496                 } else {
497                     elementPersistentAttributeType = Attribute.PersistentAttributeType.BASIC;
498                     persistentAttributeType = Attribute.PersistentAttributeType.ELEMENT_COLLECTION;
499                 }
500 
501                 final Attribute.PersistentAttributeType keyPersistentAttributeType;
502 
503                 // Finally, we determine the type of the map key (if needed)
504                 if (value instanceof Map) {
505                     final Value keyValue = ((Map)value).getIndex();
506                     final org.hibernate.type.Type keyType = keyValue.getType();
507 
508                     if (keyType.isAnyType()) {
509                     	if ( context.isIgnoreUnsupported() ) {
510                             return null;
511                         }
512             			else {
513             				throw new UnsupportedOperationException("collection of any not supported yet");
514             			}
515                     }
516                     if (keyValue instanceof Component) keyPersistentAttributeType = Attribute.PersistentAttributeType.EMBEDDED;
517                     else if (keyType.isAssociationType()) keyPersistentAttributeType = Attribute.PersistentAttributeType.MANY_TO_ONE;
518                     else keyPersistentAttributeType = Attribute.PersistentAttributeType.BASIC;
519                 } else keyPersistentAttributeType = null;
520                 return new PluralAttributeMetadataImpl(attributeContext.getPropertyMapping(), attributeContext.getOwnerType(),
521                                                        member, persistentAttributeType, elementPersistentAttributeType,
522                                                        keyPersistentAttributeType);
523             } else if (value instanceof OneToMany) {
524                 // TODO : is this even possible??? Really OneToMany should be describing the
525                 // element value within a o.h.mapping.Collection (see logic branch above)
526                 throw new IllegalArgumentException("HUH???");
527 //					final boolean isManyToMany = isManyToMany( member );
528 //					//one to many with FK => entity
529 //					return new PluralAttributeMetadataImpl(
530 //							attributeContext.getPropertyMapping(),
531 //							attributeContext.getOwnerType(),
532 //							member,
533 //							isManyToMany
534 //									? Attribute.PersistentAttributeType.MANY_TO_MANY
535 //									: Attribute.PersistentAttributeType.ONE_TO_MANY
536 //							value,
537 //							AttributeContext.TypeStatus.ENTITY,
538 //							Attribute.PersistentAttributeType.ONE_TO_MANY,
539 //							null, null, null
540 //					);
541 			}
542 		}
543 		else if ( attributeContext.getPropertyMapping().isComposite() ) {
544 			// component
545 			return new SingularAttributeMetadataImpl<X,Y>(
546 					attributeContext.getPropertyMapping(),
547 					attributeContext.getOwnerType(),
548 					member,
549 					Attribute.PersistentAttributeType.EMBEDDED
550 			);
551 		}
552 		else {
553 			// basic type
554 			return new SingularAttributeMetadataImpl<X,Y>(
555 					attributeContext.getPropertyMapping(),
556 					attributeContext.getOwnerType(),
557 					member,
558 					Attribute.PersistentAttributeType.BASIC
559 			);
560 		}
561 		throw new UnsupportedOperationException( "oops, we are missing something: " + attributeContext.getPropertyMapping() );
562 	}

564 	public static Attribute.PersistentAttributeType determineSingularAssociationAttributeType(Member member) {
565 		if ( Field.class.isInstance( member ) ) {
566 			return ( (Field) member ).getAnnotation( OneToOne.class ) != null
567 					? Attribute.PersistentAttributeType.ONE_TO_ONE
568 					: Attribute.PersistentAttributeType.MANY_TO_ONE;
569 		}
570         else if (MapMember.class.isInstance( member ))  {
571             return  Attribute.PersistentAttributeType.MANY_TO_ONE; // curious to see how this works for non-annotated methods
572         }
573 		else {
574 			return ( (Method) member ).getAnnotation( OneToOne.class ) != null
575 					? Attribute.PersistentAttributeType.ONE_TO_ONE
576 					: Attribute.PersistentAttributeType.MANY_TO_ONE;
577 		}

580 	private abstract class BaseAttributeMetadata<X,Y> implements AttributeMetadata<X,Y> {
581 		private final Property propertyMapping;
582 		private final AbstractManagedType<X> ownerType;
583 		private final Member member;
584 		private final Class<Y> javaType;
585 		private final Attribute.PersistentAttributeType persistentAttributeType;
586 
587 		@SuppressWarnings({ "unchecked" })
588 		protected BaseAttributeMetadata(
589 				Property propertyMapping,
590 				AbstractManagedType<X> ownerType,
591 				Member member,
592 				Attribute.PersistentAttributeType persistentAttributeType) {
593 			this.propertyMapping = propertyMapping;
594 			this.ownerType = ownerType;
595 			this.member = member;
596 			this.persistentAttributeType = persistentAttributeType;
597 
598 			final Class declaredType;
599 
600 			if ( member == null ) {
601 				// assume we have a MAP entity-mode "class"
602 				declaredType = propertyMapping.getType().getReturnedClass();
603 			}
604 			else if ( Field.class.isInstance( member ) ) {
605 				declaredType = ( (Field) member ).getType();
606 			}
607 			else if ( Method.class.isInstance( member ) ) {
608 				declaredType = ( (Method) member ).getReturnType();
609 			}
610             else if ( MapMember.class.isInstance( member ) ) {
611                 declaredType = ((MapMember) member).getType();
612             }
613 			else {
614 				throw new IllegalArgumentException( "Cannot determine java-type from given member [" + member + "]" );
615 			}
616 			this.javaType = accountForPrimitiveTypes( declaredType );
617 		}

621 		}

624 			return member;

839 		private Class<?> getClassFromGenericArgument(java.lang.reflect.Type type) {
840 			if ( type instanceof Class ) {
841 				return (Class) type;
842 			}
843 			else if ( type instanceof TypeVariable ) {
844 				final java.lang.reflect.Type upperBound = ( ( TypeVariable ) type ).getBounds()[0];
845 				return getClassFromGenericArgument( upperBound );
846 			}
847 			else if ( type instanceof ParameterizedType ) {
848 				final java.lang.reflect.Type rawType = ( (ParameterizedType) type ).getRawType();
849 				return getClassFromGenericArgument( rawType );
850 			}
851 			else {
852 				throw new AssertionFailure(
853 						"Fail to process type argument in a generic declaration. Member : " + getMemberDescription()
854 								+ " Type: " + type.getClass()
855 				);
856 			}
857 		}

872     public static ParameterizedType getSignatureType(Member member) {
873         final java.lang.reflect.Type type;
874         if (Field.class.isInstance( member )) {
875             type =  ( ( Field ) member ).getGenericType();
876         }
877         else if ( Method.class.isInstance( member ) ) {
878             type = ( ( Method ) member ).getGenericReturnType();
879         }
880         else {
881             type = ( (MapMember) member ).getType();
882         }
883         //this is a raw type
884         if ( type instanceof Class ) return null;
885         return (ParameterizedType) type;
886     }

888 	public static PluralAttribute.CollectionType determineCollectionType(Class javaType) {
889 		if ( java.util.List.class.isAssignableFrom( javaType ) ) {
890 			return PluralAttribute.CollectionType.LIST;
891 		}
892 		else if ( java.util.Set.class.isAssignableFrom( javaType ) ) {
893 			return PluralAttribute.CollectionType.SET;
894 		}
895 		else if ( java.util.Map.class.isAssignableFrom( javaType ) ) {
896 			return PluralAttribute.CollectionType.MAP;
897 		}
898 		else if ( java.util.Collection.class.isAssignableFrom( javaType ) ) {
899 			return PluralAttribute.CollectionType.COLLECTION;
900 		}
901 		else {
902 			throw new IllegalArgumentException( "Expecting collection type [" + javaType.getName() + "]" );
903 		}

934 		@Override
935 		public Member resolveMember(AttributeContext attributeContext) {
936             final AbstractIdentifiableType identifiableType = (AbstractIdentifiableType) attributeContext.getOwnerType();
937 			final EntityMetamodel entityMetamodel = getDeclarerEntityMetamodel( identifiableType );
938 			if ( ! entityMetamodel.getIdentifierProperty().isVirtual() ) {
939 				throw new IllegalArgumentException( "expecting IdClass mapping" );
940 			}
941 			org.hibernate.type.Type type = entityMetamodel.getIdentifierProperty().getType();
942 			if ( ! EmbeddedComponentType.class.isInstance( type ) ) {
943 				throw new IllegalArgumentException( "expecting IdClass mapping" );
944 			}
945 
946 			final EmbeddedComponentType componentType = (EmbeddedComponentType) type;
947 			final String attributeName = attributeContext.getPropertyMapping().getName();
948 
949 			final Getter getter = componentType.getComponentTuplizer()
950 					.getGetter( componentType.getPropertyIndex( attributeName ) );
951 
952 			return MapAccessor.MapGetter.class.isInstance( getter )
953 					? new MapMember( attributeName, attributeContext.getPropertyMapping().getType().getReturnedClass() )
954 					: getter.getMember();
955 		}

958 	/**
959 	 * A {@link Member} resolver for normal attributes.
960 	 */
961 	private final MemberResolver NORMAL_MEMBER_RESOLVER = new MemberResolver() {
962 		@Override
963 		public Member resolveMember(AttributeContext attributeContext) {
964 			final AbstractManagedType ownerType = attributeContext.getOwnerType();
965 			final Property property = attributeContext.getPropertyMapping();
966 			final Type.PersistenceType persistenceType = ownerType.getPersistenceType();
967 			if ( Type.PersistenceType.EMBEDDABLE == persistenceType ) {
968 				return EMBEDDED_MEMBER_RESOLVER.resolveMember( attributeContext );
969 			}
970 			else if ( Type.PersistenceType.ENTITY == persistenceType
971 					|| Type.PersistenceType.MAPPED_SUPERCLASS == persistenceType ) {
972                 final AbstractIdentifiableType identifiableType = (AbstractIdentifiableType) ownerType;
973 				final EntityMetamodel entityMetamodel = getDeclarerEntityMetamodel( identifiableType );
974 				final String propertyName = property.getName();
975 				final Integer index = entityMetamodel.getPropertyIndexOrNull( propertyName );
976 				if ( index == null ) {
977 					// just like in #determineIdentifierJavaMember , this *should* indicate we have an IdClass mapping
978 					return VIRTUAL_IDENTIFIER_MEMBER_RESOLVER.resolveMember( attributeContext );
979 				}
980 				else {
981 					final Getter getter = entityMetamodel.getTuplizer().getGetter( index );
982 					return MapAccessor.MapGetter.class.isInstance( getter )
983 							? new MapMember( propertyName, property.getType().getReturnedClass() )
984 							: getter.getMember();
985 				}
986 			}
987 			else {
988 				throw new IllegalArgumentException( "Unexpected owner type : " + persistenceType );
989 			}
990 		}
991 	};
