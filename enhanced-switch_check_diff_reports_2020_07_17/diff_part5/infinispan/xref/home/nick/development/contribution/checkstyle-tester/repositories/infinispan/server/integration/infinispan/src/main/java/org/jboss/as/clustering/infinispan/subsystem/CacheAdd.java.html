135     public static synchronized Configuration getDefaultConfiguration(CacheMode cacheMode) {
136         if (defaults == null) {
137             ConfigurationBuilderHolder holder = load(DEFAULTS);
138             Configuration defaultConfig = holder.getDefaultConfigurationBuilder().build();
139             Map<CacheMode, Configuration> map = new EnumMap<CacheMode, Configuration>(CacheMode.class);
140             map.put(defaultConfig.clustering().cacheMode(), defaultConfig);
141             for (ConfigurationBuilder builder : holder.getNamedConfigurationBuilders().values()) {
142                 Configuration config = builder.build();
143                 map.put(config.clustering().cacheMode(), config);
144             }
145             for (CacheMode mode : CacheMode.values()) {
146                 if (!map.containsKey(mode)) {
147                     map.put(mode, new ConfigurationBuilder().read(defaultConfig).clustering().cacheMode(mode).build());
148                 }
149             }
150             defaults = map;
151         }
152         return defaults.get(cacheMode);
153     }

155     private static ConfigurationBuilderHolder load(String resource) {
156         URL url = find(resource, CacheAdd.class.getClassLoader());
157         log.debugf("Loading Infinispan defaults from %s", url.toString());
158         try {
159             InputStream input = url.openStream();
160             ParserRegistry parser = new ParserRegistry(InfinispanExtension.class.getClassLoader());
161             try {
162                 return parser.parse(input);
163             } finally {
164                 try {
165                     input.close();
166                 } catch (IOException e) {
167                     log.warn(e.getLocalizedMessage(), e);
168                 }
169             }
170         } catch (IOException e) {
171             throw new IllegalStateException(String.format("Failed to parse %s", url), e);
172         }
173     }
174 
175     private static URL find(String resource, ClassLoader... loaders) {
176         for (ClassLoader loader : loaders) {
177             if (loader != null) {
178                 URL url = FileLookupFactory.newInstance().lookupFileLocation(resource, loader);
179                 if (url != null) {
180                     return url;
181                 }
182             }
183         }
184         throw new IllegalArgumentException(String.format("Failed to locate %s", resource));
185     }

199     @Override
200     protected void performRuntime(OperationContext context, ModelNode operation, ModelNode model, ServiceVerificationHandler verificationHandler, List<ServiceController<?>> newControllers) throws OperationFailedException {
201 
202         // Because we use child resources in a read-only manner to configure the cache, replace the local model with the full model
203         ModelNode cacheModel = Resource.Tools.readModel(context.readResource(PathAddress.EMPTY_ADDRESS));
204 
205         // we also need the containerModel
206         PathAddress containerAddress = getCacheContainerAddressFromOperation(operation);
207         ModelNode containerModel = context.readResourceFromRoot(containerAddress).getModel();
208 
209         // install the services from a reusable method
210         newControllers.addAll(this.installRuntimeServices(context, operation, containerModel, cacheModel, verificationHandler));
211     }

213     Collection<ServiceController<?>> installRuntimeServices(OperationContext context, ModelNode operation, ModelNode containerModel, ModelNode cacheModel, ServiceVerificationHandler verificationHandler) throws OperationFailedException {
214 
215         // get all required addresses, names and service names
216         PathAddress cacheAddress = getCacheAddressFromOperation(operation);
217         PathAddress containerAddress = getCacheContainerAddressFromOperation(operation);
218         String cacheName = cacheAddress.getLastElement().getValue();
219         String containerName = containerAddress.getLastElement().getValue();
220 
221         // get model attributes
222         ModelNode resolvedValue = null;
223         final String jndiName = ((resolvedValue = CacheResource.JNDI_NAME.resolveModelAttribute(context, cacheModel)).isDefined()) ? resolvedValue.asString() : null;
224         StartMode startMode = StartMode.valueOf(CacheResource.START.resolveModelAttribute(context, cacheModel).asString());
225         if (startMode != StartMode.EAGER) {
226            log.warnf("Ignoring start mode [%s] of cache service [%s], as EAGER is the only supported mode", startMode, cacheName);
227            startMode = StartMode.EAGER;
228         }
229         final ServiceController.Mode initialMode = startMode.getMode();
230 
231         final ModuleIdentifier moduleId = (resolvedValue = CacheResource.CACHE_MODULE.resolveModelAttribute(context, cacheModel)).isDefined() ? ModuleIdentifier.fromString(resolvedValue.asString()) : null;
232 
233         // create a list for dependencies which may need to be added during processing
234         List<Dependency<?>> dependencies = new LinkedList<Dependency<?>>();
235         // Infinispan Configuration to hold the operation data
236         ConfigurationBuilder builder = new ConfigurationBuilder().read(getDefaultConfiguration(this.mode));
237 
238         // process cache configuration ModelNode describing overrides to defaults
239         processModelNode(context, containerName, cacheModel, builder, dependencies);
240 
241         // get container Model to pick up the value of the default cache of the container
242         // AS7-3488 make default-cache no required attribute
243         String defaultCache = CacheContainerResource.DEFAULT_CACHE.resolveModelAttribute(context, containerModel).asString();
244 
245         ServiceTarget target = context.getServiceTarget();
246         Configuration config = builder.build();
247 
248         Collection<ServiceController<?>> controllers = new ArrayList<ServiceController<?>>(3);
249 
250         // install the cache configuration service (configures a cache)
251         controllers.add(this.installCacheConfigurationService(target, containerName, cacheName, defaultCache, moduleId,
252                         builder, config, dependencies, verificationHandler));
253         log.debugf("Cache configuration service for %s installed for container %s", cacheName, containerName);
254 
255         // now install the corresponding cache service (starts a configured cache)
256         controllers.add(this.installCacheService(target, containerName, cacheName, defaultCache, initialMode, config, verificationHandler));
257 
258         // install a name service entry for the cache
259         controllers.add(this.installJndiService(target, containerName, cacheName, InfinispanJndiName.createCacheJndiName(jndiName, containerName, cacheName), verificationHandler));
260         log.debugf("Cache service for cache %s installed for container %s", cacheName, containerName);
261 
262         return controllers;
263     }

265     void removeRuntimeServices(OperationContext context, ModelNode operation, ModelNode model)
266             throws OperationFailedException {
267         // get container and cache addresses
268         final PathAddress cacheAddress = getCacheAddressFromOperation(operation) ;
269         final PathAddress containerAddress = getCacheContainerAddressFromOperation(operation) ;
270         // get container and cache names
271         final String cacheName = cacheAddress.getLastElement().getValue() ;
272         final String containerName = containerAddress.getLastElement().getValue() ;
273 
274         // remove all services started by CacheAdd, in reverse order
275         // remove the binder service
276         ModelNode resolvedValue = null;
277         final String jndiName = (resolvedValue = CacheResource.JNDI_NAME.resolveModelAttribute(context, model)).isDefined() ? resolvedValue.asString() : null;
278         ContextNames.BindInfo bindInfo = ContextNames.bindInfoFor(InfinispanJndiName.createCacheJndiName(jndiName, containerName, cacheName));
279         context.removeService(bindInfo.getBinderServiceName()) ;
280         // remove the CacheService instance
281         context.removeService(CacheService.getServiceName(containerName, cacheName));
282         // remove the cache configuration service
283         context.removeService(CacheConfigurationService.getServiceName(containerName, cacheName));
284 
285         log.debugf("cache %s removed for container %s", cacheName, containerName);
286     }

298     ServiceController<?> installCacheConfigurationService(ServiceTarget target, String containerName, String cacheName, String defaultCache, ModuleIdentifier moduleId,
299             ConfigurationBuilder builder, Configuration config, List<Dependency<?>> dependencies, ServiceVerificationHandler verificationHandler) {
300 
301         final InjectedValue<EmbeddedCacheManager> container = new InjectedValue<EmbeddedCacheManager>();
302         final CacheConfigurationDependencies cacheConfigurationDependencies = new CacheConfigurationDependencies(container);
303         final Service<Configuration> service = new CacheConfigurationService(cacheName, builder, moduleId, cacheConfigurationDependencies);
304         final ServiceBuilder<?> configBuilder = target.addService(CacheConfigurationService.getServiceName(containerName, cacheName), service)
305                 .addDependency(EmbeddedCacheManagerService.getServiceName(containerName), EmbeddedCacheManager.class, container)
306                 .addDependency(Services.JBOSS_SERVICE_MODULE_LOADER, ModuleLoader.class, cacheConfigurationDependencies.getModuleLoaderInjector())
307                 .setInitialMode(ServiceController.Mode.PASSIVE)
308         ;
309         if (config.invocationBatching().enabled()) {
310             cacheConfigurationDependencies.getTransactionManagerInjector().inject(BatchModeTransactionManager.getInstance());
311         } else if (config.transaction().transactionMode() == org.infinispan.transaction.TransactionMode.TRANSACTIONAL) {
312             configBuilder.addDependency(TxnServices.JBOSS_TXN_TRANSACTION_MANAGER, TransactionManager.class, cacheConfigurationDependencies.getTransactionManagerInjector());
313             if (config.transaction().useSynchronization()) {
314                 configBuilder.addDependency(TxnServices.JBOSS_TXN_SYNCHRONIZATION_REGISTRY, TransactionSynchronizationRegistry.class, cacheConfigurationDependencies.getTransactionSynchronizationRegistryInjector());
315             }
316         }
317 
318         // add in any additional dependencies resulting from ModelNode parsing
319         for (Dependency<?> dependency : dependencies) {
320             this.addDependency(configBuilder, dependency);
321         }
322         // add an alias for the default cache
323         if (cacheName.equals(defaultCache)) {
324             configBuilder.addAliases(CacheConfigurationService.getServiceName(containerName, null));
325         }
326         return configBuilder.install();
327     }

329     ServiceController<?> installCacheService(ServiceTarget target, String containerName, String cacheName, String defaultCache, ServiceController.Mode initialMode,
330             Configuration config, ServiceVerificationHandler verificationHandler) {
331 
332         final InjectedValue<EmbeddedCacheManager> container = new InjectedValue<EmbeddedCacheManager>();
333         final CacheDependencies cacheDependencies = new CacheDependencies(container);
334         final Service<Cache<Object, Object>> service = new CacheService<Object, Object>(cacheName, cacheDependencies);
335         final ServiceBuilder<?> builder = target.addService(CacheService.getServiceName(containerName, cacheName), service)
336                 .addDependency(CacheConfigurationService.getServiceName(containerName, cacheName))
337                 .addDependency(EmbeddedCacheManagerService.getServiceName(containerName), EmbeddedCacheManager.class, container)
338                 .setInitialMode(initialMode)
339         ;
340         if (config.transaction().recovery().enabled()) {
341             builder.addDependency(TxnServices.JBOSS_TXN_ARJUNA_RECOVERY_MANAGER, XAResourceRecoveryRegistry.class, cacheDependencies.getRecoveryRegistryInjector());
342         }
343 
344         // add an alias for the default cache
345         if (cacheName.equals(defaultCache)) {
346             builder.addAliases(CacheService.getServiceName(containerName, null));
347         }
348 
349         builder.addDependency(DeployedCacheStoreFactoryService.SERVICE_NAME, DeployedCacheStoreFactory.class, cacheDependencies.getDeployedCacheStoreFactoryInjector());
350 
351         return builder.install();
352     }

354     @SuppressWarnings("rawtypes")
355     ServiceController<?> installJndiService(ServiceTarget target, String containerName, String cacheName, String jndiName, ServiceVerificationHandler verificationHandler) {
356 
357         final ServiceName cacheServiceName = CacheService.getServiceName(containerName, cacheName);
358         final ContextNames.BindInfo bindInfo = ContextNames.bindInfoFor(jndiName);
359         final BinderService binder = new BinderService(bindInfo.getBindName());
360         return target.addService(bindInfo.getBinderServiceName(), binder)
361                 .addAliases(ContextNames.JAVA_CONTEXT_SERVICE_NAME.append(jndiName))
362                 .addDependency(cacheServiceName, Cache.class, new ManagedReferenceInjector<Cache>(binder.getManagedObjectInjector()))
363                 .addDependency(bindInfo.getParentContextServiceName(), ServiceBasedNamingStore.class, binder.getNamingStoreInjector())
364                 .setInitialMode(ServiceController.Mode.PASSIVE)
365                 .install()
366         ;
367     }

369     private <T> void addDependency(ServiceBuilder<?> builder, Dependency<T> dependency) {
370         final ServiceName name = dependency.getName();
371         final Injector<T> injector = dependency.getInjector();
372         if (injector != null) {
373             builder.addDependency(name, dependency.getType(), injector);
374         } else {
375             builder.addDependency(name);
376         }
377     }

399     /**
400      * Create a Configuration object initialized from the operation ModelNode
401      *
402      * @param containerName the name of the cache container
403      * @param cache         ModelNode representing cache configuration
404      * @param builder       ConfigurationBuilder object to add data to
405      * @return initialised Configuration object
406      */
407     void processModelNode(OperationContext context, String containerName, ModelNode cache, ConfigurationBuilder builder, List<Dependency<?>> dependencies)
408             throws OperationFailedException {
409 
410         builder.jmxStatistics().enabled(CacheResource.STATISTICS.resolveModelAttribute(context, cache).asBoolean());
411 
412         final Indexing indexing = Indexing.valueOf(CacheResource.INDEXING.resolveModelAttribute(context, cache).asString());
413         final boolean autoConfig = CacheResource.INDEXING_AUTO_CONFIG.resolveModelAttribute(context, cache).asBoolean();
414         final boolean batching = CacheResource.BATCHING.resolveModelAttribute(context, cache).asBoolean();
415 
416         // set the cache mode (may be modified when setting up clustering attributes)
417         builder.clustering().cacheMode(this.mode);
418         final ModelNode indexingPropertiesModel = CacheResource.INDEXING_PROPERTIES.resolveModelAttribute(context, cache);
419         Properties indexingProperties = new Properties();
420         if (indexing.isEnabled() && indexingPropertiesModel.isDefined()) {
421             for (Property p : indexingPropertiesModel.asPropertyList()) {
422                 String value = p.getValue().asString();
423                 indexingProperties.put(p.getName(), value);
424             }
425         }
426         builder.indexing()
427                 .index(indexing.isEnabled() ? indexing.isLocalOnly() ? Index.LOCAL : Index.ALL : Index.NONE)
428                 .withProperties(indexingProperties)
429                 .autoConfig(autoConfig)
430         ;
431 
432         if (cache.hasDefined(ModelKeys.REMOTE_CACHE)) {
433              builder.sites().backupFor().remoteCache(CacheResource.REMOTE_CACHE.resolveModelAttribute(context, cache).asString());
434         }
435         if (cache.hasDefined(ModelKeys.REMOTE_SITE)) {
436              builder.sites().backupFor().remoteSite(CacheResource.REMOTE_SITE.resolveModelAttribute(context, cache).asString());
437         }
438 
439 
440         // locking is a child resource
441         if (cache.hasDefined(ModelKeys.LOCKING) && cache.get(ModelKeys.LOCKING, ModelKeys.LOCKING_NAME).isDefined()) {
442             ModelNode locking = cache.get(ModelKeys.LOCKING, ModelKeys.LOCKING_NAME);
443 
444             final IsolationLevel isolationLevel = IsolationLevel.READ_COMMITTED;
445             if (locking.hasDefined(ModelKeys.ISOLATION)) {
446                log.warn("Ignoring XML attribute " + ModelKeys.ISOLATION + ", please remove from configuration file");
447             }
448             final boolean striping = LockingResource.STRIPING.resolveModelAttribute(context, locking).asBoolean();
449             final long acquireTimeout = LockingResource.ACQUIRE_TIMEOUT.resolveModelAttribute(context, locking).asLong();
450             final int concurrencyLevel = LockingResource.CONCURRENCY_LEVEL.resolveModelAttribute(context, locking).asInt();
451 
452             builder.locking()
453                     .isolationLevel(isolationLevel)
454                     .useLockStriping(striping)
455                     .lockAcquisitionTimeout(acquireTimeout)
456                     .concurrencyLevel(concurrencyLevel)
457             ;
458         }
459 
460         TransactionMode txMode = TransactionMode.NONE;
461         LockingMode lockingMode = LockingMode.OPTIMISTIC;
462         // locking is a child resource
463         if (cache.hasDefined(ModelKeys.TRANSACTION) && cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME).isDefined()) {
464             ModelNode transaction = cache.get(ModelKeys.TRANSACTION, ModelKeys.TRANSACTION_NAME);
465 
466             final long stopTimeout = TransactionResource.STOP_TIMEOUT.resolveModelAttribute(context, transaction).asLong();
467             txMode = TransactionMode.valueOf(TransactionResource.MODE.resolveModelAttribute(context, transaction).asString());
468             lockingMode = LockingMode.valueOf(TransactionResource.LOCKING.resolveModelAttribute(context, transaction).asString());
469 
470             builder.transaction().cacheStopTimeout(stopTimeout);
471         }
472         builder.transaction()
473                 .transactionMode(txMode.getMode())
474                 .lockingMode(lockingMode)
475                 .useSynchronization(!txMode.isXAEnabled())
476                 .recovery().enabled(txMode.isRecoveryEnabled())
477         ;
478         if (txMode.isRecoveryEnabled()) {
479             builder.transaction().syncCommitPhase(true).syncRollbackPhase(true);
480         }
481 
482         if (batching) {
483             builder.transaction().transactionMode(org.infinispan.transaction.TransactionMode.TRANSACTIONAL).invocationBatching().enable();
484         } else {
485             builder.transaction().invocationBatching().disable();
486         }
487 
488         // eviction is a child resource
489         if (cache.hasDefined(ModelKeys.EVICTION) && cache.get(ModelKeys.EVICTION, ModelKeys.EVICTION_NAME).isDefined()) {
490             ModelNode eviction = cache.get(ModelKeys.EVICTION, ModelKeys.EVICTION_NAME);
491 
492             final EvictionStrategy strategy = EvictionStrategy.valueOf(EvictionResource.EVICTION_STRATEGY.resolveModelAttribute(context, eviction).asString());
493             builder.eviction().strategy(strategy);
494 
495             if (strategy.isEnabled()) {
496                 final long maxEntries = EvictionResource.MAX_ENTRIES.resolveModelAttribute(context, eviction).asLong();
497                 builder.eviction().maxEntries(maxEntries);
498             }
499         }
500         // expiration is a child resource
501         if (cache.hasDefined(ModelKeys.EXPIRATION) && cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME).isDefined()) {
502 
503             ModelNode expiration = cache.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME);
504 
505             final long maxIdle = ExpirationResource.MAX_IDLE.resolveModelAttribute(context, expiration).asLong();
506             final long lifespan = ExpirationResource.LIFESPAN.resolveModelAttribute(context, expiration).asLong();
507             final long interval = ExpirationResource.INTERVAL.resolveModelAttribute(context, expiration).asLong();
508 
509             builder.expiration()
510                     .maxIdle(maxIdle)
511                     .lifespan(lifespan)
512                     .wakeUpInterval(interval)
513             ;
514             // Only enable the reaper thread if we need it
515             if (interval > 0) {
516                 builder.expiration().enableReaper();
517             } else {
518                 builder.expiration().disableReaper();
519             }
520         }
521 
522         // compatibility is a child resource
523         if (cache.hasDefined(ModelKeys.COMPATIBILITY) && cache.get(ModelKeys.COMPATIBILITY, ModelKeys.COMPATIBILITY_NAME).isDefined()) {
524 
525             ModelNode compatibility = cache.get(ModelKeys.COMPATIBILITY, ModelKeys.COMPATIBILITY_NAME);
526 
527             final boolean enabled = CompatibilityResource.ENABLED.resolveModelAttribute(context, compatibility).asBoolean();
528 
529             builder.compatibility().enabled(enabled);
530 
531             if (compatibility.hasDefined(ModelKeys.MARSHALLER)) {
532                 String marshaller = CompatibilityResource.MARSHALLER.resolveModelAttribute(context, compatibility).asString();
533                 try {
534                     Class<? extends Marshaller> marshallerClass = CacheLoader.class.getClassLoader().loadClass(marshaller).asSubclass(Marshaller.class);
535                     builder.compatibility().marshaller(marshallerClass.newInstance());
536                 } catch (Exception e) {
537                     throw InfinispanMessages.MESSAGES.invalidCompatibilityMarshaller(e, marshaller);
538                 }
539             }
540         }
541 
542         if (cache.hasDefined(ModelKeys.SECURITY) && cache.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME).isDefined()) {
543             ModelNode securityModel = cache.get(ModelKeys.SECURITY, ModelKeys.SECURITY_NAME);
544 
545             if (securityModel.hasDefined(ModelKeys.AUTHORIZATION) && securityModel.get(ModelKeys.AUTHORIZATION).hasDefined(ModelKeys.AUTHORIZATION_NAME)) {
546                 ModelNode authzModel = securityModel.get(ModelKeys.AUTHORIZATION, ModelKeys.AUTHORIZATION_NAME);
547 
548                 AuthorizationConfigurationBuilder authzBuilder = builder.security().authorization();
549                 authzBuilder.enabled(CacheAuthorizationResource.ENABLED.resolveModelAttribute(context, authzModel).asBoolean());
550                 for(ModelNode role : CacheAuthorizationResource.ROLES.resolveModelAttribute(context, authzModel).asList()) {
551                     authzBuilder.role(role.asString());
552                 }
553             }
554         }
555 
556         // loaders are a child resource
557         for (String loaderKey : loaderKeys) {
558             handleLoaderProperties(context, cache, loaderKey, containerName, builder, dependencies);
559         }
560 
561         // stores are a child resource
562         for (String storeKey : storeKeys) {
563             handleStoreProperties(context, cache, storeKey, containerName, builder, dependencies);
564         }
565 
566         if (cache.hasDefined(ModelKeys.BACKUP)) {
567             SitesConfigurationBuilder sitesBuilder = builder.sites();
568             for (Property property : cache.get(ModelKeys.BACKUP).asPropertyList()) {
569                 String siteName = property.getName();
570                 ModelNode site = property.getValue();
571                 BackupConfigurationBuilder backupConfigurationBuilder = sitesBuilder.addBackup();
572                 backupConfigurationBuilder.site(siteName)
573                         .backupFailurePolicy(BackupFailurePolicy.valueOf(BackupSiteResource.FAILURE_POLICY.resolveModelAttribute(context, site).asString()))
574                         .strategy(BackupStrategy.valueOf(BackupSiteResource.STRATEGY.resolveModelAttribute(context, site).asString()))
575                         .replicationTimeout(BackupSiteResource.REPLICATION_TIMEOUT.resolveModelAttribute(context, site).asLong());
576                 if (BackupSiteResource.ENABLED.resolveModelAttribute(context, site).asBoolean()) {
577                     sitesBuilder.addInUseBackupSite(siteName);
578                 }
579                 if (site.hasDefined(ModelKeys.TAKE_OFFLINE)) {
580                     ModelNode takeOfflineModel = site.get(ModelKeys.TAKE_OFFLINE);
581                     backupConfigurationBuilder.takeOffline()
582                           .afterFailures(BackupSiteResource.TAKE_OFFLINE_AFTER_FAILURES.resolveModelAttribute(context, takeOfflineModel).asInt())
583                           .minTimeToWait(BackupSiteResource.TAKE_OFFLINE_MIN_WAIT.resolveModelAttribute(context, takeOfflineModel).asLong());
584                 }
585                 if (site.hasDefined(ModelKeys.STATE_TRANSFER) && site.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME).isDefined()) {
586                     ModelNode stateTransferModel = site.get(ModelKeys.STATE_TRANSFER, ModelKeys.STATE_TRANSFER_NAME);
587                     backupConfigurationBuilder.stateTransfer()
588                           .chunkSize(BackupSiteStateTransferResource.STATE_TRANSFER_CHUNK_SIZE.resolveModelAttribute(context, stateTransferModel).asInt())
589                           .timeout(BackupSiteStateTransferResource.STATE_TRANSFER_TIMEOUT.resolveModelAttribute(context, stateTransferModel).asLong())
590                           .maxRetries(BackupSiteStateTransferResource.STATE_TRANSFER_MAX_RETRIES.resolveModelAttribute(context, stateTransferModel).asInt())
591                           .waitTime(BackupSiteStateTransferResource.STATE_TRANSFER_WAIT_TIME.resolveModelAttribute(context, stateTransferModel).asLong());
592                 }
593 
594             }
595         }

643    private StoreConfigurationBuilder<?, ?> buildCacheLoader(PersistenceConfigurationBuilder persistenceBuilder, ModelNode loader, String loaderKey) throws OperationFailedException {
644         if (loaderKey.equals(ModelKeys.CLUSTER_LOADER)) {

677    private StoreConfigurationBuilder createStoreConfigurationFromLocalClassloader(String className, PersistenceConfigurationBuilder persistenceBuilder) throws ClassNotFoundException {
678       Class<?> storeImplClass = CacheLoader.class.getClassLoader().loadClass(className);
679       ConfiguredBy annotation = storeImplClass.getAnnotation(ConfiguredBy.class);
680       Class<? extends StoreConfigurationBuilder> builderClass = null;
681       if (annotation != null) {
682          Class<?> configuredBy = annotation.value();
683          if (configuredBy != null) {
684             BuiltBy builtBy = configuredBy.getAnnotation(BuiltBy.class);
685             builderClass = builtBy.value().asSubclass(StoreConfigurationBuilder.class);
686          }
687       }
688       if (builderClass == null) {
689          return persistenceBuilder.addStore(CustomStoreConfigurationBuilder.class).customStoreClass(storeImplClass);
690       }
691       return persistenceBuilder.addStore(builderClass);
692    }

694    private boolean isPresentInLoadClassLoader(String className) {
695       try {
696          newInstance(className);
697          return true;
698       } catch (InstantiationException e) {
699          throw new IllegalStateException("Could not instantiate class " + className, e);
700       } catch (IllegalAccessException e) {
701          throw new IllegalStateException("Class " + className + " seems not to have a default constructor", e);
702       } catch (ClassNotFoundException e) {
703          return false;
704       }
705    }

710         if (storeKey.equals(ModelKeys.FILE_STORE)) {

712             if (store.hasDefined(ModelKeys.MAX_ENTRIES)) {

714             }
715             final String path = ((resolvedValue = FileStoreResource.PATH.resolveModelAttribute(context, store)).isDefined()) ? resolvedValue.asString() : InfinispanExtension.SUBSYSTEM_NAME + File.separatorChar + containerName;
716             final String relativeTo = ((resolvedValue = FileStoreResource.RELATIVE_TO.resolveModelAttribute(context, store)).isDefined()) ? resolvedValue.asString() : ServerEnvironment.SERVER_DATA_DIR;
717             Injector<PathManager> injector = new SimpleInjector<PathManager>() {
718                 volatile PathManager.Callback.Handle callbackHandle;
719                 @Override
720                 public void inject(PathManager value) {
721                     callbackHandle = value.registerCallback(relativeTo, PathManager.ReloadServerCallback.create(), PathManager.Event.UPDATED, PathManager.Event.REMOVED);
722                     builder.location(value.resolveRelativePathEntry(path, relativeTo));
723                 }
724 
725                 @Override
726                 public void uninject() {
727                     super.uninject();
728                     if (callbackHandle != null) {
729                         callbackHandle.remove();
730                     }
731                 }
732             };
733             dependencies.add(new Dependency<PathManager>(PathManagerService.SERVICE_NAME, PathManager.class, injector));
734             return builder;
735         } else if (storeKey.equals(ModelKeys.STRING_KEYED_JDBC_STORE) || storeKey.equals(ModelKeys.BINARY_KEYED_JDBC_STORE) || storeKey.equals(ModelKeys.MIXED_KEYED_JDBC_STORE)) {

737             DatabaseType databaseType = dialectNode.isDefined() ? DatabaseType.valueOf(dialectNode.asString()) : null;
738 
739             AbstractJdbcStoreConfigurationBuilder<?, ?> builder = buildJdbcStore(persistenceBuilder, context, store, databaseType);

751                     @Override

755                         } catch (UnknownHostException e) {
756                             throw InfinispanMessages.MESSAGES.failedToInjectSocketBinding(e, value);
757                         }
758                     }
759                 };
760                 dependencies.add(new Dependency<OutboundSocketBinding>(OutboundSocketBinding.OUTBOUND_SOCKET_BINDING_BASE_SERVICE_NAME.append(outboundSocketBinding), OutboundSocketBinding.class, injector));
761             }
762             if (store.hasDefined(ModelKeys.CACHE)) {
763                 builder.remoteCacheName(store.get(ModelKeys.CACHE).asString());
764             }
765             if (store.hasDefined(ModelKeys.HOTROD_WRAPPING)) {
766                 builder.hotRodWrapping(store.require(ModelKeys.HOTROD_WRAPPING).asBoolean());
767             }
768             if (store.hasDefined(ModelKeys.RAW_VALUES)) {
769                 builder.rawValues(store.require(ModelKeys.RAW_VALUES).asBoolean());
770             }
771             if (store.hasDefined(ModelKeys.SOCKET_TIMEOUT)) {
772                 builder.socketTimeout(store.require(ModelKeys.SOCKET_TIMEOUT).asLong());
773             }
774             if (store.hasDefined(ModelKeys.TCP_NO_DELAY)) {
775                 builder.tcpNoDelay(store.require(ModelKeys.TCP_NO_DELAY).asBoolean());
776             }
777             return builder;
778         } else if (storeKey.equals(ModelKeys.LEVELDB_STORE)) {
779             final LevelDBStoreConfigurationBuilder builder = persistenceBuilder.addStore(LevelDBStoreConfigurationBuilder.class);
780             final String path = ((resolvedValue = LevelDBStoreResource.PATH.resolveModelAttribute(context, store)).isDefined()) ? resolvedValue.asString() : InfinispanExtension.SUBSYSTEM_NAME + File.separatorChar + containerName + File.separatorChar + "data";
781             final String relativeTo = ((resolvedValue = LevelDBStoreResource.RELATIVE_TO.resolveModelAttribute(context, store)).isDefined()) ? resolvedValue.asString() : ServerEnvironment.SERVER_DATA_DIR;
782             Injector<PathManager> injector = new SimpleInjector<PathManager>() {
783                 volatile PathManager.Callback.Handle callbackHandle;
784                 @Override
785                 public void inject(PathManager value) {
786                     callbackHandle = value.registerCallback(relativeTo, PathManager.ReloadServerCallback.create(), PathManager.Event.UPDATED, PathManager.Event.REMOVED);
787                     builder.location(value.resolveRelativePathEntry(path, relativeTo));
788                 }
789 
790                 @Override
791                 public void uninject() {
792                     super.uninject();
793                     if (callbackHandle != null) {
794                         callbackHandle.remove();
795                     }
796                 }
797             };
798             dependencies.add(new Dependency<PathManager>(PathManagerService.SERVICE_NAME, PathManager.class, injector));
799 
800             final boolean expirationDefined = store.hasDefined(ModelKeys.EXPIRATION) && store.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME).isDefined();
801             final String expirationPath;
802             if (expirationDefined) {
803                 ModelNode expiration = store.get(ModelKeys.EXPIRATION, ModelKeys.EXPIRATION_NAME);
804                 expirationPath = LevelDBExpirationResource.PATH.resolveModelAttribute(context, expiration).asString();

841             if (store.hasDefined(ModelKeys.IMPLEMENTATION)) {
842                 ModelNode node = store.get(ModelKeys.IMPLEMENTATION, ModelKeys.IMPLEMENTATION_NAME);
843                 final LevelDBStoreConfiguration.ImplementationType implementationType = LevelDBStoreConfiguration.ImplementationType.valueOf(LevelDBImplementationResource.TYPE.resolveModelAttribute(context, node).asString());
844                 builder.implementationType(implementationType);
845             }
846             return builder;
847         } else if (storeKey.equals(ModelKeys.REST_STORE)) {
848                 final RestStoreConfigurationBuilder builder = persistenceBuilder.addStore(RestStoreConfigurationBuilder.class);
849                 builder.host("localhost"); // To pass builder validation, the builder will be configured properly when the outbound socket is ready to be injected

897            try {
898               return handleStoreOrLoaderClass(className, persistenceBuilder);

902         } else {
903            throw new IllegalStateException();

911    private void parseCommonAttributes(OperationContext context, PersistenceConfigurationBuilder persistenceBuilder, ModelNode store, StoreConfigurationBuilder storeConfigurationBuilder) throws OperationFailedException {
912       ModelNode shared = store.get(ModelKeys.SHARED);
913       if (shared != null && shared.isDefined()) {
914          storeConfigurationBuilder.shared(shared.asBoolean());
915       }
916       ModelNode preload = store.get(ModelKeys.PRELOAD);
917       if (preload != null && preload.isDefined()) {
918          storeConfigurationBuilder.preload(preload.asBoolean());
919       }
920       ModelNode fetchState = store.get(ModelKeys.FETCH_STATE);
921       if (fetchState != null && fetchState.isDefined()) {
922          storeConfigurationBuilder.fetchPersistentState(fetchState.asBoolean());
923       }
924       ModelNode passivation = store.get(ModelKeys.PASSIVATION);
925       if (passivation != null && passivation.isDefined()) {
926          persistenceBuilder.passivation(passivation.asBoolean());
927       }
928       ModelNode purge = store.get(ModelKeys.PURGE);
929       if (purge != null && purge.isDefined()) {
930          storeConfigurationBuilder.purgeOnStartup(purge.asBoolean());
931       }
932       ModelNode singleton = store.get(ModelKeys.SINGLETON);
933       if (singleton != null && singleton.isDefined()) {
934          if (singleton.asBoolean())
935             storeConfigurationBuilder.singleton().enable();
936       }
937       ModelNode readOnly = store.get(ModelKeys.READ_ONLY);
938       if (readOnly != null && readOnly.isDefined()) {
939          storeConfigurationBuilder.ignoreModifications(readOnly.asBoolean());
940       }
941       final boolean async = store.hasDefined(ModelKeys.WRITE_BEHIND) && store.get(ModelKeys.WRITE_BEHIND, ModelKeys.WRITE_BEHIND_NAME).isDefined();
942       if (async) {
943          ModelNode writeBehind = store.get(ModelKeys.WRITE_BEHIND, ModelKeys.WRITE_BEHIND_NAME);
944          storeConfigurationBuilder.async().enable()
945                .flushLockTimeout(StoreWriteBehindResource.FLUSH_LOCK_TIMEOUT.resolveModelAttribute(context, writeBehind).asLong())
946                .modificationQueueSize(StoreWriteBehindResource.MODIFICATION_QUEUE_SIZE.resolveModelAttribute(context, writeBehind).asInt())
947                .shutdownTimeout(StoreWriteBehindResource.SHUTDOWN_TIMEOUT.resolveModelAttribute(context, writeBehind).asLong())
948                .threadPoolSize(StoreWriteBehindResource.THREAD_POOL_SIZE.resolveModelAttribute(context, writeBehind).asInt())
949          ;
950       }
951 
952       final Properties properties = new TypedProperties();
953       if (store.hasDefined(ModelKeys.PROPERTY)) {
954          for (Property property : store.get(ModelKeys.PROPERTY).asPropertyList()) {
955             // format of properties
956             // "property" => {
957             //   "property-name" => {"value => "property-value"}
958             // }
959             String propertyName = property.getName();
960             // get the value from ModelNode {"value" => "property-value"}
961             ModelNode propertyValue = null ;
962             propertyValue = StorePropertyResource.VALUE.resolveModelAttribute(context,property.getValue());
963             properties.setProperty(propertyName, propertyValue.asString());
964          }
965       }
966       storeConfigurationBuilder.withProperties(properties);
967 
968       ModelNode writeBehind = store.get(ModelKeys.WRITE_BEHIND);
969       if (writeBehind != null && writeBehind.isDefined()) {
970          if (writeBehind.asBoolean())
971             storeConfigurationBuilder.async();
972       }
973    }

975    private AbstractJdbcStoreConfigurationBuilder<?, ?> buildJdbcStore(PersistenceConfigurationBuilder loadersBuilder, OperationContext context, ModelNode store, DatabaseType databaseType) throws OperationFailedException {
976         boolean useStringKeyedTable = store.hasDefined(ModelKeys.STRING_KEYED_TABLE);
977         boolean useBinaryKeyedTable = store.hasDefined(ModelKeys.BINARY_KEYED_TABLE);
978         if (useStringKeyedTable && !useBinaryKeyedTable) {
979             JdbcStringBasedStoreConfigurationBuilder builder = loadersBuilder.addStore(JdbcStringBasedStoreConfigurationBuilder.class);
980             builder.dialect(databaseType);
981             this.buildStringKeyedTable(builder.table(), context, store.get(ModelKeys.STRING_KEYED_TABLE));
982             return builder;
983         } else if (useBinaryKeyedTable && !useStringKeyedTable) {
984             JdbcBinaryStoreConfigurationBuilder builder = loadersBuilder.addStore(JdbcBinaryStoreConfigurationBuilder.class);
985             builder.dialect(databaseType);
986             this.buildBinaryKeyedTable(builder.table(), context, store.get(ModelKeys.BINARY_KEYED_TABLE));
987             return builder;
988         }
989         // Else, use mixed mode
990         JdbcMixedStoreConfigurationBuilder builder = loadersBuilder.addStore(JdbcMixedStoreConfigurationBuilder.class);
991         builder.dialect(databaseType);
992         this.buildStringKeyedTable(builder.stringTable(), context, store.get(ModelKeys.STRING_KEYED_TABLE));
993         this.buildBinaryKeyedTable(builder.binaryTable(), context, store.get(ModelKeys.BINARY_KEYED_TABLE));
994         return builder;
995     }

1005     private void buildTable(TableManipulationConfigurationBuilder<?, ?> builder, OperationContext context, ModelNode table, String defaultTableNamePrefix) throws OperationFailedException {
1006         ModelNode tableNamePrefix = BaseJDBCStoreResource.PREFIX.resolveModelAttribute(context, table);
1007         builder.batchSize(BaseJDBCStoreResource.BATCH_SIZE.resolveModelAttribute(context, table).asInt())
1008                 .fetchSize(BaseJDBCStoreResource.FETCH_SIZE.resolveModelAttribute(context, table).asInt())
1009                 .tableNamePrefix(tableNamePrefix.isDefined() ? tableNamePrefix.asString() : defaultTableNamePrefix)
1010                 .idColumnName(this.getColumnProperty(context, table, ModelKeys.ID_COLUMN, BaseJDBCStoreResource.COLUMN_NAME, "id"))
1011                 .idColumnType(this.getColumnProperty(context, table, ModelKeys.ID_COLUMN, BaseJDBCStoreResource.COLUMN_TYPE, "VARCHAR"))
1012                 .dataColumnName(this.getColumnProperty(context, table, ModelKeys.DATA_COLUMN, BaseJDBCStoreResource.COLUMN_NAME, "datum"))
1013                 .dataColumnType(this.getColumnProperty(context, table, ModelKeys.DATA_COLUMN, BaseJDBCStoreResource.COLUMN_TYPE, "BINARY"))
1014                 .timestampColumnName(this.getColumnProperty(context, table, ModelKeys.TIMESTAMP_COLUMN, BaseJDBCStoreResource.COLUMN_NAME, "version"))
1015                 .timestampColumnType(this.getColumnProperty(context, table, ModelKeys.TIMESTAMP_COLUMN, BaseJDBCStoreResource.COLUMN_TYPE, "BIGINT"));
1016     }

1018     private String getColumnProperty(OperationContext context, ModelNode table, String columnKey, AttributeDefinition columnAttribute, String defaultValue) throws OperationFailedException
1019     {
1020         if (!table.isDefined() || !table.hasDefined(columnKey)) return defaultValue;
1021         ModelNode column = table.get(columnKey);
1022         ModelNode resolvedValue = null ;
1023         return ((resolvedValue = columnAttribute.resolveModelAttribute(context, column)).isDefined()) ? resolvedValue.asString() : defaultValue;
1024     }
