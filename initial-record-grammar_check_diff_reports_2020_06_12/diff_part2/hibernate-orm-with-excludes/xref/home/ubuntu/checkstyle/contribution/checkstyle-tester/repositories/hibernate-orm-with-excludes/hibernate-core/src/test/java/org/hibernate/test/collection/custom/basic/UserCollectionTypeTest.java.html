24  package org.hibernate.test.collection.custom.basic;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Hibernate;
29  import org.hibernate.Session;
30  import org.hibernate.Transaction;
31  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
32  
33  import static org.junit.Assert.assertEquals;
34  import static org.junit.Assert.assertTrue;
35  
36  /**
37   * @author Max Rydahl Andersen
38   */
39  public abstract class UserCollectionTypeTest extends BaseCoreFunctionalTestCase {
40  
41  	@Override
42  	protected String getCacheConcurrencyStrategy() {
43  		return "nonstrict-read-write";
44  	}
45  
46  	@Test
47  	public void testBasicOperation() {
48  		Session s = openSession();
49  		Transaction t = s.beginTransaction();
50  		User u = new User("max");
51  		u.getEmailAddresses().add( new Email("max@hibernate.org") );
52  		u.getEmailAddresses().add( new Email("max.andersen@jboss.com") );
53  		s.persist(u);
54  		t.commit();
55  		s.close();
56  		
57  		s = openSession();
58  		t = s.beginTransaction();
59  		User u2 = (User) s.createCriteria(User.class).uniqueResult();
60  		assertTrue( Hibernate.isInitialized( u2.getEmailAddresses() ) );
61  		assertEquals( u2.getEmailAddresses().size(), 2 );
62  		t.commit();
63  		s.close();
64  
65  		s = openSession();
66  		t = s.beginTransaction();
67  		u2 = ( User ) s.get( User.class, u.getUserName() );
68  		u2.getEmailAddresses().size();
69  		assertEquals( 2, MyListType.lastInstantiationRequest );
70  		t.commit();
71  		s.close();
72  
73  		s = openSession();
74  		t = s.beginTransaction();
75  		s.delete( u );
76  		t.commit();
77  		s.close();
78  	}
79  
80  }
