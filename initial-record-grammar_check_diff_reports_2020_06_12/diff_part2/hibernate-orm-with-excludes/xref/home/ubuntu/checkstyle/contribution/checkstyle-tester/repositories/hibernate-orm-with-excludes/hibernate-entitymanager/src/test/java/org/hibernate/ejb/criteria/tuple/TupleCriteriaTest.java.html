40  import static org.junit.Assert.assertEquals;
41  import static org.junit.Assert.assertTrue;

43  /**
44   * @author Emmanuel Bernard
45   */
46  public class TupleCriteriaTest extends AbstractMetamodelSpecificTest {
47  	@Test
48  	public void testArray() {
49  		EntityManager em = entityManagerFactory().createEntityManager();
50  		em.getTransaction().begin();
51  		Customer c1 = new Customer();
52  		c1.setId( "c1" );
53  		c1.setAge( 18 );
54  		c1.setName( "Bob" );
55  		em.persist( c1 );
56  		em.getTransaction().commit();
57  		em.close();
58  
59  		em = entityManagerFactory().createEntityManager();
60  		em.getTransaction().begin();
61  		final CriteriaBuilder cb = em.getCriteriaBuilder();
62  		CriteriaQuery<Object[]> q = cb.createQuery(Object[].class);
63  		Root<Customer> c = q.from(Customer.class);
64  		q.select( cb.array( c.get(Customer_.name), c.get(Customer_.age) ) );
65  		List<Object[]> result = em.createQuery(q).getResultList();
66  		assertEquals( 1, result.size() );
67  		assertEquals( c1.getName(), result.get( 0 )[0] );
68  		assertEquals( c1.getAge(), result.get( 0 )[1] );
69  		em.getTransaction().commit();
70  		em.close();
71  
72  		em = entityManagerFactory().createEntityManager();
73  		em.getTransaction().begin();
74  		em.createQuery( "delete Customer" ).executeUpdate();
75  		em.getTransaction().commit();
76  		em.close();
77  	}
78  
79  	@Test
80  	public void testTuple() {
81  		EntityManager em = entityManagerFactory().createEntityManager();
82  
83  		em.getTransaction().begin();
84  		Customer c1 = new Customer();
85  		c1.setId( "c1" );
86  		c1.setAge( 18 );
87  		c1.setName( "Bob" );
88  		em.persist( c1 );
89  		em.getTransaction().commit();
90  		em.close();
91  
92  		em = entityManagerFactory().createEntityManager();
93  		em.getTransaction().begin();
94  		final CriteriaBuilder builder = em.getCriteriaBuilder();
95  		CriteriaQuery<Tuple> criteria = builder.createTupleQuery();
96  		Root<Customer> customerRoot = criteria.from( Customer.class );
97  		Path<String> namePath = customerRoot.get( Customer_.name );
98  		Path<Integer> agePath = customerRoot.get( Customer_.age );
99  		agePath.alias( "age" );
100 		criteria.multiselect( namePath, agePath );
101 		List<Tuple> results = em.createQuery( criteria ).getResultList();
102 		assertEquals( 1, results.size() );
103 		Object resultElement = results.get( 0 );
104 		assertTrue( "Check  result 'row' as Tuple", Tuple.class.isInstance( resultElement ) );
105 		Tuple resultElementTuple = (Tuple) resultElement;
106 		Object[] tupleArray = resultElementTuple.toArray();
107 		assertEquals( 2, tupleArray.length );
108 		assertEquals( tupleArray[0], resultElementTuple.get( 0 ) );
109 		assertEquals( resultElementTuple.get( namePath ), resultElementTuple.get( 0 ) );
110 		assertEquals( tupleArray[1], resultElementTuple.get( 1 ) );
111 		assertEquals( resultElementTuple.get( agePath ), resultElementTuple.get( 1 ) );
112 		assertEquals( resultElementTuple.get( agePath ), resultElementTuple.get( "age" ) );
113 		em.getTransaction().commit();
114 		em.close();
115 
116 		em = entityManagerFactory().createEntityManager();
117 		em.getTransaction().begin();
118 		em.createQuery( "delete Customer" ).executeUpdate();
119 		em.getTransaction().commit();
120 		em.close();
121 	}
