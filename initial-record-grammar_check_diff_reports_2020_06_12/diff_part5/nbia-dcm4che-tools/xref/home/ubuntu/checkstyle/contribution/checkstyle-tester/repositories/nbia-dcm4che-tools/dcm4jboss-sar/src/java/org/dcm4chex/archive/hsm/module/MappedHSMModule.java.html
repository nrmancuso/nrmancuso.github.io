40  package org.dcm4chex.archive.hsm.module;
41  
42  import java.io.File;
43  import java.util.HashMap;
44  import java.util.Map.Entry;
45  import java.util.StringTokenizer;
46  
47  import javax.management.MalformedObjectNameException;
48  import javax.management.ObjectName;
49  
50  /**
51   * @author franz.willer@gmail.com
52   * @version $Revision:  $ $Date: $
53   * @since June 22, 2012
54   */
55  public class MappedHSMModule extends AbstractHSMModule {
56  
57      public static final String NEW_LINE = System.getProperty("line.separator", "\n");
58      private static final String STRING = String.class.getName();
59      private static final String FILE = File.class.getName();
60  
61      HashMap<String, ObjectName> mapping = new HashMap<String, ObjectName>();
62          
63      protected void startService() throws Exception {}
64  
65      protected void stopService() throws Exception {}
66      
67      public String getModuleMapping() {
68          if (mapping.isEmpty())
69              return NONE;
70          StringBuilder sb = new StringBuilder();
71          for (Entry<String, ObjectName> entry : mapping.entrySet()) {
72              sb.append(entry.getKey()).append('=').append(entry.getValue()).append(NEW_LINE);
73          }
74          return sb.toString();
75      }
76      
77      public void setModuleMapping(String s) throws MalformedObjectNameException, NullPointerException {
78          s = s.trim();
79          if(NONE.equals(s)) {
80              mapping.clear();
81          } else {
82              StringTokenizer st = new StringTokenizer(s, " \t\n\r;");
83              int pos;
84              HashMap<String, ObjectName> newMap = new HashMap<String, ObjectName>();
85              for (String tk ; st.hasMoreElements() ;) {
86                  tk = st.nextToken();
87                  pos = tk.indexOf('=');
88                  if (pos == -1)
89                      throw new IllegalArgumentException("Wrong format: must be <FilesystemID>=<HSMModule Service Name>[<NewLine>...]");
90                  newMap.put(tk.substring(0, pos), new ObjectName(tk.substring(++pos)));
91              }
92              mapping = newMap;
93          }
94      }
95      
96      @Override
97      public File prepareHSMFile(String fsID, String filePath) throws HSMException {
98          return (File) invoke(fsID, "prepareHSMFile", new Object[]{fsID, filePath}, new String[]{STRING, STRING});
99      }
100 
101     @Override
102     public String storeHSMFile(File file, String fsID, String filePath) throws HSMException {
103         return (String)invoke(fsID, "storeHSMFile", new Object[]{file, fsID, filePath}, new String[]{FILE, STRING, STRING});
104     }
105     
106     @Override
107     public void failedHSMFile(File file, String fsID, String filePath) throws HSMException {
108         invoke(fsID, "failedHSMFile", new Object[]{file, fsID, filePath}, new String[]{FILE, STRING, STRING});
109     }
110 
111     @Override
112     public File fetchHSMFile(String fsID, String filePath) throws HSMException {
113         return (File)invoke(fsID, "fetchHSMFile", new Object[]{fsID, filePath}, new String[]{STRING, STRING});
114     }
115 
116     @Override
117     public void fetchHSMFileFinished(String fsID, String filePath, File file) throws HSMException {
118         invoke(fsID, "fetchHSMFileFinished", new Object[]{fsID, filePath, file}, new String[]{STRING, STRING, FILE});
119     }
120     
121     @Override
122     public Integer queryStatus(String fsID, String filePath, String userInfo) throws HSMException {
123         return (Integer)invoke(fsID, "queryStatus", new Object[]{fsID, filePath, userInfo}, new String[]{STRING, STRING, STRING});
124     }
125     
126     private Object invoke(String fsID, String cmd, Object[] params, String[] paramTypes) throws HSMException {
127         ObjectName hsmModuleServicename = mapping.get(fsID);
128         if (hsmModuleServicename == null) {
129             throw new HSMException("Invoke command '"+cmd+"' on mapped HSMModule failed! No mapping for fsID:"+fsID);
130         }
131         log.debug("Invoke "+cmd+" on MappedHSMModule! module:"+hsmModuleServicename);
132         try {
133             return server.invoke(hsmModuleServicename, cmd, params, paramTypes);
134         } catch (Exception x) {
135             throw new HSMException("Invoke command '"+cmd+"' on mapped HSMModule failed! HSMModule:"+
136                     hsmModuleServicename+" (for fsID:"+fsID+")", x);
137         }
138     }
139 }
