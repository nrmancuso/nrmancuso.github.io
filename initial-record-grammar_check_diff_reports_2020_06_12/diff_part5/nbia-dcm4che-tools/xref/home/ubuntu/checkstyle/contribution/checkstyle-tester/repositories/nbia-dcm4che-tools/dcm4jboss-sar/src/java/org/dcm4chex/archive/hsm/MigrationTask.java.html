62  public class MigrationTask implements Runnable {
63  
64      private static long taskCounter = 0;
65      private long taskNumber;
66      
67      private HSMMigrateService service;
68      private Iterator<String> tarfilenames;
69      FileSystemMgt2 mgr;
70      String sourceFs, targetFs;
71      int[] counts;
72      
73      private byte[] buf = new byte[8192];
74  
75      private static final Logger log = Logger.getLogger(MigrationTask.class);
76      
77      public MigrationTask(HSMMigrateService service, Iterator<String> tarfilenames, 
78              FileSystemMgt2 mgr, int[] count) {
79          taskNumber = ++taskCounter;
80          this.service = service;
81          this.tarfilenames = tarfilenames;
82          this.mgr = mgr;
83          sourceFs = service.getSourceFileSystem();
84          targetFs = service.getTargetFileSystem();
85          this.counts = count;
86      }
87      private void migrateTarFile(String srcFsId, String srcTarFilename, String targetFsId) throws Exception {
88          int nrOfCopies = new QueryHSMMigrateCmd().countFileCopiesOfTarFile(service.getSrcFsPk(), srcTarFilename, service.getTargetFsPk());
89          log.debug("########### nrOfCopies:"+nrOfCopies);
90          FileDTO[] dtos = null;
91          if (nrOfCopies > 0 || service.isVerifyTar()) {
92              dtos = mgr.getFilesOfTarFile(srcFsId, srcTarFilename);
93              log.debug("########### dtos.length:"+dtos.length);
94              if (nrOfCopies > 0 && dtos.length == nrOfCopies) {
95                  log.warn(this+" - File copies of all files in source tar file "+srcTarFilename+" already exists! Set file status of source to MIGRATED.");
96                  mgr.setFilestatusOfFilesOfTarFile(srcFsId, srcTarFilename, FileStatus.MIGRATED);
97                  return;
98              }
99          }
100         File src = service.fetchTarFile(srcFsId, srcTarFilename);
101         if (service.isVerifyTar()) {
102             try {
103                 List<FileDTO> missingFiles = verifyTar(src, dtos, true);
104                 for (int i = 0, len = missingFiles.size() ; i < len ; i++) {
105                     mgr.setFileStatus(missingFiles.get(i).getPk(), FileStatus.MD5_CHECK_FAILED);
106                 }
107             } catch (Exception x) {
108                 log.error(x.getMessage()+" Set file status of source tar file entities to MD5_CHECK_FAILED and skip migration of "+srcTarFilename);
109                 mgr.setFilestatusOfFilesOfTarFile(srcFsId, srcTarFilename, FileStatus.MD5_CHECK_FAILED);
110                 service.fetchTarFileFinished(srcFsId, srcTarFilename, src);
111                 return;
112             }
113         }
114         int nrOfFiles = 0;
115         String targetTarFilename = srcTarFilename;
116         File target = null;
117         try {
118             target = service.prepareHSMFile(targetFsId, srcTarFilename);
119             if(!target.exists()) {
120                 if (!target.getParentFile().exists()) {
121                     log.info("M-CREATE Directory "+target);
122                     target.getParentFile().mkdirs();
123                 }
124                 log.info("M-CREATE "+target);
125                 target.createNewFile();
126             }
127             if (target.length() > 0) {
128                 log.warn(this+" - Target tar file "+target+" already exists! Skip migration with assumption that this file is already migrated and set file status of source to MIGRATED.");
129                 mgr.setFilestatusOfFilesOfTarFile(srcFsId, srcTarFilename, FileStatus.MIGRATED);
130                 return;
131             } else {
132                 FileChannel srcCh = null;
133                 FileChannel destCh = null;
134                 try {
135                     srcCh = new FileInputStream(src).getChannel();
136                     destCh = new FileOutputStream(target).getChannel();
137                     destCh.transferFrom(srcCh, 0, srcCh.size());
138                 } finally {
139                   if(srcCh != null)
140                       srcCh.close();
141                   if(destCh != null)
142                       destCh.close();
143                 }
144             }
145             targetTarFilename = service.storeHSMFile(target, targetFsId, srcTarFilename);
146             nrOfFiles = mgr.migrateFilesOfTarFile(srcFsId, srcTarFilename, targetFsId, targetTarFilename, 
147                     FileStatus.toInt(service.getTargetFileStatus()));
148             if (service.isVerifyTar()) {
149                 verifyTar(service.fetchTarFile(targetFsId, targetTarFilename), mgr.getFilesOfTarFile(targetFsId, targetTarFilename), false);
150             }
151         } catch (Exception x) {
152             if (target != null) {
153                 service.failedHSMFile(target, targetFsId, targetTarFilename);
154                 target.delete();
155                 log.error("Remove file entities of failed migrated tar file!");
156                 mgr.deleteFilesOfInvalidTarFile(targetFsId, targetTarFilename);
157             }
158             throw x;
159         } finally {
160             service.fetchTarFileFinished(srcFsId, srcTarFilename, src);
161         }
162         synchronized(counts) {
163             counts[1] += nrOfFiles;
164             log.info(nrOfFiles+" files migrated by "+this+" - Tasks in progress:"+counts[0]);
165         }
166     }
167 
168     private List<FileDTO> verifyTar(File file, FileDTO[] files, boolean markMissingFiles) throws IOException, VerifyTarException {
169         Map<String, byte[]> entries = VerifyTar.verify(file, buf);
170         if (entries == null) 
171             throw new VerifyTarException("Verify tar failed! Tar file has no entries.");
172         List<FileDTO> missingFiles = new ArrayList<FileDTO>();
173         for (int i = 0 ; i < files.length ; i++) {
174             String filePath = files[i].getFilePath();
175             String filepathInTar = files[i].getFilePath().substring(filePath.indexOf('!')+1);
176             byte[] md5 = entries.get(filepathInTar);
177             if (md5 == null) {
178                 String msg = this+" - Verify tar failed! "+filepathInTar+" not found in tar file "+file;
179                 if (markMissingFiles) {
180                     log.error(msg+". Set file status to MD5_CHECK_FAILED");
181                     missingFiles.add(files[i]);
182                 } else {
183                     throw new VerifyTarException(msg);
184                 }
185             }
186             if (!Arrays.equals(files[i].getFileMd5(), md5)) {
187                 String msg = this+" - Verify tar failed! Different MD5 of file "+filepathInTar+" and file entity! ("+
188                               MD5.toString(md5)+" vs. "+files[i].getMd5String()+ ")";
189                 if (markMissingFiles) {
190                     log.error(msg+". Set file status to MD5_CHECK_FAILED");
191                     missingFiles.add(files[i]);
192                 } else {
193                     throw new VerifyTarException(msg);
194                 }
195             }
196         }
197         if (missingFiles.size() == files.length)
198             throw new VerifyTarException(this+" - Verify tar failed! No file entity found in tar file.");
199         return missingFiles;
200     }
201 
202     /**
203      * Get counts of migrationTasks
204      * [0]: remaining tasks for a specific migration run
205      * [1]: count migrated tar files.
206      * @return
207      */
208     public int[] getCounts() {
209         return counts;
210     }
211 
212     @Override
213     public void run() {
214         String fn = "";
215         synchronized (tarfilenames) {
216             fn = tarfilenames.hasNext() ? tarfilenames.next() : null;
217         }
218         while(fn != null) {
219             log.info(this+" - Migrate tar file: "+fn);
220             try {
221                 migrateTarFile(sourceFs, fn, targetFs);
222             } catch (Exception e) {
223                 try {
224                     log.error(this+" - Migration of "+fn+" failed!", e);
225                     mgr.setFilestatusOfFilesOfTarFile(sourceFs, fn, FileStatus.MIGRATION_FAILED);
226                 } catch (Exception x) {
227                     log.error(this+" - Set FileStatus (MIGRATION_FAILED) of "+fn+" failed!", x);
228                 }
229             }
230             synchronized (tarfilenames) {
231                 fn = tarfilenames.hasNext() ? tarfilenames.next() : null;
232             }
233         }
234         service.removeMigrationTask(this);
235     }
236 
237     public String toString() {
238         return "MigrationTask:"+taskNumber;
239     }
