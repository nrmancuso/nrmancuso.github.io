446     private void setupFontList(int rangeStart, int rangeEnd) {
447 
448         int listCount = fontMenu.getItemCount();
449         int size = 16;
450 
451         try {
452             size =  Float.valueOf(sizeField.getText()).intValue();
453         }
454         catch ( Exception e ) {
455             System.out.println("Invalid font size in the size textField. Using default value of 16");
456         }
457 
458         int style = fontStyles[styleMenu.getSelectedIndex()];
459         Font f;
460         for (int i = 0; i < listCount; i++) {
461             String fontName = (String)fontMenu.getItemAt(i);
462             f = new Font(fontName, style, size);
463             if ((rm.getSelectedIndex() != RangeMenu.SURROGATES_AREA_INDEX) &&
464                 canDisplayRange(f, rangeStart, rangeEnd)) {
465                 fontMenu.setBit(i, true);
466             }
467             else {
468                 fontMenu.setBit(i, false);
469             }
470         }
471 
472         fontMenu.repaint();
473     }

475     protected boolean canDisplayRange(Font font, int rangeStart, int rangeEnd) {
476         for (int i = rangeStart; i < rangeEnd; i++) {
477             if (font.canDisplay(i)) {
478                 return true;
479             }
480         }
481         return false;
482     }

521     /// Converts user text into arrays of String, delimited at newline character
522     /// Also replaces any valid escape sequence with appropriate unicode character
523     /// Support \\UXXXXXX notation for surrogates
524     private String[] parseUserText( String orig ) {
525         int length = orig.length();
526         StringTokenizer perLine = new StringTokenizer( orig, "\n" );
527         String textLines[] = new String[ perLine.countTokens() ];
528         int lineNumber = 0;
529 
530         while ( perLine.hasMoreElements() ) {
531             StringBuffer converted = new StringBuffer();
532             String oneLine = perLine.nextToken();
533             int lineLength = oneLine.length();
534             int prevEscapeEnd = 0;
535             int nextEscape = -1;
536             do {
537                 int nextBMPEscape = oneLine.indexOf( "\\u", prevEscapeEnd );
538                 int nextSupEscape = oneLine.indexOf( "\\U", prevEscapeEnd );
539                 nextEscape = (nextBMPEscape < 0)
540                     ? ((nextSupEscape < 0)
541                        ? -1
542                        : nextSupEscape)
543                     : ((nextSupEscape < 0)
544                        ? nextBMPEscape
545                        : Math.min(nextBMPEscape, nextSupEscape));
546 
547                 if ( nextEscape != -1 ) {
548                     if ( prevEscapeEnd < nextEscape )
549                         converted.append( oneLine.substring( prevEscapeEnd, nextEscape ));
550 
551                     prevEscapeEnd = nextEscape + (nextEscape == nextBMPEscape ? 6 : 8);
552                     try {
553                         String hex = oneLine.substring( nextEscape + 2, prevEscapeEnd );
554                         if (nextEscape == nextBMPEscape) {
555                             converted.append( (char) Integer.parseInt( hex, 16 ));
556                         } else {
557                             converted.append( new String( Character.toChars( Integer.parseInt( hex, 16 ))));
558                         }
559                     }
560                     catch ( Exception e ) {
561                         int copyLimit = Math.min(lineLength, prevEscapeEnd);
562                         converted.append( oneLine.substring( nextEscape, copyLimit ));
563                     }
564                 }
565             } while (nextEscape != -1);
566             if ( prevEscapeEnd < lineLength )
567               converted.append( oneLine.substring( prevEscapeEnd, lineLength ));
568             textLines[ lineNumber++ ] = converted.toString();
569         }
570         return textLines;
571     }

573     /// Reads the text from specified file, detecting UTF-16 encoding
574     /// Then breaks the text into String array, delimited at every line break
575     private void readTextFile( String fileName ) {
576         try {
577             String fileText, textLines[];
578             BufferedInputStream bis =
579               new BufferedInputStream( new FileInputStream( fileName ));
580             int numBytes = bis.available();
581             if (numBytes == 0) {
582                 throw new Exception("Text file " + fileName + " is empty");
583             }
584             byte byteData[] = new byte[ numBytes ];
585             bis.read( byteData, 0, numBytes );
586             bis.close();
587 
588             /// If byte mark is found, then use UTF-16 encoding to convert bytes...
589             if (numBytes >= 2 &&
590                 (( byteData[0] == (byte) 0xFF && byteData[1] == (byte) 0xFE ) ||
591                  ( byteData[0] == (byte) 0xFE && byteData[1] == (byte) 0xFF )))
592               fileText = new String( byteData, "UTF-16" );
593             /// Otherwise, use system default encoding
594             else
595               fileText = new String( byteData );
596 
597             int length = fileText.length();
598             StringTokenizer perLine = new StringTokenizer( fileText, "\n" );
599             /// Determine "Return Char" used in this file
600             /// This simply finds first occurrence of CR, CR+LF or LF...
601             for ( int i = 0; i < length; i++ ) {
602                 char iTh = fileText.charAt( i );
603                 if ( iTh == '\r' ) {
604                     if ( i < length - 1 && fileText.charAt( i + 1 ) == '\n' )
605                       perLine = new StringTokenizer( fileText, "\r\n" );
606                     else
607                       perLine = new StringTokenizer( fileText, "\r" );
608                     break;
609                 }
610                 else if ( iTh == '\n' )
611                   /// Use the one already created
612                   break;
613             }
614             int lineNumber = 0, numLines = perLine.countTokens();
615             textLines = new String[ numLines ];
616 
617             while ( perLine.hasMoreElements() ) {
618                 String oneLine = perLine.nextToken();
619                 if ( oneLine == null )
620                   /// To make LineBreakMeasurer to return a valid TextLayout
621                   /// on an empty line, simply feed it a space char...
622                   oneLine = " ";
623                 textLines[ lineNumber++ ] = oneLine;
624             }
625             fp.setTextToDraw( fp.FILE_TEXT, null, null, textLines );
626             rm.setEnabled( false );
627             methodsMenu.setEnabled( false );
628         }
629         catch ( Exception ex ) {
630             fireChangeStatus( "ERROR: Failed to Read Text File; See Stack Trace", true );
631             ex.printStackTrace();
632         }
633     }

638             String curOptions = fp.getCurrentOptions();
639             BufferedOutputStream bos =
640               new BufferedOutputStream( new FileOutputStream( fileName ));
641             /// Prepend title and the option that is only obtainable here
642             int range[] = rm.getSelectedRange();

644               ( "Font2DTest Option File\n" +

702     /// Loads saved options and applies them
703     private void loadOptions( String fileName ) {
704         try {
705             BufferedInputStream bis =
706               new BufferedInputStream( new FileInputStream( fileName ));
707             int numBytes = bis.available();
708             byte byteData[] = new byte[ numBytes ];
709             bis.read( byteData, 0, numBytes );
710             bis.close();
711             if ( numBytes < 2 ||
712                 (byteData[0] != (byte) 0xFE || byteData[1] != (byte) 0xFF) )
713               throw new Exception( "Not a Font2DTest options file" );
714 
715             String options = new String( byteData, "UTF-16" );
716             StringTokenizer perLine = new StringTokenizer( options, "\n" );
717             String title = perLine.nextToken();
718             if ( !title.equals( "Font2DTest Option File" ))
719               throw new Exception( "Not a Font2DTest options file" );
720 
721             /// Parse all options
722             boolean displayGridOpt = Boolean.parseBoolean( perLine.nextToken() );
723             boolean force16ColsOpt = Boolean.parseBoolean( perLine.nextToken() );
724             boolean showFontInfoOpt = Boolean.parseBoolean( perLine.nextToken() );
725             String rangeNameOpt = perLine.nextToken();
726             int rangeStartOpt = Integer.parseInt( perLine.nextToken() );
727             int rangeEndOpt = Integer.parseInt( perLine.nextToken() );
728             String fontNameOpt = perLine.nextToken();
729             float fontSizeOpt = Float.parseFloat( perLine.nextToken() );
730             int fontStyleOpt = Integer.parseInt( perLine.nextToken() );
731             int fontTransformOpt = Integer.parseInt( perLine.nextToken() );
732             int g2TransformOpt = Integer.parseInt( perLine.nextToken() );
733             int textToUseOpt = Integer.parseInt( perLine.nextToken() );
734             int drawMethodOpt = Integer.parseInt( perLine.nextToken() );
735             int antialiasOpt = Integer.parseInt(perLine.nextToken());
736             int fractionalOpt = Integer.parseInt(perLine.nextToken());
737             int lcdContrast = Integer.parseInt(perLine.nextToken());
738             String userTextOpt[] = { "Font2DTest!" };
739             String dialogEntry = "Font2DTest!";
740             if (textToUseOpt == fp.USER_TEXT )  {
741                 int numLines = perLine.countTokens(), lineNumber = 0;
742                 if ( numLines != 0 ) {
743                     userTextOpt = new String[ numLines ];
744                     dialogEntry = "";
745                     for ( ; perLine.hasMoreElements(); lineNumber++ ) {
746                         userTextOpt[ lineNumber ] = perLine.nextToken();
747                         dialogEntry += userTextOpt[ lineNumber ] + "\n";
748                     }
749                 }
750             }
751 
752             /// Reset GUIs
753             displayGridCBMI.setState( displayGridOpt );
754             force16ColsCBMI.setState( force16ColsOpt );
755             showFontInfoCBMI.setState( showFontInfoOpt );
756             rm.setSelectedRange( rangeNameOpt, rangeStartOpt, rangeEndOpt );
757             fontMenu.setSelectedItem( fontNameOpt );
758             sizeField.setText( String.valueOf( fontSizeOpt ));
759             styleMenu.setSelectedIndex( fontStyleOpt );
760             transformMenu.setSelectedIndex( fontTransformOpt );
761             transformMenuG2.setSelectedIndex( g2TransformOpt );
762             textMenu.setSelectedIndex( textToUseOpt );
763             methodsMenu.setSelectedIndex( drawMethodOpt );
764             antiAliasMenu.setSelectedIndex( antialiasOpt );
765             fracMetricsMenu.setSelectedIndex( fractionalOpt );
766             contrastSlider.setValue(lcdContrast);
767 
768             userTextArea.setText( dialogEntry );
769             updateGUI();
770 
771             if ( textToUseOpt == fp.FILE_TEXT ) {
772               tFileName = perLine.nextToken();
773               readTextFile(tFileName );
774             }
775 
776             /// Reset option variables and repaint
777             fp.loadOptions( displayGridOpt, force16ColsOpt,
778                             rangeStartOpt, rangeEndOpt,
779                             fontNameOpt, fontSizeOpt,
780                             fontStyleOpt, fontTransformOpt, g2TransformOpt,
781                             textToUseOpt, drawMethodOpt,
782                             antialiasOpt, fractionalOpt,
783                             lcdContrast, userTextOpt );
784             if ( showFontInfoOpt ) {
785                 fireUpdateFontInfo();
786                 fontInfoDialog.show();
787             }
788             else
789               fontInfoDialog.hide();
790         }
791         catch ( Exception ex ) {
792             fireChangeStatus( "ERROR: Failed to Load Options File; See Stack Trace", true );
793             ex.printStackTrace();
794         }
795     }

822     /// ActionListener interface function
823     /// Responds to JMenuItem, JTextField and JButton actions
824     public void actionPerformed( ActionEvent e ) {
825         Object source = e.getSource();
826 
827         if ( source instanceof JMenuItem ) {
828             JMenuItem mi = (JMenuItem) source;
829             String itemName = mi.getText();
830 
831             if ( itemName.equals( "Save Selected Options..." )) {
832                 String fileName = promptFile( true, "options.txt" );
833                 if ( fileName != null )
834                   writeCurrentOptions( fileName );
835             }
836             else if ( itemName.equals( "Load Options..." )) {
837                 String fileName = promptFile( false, "options.txt" );
838                 if ( fileName != null )
839                   loadOptions( fileName );
840             }
841             else if ( itemName.equals( "Save as PNG..." )) {
842                 String fileName = promptFile( true, fontMenu.getSelectedItem() + ".png" );
843                 if ( fileName != null )
844                   fp.doSavePNG( fileName );
845             }
846             else if ( itemName.equals( "Load PNG File to Compare..." )) {
847                 String fileName = promptFile( false, null );
848                 if ( fileName != null )
849                   loadComparisonPNG( fileName );
850             }
851             else if ( itemName.equals( "Page Setup..." ))
852               fp.doPageSetup();
853             else if ( itemName.equals( "Print..." ))
854               printDialog.show();
855             else if ( itemName.equals( "Close" ))
856               parent.dispose();
857             else if ( itemName.equals( "Exit" ))
858               System.exit(0);
859         }
860 
861         else if ( source instanceof JTextField ) {
862             JTextField tf = (JTextField) source;
863             float sz = 12f;
864             try {
865                  sz = Float.parseFloat(sizeField.getText());
866                  if (sz < 1f || sz > 120f) {
867                       sz = 12f;
868                       sizeField.setText("12");
869                  }
870             } catch (Exception se) {
871                  sizeField.setText("12");
872             }
873             if ( tf == sizeField )
874               fp.setFontParams( fontMenu.getSelectedItem(),
875                                 sz,
876                                 styleMenu.getSelectedIndex(),
877                                 transformMenu.getSelectedIndex() );
878         }
879 
880         else if ( source instanceof JButton ) {
881             String itemName = ( (JButton) source ).getText();
882             /// Print dialog buttons...
883             if ( itemName.equals( "Print" )) {
884                 for ( int i = 0; i < printModeCBs.length; i++ )
885                   if ( printModeCBs[i].isSelected() ) {
886                       printDialog.hide();
887                       fp.doPrint( i );
888                   }
889             }
890             else if ( itemName.equals( "Cancel" ))
891               printDialog.hide();
892             /// Update button from Usert Text JDialog...
893             else if ( itemName.equals( "Update" ))
894               fp.setTextToDraw( fp.USER_TEXT, null,
895                                 parseUserText( userTextArea.getText() ), null );
896         }
897         else if ( source instanceof JComboBox ) {
898             JComboBox c = (JComboBox) source;
899 
900             /// RangeMenu handles actions by itself and then calls fireRangeChanged,
901             /// so it is not listed or handled here
902             if ( c == fontMenu || c == styleMenu || c == transformMenu ) {
903                 float sz = 12f;
904                 try {
905                     sz = Float.parseFloat(sizeField.getText());
906                     if (sz < 1f || sz > 120f) {
907                         sz = 12f;
908                         sizeField.setText("12");
909                     }
910                 } catch (Exception se) {
911                     sizeField.setText("12");
912                 }
913                 fp.setFontParams(fontMenu.getSelectedItem(),
914                                  sz,
915                                  styleMenu.getSelectedIndex(),
916                                  transformMenu.getSelectedIndex());
917             } else if ( c == methodsMenu )
918               fp.setDrawMethod( methodsMenu.getSelectedIndex() );
919             else if ( c == textMenu ) {
920 
921                 if(canDisplayCheck) {
922                     fireRangeChanged();
923                 }
924 
925                 int selected = textMenu.getSelectedIndex();
926 
927                 if ( selected == fp.RANGE_TEXT )
928                   fp.setTextToDraw( fp.RANGE_TEXT, rm.getSelectedRange(),
929                                     null, null );
930                 else if ( selected == fp.USER_TEXT )
931                   fp.setTextToDraw( fp.USER_TEXT, null,
932                                     parseUserText( userTextArea.getText() ), null );
933                 else if ( selected == fp.FILE_TEXT ) {
934                     String fileName = promptFile( false, null );
935                     if ( fileName != null ) {
936                       tFileName = fileName;
937                       readTextFile( fileName );
938                     } else {
939                         /// User cancelled selection; reset to previous choice
940                         c.setSelectedIndex( currentTextChoice );
941                         return;
942                     }
943                 }
944                 else if ( selected == fp.ALL_GLYPHS )
945                   fp.setTextToDraw( fp.ALL_GLYPHS, null, null, null );
946 
947                 updateGUI();
948                 currentTextChoice = selected;
949             }
950             else if ( c == transformMenuG2 ) {
951                 fp.setTransformG2( transformMenuG2.getSelectedIndex() );
952             }
953             else if (c == antiAliasMenu || c == fracMetricsMenu) {
954                 if (c == antiAliasMenu) {
955                     boolean enabled = FontPanel.AAValues.
956                         isLCDMode(antiAliasMenu.getSelectedItem());
957                         contrastSlider.setEnabled(enabled);
958                 }
959                 fp.setRenderingHints(antiAliasMenu.getSelectedItem(),
960                                      fracMetricsMenu.getSelectedItem(),
961                                      contrastSlider.getValue());
962             }
963         }
964     }
