113     @Override
114     public Object[] getChildren(Object parent) {
115         if (grouping == null) {
116             // on initialization
117             return EMPTY;
118         }
119         Object[] children = EMPTY;
120         if (parent instanceof BugGroup) {
121             BugGroup group = (BugGroup) parent;
122             children = group.getChildren();
123             if (rootElement == parent && rootElement.size() == 0) {
124                 if (DEBUG) {
125                     System.out.println("Root is empty...");
126                 }
127             }
128         } else {
129             if (parent instanceof IWorkspaceRoot || parent instanceof IWorkingSet) {
130                 if (input == parent) {
131                     Object[] objects = rootElement.getChildren();
132                     if (objects.length > 0) {
133                         if (bugFilterActive != isBugFilterActive()) {
134                             refreshFilters();
135                         }
136                         return objects;
137                     }
138                 }
139                 BugGroup root = new BugGroup(null, parent, GroupType.getType(parent));
140                 clearFilters();
141                 children = createChildren(grouping.getFirstType(), getResources(parent), root);
142                 if (input == parent) {
143                     // remember the root
144                     rootElement = root;
145                 }
146             }
147         }
148         return children;
149     }

151     private Set<IResource> getResources(Object parent) {
152         Set<IResource> resources = new HashSet<>();
153         if (parent instanceof IWorkingSet) {
154             IWorkingSet workingSet = (IWorkingSet) parent;
155             IAdaptable[] elements = workingSet.getElements();
156             // elements may contain NON-resource elements, which we have to
157             // convert to
158             // resources
159             for (IAdaptable adaptable : elements) {
160                 IResource resource = (IResource) adaptable.getAdapter(IResource.class);
161                 // TODO get only java projects or children of them
162                 if (resource != null) {
163                     resources.add(resource);
164                 }
165             }
166         } else if (parent instanceof IWorkspaceRoot) {
167             IWorkspaceRoot workspaceRoot = (IWorkspaceRoot) parent;
168             // TODO get only java projects
169             IProject[] projects = workspaceRoot.getProjects();
170             for (IProject project : projects) {
171                 resources.add(project);
172             }
173         }
174         return resources;
175     }

204     @Override
205     public void dispose() {
206         if (DEBUG) {
207             System.out.println("Disposing content provider!");
208         }
209         refreshJob.dispose();
210         rootElement.dispose();
211         clearFilters();
212 
213         IPreferenceStore store = FindbugsPlugin.getDefault().getPreferenceStore();
214         store.setValue(FindBugsConstants.LAST_USED_GROUPING, getGrouping().toString());
215         IWorkingSet workingSet = getCurrentWorkingSet();
216         String name = workingSet != null ? workingSet.getName() : "";
217         store.setValue(FindBugsConstants.LAST_USED_WORKING_SET, name);
218     }

220     @Override
221     public void inputChanged(Viewer newViewer, Object oldInput, Object newInput) {
222         viewer = (CommonViewer) newViewer;
223         if (newInput == null || newInput instanceof IWorkingSet || newInput instanceof IWorkspaceRoot) {
224             rootElement.dispose();
225         }
226         input = newInput;
227         if (newInput == null) {
228             refreshJob.setViewer(null);
229         } else {
230             refreshJob.setViewer((CommonViewer) newViewer);
231             bugFilterActive = isBugFilterActive();
232         }
233         clearFilters();
234     }

236     public void reSetInput() {
237         clearFilters();
238         Object oldInput = getInput();
239         viewer.setInput(null);
240         rootElement.dispose();
241         rootElement = new BugGroup(null, null, GroupType.Undefined);
242         if (oldInput instanceof IWorkingSet || oldInput instanceof IWorkspaceRoot) {
243             viewer.setInput(oldInput);
244         } else {
245             IWorkingSet workingSet = getCurrentWorkingSet();
246             if (workingSet != null) {
247                 viewer.setInput(workingSet);
248             } else {
249                 viewer.setInput(ResourcesPlugin.getWorkspace().getRoot());
250             }
251         }
252     }

262     public void refreshFilters() {
263         clearFilters();
264         bugFilterActive = isBugFilterActive();
265         if (!bugFilterActive) {
266             return;
267         }
268         if (DEBUG) {
269             System.out.println("Refreshing filters!");
270         }
271         Set<String> patternFilter = getPatternFilter();
272         for (IMarker marker : rootElement.getAllMarkers()) {
273             if (MarkerUtil.isFiltered(marker, patternFilter)) {
274                 filteredMarkers.add(marker);
275             }
276         }
277     }

279     public synchronized int getFilteredMarkersCount(BugGroup bugGroup) {
280         if (!isBugFilterActive()) {
281             return 0;
282         }
283         Integer bugCount;
284         bugCount = filteredMarkersMap.get(bugGroup);
285         if (bugCount == null) {
286             int count = 0;
287             for (IMarker marker : bugGroup.getAllMarkers()) {
288                 if (isFiltered(marker)) {
289                     count++;
290                 }
291             }
292             bugCount = Integer.valueOf(count);
293             filteredMarkersMap.put(bugGroup, bugCount);
294         }
295         return bugCount.intValue();
296     }

318     /**
319      * @param desiredType
320      *            type of the child groups
321      * @param parents
322      *            all the common parents
323      * @param parent
324      *            "empty" root node, which must be filled in with markers from
325      *            set of given parents
326      * @return
327      */
328     private synchronized Object[] createChildren(GroupType desiredType, Set<IResource> parents, BugGroup parent) {
329         Set<IMarker> markerSet = new HashSet<>();
330         boolean filterActive = isBugFilterActive();
331         Set<String> patternFilter = getPatternFilter();
332         for (IResource resource : parents) {
333             IMarker[] markers = getMarkers(resource);
334             for (IMarker marker : markers) {
335                 boolean added = markerSet.add(marker);
336                 if (filterActive && added && MarkerUtil.isFiltered(marker, patternFilter)) {
337                     filteredMarkers.add(marker);
338                 }
339             }
340         }
341         parent.setMarkers(markerSet);
342         return createGroups(parent, desiredType.getMapper());
343     }

345     /**
346      * @param <Identifier>
347      *            object type, like String or Integer
348      * @param mapper
349      *            maps between BugInstance and group
350      */
351     private synchronized <Identifier> Object[] createGroups(BugGroup parent, MarkerMapper<Identifier> mapper) {
352         if (mapper == MarkerMapper.NO_MAPPING) {
353             return parent.getAllMarkers().toArray(new IMarker[0]);
354         }
355         Set<IMarker> allMarkers = parent.getAllMarkers();
356         GroupType childType = grouping.getChildType(mapper.getType());
357         Map<Identifier, Set<IMarker>> groupIds = new HashMap<>();
358         UserPreferences prefs = FindbugsPlugin.getCorePreferences(null, false);
359         Set<String> disabledPlugins = prefs.getCustomPlugins(false);
360 
361         // first, sort all bugs to the sets with same identifier type
362         Set<String> errorMessages = new HashSet<>();
363         for (IMarker marker : allMarkers) {
364             Identifier id = mapper.getIdentifier(marker);
365             if (id == null) {
366                 String pluginId = MarkerUtil.getPluginId(marker);
367                 if(pluginId.length() == 0 || disabledPlugins.contains(pluginId)){
368                     // do not report errors for disabled plugins
369                     continue;
370                 }
371                 try {
372                     String debugDescription = mapper.getDebugDescription(marker);
373                     if(errorMessages.contains(debugDescription)) {
374                         continue;
375                     }
376                     errorMessages.add(debugDescription);
377                     if(FindbugsPlugin.getDefault().isDebugging()){
378                         FindbugsPlugin.getDefault().logWarning(
379                             "BugContentProvider.createGroups: failed to find " + debugDescription + " for marker on file "
380                                     + marker.getResource());
381                     }
382                 } catch (CoreException e) {
383                     FindbugsPlugin.getDefault().logException(e, "Exception on retrieving debug data for: " + mapper.getType());
384                 }
385                 continue;
386             }
387             if (!groupIds.containsKey(id)) {
388                 groupIds.put(id, new HashSet<IMarker>());
389             }
390             groupIds.get(id).add(marker);
391         }
392 
393         // now create groups from the sorted bug sets
394         Set<Entry<Identifier, Set<IMarker>>> typesSet = groupIds.entrySet();
395         BugGroup[] children = new BugGroup[typesSet.size()];
396         boolean lastlevel = grouping.getChildType(mapper.getType()) == GroupType.Marker;
397         int i = 0;
398         for (Entry<Identifier, Set<IMarker>> entry : typesSet) {
399             Identifier groupId = entry.getKey();
400             children[i] = new BugGroup(parent, groupId, mapper.getType());
401             children[i].setMarkers(entry.getValue());
402             i++;
403         }
404         if (!lastlevel) {
405             for (BugGroup bugGroup : children) {
406                 // recursive call
407                 createGroups(bugGroup, childType.getMapper());
408             }
409         }
410         return children;
411     }

475     /**
476      * @return list of the *visible* parents with changed children to refresh
477      *         the viewer. Returns empty list if the full refresh is needed
478      */
479     public synchronized Set<BugGroup> updateContent(List<DeltaInfo> deltas) {
480         int oldRootSize = rootElement.getChildren().length;
481         Set<BugGroup> changedParents = new HashSet<>();
482         bugFilterActive = isBugFilterActive();
483         Set<String> patternFilter = getPatternFilter();
484         for (DeltaInfo delta : deltas) {
485             if (DEBUG) {
486                 System.out.println(delta + " (contentProvider.updateContent)");
487             }
488             IMarker changedMarker = delta.marker;
489             switch (delta.changeKind) {
490             case IResourceDelta.REMOVED:
491                 BugGroup parent = findParent(changedMarker);
492                 if (parent == null) {
493                     if (DEBUG) {
494                         System.out.println(delta + " IGNORED because marker does not have parent!");
495                     }
496                     continue;
497                 }
498                 removeMarker(parent, changedMarker, changedParents);
499                 break;
500             case IResourceDelta.ADDED:
501                 if (!changedMarker.exists()) {
502                     if (DEBUG) {
503                         System.out.println(delta + " IGNORED because marker does not exists anymore!");
504                     }
505                     continue;
506                 }
507                 addMarker(changedMarker, changedParents, patternFilter);
508                 break;
509             default:
510                     FindbugsPlugin.getDefault()
511                     .logWarning("UKNOWN delta change kind" + delta.changeKind);
512 
513             }
514         }
515         if (rootElement.getMarkersCount() == 0 || rootElement.getChildren().length != oldRootSize) {
516             if (oldRootSize == 0 || rootElement.getChildren().length == 0) {
517                 changedParents.clear();
518             } else {
519                 changedParents.add(rootElement);
520             }
521             return changedParents;
522         }
523         // XXX this is a fix for not updating of children on incremental build
524         // and
525         // for "empty" groups which can never be empty... I don't know where the
526         // bug is...
527         changedParents.add(rootElement);
528         return changedParents;
529     }

531     private void removeMarker(BugGroup parent, IMarker marker, Set<BugGroup> changedParents) {
532         BugGroup accessibleParent = getFirstAccessibleParent(parent);
533         changedParents.add(accessibleParent);
534         removeMarker(parent, marker);
535     }

537     private void addMarker(IMarker toAdd, Set<BugGroup> changedParents, Set<String> patternFilter) {
538         MarkerMapper<?> mapper = grouping.getFirstType().getMapper();
539         // filter through working set
540         IMarker marker = toAdd;
541         if (!resourceFilter.contains(marker.getResource())) {
542             return;
543         }
544         rootElement.addMarker(marker);
545         addMarker(marker, mapper, rootElement, changedParents, patternFilter);
546     }

548     private <Identifier> void addMarker(IMarker marker, MarkerMapper<Identifier> mapper, BugGroup parent,
549             Set<BugGroup> changedParents, Set<String> patternFilter) {
550 
551         if (mapper == MarkerMapper.NO_MAPPING) {
552             return;
553         }
554 
555         Identifier id = mapper.getIdentifier(marker);
556         if (id == null) {
557             if(FindbugsPlugin.getDefault().isDebugging()){
558                 FindbugsPlugin.getDefault().logWarning(
559                     "BugContentProvider.createPatternGroups: " + "Failed to find bug id of type " + mapper.getType()
560                             + " for marker on file " + marker.getResource());
561             }
562             return;
563         }
564 
565         // search for the right node to insert. there cannot be more then one
566         BugGroup matchingChild = null;
567         for (Object object : parent.getChildren()) {
568             if (!(object instanceof BugGroup)) {
569                 break;
570             }
571             BugGroup group = (BugGroup) object;
572             if (id.equals(group.getData())) {
573                 matchingChild = group;
574                 break;
575             }
576         }
577 
578         GroupType childType = grouping.getChildType(mapper.getType());
579         boolean filtered = bugFilterActive && MarkerUtil.isFiltered(marker, patternFilter);
580         if (filtered) {
581             filteredMarkers.add(marker);
582         }
583 
584         if (matchingChild != null) {
585             // node exists? fine, add marker and update children
586             matchingChild.addMarker(marker);
587             if (filtered) {
588                 Integer count = filteredMarkersMap.get(matchingChild);
589                 if (count == null) {
590                     count = Integer.valueOf(0);
591                 }
592                 filteredMarkersMap.put(matchingChild, Integer.valueOf(count.intValue() + 1));
593             }
594 
595             if (isAccessible(matchingChild)) {
596                 // update only first visible parent element to avoid multiple
597                 // refreshes
598                 changedParents.add(getFirstAccessibleParent(matchingChild));
599             }
600             boolean lastlevel = childType == GroupType.Marker;
601             if (!lastlevel) {
602                 addMarker(marker, childType.getMapper(), matchingChild, changedParents, patternFilter);
603             }
604         } else {
605             // if there is no node, create one and recursively all children to
606             // the last
607             // level
608             BugGroup group = new BugGroup(parent, id, mapper.getType());
609             group.addMarker(marker);
610             if (filtered) {
611                 filteredMarkersMap.put(group, Integer.valueOf(1));
612             }
613             createGroups(group, childType.getMapper());
614         }
615     }

617     /**
618      * @return true if the element is accessible from the content viewer point
619      *         of view. After "go into" action parent elements can became
620      *         inaccessible to the viewer
621      */
622     public boolean isAccessible(BugGroup group) {
623         BugGroup rootGroup;
624         if (!(input instanceof BugGroup)) {
625             rootGroup = rootElement;
626         } else {
627             rootGroup = (BugGroup) input;
628         }
629 
630         if (grouping.compare(rootGroup.getType(), group.getType()) < 0) {
631             return true;
632         }
633         return false;
634     }

636     private void removeMarker(BugGroup parent, IMarker marker) {
637         parent.removeMarker(marker);
638         List<BugGroup> parents = getSelfAndParents(parent);
639         boolean filtered = isFiltered(marker);
640         for (BugGroup group : parents) {
641             if (group.getMarkersCount() == 0 && group.getParent() instanceof BugGroup) {
642                 removeGroup((BugGroup) group.getParent(), group);
643             }
644             if (filtered) {
645                 Integer count = filteredMarkersMap.get(group);
646                 if (count != null && count.intValue() > 0) {
647                     filteredMarkersMap.put(group, Integer.valueOf(count.intValue() - 1));
648                 }
649             }
650         }
651         filteredMarkers.remove(marker);
652     }
653 
654     private void removeGroup(BugGroup parent, BugGroup child) {
655         if (parent.removeChild(child)) {
656             filteredMarkersMap.remove(child);
657             if (parent.getMarkersCount() == 0 && parent.getParent() instanceof BugGroup) {
658                 removeGroup((BugGroup) parent.getParent(), parent);
659             }
660         }
661     }
662 
663     private List<BugGroup> getSelfAndParents(BugGroup child) {
664         List<BugGroup> parents = new ArrayList<>();
665         parents.add(child);
666         while (child.getParent() instanceof BugGroup) {
667             child = (BugGroup) child.getParent();
668             parents.add(child);
669         }
670         return parents;
671     }

673     private BugGroup getFirstAccessibleParent(BugGroup element) {
674         if (element.getParent() instanceof BugGroup) {
675             BugGroup parent = (BugGroup) element.getParent();
676             if (!isAccessible(parent)) {
677                 return element;
678             }
679             return getFirstAccessibleParent(parent);
680         }
681         return element;
682     }

684     private BugGroup findParent(IMarker marker) {
685 
686         Object[] rootObjects = rootElement.getChildren();
687 
688         GroupType parentType = grouping.getParentType(GroupType.Marker);
689         if (parentType == GroupType.Undefined) {
690             return null;
691         }
692         // traverse through all groups looking for one with marker
693         for (int i = 0; i < rootObjects.length; i++) {
694             Object object = rootObjects[i];
695             if (!(object instanceof BugGroup)) {
696                 break;
697             }
698             BugGroup group = (BugGroup) object;
699             if (group.contains(marker)) {
700                 if (group.getType() == parentType) {
701                     return group;
702                 }
703                 rootObjects = group.getChildren();
704                 i = -1;
705             }
706         }
707         return null;
708     }

733     public Set<Object> getShowInTargets(Object obj) {
734         Set<Object> supported = new HashSet<>();
735         if (obj instanceof BugGroup) {
736             supported.add(obj);
737         } else if (obj instanceof IMarker) {
738             supported.add(obj);
739         } else if (obj instanceof IJavaProject) {
740             return getShowInTargets(((IJavaProject) obj).getProject());
741         } else if (obj instanceof IClassFile) {
742             return getShowInTargets(((IClassFile) obj).getType());
743         } else if (obj instanceof IFile) {
744             IJavaElement javaElement = JavaCore.create((IFile) obj);
745             return getShowInTargets(javaElement);
746         } else if (obj instanceof IFolder) {
747             IJavaElement javaElement = JavaCore.create((IFolder) obj);
748             return getShowInTargets(javaElement);
749         } else if (obj instanceof IStructuredSelection) {
750             IStructuredSelection selection = (IStructuredSelection) obj;
751             Iterator<?> iter = selection.iterator();
752             while (iter.hasNext()) {
753                 Object object = iter.next();
754                 supported.add(getShowInTargets(object));
755             }
756         } else {
757             // TODO think how improve performance for project/package objects?
758             Set<IMarker> markers = MarkerUtil.getMarkers(obj);
759             boolean found = false;
760             main: for (IMarker marker : markers) {
761                 BugGroup group = findParent(marker);
762                 if (group == null) {
763                     continue;
764                 }
765                 List<BugGroup> selfAndParents = getSelfAndParents(group);
766                 for (BugGroup bugGroup : selfAndParents) {
767                     if (obj.equals(bugGroup.getData())) {
768                         supported.add(bugGroup);
769                         found = true;
770                         break main;
771                     }
772                 }
773             }
774             if (!found) {
775                 supported.addAll(markers);
776             }
777         }
778         return supported;
779     }
