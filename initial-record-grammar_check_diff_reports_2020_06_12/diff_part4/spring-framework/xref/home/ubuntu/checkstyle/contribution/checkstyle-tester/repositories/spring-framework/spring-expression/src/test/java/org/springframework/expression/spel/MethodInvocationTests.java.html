86  	@Test
87  	public void testMethodThrowingException_SPR6760() {
88  		// Test method on inventor: throwException()
89  		// On 1 it will throw an IllegalArgumentException
90  		// On 2 it will throw a RuntimeException
91  		// On 3 it will exit normally
92  		// In each case it increments the Inventor field 'counter' when invoked
93  
94  		SpelExpressionParser parser = new SpelExpressionParser();
95  		Expression expr = parser.parseExpression("throwException(#bar)");
96  
97  		// Normal exit
98  		StandardEvaluationContext eContext = TestScenarioCreator.getTestEvaluationContext();
99  		eContext.setVariable("bar", 3);
100 		Object o = expr.getValue(eContext);
101 		assertEquals(o, 3);
102 		assertEquals(1, parser.parseExpression("counter").getValue(eContext));
103 
104 		// Now the expression has cached that throwException(int) is the right thing to call
105 		// Let's change 'bar' to be a PlaceOfBirth which indicates the cached reference is
106 		// out of date.
107 		eContext.setVariable("bar", new PlaceOfBirth("London"));
108 		o = expr.getValue(eContext);
109 		assertEquals("London", o);
110 		// That confirms the logic to mark the cached reference stale and retry is working
111 
112 		// Now let's cause the method to exit via exception and ensure it doesn't cause a retry.
113 
114 		// First, switch back to throwException(int)
115 		eContext.setVariable("bar", 3);
116 		o = expr.getValue(eContext);
117 		assertEquals(3, o);
118 		assertEquals(2, parser.parseExpression("counter").getValue(eContext));
119 
120 
121 		// Now cause it to throw an exception:
122 		eContext.setVariable("bar", 1);
123 		try {
124 			o = expr.getValue(eContext);
125 			fail();
126 		}
127 		catch (Exception ex) {
128 			if (ex instanceof SpelEvaluationException) {
129 				fail("Should not be a SpelEvaluationException: " + ex);
130 			}
131 			// normal
132 		}
133 		// If counter is 4 then the method got called twice!
134 		assertEquals(3, parser.parseExpression("counter").getValue(eContext));
135 
136 		eContext.setVariable("bar", 4);
137 		try {
138 			o = expr.getValue(eContext);
139 			fail();
140 		}
141 		catch (Exception ex) {
142 			// 4 means it will throw a checked exception - this will be wrapped
143 			if (!(ex instanceof ExpressionInvocationTargetException)) {
144 				fail("Should have been wrapped: " + ex);
145 			}
146 			// normal
147 		}
148 		// If counter is 5 then the method got called twice!
149 		assertEquals(4, parser.parseExpression("counter").getValue(eContext));
150 	}

202 	@Test
203 	public void testMethodFiltering_SPR6764() {
204 		SpelExpressionParser parser = new SpelExpressionParser();
205 		StandardEvaluationContext context = new StandardEvaluationContext();
206 		context.setRootObject(new TestObject());
207 		LocalFilter filter = new LocalFilter();
208 		context.registerMethodFilter(TestObject.class,filter);
209 
210 		// Filter will be called but not do anything, so first doit() will be invoked
211 		SpelExpression expr = (SpelExpression) parser.parseExpression("doit(1)");
212 		String result = expr.getValue(context, String.class);
213 		assertEquals("1", result);
214 		assertTrue(filter.filterCalled);
215 
216 		// Filter will now remove non @Anno annotated methods
217 		filter.removeIfNotAnnotated = true;
218 		filter.filterCalled = false;
219 		expr = (SpelExpression) parser.parseExpression("doit(1)");
220 		result = expr.getValue(context, String.class);
221 		assertEquals("double 1.0", result);
222 		assertTrue(filter.filterCalled);
223 
224 		// check not called for other types
225 		filter.filterCalled = false;
226 		context.setRootObject(new String("abc"));
227 		expr = (SpelExpression) parser.parseExpression("charAt(0)");
228 		result = expr.getValue(context, String.class);
229 		assertEquals("a", result);
230 		assertFalse(filter.filterCalled);
231 
232 		// check de-registration works
233 		filter.filterCalled = false;
234 		context.registerMethodFilter(TestObject.class,null);//clear filter
235 		context.setRootObject(new TestObject());
236 		expr = (SpelExpression) parser.parseExpression("doit(1)");
237 		result = expr.getValue(context, String.class);
238 		assertEquals("1", result);
239 		assertFalse(filter.filterCalled);
240 	}

320 	// Simple filter
321 	static class LocalFilter implements MethodFilter {
322 
323 		public boolean removeIfNotAnnotated = false;
324 
325 		public boolean filterCalled = false;
326 
327 		private boolean isAnnotated(Method method) {
328 			Annotation[] anns = method.getAnnotations();
329 			if (anns == null) {
330 				return false;
331 			}
332 			for (Annotation ann : anns) {
333 				String name = ann.annotationType().getName();
334 				if (name.endsWith("Anno")) {
335 					return true;
336 				}
337 			}
338 			return false;
339 		}
340 
341 		@Override
342 		public List<Method> filter(List<Method> methods) {
343 			filterCalled = true;
344 			List<Method> forRemoval = new ArrayList<Method>();
345 			for (Method method: methods) {
346 				if (removeIfNotAnnotated && !isAnnotated(method)) {
347 					forRemoval.add(method);
348 				}
349 			}
350 			for (Method method: forRemoval) {
351 				methods.remove(method);
352 			}
353 			return methods;
354 		}
355 	}
