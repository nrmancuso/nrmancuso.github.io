53  /**
54   * Builder for standard {@link ServiceRegistry} instances.
55   *
56   * @author Steve Ebersole
57   * 
58   * @see StandardServiceRegistryImpl
59   * @see BootstrapServiceRegistryBuilder
60   */
61  public class ServiceRegistryBuilder {
62  	private static final Logger log = Logger.getLogger( ServiceRegistryBuilder.class );
63  
64  	public static final String DEFAULT_CFG_RESOURCE_NAME = "hibernate.cfg.xml";
65  
66  	private final Map settings;
67  	private final List<BasicServiceInitiator> initiators = standardInitiatorList();
68  	private final List<ProvidedService> providedServices = new ArrayList<ProvidedService>();
69  	
70  	private boolean autoCloseRegistry = true;
71  
72  	private final BootstrapServiceRegistry bootstrapServiceRegistry;
73  
74  	/**
75  	 * Create a default builder
76  	 */
77  	public ServiceRegistryBuilder() {
78  		this( new BootstrapServiceRegistryImpl() );
79  	}
80  
81  	/**
82  	 * Create a builder with the specified bootstrap services.
83  	 *
84  	 * @param bootstrapServiceRegistry Provided bootstrap registry to use.
85  	 */
86  	public ServiceRegistryBuilder(BootstrapServiceRegistry bootstrapServiceRegistry) {
87  		this.settings = Environment.getProperties();
88  		this.bootstrapServiceRegistry = bootstrapServiceRegistry;
89  	}
90  
91  	/**
92  	 * Used from the {@link #initiators} variable initializer
93  	 *
94  	 * @return List of standard initiators
95  	 */
96  	private static List<BasicServiceInitiator> standardInitiatorList() {
97  		final List<BasicServiceInitiator> initiators = new ArrayList<BasicServiceInitiator>();
98  		initiators.addAll( StandardServiceInitiators.LIST );
99  		return initiators;
100 	}
101 
102 	/**
103 	 * Read settings from a {@link Properties} file.  Differs from {@link #configure()} and {@link #configure(String)}
104 	 * in that here we read a {@link Properties} file while for {@link #configure} we read the XML variant.
105 	 *
106 	 * @param resourceName The name by which to perform a resource look up for the properties file.
107 	 *
108 	 * @return this, for method chaining
109 	 *
110 	 * @see #configure()
111 	 * @see #configure(String)
112 	 */
113 	@SuppressWarnings( {"unchecked"})
114 	public ServiceRegistryBuilder loadProperties(String resourceName) {
115 		InputStream stream = bootstrapServiceRegistry.getService( ClassLoaderService.class ).locateResourceStream( resourceName );
116 		try {
117 			Properties properties = new Properties();
118 			properties.load( stream );
119 			settings.putAll( properties );
120 		}
121 		catch (IOException e) {
122 			throw new ConfigurationException( "Unable to apply settings from properties file [" + resourceName + "]", e );
123 		}
124 		finally {
125 			try {
126 				stream.close();
127 			}
128 			catch (IOException e) {
129 				log.debug(
130 						String.format( "Unable to close properties file [%s] stream", resourceName ),
131 						e
132 				);
133 			}
134 		}
135 
136 		return this;
137 	}
138 
139 	/**
140 	 * Read setting information from an XML file using the standard resource location
141 	 *
142 	 * @return this, for method chaining
143 	 *
144 	 * @see #DEFAULT_CFG_RESOURCE_NAME
145 	 * @see #configure(String)
146 	 * @see #loadProperties(String)
147 	 */
148 	public ServiceRegistryBuilder configure() {
149 		return configure( DEFAULT_CFG_RESOURCE_NAME );
150 	}
151 
152 	/**
153 	 * Read setting information from an XML file using the named resource location
154 	 *
155 	 * @param resourceName The named resource
156 	 *
157 	 * @return this, for method chaining
158 	 *
159 	 * @see #loadProperties(String)
160 	 */
161 	@SuppressWarnings( {"unchecked"})
162 	public ServiceRegistryBuilder configure(String resourceName) {
163 		InputStream stream = bootstrapServiceRegistry.getService( ClassLoaderService.class ).locateResourceStream( resourceName );
164 		JaxbHibernateConfiguration configurationElement = jaxbProcessorHolder.getValue().unmarshal(
165 				stream,
166 				new Origin( SourceType.RESOURCE, resourceName )
167 		);
168 		for ( JaxbHibernateConfiguration.JaxbSessionFactory.JaxbProperty xmlProperty : configurationElement.getSessionFactory().getProperty() ) {
169 			settings.put( xmlProperty.getName(), xmlProperty.getValue() );
170 		}
171 
172 		return this;
173 	}
174 
175 	private ValueHolder<JaxbProcessor> jaxbProcessorHolder = new ValueHolder<JaxbProcessor>(
176 			new ValueHolder.DeferredInitializer<JaxbProcessor>() {
177 				@Override
178 				public JaxbProcessor initialize() {
179 					return new JaxbProcessor( bootstrapServiceRegistry.getService( ClassLoaderService.class ) );
180 				}
181 			}
182 	);
183 
184 	/**
185 	 * Apply a setting value
186 	 *
187 	 * @param settingName The name of the setting
188 	 * @param value The value to use.
189 	 *
190 	 * @return this, for method chaining
191 	 */
192 	@SuppressWarnings( {"unchecked", "UnusedDeclaration"})
193 	public ServiceRegistryBuilder applySetting(String settingName, Object value) {
194 		settings.put( settingName, value );
195 		return this;
196 	}
197 
198 	/**
199 	 * Apply a groups of setting values
200 	 *
201 	 * @param settings The incoming settings to apply
202 	 *
203 	 * @return this, for method chaining
204 	 */
205 	@SuppressWarnings( {"unchecked", "UnusedDeclaration"})
206 	public ServiceRegistryBuilder applySettings(Map settings) {
207 		this.settings.putAll( settings );
208 		return this;
209 	}
210 
211 	/**
212 	 * Adds a service initiator.
213 	 *
214 	 * @param initiator The initiator to be added
215 	 *
216 	 * @return this, for method chaining
217 	 */
218 	@SuppressWarnings( {"UnusedDeclaration"})
219 	public ServiceRegistryBuilder addInitiator(BasicServiceInitiator initiator) {
220 		initiators.add( initiator );
221 		return this;
222 	}
223 
224 	/**
225 	 * Adds a user-provided service
226 	 *
227 	 * @param serviceRole The role of the service being added
228 	 * @param service The service implementation
229 	 *
230 	 * @return this, for method chaining
231 	 */
232 	@SuppressWarnings( {"unchecked"})
233 	public ServiceRegistryBuilder addService(final Class serviceRole, final Service service) {
234 		providedServices.add( new ProvidedService( serviceRole, service ) );
235 		return this;
236 	}
237 	/**
238 	* By default, when a ServiceRegistry is no longer referenced by any other
239 	* registries as a parent it will be closed.
240 	* <p/>
241 	* Some applications that explicitly build "shared registries" may want to
242 	* circumvent that behavior.
243 	* <p/>
244 	* This method indicates that the registry being built should not be
245 	* automatically closed. The caller agrees to take responsibility to
246 	* close it themselves.
247 	*
248 	* @return this, for method chaining
249 	*/
250 	public ServiceRegistryBuilder disableAutoClose() {
251 		this.autoCloseRegistry = false;
252 		return this;
253 	}
254 
255 	/**
256 	* See the discussion on {@link #disableAutoClose}. This method enables
257 	* the auto-closing.
258 	*
259 	* @return this, for method chaining
260 	*/
261 	public ServiceRegistryBuilder enableAutoClose() {
262 		this.autoCloseRegistry = true;
263 		return this;
264 	}
265 
266 	/**
267 	 * Build the service registry accounting for all settings and service initiators and services.
268 	 *
269 	 * @return The built service registry
270 	 */
271 	public ServiceRegistry buildServiceRegistry() {
272 		Map<?,?> settingsCopy = new HashMap();
273 		settingsCopy.putAll( settings );
274 		Environment.verifyProperties( settingsCopy );
275 		ConfigurationHelper.resolvePlaceHolders( settingsCopy );
276 
277 		for ( Integrator integrator : bootstrapServiceRegistry.getService( IntegratorService.class ).getIntegrators() ) {
278 			if ( ServiceContributingIntegrator.class.isInstance( integrator ) ) {
279 				ServiceContributingIntegrator.class.cast( integrator ).prepareServices( this );
280 			}
281 		}
282 
283 		return new StandardServiceRegistryImpl( autoCloseRegistry, bootstrapServiceRegistry, initiators, providedServices, settingsCopy );
284 	}
285 
286 	/**
287 	 * Destroy a service registry.  Applications should only destroy registries they have explicitly created.
288 	 *
289 	 * @param serviceRegistry The registry to be closed.
290 	 */
291 	public static void destroy(ServiceRegistry serviceRegistry) {
292 		( (StandardServiceRegistryImpl) serviceRegistry ).destroy();
293 	}
