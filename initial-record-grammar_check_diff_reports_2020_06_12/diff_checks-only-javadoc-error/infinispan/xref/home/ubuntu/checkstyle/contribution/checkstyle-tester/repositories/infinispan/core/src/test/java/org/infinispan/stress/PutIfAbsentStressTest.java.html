20  /**
21   * Verifies the atomic semantic of Infinispan's implementations of
22   * java.util.concurrent.ConcurrentMap<K, V>.putIfAbsent(K key, V value); which is an interesting
23   * concurrent locking case.
24   *
25   * @since 4.0
26   * @see java.util.concurrent.ConcurrentMap#putIfAbsent(Object, Object)
27   * @author Sanne Grinovero
28   */

30        description = "Since this test is slow to run, it should be disabled by default and run by hand as necessary.")

103    /**
104     * Adapter to run the test on any configuration
105     */
106    private void testConcurrentLockingOnMultipleManagers(ConfigurationBuilder cfg) throws InterruptedException {
107       List<EmbeddedCacheManager> cacheContainers = new ArrayList<EmbeddedCacheManager>(NODES_NUM);
108       List<Cache<String, String>> caches = new ArrayList<Cache<String, String>>();
109       List<ConcurrentMap<String, String>> maps = new ArrayList<ConcurrentMap<String, String>>(NODES_NUM
110                * THREAD_PER_NODE);
111       for (int nodeNum = 0; nodeNum < NODES_NUM; nodeNum++) {
112          EmbeddedCacheManager cm = TestCacheManagerFactory.createClusteredCacheManager(cfg);
113          cacheContainers.add(cm);
114          Cache<String, String> cache = cm.getCache();
115          caches.add(cache);
116          for (int threadNum = 0; threadNum < THREAD_PER_NODE; threadNum++) {
117             maps.add(cache);
118          }
119       }
120       TestingUtil.blockUntilViewsReceived(10000, caches);
121       try {
122          testConcurrentLocking(maps);
123       } finally {
124          TestingUtil.killCacheManagers(cacheContainers);
125       }
126    }

128    /**
129     * Adapter for tests sharing a single Cache instance
130     */
131    private void testConcurrentLocking(ConcurrentMap<String, String> map) throws InterruptedException {
132       int size = NODES_NUM * THREAD_PER_NODE;
133       List<ConcurrentMap<String, String>> maps = new ArrayList<ConcurrentMap<String, String>>(size);
134       for (int i = 0; i < size; i++) {
135          maps.add(map);
136       }
137       testConcurrentLocking(maps);
138    }

140    /**
141     * Drives the actual test on an Executor and verifies the result
142     *
143     * @param maps the caches to be tested
144     */
145    private void testConcurrentLocking(List<ConcurrentMap<String, String>> maps) throws InterruptedException {
146       SharedStats stats = new SharedStats();
147       ThreadPoolExecutor executor = (ThreadPoolExecutor) Executors.newFixedThreadPool(NODES_NUM);
148       List<StressingThread> threads = new ArrayList<StressingThread>();
149       int i=0;
150       for (ConcurrentMap<String, String> map : maps) {
151          StressingThread thread = new StressingThread(stats, map, i++);
152          threads.add(thread);
153          executor.execute(thread);
154       }
155       executor.shutdown();
156       Thread.sleep(5000);
157       int putsAfter5Seconds = stats.succesfullPutsCounter.get();
158       System.out.println("\nSituation after 5 seconds:");
159       System.out.println(stats.toString());
160       executor.awaitTermination(STRESS_TIME_MINUTES, TimeUnit.MINUTES);
161       stats.globalQuit = true;
162       executor.awaitTermination(10, TimeUnit.SECONDS); // give some time to awake and quit
163       executor.shutdownNow();
164       System.out.println("\nFinal situation:");
165       System.out.println(stats.toString());
166       assert !stats.seenFailures : "at least one thread has seen unexpected state";
167       assert stats.succesfullPutsCounter.get() > 0 : "the lock should have been taken at least once";
168       assert stats.succesfullPutsCounter.get() > putsAfter5Seconds : "the lock count didn't improve since the first 5 seconds. Deadlock?";
169       assert stats.succesfullPutsCounter.get() == stats.lockReleasedCounter.get() : "there's a mismatch in acquires and releases count";
170       assert stats.lockOwnersCounter.get() == 0 : "the lock is still held at test finish";
171    }
172 
173    private static class StressingThread implements Runnable {
174 
175       private final SharedStats stats;
176       private final ConcurrentMap<String, String> cache;
177       private final String ourValue;
178 
179       public StressingThread(SharedStats stats, ConcurrentMap<String, String> cache, int threadId) {
180          this.stats = stats;
181          this.cache = cache;
182          this.ourValue = "v" + threadId;
183       }
184 
185       @Override
186       public void run() {
187          while (!(stats.seenFailures || stats.globalQuit || Thread.interrupted())) {
188                doCycle();
189          }
190       }
191 
192       private void doCycle() {
193          String beforePut = cache.putIfAbsent(SHARED_KEY, ourValue);
194          if (beforePut != null) {
195             stats.canceledPutsCounter.incrementAndGet();
196          } else {
197             final String currentCacheValue = cache.get(SHARED_KEY);
198             boolean lockIsFine = stats.lockOwnersCounter.compareAndSet(0, 1) && ourValue.equals(currentCacheValue);
199             stats.succesfullPutsCounter.incrementAndGet();
200             checkIsTrue(lockIsFine, "I got the lock, some other thread is owning the lock AS WELL.");
201             lockIsFine = stats.lockOwnersCounter.compareAndSet(1, 0);
202             checkIsTrue(lockIsFine, "Some other thread changed the lock count while I was having it!");
203             cache.remove(SHARED_KEY);
204             stats.lockReleasedCounter.incrementAndGet();
205          }
206       }
207 
208       private void checkIsTrue(boolean assertion, String message) {
209          if (assertion == false) {
210             stats.seenFailures = true;
211             System.out.println(message);
212          }
213       }
214 
215    }
216 
217    /**
218     * Common state to verify cache behaviour
219     */
220    public static class SharedStats {
221 
222       final AtomicInteger canceledPutsCounter = new AtomicInteger(0);
223       final AtomicInteger succesfullPutsCounter = new AtomicInteger(0);
224       final AtomicInteger lockReleasedCounter = new AtomicInteger(0);
225       final AtomicInteger lockOwnersCounter = new AtomicInteger(0);
226       Throwable throwable = null;
227       volatile boolean globalQuit = false; // when it's true the threads quit
228       volatile boolean seenFailures = false; // set to true by a thread if it has experienced
229                                              // illegal state
230 
231       @Override
232       public String toString() {
233          return "\n\tCanceled puts count:\t" + canceledPutsCounter.get() +
234                 "\n\tSuccesfull puts count:\t" + succesfullPutsCounter.get() +
235                 "\n\tRemoved count:\t" + lockReleasedCounter.get() +
236                 "\n\tIllegal state detected:\t" + seenFailures;
237       }
238 
239    }
