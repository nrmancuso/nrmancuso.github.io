63  public class CompressionRules {
64  
65      static final Logger LOG = Logger.getLogger(CompressionRules.class);
66  
67      static final int NONE = 0;
68  
69      static final int J2LL = 1;
70  
71      static final int JLSL = 2;
72  
73      static final int J2KR = 3;
74  
75      static final int JPLY = 4;
76  
77      static final String[] CODES = { "NONE", "JPLL", "JLSL", "J2KR", "JPLY" };
78  
79      static final String[] TSUIDS = { null, UIDs.JPEGLossless,
80              UIDs.JPEGLSLossless, UIDs.JPEG2000Lossless, UIDs.JPEGBaseline};
81  
82      private final ArrayList list = new ArrayList();
83  
84      private static final class Entry {
85  
86          final Condition condition;
87          final int compression;
88          final float quality;
89          final String derivationDescription;
90          final float ratio;
91          
92          Entry(Condition condition, int compression, float quality,
93                  String derivationDescription, float ratio) {
94              this.condition = condition;
95              this.compression = compression;
96              this.quality = quality;
97              this.derivationDescription = derivationDescription;
98              this.ratio = ratio;
99         }
100     }
101 
102     public CompressionRules(String s) {
103         StringTokenizer stk = new StringTokenizer(s, "\r\n;");
104         while (stk.hasMoreTokens()) {
105             String tk = stk.nextToken().trim();
106             if (tk.length() == 0)
107                 continue;
108             try {
109                 int endCond = tk.indexOf(']') + 1;
110                 Condition cond = new Condition(tk.substring(0, endCond));
111                 String codec = tk.substring(endCond);
112                 int compression;
113                 float quality = 0.75f;
114                 float ratio = 5.f;
115                 String derivationDescription = "JPEG Lossy Compression with quality=0.75";
116                 if (codec.startsWith("JPLY(")) {
117                     if (!codec.endsWith(")"))
118                         throw new IllegalArgumentException();
119                     int endQuality = codec.indexOf(':');
120                     if (endQuality == -1)
121                         throw new IllegalArgumentException();
122                     int endDesc = codec.indexOf(':',endQuality+1); 
123                     if (endDesc == -1)
124                         throw new IllegalArgumentException();
125                     compression = JPLY;
126                     quality = Float.parseFloat(codec.substring(5, endQuality));
127                     derivationDescription =
128                         codec.substring(endQuality + 1, endDesc);
129                     ratio =  Float.parseFloat(
130                             codec.substring(endDesc + 1, codec.length()-1));
131                 } else {
132                     compression = Arrays.asList(CODES).indexOf(codec);
133                     if (compression == -1)
134                         throw new IllegalArgumentException();
135                 }
136                 list.add(new Entry(cond, compression, quality, derivationDescription,
137                         ratio));
138             } catch (IllegalArgumentException e) {
139                 throw new IllegalArgumentException(tk);
140             }
141         }
142     }
143 
144     public CompressCmd getCompressFor(Association assoc, Dataset ds) {
145         Map param = new HashMap();
146         param.put("calling", new String[] { assoc.getCallingAET() });
147         param.put("called", new String[] { assoc.getCalledAET() });
148         if (ds != null) {
149             putIntoIfNotNull(param, "cuid", ds, Tags.SOPClassUID);
150             putIntoIfNotNull(param, "pmi", ds, Tags.PhotometricInterpretation);
151             putIntoIfNotNull(param, "imgtype", ds, Tags.ImageType);
152             putIntoIfNotNull(param, "bodypart", ds, Tags.BodyPartExamined);
153         }
154         for (Iterator it = list.iterator(); it.hasNext();) {
155             Entry e = (Entry) it.next();
156             if (e.condition.isTrueFor(param)) {
157                 try {
158                     return (e.compression == NONE)? null : (e.compression == JPLY) 
159                             ? CompressCmd.createJPEGLossyCompressCmd(
160                                     ds, e.quality, e.derivationDescription,
161                                     e.ratio, null, null)
162                             : CompressCmd.createCompressCmd(
163                                     ds, TSUIDS[e.compression]);
164                 } catch (CompressionFailedException e1) {
165                     LOG.info(e1.getMessage());
166                     continue;
167                 }
168             }
169         }
170         return null;
171     }
172 
173     private void putIntoIfNotNull(Map param, String key, Dataset ds, int tag) {
174         String[] val = ds.getStrings(tag);
175         if (val != null && val.length != 0) {
176             param.put(key, val);
177         }
178     }
179 
180     public String toString() {
181         final String newline = System.getProperty("line.separator", "\n");
182         if (list.isEmpty())
183             return newline;
184         StringBuffer sb = new StringBuffer();
185         for (Iterator it = list.iterator(); it.hasNext();) {
186             Entry e = (Entry) it.next();
187             e.condition.toStringBuffer(sb);
188             sb.append(CODES[e.compression]);
189             if (e.compression == JPLY) {
190                 sb.append('(').append(e.quality)
191                   .append(':').append(e.derivationDescription)
192                   .append(':').append(e.ratio)
193                   .append(')');
194             }
195             sb.append(newline);
196         }
197         return sb.toString();
198     }
