93     public final Object visitGetKeysInGroupCommand(InvocationContext ctx, GetKeysInGroupCommand command) throws Throwable {
94        final String groupName = command.getGroupName();
95        if (command.isGroupOwner()) {
96           //don't go remote if we are an owner.
97           return invokeNextInterceptor(ctx, command);
98        }
99        Map<Address, Response> responseMap = rpcManager.invokeRemotely(Collections.singleton(groupManager.getPrimaryOwner(groupName)), command,
100                                                                      rpcManager.getDefaultRpcOptions(true));
101       if (!responseMap.isEmpty()) {
102          Response response = responseMap.values().iterator().next();
103          if (response instanceof SuccessfulResponse) {
104             //noinspection unchecked
105             List<CacheEntry> cacheEntries = (List<CacheEntry>) ((SuccessfulResponse) response).getResponseValue();
106             for (CacheEntry entry : cacheEntries) {
107                entryFactory.wrapEntryForReading(ctx, entry.getKey(), entry);
108             }
109          }
110       }
111       return invokeNextInterceptor(ctx, command);
112    }

116       if (ctx.isOriginLocal() && !isLocalModeForced(command)) {
117          rpcManager.invokeRemotely(null, command, rpcManager.getDefaultRpcOptions(isSynchronous(command)));
118       }
119       return invokeNextInterceptor(ctx, command);

122    @Override
123    protected final InternalCacheEntry retrieveFromRemoteSource(Object key, InvocationContext ctx, boolean acquireRemoteLock, FlagAffectedCommand command, boolean isWrite) throws Exception {
124       GlobalTransaction gtx = ctx.isInTxScope() ? ((TxInvocationContext)ctx).getGlobalTransaction() : null;
125       ClusteredGetCommand get = cf.buildClusteredGetCommand(key, command.getFlags(), acquireRemoteLock, gtx);
126       get.setWrite(isWrite);
127 
128       RpcOptionsBuilder rpcOptionsBuilder = rpcManager.getRpcOptionsBuilder(ResponseMode.WAIT_FOR_VALID_RESPONSE, DeliverOrder.NONE);
129       int lastTopologyId = -1;
130       InternalCacheEntry value = null;
131       while (value == null) {
132          final CacheTopology cacheTopology = stateTransferManager.getCacheTopology();
133          final int currentTopologyId = cacheTopology.getTopologyId();
134 
135          if (trace) {
136             log.tracef("Perform remote get for key %s. topologyId=%s, currentTopologyId=%s",
137                        key, lastTopologyId, currentTopologyId);
138          }
139          List<Address> targets;
140          if (lastTopologyId < currentTopologyId) {
141             // Cache topology has changed or it is the first time.
142             lastTopologyId = currentTopologyId;
143             targets = new ArrayList<>(cacheTopology.getReadConsistentHash().locateOwners(key));
144          } else if (lastTopologyId == currentTopologyId && cacheTopology.getPendingCH() != null) {
145             // Same topologyId, but the owners could have already installed the next topology
146             // Lets try with pending consistent owners (the read owners in the next topology)
147             lastTopologyId = currentTopologyId + 1;
148             targets = new ArrayList<>(cacheTopology.getPendingCH().locateOwners(key));
149             // Remove already contacted nodes
150             targets.removeAll(cacheTopology.getReadConsistentHash().locateOwners(key));
151             if (targets.isEmpty()) {
152                if (trace) {
153                   log.tracef("No valid values found for key '%s' (topologyId=%s).", key, currentTopologyId);
154                }
155                break;
156             }
157          } else { // lastTopologyId > currentTopologyId || cacheTopology.getPendingCH() == null
158             // We have not received a valid value from the pending CH owners either, and the topology id hasn't changed
159             if (trace) {
160                log.tracef("No valid values found for key '%s' (topologyId=%s).", key, currentTopologyId);
161             }
162             break;
163          }
164 
165          value = invokeClusterGetCommandRemotely(targets, rpcOptionsBuilder, get, key);
166          if (trace) {
167             log.tracef("Remote get of key '%s' (topologyId=%s) returns %s", key, currentTopologyId, value);
168          }
169       }
170       return value;
171    }

173    private InternalCacheEntry invokeClusterGetCommandRemotely(List<Address> targets, RpcOptionsBuilder rpcOptionsBuilder,
174                                                       ClusteredGetCommand get, Object key) {
175       ResponseFilter filter = new ClusteredGetResponseValidityFilter(targets, rpcManager.getAddress());
176       RpcOptions options = rpcOptionsBuilder.responseFilter(filter).build();
177       Map<Address, Response> responses = rpcManager.invokeRemotely(targets, get, options);
178 
179       if (!responses.isEmpty()) {
180          for (Response r : responses.values()) {
181             if (r instanceof SuccessfulResponse) {
182 
183                // The response value might be null.
184                SuccessfulResponse response = (SuccessfulResponse) r;
185                Object responseValue = response.getResponseValue();
186                if (responseValue == null) {
187                   continue;
188                }
189 
190                InternalCacheValue cacheValue = (InternalCacheValue) responseValue;
191                InternalCacheEntry ice = cacheValue.toInternalCacheEntry(key);
192                if (rvrl != null) {
193                   rvrl.remoteValueFound(ice);
194                }
195                return ice;
196             }
197          }
198       }
199       if (rvrl != null) {
200          rvrl.remoteValueNotFound(key);
201       }
202       return null;
203    }

205    protected Map<Object, InternalCacheEntry> retrieveFromRemoteSources(Set<?> requestedKeys, InvocationContext ctx, Set<Flag> flags) throws Throwable {
206       GlobalTransaction gtx = ctx.isInTxScope() ? ((TxInvocationContext)ctx).getGlobalTransaction() : null;
207       CacheTopology cacheTopology = stateTransferManager.getCacheTopology();
208       ConsistentHash ch = cacheTopology.getReadConsistentHash();
209 
210       Map<Address, List<Object>> ownerKeys = new HashMap<>();
211       for (Object key : requestedKeys) {
212          Address owner = ch.locatePrimaryOwner(key);
213          List<Object> requestedKeysFromNode = ownerKeys.get(owner);
214          if (requestedKeysFromNode == null) {
215             ownerKeys.put(owner, requestedKeysFromNode = new ArrayList<>());
216          }
217          requestedKeysFromNode.add(key);
218       }
219 
220       Map<Address, ReplicableCommand> commands = new HashMap<>();
221       for (Map.Entry<Address, List<Object>> entry : ownerKeys.entrySet()) {
222          List<Object> keys = entry.getValue();
223          ClusteredGetAllCommand remoteGetAll = cf.buildClusteredGetAllCommand(keys, flags, gtx);
224          commands.put(entry.getKey(), remoteGetAll);
225       }
226 
227       RpcOptionsBuilder rpcOptionsBuilder = rpcManager.getRpcOptionsBuilder(
228             ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS, DeliverOrder.NONE);
229       RpcOptions options = rpcOptionsBuilder.build();
230       Map<Address, Response> responses = rpcManager.invokeRemotely(commands, options);
231 
232       Map<Object, InternalCacheEntry> entries = new HashMap<>();
233       for (Map.Entry<Address, Response> entry : responses.entrySet()) {
234          updateWithValues(((ClusteredGetAllCommand) commands.get(entry.getKey())).getKeys(),
235                entry.getValue(), entries);
236       }
237 
238       return entries;
239    }

241    private void updateWithValues(List<?> keys, Response r, Map<Object, InternalCacheEntry> entries) {
242       if (r instanceof SuccessfulResponse) {
243          SuccessfulResponse response = (SuccessfulResponse) r;
244          List<InternalCacheValue> values = (List<InternalCacheValue>) response.getResponseValue();
245          // Only process if we got a return value - this can happen if the node is shutting
246          // down when it received the request
247          if (values != null) {
248             for (int i = 0; i < keys.size(); ++i) {
249                InternalCacheValue icv = values.get(i);
250                if (icv != null) {
251                   Object key = keys.get(i);
252                   Object value = icv.getValue();
253                   if (value == null) {
254                      entries.put(key, null);
255                   } else {
256                      InternalCacheEntry ice = icv.toInternalCacheEntry(key);
257                      entries.put(key, ice);
258                   }
259                }
260             }
261          }
262       }
263    }

265    protected final Object handleNonTxWriteCommand(InvocationContext ctx, DataWriteCommand command) throws Throwable {
266       if (ctx.isInTxScope()) {
267          throw new CacheException("Attempted execution of non-transactional write command in a transactional invocation context");
268       }
269 
270       RecipientGenerator recipientGenerator = new SingleKeyRecipientGenerator(command.getKey());
271 
272       // see if we need to load values from remote sources first
273       if (needValuesFromPreviousOwners(ctx, command)) {
274          remoteGetBeforeWrite(ctx, command, recipientGenerator);
275       }
276 
277       // invoke the command locally, we need to know if it's successful or not
278       Object localResult = invokeNextInterceptor(ctx, command);
279 
280       // if this is local mode then skip distributing
281       if (isLocalModeForced(command)) {
282          return localResult;
283       }
284 
285 
286       boolean isSync = isSynchronous(command);
287       Address primaryOwner = cdl.getPrimaryOwner(command.getKey());
288       int commandTopologyId = command.getTopologyId();
289       int currentTopologyId = stateTransferManager.getCacheTopology().getTopologyId();
290       // TotalOrderStateTransferInterceptor doesn't set the topology id for PFERs.
291       // TODO Shouldn't PFERs be executed in a tx with total order?
292       boolean topologyChanged = isSync && currentTopologyId != commandTopologyId && commandTopologyId != -1;
293       if (trace) {
294          log.tracef("Command topology id is %d, current topology id is %d, successful? %s",
295                (Object)commandTopologyId, currentTopologyId, command.isSuccessful());
296       }
297       // We need to check for topology changes on the origin even if the command was unsuccessful
298       // otherwise we execute the command on the correct primary owner, and then we still
299       // throw an OutdatedTopologyInterceptor when we return in EntryWrappingInterceptor.
300       if (topologyChanged) {
301          throw new OutdatedTopologyException("Cache topology changed while the command was executing: expected " +
302                commandTopologyId + ", got " + currentTopologyId);
303       }
304 
305       ValueMatcher valueMatcher = command.getValueMatcher();
306       if (!ctx.isOriginLocal()) {
307          if (primaryOwner.equals(rpcManager.getAddress())) {
308             if (!command.isSuccessful()) {
309                log.tracef("Skipping the replication of the conditional command as it did not succeed on primary owner (%s).", command);
310                return localResult;
311             }
312             List<Address> recipients = recipientGenerator.generateRecipients();
313             // Ignore the previous value on the backup owners
314             command.setValueMatcher(ValueMatcher.MATCH_ALWAYS);
315             try {
316                rpcManager.invokeRemotely(recipients, command, determineRpcOptionsForBackupReplication(rpcManager,
317                                                                                                       isSync, recipients));
318             } finally {
319                // Switch to the retry policy, in case the primary owner changed and the write already succeeded on the new primary
320                command.setValueMatcher(valueMatcher.matcherForRetry());
321             }
322          }
323          return localResult;
324       } else {
325          if (primaryOwner.equals(rpcManager.getAddress())) {
326             if (!command.isSuccessful()) {
327                log.tracef("Skipping the replication of the command as it did not succeed on primary owner (%s).", command);
328                return localResult;
329             }
330             List<Address> recipients = recipientGenerator.generateRecipients();
331             log.tracef("I'm the primary owner, sending the command to all the backups (%s) in order to be applied.",
332                   recipients);
333             // check if a single owner has been configured and the target for the key is the local address
334             boolean isSingleOwnerAndLocal = cacheConfiguration.clustering().hash().numOwners() == 1;
335             if (!isSingleOwnerAndLocal) {
336                // Ignore the previous value on the backup owners
337                command.setValueMatcher(ValueMatcher.MATCH_ALWAYS);
338                try {
339                   rpcManager.invokeRemotely(recipients, command, determineRpcOptionsForBackupReplication(rpcManager,
340                                                                                                          isSync, recipients));
341                } finally {
342                   // Switch to the retry policy, in case the primary owner changed and the write already succeeded on the new primary
343                   command.setValueMatcher(valueMatcher.matcherForRetry());
344                }
345             }
346             return localResult;
347          } else {
348             log.tracef("I'm not the primary owner, so sending the command to the primary owner(%s) in order to be forwarded", primaryOwner);
349             boolean isSyncForwarding = isSync || isNeedReliableReturnValues(command);
350 
351             Map<Address, Response> addressResponseMap;
352             try {
353                addressResponseMap = rpcManager.invokeRemotely(Collections.singletonList(primaryOwner), command,
354                      rpcManager.getDefaultRpcOptions(isSyncForwarding));
355             } catch (RemoteException e) {
356                Throwable ce = e;
357                while (ce instanceof RemoteException) {
358                   ce = ce.getCause();
359                }
360                if (ce instanceof OutdatedTopologyException) {
361                   // If the primary owner throws an OutdatedTopologyException, it must be because the command succeeded there
362                   if (trace) log.tracef("Changing the value matching policy from %s to %s (original value was %s)",
363                         command.getValueMatcher(), valueMatcher.matcherForRetry(), valueMatcher);
364                   command.setValueMatcher(valueMatcher.matcherForRetry());
365                }
366                throw e;
367             } catch (SuspectException e) {
368                // If the primary owner became suspected, we don't know if it was able to replicate it's data properly
369                // to all backup owners and notify all listeners, thus we need to retry with new matcher in case if
370                // it had updated the backup owners
371                if (trace) log.tracef("Primary owner suspected - Changing the value matching policy from %s to %s " +
372                                            "(original value was %s)", command.getValueMatcher(),
373                                      valueMatcher.matcherForRetry(), valueMatcher);
374                command.setValueMatcher(valueMatcher.matcherForRetry());
375                throw e;
376             }
377             if (!isSyncForwarding) return localResult;
378 
379             Object primaryResult = getResponseFromPrimaryOwner(primaryOwner, addressResponseMap);
380             command.updateStatusFromRemoteResponse(primaryResult);
381             return primaryResult;
382          }
383       }
384    }

386    private RpcOptions determineRpcOptionsForBackupReplication(RpcManager rpc, boolean isSync, List<Address> recipients) {
387       RpcOptions options;
388       if (isSync) {
389          // If no recipients, means a broadcast, so we can ignore leavers
390          if (recipients == null) {
391             options = rpc.getRpcOptionsBuilder(ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS).build();
392          } else {
393             options = rpc.getDefaultRpcOptions(true);
394          }
395       } else {
396          options = rpc.getDefaultRpcOptions(false);
397       }
398       return options;
399    }

401    private Object getResponseFromPrimaryOwner(Address primaryOwner, Map<Address, Response> addressResponseMap) {
402       Response fromPrimaryOwner = addressResponseMap.get(primaryOwner);
403       if (fromPrimaryOwner == null) {
404          log.tracef("Primary owner %s returned null", primaryOwner);
405          return null;
406       }
407       if (fromPrimaryOwner.isSuccessful()) {
408          return ((SuccessfulResponse) fromPrimaryOwner).getResponseValue();
409       }
410 
411       if (addressResponseMap.get(primaryOwner) instanceof CacheNotFoundResponse) {
412          // This means the cache wasn't running on the primary owner, so the command wasn't executed.
413          // We throw an OutdatedTopologyException, StateTransferInterceptor will catch the exception and
414          // it will then retry the command.
415          throw new OutdatedTopologyException("Cache is no longer running on primary owner " + primaryOwner);
416       }
417 
418       Throwable cause = fromPrimaryOwner instanceof ExceptionResponse ? ((ExceptionResponse)fromPrimaryOwner).getException() : null;
419       throw new CacheException("Got unsuccessful response from primary owner: " + fromPrimaryOwner, cause);
420    }

422    @Override
423    public Object visitGetAllCommand(InvocationContext ctx, GetAllCommand command) throws Throwable {
424       if (command.hasFlag(Flag.CACHE_MODE_LOCAL)
425             || command.hasFlag(Flag.SKIP_REMOTE_LOOKUP)
426             || command.hasFlag(Flag.IGNORE_RETURN_VALUES)) {
427          return invokeNextInterceptor(ctx, command);
428       }
429 
430       int commandTopologyId = command.getTopologyId();
431       if (ctx.isOriginLocal()) {
432          int currentTopologyId = stateTransferManager.getCacheTopology().getTopologyId();
433          boolean topologyChanged = currentTopologyId != commandTopologyId && commandTopologyId != -1;
434          if (trace) {
435             log.tracef("Command topology id is %d, current topology id is %d", commandTopologyId, currentTopologyId);
436          }
437          if (topologyChanged) {
438             throw new OutdatedTopologyException("Cache topology changed while the command was executing: expected " +
439                   commandTopologyId + ", got " + currentTopologyId);
440          }
441 
442          // At this point, we know that an entry located on this node that exists in the data container/store
443          // must also exist in the context.
444          ConsistentHash ch = command.getConsistentHash();
445          Set<Object> requestedKeys = new HashSet<>();
446          for (Object key : command.getKeys()) {
447             CacheEntry entry = ctx.lookupEntry(key);
448             if (entry == null) {
449                if (!isValueAvailableLocally(ch, key)) {
450                   requestedKeys.add(key);
451                } else {
452                   if (trace) {
453                      log.tracef("Not doing a remote get for missing key %s since entry is "
454                                  + "mapped to current node (%s). Owners are %s",
455                            toStr(key), rpcManager.getAddress(), ch.locateOwners(key));
456                   }
457                   // Force a map entry to be created, because we know this entry is local
458                   entryFactory.wrapEntryForPut(ctx, key, null, false, command, false);
459                }
460             }
461          }
462 
463          boolean missingRemoteValues = false;
464          if (!requestedKeys.isEmpty()) {
465             if (trace) {
466                log.tracef("Fetching entries for keys %s from remote nodes", requestedKeys);
467             }
468 
469             Map<Object, InternalCacheEntry> justRetrieved = retrieveFromRemoteSources(
470                   requestedKeys, ctx, command.getFlags());
471             Map<Object, InternalCacheEntry> previouslyFetched = command.getRemotelyFetched();
472             if (previouslyFetched != null) {
473                previouslyFetched.putAll(justRetrieved);
474             } else {
475                command.setRemotelyFetched(justRetrieved);
476             }
477             for (Object key : requestedKeys) {
478                if (!justRetrieved.containsKey(key)) {
479                   missingRemoteValues = true;
480                } else {
481                   entryFactory.wrapEntryForPut(ctx, key, justRetrieved.get(key), false, command, false);
482                }
483             }
484          }
485 
486          if (missingRemoteValues) {
487             throw new OutdatedTopologyException("Remote values are missing because of a topology change");
488          }
489          return invokeNextInterceptor(ctx, command);
490       } else { // remote
491          int currentTopologyId = stateTransferManager.getCacheTopology().getTopologyId();
492          boolean topologyChanged = currentTopologyId != commandTopologyId && commandTopologyId != -1;
493          // If the topology changed while invoking, this means we cannot trust any null values
494          // so we shouldn't return them
495          ConsistentHash ch = command.getConsistentHash();
496          for (Object key : command.getKeys()) {
497             CacheEntry entry = ctx.lookupEntry(key);
498             if (entry == null || entry.isNull()) {
499                if (isValueAvailableLocally(ch, key) && !topologyChanged) {
500                   if (trace) {
501                      log.tracef("Not doing a remote get for missing key %s since entry is "
502                                  + "mapped to current node (%s). Owners are %s",
503                            toStr(key), rpcManager.getAddress(), ch.locateOwners(key));
504                   }
505                   // Force a map entry to be created, because we know this entry is local
506                   entryFactory.wrapEntryForPut(ctx, key, null, false, command, false);
507                }
508             }
509          }
510          Map<Object, Object> values = (Map<Object, Object>) invokeNextInterceptor(ctx, command);
511          return values;
512       }
513    }

525

529    class SingleKeyRecipientGenerator implements RecipientGenerator {
530       private final Object key;
531       private final Set<Object> keys;
532       private List<Address> recipients = null;
533 
534       SingleKeyRecipientGenerator(Object key) {
535          this.key = key;
536          keys = Collections.singleton(key);
537       }
538 
539       @Override
540       public List<Address> generateRecipients() {
541          if (recipients == null) {
542             recipients = cdl.getOwners(key);
543          }
544          return recipients;
545       }
546 
547       @Override
548       public Collection<Object> getKeys() {
549          return keys;
550       }
551    }

557 
558       MultipleKeysRecipientGenerator(Collection<Object> keys) {
559          this.keys = keys;

567          return recipients;

570       @Override
571       public Collection<Object> getKeys() {
572          return keys;
