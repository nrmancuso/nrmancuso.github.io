5   public class MultipleCaseLabels {
6       private static final int MONDAY = 1;
7       private static final int TUESDAY = 2;
8       private static final int WEDNESDAY = 3;
9       private static final int THURSDAY = 4;
10      private static final int FRIDAY = 5;
11      private static final int SATURDAY = 6;
12      private static final int SUNDAY = 7;
13  
14  
15      public static void main(String[] args) {
16          int day = THURSDAY;
17  
18          switch (day) {
19              case MONDAY, FRIDAY, SUNDAY: System.out.println("  6"); break;
20              case TUESDAY               : System.out.println("  7"); break;
21              case THURSDAY, SATURDAY    : System.out.println("  8"); break;
22              case WEDNESDAY             : System.out.println("  9"); break;
23          }
24      }
25  }
