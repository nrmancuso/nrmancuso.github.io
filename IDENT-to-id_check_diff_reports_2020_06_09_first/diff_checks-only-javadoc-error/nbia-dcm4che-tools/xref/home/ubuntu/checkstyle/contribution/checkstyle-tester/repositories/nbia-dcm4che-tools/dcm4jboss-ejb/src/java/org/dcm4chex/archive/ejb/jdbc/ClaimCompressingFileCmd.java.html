38  package org.dcm4chex.archive.ejb.jdbc;
39  
40  import java.sql.PreparedStatement;
41  import java.sql.SQLException;
42  
43  /**
44   * @author Gunter Zeilinger <gunterze@gmail.com>
45   * @version $Revision$ $Date$
46   * @since Oct 23, 2008
47   */
48  public class ClaimCompressingFileCmd extends BaseUpdateCmd {
49  
50      public static final int transactionIsolationLevel = 0;
51  
52      public static boolean claim(long pk) throws SQLException {
53          ClaimCompressingFileCmd cmd = new ClaimCompressingFileCmd();
54          try {
55              cmd.setFilePk(pk);
56              return cmd.execute() > 0;
57         } finally {
58              cmd.close();
59          }
60      }
61  
62      public ClaimCompressingFileCmd() throws SQLException {
63          super(JdbcProperties.getInstance().getDataSource(),
64                  transactionIsolationLevel,
65                  JdbcProperties.getInstance().getProperty("ClaimCompressingFileCmd"));
66      }
67  
68      public void setFilePk(long pk) throws SQLException {
69          ((PreparedStatement) stmt).setLong(1, pk);
70      }
71  }
