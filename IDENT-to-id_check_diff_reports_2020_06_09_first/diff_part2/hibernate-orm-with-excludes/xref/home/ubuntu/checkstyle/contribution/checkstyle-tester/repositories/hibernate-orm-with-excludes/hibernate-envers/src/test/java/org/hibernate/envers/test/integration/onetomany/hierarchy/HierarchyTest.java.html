18  @TestForIssue(jiraKey = "HHH-6661")
19  public class HierarchyTest extends BaseEnversJPAFunctionalTestCase {
20      private Long parentId = null;
21      private Long child1Id = null;
22      private Long child2Id = null;
23  
24      public void configure(Ejb3Configuration cfg) {
25          cfg.addAnnotatedClass(Node.class);
26      }
27  
28      @Test
29      @Priority(10)
30      public void initData() {
31          EntityManager em = getEntityManager();
32  
33          // Revision 1
34          em.getTransaction().begin();
35          Node parent = new Node("parent", (Node)null);
36          Node child1 = new Node("child1", parent);
37          Node child2 = new Node("child2", parent);
38          parent.getChildren().add(child1); 
39          parent.getChildren().add(child2);
40          em.persist(parent);
41          em.persist(child1);
42          em.persist(child2);
43          em.getTransaction().commit();
44  
45          parentId = parent.getId();
46          child1Id = child1.getId();
47          child2Id = child2.getId();
48  
49          // Revision 2
50          em.getTransaction().begin();
51          parent = em.find(Node.class, parent.getId());
52          parent.getChildren().get(0).setData("child1 modified");
53          em.getTransaction().commit();
54  
55          // Revision 3
56          em.getTransaction().begin();
57          child2 = em.find(Node.class, child2.getId());
58          em.remove(child2);
59          em.getTransaction().commit();
60      }
61  
62      @Test
63      public void testRevisionsCounts() {
64          Assert.assertEquals(Arrays.asList(1, 3), getAuditReader().getRevisions(Node.class, parentId));
65          Assert.assertEquals(Arrays.asList(1, 2), getAuditReader().getRevisions(Node.class, child1Id));
66          Assert.assertEquals(Arrays.asList(1, 3), getAuditReader().getRevisions(Node.class, child2Id));
67      }
68  
69      @Test
70      public void testHistoryOfParentNode() {
71          Node parent = new Node("parent", parentId);
72          Node child1 = new Node("child1", child1Id);
73          Node child2 = new Node("child2", child2Id);
74  
75          Node ver1 = getAuditReader().find(Node.class, parentId, 1);
76          Assert.assertEquals(parent, ver1);
77          Assert.assertTrue(TestTools.checkCollection( ver1.getChildren(), child1, child2 ));
78  
79          child1.setData("child1 modified");
80  
81          Node ver2 = getAuditReader().find(Node.class, parentId, 2);
82          Assert.assertEquals(parent, ver2);
83          Assert.assertTrue(TestTools.checkCollection( ver2.getChildren(), child1, child2 ));
84  
85          Node ver3 = getAuditReader().find(Node.class, parentId, 3);
86          Assert.assertEquals(parent, ver3);
87          Assert.assertTrue(TestTools.checkCollection( ver3.getChildren(), child1 ));
88      }
89  
90      @Test
91      public void testHistoryOfChild1Node() {
92          Node parent = new Node("parent", parentId);
93          Node child1 = new Node("child1", child1Id);
94  
95          Node ver1 = getAuditReader().find(Node.class, child1Id, 1);
96          Assert.assertEquals(child1, ver1);
97          Assert.assertEquals(parent.getId(), ver1.getParent().getId());
98          Assert.assertEquals(parent.getData(), ver1.getParent().getData());
99  
100         child1.setData("child1 modified");
101 
102         Node ver2 = getAuditReader().find(Node.class, child1Id, 2);
103         Assert.assertEquals(child1, ver2);
104         Assert.assertEquals(parent.getId(), ver2.getParent().getId());
105         Assert.assertEquals(parent.getData(), ver2.getParent().getData());
106     }
