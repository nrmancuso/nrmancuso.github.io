24  package org.hibernate.test.annotations.beanvalidation;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.mapping.Column;
29  import org.hibernate.mapping.PersistentClass;
30  import org.hibernate.mapping.Property;
31  import org.hibernate.testing.TestForIssue;
32  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
33  
34  import static org.junit.Assert.assertEquals;
35  import static org.junit.Assert.assertFalse;
36  import static org.junit.Assert.assertTrue;
37  
38  /**
39   * Test verifying that DDL constraints get applied when Bean Validation / Hibernate Validator are enabled.
40   *
41   * @author Emmanuel Bernard
42   * @author Hardy Ferentschik
43   */
44  public class DDLTest extends BaseCoreFunctionalTestCase {
45  	@Test
46  	public void testBasicDDL() {
47  		PersistentClass classMapping = configuration().getClassMapping( Address.class.getName() );
48  		Column stateColumn = (Column) classMapping.getProperty( "state" ).getColumnIterator().next();
49  		assertEquals( stateColumn.getLength(), 3 );
50  		Column zipColumn = (Column) classMapping.getProperty( "zip" ).getColumnIterator().next();
51  		assertEquals( zipColumn.getLength(), 5 );
52  		assertFalse( zipColumn.isNullable() );
53  	}
54  
55  	@Test
56  	public void testApplyOnIdColumn() throws Exception {
57  		PersistentClass classMapping = configuration().getClassMapping( Tv.class.getName() );
58  		Column serialColumn = (Column) classMapping.getIdentifierProperty().getColumnIterator().next();
59  		assertEquals( "Validator annotation not applied on ids", 2, serialColumn.getLength() );
60  	}
61  
62  	@Test
63  	@TestForIssue( jiraKey = "HHH-5281" )
64  	public void testLengthConstraint() throws Exception {
65  		PersistentClass classMapping = configuration().getClassMapping( Tv.class.getName() );
66  		Column modelColumn = (Column) classMapping.getProperty( "model" ).getColumnIterator().next();
67  		assertEquals( modelColumn.getLength(), 5 );
68  	}
69  
70  	@Test
71  	public void testApplyOnManyToOne() throws Exception {
72  		PersistentClass classMapping = configuration().getClassMapping( TvOwner.class.getName() );
73  		Column serialColumn = (Column) classMapping.getProperty( "tv" ).getColumnIterator().next();
74  		assertEquals( "Validator annotations not applied on associations", false, serialColumn.isNullable() );
75  	}
76  
77  	@Test
78  	public void testSingleTableAvoidNotNull() throws Exception {
79  		PersistentClass classMapping = configuration().getClassMapping( Rock.class.getName() );
80  		Column serialColumn = (Column) classMapping.getProperty( "bit" ).getColumnIterator().next();
81  		assertTrue( "Notnull should not be applied on single tables", serialColumn.isNullable() );
82  	}
83  
84  	@Test
85  	public void testNotNullOnlyAppliedIfEmbeddedIsNotNullItself() throws Exception {
86  		PersistentClass classMapping = configuration().getClassMapping( Tv.class.getName() );
87  		Property property = classMapping.getProperty( "tuner.frequency" );
88  		Column serialColumn = (Column) property.getColumnIterator().next();
89  		assertEquals(
90  				"Validator annotations are applied on tuner as it is @NotNull", false, serialColumn.isNullable()
91  		);
92  
93  		property = classMapping.getProperty( "recorder.time" );
94  		serialColumn = (Column) property.getColumnIterator().next();
95  		assertEquals(
96  				"Validator annotations are applied on tuner as it is @NotNull", true, serialColumn.isNullable()
97  		);
98  	}
99  
100 	@Override
101 	protected Class<?>[] getAnnotatedClasses() {
102 		return new Class<?>[] {
103 				Address.class,
104 				Tv.class,
105 				TvOwner.class,
106 				Rock.class
107 		};
108 	}
109 }
