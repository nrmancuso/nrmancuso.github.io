47  	@Test
48  	public void testDefaultConfigurationModeIsInherited() throws Exception {
49  		User john = new User();
50  		john.setFirstname( "John" );
51  		john.setLastname( "Doe" );
52  		List<User> friends = new ArrayList<User>();
53  		User friend = new User();
54  		friend.setFirstname( "Jane" );
55  		friend.setLastname( "Doe" );
56  		friends.add( friend );
57  		john.setFriends( friends );
58  
59  		Session s = openSession();
60  		s.persist( john );
61  		Transaction tx = s.beginTransaction();
62  		tx.commit();
63  		s.clear();
64  		tx = s.beginTransaction();
65  		john = ( User ) s.get( User.class, john.getId() );
66  		assertEquals( "Wrong number of friends", 1, john.getFriends().size() );
67  		assertNull( john.firstname );
68  
69  		s.delete( john );
70  		tx.commit();
71  		s.close();
72  	}

78  		fur.setName( "Beech" );

82  		Transaction tx = s.beginTransaction();
83  		tx.commit();

86  		fur = ( Furniture ) s.get( Furniture.class, fur.getId() );
87  		assertFalse( fur.isAlive );

183 	@Test
184 	public void testEmbeddableUsesAccessStrategyOfContainingClass() throws Exception {
185 		Circle circle = new Circle();
186 		Color color = new Color( 5, 10, 15 );
187 		circle.setColor( color );
188 		Session s = openSession();
189 		s.persist( circle );
190 		Transaction tx = s.beginTransaction();
191 		tx.commit();
192 		s.clear();
193 		tx = s.beginTransaction();
194 		circle = ( Circle ) s.get( Circle.class, circle.getId() );
195 		assertEquals( 5, circle.getColor().r );
196 		try {
197 			circle.getColor().getR();
198 			fail();
199 		} catch (RuntimeException e) {
200 			// success		
201 		}
202 		s.delete( circle );
203 		tx.commit();
204 		s.close();
205 	}
