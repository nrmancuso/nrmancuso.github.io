851 	private String buildLongString(int size, char baseChar) {
852 		StringBuilder buff = new StringBuilder();
853 		for( int i = 0; i < size; i++ ) {
854 			buff.append( baseChar );
855 		}
856 		return buff.toString();
857 	}

859 	private byte[] buildLongByteArray(int size, boolean on) {
860 		byte[] data = new byte[size];
861 		data[0] = mask( on );
862 		for ( int i = 0; i < size; i++ ) {
863 			data[i] = mask( on );
864 			on = !on;
865 		}
866 		return data;
867 	}
868 
869 	private byte mask(boolean on) {
870 		return on ? ( byte ) 1 : ( byte ) 0;
871 	}

873 	@SuppressWarnings( {"unchecked"})
874 	private static class UpperCasedAliasToEntityMapResultTransformer extends BasicTransformerAdapter implements Serializable {
875 		public Object transformTuple(Object[] tuple, String[] aliases) {
876 			Map result = new HashMap( tuple.length );
877 			for ( int i = 0; i < tuple.length; i++ ) {
878 				String alias = aliases[i];
879 				if ( alias != null ) {
880 					result.put( alias.toUpperCase(), tuple[i] );
881 				}
882 			}
883 			return result;
884 		}
