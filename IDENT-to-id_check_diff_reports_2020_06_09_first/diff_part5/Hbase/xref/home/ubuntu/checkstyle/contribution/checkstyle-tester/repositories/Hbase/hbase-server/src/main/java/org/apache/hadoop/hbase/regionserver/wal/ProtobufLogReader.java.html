50  /**
51   * A Protobuf based WAL has the following structure:
52   * <p>
53   * &lt;PB_WAL_MAGIC&gt;&lt;WALHeader&gt;&lt;WALEdits&gt;...&lt;WALEdits&gt;&lt;Trailer&gt;
54   * &lt;TrailerSize&gt; &lt;PB_WAL_COMPLETE_MAGIC&gt;
55   * </p>
56   * The Reader reads meta information (WAL Compression state, WALTrailer, etc) in
57   * {@link ProtobufLogReader#initReader(FSDataInputStream)}. A WALTrailer is an extensible structure
58   * which is appended at the end of the WAL. This is empty for now; it can contain some meta
59   * information such as Region level stats, etc in future.
60   */
61  @InterfaceAudience.LimitedPrivate({HBaseInterfaceAudience.COPROC, HBaseInterfaceAudience.PHOENIX, HBaseInterfaceAudience.CONFIG})
62  public class ProtobufLogReader extends ReaderBase {
63    private static final Log LOG = LogFactory.getLog(ProtobufLogReader.class);
64    // public for WALFactory until we move everything to o.a.h.h.wal
65    @InterfaceAudience.Private
66    public static final byte[] PB_WAL_MAGIC = Bytes.toBytes("PWAL");
67    // public for TestWALSplit
68    @InterfaceAudience.Private
69    public static final byte[] PB_WAL_COMPLETE_MAGIC = Bytes.toBytes("LAWP");
70    /**
71     * Configuration name of WAL Trailer's warning size. If a waltrailer's size is greater than the
72     * configured size, providers should log a warning. e.g. this is used with Protobuf reader/writer.
73     */
74    static final String WAL_TRAILER_WARN_SIZE = "hbase.regionserver.waltrailer.warn.size";
75    static final int DEFAULT_WAL_TRAILER_WARN_SIZE = 1024 * 1024; // 1MB

87    // maximum size of the wal Trailer in bytes. If a user writes/reads a trailer with size larger
88    // than this size, it is written/read respectively, with a WARN message in the log.

170   protected WALHdrContext readHeader(Builder builder, FSDataInputStream stream)
171       throws IOException {
172      boolean res = builder.mergeDelimitedFrom(stream);
173      if (!res) return new WALHdrContext(WALHdrResult.EOF, null);
174      if (builder.hasWriterClsName() &&
175          !getWriterClsNames().contains(builder.getWriterClsName())) {
176        return new WALHdrContext(WALHdrResult.UNKNOWN_WRITER_CLS, null);
177      }
178      String clsName = null;
179      if (builder.hasCellCodecClsName()) {
180        clsName = builder.getCellCodecClsName();
181      }
182      return new WALHdrContext(WALHdrResult.SUCCESS, clsName);
183   }

185   private String initInternal(FSDataInputStream stream, boolean isFirst)
186       throws IOException {
187     close();
188     long expectedPos = PB_WAL_MAGIC.length;
189     if (stream == null) {
190       stream = fs.open(path);
191       stream.seek(expectedPos);
192     }
193     if (stream.getPos() != expectedPos) {
194       throw new IOException("The stream is at invalid position: " + stream.getPos());
195     }
196     // Initialize metadata or, when we reset, just skip the header.
197     WALProtos.WALHeader.Builder builder = WALProtos.WALHeader.newBuilder();
198     WALHdrContext hdrCtxt = readHeader(builder, stream);
199     WALHdrResult walHdrRes = hdrCtxt.getResult();
200     if (walHdrRes == WALHdrResult.EOF) {
201       throw new EOFException("Couldn't read WAL PB header");
202     }
203     if (walHdrRes == WALHdrResult.UNKNOWN_WRITER_CLS) {
204       throw new IOException("Got unknown writer class: " + builder.getWriterClsName());
205     }
206     if (isFirst) {
207       WALProtos.WALHeader header = builder.build();
208       this.hasCompression = header.hasHasCompression() && header.getHasCompression();
209       this.hasTagCompression = header.hasHasTagCompression() && header.getHasTagCompression();
210     }
211     this.inputStream = stream;
212     this.walEditsStopOffset = this.fileLength;
213     long currentPosition = stream.getPos();
214     trailerPresent = setTrailerIfPresent();
215     this.seekOnFs(currentPosition);
216     if (LOG.isTraceEnabled()) {
217       LOG.trace("After reading the trailer: walEditsStopOffset: " + this.walEditsStopOffset
218           + ", fileLength: " + this.fileLength + ", " + "trailerPresent: " + trailerPresent);
219     }
220     
221     codecClsName = hdrCtxt.getCellCodecClsName();
222     
223     return hdrCtxt.getCellCodecClsName();
224   }

226   /**
227    * To check whether a trailer is present in a WAL, it seeks to position (fileLength -
228    * PB_WAL_COMPLETE_MAGIC.size() - Bytes.SIZEOF_INT). It reads the int value to know the size of
229    * the trailer, and checks whether the trailer is present at the end or not by comparing the last
230    * PB_WAL_COMPLETE_MAGIC.size() bytes. In case trailer is not present, it returns false;
231    * otherwise, sets the trailer and sets this.walEditsStopOffset variable up to the point just
232    * before the trailer.
233    * <ul>
234    * The trailer is ignored in case:
235    * <li>fileLength is 0 or not correct (when file is under recovery, etc).
236    * <li>the trailer size is negative.
237    * </ul>
238    * <p>
239    * In case the trailer size > this.trailerMaxSize, it is read after a WARN message.
240    * @return true if a valid trailer is present
241    * @throws IOException
242    */
243   private boolean setTrailerIfPresent() {
244     try {
245       long trailerSizeOffset = this.fileLength - (PB_WAL_COMPLETE_MAGIC.length + Bytes.SIZEOF_INT);
246       if (trailerSizeOffset <= 0) return false;// no trailer possible.
247       this.seekOnFs(trailerSizeOffset);
248       // read the int as trailer size.
249       int trailerSize = this.inputStream.readInt();
250       ByteBuffer buf = ByteBuffer.allocate(ProtobufLogReader.PB_WAL_COMPLETE_MAGIC.length);
251       this.inputStream.readFully(buf.array(), buf.arrayOffset(), buf.capacity());
252       if (!Arrays.equals(buf.array(), PB_WAL_COMPLETE_MAGIC)) {
253         LOG.trace("No trailer found.");
254         return false;
255       }
256       if (trailerSize < 0) {
257         LOG.warn("Invalid trailer Size " + trailerSize + ", ignoring the trailer");
258         return false;
259       } else if (trailerSize > this.trailerWarnSize) {
260         // continue reading after warning the user.
261         LOG.warn("Please investigate WALTrailer usage. Trailer size > maximum configured size : "
262           + trailerSize + " > " + this.trailerWarnSize);
263       }
264       // seek to the position where trailer starts.
265       long positionOfTrailer = trailerSizeOffset - trailerSize;
266       this.seekOnFs(positionOfTrailer);
267       // read the trailer.
268       buf = ByteBuffer.allocate(trailerSize);// for trailer.
269       this.inputStream.readFully(buf.array(), buf.arrayOffset(), buf.capacity());
270       trailer = WALTrailer.parseFrom(buf.array());
271       this.walEditsStopOffset = positionOfTrailer;
272       return true;
273     } catch (IOException ioe) {
274       LOG.warn("Got IOE while reading the trailer. Continuing as if no trailer is present.", ioe);
275     }
276     return false;
277   }

308   @Override
309   protected boolean readNext(Entry entry) throws IOException {
310     while (true) {
311       // OriginalPosition might be < 0 on local fs; if so, it is useless to us.
312       long originalPosition = this.inputStream.getPos();
313       if (trailerPresent && originalPosition > 0 && originalPosition == this.walEditsStopOffset) {
314         return false;
315       }
316       WALKey.Builder builder = WALKey.newBuilder();
317       long size = 0;
318       try {
319         long available = -1;
320         try {
321           int firstByte = this.inputStream.read();
322           if (firstByte == -1) {
323             throw new EOFException("First byte is negative");
324           }
325           size = CodedInputStream.readRawVarint32(firstByte, this.inputStream);
326           // available may be < 0 on local fs for instance.  If so, can't depend on it.
327           available = this.inputStream.available();
328           if (available > 0 && available < size) {
329             throw new EOFException("Available stream not enough for edit, " +
330                 "inputStream.available()= " + this.inputStream.available() + ", " +
331                 "entry size= " + size);
332           }
333           final InputStream limitedInput = new LimitInputStream(this.inputStream, size);
334           builder.mergeFrom(limitedInput);
335         } catch (InvalidProtocolBufferException ipbe) {
336           throw (EOFException) new EOFException("Invalid PB, EOF? Ignoring; originalPosition=" +
337             originalPosition + ", currentPosition=" + this.inputStream.getPos() +
338             ", messageSize=" + size + ", currentAvailable=" + available).initCause(ipbe);
339         }
340         if (!builder.isInitialized()) {
341           // TODO: not clear if we should try to recover from corrupt PB that looks semi-legit.
342           //       If we can get the KV count, we could, theoretically, try to get next record.
343           throw new EOFException("Partial PB while reading WAL, " +
344               "probably an unexpected EOF, ignoring");
345         }
346         WALKey walKey = builder.build();
347         entry.getKey().readFieldsFromPb(walKey, this.byteStringUncompressor);
348         if (!walKey.hasFollowingKvCount() || 0 == walKey.getFollowingKvCount()) {
349           LOG.trace("WALKey has no KVs that follow it; trying the next one");
350           continue;
351         }
352         int expectedCells = walKey.getFollowingKvCount();
353         long posBefore = this.inputStream.getPos();
354         try {
355           int actualCells = entry.getEdit().readFromCells(cellDecoder, expectedCells);
356           if (expectedCells != actualCells) {
357             throw new EOFException("Only read " + actualCells); // other info added in catch
358           }
359         } catch (Exception ex) {
360           String posAfterStr = "<unknown>";
361           try {
362             posAfterStr = this.inputStream.getPos() + "";
363           } catch (Throwable t) {
364             LOG.trace("Error getting pos for error message - ignoring", t);
365           }
366           String message = " while reading " + expectedCells + " WAL KVs; started reading at "
367               + posBefore + " and read up to " + posAfterStr;
368           IOException realEofEx = extractHiddenEof(ex);
369           throw (EOFException) new EOFException("EOF " + message).
370               initCause(realEofEx != null ? realEofEx : ex);
371         }
372         if (trailerPresent && this.inputStream.getPos() > this.walEditsStopOffset) {
373           LOG.error("Read WALTrailer while reading WALEdits. wal: " + this.path
374               + ", inputStream.getPos(): " + this.inputStream.getPos() + ", walEditsStopOffset: "
375               + this.walEditsStopOffset);
376           throw new EOFException("Read WALTrailer while reading WALEdits");
377         }
378       } catch (EOFException eof) {
379         LOG.trace("Encountered a malformed edit, seeking back to last good position in file", eof);
380         // If originalPosition is < 0, it is rubbish and we cannot use it (probably local fs)
381         if (originalPosition < 0) throw eof;
382         // Else restore our position to original location in hope that next time through we will
383         // read successfully.
384         seekOnFs(originalPosition);
385         return false;
386       }
387       return true;
388     }
389   }

391   private IOException extractHiddenEof(Exception ex) {
392     // There are two problems we are dealing with here. Hadoop stream throws generic exception
393     // for EOF, not EOFException; and scanner further hides it inside RuntimeException.
394     IOException ioEx = null;
395     if (ex instanceof EOFException) {
396       return (EOFException)ex;
397     } else if (ex instanceof IOException) {
398       ioEx = (IOException)ex;
399     } else if (ex instanceof RuntimeException
400         && ex.getCause() != null && ex.getCause() instanceof IOException) {
401       ioEx = (IOException)ex.getCause();
402     }
403     if (ioEx != null) {
404       if (ioEx.getMessage().contains("EOF")) return ioEx;
405       return null;
406     }
407     return null;
408   }
