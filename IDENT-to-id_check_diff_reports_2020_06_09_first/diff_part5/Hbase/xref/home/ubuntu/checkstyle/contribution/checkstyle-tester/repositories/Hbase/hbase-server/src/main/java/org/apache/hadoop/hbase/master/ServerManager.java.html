85  /**
86   * The ServerManager class manages info about region servers.
87   * <p>
88   * Maintains lists of online and dead servers.  Processes the startups,
89   * shutdowns, and deaths of region servers.
90   * <p>
91   * Servers are distinguished in two different ways.  A given server has a
92   * location, specified by hostname and port, and of which there can only be one
93   * online at any given time.  A server instance is specified by the location
94   * (hostname and port) as well as the startcode (timestamp from when the server
95   * was started).  This is used to differentiate a restarted instance of a given
96   * server from the original instance.
97   * <p>
98   * If a sever is known not to be running any more, it is called dead. The dead
99   * server needs to be handled by a ServerShutdownHandler.  If the handler is not
100  * enabled yet, the server can't be handled right away so it is queued up.
101  * After the handler is enabled, the server will be submitted to a handler to handle.
102  * However, the handler may be just partially enabled.  If so,
103  * the server cannot be fully processed, and be queued up for further processing.
104  * A server is fully processed only after the handler is fully enabled
105  * and has completed the handling.
106  */
107 @InterfaceAudience.Private

243   /**
244    * Let the server manager know a new regionserver has come online
245    * @param request the startup request
246    * @param ia the InetAddress from which request is received
247    * @return The ServerName we know this server as.
248    * @throws IOException
249    */
250   ServerName regionServerStartup(RegionServerStartupRequest request, InetAddress ia)
251       throws IOException {
252     // Test for case where we get a region startup message from a regionserver
253     // that has been quickly restarted but whose znode expiration handler has
254     // not yet run, or from a server whose fail we are currently processing.
255     // Test its host+port combo is present in serverAddresstoServerInfo.  If it
256     // is, reject the server and trigger its expiration. The next time it comes
257     // in, it should have been removed from serverAddressToServerInfo and queued
258     // for processing by ProcessServerShutdown.
259 
260     final String hostname = request.hasUseThisHostnameInstead() ?
261         request.getUseThisHostnameInstead() :ia.getHostName();
262     ServerName sn = ServerName.valueOf(hostname, request.getPort(),
263       request.getServerStartCode());
264     checkClockSkew(sn, request.getServerCurrentTime());
265     checkIsDead(sn, "STARTUP");
266     if (!checkAndRecordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD)) {
267       LOG.warn("THIS SHOULD NOT HAPPEN, RegionServerStartup"
268         + " could not record the server: " + sn);
269     }
270     return sn;
271   }

273   private ConcurrentNavigableMap<byte[], Long> getOrCreateStoreFlushedSequenceId(
274     byte[] regionName) {
275     ConcurrentNavigableMap<byte[], Long> storeFlushedSequenceId =
276         storeFlushedSequenceIdsByRegion.get(regionName);
277     if (storeFlushedSequenceId != null) {
278       return storeFlushedSequenceId;
279     }
280     storeFlushedSequenceId = new ConcurrentSkipListMap<byte[], Long>(Bytes.BYTES_COMPARATOR);
281     ConcurrentNavigableMap<byte[], Long> alreadyPut =
282         storeFlushedSequenceIdsByRegion.putIfAbsent(regionName, storeFlushedSequenceId);
283     return alreadyPut == null ? storeFlushedSequenceId : alreadyPut;
284   }
285   /**
286    * Updates last flushed sequence Ids for the regions on server sn
287    * @param sn
288    * @param hsl
289    */
290   private void updateLastFlushedSequenceIds(ServerName sn, ServerLoad hsl) {
291     Map<byte[], RegionLoad> regionsLoad = hsl.getRegionsLoad();
292     for (Entry<byte[], RegionLoad> entry : regionsLoad.entrySet()) {
293       byte[] encodedRegionName = Bytes.toBytes(HRegionInfo.encodeRegionName(entry.getKey()));
294       Long existingValue = flushedSequenceIdByRegion.get(encodedRegionName);
295       long l = entry.getValue().getCompleteSequenceId();
296       // Don't let smaller sequence ids override greater sequence ids.
297       if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue)) {
298         flushedSequenceIdByRegion.put(encodedRegionName, l);
299       } else if (l != HConstants.NO_SEQNUM && l < existingValue) {
300         LOG.warn("RegionServer " + sn + " indicates a last flushed sequence id ("
301             + l + ") that is less than the previous last flushed sequence id ("
302             + existingValue + ") for region " + Bytes.toString(entry.getKey()) + " Ignoring.");
303       }
304       ConcurrentNavigableMap<byte[], Long> storeFlushedSequenceId =
305           getOrCreateStoreFlushedSequenceId(encodedRegionName);
306       for (StoreSequenceId storeSeqId : entry.getValue().getStoreCompleteSequenceId()) {
307         byte[] family = storeSeqId.getFamilyName().toByteArray();
308         existingValue = storeFlushedSequenceId.get(family);
309         l = storeSeqId.getSequenceId();
310         // Don't let smaller sequence ids override greater sequence ids.
311         if (existingValue == null || (l != HConstants.NO_SEQNUM && l > existingValue.longValue())) {
312           storeFlushedSequenceId.put(family, l);
313         }
314       }
315     }
316   }

318   void regionServerReport(ServerName sn,
319       ServerLoad sl) throws YouAreDeadException {
320     checkIsDead(sn, "REPORT");
321     if (null == this.onlineServers.replace(sn, sl)) {
322       // Already have this host+port combo and its just different start code?
323       // Just let the server in. Presume master joining a running cluster.
324       // recordNewServer is what happens at the end of reportServerStartup.
325       // The only thing we are skipping is passing back to the regionserver
326       // the ServerName to use. Here we presume a master has already done
327       // that so we'll press on with whatever it gave us for ServerName.
328       if (!checkAndRecordNewServer(sn, sl)) {
329         LOG.info("RegionServerReport ignored, could not record the server: " + sn);
330         return; // Not recorded, so no need to move on
331       }
332     }
333     updateLastFlushedSequenceIds(sn, sl);
334   }

336   /**
337    * Check is a server of same host and port already exists,
338    * if not, or the existed one got a smaller start code, record it.
339    *
340    * @param sn the server to check and record
341    * @param sl the server load on the server
342    * @return true if the server is recorded, otherwise, false
343    */
344   boolean checkAndRecordNewServer(
345       final ServerName serverName, final ServerLoad sl) {
346     ServerName existingServer = null;
347     synchronized (this.onlineServers) {
348       existingServer = findServerWithSameHostnamePortWithLock(serverName);
349       if (existingServer != null && (existingServer.getStartcode() > serverName.getStartcode())) {
350         LOG.info("Server serverName=" + serverName + " rejected; we already have "
351             + existingServer.toString() + " registered with same hostname and port");
352         return false;
353       }
354       recordNewServerWithLock(serverName, sl);
355     }
356 
357     // Tell our listeners that a server was added
358     if (!this.listeners.isEmpty()) {
359       for (ServerListener listener : this.listeners) {
360         listener.serverAdded(serverName);
361       }
362     }
363 
364     // Note that we assume that same ts means same server, and don't expire in that case.
365     //  TODO: ts can theoretically collide due to clock shifts, so this is a bit hacky.
366     if (existingServer != null && (existingServer.getStartcode() < serverName.getStartcode())) {
367       LOG.info("Triggering server recovery; existingServer " +
368           existingServer + " looks stale, new server:" + serverName);
369       expireServer(existingServer);
370     }
371     return true;
372   }

374   /**
375    * Checks if the clock skew between the server and the master. If the clock skew exceeds the
376    * configured max, it will throw an exception; if it exceeds the configured warning threshold,
377    * it will log a warning but start normally.
378    * @param serverName Incoming servers's name
379    * @param serverCurrentTime
380    * @throws ClockOutOfSyncException if the skew exceeds the configured max value
381    */
382   private void checkClockSkew(final ServerName serverName, final long serverCurrentTime)
383   throws ClockOutOfSyncException {
384     long skew = Math.abs(System.currentTimeMillis() - serverCurrentTime);
385     if (skew > maxSkew) {
386       String message = "Server " + serverName + " has been " +
387         "rejected; Reported time is too far out of sync with master.  " +
388         "Time difference of " + skew + "ms > max allowed of " + maxSkew + "ms";
389       LOG.warn(message);
390       throw new ClockOutOfSyncException(message);
391     } else if (skew > warningSkew){
392       String message = "Reported time for server " + serverName + " is out of sync with master " +
393         "by " + skew + "ms. (Warning threshold is " + warningSkew + "ms; " +
394         "error threshold is " + maxSkew + "ms)";
395       LOG.warn(message);
396     }
397   }

399   /**
400    * If this server is on the dead list, reject it with a YouAreDeadException.
401    * If it was dead but came back with a new start code, remove the old entry
402    * from the dead list.
403    * @param serverName
404    * @param what START or REPORT
405    * @throws org.apache.hadoop.hbase.YouAreDeadException
406    */
407   private void checkIsDead(final ServerName serverName, final String what)
408       throws YouAreDeadException {
409     if (this.deadservers.isDeadServer(serverName)) {
410       // host name, port and start code all match with existing one of the
411       // dead servers. So, this server must be dead.
412       String message = "Server " + what + " rejected; currently processing " +
413           serverName + " as dead server";
414       LOG.debug(message);
415       throw new YouAreDeadException(message);
416     }
417     // remove dead server with same hostname and port of newly checking in rs after master
418     // initialization.See HBASE-5916 for more information.
419     if ((this.services == null || ((HMaster) this.services).isInitialized())
420         && this.deadservers.cleanPreviousInstance(serverName)) {
421       // This server has now become alive after we marked it as dead.
422       // We removed it's previous entry from the dead list to reflect it.
423       LOG.debug(what + ":" + " Server " + serverName + " came back up," +
424           " removed it from the dead servers list");
425     }
426   }

453   public RegionStoreSequenceIds getLastFlushedSequenceId(byte[] encodedRegionName) {
454     RegionStoreSequenceIds.Builder builder = RegionStoreSequenceIds.newBuilder();
455     Long seqId = flushedSequenceIdByRegion.get(encodedRegionName);
456     builder.setLastFlushedSequenceId(seqId != null ? seqId.longValue() : HConstants.NO_SEQNUM);
457     Map<byte[], Long> storeFlushedSequenceId =
458         storeFlushedSequenceIdsByRegion.get(encodedRegionName);
459     if (storeFlushedSequenceId != null) {
460       for (Map.Entry<byte[], Long> entry : storeFlushedSequenceId.entrySet()) {
461         builder.addStoreSequenceId(StoreSequenceId.newBuilder()
462             .setFamilyName(ByteString.copyFrom(entry.getKey()))
463             .setSequenceId(entry.getValue().longValue()).build());
464       }
465     }
466     return builder.build();
467   }

477   /**
478    * Compute the average load across all region servers.
479    * Currently, this uses a very naive computation - just uses the number of
480    * regions being served, ignoring stats about number of requests.
481    * @return the average load
482    */
483   public double getAverageLoad() {
484     int totalLoad = 0;
485     int numServers = 0;
486     for (ServerLoad sl: this.onlineServers.values()) {
487         numServers++;
488         totalLoad += sl.getNumberOfRegions();
489     }
490     return numServers == 0 ? 0 :
491       (double)totalLoad / (double)numServers;
492   }

523   void letRegionServersShutdown() {
524     long previousLogTime = 0;
525     ServerName sn = master.getServerName();
526     ZooKeeperWatcher zkw = master.getZooKeeper();
527     int onlineServersCt;
528     while ((onlineServersCt = onlineServers.size()) > 0){
529 
530       if (System.currentTimeMillis() > (previousLogTime + 1000)) {
531         Set<ServerName> remainingServers = onlineServers.keySet();
532         synchronized (onlineServers) {
533           if (remainingServers.size() == 1 && remainingServers.contains(sn)) {
534             // Master will delete itself later.
535             return;
536           }
537         }
538         StringBuilder sb = new StringBuilder();
539         // It's ok here to not sync on onlineServers - merely logging
540         for (ServerName key : remainingServers) {
541           if (sb.length() > 0) {
542             sb.append(", ");
543           }
544           sb.append(key);
545         }
546         LOG.info("Waiting on regionserver(s) to go down " + sb.toString());
547         previousLogTime = System.currentTimeMillis();
548       }
549 
550       try {
551         List<String> servers = ZKUtil.listChildrenNoWatch(zkw, zkw.rsZNode);
552         if (servers == null || servers.size() == 0 || (servers.size() == 1
553             && servers.contains(sn.toString()))) {
554           LOG.info("ZK shows there is only the master self online, exiting now");
555           // Master could have lost some ZK events, no need to wait more.
556           break;
557         }
558       } catch (KeeperException ke) {
559         LOG.warn("Failed to list regionservers", ke);
560         // ZK is malfunctioning, don't hang here
561         break;
562       }
563       synchronized (onlineServers) {
564         try {
565           if (onlineServersCt == onlineServers.size()) onlineServers.wait(100);
566         } catch (InterruptedException ignored) {
567           // continue
568         }
569       }
570     }
571   }

573   /*
574    * Expire the passed server.  Add it to list of dead servers and queue a
575    * shutdown processing.
576    */
577   public synchronized void expireServer(final ServerName serverName) {
578     if (serverName.equals(master.getServerName())) {
579       if (!(master.isAborted() || master.isStopped())) {
580         master.stop("We lost our znode?");
581       }
582       return;
583     }
584     if (!services.isServerShutdownHandlerEnabled()) {
585       LOG.info("Master doesn't enable ServerShutdownHandler during initialization, "
586           + "delay expiring server " + serverName);
587       this.queuedDeadServers.add(serverName);
588       return;
589     }
590     if (this.deadservers.isDeadServer(serverName)) {
591       // TODO: Can this happen?  It shouldn't be online in this case?
592       LOG.warn("Expiration of " + serverName +
593           " but server shutdown already in progress");
594       return;
595     }
596     synchronized (onlineServers) {
597       if (!this.onlineServers.containsKey(serverName)) {
598         LOG.warn("Expiration of " + serverName + " but server not online");
599       }
600       // Remove the server from the known servers lists and update load info BUT
601       // add to deadservers first; do this so it'll show in dead servers list if
602       // not in online servers list.
603       this.deadservers.add(serverName);
604       this.onlineServers.remove(serverName);
605       onlineServers.notifyAll();
606     }
607     this.rsAdmins.remove(serverName);
608     // If cluster is going down, yes, servers are going to be expiring; don't
609     // process as a dead server
610     if (this.clusterShutdown) {
611       LOG.info("Cluster shutdown set; " + serverName +
612         " expired; onlineServers=" + this.onlineServers.size());
613       if (this.onlineServers.isEmpty()) {
614         master.stop("Cluster shutdown set; onlineServer=0");
615       }
616       return;
617     }
618 
619     boolean carryingMeta = services.getAssignmentManager().isCarryingMeta(serverName);
620     if (carryingMeta) {
621       this.services.getExecutorService().submit(new MetaServerShutdownHandler(this.master,
622         this.services, this.deadservers, serverName));
623     } else {
624       this.services.getExecutorService().submit(new ServerShutdownHandler(this.master,
625         this.services, this.deadservers, serverName, true));
626     }
627     LOG.debug("Added=" + serverName +
628       " to dead servers, submitted shutdown handler to be executed meta=" + carryingMeta);
629 
630     // Tell our listeners that a server was removed
631     if (!this.listeners.isEmpty()) {
632       for (ServerListener listener : this.listeners) {
633         listener.serverRemoved(serverName);
634       }
635     }
636   }

642   public synchronized void processDeadServer(final ServerName serverName, boolean shouldSplitWal) {
643     // When assignment manager is cleaning up the zookeeper nodes and rebuilding the
644     // in-memory region states, region servers could be down. Meta table can and
645     // should be re-assigned, log splitting can be done too. However, it is better to
646     // wait till the cleanup is done before re-assigning user regions.
647     //
648     // We should not wait in the server shutdown handler thread since it can clog
649     // the handler threads and meta table could not be re-assigned in case
650     // the corresponding server is down. So we queue them up here instead.
651     if (!services.getAssignmentManager().isFailoverCleanupDone()) {
652       requeuedDeadServers.put(serverName, shouldSplitWal);
653       return;
654     }
655 
656     this.deadservers.add(serverName);
657     this.services.getExecutorService().submit(
658       new ServerShutdownHandler(this.master, this.services, this.deadservers, serverName,
659           shouldSplitWal));
660   }

662   /**
663    * Process the servers which died during master's initialization. It will be
664    * called after HMaster#assignMeta and AssignmentManager#joinCluster.
665    * */
666   synchronized void processQueuedDeadServers() {
667     if (!services.isServerShutdownHandlerEnabled()) {
668       LOG.info("Master hasn't enabled ServerShutdownHandler");
669     }
670     Iterator<ServerName> serverIterator = queuedDeadServers.iterator();
671     while (serverIterator.hasNext()) {
672       ServerName tmpServerName = serverIterator.next();
673       expireServer(tmpServerName);
674       serverIterator.remove();
675       requeuedDeadServers.remove(tmpServerName);
676     }
677 
678     if (!services.getAssignmentManager().isFailoverCleanupDone()) {
679       LOG.info("AssignmentManager hasn't finished failover cleanup; waiting");
680     }
681 
682     for(ServerName tmpServerName : requeuedDeadServers.keySet()){
683       processDeadServer(tmpServerName, requeuedDeadServers.get(tmpServerName));
684     }
685     requeuedDeadServers.clear();
686   }

843    * to close the region.  This bypasses the active hmaster.
844    */
845   public static void closeRegionSilentlyAndWait(ClusterConnection connection, 
846     ServerName server, HRegionInfo region, long timeout) throws IOException, InterruptedException {

849       ProtobufUtil.closeRegion(rs, server, region.getRegionName(), false);

863       }
864       Thread.sleep(1000);
865     }
866     throw new IOException("Region " + region + " failed to close within"
867         + " timeout " + timeout);

900   /**
901    * Check if a region server is reachable and has the expected start code
902    */
903   public boolean isServerReachable(ServerName server) {
904     if (server == null) throw new NullPointerException("Passed server is null");
905 
906     RetryCounter retryCounter = pingRetryCounterFactory.create();
907     while (retryCounter.shouldRetry()) {
908       synchronized (this.onlineServers) {
909         if (this.deadservers.isDeadServer(server)) {
910           return false;
911         }
912       }
913       try {
914         AdminService.BlockingInterface admin = getRsAdmin(server);
915         if (admin != null) {
916           ServerInfo info = ProtobufUtil.getServerInfo(admin);
917           return info != null && info.hasServerName()
918             && server.getStartcode() == info.getServerName().getStartCode();
919         }
920       } catch (RegionServerStoppedException | ServerNotRunningYetException e) {
921         if (LOG.isDebugEnabled()) {
922           LOG.debug("Couldn't reach " + server, e);
923         }
924         break;
925       } catch (IOException ioe) {
926         if (LOG.isDebugEnabled()) {
927           LOG.debug("Couldn't reach " + server + ", try=" + retryCounter.getAttemptTimes() + " of "
928               + retryCounter.getMaxAttempts(), ioe);
929         }
930         try {
931           retryCounter.sleepUntilNextRetry();
932         } catch(InterruptedException ie) {
933           Thread.currentThread().interrupt();
934           break;
935         }
936       }
937     }
938     return false;
939   }

963   /**
964    * Wait for the region servers to report in.
965    * We will wait until one of this condition is met:
966    *  - the master is stopped
967    *  - the 'hbase.master.wait.on.regionservers.maxtostart' number of
968    *    region servers is reached
969    *  - the 'hbase.master.wait.on.regionservers.mintostart' is reached AND
970    *   there have been no new region server in for
971    *      'hbase.master.wait.on.regionservers.interval' time AND
972    *   the 'hbase.master.wait.on.regionservers.timeout' is reached
973    *
974    * @throws InterruptedException
975    */
976   public void waitForRegionServers(MonitoredTask status)
977   throws InterruptedException {
978     final long interval = this.master.getConfiguration().
979       getLong(WAIT_ON_REGIONSERVERS_INTERVAL, 1500);
980     final long timeout = this.master.getConfiguration().
981       getLong(WAIT_ON_REGIONSERVERS_TIMEOUT, 4500);
982     int defaultMinToStart = 1;
983     if (BaseLoadBalancer.tablesOnMaster(master.getConfiguration())) {
984       // If we assign regions to master, we'd like to start
985       // at least another region server so that we don't
986       // assign all regions to master if other region servers
987       // don't come up in time.
988       defaultMinToStart = 2;
989     }
990     int minToStart = this.master.getConfiguration().
991       getInt(WAIT_ON_REGIONSERVERS_MINTOSTART, defaultMinToStart);
992     if (minToStart < 1) {
993       LOG.warn(String.format(
994         "The value of '%s' (%d) can not be less than 1, ignoring.",
995         WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
996       minToStart = 1;
997     }
998     int maxToStart = this.master.getConfiguration().
999       getInt(WAIT_ON_REGIONSERVERS_MAXTOSTART, Integer.MAX_VALUE);
1000     if (maxToStart < minToStart) {
1001         LOG.warn(String.format(
1002             "The value of '%s' (%d) is set less than '%s' (%d), ignoring.",
1003             WAIT_ON_REGIONSERVERS_MAXTOSTART, maxToStart,
1004             WAIT_ON_REGIONSERVERS_MINTOSTART, minToStart));
1005         maxToStart = Integer.MAX_VALUE;
1006     }
1007 
1008     long now =  System.currentTimeMillis();
1009     final long startTime = now;
1010     long slept = 0;
1011     long lastLogTime = 0;
1012     long lastCountChange = startTime;
1013     int count = countOfRegionServers();
1014     int oldCount = 0;
1015     while (!this.master.isStopped() && count < maxToStart
1016         && (lastCountChange+interval > now || timeout > slept || count < minToStart)) {
1017       // Log some info at every interval time or if there is a change
1018       if (oldCount != count || lastLogTime+interval < now){
1019         lastLogTime = now;
1020         String msg =
1021           "Waiting for region servers count to settle; currently"+
1022             " checked in " + count + ", slept for " + slept + " ms," +
1023             " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+
1024             ", timeout of "+timeout+" ms, interval of "+interval+" ms.";
1025         LOG.info(msg);
1026         status.setStatus(msg);
1027       }
1028 
1029       // We sleep for some time
1030       final long sleepTime = 50;
1031       Thread.sleep(sleepTime);
1032       now =  System.currentTimeMillis();
1033       slept = now - startTime;
1034 
1035       oldCount = count;
1036       count = countOfRegionServers();
1037       if (count != oldCount) {
1038         lastCountChange = now;
1039       }
1040     }
1041 
1042     LOG.info("Finished waiting for region servers count to settle;" +
1043       " checked in " + count + ", slept for " + slept + " ms," +
1044       " expecting minimum of " + minToStart + ", maximum of "+ maxToStart+","+
1045       " master is "+ (this.master.isStopped() ? "stopped.": "running")
1046     );
1047   }
