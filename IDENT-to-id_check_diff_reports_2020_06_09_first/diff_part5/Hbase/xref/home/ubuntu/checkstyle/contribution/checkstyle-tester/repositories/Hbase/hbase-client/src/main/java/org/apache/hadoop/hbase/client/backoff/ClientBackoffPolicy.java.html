24  /**
25   * Configurable policy for the amount of time a client should wait for a new request to the
26   * server when given the server load statistics.
27   * <p>
28   * Must have a single-argument constructor that takes a {@link org.apache.hadoop.conf.Configuration}
29   * </p>
30   */
31  @InterfaceAudience.Public
32  @InterfaceStability.Unstable
33  public interface ClientBackoffPolicy {
34  
35    public static final String BACKOFF_POLICY_CLASS =
36        "hbase.client.statistics.backoff-policy";
37  
38    /**
39     * @return the number of ms to wait on the client based on the
40     */
41    public long getBackoffTime(ServerName serverName, byte[] region, ServerStatistics stats);
42  }
