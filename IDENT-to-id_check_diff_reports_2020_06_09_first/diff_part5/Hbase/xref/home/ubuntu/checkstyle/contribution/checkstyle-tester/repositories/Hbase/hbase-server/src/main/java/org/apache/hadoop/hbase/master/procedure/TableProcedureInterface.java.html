19  package org.apache.hadoop.hbase.master.procedure;
20  
21  import org.apache.hadoop.hbase.TableName;
22  import org.apache.hadoop.hbase.classification.InterfaceAudience;
23  import org.apache.hadoop.hbase.classification.InterfaceStability;
24  
25  /**
26   * Procedures that operates on a specific Table (e.g. create, delete, snapshot, ...)
27   * must implement this interface to allow the system handle the lock/concurrency problems.
28   */
29  @InterfaceAudience.Private
30  @InterfaceStability.Evolving
31  public interface TableProcedureInterface {
32    public enum TableOperationType {
33      CREATE, DELETE, DISABLE, EDIT, ENABLE, READ,
34    };
35  
36    /**
37     * @return the name of the table the procedure is operating on
38     */
39    TableName getTableName();
40  
41    /**
42     * Given an operation type we can take decisions about what to do with pending operations.
43     * e.g. if we get a delete and we have some table operation pending (e.g. add column)
44     * we can abort those operations.
45     * @return the operation type that the procedure is executing.
46     */
47    TableOperationType getTableOperationType();
48  }
