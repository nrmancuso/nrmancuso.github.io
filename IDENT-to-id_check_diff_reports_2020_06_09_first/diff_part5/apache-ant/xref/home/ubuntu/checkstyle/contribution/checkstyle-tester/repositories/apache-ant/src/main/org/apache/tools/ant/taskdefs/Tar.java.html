225     /**
226      * do the business
227      * @throws BuildException on error
228      */
229     public void execute() throws BuildException {
230         if (tarFile == null) {
231             throw new BuildException("tarfile attribute must be set!",
232                                      getLocation());
233         }
234 
235         if (tarFile.exists() && tarFile.isDirectory()) {
236             throw new BuildException("tarfile is a directory!",
237                                      getLocation());
238         }
239 
240         if (tarFile.exists() && !tarFile.canWrite()) {
241             throw new BuildException("Can not write to the specified tarfile!",
242                                      getLocation());
243         }
244 
245         Vector savedFileSets = (Vector) filesets.clone();
246         try {
247             if (baseDir != null) {
248                 if (!baseDir.exists()) {
249                     throw new BuildException("basedir does not exist!",
250                                              getLocation());
251                 }
252 
253                 // add the main fileset to the list of filesets to process.
254                 TarFileSet mainFileSet = new TarFileSet(fileset);
255                 mainFileSet.setDir(baseDir);
256                 filesets.addElement(mainFileSet);
257             }
258 
259             if (filesets.size() == 0 && resourceCollections.size() == 0) {
260                 throw new BuildException("You must supply either a basedir "
261                                          + "attribute or some nested resource"
262                                          + " collections.",
263                                          getLocation());
264             }
265 
266             // check if tar is out of date with respect to each
267             // fileset
268             boolean upToDate = true;
269             for (Enumeration e = filesets.elements(); e.hasMoreElements();) {
270                 upToDate &= check((TarFileSet) e.nextElement());
271             }
272             for (Enumeration e = resourceCollections.elements();
273                  e.hasMoreElements();) {
274                 upToDate &= check((ResourceCollection) e.nextElement());
275             }
276 
277             if (upToDate) {
278                 log("Nothing to do: " + tarFile.getAbsolutePath()
279                     + " is up to date.", Project.MSG_INFO);
280                 return;
281             }
282 
283             File parent = tarFile.getParentFile();
284             if (parent != null && !parent.isDirectory()
285                 && !(parent.mkdirs() || parent.isDirectory())) {
286                 throw new BuildException("Failed to create missing parent"
287                                          + " directory for " + tarFile);
288             }
289 
290             log("Building tar: " + tarFile.getAbsolutePath(), Project.MSG_INFO);
291 
292             TarOutputStream tOut = null;
293             try {
294                 tOut = new TarOutputStream(
295                     compression.compress(
296                         new BufferedOutputStream(
297                             new FileOutputStream(tarFile))));
298                 tOut.setDebug(true);
299                 if (longFileMode.isTruncateMode()) {
300                     tOut.setLongFileMode(TarOutputStream.LONGFILE_TRUNCATE);
301                 } else if (longFileMode.isFailMode()
302                             || longFileMode.isOmitMode()) {
303                     tOut.setLongFileMode(TarOutputStream.LONGFILE_ERROR);
304                 } else if (longFileMode.isPosixMode()) {
305                     tOut.setLongFileMode(TarOutputStream.LONGFILE_POSIX);
306                 } else {
307                     // warn or GNU
308                     tOut.setLongFileMode(TarOutputStream.LONGFILE_GNU);
309                 }
310 
311                 longWarningGiven = false;
312                 for (Enumeration e = filesets.elements();
313                      e.hasMoreElements();) {
314                     tar((TarFileSet) e.nextElement(), tOut);
315                 }
316                 for (Enumeration e = resourceCollections.elements();
317                      e.hasMoreElements();) {
318                     tar((ResourceCollection) e.nextElement(), tOut);
319                 }
320             } catch (IOException ioe) {
321                 String msg = "Problem creating TAR: " + ioe.getMessage();
322                 throw new BuildException(msg, ioe, getLocation());
323             } finally {
324                 FileUtils.close(tOut);
325             }
326         } finally {
327             filesets = savedFileSets;
328         }
329     }

331     /**
332      * tar a file
333      * @param file the file to tar
334      * @param tOut the output stream
335      * @param vPath the path name of the file to tar
336      * @param tarFileSet the fileset that the file came from.
337      * @throws IOException on error
338      */
339     protected void tarFile(File file, TarOutputStream tOut, String vPath,
340                            TarFileSet tarFileSet)
341         throws IOException {
342         if (file.equals(tarFile)) {
343             // If the archive is built for the first time and it is
344             // matched by a resource collection, then it hasn't been
345             // found in check (it hasn't been there) but will be
346             // included now.
347             //
348             // for some strange reason the old code would simply skip
349             // the entry and not fail, do the same now for backwards
350             // compatibility reasons.  Without this, the which4j build
351             // fails in Gump
352             return;
353         }
354         tarResource(new FileResource(file), tOut, vPath, tarFileSet);
355     }

357     /**
358      * tar a resource
359      * @param r the resource to tar
360      * @param tOut the output stream
361      * @param vPath the path name of the file to tar
362      * @param tarFileSet the fileset that the file came from, may be null.
363      * @throws IOException on error
364      * @since Ant 1.7
365      */
366     protected void tarResource(Resource r, TarOutputStream tOut, String vPath,
367                                TarFileSet tarFileSet)
368         throws IOException {
369 
370         if (!r.isExists()) {
371             return;
372         }
373 
374         boolean preserveLeadingSlashes = false;
375 
376         if (tarFileSet != null) {
377             String fullpath = tarFileSet.getFullpath(this.getProject());
378             if (fullpath.length() > 0) {
379                 vPath = fullpath;
380             } else {
381                 // don't add "" to the archive
382                 if (vPath.length() <= 0) {
383                     return;
384                 }
385 
386                 String prefix = tarFileSet.getPrefix(this.getProject());
387                 // '/' is appended for compatibility with the zip task.
388                 if (prefix.length() > 0 && !prefix.endsWith("/")) {
389                     prefix = prefix + "/";
390                 }
391                 vPath = prefix + vPath;
392             }
393 
394             preserveLeadingSlashes = tarFileSet.getPreserveLeadingSlashes();
395 
396             if (vPath.startsWith("/") && !preserveLeadingSlashes) {
397                 int l = vPath.length();
398                 if (l <= 1) {
399                     // we would end up adding "" to the archive
400                     return;
401                 }
402                 vPath = vPath.substring(1, l);
403             }
404         }
405 
406         if (r.isDirectory() && !vPath.endsWith("/")) {
407             vPath += "/";
408         }
409 
410         if (vPath.length() >= TarConstants.NAMELEN) {
411             if (longFileMode.isOmitMode()) {
412                 log("Omitting: " + vPath, Project.MSG_INFO);
413                 return;
414             } else if (longFileMode.isWarnMode()) {
415                 log("Entry: " + vPath + " longer than "
416                     + TarConstants.NAMELEN + " characters.",
417                     Project.MSG_WARN);
418                 if (!longWarningGiven) {
419                     log("Resulting tar file can only be processed "
420                         + "successfully by GNU compatible tar commands",
421                         Project.MSG_WARN);
422                     longWarningGiven = true;
423                 }
424             } else if (longFileMode.isFailMode()) {
425                 throw new BuildException("Entry: " + vPath
426                         + " longer than " + TarConstants.NAMELEN
427                         + "characters.", getLocation());
428             }
429         }
430 
431         TarEntry te = new TarEntry(vPath, preserveLeadingSlashes);
432         te.setModTime(r.getLastModified());
433         // preserve permissions
434         if (r instanceof ArchiveResource) {
435             ArchiveResource ar = (ArchiveResource) r;
436             te.setMode(ar.getMode());
437             if (r instanceof TarResource) {
438                 TarResource tr = (TarResource) r;
439                 te.setUserName(tr.getUserName());
440                 te.setUserId(tr.getUid());
441                 te.setGroupName(tr.getGroup());
442                 te.setGroupId(tr.getGid());
443             }
444         }
445 
446         if (!r.isDirectory()) {
447             if (r.size() > TarConstants.MAXSIZE) {
448                 throw new BuildException(
449                     "Resource: " + r + " larger than "
450                     + TarConstants.MAXSIZE + " bytes.");
451             }
452             te.setSize(r.getSize());
453             // override permissions if set explicitly
454             if (tarFileSet != null && tarFileSet.hasFileModeBeenSet()) {
455                 te.setMode(tarFileSet.getMode());
456             }
457         } else if (tarFileSet != null && tarFileSet.hasDirModeBeenSet()) {
458             // override permissions if set explicitly
459             te.setMode(tarFileSet.getDirMode(this.getProject()));
460         }
461 
462         if (tarFileSet != null) {
463             // only override permissions if set explicitly
464             if (tarFileSet.hasUserNameBeenSet()) {
465                 te.setUserName(tarFileSet.getUserName());
466             }
467             if (tarFileSet.hasGroupBeenSet()) {
468                 te.setGroupName(tarFileSet.getGroup());
469             }
470             if (tarFileSet.hasUserIdBeenSet()) {
471                 te.setUserId(tarFileSet.getUid());
472             }
473             if (tarFileSet.hasGroupIdBeenSet()) {
474                 te.setGroupId(tarFileSet.getGid());
475             }
476         }
477 
478         InputStream in = null;
479         try {
480             tOut.putNextEntry(te);
481 
482             if (!r.isDirectory()) {
483                 in = r.getInputStream();
484 
485                 byte[] buffer = new byte[BUFFER_SIZE];
486                 int count = 0;
487                 do {
488                     tOut.write(buffer, 0, count);
489                     count = in.read(buffer, 0, buffer.length);
490                 } while (count != -1);
491             }
492 
493             tOut.closeEntry();
494         } finally {
495             FileUtils.close(in);
496         }
497     }

551     /**
552      * Checks whether the archive is out-of-date with respect to the resources
553      * of the given collection.
554      *
555      * <p>Also checks that either all collections only contain file
556      * resources or this class supports non-file collections.</p>
557      *
558      * <p>And - in case of file-collections - ensures that the archive won't
559      * contain itself.</p>
560      *
561      * @param rc the resource collection to check
562      * @return whether the archive is up-to-date
563      * @since Ant 1.7
564      */
565     protected boolean check(ResourceCollection rc) {
566         boolean upToDate = true;
567         if (isFileFileSet(rc)) {
568             FileSet fs = (FileSet) rc;
569             upToDate = check(fs.getDir(getProject()), getFileNames(fs));
570         } else if (!rc.isFilesystemOnly() && !supportsNonFileResources()) {
571             throw new BuildException("only filesystem resources are supported");
572         } else if (rc.isFilesystemOnly()) {
573             HashSet basedirs = new HashSet();
574             HashMap basedirToFilesMap = new HashMap();
575             for (Resource res : rc) {
576                 FileResource r = ResourceUtils
577                     .asFileResource(res.as(FileProvider.class));
578                 File base = r.getBaseDir();
579                 if (base == null) {
580                     base = Copy.NULL_FILE_PLACEHOLDER;
581                 }
582                 basedirs.add(base);
583                 Vector files = (Vector) basedirToFilesMap.get(base);
584                 if (files == null) {
585                     files = new Vector();
586                     basedirToFilesMap.put(base, files);
587                 }
588                 if (base == Copy.NULL_FILE_PLACEHOLDER) {
589                     files.add(r.getFile().getAbsolutePath());
590                 } else {
591                     files.add(r.getName());
592                 }
593             }
594             Iterator iter = basedirs.iterator();
595             while (iter.hasNext()) {
596                 File base = (File) iter.next();
597                 Vector f = (Vector) basedirToFilesMap.get(base);
598                 String[] files = (String[]) f.toArray(new String[f.size()]);
599                 upToDate &=
600                     check(base == Copy.NULL_FILE_PLACEHOLDER ? null : base,
601                           files);
602             }
603         } else { // non-file resources
604             Iterator<Resource> iter = rc.iterator();
605             while (upToDate && iter.hasNext()) {
606                 Resource r = iter.next();
607                 upToDate = archiveIsUpToDate(r);
608             }
609         }
610         return upToDate;
611     }

613     /**
614      * <p>Checks whether the archive is out-of-date with respect to the
615      * given files, ensures that the archive won't contain itself.</p>
616      *
617      * @param basedir base directory for file names
618      * @param files array of relative file names
619      * @return whether the archive is up-to-date
620      * @since Ant 1.7
621      */
622     protected boolean check(File basedir, String[] files) {
623         boolean upToDate = true;
624         if (!archiveIsUpToDate(files, basedir)) {
625             upToDate = false;
626         }
627 
628         for (int i = 0; i < files.length; ++i) {
629             if (tarFile.equals(new File(basedir, files[i]))) {
630                 throw new BuildException("A tar file cannot include "
631                                          + "itself", getLocation());
632             }
633         }
634         return upToDate;
635     }

637     /**
638      * Adds the resources contained in this collection to the archive.
639      *
640      * <p>Uses the file based methods for file resources for backwards
641      * compatibility.</p>
642      *
643      * @param rc the collection containing resources to add
644      * @param tOut stream writing to the archive.
645      * @throws IOException on error.
646      * @since Ant 1.7
647      */
648     protected void tar(ResourceCollection rc, TarOutputStream tOut)
649         throws IOException {
650         ArchiveFileSet afs = null;
651         if (rc instanceof ArchiveFileSet) {
652             afs = (ArchiveFileSet) rc;
653         }
654         if (afs != null && afs.size() > 1
655             && afs.getFullpath(this.getProject()).length() > 0) {
656             throw new BuildException("fullpath attribute may only "
657                                      + "be specified for "
658                                      + "filesets that specify a "
659                                      + "single file.");
660         }
661         TarFileSet tfs = asTarFileSet(afs);
662 
663         if (isFileFileSet(rc)) {
664             FileSet fs = (FileSet) rc;
665             String[] files = getFileNames(fs);
666             for (int i = 0; i < files.length; i++) {
667                 File f = new File(fs.getDir(getProject()), files[i]);
668                 String name = files[i].replace(File.separatorChar, '/');
669                 tarFile(f, tOut, name, tfs);
670             }
671         } else if (rc.isFilesystemOnly()) {
672             for (Resource r : rc) {
673                 File f = r.as(FileProvider.class).getFile();
674                 tarFile(f, tOut, f.getName(), tfs);
675             }
676         } else { // non-file resources
677             for (Resource r : rc) {
678                 tarResource(r, tOut, r.getName(), tfs);
679             }
680         }
681     }

686      * @param rc the resource collection to check.

690     protected static boolean isFileFileSet(ResourceCollection rc) {
691         return rc instanceof FileSet && rc.isFilesystemOnly();
692     }

694     /**
695      * Grabs all included files and directors from the FileSet and
696      * returns them as an array of (relative) file names.
697      * @param fs the fileset to operate on.
698      * @return a list of the filenames.
699      * @since Ant 1.7
700      */
701     protected static String[] getFileNames(FileSet fs) {
702         DirectoryScanner ds = fs.getDirectoryScanner(fs.getProject());
703         String[] directories = ds.getIncludedDirectories();
704         String[] filesPerSe = ds.getIncludedFiles();
705         String[] files = new String [directories.length + filesPerSe.length];
706         System.arraycopy(directories, 0, files, 0, directories.length);
707         System.arraycopy(filesPerSe, 0, files, directories.length,
708                          filesPerSe.length);
709         return files;
710     }

712     /**
713      * Copies fullpath, prefix and permission attributes from the
714      * ArchiveFileSet to a new TarFileSet (or returns it unchanged if
715      * it already is a TarFileSet).
716      *
717      * @param archiveFileSet fileset to copy attributes from, may be null
718      * @return a new TarFileSet.
719      * @since Ant 1.7
720      */
721     protected TarFileSet asTarFileSet(ArchiveFileSet archiveFileSet) {
722         TarFileSet tfs = null;
723         if (archiveFileSet != null && archiveFileSet instanceof TarFileSet) {
724             tfs = (TarFileSet) archiveFileSet;
725         } else {
726             tfs = new TarFileSet();
727             tfs.setProject(getProject());
728             if (archiveFileSet != null) {
729                 tfs.setPrefix(archiveFileSet.getPrefix(getProject()));
730                 tfs.setFullpath(archiveFileSet.getFullpath(getProject()));
731                 if (archiveFileSet.hasFileModeBeenSet()) {
732                     tfs.integerSetFileMode(archiveFileSet
733                                            .getFileMode(getProject()));
734                 }
735                 if (archiveFileSet.hasDirModeBeenSet()) {
736                     tfs.integerSetDirMode(archiveFileSet
737                                           .getDirMode(getProject()));
738                 }
739 
740                 if (archiveFileSet
741                     instanceof org.apache.tools.ant.types.TarFileSet) {
742                     org.apache.tools.ant.types.TarFileSet t =
743                         (org.apache.tools.ant.types.TarFileSet) archiveFileSet;
744                     if (t.hasUserNameBeenSet()) {
745                         tfs.setUserName(t.getUserName());
746                     }
747                     if (t.hasGroupBeenSet()) {
748                         tfs.setGroup(t.getGroup());
749                     }
750                     if (t.hasUserIdBeenSet()) {
751                         tfs.setUid(t.getUid());
752                     }
753                     if (t.hasGroupIdBeenSet()) {
754                         tfs.setGid(t.getGid());
755                     }
756                 }
757             }
758         }
759         return tfs;
760     }

914     /**
915      * Valid Modes for Compression attribute to Tar Task
916      *
917      */
918     public static final class TarCompressionMethod extends EnumeratedAttribute {
919 
920         // permissible values for compression attribute
921         /**
922          *    No compression
923          */
924         private static final String NONE = "none";
925         /**
926          *    GZIP compression
927          */
928         private static final String GZIP = "gzip";
929         /**
930          *    BZIP2 compression
931          */
932         private static final String BZIP2 = "bzip2";
933 
934 
935         /**
936          * Default constructor
937          */
938         public TarCompressionMethod() {
939             super();
940             setValue(NONE);
941         }
942 
943         /**
944          *  Get valid enumeration values.
945          *  @return valid enumeration values
946          */
947         public String[] getValues() {
948             return new String[] {NONE, GZIP, BZIP2 };
949         }
950 
951         /**
952          *  This method wraps the output stream with the
953          *     corresponding compression method
954          *
955          *  @param ostream output stream
956          *  @return output stream with on-the-fly compression
957          *  @exception IOException thrown if file is not writable
958          */
959         private OutputStream compress(final OutputStream ostream)
960             throws IOException {
961             final String v = getValue();
962             if (GZIP.equals(v)) {
963                 return new GZIPOutputStream(ostream);
964             } else {
965                 if (BZIP2.equals(v)) {
966                     ostream.write('B');
967                     ostream.write('Z');
968                     return new CBZip2OutputStream(ostream);
969                 }
970             }
971             return ostream;
972         }
