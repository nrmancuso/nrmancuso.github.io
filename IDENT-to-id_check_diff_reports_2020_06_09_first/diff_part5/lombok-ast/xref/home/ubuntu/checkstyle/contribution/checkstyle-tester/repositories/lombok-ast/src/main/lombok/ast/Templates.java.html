249 @GenerateAstNode
250 class ModifiersTemplate {
251 	@ParentAccessor List<KeywordModifier> keywords2;
252 	@ParentAccessor List<Annotation> annotations1;
253 	
254 	/**
255 	 * Returns the keyword-based modifiers the way {@link java.lang.reflect.Modifiers} works.
256 	 * Only those keywords that are explicitly in the AST are reported; to also include implicit flags, such
257 	 * as for example the idea that methods in interfaces are always public and abstract even if not marked as such,
258 	 * use {@link #getEffectiveModifierFlags()}.
259 	 */
260 	@CopyMethod
261 	static int getExplicitModifierFlags(Modifiers m) {
262 		int out = 0;
263 		for (KeywordModifier n : m.astKeywords()) out |= n.asReflectModifiers();
264 		return out;
265 	}
266 	
267 	/**
268 	 * Returns the keyword-based modifiers the way {@link java.lang.reflect.Modifiers} works. Also sets flags that are implicitly true due to the nature
269 	 * of the node that the modifiers are attached to (for example, inner interfaces are implicitly static and thus if the Modifiers object is a child of
270 	 * such a declaration, its static bit will be set. Similarly, method declarations in interfaces are abstract and public whether or not those keywords
271 	 * have been applied to the node).
272 	 */
273 	@CopyMethod
274 	static int getEffectiveModifierFlags(Modifiers m) {
275 		int explicit = getExplicitModifierFlags(m);
276 		int out = explicit;
277 		Node declaration = m.getParent();
278 		
279 		// Interfaces and Enums can only be static by their very nature.
280 		if (declaration instanceof TypeDeclaration && !(declaration instanceof ClassDeclaration)) {
281 			out |= Modifier.STATIC;
282 		}
283 		
284 		// We consider top-level types as static, because semantically that makes sense.
285 		if (declaration instanceof ClassDeclaration && ((ClassDeclaration)declaration).upIfTopLevelToCompilationUnit() != null) {
286 			out |= Modifier.STATIC;
287 		}
288 		
289 		// We consider interfaces as abstract, because semantically that makes sense.
290 		if (declaration instanceof TypeDeclaration && ((TypeDeclaration)declaration).isInterface()) {
291 			out |= Modifier.ABSTRACT;
292 		}
293 		
294 		// Types in interfaces are always static.
295 		if (declaration instanceof ClassDeclaration) {
296 			TypeDeclaration container = ((ClassDeclaration)declaration).upUpToTypeDeclaration();
297 			if (container != null && container.isInterface()) out |= Modifier.STATIC;
298 		}
299 		
300 		if (declaration instanceof MethodDeclaration) {
301 			TypeDeclaration container = ((MethodDeclaration)declaration).upUpToTypeDeclaration();
302 			if (container != null && container.isInterface() && (explicit & Modifier.STATIC) == 0) {
303 				out |= Modifier.PUBLIC | Modifier.ABSTRACT;
304 			}
305 		}
306 		
307 		if (declaration instanceof VariableDeclaration) {
308 			TypeDeclaration container = ((VariableDeclaration)declaration).upUpToTypeDeclaration();
309 			if (container != null && container.isInterface()) {
310 				out |= Modifier.PUBLIC | Modifier.FINAL | Modifier.STATIC;
311 			}
312 		}
313 		
314 		return out;
315 	}
316 	
317 	@CopyMethod
318 	static boolean isPublic(Modifiers m) {
319 		return 0 != (getEffectiveModifierFlags(m) & java.lang.reflect.Modifier.PUBLIC);
320 	}
321 	
322 	@CopyMethod
323 	static boolean isProtected(Modifiers m) {
324 		return 0 != (getEffectiveModifierFlags(m) & java.lang.reflect.Modifier.PROTECTED);
325 	}
326 	
327 	@CopyMethod
328 	static boolean isPrivate(Modifiers m) {
329 		return 0 != (getEffectiveModifierFlags(m) & java.lang.reflect.Modifier.PRIVATE);
330 	}
331 	
332 	@CopyMethod
333 	static boolean isPackagePrivate(Modifiers m) {
334 		return 0 == (getEffectiveModifierFlags(m) & (
335 				Modifier.PUBLIC | Modifier.PRIVATE | Modifier.PROTECTED
336 				));
337 	}
338 	
339 	@CopyMethod
340 	static boolean isStatic(Modifiers m) {
341 		return 0 != (getEffectiveModifierFlags(m) & java.lang.reflect.Modifier.STATIC);
342 	}
343 	
344 	@CopyMethod
345 	static boolean isFinal(Modifiers m) {
346 		return 0 != (getEffectiveModifierFlags(m) & java.lang.reflect.Modifier.FINAL);
347 	}
348 	
349 	@CopyMethod
350 	static boolean isAbstract(Modifiers m) {

352 	}
353 	
354 	@CopyMethod
355 	static boolean isTransient(Modifiers m) {

357 	}
358 	
359 	@CopyMethod
360 	static boolean isEmpty(Modifiers m) {

362 	}

482 @GenerateAstNode(implementing={Expression.class, DescribedNode.class}, mixin=ExpressionMixin.class)
483 class BinaryExpressionTemplate {
484 	@Mandatory Expression left1;
485 	@Mandatory Expression right3;
486 	@NotChildOfNode(rawFormParser="parseOperator", rawFormGenerator="generateOperator")
487 	@Mandatory BinaryOperator operator2;
488 	
489 	@CopyMethod
490 	static String getDescription(BinaryExpression self) {
491 		try {
492 			return self.astOperator().getSymbol();
493 		} catch (Exception e) {
494 			return self.rawOperator();
495 		}
496 	}
497 	
498 	static String generateOperator(BinaryOperator op) {
499 		return op.getSymbol();
500 	}
501 	
502 	static BinaryOperator parseOperator(String op) {
503 		if (op == null) throw new IllegalArgumentException("missing operator");
504 		BinaryOperator result = BinaryOperator.fromSymbol(op.trim());
505 		if (result == null) throw new IllegalArgumentException("unknown binary operator: " + op.trim());
506 		return result;
507 	}
508 	
509 	@CopyMethod
510 	static boolean needsParentheses(Expression self) {
511 		try {
512 			return needsParentheses(self, ((BinaryExpression)self).astOperator().pLevel());
513 		} catch (Throwable ignore) {
514 			return true;
515 		}
516 	}
517 	
518 	static boolean needsParentheses(Node self, int pLevel) {
519 		Node parent = self.getParent();
520 		
521 		if (parent instanceof InlineIfExpression) {
522 			if (!(self instanceof InlineIfExpression)) {
523 				return pLevel >= BinaryOperator.ASSIGN.pLevel();
524 			}
525 			return ((InlineIfExpression)parent).rawIfFalse() != self;
526 		}
527 		
528 		if (parent instanceof UnaryExpression || parent instanceof Cast || parent instanceof ConstructorInvocation) {
529 			//If we're the child of a ConstructorInvocation, are we its parameter (definitely no parens needed) or its qualifier?
530 			if (parent instanceof ConstructorInvocation && ((ConstructorInvocation)parent).rawQualifier() != self) return false;
531 			final int otherPLevel = 1;
532 			if (otherPLevel > pLevel) return false;
533 			if (otherPLevel < pLevel) return true;
534 			
535 			boolean otherIsPostfix = false;
536 			boolean selfIsPostfix = false;
537 			
538 			try {
539 				if (parent instanceof ConstructorInvocation) otherIsPostfix = true;
540 				else otherIsPostfix = ((UnaryExpression)parent).astOperator().isPostfix();
541 			} catch (Throwable ignore) {}
542 			try {
543 				if (self instanceof ConstructorInvocation) selfIsPostfix = true;
544 				else selfIsPostfix = ((UnaryExpression)self).astOperator().isPostfix();
545 			} catch (Throwable ignore) {}
546 			return (!selfIsPostfix && otherIsPostfix);
547 		}
548 		
549 		if (parent instanceof ConstructorInvocation) return self == ((ConstructorInvocation)parent).rawQualifier();
550 		if (parent instanceof MethodInvocation) return self == ((MethodInvocation)parent).rawOperand();
551 		if (parent instanceof ArrayAccess) return self == ((ArrayAccess)parent).rawOperand();
552 		if (parent instanceof Select) return self == ((Select)parent).rawOperand();
553 		if (parent instanceof InstanceOf) return pLevel > BinaryOperator.LESS.pLevel();
554 		if (parent instanceof BinaryExpression) {
555 			BinaryExpression be = (BinaryExpression)parent;
556 			int otherPLevel;
557 			try {
558 				otherPLevel = be.astOperator().pLevel();
559 			} catch (Throwable ignore) {
560 				return true;
561 			}
562 			if (otherPLevel > pLevel) return false;
563 			if (otherPLevel < pLevel) return true;
564 			if (be.rawLeft() == self) {
565 				return pLevel == BinaryOperator.ASSIGN.pLevel();
566 			}
567 			if (be.rawRight() == self) {
568 				return pLevel != BinaryOperator.ASSIGN.pLevel();
569 			}
570 			return true;
571 		}
572 		
573 		return false;
574 	}
