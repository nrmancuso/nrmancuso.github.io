183 	private void set(ASTNode node, Node value) {
184 		if (result != null) throw new IllegalStateException("result is already set");
185 		
186 		if (value instanceof lombok.ast.Expression && hasFlag(FlagKey.AS_STATEMENT)) {
187 			lombok.ast.ExpressionStatement stat = new lombok.ast.ExpressionStatement();
188 			stat.astExpression((lombok.ast.Expression)value);
189 			int start = node.sourceStart;
190 			int end = node.sourceEnd;
191 			try {
192 				end = (Integer)node.getClass().getField("statementEnd").get(node);
193 			} catch (Exception e) {
194 				// Not all these classes may have a statementEnd.
195 			}
196 			
197 			set(node, stat.setPosition(toPosition(start, end)));
198 			return;
199 		}
200 		
201 		if (value instanceof lombok.ast.Expression) {
202 			int parenCount = (node.bits & ASTNode.ParenthesizedMASK) >> ASTNode.ParenthesizedSHIFT;
203 			for (int i = 0; i < parenCount; i++) {
204 				((lombok.ast.Expression) value).astParensPositions().add(value.getPosition());
205 			}
206 		}
207 		
208 		List<Node> result = Lists.newArrayList();
209 		if (value != null) result.add(value);
210 		this.result = result;
211 	}

227 	private Node toTree(ASTNode node, Map<FlagKey, Object> params) {
228 		if (node == null) return null;
229 		EcjTreeConverter newConverter = new EcjTreeConverter();
230 		if (params != null) newConverter.params = params;
231 		newConverter.visit(rawInput, node);
232 		try {
233 			return newConverter.get();
234 		} catch (RuntimeException e) {
235 			System.err.printf("Node '%s' (%s) did not produce any results\n", node, node.getClass().getSimpleName());
236 			throw e;
237 		}
238 	}

250 	private void fillList(ASTNode[] nodes, RawListAccessor<?, ?> list, Map<FlagKey, Object> params) {
251 		if (nodes == null) return;
252 		
253 		// int i, j; is represented with multiple AVDs, but in lombok.ast, it's 1 node. We need to
254 		// gather up sequential AVD nodes, check if the start position of each type is equal, and convert
255 		// them to one VariableDefinition by calling a special method.
256 		java.util.List<AbstractVariableDeclaration> varDeclQueue = new ArrayList<AbstractVariableDeclaration>();
257 		
258 		boolean fold = !params.containsKey(FlagKey.NO_VARDECL_FOLDING);
259 		
260 		for (ASTNode node : nodes) {
261 			if ((node instanceof FieldDeclaration || node instanceof LocalDeclaration) &&
262 					((AbstractVariableDeclaration)node).type != null) {
263 				
264 				if (fold && (varDeclQueue.isEmpty() || varDeclQueue.get(0).type.sourceStart == ((AbstractVariableDeclaration)node).type.sourceStart)) {
265 					varDeclQueue.add((AbstractVariableDeclaration) node);
266 					continue;
267 				} else {
268 					if (!varDeclQueue.isEmpty()) list.addToEnd(toVariableDefinition(varDeclQueue, params));
269 					varDeclQueue.clear();
270 					varDeclQueue.add((AbstractVariableDeclaration) node);
271 					continue;
272 				}
273 			}
274 			
275 			if (!varDeclQueue.isEmpty()) list.addToEnd(toVariableDefinition(varDeclQueue, params));
276 			varDeclQueue.clear();
277 			list.addToEnd(toTree(node, params));
278 		}
279 		
280 		if (!varDeclQueue.isEmpty()) list.addToEnd(toVariableDefinition(varDeclQueue, params));
281 	}

299 	private Node toVariableDefinition(List<AbstractVariableDeclaration> decls, Map<FlagKey, Object> params) {
300 		lombok.ast.VariableDefinition def = createVariableDefinition(decls, params);
301 		AbstractVariableDeclaration first = decls.get(0);
302 		def.setPosition(toPosition(first.declarationSourceStart, first.sourceEnd));
303 		
304 		if (params.containsKey(FlagKey.AS_DEFINITION)) return def;
305 		
306 		lombok.ast.VariableDeclaration decl = new lombok.ast.VariableDeclaration();
307 		if (first instanceof FieldDeclaration) {
308 			decl.astJavadoc((lombok.ast.Comment) toTree(((FieldDeclaration)first).javadoc));
309 		}
310 		
311 		decl.astDefinition(def);
312 		decl.setPosition(toPosition(first.declarationSourceStart, first.declarationEnd));
313 		return decl;
314 	}

316 	private lombok.ast.VariableDefinition createVariableDefinition(List<AbstractVariableDeclaration> decls, Map<FlagKey, Object> params) {
317 		int dims = Integer.MAX_VALUE;
318 		TypeReference winner = null;
319 		for (AbstractVariableDeclaration decl : decls) {
320 			TypeReference tr = decl.type;
321 			int newDims = tr.dimensions();
322 			if (newDims < dims) {
323 				dims = newDims;
324 				winner = tr;
325 			}
326 			if (dims == 0) break;
327 		}
328 		
329 		AbstractVariableDeclaration first = decls.get(0);
330 		lombok.ast.VariableDefinition varDef = new lombok.ast.VariableDefinition();
331 		varDef.astModifiers(toModifiers(first.modifiers, first.annotations, first.modifiersSourceStart, first.declarationSourceStart));
332 		varDef.astTypeReference((lombok.ast.TypeReference) toTree(winner));
333 		if ((first.type.bits & ASTNode.IsVarArgs) != 0) {
334 			varDef.astVarargs(true);
335 			setConversionPositionInfo(varDef, "typeref", toPosition(first.type.sourceStart, first.type.sourceEnd));
336 		}
337 		
338 		for (AbstractVariableDeclaration decl : decls) {
339 			lombok.ast.VariableDefinitionEntry varDefEntry = new lombok.ast.VariableDefinitionEntry();
340 			if (first instanceof FieldDeclaration) {
341 				setConversionPositionInfo(varDefEntry, "varDeclPart1", toPosition(decl.sourceStart, ((FieldDeclaration) decl).endPart1Position));
342 				setConversionPositionInfo(varDefEntry, "varDeclPart2", toPosition(decl.sourceStart, ((FieldDeclaration) decl).endPart2Position));
343 			}
344 			setConversionPositionInfo(varDefEntry, "declarationSource", toPosition(decl.declarationSourceStart, decl.declarationSourceEnd));
345 			setConversionPositionInfo(varDefEntry, "typeSourcePos", toPosition(decl.type.sourceStart, decl.type.sourceEnd));
346 			varDefEntry.astInitializer((lombok.ast.Expression) toTree(decl.initialization));
347 			varDefEntry.astName(toIdentifier(decl.name, decl.sourceStart, decl.sourceEnd));
348 			int delta = decl.type.dimensions() - winner.dimensions();
349 			varDefEntry.astArrayDimensions(delta);
350 			varDef.astVariables().addToEnd(varDefEntry);
351 		}
352 		return varDef;
353 	}

415 	private final EcjTreeVisitor visitor = new EcjTreeVisitor() {
416 		@Override public void visitCompilationUnitDeclaration(CompilationUnitDeclaration node) {
417 			lombok.ast.CompilationUnit unit = new lombok.ast.CompilationUnit();
418 			unit.rawPackageDeclaration(toTree(node.currentPackage, FlagKey.IMPORTDECLARATION_IS_PACKAGE));
419 			if (node.javadoc != null) {
420 				lombok.ast.PackageDeclaration lombokJavadoc = unit.astPackageDeclaration();
421 				if (lombokJavadoc != null) {
422 					lombokJavadoc.rawJavadoc(toTree(node.javadoc));
423 				}
424 			}
425 			fillList(node.imports, unit.rawImportDeclarations());
426 			
427 			TypeDeclaration[] newTypes = null;
428 			if (node.types != null && node.types.length > 0 && CharOperation.equals(EcjTreeBuilder.PACKAGE_INFO, node.types[0].name)) {
429 				newTypes = new TypeDeclaration[node.types.length - 1];
430 				System.arraycopy(node.types, 1, newTypes, 0, node.types.length - 1);
431 			} else {
432 				newTypes = node.types;
433 			}
434 			
435 			fillList(newTypes, unit.rawTypeDeclarations());
436 			set(node, unit);
437 		}
438 		
439 		@Override public void visitImportReference(ImportReference node) {
440 			if (hasFlag(FlagKey.IMPORTDECLARATION_IS_PACKAGE)) {
441 				lombok.ast.PackageDeclaration pkg = new lombok.ast.PackageDeclaration();
442 				fillIdentifiers(node.tokens, node.sourcePositions, pkg.astParts());
443 				fillList(node.annotations, pkg.rawAnnotations());
444 				pkg.setPosition(toPosition(node.declarationSourceStart, node.declarationSourceEnd));
445 				set(node, pkg);
446 				return;
447 			}
448 			
449 			lombok.ast.ImportDeclaration imp = new lombok.ast.ImportDeclaration();
450 			fillIdentifiers(node.tokens, node.sourcePositions, imp.astParts());
451 			imp.astStarImport((node.bits & ASTNode.OnDemand) != 0);
452 			imp.astStaticImport((node.modifiers & ClassFileConstants.AccStatic) != 0);
453 			imp.setPosition(toPosition(node.declarationSourceStart, node.declarationSourceEnd));
454 			set(node, imp);
455 		}
456 		
457 		@Override public void visitInitializer(Initializer node) {
458 			if ((node.modifiers & ClassFileConstants.AccStatic) != 0) {
459 				lombok.ast.StaticInitializer staticInit = new lombok.ast.StaticInitializer();
460 				staticInit.astBody((lombok.ast.Block) toTree(node.block));
461 				staticInit.setPosition(toPosition(node.declarationSourceStart, node.sourceEnd));
462 				set(node, staticInit);
463 				return;
464 			} else {
465 				lombok.ast.InstanceInitializer instanceInit = new lombok.ast.InstanceInitializer();
466 				instanceInit.astBody((lombok.ast.Block) toTree(node.block));
467 				set(node, setPosition(node, instanceInit));
468 				return;
469 			}
470 		}

472 		@Override 
473 		public void visitTypeDeclaration(TypeDeclaration node) {
474 			lombok.ast.TypeDeclaration decl = null;
475 			switch (TypeDeclaration.kind(node.modifiers)) {
476 				case TypeDeclaration.CLASS_DECL: {
477 					lombok.ast.ClassDeclaration cDecl = new lombok.ast.ClassDeclaration();
478 					
479 					cDecl.rawExtending(toTree(node.superclass));
480 					cDecl.astBody(createNormalTypeBody(node));
481 					fillList(node.superInterfaces, cDecl.rawImplementing());
482 					fillList(node.typeParameters, cDecl.rawTypeVariables());
483 					
484 					decl = cDecl;
485 					break;
486 				}
487 				case TypeDeclaration.INTERFACE_DECL: {
488 					lombok.ast.InterfaceDeclaration iDecl = new lombok.ast.InterfaceDeclaration();
489 					iDecl.astBody(createNormalTypeBody(node));
490 					fillList(node.superInterfaces, iDecl.rawExtending());
491 					fillList(node.typeParameters, iDecl.rawTypeVariables());
492 					
493 					decl = iDecl;
494 					break;
495 				}
496 				case TypeDeclaration.ENUM_DECL: {
497 					lombok.ast.EnumDeclaration eDecl = new lombok.ast.EnumDeclaration();
498 					lombok.ast.EnumTypeBody enumTypeBody = createEnumTypeBody(node);
499 					
500 					fillList(node.superInterfaces, eDecl.rawImplementing());
501 					eDecl.astBody(enumTypeBody);

507 					aDecl.astBody(createNormalTypeBody(node));

509 					decl = aDecl;

511 				}
512 			}
513 			decl.astJavadoc((lombok.ast.Comment) toTree(node.javadoc));
514 			decl.astModifiers(toModifiers(node.modifiers, node.annotations, node.modifiersSourceStart, node.declarationSourceStart));
515 			decl.astName(toIdentifier(node.name, node.sourceStart, node.sourceEnd));
516 			decl.setPosition(toPosition(node.declarationSourceStart, node.declarationSourceEnd));
517
