25  package sun.swing.text;
26  
27  import java.util.*;
28  import java.awt.Graphics;
29  import java.awt.print.*;
30  
31  
32  /**
33   * Printable to merge multiple printables into one.
34   *
35   * @author Igor Kushnirskiy
36   *
37   * @since 1.6
38   */
39  class CompoundPrintable implements CountingPrintable {
40      private final Queue<CountingPrintable> printables;
41      private int offset = 0;
42  
43      public CompoundPrintable(List<CountingPrintable> printables) {
44          this.printables = new LinkedList<CountingPrintable>(printables);
45      }
46  
47      public int print(final Graphics graphics,
48                       final PageFormat pf,
49                       final int pageIndex) throws PrinterException {
50          int ret = NO_SUCH_PAGE;
51          while (printables.peek() != null) {
52              ret = printables.peek().print(graphics, pf, pageIndex - offset);
53              if (ret == PAGE_EXISTS) {
54                  break;
55              } else {
56                  offset += printables.poll().getNumberOfPages();
57              }
58          }
59          return ret;
60      }
61  
62      /**
63       * Returns the number of pages in this printable.
64       * <p>
65       * This number is defined only after {@code print} returns NO_SUCH_PAGE.
66       *
67       * @return the number of pages.
68       */
69      public int getNumberOfPages() {
70          return offset;
71      }
72  
73  }
