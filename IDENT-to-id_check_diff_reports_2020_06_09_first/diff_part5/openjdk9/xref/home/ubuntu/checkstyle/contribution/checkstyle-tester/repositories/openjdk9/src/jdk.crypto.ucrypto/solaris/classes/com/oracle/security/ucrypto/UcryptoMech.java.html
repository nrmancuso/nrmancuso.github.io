141     private int mech = 0;
142     private final ServiceDesc[] serviceDescs;
143 
144     private static ServiceDesc sd(String type, String algo, String cn, String... aliases) {
145         return new ServiceDesc(type, algo, cn, aliases);
146     }
147 
148     UcryptoMech(ServiceDesc[] serviceDescs) {
149         this.serviceDescs = serviceDescs;
150     }
151 
152     public void setValue(int nativeMechValue) {

154     }
155 
156     public int value() { return mech; }
157     public ServiceDesc[] getServiceDescriptions() { return serviceDescs; }
