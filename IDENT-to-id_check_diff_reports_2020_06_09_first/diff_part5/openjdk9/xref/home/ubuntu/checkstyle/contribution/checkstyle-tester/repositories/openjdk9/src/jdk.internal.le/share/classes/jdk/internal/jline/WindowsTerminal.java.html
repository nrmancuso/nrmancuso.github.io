27  /**
28   * Terminal implementation for Microsoft Windows. Terminal initialization in
29   * {@link #init} is accomplished by extracting the
30   * <em>jline_<i>version</i>.dll</em>, saving it to the system temporary
31   * directoy (determined by the setting of the <em>java.io.tmpdir</em> System
32   * property), loading the library, and then calling the Win32 APIs <a
33   * href="http://msdn.microsoft.com/library/default.asp?
34   * url=/library/en-us/dllproc/base/setconsolemode.asp">SetConsoleMode</a> and
35   * <a href="http://msdn.microsoft.com/library/default.asp?
36   * url=/library/en-us/dllproc/base/getconsolemode.asp">GetConsoleMode</a> to
37   * disable character echoing.
38   * <p/>
39   * <p>
40   * By default, the {@link #wrapInIfNeeded(java.io.InputStream)} method will attempt
41   * to test to see if the specified {@link InputStream} is {@link System#in} or a wrapper
42   * around {@link FileDescriptor#in}, and if so, will bypass the character reading to
43   * directly invoke the readc() method in the JNI library. This is so the class
44   * can read special keys (like arrow keys) which are otherwise inaccessible via
45   * the {@link System#in} stream. Using JNI reading can be bypassed by setting
46   * the <code>jline.WindowsTerminal.directConsole</code> system property
47   * to <code>false</code>.
48   * </p>
49   *
50   * @author <a href="mailto:mwp1@cornell.edu">Marc Prud'hommeaux</a>
51   * @author <a href="mailto:jason@planet57.com">Jason Dillon</a>
52   * @since 2.0
53   */

111     @Override
112     public void setEchoEnabled(final boolean enabled) {
113         // Must set these four modes at the same time to make it work fine.
114         if (enabled) {
115             setConsoleMode(getConsoleMode() |
116                 ENABLE_ECHO_INPUT.code |
117                 ENABLE_LINE_INPUT.code |
118                 ENABLE_PROCESSED_INPUT.code |
119                 ENABLE_WINDOW_INPUT.code);
120         }
121         else {
122             setConsoleMode(getConsoleMode() &
123                 ~(ENABLE_LINE_INPUT.code |
124                     ENABLE_ECHO_INPUT.code |
125                     ENABLE_PROCESSED_INPUT.code |
126                     ENABLE_WINDOW_INPUT.code));
127         }
128         super.setEchoEnabled(enabled);
129     }

147     @Override
148     public InputStream wrapInIfNeeded(InputStream in) throws IOException {
149         if (directConsole && isSystemIn(in)) {
150             return new InputStream() {
151                 private byte[] buf = null;
152                 int bufIdx = 0;
153 
154                 @Override
155                 public int read() throws IOException {
156                     while (buf == null || bufIdx == buf.length) {
157                         buf = readConsoleInput();
158                         bufIdx = 0;
159                     }
160                     int c = buf[bufIdx] & 0xFF;
161                     bufIdx++;
162                     return c;
163                 }
164             };
165         } else {
166             return super.wrapInIfNeeded(in);
167         }
168     }

170     protected boolean isSystemIn(final InputStream in) throws IOException {
171         if (in == null) {
172             return false;
173         }
174         else if (in == System.in) {
175             return true;
176         }
177         else if (in instanceof FileInputStream && ((FileInputStream) in).getFD() == FileDescriptor.in) {
178             return true;
179         }
180 
181         return false;
182     }

184     @Override
185     public String getOutputEncoding() {
186         int codepage = getConsoleOutputCodepage();
187         //http://docs.oracle.com/javase/6/docs/technotes/guides/intl/encoding.doc.html
188         String charsetMS = "ms" + codepage;
189         if (java.nio.charset.Charset.isSupported(charsetMS)) {
190             return charsetMS;
191         }
192         String charsetCP = "cp" + codepage;
193         if (java.nio.charset.Charset.isSupported(charsetCP)) {
194             return charsetCP;
195         }
196         Log.debug("can't figure out the Java Charset of this code page (" + codepage + ")...");
197         return super.getOutputEncoding();
198     }

321     private byte[] readConsoleInput() {
322         KEY_EVENT_RECORD keyEvent = readKeyEvent();
323 
324         return convertKeys(keyEvent).getBytes();
325     }

327     public static String convertKeys(KEY_EVENT_RECORD keyEvent) {
328         if (keyEvent == null) {
329             return "";
330         }
331 
332         StringBuilder sb = new StringBuilder();
333 
334         if (keyEvent.keyDown) {
335             if (keyEvent.uchar > 0) {
336                 // support some C1 control sequences: ALT + [@-_] (and [a-z]?) => ESC <ascii>
337                 // http://en.wikipedia.org/wiki/C0_and_C1_control_codes#C1_set
338                 final int altState = KEY_EVENT_RECORD.ALT_PRESSED;
339                 // Pressing "Alt Gr" is translated to Alt-Ctrl, hence it has to be checked that Ctrl is _not_ pressed,
340                 // otherwise inserting of "Alt Gr" codes on non-US keyboards would yield errors
341                 final int ctrlState = KEY_EVENT_RECORD.CTRL_PRESSED;
342 
343                 boolean handled = false;
344 
345                 if ((keyEvent.controlKeyState & ctrlState) != 0) {
346                     switch (keyEvent.keyCode) {
347                         case 0x43: //Ctrl-C
348                             sb.append("\003");
349                             handled = true;
350                             break;
351                     }
352                 }
353 
354                 if ((keyEvent.controlKeyState & KEY_EVENT_RECORD.SHIFT_PRESSED) != 0) {
355                     switch (keyEvent.keyCode) {
356                         case 0x09: //Shift-Tab
357                             sb.append("\033\133\132");
358                             handled = true;
359                             break;
360                     }
361                 }
362 
363                 if (!handled) {
364                     if (((keyEvent.uchar >= '@' && keyEvent.uchar <= '_') || (keyEvent.uchar >= 'a' && keyEvent.uchar <= 'z'))
365                         && ((keyEvent.controlKeyState & altState) != 0) && ((keyEvent.controlKeyState & ctrlState) == 0)) {
366                         sb.append('\u001B'); // ESC
367                     }
368 
369                     sb.append(keyEvent.uchar);
370                 }
371             } else {
372                 // virtual keycodes: http://msdn.microsoft.com/en-us/library/windows/desktop/dd375731(v=vs.85).aspx
373                 // xterm escape codes: E. Moy, S. Gildea and T. Dickey, "XTerm Control Sequences":
374                 // http://invisible-island.net/xterm/ctlseqs/ctlseqs.html
375                 // http://xorg.freedesktop.org/releases/X11R6.8.1/PDF/ctlseqs.pdf
376                 // just add support for basic editing keys and function keys
377                 String escapeSequence = null;
378                 switch (keyEvent.keyCode) {
379                 case 0x21: // VK_PRIOR PageUp
380                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[5~", "\u001B[5;%d~");
381                     break;
382                 case 0x22: // VK_NEXT PageDown
383                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[6~", "\u001B[6;%d~");
384                     break;
385                 case 0x23: // VK_END
386                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[4~", "\u001B[4;%d~");
387                     break;
388                 case 0x24: // VK_HOME
389                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[1~", "\u001B[1;%d~");
390                     break;
391                 case 0x25: // VK_LEFT
392                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[D", "\u001B[1;%dD");
393                     break;
394                 case 0x26: // VK_UP
395                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[A", "\u001B[1;%dA");
396                     break;
397                 case 0x27: // VK_RIGHT
398                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[C", "\u001B[1;%dC");
399                     break;
400                 case 0x28: // VK_DOWN
401                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[B", "\u001B[1;%dB");
402                     break;
403                 case 0x2D: // VK_INSERT
404                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[2~", "\u001B[2;%d~");
405                     break;
406                 case 0x2E: // VK_DELETE
407                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[3~", "\u001B[3;%d~");
408                     break;
409                 case 0x70: // VK_F1
410                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001BOP", "\u001BO%dP");
411                     break;
412                 case 0x71: // VK_F2
413                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001BOQ", "\u001BO%dQ");
414                     break;
415                 case 0x72: // VK_F3
416                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001BOR", "\u001BO%dR");
417                     break;
418                 case 0x73: // VK_F4
419                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001BOS", "\u001BO%dS");
420                     break;
421                 case 0x74: // VK_F5
422                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[15~", "\u001B[15;%d~");
423                     break;
424                 case 0x75: // VK_F6
425                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[17~", "\u001B[17;%d~");
426                     break;
427                 case 0x76: // VK_F7
428                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[18~", "\u001B[18;%d~");
429                     break;
430                 case 0x77: // VK_F8
431                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[19~", "\u001B[19;%d~");
432                     break;
433                 case 0x78: // VK_F9
434                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[20~", "\u001B[20;%d~");
435                     break;
436                 case 0x79: // VK_F10
437                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[21~", "\u001B[21;%d~");
438                     break;
439                 case 0x7A: // VK_F11
440                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[23~", "\u001B[23;%d~");
441                     break;
442                 case 0x7B: // VK_F12
443                     escapeSequence = escapeSequence(keyEvent.controlKeyState, "\u001B[24~", "\u001B[24;%d~");
444                     break;
445                 default:
446                     break;
447                 }
448                 if (escapeSequence != null) {
449                     for (int k = 0; k < keyEvent.repeatCount; k++) {
450                         sb.append(escapeSequence);
451                     }
452                 }
453             }
454         } else {
455             // key up event
456             // support ALT+NumPad input method
457             if (keyEvent.keyCode == 0x12/*VK_MENU ALT key*/ && keyEvent.uchar > 0) {
458                 sb.append(keyEvent.uchar);
459             }
460         }
461         return sb.toString();
462     }

464     private static String escapeSequence(int controlKeyState, String noControlSequence, String withControlSequence) {
465         int controlNum = 1;
466 
467         if ((controlKeyState & KEY_EVENT_RECORD.SHIFT_PRESSED) != 0) {
468             controlNum += 1;
469         }
470 
471         if ((controlKeyState & KEY_EVENT_RECORD.ALT_PRESSED) != 0) {
472             controlNum += 2;
473         }
474 
475         if ((controlKeyState & KEY_EVENT_RECORD.CTRL_PRESSED) != 0) {
476             controlNum += 4;
477         }
478 
479         if (controlNum > 1) {
480             return String.format(withControlSequence, controlNum);
481         } else {
482             return noControlSequence;
483         }
484     }

494         public final int controlKeyState;
495         public final int keyCode;
496         public final int repeatCount;

498         public KEY_EVENT_RECORD(boolean keyDown, char uchar, int controlKeyState, int keyCode, int repeatCount) {
499             this.keyDown = keyDown;
500             this.uchar = uchar;
501             this.controlKeyState = controlKeyState;
502             this.keyCode = keyCode;
503             this.repeatCount = repeatCount;
504         }

514     /**
515      * Console mode
516      * <p/>
517      * Constants copied <tt>wincon.h</tt>.
518      */
519     public static enum ConsoleMode
520     {
521         /**
522          * The ReadFile or ReadConsole function returns only when a carriage return
523          * character is read. If this mode is disable, the functions return when one
524          * or more characters are available.
525          */
526         ENABLE_LINE_INPUT(2),
527 
528         /**
529          * Characters read by the ReadFile or ReadConsole function are written to
530          * the active screen buffer as they are read. This mode can be used only if
531          * the ENABLE_LINE_INPUT mode is also enabled.
532          */
533         ENABLE_ECHO_INPUT(4),
534 
535         /**
536          * CTRL+C is processed by the system and is not placed in the input buffer.
537          * If the input buffer is being read by ReadFile or ReadConsole, other
538          * control keys are processed by the system and are not returned in the
539          * ReadFile or ReadConsole buffer. If the ENABLE_LINE_INPUT mode is also
540          * enabled, backspace, carriage return, and linefeed characters are handled
541          * by the system.
542          */
543         ENABLE_PROCESSED_INPUT(1),
544 
545         /**
546          * User interactions that change the size of the console screen buffer are
547          * reported in the console's input buffee. Information about these events
548          * can be read from the input buffer by applications using
549          * theReadConsoleInput function, but not by those using ReadFile
550          * orReadConsole.
551          */
552         ENABLE_WINDOW_INPUT(8),
553 
554         /**
555          * If the mouse pointer is within the borders of the console window and the
556          * window has the keyboard focus, mouse events generated by mouse movement
557          * and button presses are placed in the input buffer. These events are
558          * discarded by ReadFile or ReadConsole, even when this mode is enabled.
559          */
560         ENABLE_MOUSE_INPUT(16),
561 
562         /**
563          * When enabled, text entered in a console window will be inserted at the
564          * current cursor location and all text following that location will not be
565          * overwritten. When disabled, all following text will be overwritten. An OR
566          * operation must be performed with this flag and the ENABLE_EXTENDED_FLAGS
567          * flag to enable this functionality.
568          */
569         ENABLE_PROCESSED_OUTPUT(1),
570 
571         /**
572          * This flag enables the user to use the mouse to select and edit text. To
573          * enable this option, use the OR to combine this flag with
574          * ENABLE_EXTENDED_FLAGS.
575          */
576         ENABLE_WRAP_AT_EOL_OUTPUT(2),;
577 
578         public final int code;
579 
580         ConsoleMode(final int code) {
581             this.code = code;
582         }
583     }
