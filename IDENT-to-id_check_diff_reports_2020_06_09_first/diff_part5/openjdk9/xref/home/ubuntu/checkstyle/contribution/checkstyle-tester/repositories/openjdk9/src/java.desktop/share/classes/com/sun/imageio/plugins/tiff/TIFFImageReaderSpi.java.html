25  package com.sun.imageio.plugins.tiff;
26  
27  import java.io.IOException;
28  import java.util.Locale;
29  import javax.imageio.ImageReader;
30  import javax.imageio.spi.ImageReaderSpi;
31  import javax.imageio.spi.ServiceRegistry;
32  import javax.imageio.stream.ImageInputStream;
33  
34  public class TIFFImageReaderSpi extends ImageReaderSpi {
35  
36      private boolean registered = false;
37  
38      public TIFFImageReaderSpi() {
39          super("Oracle Corporation",
40                "1.0",
41                new String[] {"tif", "TIF", "tiff", "TIFF"},
42                new String[] {"tif", "tiff"},
43                new String[] {"image/tiff"},
44                "com.sun.imageio.plugins.tiff.TIFFImageReader",
45                new Class<?>[] { ImageInputStream.class },
46                new String[] {"com.sun.imageio.plugins.tiff.TIFFImageWriterSpi"},
47                false,
48                TIFFStreamMetadata.NATIVE_METADATA_FORMAT_NAME,
49                "com.sun.imageio.plugins.tiff.TIFFStreamMetadataFormat",
50                null, null,
51                true,
52                TIFFImageMetadata.NATIVE_METADATA_FORMAT_NAME,
53                "com.sun.imageio.plugins.tiff.TIFFImageMetadataFormat",
54                null, null
55                );
56      }
57  
58      public String getDescription(Locale locale) {
59          return "Standard TIFF image reader";
60      }
61  
62      public boolean canDecodeInput(Object input) throws IOException {
63          if (!(input instanceof ImageInputStream)) {
64              return false;
65          }
66  
67          ImageInputStream stream = (ImageInputStream)input;
68          byte[] b = new byte[4];
69          stream.mark();
70          stream.readFully(b);
71          stream.reset();
72  
73          return ((b[0] == (byte)0x49 && b[1] == (byte)0x49 &&
74                   b[2] == (byte)0x2a && b[3] == (byte)0x00) ||
75                  (b[0] == (byte)0x4d && b[1] == (byte)0x4d &&
76                   b[2] == (byte)0x00 && b[3] == (byte)0x2a));
77      }
78  
79      public ImageReader createReaderInstance(Object extension) {
80          return new TIFFImageReader(this);
81      }
82  
83      public void onRegistration(ServiceRegistry registry,
84                                 Class<?> category) {
85          if (registered) {
86              return;
87          }
88  
89          registered = true;
90      }
91  }
