25     @bug 4239446
26     @summary Make sure the ZipEntry fields are correct.
27   */
28  
29  import java.io.*;
30  import java.util.zip.*;
31  
32  public class CopyJar {
33      public static void main(String args[]) throws Exception {
34          try (ZipFile zf = new ZipFile(new File(System.getProperty("test.src", "."),
35                                                 "input.jar"))) {
36              ZipEntry ze = zf.getEntry("ReleaseInflater.java");
37              ZipOutputStream zos = new ZipOutputStream(new ByteArrayOutputStream());
38              InputStream in = zf.getInputStream(ze);
39              byte[] b = new byte[128];
40              int n;
41              zos.putNextEntry(ze);
42              while((n = in.read(b)) != -1) {
43                  zos.write(b, 0, n);
44              }
45              zos.close();
46          }
47      }
48  }
