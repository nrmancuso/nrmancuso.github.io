49  /**
50   * @author franz.willer
51   *
52   * The Model for Teaching File Selector WEB interface.
53   */
54  public class TFModel extends BasicFormModel {
55  
56      public static final String[] DOC_TITLES = new String[] {
57          "TCE001", "IHERADTF", "For Teaching File Export",
58          "TCE002", "IHERADTF", "For Clinical Trial Export",
59          "TCE007", "IHERADTF", "For Research Collection Export",
60          "113019", "99DCM4CHE", "For Media Export"};
61  
62      public static final String[] DELAY_REASONS = new String[] {
63          "TCE011" ,"IHERADTF", "Delay export until final report is available",
64          "TCE012", "IHERADTF", "Delay export until clinical information is available",
65          "TCE013", "IHERADTF", "Delay export until confirmation of diagnosis is available",
66          "TCE014", "IHERADTF", "Delay export until histopathology is available",
67          "TCE015", "IHERADTF", "Delay export until other laboratory results is available",
68          "TCE016", "IHERADTF", "Delay export until patient is discharged",
69          "TCE017", "IHERADTF", "Delay export until patient dies",
70          "TCE018", "IHERADTF", "Delay export until expert review is available"};
71  
72      private static final int CODE = 0;
73      private static final int DESIGNATOR = 1;
74      private static final int MEANING = 2;
75  
76      /** The session attribute name to store the model in http session. */
77      public static final String TF_ATTR_NAME = "tfModel";
78  
79      private SRManifestModel manifestModel;
80  
81      private String user;
82  
83      private Set instances;
84  
85      private int selectedTitle;
86      private int selectedDelayReason = -1;
87      private Collection dispositions;
88      private String disposition;
89  
90      /**
91       * Creates the model.
92       * <p>
93       */
94      private TFModel(String user, HttpServletRequest request) {
95          super(request);
96          this.user = user;
97          manifestModel = new SRManifestModel();
98      }
99  
100 
101     /**
102      * @return Returns the user.
103      */
104     public String getUser() {
105         return user;
106     }
107     /**
108      * @return Returns the manifestModel.
109      */
110     public SRManifestModel getManifestModel() {
111         return manifestModel;
112     }
113     /**
114      * @param disposition The disposition to set.
115      */
116     public void setDisposition(String disposition) {
117         if ( disposition != null ) {
118             this.disposition = disposition;
119         }
120     }
121     /**
122      * @param selectedDelayReason The selectedDelayReason to set.
123      */
124     public void setSelectedDelayReason(int selectedDelayReason) {
125         this.selectedDelayReason = selectedDelayReason;
126     }
127     /**
128      * @param selectedTitle The selectedTitle to set.
129      */
130     public void setSelectedTitle(int selectedTitle) {
131         this.selectedTitle = selectedTitle;
132     }
133 
134     /**
135      * Get the model for an http request.
136      * <p>
137      * Look in the session for an associated model via <code>TF_ATTR_NAME</code><br>
138      * If there is no model stored in session (first request) a new model is created and stored in session.
139      * 
140      * @param request A http request.
141      * 
142      * @return The model for given request.
143      */
144     public static final TFModel getModel( HttpServletRequest request ) {
145         TFModel model = (TFModel) request.getSession().getAttribute(TF_ATTR_NAME);
146         if (model == null) {
147             model = new TFModel(request.getUserPrincipal().getName(), request);
148             request.getSession().setAttribute(TF_ATTR_NAME, model);
149         }
150         return model;
151     }
152 
153     public String getModelName() { return "TF"; }
154 
155     private String selectFrom(int off, int index, String[] values) {
156         return index < 0 ? null : values[off + index * 3];
157     }
158 
159     private String[] meaning(String[] values) {
160         String[] ss = new String[values.length / 3];
161         for (int i = 0; i < ss.length; i++) {
162             ss[i] = selectFrom(MEANING, i, values);           
163         }
164         return ss;
165     }
166 
167 
168     public String[] getDocTitles() {
169         return meaning(DOC_TITLES);
170     }
171 
172     public String[] getDelayReasons() {
173         return meaning(DELAY_REASONS);
174     }
175 
176     /**
177      * @return Returns the instances.
178      */
179     public Set getInstances() {
180         return instances;
181     }
182     /**
183      * @param instances The instances to set.
184      */
185     public void setInstances(Set instances) {
186         this.instances = instances;
187     }
188 
189     public int getNumberOfInstances() {
190         return instances == null ? 0 : instances.size();
191     }
192 
193     public int getSelectedDocTitle() {
194         return selectedTitle;
195     }
196 
197     public String selectedDocTitle() {
198         return selectFrom(MEANING, selectedTitle, DOC_TITLES);
199     }    
200 
201     public String selectedDocTitleCode() {
202         return selectFrom(CODE, selectedTitle, DOC_TITLES);
203     }
204 
205     public String selectedDocTitleDesignator() {

207     }
208 
209     public int getSelectedDelayReason() {
210         return selectedDelayReason;
211     }
212 
213     public String selectedDelayReason() {
214         return selectFrom(MEANING, selectedDelayReason, DELAY_REASONS);
215     }
216 
217     public String selectedDelayReasonCode() {
218         return selectFrom(CODE, selectedDelayReason, DELAY_REASONS);
219     }
220 
221     public String selectedDelayReasonDesignator() {
222         return selectFrom(DESIGNATOR, selectedDelayReason, DELAY_REASONS);
223     }
224 
225     public String getDisposition() {
226         return this.disposition;
227     }
228     /**
229      * @param configuredDispositions
230      */
231     public void setDispositions(Collection dispositions) {
232         this.dispositions = dispositions;
233     }
234     /**
235      * @return Returns the list of configured dispositions.
236      */
237     public Collection getDispositions() {
238         return dispositions;
239     }
240 
241 
242     /**
243      * 
244      */
245     public void clear() {
246         selectedTitle = 0;
247         selectedDelayReason = -1;
248         disposition = null;
249         manifestModel.clear();
250     }
