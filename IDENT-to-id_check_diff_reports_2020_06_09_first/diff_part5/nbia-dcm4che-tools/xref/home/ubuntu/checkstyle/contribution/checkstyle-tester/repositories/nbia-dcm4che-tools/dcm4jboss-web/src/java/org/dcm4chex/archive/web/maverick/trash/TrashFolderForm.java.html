53   * @author franz.willer@gwi-ag.com
54   * @version $Revision: 11261 $ $Date: 2009-05-08 10:15:19 +0200 (Fr, 08 Mai 2009) $
55   * @since 19.12.2005

57  public class TrashFolderForm extends BasicFolderForm {
58  
59      static final String FOLDER_ATTRNAME = "trashFolderFrom";
60  
61      protected static Logger log = Logger.getLogger(TrashFolderForm.class);
62  
63      private Dataset ds = DcmObjectFactory.getInstance().newDataset();
64  
65      static TrashFolderForm getTrashFolderForm(ControllerContext ctx) {
66          HttpServletRequest request = ctx.getRequest();
67          TrashFolderForm form = (TrashFolderForm) request.getSession()
68          .getAttribute(FOLDER_ATTRNAME);
69          if (form == null) {
70              form = new TrashFolderForm(request);
71              request.getSession().setAttribute(FOLDER_ATTRNAME, form);
72              initLimit(ctx.getServletConfig().getInitParameter("limitNrOfStudies"), form);
73          }
74          initLimit(request.getParameter("limitNrOfStudies"), form);
75          form.clearPopupMsg();
76  
77          return form;
78      }
79  
80      private TrashFolderForm( HttpServletRequest request ) {
81          super(request);
82      }
83  
84      public final String getPatientID() {
85          return ds.getString(Tags.PatientID);
86      }
87  
88      public final void setPatientID(String patientID) {
89          ds.putLO(Tags.PatientID, patientID);
90      }
91  
92      public final String getPatientName() {
93          return ds.getString(Tags.PatientName);
94      }
95  
96      public final void setPatientName(String patientName) {
97          ds.putPN(Tags.PatientName, patientName);
98      }
99  
100     /**
101      * @return Returns the studyUID.
102      */
103     public String getStudyUID() {
104         return ds.getString(Tags.StudyInstanceUID);
105     }
106     /**
107      * @param studyUID The studyUID to set.
108      */
109     public void setStudyUID(String studyUID) {
110         ds.putUI(Tags.StudyInstanceUID, studyUID);
111     }
112     public final String getAccessionNumber() {
113         return ds.getString(Tags.AccessionNumber);
114     }
115 
116     public final void setAccessionNumber(String accessionNumber) {
117         ds.putSH(Tags.AccessionNumber, accessionNumber);
118     }
119 
120     public final void setCallingAETs(String[] aets ) {
121         ds.setPrivateCreatorID(PrivateTags.CreatorID);
122         ds.putAE(PrivateTags.CallingAET, aets);
123     }
124 
125     public Dataset filterDS() {
126         return ds;
127     }
128 
129     public String getModelName() { return "TRASH"; }
130 
131     /* (non-Javadoc)
132      * @see org.dcm4chex.archive.web.maverick.BasicFormPagingModel#gotoCurrentPage()
133      */
134     public void gotoCurrentPage() {
135         //We doesnt need this method here. TrashFolderCtrl does not use performPrevious/performNext!
136     }
137 
138 }
