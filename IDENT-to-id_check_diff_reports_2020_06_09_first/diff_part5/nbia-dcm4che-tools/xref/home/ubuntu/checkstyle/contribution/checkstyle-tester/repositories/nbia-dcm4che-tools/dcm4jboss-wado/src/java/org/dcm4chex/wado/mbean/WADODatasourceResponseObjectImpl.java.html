40  package org.dcm4chex.wado.mbean;
41  
42  import java.io.IOException;
43  import java.io.OutputStream;
44  
45  import javax.xml.transform.TransformerConfigurationException;
46  
47  import org.dcm4che.data.Dataset;
48  import org.dcm4che.data.DcmEncodeParam;
49  import org.dcm4che.net.DataSource;
50  import org.xml.sax.SAXException;
51  
52  /**
53   * @author franz.willer
54   *
55   * TODO To change the template for this generated type comment go to
56   * Window - Preferences - Java - Code Style - Code Templates
57   */
58  public class WADODatasourceResponseObjectImpl extends BasicWADOResponseObject {
59  
60  	private DataSource datasource;
61  	private Dataset dataset;
62  	String transferSyntax = null;
63  
64  	public WADODatasourceResponseObjectImpl( DataSource ds, String ts, String contentType, int retCode, String errMsg ) {
65  		super(contentType,retCode,errMsg);
66  		this.datasource = ds;
67  		this.transferSyntax = ts;
68  	}
69  	public WADODatasourceResponseObjectImpl( Dataset ds, String ts, String contentType, int retCode, String errMsg ) {
70  		super(contentType,retCode,errMsg);
71  		this.dataset = ds;
72  		this.transferSyntax = ts;
73  	}
74  	
75  	/* (non-Javadoc)
76  	 * @see org.dcm4chex.wado.common.WADOResponseObject#getFile()
77  	 */
78  	public void execute( OutputStream out ) throws TransformerConfigurationException, SAXException, IOException {
79  		if ( datasource != null ) {
80  			datasource.writeTo( out, transferSyntax );
81  		} else if ( dataset != null ) {
82  			dataset.writeFile( out, DcmEncodeParam.valueOf( transferSyntax ));
83  		} else {
84  			throw new IllegalArgumentException("Cant execute WADO datasource response! Neither a datasource nor dataset object is set!");
85  		}
86  	}
87  }
