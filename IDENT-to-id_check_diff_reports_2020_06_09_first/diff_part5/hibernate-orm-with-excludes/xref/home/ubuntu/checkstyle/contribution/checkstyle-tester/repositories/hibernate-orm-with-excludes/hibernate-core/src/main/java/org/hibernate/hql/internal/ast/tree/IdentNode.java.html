45  /**
46   * Represents an identifier all by itself, which may be a function name,
47   * a class alias, or a form of naked property-ref depending on the
48   * context.
49   *
50   * @author josh
51   */
52  public class IdentNode extends FromReferenceNode implements SelectExpression {
53  
54  	private static final int UNKNOWN = 0;
55  	private static final int PROPERTY_REF = 1;
56  	private static final int COMPONENT_REF = 2;
57  
58  	private boolean nakedPropertyRef = false;
59  
60  	public void resolveIndex(AST parent) throws SemanticException {
61  		// An ident node can represent an index expression if the ident
62  		// represents a naked property ref
63  		//      *Note: this makes the assumption (which is currently the case
64  		//      in the hql-sql grammar) that the ident is first resolved
65  		//      itself (addrExpr -> resolve()).  The other option, if that
66  		//      changes, is to call resolve from here; but it is
67  		//      currently un-needed overhead.
68  		if (!(isResolved() && nakedPropertyRef)) {
69  			throw new UnsupportedOperationException();
70  		}
71  
72  		String propertyName = getOriginalText();
73  		if (!getDataType().isCollectionType()) {
74  			throw new SemanticException("Collection expected; [" + propertyName + "] does not refer to a collection property");
75  		}
76  
77  		// TODO : most of below was taken verbatim from DotNode; should either delegate this logic or super-type it
78  		CollectionType type = (CollectionType) getDataType();
79  		String role = type.getRole();
80  		QueryableCollection queryableCollection = getSessionFactoryHelper().requireQueryableCollection(role);
81  
82  		String alias = null;  // DotNode uses null here...
83  		String columnTableAlias = getFromElement().getTableAlias();
84  		JoinType joinType = JoinType.INNER_JOIN;
85  		boolean fetch = false;
86  
87  		FromElementFactory factory = new FromElementFactory(
88  				getWalker().getCurrentFromClause(),
89  				getFromElement(),
90  				propertyName,
91  				alias,
92  				getFromElement().toColumns(columnTableAlias, propertyName, false),
93  				true
94  		);
95  		FromElement elem = factory.createCollection(queryableCollection, role, joinType, fetch, true);
96  		setFromElement(elem);
97  		getWalker().addQuerySpaces(queryableCollection.getCollectionSpaces());	// Always add the collection's query spaces.
98  	}

100 	public void resolve(boolean generateJoin, boolean implicitJoin, String classAlias, AST parent) {
101 		if (!isResolved()) {
102 			if (getWalker().getCurrentFromClause().isFromElementAlias(getText())) {
103 				if (resolveAsAlias()) {
104 					setResolved();
105 					// We represent a from-clause alias
106 				}
107 			}
108 			else if (parent != null && parent.getType() == SqlTokenTypes.DOT) {
109 				DotNode dot = (DotNode) parent;
110 				if (parent.getFirstChild() == this) {
111 					if (resolveAsNakedComponentPropertyRefLHS(dot)) {
112 						// we are the LHS of the DOT representing a naked comp-prop-ref
113 						setResolved();
114 					}
115 				}
116 				else {
117 					if (resolveAsNakedComponentPropertyRefRHS(dot)) {
118 						// we are the RHS of the DOT representing a naked comp-prop-ref
119 						setResolved();
120 					}
121 				}
122 			}
123 			else {
124 				int result = resolveAsNakedPropertyRef();
125 				if (result == PROPERTY_REF) {
126 					// we represent a naked (simple) prop-ref
127 					setResolved();
128 				}
129 				else if (result == COMPONENT_REF) {
130 					// EARLY EXIT!!!  return so the resolve call explicitly coming from DotNode can
131 					// resolve this...
132 					return;
133 				}
134 			}
135 
136 			// if we are still not resolved, we might represent a constant.
137 			//      needed to add this here because the allowance of
138 			//      naked-prop-refs in the grammar collides with the
139 			//      definition of literals/constants ("nondeterminism").
140 			//      TODO: cleanup the grammar so that "processConstants" is always just handled from here
141 			if (!isResolved()) {
142 				try {
143 					getWalker().getLiteralProcessor().processConstant(this, false);
144 				}
145 				catch (Throwable ignore) {
146 					// just ignore it for now, it'll get resolved later...
147 				}
148 			}
149 		}
150 	}

152 	private boolean resolveAsAlias() {
153 		// This is not actually a constant, but a reference to FROM element.
154 		final FromElement element = getWalker().getCurrentFromClause().getFromElement( getText() );
155 		if ( element == null ) {
156 			return false;
157 		}
158 
159 		setType( SqlTokenTypes.ALIAS_REF );
160 		setFromElement( element );
161 
162 		String[] columnExpressions = element.getIdentityColumns();
163 
164 		// determine whether to apply qualification (table alias) to the column(s)...
165 		if ( ! isFromElementUpdateOrDeleteRoot( element ) ) {
166 			if ( StringHelper.isNotEmpty( element.getTableAlias() ) ) {
167 				// apparently we also need to check that they are not already qualified.  Ugh!
168 				columnExpressions = StringHelper.qualifyIfNot( element.getTableAlias(), columnExpressions );
169 			}
170 		}
171 
172 		final Dialect dialect = getWalker().getSessionFactoryHelper().getFactory().getDialect();
173 		final boolean isInCount = getWalker().isInCount();
174 		final boolean isInDistinctCount = isInCount && getWalker().isInCountDistinct();
175 		final boolean isInNonDistinctCount = isInCount && ! getWalker().isInCountDistinct();
176 		final boolean isCompositeValue = columnExpressions.length > 1;
177 		if ( isCompositeValue ) {
178 			if ( isInNonDistinctCount && ! dialect.supportsTupleCounts() ) {
179 				// TODO: #supportsTupleCounts currently false for all Dialects -- could this be cleaned up?
180 				setText( columnExpressions[0] );
181 			}
182 			else {
183 				String joinedFragment = StringHelper.join( ", ", columnExpressions );
184 				// avoid wrapping in parenthesis (explicit tuple treatment) if possible due to varied support for
185 				// tuple syntax across databases..
186 				final boolean shouldSkipWrappingInParenthesis =
187 						(isInDistinctCount && ! dialect.requiresParensForTupleDistinctCounts())
188 						|| isInNonDistinctCount
189 						|| getWalker().getCurrentTopLevelClauseType() == HqlSqlTokenTypes.ORDER
190 						|| getWalker().getCurrentTopLevelClauseType() == HqlSqlTokenTypes.GROUP;
191 				if ( ! shouldSkipWrappingInParenthesis ) {
192 					joinedFragment = "(" + joinedFragment + ")";
193 				}
194 				setText( joinedFragment );
195 			}
196 			return true;
197 		}
198 		else if ( columnExpressions.length > 0 ) {
199 			setText( columnExpressions[0] );
200 			return true;
201 		}
202 
203 		return false;
204 	}

206 	private Type getNakedPropertyType(FromElement fromElement) {
207 		if (fromElement == null) {
208 			return null;
209 		}
210 		String property = getOriginalText();
211 		Type propertyType = null;
212 		try {
213 			propertyType = fromElement.getPropertyType(property, property);
214 		}
215 		catch (Throwable t) {
216 		}
217 		return propertyType;
218 	}
219 
220 	private int resolveAsNakedPropertyRef() {
221 		FromElement fromElement = locateSingleFromElement();
222 		if (fromElement == null) {
223 			return UNKNOWN;
224 		}
225 		Queryable persister = fromElement.getQueryable();
226 		if (persister == null) {
227 			return UNKNOWN;
228 		}
229 		Type propertyType = getNakedPropertyType(fromElement);
230 		if (propertyType == null) {
231 			// assume this ident's text does *not* refer to a property on the given persister
232 			return UNKNOWN;
233 		}
234 
235 		if ((propertyType.isComponentType() || propertyType.isAssociationType() )) {
236 			return COMPONENT_REF;
237 		}
238 
239 		setFromElement(fromElement);
240 		String property = getText();
241 		String[] columns = getWalker().isSelectStatement()
242 				? persister.toColumns(fromElement.getTableAlias(), property)
243 				: persister.toColumns(property);
244 		String text = StringHelper.join(", ", columns);
245 		setText(columns.length == 1 ? text : "(" + text + ")");
246 		setType(SqlTokenTypes.SQL_TOKEN);
247 
248 		// these pieces are needed for usage in select clause
249 		super.setDataType(propertyType);
250 		nakedPropertyRef = true;
251 
252 		return PROPERTY_REF;
253 	}
254 
255 	private boolean resolveAsNakedComponentPropertyRefLHS(DotNode parent) {
256 		FromElement fromElement = locateSingleFromElement();
257 		if (fromElement == null) {
258 			return false;
259 		}
260 
261 		Type componentType = getNakedPropertyType(fromElement);
262 		if ( componentType == null ) {
263 			throw new QueryException( "Unable to resolve path [" + parent.getPath() + "], unexpected token [" + getOriginalText() + "]" );
264 		}
265 		if (!componentType.isComponentType()) {
266 			throw new QueryException("Property '" + getOriginalText() + "' is not a component.  Use an alias to reference associations or collections.");
267 		}
268 
269 		Type propertyType = null;  // used to set the type of the parent dot node
270 		String propertyPath = getText() + "." + getNextSibling().getText();
271 		try {
272 			// check to see if our "propPath" actually
273 			// represents a property on the persister
274 			propertyType = fromElement.getPropertyType(getText(), propertyPath);
275 		}
276 		catch (Throwable t) {
277 			// assume we do *not* refer to a property on the given persister
278 			return false;
279 		}
280 
281 		setFromElement(fromElement);
282 		parent.setPropertyPath(propertyPath);
283 		parent.setDataType(propertyType);
284 
285 		return true;
286 	}
287 
288 	private boolean resolveAsNakedComponentPropertyRefRHS(DotNode parent) {
289 		FromElement fromElement = locateSingleFromElement();
290 		if (fromElement == null) {
291 			return false;
292 		}
293 
294 		Type propertyType = null;
295 		String propertyPath = parent.getLhs().getText() + "." + getText();
296 		try {
297 			// check to see if our "propPath" actually
298 			// represents a property on the persister
299 			propertyType = fromElement.getPropertyType(getText(), propertyPath);
300 		}
301 		catch (Throwable t) {
302 			// assume we do *not* refer to a property on the given persister
303 			return false;
304 		}
305 
306 		setFromElement(fromElement);
307 		// this piece is needed for usage in select clause
308 		super.setDataType(propertyType);
309 		nakedPropertyRef = true;
310 
311 		return true;
312 	}
313 
314 	private FromElement locateSingleFromElement() {
315 		List fromElements = getWalker().getCurrentFromClause().getFromElements();
316 		if (fromElements == null || fromElements.size() != 1) {
317 			// TODO : should this be an error?
318 			return null;
319 		}
320 		FromElement element = (FromElement) fromElements.get(0);
321 		if (element.getClassAlias() != null) {
322 			// naked property-refs cannot be used with an aliased from element
323 			return null;
324 		}
325 		return element;
326 	}

328 	@Override
329     public Type getDataType() {
330 		Type type = super.getDataType();
331 		if ( type != null ) {
332 			return type;
333 		}
334 		FromElement fe = getFromElement();
335 		if ( fe != null ) {
336 			return fe.getDataType();
337 		}
338 		SQLFunction sf = getWalker().getSessionFactoryHelper().findSQLFunction( getText() );
339 		if ( sf != null ) {
340 			return sf.getReturnType( null, getWalker().getSessionFactoryHelper().getFactory() );
341 		}
342 		return null;
343 	}

345 	public void setScalarColumnText(int i) throws SemanticException {
346 		if (nakedPropertyRef) {
347 			// do *not* over-write the column text, as that has already been
348 			// "rendered" during resolve
349 			ColumnHelper.generateSingleScalarColumn(this, i);
350 		}
351 		else {
352 			FromElement fe = getFromElement();
353 			if (fe != null) {
354 				setText(fe.renderScalarIdentifierSelect(i));
355 			}
356 			else {
357 				ColumnHelper.generateSingleScalarColumn(this, i);
358 			}
359 		}
360 	}

362 	@Override
363     public String getDisplayText() {
364 		StringBuilder buf = new StringBuilder();
365 
366 		if (getType() == SqlTokenTypes.ALIAS_REF) {
367 			buf.append("{alias=").append(getOriginalText());
368 			if (getFromElement() == null) {
369 				buf.append(", no from element");
370 			}
371 			else {
372 				buf.append(", className=").append(getFromElement().getClassName());
373 				buf.append(", tableAlias=").append(getFromElement().getTableAlias());
374 			}
375 			buf.append("}");
376 		}
377 		else {
378 			buf.append("{originalText=" + getOriginalText()).append("}");
379 		}
380 		return buf.toString();
381 	}
