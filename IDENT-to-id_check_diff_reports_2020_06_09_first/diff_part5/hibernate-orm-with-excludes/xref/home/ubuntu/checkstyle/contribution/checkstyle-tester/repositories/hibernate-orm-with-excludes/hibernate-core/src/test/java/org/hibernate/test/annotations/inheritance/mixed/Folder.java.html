2   package org.hibernate.test.annotations.inheritance.mixed;
3   import java.util.HashSet;
4   import java.util.Set;
5   import javax.persistence.DiscriminatorValue;
6   import javax.persistence.Entity;
7   import javax.persistence.OneToMany;
8   
9   /**
10   * @author Emmanuel Bernard
11   */
12  @Entity
13  @DiscriminatorValue("F")
14  public class Folder extends File {
15  	private Set<File> children = new HashSet<File>();
16  
17  	Folder() {
18  	}
19  
20  	public Folder(String name) {
21  		super( name );
22  	}
23  
24  	@OneToMany(mappedBy = "parent")
25  	public Set<File> getChildren() {
26  		return children;
27  	}
28  
29  	public void setChildren(Set children) {
30  		this.children = children;
31  	}
32  }
