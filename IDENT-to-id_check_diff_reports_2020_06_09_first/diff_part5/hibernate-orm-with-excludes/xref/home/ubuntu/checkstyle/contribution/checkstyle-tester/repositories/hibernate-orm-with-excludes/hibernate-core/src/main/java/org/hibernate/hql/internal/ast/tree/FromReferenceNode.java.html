25  package org.hibernate.hql.internal.ast.tree;
26  import antlr.SemanticException;
27  import antlr.collections.AST;
28  import org.jboss.logging.Logger;
29  
30  import org.hibernate.hql.internal.antlr.HqlSqlTokenTypes;
31  import org.hibernate.internal.CoreMessageLogger;
32  
33  /**
34   * Represents a reference to a FROM element, for example a class alias in a WHERE clause.
35   *
36   * @author josh
37   */
38  public abstract class FromReferenceNode extends AbstractSelectExpression
39          implements ResolvableNode, DisplayableNode, InitializeableNode, PathNode {
40  
41  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, FromReferenceNode.class.getName() );
42  
43  	private FromElement fromElement;
44  	private boolean resolved = false;
45  	public static final int ROOT_LEVEL = 0;
46  
47  	@Override
48      public FromElement getFromElement() {
49  		return fromElement;
50  	}
51  
52  	public void setFromElement(FromElement fromElement) {
53  		this.fromElement = fromElement;
54  	}
55  
56  	/**
57  	 * Resolves the left hand side of the DOT.
58  	 *
59  	 * @throws SemanticException
60  	 */
61  	public void resolveFirstChild() throws SemanticException {
62  	}
63  
64  	public String getPath() {
65  		return getOriginalText();
66  	}
67  
68  	public boolean isResolved() {
69  		return resolved;
70  	}
71  
72  	public void setResolved() {
73  		this.resolved = true;
74  		if ( LOG.isDebugEnabled() ) {
75  			LOG.debugf( "Resolved : %s -> %s", this.getPath(), this.getText() );
76  		}
77  	}
78  
79  	public String getDisplayText() {
80  		StringBuilder buf = new StringBuilder();
81  		buf.append( "{" ).append( ( fromElement == null ) ? "no fromElement" : fromElement.getDisplayText() );
82  		buf.append( "}" );
83  		return buf.toString();
84  	}
85  
86  	public void recursiveResolve(int level, boolean impliedAtRoot, String classAlias) throws SemanticException {
87  		recursiveResolve( level, impliedAtRoot, classAlias, this );
88  	}
89  
90  	public void recursiveResolve(int level, boolean impliedAtRoot, String classAlias, AST parent) throws SemanticException {
91  		AST lhs = getFirstChild();
92  		int nextLevel = level + 1;
93  		if ( lhs != null ) {
94  			FromReferenceNode n = ( FromReferenceNode ) lhs;
95  			n.recursiveResolve( nextLevel, impliedAtRoot, null, this );
96  		}
97  		resolveFirstChild();
98  		boolean impliedJoin = true;
99  		if ( level == ROOT_LEVEL && !impliedAtRoot ) {
100 			impliedJoin = false;
101 		}
102 		resolve( true, impliedJoin, classAlias, parent );
103 	}
104 
105 	@Override
106     public boolean isReturnableEntity() throws SemanticException {
107 		return !isScalar() && fromElement.isEntity();
108 	}
109 
110 	public void resolveInFunctionCall(boolean generateJoin, boolean implicitJoin) throws SemanticException {
111 		resolve( generateJoin, implicitJoin );
112 	}
113 
114 	public void resolve(boolean generateJoin, boolean implicitJoin) throws SemanticException {
115 		resolve( generateJoin, implicitJoin, null );
116 	}
117 
118 	public void resolve(boolean generateJoin, boolean implicitJoin, String classAlias) throws SemanticException {
119 		resolve( generateJoin, implicitJoin, classAlias, null );
120 	}
121 
122 	public void prepareForDot(String propertyName) throws SemanticException {
123 	}
124 
125 	/**
126 	 * Sub-classes can override this method if they produce implied joins (e.g. DotNode).
127 	 *
128 	 * @return an implied join created by this from reference.
129 	 */
130 	public FromElement getImpliedJoin() {
131 		return null;
132 	}
133 
134 	@SuppressWarnings("SimplifiableIfStatement")
135 	protected boolean isFromElementUpdateOrDeleteRoot(FromElement element) {
136 		if ( element.getFromClause().getParentFromClause() != null ) {
137 			// its not even a root...
138 			return false;
139 		}
140 
141 		return getWalker().getStatementType() == HqlSqlTokenTypes.DELETE
142 				|| getWalker().getStatementType() == HqlSqlTokenTypes.UPDATE;
143 	}
144 
145 }
