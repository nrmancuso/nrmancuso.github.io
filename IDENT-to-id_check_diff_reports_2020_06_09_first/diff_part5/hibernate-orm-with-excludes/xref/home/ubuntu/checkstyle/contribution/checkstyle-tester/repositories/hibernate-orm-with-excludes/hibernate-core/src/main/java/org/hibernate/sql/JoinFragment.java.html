25  package org.hibernate.sql;
26  
27  import org.hibernate.internal.util.StringHelper;
28  
29  /**
30   * An abstract SQL join fragment renderer
31   *
32   * @author Gavin King
33   */
34  public abstract class JoinFragment {
35  
36  	public abstract void addJoin(String tableName, String alias, String[] fkColumns, String[] pkColumns, JoinType joinType);
37  
38  	public abstract void addJoin(String tableName, String alias, String[] fkColumns, String[] pkColumns, JoinType joinType, String on);
39  
40  	public abstract void addCrossJoin(String tableName, String alias);
41  
42  	public abstract void addJoins(String fromFragment, String whereFragment);
43  
44  	public abstract String toFromFragmentString();
45  
46  	public abstract String toWhereFragmentString();
47  
48  	// --Commented out by Inspection (12/4/04 9:10 AM): public abstract void addCondition(String alias, String[] columns, String condition);
49  	public abstract void addCondition(String alias, String[] fkColumns, String[] pkColumns);
50  
51  	public abstract boolean addCondition(String condition);
52  	// --Commented out by Inspection (12/4/04 9:10 AM): public abstract void addFromFragmentString(String fromFragmentString);
53  
54  	public abstract JoinFragment copy();
55  
56  	/**
57  	 * @deprecated use {@link JoinType#INNER_JOIN} instead.
58  	 */
59  	@Deprecated
60  	public static final int INNER_JOIN = 0;
61  	/**
62  	 * @deprecated use {@link JoinType#FULL_JOIN} instead.
63  	 */
64  	@Deprecated
65  	public static final int FULL_JOIN = 4;
66  	/**
67  	 * @deprecated use {@link JoinType#LEFT_OUTER_JOIN} instead.
68  	 */
69  	@Deprecated
70  	public static final int LEFT_OUTER_JOIN = 1;
71  	/**
72  	 * @deprecated use {@link JoinType#RIGHT_OUTER_JOIN} instead.
73  	 */
74  	@Deprecated
75  	public static final int RIGHT_OUTER_JOIN = 2;
76  	private boolean hasFilterCondition = false;
77  	private boolean hasThetaJoins = false;
78  
79  	public void addFragment(JoinFragment ojf) {
80  		if ( ojf.hasThetaJoins() ) {
81  			hasThetaJoins = true;
82  		}
83  		addJoins( ojf.toFromFragmentString(), ojf.toWhereFragmentString() );
84  	}
85  
86  	/**
87  	 * Appends the 'on' condition to the buffer, returning true if the condition was added.
88  	 * Returns false if the 'on' condition was empty.
89  	 *
90  	 * @param buffer The buffer to append the 'on' condition to.
91  	 * @param on     The 'on' condition.
92  	 * @return Returns true if the condition was added, false if the condition was already in 'on' string.
93  	 */
94  	protected boolean addCondition(StringBuilder buffer, String on) {
95  		if ( StringHelper.isNotEmpty( on ) ) {
96  			if ( !on.startsWith( " and" ) ) buffer.append( " and " );
97  			buffer.append( on );
98  			return true;
99  		}
100 		else {
101 			return false;
102 		}
103 	}
104 
105 	/**
106 	 * True if the where fragment is from a filter condition.
107 	 *
108 	 * @return True if the where fragment is from a filter condition.
109 	 */
110 	public boolean hasFilterCondition() {
111 		return hasFilterCondition;
112 	}
113 
114 	public void setHasFilterCondition(boolean b) {
115 		this.hasFilterCondition = b;
116 	}
117 
118 	public boolean hasThetaJoins() {
119 		return hasThetaJoins;
120 	}
121 
122 	public void setHasThetaJoins(boolean hasThetaJoins) {
123 		this.hasThetaJoins = hasThetaJoins;
124 	}
125 }
