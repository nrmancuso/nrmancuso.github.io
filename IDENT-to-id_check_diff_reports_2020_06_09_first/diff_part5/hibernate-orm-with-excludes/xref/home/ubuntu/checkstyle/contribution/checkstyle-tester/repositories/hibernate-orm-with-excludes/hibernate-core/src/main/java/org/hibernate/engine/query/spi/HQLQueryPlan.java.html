82  	protected HQLQueryPlan(String hql, String collectionRole, boolean shallow, Map enabledFilters,  SessionFactoryImplementor factory){
83  		this.sourceQuery = hql;
84  		this.shallow = shallow;
85  		Set copy = new HashSet();
86  		copy.addAll( enabledFilters.keySet() );
87  		this.enabledFilterNames = java.util.Collections.unmodifiableSet( copy );
88  
89  		final String[] concreteQueryStrings = QuerySplitter.concreteQueries( hql, factory );
90  		final int length = concreteQueryStrings.length;
91  		this.translators = new QueryTranslator[length];
92  
93  		List<String> sqlStringList = new ArrayList<String>();
94  		Set combinedQuerySpaces = new HashSet();
95  
96  		final boolean hasCollectionRole = (collectionRole == null);
97  		final Map querySubstitutions = factory.getSettings().getQuerySubstitutions();
98  		final QueryTranslatorFactory queryTranslatorFactory = factory.getSettings().getQueryTranslatorFactory();
99  
100 		for ( int i=0; i<length; i++ ) {
101 			if ( hasCollectionRole ) {
102 				translators[i] = queryTranslatorFactory
103 						.createQueryTranslator( hql, concreteQueryStrings[i], enabledFilters, factory );
104 				translators[i].compile( querySubstitutions, shallow );
105 			}
106 			else {
107 				translators[i] = queryTranslatorFactory
108 						.createFilterTranslator( hql, concreteQueryStrings[i], enabledFilters, factory );
109 				( ( FilterTranslator ) translators[i] ).compile( collectionRole, querySubstitutions, shallow );
110 			}
111 			combinedQuerySpaces.addAll( translators[i].getQuerySpaces() );
112 			sqlStringList.addAll( translators[i].collectSqlStrings() );
113 		}
114 
115 		this.sqlStrings = ArrayHelper.toStringArray( sqlStringList );
116 		this.querySpaces = combinedQuerySpaces;
117 
118 		if ( length == 0 ) {
119 			parameterMetadata = new ParameterMetadata( null, null );
120 			returnMetadata = null;
121 		}
122 		else {
123 			this.parameterMetadata = buildParameterMetadata( translators[0].getParameterTranslations(), hql );
124 			if ( translators[0].isManipulationStatement() ) {
125 				returnMetadata = null;
126 			}
127 			else {
128 				final Type[] types = ( length > 1 ) ? new Type[translators[0].getReturnTypes().length] : translators[0].getReturnTypes();
129 				returnMetadata = new ReturnMetadata( translators[0].getReturnAliases(), types );
130 			}
131 		}
132 	}

167 	public List performList(
168 			QueryParameters queryParameters,
169 	        SessionImplementor session) throws HibernateException {
170 		if ( LOG.isTraceEnabled() ) {
171 			LOG.tracev( "Find: {0}", getSourceQuery() );
172 			queryParameters.traceParameters( session.getFactory() );
173 		}
174 
175 		final RowSelection rowSelection = queryParameters.getRowSelection();
176 		final boolean hasLimit = rowSelection != null
177 				&& rowSelection.definesLimits();
178 		final boolean needsLimit = hasLimit && translators.length > 1;
179 
180 		final QueryParameters queryParametersToUse;
181 		if ( needsLimit ) {
182 			LOG.needsLimit();
183 			RowSelection selection = new RowSelection();
184 			selection.setFetchSize( queryParameters.getRowSelection().getFetchSize() );
185 			selection.setTimeout( queryParameters.getRowSelection().getTimeout() );
186 			queryParametersToUse = queryParameters.createCopyUsing( selection );
187 		}
188 		else {
189 			queryParametersToUse = queryParameters;
190 		}
191 
192 		final int guessedResultSize = guessResultSize( rowSelection );
193 		final List combinedResults = new ArrayList( guessedResultSize );
194 		final IdentitySet distinction = new IdentitySet( guessedResultSize );
195 		int includedCount = -1;
196 		translator_loop:
197 		for ( QueryTranslator translator : translators ) {
198 			List tmp = translator.list( session, queryParametersToUse );
199 			if ( needsLimit ) {
200 				// NOTE : firstRow is zero-based
201 				final int first = queryParameters.getRowSelection().getFirstRow() == null
202 						? 0
203 						: queryParameters.getRowSelection().getFirstRow();
204 				final int max = queryParameters.getRowSelection().getMaxRows() == null
205 						? -1
206 						: queryParameters.getRowSelection().getMaxRows();
207 				for ( final Object result : tmp ) {
208 					if ( !distinction.add( result ) ) {
209 						continue;
210 					}
211 					includedCount++;
212 					if ( includedCount < first ) {
213 						continue;
214 					}
215 					combinedResults.add( result );
216 					if ( max >= 0 && includedCount > max ) {
217 						// break the outer loop !!!
218 						break translator_loop;
219 					}
220 				}
221 			}
222 			else {
223 				combinedResults.addAll( tmp );
224 			}
225 		}
226 		return combinedResults;
227 	}

229 	/**
230 	 * If we're able to guess a likely size of the results we can optimize allocation
231 	 * of our datastructures.
232 	 * Essentially if we detect the user is not using pagination, we attempt to use the FetchSize
233 	 * as a reasonable hint. If fetch size is not being set either, it is reasonable to expect
234 	 * that we're going to have a single hit. In such a case it would be tempting to return a constant
235 	 * of value one, but that's dangerous as it doesn't scale up appropriately for example
236 	 * with an ArrayList if the guess is wrong.
237 	 *
238 	 * @param rowSelection
239 	 * @return a reasonable size to use for allocation
240 	 */
241 	private final int guessResultSize(RowSelection rowSelection) {
242 		if ( rowSelection != null ) {
243 			final int maxReasonableAllocation = rowSelection.getFetchSize() != null ? rowSelection.getFetchSize().intValue() : 100;
244 			if ( rowSelection.getMaxRows() != null && rowSelection.getMaxRows().intValue() > 0 ) {
245 				return Math.min( maxReasonableAllocation, rowSelection.getMaxRows().intValue() );
246 			}
247 			else if ( rowSelection.getFetchSize() != null && rowSelection.getFetchSize().intValue() > 0 ) {
248 				return rowSelection.getFetchSize().intValue();
249 			}
250 		}
251 		return 7;//magic number guessed as a reasonable default.
252 	}

254 	/**
255 	 * Coordinates the efforts to perform an iterate across all the included query translators.
256 	 *
257 	 * @param queryParameters The query parameters
258 	 * @param session The session
259 	 *
260 	 * @return The query result iterator
261 	 *
262 	 * @throws HibernateException Indicates a problem performing the query
263 	 */
264 	@SuppressWarnings("unchecked")
265 	public Iterator performIterate(
266 			QueryParameters queryParameters,
267 	        EventSource session) throws HibernateException {
268 		if ( LOG.isTraceEnabled() ) {
269 			LOG.tracev( "Iterate: {0}", getSourceQuery() );
270 			queryParameters.traceParameters( session.getFactory() );
271 		}
272 		if ( translators.length == 0 ) {
273 			return EmptyIterator.INSTANCE;
274 		}
275 
276 		Iterator[] results = null;
277 		boolean many = translators.length > 1;
278 		if ( many ) {
279 			results = new Iterator[translators.length];
280 		}
281 
282 		Iterator result = null;
283 		for ( int i = 0; i < translators.length; i++ ) {
284 			result = translators[i].iterate( queryParameters, session );
285 			if ( many ) {
286 				results[i] = result;
287 			}
288 		}
289 
290 		return many ? new JoinedIterator(results) : result;
291 	}

293 	public ScrollableResults performScroll(
294 			QueryParameters queryParameters,
295 	        SessionImplementor session) throws HibernateException {
296 		if ( LOG.isTraceEnabled() ) {
297 			LOG.tracev( "Iterate: {0}", getSourceQuery() );
298 			queryParameters.traceParameters( session.getFactory() );
299 		}
300 		if ( translators.length != 1 ) {
301 			throw new QueryException( "implicit polymorphism not supported for scroll() queries" );
302 		}
303 		if ( queryParameters.getRowSelection().definesLimits() && translators[0].containsCollectionFetches() ) {
304 			throw new QueryException( "firstResult/maxResults not supported in conjunction with scroll() of a query containing collection fetches" );
305 		}
306 
307 		return translators[0].scroll( queryParameters, session );
308 	}

310 	public int performExecuteUpdate(QueryParameters queryParameters, SessionImplementor session)
311 			throws HibernateException {
312 		if ( LOG.isTraceEnabled() ) {
313 			LOG.tracev( "Execute update: {0}", getSourceQuery() );
314 			queryParameters.traceParameters( session.getFactory() );
315 		}
316 		if ( translators.length != 1 ) {
317 			LOG.splitQueries( getSourceQuery(), translators.length );
318 		}
319 		int result = 0;
320 		for ( QueryTranslator translator : translators ) {
321 			result += translator.executeUpdate( queryParameters, session );
322 		}
323 		return result;
324 	}

326 	private ParameterMetadata buildParameterMetadata(ParameterTranslations parameterTranslations, String hql) {
327 		long start = System.currentTimeMillis();
328 		ParamLocationRecognizer recognizer = ParamLocationRecognizer.parseLocations( hql );
329 		long end = System.currentTimeMillis();
330 		if ( LOG.isTraceEnabled() ) {
331 			LOG.tracev( "HQL param location recognition took {0} mills ({1})", ( end - start ), hql );
332 		}
333 
334 		int ordinalParamCount = parameterTranslations.getOrdinalParameterCount();
335 		int[] locations = ArrayHelper.toIntArray( recognizer.getOrdinalParameterLocationList() );
336 		if ( parameterTranslations.supportsOrdinalParameterMetadata() && locations.length != ordinalParamCount ) {
337 			throw new HibernateException( "ordinal parameter mismatch" );
338 		}
339 		ordinalParamCount = locations.length;
340 		OrdinalParameterDescriptor[] ordinalParamDescriptors = new OrdinalParameterDescriptor[ordinalParamCount];
341 		for ( int i = 1; i <= ordinalParamCount; i++ ) {
342 			ordinalParamDescriptors[ i - 1 ] = new OrdinalParameterDescriptor(
343 					i,
344 			        parameterTranslations.supportsOrdinalParameterMetadata()
345 		                    ? parameterTranslations.getOrdinalParameterExpectedType( i )
346 		                    : null,
347 			        locations[ i - 1 ]
348 			);
349 		}
350 
351 		Map<String, NamedParameterDescriptor> namedParamDescriptorMap = new HashMap<String, NamedParameterDescriptor>();
352 		Map<String, ParamLocationRecognizer.NamedParameterDescription> map = recognizer.getNamedParameterDescriptionMap();
353 		for ( final String name : map.keySet() ) {
354 			final ParamLocationRecognizer.NamedParameterDescription description = map.get( name );
355 			namedParamDescriptorMap.put(
356 					name,
357 					new NamedParameterDescriptor(
358 							name,
359 							parameterTranslations.getNamedParameterExpectedType( name ),
360 							description.buildPositionsArray(),
361 							description.isJpaStyle()
362 					)
363 			);
364 		}
365 		return new ParameterMetadata( ordinalParamDescriptors, namedParamDescriptorMap );
366 	}

369 		System.arraycopy(translators, 0, copy, 0, copy.length);
370 		return copy;
