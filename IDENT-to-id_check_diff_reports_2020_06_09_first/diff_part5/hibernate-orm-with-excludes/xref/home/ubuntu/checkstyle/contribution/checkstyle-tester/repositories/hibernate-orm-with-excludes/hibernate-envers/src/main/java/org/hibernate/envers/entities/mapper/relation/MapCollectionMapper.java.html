24  package org.hibernate.envers.entities.mapper.relation;
25  
26  import java.io.Serializable;
27  import java.util.Collection;
28  import java.util.Map;
29  
30  import org.hibernate.collection.spi.PersistentCollection;
31  import org.hibernate.engine.spi.SessionImplementor;
32  import org.hibernate.envers.configuration.AuditConfiguration;
33  import org.hibernate.envers.entities.mapper.PropertyMapper;
34  import org.hibernate.envers.entities.mapper.relation.lazy.initializor.Initializor;
35  import org.hibernate.envers.entities.mapper.relation.lazy.initializor.MapCollectionInitializor;
36  import org.hibernate.envers.reader.AuditReaderImplementor;
37  
38  /**
39   * @author Adam Warski (adam at warski dot org)
40   */
41  public class MapCollectionMapper<T extends Map> extends AbstractCollectionMapper<T> implements PropertyMapper {
42      protected final MiddleComponentData elementComponentData;
43      protected final MiddleComponentData indexComponentData;
44  
45      public MapCollectionMapper(CommonCollectionMapperData commonCollectionMapperData,
46                                 Class<? extends T> collectionClass, Class<? extends T> proxyClass,
47                                 MiddleComponentData elementComponentData, MiddleComponentData indexComponentData,
48  							   boolean revisionTypeInId) {
49  		super( commonCollectionMapperData, collectionClass, proxyClass, false, revisionTypeInId );
50          this.elementComponentData = elementComponentData;
51          this.indexComponentData = indexComponentData;
52      }
53  
54      protected Initializor<T> getInitializor(AuditConfiguration verCfg, AuditReaderImplementor versionsReader,
55                                              Object primaryKey, Number revision, boolean removed) {
56          return new MapCollectionInitializor<T>(verCfg, versionsReader, commonCollectionMapperData.getQueryGenerator(),
57                  primaryKey, revision, removed, collectionClass, elementComponentData, indexComponentData);
58      }
59  
60      protected Collection getNewCollectionContent(PersistentCollection newCollection) {
61          if (newCollection == null) {
62              return null;
63          } else {
64              return ((Map) newCollection).entrySet();
65          }
66      }
67  
68      protected Collection getOldCollectionContent(Serializable oldCollection) {
69          if (oldCollection == null) {
70              return null;
71          } else {
72              return ((Map) oldCollection).entrySet();
73          }
74      }
75  
76      protected void mapToMapFromObject(SessionImplementor session, Map<String, Object> idData, Map<String, Object> data, Object changed) {
77          elementComponentData.getComponentMapper().mapToMapFromObject(session, idData, data, ((Map.Entry) changed).getValue());
78          indexComponentData.getComponentMapper().mapToMapFromObject(session, idData, data, ((Map.Entry) changed).getKey());
79      }
80  }
