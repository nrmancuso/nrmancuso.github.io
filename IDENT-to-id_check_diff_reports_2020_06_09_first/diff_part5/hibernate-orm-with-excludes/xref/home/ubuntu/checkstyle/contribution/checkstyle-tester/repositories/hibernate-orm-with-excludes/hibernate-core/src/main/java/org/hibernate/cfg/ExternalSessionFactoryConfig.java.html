252 	public final void addAdditionalProperty(String name, String value) {
253 		if ( !getExcludedPropertyNames().contains( name ) ) {

257 			additionalProperties.put( name, value );
258 		}

261 	protected final Configuration buildConfiguration() {
262 
263 		Configuration cfg = new Configuration().setProperties( buildProperties() );
264 
265 
266 		String[] mappingFiles = ConfigurationHelper.toStringArray( mapResources, " ,\n\t\r\f" );
267 		for ( int i = 0; i < mappingFiles.length; i++ ) {
268 			cfg.addResource( mappingFiles[i] );
269 		}
270 
271 		return cfg;
272 	}

274 	protected final Properties buildProperties() {
275 		Properties props = new Properties();
276 		setUnlessNull( props, Environment.DIALECT, dialect );
277 		setUnlessNull( props, Environment.DEFAULT_SCHEMA, defaultSchema );
278 		setUnlessNull( props, Environment.DEFAULT_CATALOG, defaultCatalog );
279 		setUnlessNull( props, Environment.MAX_FETCH_DEPTH, maximumFetchDepth );
280 		setUnlessNull( props, Environment.STATEMENT_FETCH_SIZE, jdbcFetchSize );
281 		setUnlessNull( props, Environment.STATEMENT_BATCH_SIZE, jdbcBatchSize );
282 		setUnlessNull( props, Environment.BATCH_VERSIONED_DATA, batchVersionedDataEnabled );
283 		setUnlessNull( props, Environment.USE_SCROLLABLE_RESULTSET, jdbcScrollableResultSetEnabled );
284 		setUnlessNull( props, Environment.USE_GET_GENERATED_KEYS, getGeneratedKeysEnabled );
285 		setUnlessNull( props, Environment.USE_STREAMS_FOR_BINARY, streamsForBinaryEnabled );
286 		setUnlessNull( props, Environment.USE_REFLECTION_OPTIMIZER, reflectionOptimizationEnabled );
287 		setUnlessNull( props, Environment.QUERY_SUBSTITUTIONS, querySubstitutions );
288 		setUnlessNull( props, Environment.SHOW_SQL, showSqlEnabled );
289 		setUnlessNull( props, Environment.USE_SQL_COMMENTS, commentsEnabled );
290 		setUnlessNull( props, Environment.CACHE_REGION_FACTORY, cacheRegionFactory );
291 		setUnlessNull( props, Environment.CACHE_PROVIDER_CONFIG, cacheProviderConfig );
292 		setUnlessNull( props, Environment.CACHE_REGION_PREFIX, cacheRegionPrefix );
293 		setUnlessNull( props, Environment.USE_MINIMAL_PUTS, minimalPutsEnabled );
294 		setUnlessNull( props, Environment.USE_SECOND_LEVEL_CACHE, secondLevelCacheEnabled );
295 		setUnlessNull( props, Environment.USE_QUERY_CACHE, queryCacheEnabled );
296 
297 		Map extraProperties = getExtraProperties();
298 		if ( extraProperties != null ) {
299 			addAll( props, extraProperties );
300 		}
301 
302 		if ( additionalProperties != null ) {
303 			addAll( props, additionalProperties );
304 		}
305 
306 		return props;
307 	}
308 
309 	protected void addAll( Properties target, Map source ) {
310 		Iterator itr = source.entrySet().iterator();
311 		while ( itr.hasNext() ) {
312 			final Map.Entry entry = ( Map.Entry ) itr.next();
313 			final String propertyName = ( String ) entry.getKey();
314 			final String propertyValue = ( String ) entry.getValue();
315 			if ( propertyName != null && propertyValue != null ) {
316 				// Make sure we don't override previous set values
317 				if ( !target.keySet().contains( propertyName ) ) {
318 					if ( !getExcludedPropertyNames().contains( propertyName) ) {
319 						target.put( propertyName, propertyValue );
320 					}
321 				}
322 			}
323 		}
324 	}
325 
326 	protected Map getExtraProperties() {
327 		return null;
328 	}

330 	private void setUnlessNull(Properties props, String key, String value) {
331 		if ( value != null ) {
332 			props.setProperty( key, value );
333 		}
334 	}

336 	private void verifyInt(String value)
337 	{
338 		if ( value != null ) {
339 			Integer.parseInt( value );
340 		}
