69  			@Override

71  				String databaseName = connection.getCatalog();
72  				Statement st = ((SessionImplementor)s).getTransactionCoordinator().getJdbcCoordinator().getStatementPreparer().createStatement();
73  				ResultSet rs =  ((SessionImplementor)s).getTransactionCoordinator().getJdbcCoordinator().getResultSetReturn().extract( st, "SELECT collation_name FROM sys.databases WHERE name = '"+databaseName+ "';" );

75  					return rs.getString( "collation_name" );
76  				}
77  				throw new AssertionError( "can't get collation name of database "+databaseName );
78  
79  			}
80  		} );
81  		s.getTransaction().commit();
82  		s.close();
83  
84  		Session s2 = openSession();
85  		String databaseName = s2.doReturningWork( new ReturningWork<String>() {
86  			@Override

88  				return connection.getCatalog();
89  			}
90  		} );
91  		s2.createSQLQuery( "ALTER DATABASE " + databaseName + " set single_user with rollback immediate" )
92  				.executeUpdate();
93  		s2.createSQLQuery( "ALTER DATABASE " + databaseName + " COLLATE Latin1_General_CS_AS" ).executeUpdate();
94  		s2.createSQLQuery( "ALTER DATABASE " + databaseName + " set multi_user" ).executeUpdate();
95  
96  		Transaction tx = s2.beginTransaction();
97  
98  		for ( int i = 1; i <= 20; i++ ) {
99  			s2.persist( new Product2( i, "Kit" + i ) );

225 	@Test
226 	@TestForIssue(jiraKey = "HHH-6627")
227 	public void testPaginationWithAggregation() {
228 		Session session = openSession();
229 		Transaction tx = session.beginTransaction();
230 
231 		// populating test data
232 		Category category1 = new Category( 1, "Category1" );
233 		Category category2 = new Category( 2, "Category2" );
234 		Category category3 = new Category( 3, "Category3" );
235 		session.persist( category1 );
236 		session.persist( category2 );
237 		session.persist( category3 );
238 		session.flush();
239 		session.persist( new Product2( 1, "Kit1", category1 ) );
240 		session.persist( new Product2( 2, "Kit2", category1 ) );
241 		session.persist( new Product2( 3, "Kit3", category1 ) );
242 		session.persist( new Product2( 4, "Kit4", category2 ) );
243 		session.persist( new Product2( 5, "Kit5", category2 ) );
244 		session.persist( new Product2( 6, "Kit6", category3 ) );
245 		session.flush();
246 		session.clear();
247 
248 		// count number of products in each category
249 		List<Object[]> result = session.createCriteria( Category.class, "c" ).createAlias( "products", "p" )
250 				.setProjection(
251 						Projections.projectionList()
252 								.add( Projections.groupProperty( "c.id" ) )
253 								.add( Projections.countDistinct( "p.id" ) )
254 				)
255 				.addOrder( Order.asc( "c.id" ) )
256 				.setFirstResult( 1 ).setMaxResults( 3 ).list();
257 
258 		assertEquals( 2, result.size() );
259 		assertArrayEquals( new Object[] { 2, 2L }, result.get( 0 ) ); // two products of second category
260 		assertArrayEquals( new Object[] { 3, 1L }, result.get( 1 ) ); // one products of third category
261 
262 		tx.rollback();
263 		session.close();
264 	}

321 	@Test
322 	@TestForIssue(jiraKey = "HHH-3961")
323 	public void testLockNowaitSqlServer() throws Exception {
324 		Session s = openSession();
325 		s.beginTransaction();
326 
327 		final Product2 kit = new Product2();
328 		kit.id = 4000;
329 		kit.description = "m";
330 		s.persist( kit );
331 		s.getTransaction().commit();
332 		final Transaction tx = s.beginTransaction();
333 
334 
335 		Session s2 = openSession();
336 
337 		Transaction tx2 = s2.beginTransaction();
338 
339 		Product2 kit2 = (Product2) s2.byId( Product2.class ).load( kit.id );
340 
341 		kit.description = "change!";
342 		s.flush(); // creates write lock on kit until we end the transaction
343 
344 		Thread thread = new Thread(
345 				new Runnable() {
346 					@Override
347 					public void run() {
348 						try {
349 							Thread.sleep( 3000 );
350 						}
351 						catch ( InterruptedException e ) {
352 							e.printStackTrace();
353 						}
354 						tx.commit();
355 					}
356 				}
357 		);
358 
359 		LockOptions opt = new LockOptions( LockMode.UPGRADE_NOWAIT );
360 		opt.setTimeOut( 0 ); // seems useless
361 		long start = System.currentTimeMillis();
362 		thread.start();
363 		try {
364 			s2.buildLockRequest( opt ).lock( kit2 );
365 		}
366 		catch ( LockTimeoutException e ) {
367 			// OK
368 		}
369 		long end = System.currentTimeMillis();
370 		thread.join();
371 		long differenceInMillisecs = end - start;
372 		assertTrue(
373 				"Lock NoWait blocked for " + differenceInMillisecs + " ms, this is definitely to much for Nowait",
374 				differenceInMillisecs < 2000
375 		);
376 
377 		s2.getTransaction().rollback();
378 		s.getTransaction().begin();
379 		s.delete( kit );
380 		s.getTransaction().commit();
381 	}
