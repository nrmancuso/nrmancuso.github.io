28  package org.hibernate.test.annotations.manytoonewithformula;
29  import javax.persistence.Column;
30  import javax.persistence.Entity;
31  import javax.persistence.Id;
32  import javax.persistence.Table;
33  
34  @Entity
35  @Table(name = "MANUFACTURER")
36  public class Manufacturer {
37  
38  	private ManufacturerId id;
39  
40  	private String name;
41  
42  	public Manufacturer(ManufacturerId id, String name) {
43  		this.id = id;
44  		this.name = name;
45  	}
46  
47  	public Manufacturer() {
48  	}
49  
50  	@Column(name = "MFG_NAME")
51  	public String getName() {
52  		return name;
53  	}
54  
55  	public void setName(String name) {
56  		this.name = name;
57  	}
58  
59  	@Id
60  	public ManufacturerId getId() {
61  		return id;
62  	}
63  
64  	public void setId(ManufacturerId id) {
65  		this.id = id;
66  	}
67  
68  	@Override
69  	public String toString() {
70  		return name;
71  	}
72  }
