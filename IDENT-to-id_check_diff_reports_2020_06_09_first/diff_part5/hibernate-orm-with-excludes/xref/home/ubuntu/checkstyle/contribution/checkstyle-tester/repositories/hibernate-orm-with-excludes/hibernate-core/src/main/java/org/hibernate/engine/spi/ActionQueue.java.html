67  /**
68   * Responsible for maintaining the queue of actions related to events.
69   * 
70   * The ActionQueue holds the DML operations queued as part of a session's
71   * transactional-write-behind semantics.  DML operations are queued here
72   * until a flush forces them to be executed against the database.
73   *
74   * @author Steve Ebersole
75   * @author Gail Badner
76   */
77  public class ActionQueue {

196 	private void addInsertAction(AbstractEntityInsertAction insert) {
197 		if ( insert.isEarlyInsert() ) {
198 			// For early inserts, must execute inserts before finding non-nullable transient entities.
199 			// TODO: find out why this is necessary
200 			LOG.tracev(
201 					"Executing inserts before finding non-nullable transient entities for early insert: [{0}]",
202 					insert
203 			);
204 			executeInserts();
205 		}
206 		NonNullableTransientDependencies nonNullableTransientDependencies = insert.findNonNullableTransientEntities();
207 		if ( nonNullableTransientDependencies == null ) {
208 			LOG.tracev( "Adding insert with no non-nullable, transient entities: [{0}]", insert);
209 			addResolvedEntityInsertAction( insert );
210 		}
211 		else {
212 			if ( LOG.isTraceEnabled() ) {
213 				LOG.tracev(
214 						"Adding insert with non-nullable, transient entities; insert=[{0}], dependencies=[{1}]",
215 						insert,
216 						nonNullableTransientDependencies.toLoggableString( insert.getSession() )
217 				);
218 			}
219 			unresolvedInsertions.addUnresolvedEntityInsertAction( insert, nonNullableTransientDependencies );
220 		}
221 	}
222 
223 	@SuppressWarnings({ "unchecked" })
224 	private void addResolvedEntityInsertAction(AbstractEntityInsertAction insert) {
225 		if ( insert.isEarlyInsert() ) {
226 			LOG.trace( "Executing insertions before resolved early-insert" );
227 			executeInserts();
228 			LOG.debug( "Executing identity-insert immediately" );
229 			execute( insert );
230 		}
231 		else {
232 			LOG.trace( "Adding resolved non-early insert action." );
233 			insertions.add( insert );
234 		}
235 		insert.makeEntityManaged();
236 		for ( AbstractEntityInsertAction resolvedAction :
237 				unresolvedInsertions.resolveDependentActions( insert.getInstance(), session ) ) {
238 			addResolvedEntityInsertAction( resolvedAction );
239 		}
240 	}

291 	/**
292 	 * Perform all currently queued actions.
293 	 *
294 	 * @throws HibernateException error executing queued actions.
295 	 */
296 	public void executeActions() throws HibernateException {
297 		if ( ! unresolvedInsertions.isEmpty() ) {
298 			throw new IllegalStateException(
299 					"About to execute actions, but there are unresolved entity insert actions."
300 			);
301 		}
302 		executeActions( orphanRemovals );
303 		executeActions( insertions );
304 		executeActions( updates );
305 		// do before actions are handled in the other collection queues
306 		executeActions( collectionQueuedOps );
307 		executeActions( collectionRemovals );
308 		executeActions( collectionUpdates );
309 		executeActions( collectionCreations );
310 		executeActions( deletions );
311 	}

342 	 * Check whether the given tables/query-spaces are to be executed against
343 	 * given the currently queued actions.
344 	 *
345 	 * @param tables The table/query-spaces to check.
346 	 *
347 	 * @return True if we contain pending actions against any of the given
348 	 *         tables; false otherwise.
349 	 */
350 	public boolean areTablesToBeUpdated(Set tables) {
351 		return areTablesToUpdated( updates, tables ) ||
352 				areTablesToUpdated( insertions, tables ) ||
353 				areTablesToUpdated( unresolvedInsertions.getDependentEntityInsertActions(), tables ) ||
354 				areTablesToUpdated( deletions, tables ) ||
355 				areTablesToUpdated( collectionUpdates, tables ) ||
356 				areTablesToUpdated( collectionCreations, tables ) ||
357 				areTablesToUpdated( collectionQueuedOps, tables ) ||
358 				areTablesToUpdated( collectionRemovals, tables ) ||
359 				areTablesToUpdated( orphanRemovals, tables );
360 	}

369 	}

371 	@SuppressWarnings({ "unchecked" })
372 	private static boolean areTablesToUpdated(Iterable actions, Set tableSpaces) {
373 		for ( Executable action : (Iterable<Executable>) actions ) {
374 			final Serializable[] spaces = action.getPropertySpaces();
375 			for ( Serializable space : spaces ) {
376 				if ( tableSpaces.contains( space ) ) {
377 					LOG.debugf( "Changes must be flushed to space: %s", space );
378 					return true;
379 				}
380 			}
381 		}
382 		return false;
383 	}

385 	private void executeActions(List list) throws HibernateException {
386 		for ( Object aList : list ) {
387 			execute( (Executable) aList );
388 		}
389 		list.clear();
390 		session.getTransactionCoordinator().getJdbcCoordinator().executeBatch();
391 	}

393 	public void execute(Executable executable) {
394 		try {
395 			executable.execute();
396 		}
397 		finally {
398 			registerCleanupActions( executable );
399 		}
400 	}

402 	private void registerCleanupActions(Executable executable) {
403 		beforeTransactionProcesses.register( executable.getBeforeTransactionCompletionProcess() );
404 		if ( session.getFactory().getSettings().isQueryCacheEnabled() ) {
405 			final String[] spaces = (String[]) executable.getPropertySpaces();
406 			if ( spaces != null && spaces.length > 0 ) { //HHH-6286
407 				afterTransactionProcesses.addSpacesToInvalidate( spaces );
408 				session.getFactory().getUpdateTimestampsCache().preInvalidate( spaces, session );
409 			}
410 		}
411 		afterTransactionProcesses.register( executable.getAfterTransactionCompletionProcess() );
412 	}

491 	 * original order is generated by cascade order, which in turn is based on
492 	 * the directionality of foreign-keys.  So even though we will be changing

498 		new InsertActionSorter().sort();

539 	public void unScheduleDeletion(EntityEntry entry, Object rescuedEntity) {
540 		if ( rescuedEntity instanceof HibernateProxy ) {
541 			LazyInitializer initializer = ( ( HibernateProxy ) rescuedEntity ).getHibernateLazyInitializer();
542 			if ( !initializer.isUninitialized() ) {
543 				rescuedEntity = initializer.getImplementation( session );
544 			}
545 		}
546 		for ( int i = 0; i < deletions.size(); i++ ) {
547 			EntityDeleteAction action = deletions.get( i );
548 			if ( action.getInstance() == rescuedEntity ) {
549 				deletions.remove( i );
550 				return;
551 			}
552 		}
553 		for ( int i = 0; i < orphanRemovals.size(); i++ ) {
554 			EntityDeleteAction action = orphanRemovals.get( i );
555 			if ( action.getInstance() == rescuedEntity ) {
556 				orphanRemovals.remove( i );
557 				return;
558 			}
559 		}
560 		throw new AssertionFailure( "Unable to perform un-delete for instance " + entry.getEntityName() );
561 	}

563 	/**
564 	 * Used by the owning session to explicitly control serialization of the
565 	 * action queue
566 	 *
567 	 * @param oos The stream to which the action queue should get written
568 	 *
569 	 * @throws IOException Indicates an error writing to the stream
570 	 */
571 	public void serialize(ObjectOutputStream oos) throws IOException {
572 		LOG.trace( "Serializing action-queue" );
573 
574 		unresolvedInsertions.serialize( oos );
575 
576 		int queueSize = insertions.size();
577 		LOG.tracev( "Starting serialization of [{0}] insertions entries", queueSize );
578 		oos.writeInt( queueSize );
579 		for ( int i = 0; i < queueSize; i++ ) {
580 			oos.writeObject( insertions.get( i ) );
581 		}
582 
583 		queueSize = deletions.size();
584 		LOG.tracev( "Starting serialization of [{0}] deletions entries", queueSize );
585 		oos.writeInt( queueSize );
586 		for ( int i = 0; i < queueSize; i++ ) {
587 			oos.writeObject( deletions.get( i ) );
588 		}
589 
590 		queueSize = orphanRemovals.size();
591 		LOG.tracev( "Starting serialization of [{0}] orphanRemovals entries", queueSize );
592 		oos.writeInt( queueSize );
593 		for ( int i = 0; i < queueSize; i++ ) {
594 			oos.writeObject( orphanRemovals.get( i ) );
595 		}
596 
597 		queueSize = updates.size();
598 		LOG.tracev( "Starting serialization of [{0}] updates entries", queueSize );
599 		oos.writeInt( queueSize );
600 		for ( int i = 0; i < queueSize; i++ ) {
601 			oos.writeObject( updates.get( i ) );
602 		}
603 
604 		queueSize = collectionUpdates.size();
605 		LOG.tracev( "Starting serialization of [{0}] collectionUpdates entries", queueSize );
606 		oos.writeInt( queueSize );
607 		for ( int i = 0; i < queueSize; i++ ) {
608 			oos.writeObject( collectionUpdates.get( i ) );
609 		}
610 
611 		queueSize = collectionRemovals.size();
612 		LOG.tracev( "Starting serialization of [{0}] collectionRemovals entries", queueSize );
613 		oos.writeInt( queueSize );
614 		for ( int i = 0; i < queueSize; i++ ) {
615 			oos.writeObject( collectionRemovals.get( i ) );
616 		}
617 
618 		queueSize = collectionCreations.size();
619 		LOG.tracev( "Starting serialization of [{0}] collectionCreations entries", queueSize );
620 		oos.writeInt( queueSize );
621 		for ( int i = 0; i < queueSize; i++ ) {
622 			oos.writeObject( collectionCreations.get( i ) );
623 		}
624 
625 		queueSize = collectionQueuedOps.size();
626 		LOG.tracev( "Starting serialization of [{0}] collectionQueuedOps entries", queueSize );
627 		oos.writeInt( queueSize );
628 		for ( int i = 0; i < queueSize; i++ ) {
629 			oos.writeObject( collectionQueuedOps.get( i ) );
630 		}
631 	}

633 	/**
634 	 * Used by the owning session to explicitly control deserialization of the
635 	 * action queue
636 	 *
637 	 * @param ois The stream from which to read the action queue
638 	 * @param session The session to which the action queue belongs
639 	 *
640 	 * @return The deserialized action queue
641 	 *
642 	 * @throws IOException indicates a problem reading from the stream
643 	 * @throws ClassNotFoundException Generally means we were unable to locate user classes.
644 	 */
645 	@SuppressWarnings({ "unchecked" })
646 	public static ActionQueue deserialize(
647 			ObjectInputStream ois,
648 			SessionImplementor session) throws IOException, ClassNotFoundException {
649 		LOG.trace( "Dedeserializing action-queue" );
650 		ActionQueue rtn = new ActionQueue( session );

729 		private SessionImplementor session;
730 		// Concurrency handling required when transaction completion process is dynamically registered
731 		// inside event listener (HHH-7478).
732 		private Queue<BeforeTransactionCompletionProcess> processes = new ConcurrentLinkedQueue<BeforeTransactionCompletionProcess>();
733 
734 		private BeforeTransactionCompletionProcessQueue(SessionImplementor session) {
735 			this.session = session;
736 		}
737 
738 		public void register(BeforeTransactionCompletionProcess process) {
739 			if ( process == null ) {
740 				return;
741 			}
742 			processes.add( process );
743 		}
744 
745 		public void beforeTransactionCompletion() {
746 			for ( BeforeTransactionCompletionProcess process : processes ) {
747 				try {
748 					process.doBeforeTransactionCompletion( session );
749 				}
750 				catch (HibernateException he) {
751 					throw he;
752 				}
753 				catch (Exception e) {
754 					throw new AssertionFailure( "Unable to perform beforeTransactionCompletion callback", e );
755 				}
756 			}
757 			processes.clear();
758 		}

764 		// Concurrency handling required when transaction completion process is dynamically registered
765 		// inside event listener (HHH-7478).

780 		}

784 				return;
785 			}

787 		}

789 		public void afterTransactionCompletion(boolean success) {
790 			for ( AfterTransactionCompletionProcess process : processes ) {
791 				try {
792 					process.doAfterTransactionCompletion( success, session );
793 				}
794 				catch ( CacheException ce ) {
795 					LOG.unableToReleaseCacheLock( ce );
796 					// continue loop
797 				}
798 				catch ( Exception e ) {
799 					throw new AssertionFailure( "Exception releasing cache locks", e );
800 				}
801 			}
802 			processes.clear();
803 
804 			if ( session.getFactory().getSettings().isQueryCacheEnabled() ) {
805 				session.getFactory().getUpdateTimestampsCache().invalidate(
806 						querySpacesToInvalidate.toArray( new String[querySpacesToInvalidate.size()] ) ,
807 						session
808 				);
809 			}
810 			querySpacesToInvalidate.clear();
811 		}

814 	/**
815 	 * Sorts the insert actions using more hashes.
816 	 *
817 	 * @author Jay Erb
818 	 */
819 	private class InsertActionSorter {
820 		// the mapping of entity names to their latest batch numbers.
821 		private HashMap<String,Integer> latestBatches = new HashMap<String,Integer>();
822 		private HashMap<Object,Integer> entityBatchNumber;
823 
824 		// the map of batch numbers to EntityInsertAction lists
825 		private HashMap<Integer,List<EntityInsertAction>> actionBatches = new HashMap<Integer,List<EntityInsertAction>>();
826 
827 		public InsertActionSorter() {
828 			//optimize the hash size to eliminate a rehash.
829 			entityBatchNumber = new HashMap<Object,Integer>( insertions.size() + 1, 1.0f );
830 		}
831 
832 		/**
833 		 * Sort the insert actions.
834 		 */
835 		@SuppressWarnings({ "unchecked", "UnnecessaryBoxing" })
836 		public void sort() {
837 			// the list of entity names that indicate the batch number
838 			for ( EntityInsertAction action : (List<EntityInsertAction>) insertions ) {
839 				// remove the current element from insertions. It will be added back later.
840 				String entityName = action.getEntityName();
841 
842 				// the entity associated with the current action.
843 				Object currentEntity = action.getInstance();
844 
845 				Integer batchNumber;
846 				if ( latestBatches.containsKey( entityName ) ) {
847 					// There is already an existing batch for this type of entity.
848 					// Check to see if the latest batch is acceptable.
849 					batchNumber = findBatchNumber( action, entityName );
850 				}
851 				else {
852 					// add an entry for this type of entity.
853 					// we can be assured that all referenced entities have already
854 					// been processed,
855 					// so specify that this entity is with the latest batch.
856 					// doing the batch number before adding the name to the list is
857 					// a faster way to get an accurate number.
858 
859 					batchNumber = actionBatches.size();
860 					latestBatches.put( entityName, batchNumber );
861 				}
862 				entityBatchNumber.put( currentEntity, batchNumber );
863 				addToBatch( batchNumber, action );
864 			}
865 			insertions.clear();
866 
867 			// now rebuild the insertions list. There is a batch for each entry in the name list.
868 			for ( int i = 0; i < actionBatches.size(); i++ ) {
869 				List<EntityInsertAction> batch = actionBatches.get( i );
870 				for ( EntityInsertAction action : batch ) {
871 					insertions.add( action );
872 				}
873 			}
874 		}
875 
876 		/**
877 		 * Finds an acceptable batch for this entity to be a member as part of the {@link InsertActionSorter}
878 		 *
879 		 * @param action The action being sorted
880 		 * @param entityName The name of the entity affected by the action
881 		 *
882 		 * @return An appropriate batch number; todo document this process better
883 		 */
884 		@SuppressWarnings({ "UnnecessaryBoxing", "unchecked" })
885 		private Integer findBatchNumber(
886 				EntityInsertAction action,
887 				String entityName) {
888 			// loop through all the associated entities and make sure they have been
889 			// processed before the latest
890 			// batch associated with this entity type.
891 
892 			// the current batch number is the latest batch for this entity type.
893 			Integer latestBatchNumberForType = latestBatches.get( entityName );
894 
895 			// loop through all the associations of the current entity and make sure that they are processed
896 			// before the current batch number
897 			Object[] propertyValues = action.getState();
898 			Type[] propertyTypes = action.getPersister().getClassMetadata()
899 					.getPropertyTypes();
900 
901 			for ( int i = 0; i < propertyValues.length; i++ ) {
902 				Object value = propertyValues[i];
903 				Type type = propertyTypes[i];
904 				if ( type.isEntityType() && value != null ) {
905 					// find the batch number associated with the current association, if any.
906 					Integer associationBatchNumber = entityBatchNumber.get( value );
907 					if ( associationBatchNumber != null && associationBatchNumber.compareTo( latestBatchNumberForType ) > 0 ) {
908 						// create a new batch for this type. The batch number is the number of current batches.
909 						latestBatchNumberForType = actionBatches.size();
910 						latestBatches.put( entityName, latestBatchNumberForType );
911 						// since this entity will now be processed in the latest possible batch,
912 						// we can be assured that it will come after all other associations,
913 						// there's not need to continue checking.
914 						break;
915 					}
916 				}
917 			}
918 			return latestBatchNumberForType;
919 		}
920 
921 		@SuppressWarnings({ "unchecked" })
922 		private void addToBatch(Integer batchNumber, EntityInsertAction action) {
923 			List<EntityInsertAction> actions = actionBatches.get( batchNumber );
924 
925 			if ( actions == null ) {
926 				actions = new LinkedList<EntityInsertAction>();
927 				actionBatches.put( batchNumber, actions );
928 			}
929 			actions.add( action );
930 		}
