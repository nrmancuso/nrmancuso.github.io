96  	public Binder(MetadataImplementor metadata, List<String> processedEntityNames) {

101 	/**
102 	 * Process an entity hierarchy.
103 	 *
104 	 * @param entityHierarchy THe hierarchy to process.
105 	 */
106 	public void processEntityHierarchy(EntityHierarchy entityHierarchy) {
107 		currentInheritanceType = entityHierarchy.getHierarchyInheritanceType();
108 		EntityBinding rootEntityBinding = createEntityBinding( entityHierarchy.getRootEntitySource(), null );
109 		if ( currentInheritanceType != InheritanceType.NO_INHERITANCE ) {
110 			processHierarchySubEntities( entityHierarchy.getRootEntitySource(), rootEntityBinding );
111 		}
112 		currentHierarchyEntityMode = null;
113 	}
114 
115 	private void processHierarchySubEntities(SubclassEntityContainer subclassEntitySource, EntityBinding superEntityBinding) {
116 		for ( SubclassEntitySource subEntity : subclassEntitySource.subclassEntitySources() ) {
117 			EntityBinding entityBinding = createEntityBinding( subEntity, superEntityBinding );
118 			processHierarchySubEntities( subEntity, entityBinding );
119 		}
120 	}

125 	private EntityBinding createEntityBinding(EntitySource entitySource, EntityBinding superEntityBinding) {
126 		if ( processedEntityNames.contains( entitySource.getEntityName() ) ) {
127 			return metadata.getEntityBinding( entitySource.getEntityName() );
128 		}
129 
130 		currentBindingContext = entitySource.getLocalBindingContext();
131 		try {
132 			final EntityBinding entityBinding = doCreateEntityBinding( entitySource, superEntityBinding );
133 
134 			metadata.addEntity( entityBinding );
135 			processedEntityNames.add( entityBinding.getEntity().getName() );
136 
137 			processFetchProfiles( entitySource, entityBinding );
138 
139 			return entityBinding;
140 		}
141 		finally {
142 			currentBindingContext = null;
143 		}
144 	}

146 	private EntityBinding doCreateEntityBinding(EntitySource entitySource, EntityBinding superEntityBinding) {
147 		final EntityBinding entityBinding = createBasicEntityBinding( entitySource, superEntityBinding );
148 
149 		bindSecondaryTables( entitySource, entityBinding );
150 		bindAttributes( entitySource, entityBinding );
151 
152 		bindTableUniqueConstraints( entitySource, entityBinding );
153 
154 		return entityBinding;
155 	}

157 	private EntityBinding createBasicEntityBinding(EntitySource entitySource, EntityBinding superEntityBinding) {
158 		if ( superEntityBinding == null ) {
159 			return makeRootEntityBinding( (RootEntitySource) entitySource );
160 		}
161 		else {
162 			switch ( currentInheritanceType ) {
163 				case SINGLE_TABLE:
164 					return makeDiscriminatedSubclassBinding( (SubclassEntitySource) entitySource, superEntityBinding );
165 				case JOINED:
166 					return makeJoinedSubclassBinding( (SubclassEntitySource) entitySource, superEntityBinding );
167 				case TABLE_PER_CLASS:
168 					return makeUnionedSubclassBinding( (SubclassEntitySource) entitySource, superEntityBinding );
169 				default:
170 					// extreme internal error!
171 					throw new AssertionFailure( "Internal condition failure" );
172 			}
173 		}
174 	}

176 	private EntityBinding makeRootEntityBinding(RootEntitySource entitySource) {
177 		currentHierarchyEntityMode = entitySource.getEntityMode();
178 
179 		final EntityBinding entityBinding = buildBasicEntityBinding( entitySource, null );
180 
181 		bindPrimaryTable( entitySource, entityBinding );
182 
183 		bindIdentifier( entitySource, entityBinding );
184 		bindVersion( entityBinding, entitySource );
185 		bindDiscriminator( entitySource, entityBinding );
186 
187 		entityBinding.getHierarchyDetails().setCaching( entitySource.getCaching() );
188 		entityBinding.getHierarchyDetails().setExplicitPolymorphism( entitySource.isExplicitPolymorphism() );
189 		entityBinding.getHierarchyDetails().setOptimisticLockStyle( entitySource.getOptimisticLockStyle() );
190 
191 		entityBinding.setMutable( entitySource.isMutable() );
192 		entityBinding.setWhereFilter( entitySource.getWhere() );
193 		entityBinding.setRowId( entitySource.getRowId() );
194 
195 		return entityBinding;
196 	}

198 	private EntityBinding buildBasicEntityBinding(EntitySource entitySource, EntityBinding superEntityBinding) {
199 		final EntityBinding entityBinding = superEntityBinding == null
200 				? new EntityBinding( currentInheritanceType, currentHierarchyEntityMode )
201 				: new EntityBinding( superEntityBinding );
202 
203 		final String entityName = entitySource.getEntityName();
204 		final String className = currentHierarchyEntityMode == EntityMode.POJO ? entitySource.getClassName() : null;
205 
206 		final Entity entity = new Entity(
207 				entityName,
208 				className,
209 				currentBindingContext.makeClassReference( className ),
210 				superEntityBinding == null ? null : superEntityBinding.getEntity()
211 		);
212 		entityBinding.setEntity( entity );
213 
214 		entityBinding.setJpaEntityName( entitySource.getJpaEntityName() );
215 
216 		if ( currentHierarchyEntityMode == EntityMode.POJO ) {
217 			final String proxy = entitySource.getProxy();
218 			if ( proxy != null ) {
219 				entityBinding.setProxyInterfaceType(
220 						currentBindingContext.makeClassReference(
221 								currentBindingContext.qualifyClassName( proxy )
222 						)
223 				);
224 				entityBinding.setLazy( true );
225 			}
226 			else if ( entitySource.isLazy() ) {
227 				entityBinding.setProxyInterfaceType( entityBinding.getEntity().getClassReferenceUnresolved() );
228 				entityBinding.setLazy( true );
229 			}
230 		}
231 		else {
232 			entityBinding.setProxyInterfaceType( null );
233 			entityBinding.setLazy( entitySource.isLazy() );
234 		}
235 
236 		final String customTuplizerClassName = entitySource.getCustomTuplizerClassName();
237 		if ( customTuplizerClassName != null ) {
238 			entityBinding.setCustomEntityTuplizerClass(
239 					currentBindingContext.<EntityTuplizer>locateClassByName(
240 							customTuplizerClassName
241 					)
242 			);
243 		}
244 
245 		final String customPersisterClassName = entitySource.getCustomPersisterClassName();
246 		if ( customPersisterClassName != null ) {
247 			entityBinding.setCustomEntityPersisterClass(
248 					currentBindingContext.<EntityPersister>locateClassByName(
249 							customPersisterClassName
250 					)
251 			);
252 		}
253 
254 		entityBinding.setMetaAttributeContext( buildMetaAttributeContext( entitySource ) );
255 
256 		entityBinding.setDynamicUpdate( entitySource.isDynamicUpdate() );
257 		entityBinding.setDynamicInsert( entitySource.isDynamicInsert() );
258 		entityBinding.setBatchSize( entitySource.getBatchSize() );
259 		entityBinding.setSelectBeforeUpdate( entitySource.isSelectBeforeUpdate() );
260 		entityBinding.setAbstract( entitySource.isAbstract() );
261 
262 		entityBinding.setCustomLoaderName( entitySource.getCustomLoaderName() );
263 		entityBinding.setCustomInsert( entitySource.getCustomSqlInsert() );
264 		entityBinding.setCustomUpdate( entitySource.getCustomSqlUpdate() );
265 		entityBinding.setCustomDelete( entitySource.getCustomSqlDelete() );
266 
267 		if ( entitySource.getSynchronizedTableNames() != null ) {
268 			entityBinding.addSynchronizedTableNames( entitySource.getSynchronizedTableNames() );
269 		}
270 
271 		entityBinding.setJpaCallbackClasses(entitySource.getJpaCallbackClasses());
272 
273 		return entityBinding;
274 	}

308 	private void bindIdentifier(RootEntitySource entitySource, EntityBinding entityBinding) {
309 		if ( entitySource.getIdentifierSource() == null ) {
310 			throw new AssertionFailure( "Expecting identifier information on root entity descriptor" );
311 		}
312 		switch ( entitySource.getIdentifierSource().getNature() ) {
313 			case SIMPLE: {
314 				bindSimpleIdentifier( (SimpleIdentifierSource) entitySource.getIdentifierSource(), entityBinding );
315 				break;
316 			}
317 			case AGGREGATED_COMPOSITE: {
318 				// composite id with an actual component class
319 			    break;
320 			}
321 			case COMPOSITE: {
322 				// what we used to term an "embedded composite identifier", which is not tobe confused with the JPA
323 				// term embedded. Specifically a composite id where there is no component class, though there may
324 				// be a @IdClass :/
325 			    break;
326 			}
327 		}
328 	}

330 	private void bindSimpleIdentifier(SimpleIdentifierSource identifierSource, EntityBinding entityBinding) {
331 		final BasicAttributeBinding idAttributeBinding = doBasicSingularAttributeBindingCreation(
332 				identifierSource.getIdentifierAttributeSource(), entityBinding
333 		);
334 
335 		entityBinding.getHierarchyDetails().getEntityIdentifier().setValueBinding( idAttributeBinding );
336 		IdGenerator generator = identifierSource.getIdentifierGeneratorDescriptor();
337 		if ( generator == null ) {
338 			Map<String, String> params = new HashMap<String, String>();
339 			params.put( IdentifierGenerator.ENTITY_NAME, entityBinding.getEntity().getName() );
340 			generator = new IdGenerator( "default_assign_identity_generator", "assigned", params );
341 		}
342 		entityBinding.getHierarchyDetails()
343 				.getEntityIdentifier()
344 				.setIdGenerator( generator );
345 
346 		final org.hibernate.metamodel.relational.Value relationalValue = idAttributeBinding.getValue();
347 
348 		if ( SimpleValue.class.isInstance( relationalValue ) ) {
349 			if ( !Column.class.isInstance( relationalValue ) ) {
350 				// this should never ever happen..
351 				throw new AssertionFailure( "Simple-id was not a column." );
352 			}
353 			entityBinding.getPrimaryTable().getPrimaryKey().addColumn( Column.class.cast( relationalValue ) );
354 		}
355 		else {
356 			for ( SimpleValue subValue : ( (Tuple) relationalValue ).values() ) {
357 				if ( Column.class.isInstance( subValue ) ) {
358 					entityBinding.getPrimaryTable().getPrimaryKey().addColumn( Column.class.cast( subValue ) );
359 				}
360 			}
361 		}
362 	}

365 		final SingularAttributeSource versioningAttributeSource = entitySource.getVersioningAttributeSource();
366 		if ( versioningAttributeSource == null ) {
367 			return;
368 		}
369 
370 		BasicAttributeBinding attributeBinding = doBasicSingularAttributeBindingCreation(
371 				versioningAttributeSource, entityBinding
372 		);
373 		entityBinding.getHierarchyDetails().setVersioningAttributeBinding( attributeBinding );
374 	}

376 	private void bindDiscriminator(RootEntitySource entitySource, EntityBinding entityBinding) {
377 		final DiscriminatorSource discriminatorSource = entitySource.getDiscriminatorSource();
378 		if ( discriminatorSource == null ) {
379 			return;
380 		}
381 
382 		EntityDiscriminator discriminator = new EntityDiscriminator();
383 		SimpleValue relationalValue = makeSimpleValue(
384 				entityBinding,
385 				discriminatorSource.getDiscriminatorRelationalValueSource()
386 		);
387 		discriminator.setBoundValue( relationalValue );
388 
389 		discriminator.getExplicitHibernateTypeDescriptor().setExplicitTypeName(
390 				discriminatorSource.getExplicitHibernateTypeName() != null
391 						? discriminatorSource.getExplicitHibernateTypeName()
392 						: "string"
393 		);
394 
395 		discriminator.setInserted( discriminatorSource.isInserted() );
396 		discriminator.setForced( discriminatorSource.isForced() );
397 
398 		entityBinding.getHierarchyDetails().setEntityDiscriminator( discriminator );
399 		entityBinding.setDiscriminatorMatchValue( entitySource.getDiscriminatorMatchValue() );
400 	}

410 	private void bindAttributes(AttributeSourceContainer attributeSourceContainer, AttributeBindingContainer attributeBindingContainer) {
411 		// todo : we really need the notion of a Stack here for the table from which the columns come for binding these attributes.
412 		// todo : adding the concept (interface) of a source of attribute metadata would allow reuse of this method for entity, component, unique-key, etc
413 		// for now, simply assume all columns come from the base table....
414 
415 		for ( AttributeSource attributeSource : attributeSourceContainer.attributeSources() ) {
416 			if ( attributeSource.isSingular() ) {
417 				final SingularAttributeSource singularAttributeSource = (SingularAttributeSource) attributeSource;
418 				if ( singularAttributeSource.getNature() == SingularAttributeNature.COMPONENT ) {
419 					bindComponent( (ComponentAttributeSource) singularAttributeSource, attributeBindingContainer );
420 				}
421 				else {
422 					doBasicSingularAttributeBindingCreation( singularAttributeSource, attributeBindingContainer );
423 				}
424 			}
425 			else {
426 				bindPersistentCollection( (PluralAttributeSource) attributeSource, attributeBindingContainer );
427 			}
428 		}
429 	}

431 	private void bindComponent(ComponentAttributeSource attributeSource, AttributeBindingContainer container) {
432 		final String attributeName = attributeSource.getName();
433 		SingularAttribute attribute = container.getAttributeContainer().locateComponentAttribute( attributeName );
434 		if ( attribute == null ) {
435 			final Component component = new Component(
436 					attributeSource.getPath(),
437 					attributeSource.getClassName(),
438 					attributeSource.getClassReference(),
439 					null // component inheritance not YET supported
440 			);
441 			attribute = container.getAttributeContainer().createComponentAttribute( attributeName, component );
442 		}
443 		ComponentAttributeBinding componentAttributeBinding = container.makeComponentAttributeBinding( attribute );
444 
445 		if ( StringHelper.isNotEmpty( attributeSource.getParentReferenceAttributeName() ) ) {
446 			final SingularAttribute parentReferenceAttribute =
447 					componentAttributeBinding.getComponent()
448 							.createSingularAttribute( attributeSource.getParentReferenceAttributeName() );
449 			componentAttributeBinding.setParentReference( parentReferenceAttribute );
450 		}
451 
452 		componentAttributeBinding.setMetaAttributeContext(
453 				buildMetaAttributeContext( attributeSource.metaAttributes(), container.getMetaAttributeContext() )
454 		);
455 
456 		bindAttributes( attributeSource, componentAttributeBinding );
457 	}

459 	private void bindPersistentCollection(PluralAttributeSource attributeSource, AttributeBindingContainer attributeBindingContainer) {
460 		final PluralAttribute existingAttribute = attributeBindingContainer.getAttributeContainer()
461 				.locatePluralAttribute( attributeSource.getName() );
462 		final AbstractPluralAttributeBinding pluralAttributeBinding;
463 
464 		if ( attributeSource.getPluralAttributeNature() == PluralAttributeNature.BAG ) {
465 			final PluralAttribute attribute = existingAttribute != null
466 					? existingAttribute
467 					: attributeBindingContainer.getAttributeContainer().createBag( attributeSource.getName() );
468 			pluralAttributeBinding = attributeBindingContainer.makeBagAttributeBinding(
469 					attribute,
470 					convert( attributeSource.getElementSource().getNature() )
471 			);
472 		}
473 		else if ( attributeSource.getPluralAttributeNature() == PluralAttributeNature.SET ) {
474 			final PluralAttribute attribute = existingAttribute != null
475 					? existingAttribute
476 					: attributeBindingContainer.getAttributeContainer().createSet( attributeSource.getName() );
477 			pluralAttributeBinding = attributeBindingContainer.makeSetAttributeBinding(
478 					attribute,
479 					convert( attributeSource.getElementSource().getNature() )
480 			);
481 		}
482 		else {
483 			// todo : implement other collection types
484 			throw new NotYetImplementedException( "Collections other than bag and set not yet implemented :(" );
485 		}
486 
487 		doBasicPluralAttributeBinding( attributeSource, pluralAttributeBinding );
488 
489 		bindCollectionTable( attributeSource, pluralAttributeBinding );
490 		bindSortingAndOrdering( attributeSource, pluralAttributeBinding );
491 
492 		bindCollectionKey( attributeSource, pluralAttributeBinding );
493 		bindCollectionElement( attributeSource, pluralAttributeBinding );
494 		bindCollectionIndex( attributeSource, pluralAttributeBinding );
495 
496 		metadata.addCollection( pluralAttributeBinding );
497 	}

499 	private void doBasicPluralAttributeBinding(PluralAttributeSource source, AbstractPluralAttributeBinding binding) {
500 		binding.setFetchTiming( source.getFetchTiming() );
501 		binding.setFetchStyle( source.getFetchStyle() );
502 		binding.setCascadeStyles( source.getCascadeStyles() );
503 
504 		binding.setCaching( source.getCaching() );
505 
506 		binding.getHibernateTypeDescriptor().setJavaTypeName(
507 				source.getPluralAttributeNature().reportedJavaType().getName()
508 		);
509 		binding.getHibernateTypeDescriptor().setExplicitTypeName( source.getTypeInformation().getName() );
510 		binding.getHibernateTypeDescriptor().getTypeParameters().putAll( source.getTypeInformation().getParameters() );
511 
512 		if ( StringHelper.isNotEmpty( source.getCustomPersisterClassName() ) ) {
513 			binding.setCollectionPersisterClass(
514 					currentBindingContext.<CollectionPersister>locateClassByName( source.getCustomPersisterClassName() )
515 			);
516 		}
517 
518 		if ( source.getCustomPersisterClassName() != null ) {
519 			binding.setCollectionPersisterClass(
520 					metadata.<CollectionPersister>locateClassByName( source.getCustomPersisterClassName() )
521 			);
522 		}
523 
524 		binding.setCustomLoaderName( source.getCustomLoaderName() );
525 		binding.setCustomSqlInsert( source.getCustomSqlInsert() );
526 		binding.setCustomSqlUpdate( source.getCustomSqlUpdate() );
527 		binding.setCustomSqlDelete( source.getCustomSqlDelete() );
528 		binding.setCustomSqlDeleteAll( source.getCustomSqlDeleteAll() );
529 
530 		binding.setMetaAttributeContext(
531 				buildMetaAttributeContext(
532 						source.metaAttributes(),
533 						binding.getContainer().getMetaAttributeContext()
534 				)
535 		);
536 
537 		doBasicAttributeBinding( source, binding );
538 	}

540 	private CollectionLaziness interpretLaziness(String laziness) {
541 		if ( laziness == null ) {
542 			laziness = Boolean.toString( metadata.getMappingDefaults().areAssociationsLazy() );
543 		}
544 
545 		if ( "extra".equals( laziness ) ) {
546 			return CollectionLaziness.EXTRA;
547 		}
548 		else if ( "false".equals( laziness ) ) {
549 			return CollectionLaziness.NOT;
550 		}
551 		else if ( "true".equals( laziness ) ) {
552 			return CollectionLaziness.LAZY;
553 		}
554 
555 		throw new MappingException(
556 				String.format( "Unexpected collection laziness value %s", laziness ),
557 				currentBindingContext.getOrigin()
558 		);
559 	}

561 	private void bindCollectionTable(
562 			PluralAttributeSource attributeSource,
563 			AbstractPluralAttributeBinding pluralAttributeBinding) {
564 		if ( attributeSource.getElementSource().getNature() == PluralAttributeElementNature.ONE_TO_MANY ) {
565 			return;
566 		}
567 
568 		final Schema.Name schemaName = Helper.determineDatabaseSchemaName(
569 				attributeSource.getExplicitSchemaName(),
570 				attributeSource.getExplicitCatalogName(),
571 				currentBindingContext
572 		);
573 		final Schema schema = metadata.getDatabase().locateSchema( schemaName );
574 
575 		final String tableName = attributeSource.getExplicitCollectionTableName();
576 		if ( StringHelper.isNotEmpty( tableName ) ) {
577 			final Identifier tableIdentifier = Identifier.toIdentifier(
578 					currentBindingContext.getNamingStrategy().tableName( tableName )
579 			);
580 			Table collectionTable = schema.locateTable( tableIdentifier );
581 			if ( collectionTable == null ) {
582 				collectionTable = schema.createTable( tableIdentifier );
583 			}
584 			pluralAttributeBinding.setCollectionTable( collectionTable );
585 		}
586 		else {
587 			// todo : not sure wel have all the needed info here in all cases, specifically when needing to know the "other side"
588 			final EntityBinding owner = pluralAttributeBinding.getContainer().seekEntityBinding();
589 			final String ownerTableLogicalName = Table.class.isInstance( owner.getPrimaryTable() )
590 					? Table.class.cast( owner.getPrimaryTable() ).getTableName().getName()
591 					: null;
592 			String collectionTableName = currentBindingContext.getNamingStrategy().collectionTableName(
593 					owner.getEntity().getName(),
594 					ownerTableLogicalName,

699 	private BasicAttributeBinding doBasicSingularAttributeBindingCreation(
700 			SingularAttributeSource attributeSource,
701 			AttributeBindingContainer attributeBindingContainer) {
702 		final SingularAttribute existingAttribute = attributeBindingContainer.getAttributeContainer()
703 				.locateSingularAttribute( attributeSource.getName() );
704 		final SingularAttribute attribute;
705 		if ( existingAttribute != null ) {
706 			attribute = existingAttribute;
707 		}
708 		else if ( attributeSource.isVirtualAttribute() ) {
709 			attribute = attributeBindingContainer.getAttributeContainer().createVirtualSingularAttribute(
710 					attributeSource.getName()
711 			);
712 		}
713 		else {
714 			attribute = attributeBindingContainer.getAttributeContainer()
715 					.createSingularAttribute( attributeSource.getName() );
716 		}
717 
718 		final BasicAttributeBinding attributeBinding;
719 		if ( attributeSource.getNature() == SingularAttributeNature.BASIC ) {
720 			attributeBinding = attributeBindingContainer.makeBasicAttributeBinding( attribute );
721 			resolveTypeInformation( attributeSource.getTypeInformation(), attributeBinding );
722 		}
723 		else if ( attributeSource.getNature() == SingularAttributeNature.MANY_TO_ONE ) {
724 			attributeBinding = attributeBindingContainer.makeManyToOneAttributeBinding( attribute );
725 			resolveTypeInformation( attributeSource.getTypeInformation(), attributeBinding );
726 			resolveToOneInformation(
727 					(ToOneAttributeSource) attributeSource,
728 					(ManyToOneAttributeBinding) attributeBinding
729 			);
730 		}
731 		else {
732 			throw new NotYetImplementedException();
733 		}
734 
735 		attributeBinding.setGeneration( attributeSource.getGeneration() );
736 		attributeBinding.setLazy( attributeSource.isLazy() );
737 		attributeBinding.setIncludedInOptimisticLocking( attributeSource.isIncludedInOptimisticLocking() );
738 
739 		attributeBinding.setPropertyAccessorName(
740 				Helper.getPropertyAccessorName(
741 						attributeSource.getPropertyAccessorName(),
742 						false,
743 						currentBindingContext.getMappingDefaults().getPropertyAccessorName()
744 				)
745 		);
746 
747 		bindRelationalValues( attributeSource, attributeBinding );
748 
749 		attributeBinding.setMetaAttributeContext(
750 				buildMetaAttributeContext(
751 						attributeSource.metaAttributes(),
752 						attributeBindingContainer.getMetaAttributeContext()
753 				)
754 		);
755 
756 		return attributeBinding;
757 	}

843 	private class PluralAttributeJavaTypeDeterminerDelegate implements BeanInfoHelper.BeanInfoDelegate {
844 		private final Class<?> ownerClass;
845 		private final String attributeName;
846 
847 		private Class<?> javaType = null;
848 
849 		private PluralAttributeJavaTypeDeterminerDelegate(Class<?> ownerClass, String attributeName) {
850 			this.ownerClass = ownerClass;
851 			this.attributeName = attributeName;
852 		}
853 
854 		@Override
855 		public void processBeanInfo(BeanInfo beanInfo) throws Exception {
856 			for ( PropertyDescriptor propertyDescriptor : beanInfo.getPropertyDescriptors() ) {
857 				if ( propertyDescriptor.getName().equals( attributeName ) ) {
858 					javaType = extractCollectionComponentType( beanInfo, propertyDescriptor );
859 					break;
860 				}
861 			}
862 		}
863 
864 		@SuppressWarnings( { "unchecked" })
865 		private Class<?> extractCollectionComponentType(BeanInfo beanInfo, PropertyDescriptor propertyDescriptor) {
866 			final java.lang.reflect.Type collectionAttributeType;
867 			if ( propertyDescriptor.getReadMethod() != null ) {
868 				collectionAttributeType = propertyDescriptor.getReadMethod().getGenericReturnType();
869 			}
870 			else if ( propertyDescriptor.getWriteMethod() != null ) {
871 				collectionAttributeType = propertyDescriptor.getWriteMethod().getGenericParameterTypes()[0];
872 			}
873 			else {
874 				// we need to look for the field and look at it...
875 				try {
876 					collectionAttributeType = ownerClass.getField( propertyDescriptor.getName() ).getGenericType();
877 				}
878 				catch ( Exception e ) {
879 					return null;
880 				}
881 			}
882 
883 			if ( ParameterizedType.class.isInstance( collectionAttributeType ) ) {
884 				final java.lang.reflect.Type[] types = ( (ParameterizedType) collectionAttributeType ).getActualTypeArguments();
885 				if ( types == null ) {
886 					return null;
887 				}
888 				else if ( types.length == 1 ) {
889 					return (Class<?>) types[0];
890 				}
891 				else if ( types.length == 2 ) {
892 					// Map<K,V>
893 					return (Class<?>) types[1];
894 				}
895 			}
896 			return null;
897 		}
898 	}

927 	private static MetaAttributeContext buildMetaAttributeContext(
928 			Iterable<MetaAttributeSource> metaAttributeSources,
929 			boolean onlyInheritable,
930 			MetaAttributeContext parentContext) {
931 		final MetaAttributeContext subContext = new MetaAttributeContext( parentContext );
932 
933 		for ( MetaAttributeSource metaAttributeSource : metaAttributeSources ) {
934 			if ( onlyInheritable & !metaAttributeSource.isInheritable() ) {
935 				continue;
936 			}
937 
938 			final String name = metaAttributeSource.getName();
939 			final MetaAttribute inheritedMetaAttribute = parentContext.getMetaAttribute( name );
940 			MetaAttribute metaAttribute = subContext.getLocalMetaAttribute( name );
941 			if ( metaAttribute == null || metaAttribute == inheritedMetaAttribute ) {
942 				metaAttribute = new MetaAttribute( name );
943 				subContext.add( metaAttribute );
944 			}
945 			metaAttribute.addValue( metaAttributeSource.getValue() );
946 		}
947 
948 		return subContext;
949 	}

1009 	private void bindRelationalValues(
1010 			RelationalValueSourceContainer relationalValueSourceContainer,
1011 			SingularAttributeBinding attributeBinding) {
1012 
1013 		List<SimpleValueBinding> valueBindings = new ArrayList<SimpleValueBinding>();
1014 
1015 		if ( !relationalValueSourceContainer.relationalValueSources().isEmpty() ) {
1016 			for ( RelationalValueSource valueSource : relationalValueSourceContainer.relationalValueSources() ) {
1017 				final TableSpecification table = attributeBinding.getContainer()
1018 						.seekEntityBinding()
1019 						.locateTable( valueSource.getContainingTableName() );
1020 
1021 				if ( ColumnSource.class.isInstance( valueSource ) ) {
1022 					final ColumnSource columnSource = ColumnSource.class.cast( valueSource );
1023 					final Column column = makeColumn( (ColumnSource) valueSource, table );
1024 					valueBindings.add(
1025 							new SimpleValueBinding(
1026 									column,
1027 									columnSource.isIncludedInInsert(),
1028 									columnSource.isIncludedInUpdate()
1029 							)
1030 					);
1031 				}
1032 				else {
1033 					valueBindings.add(
1034 							new SimpleValueBinding(
1035 									makeDerivedValue( ( (DerivedValueSource) valueSource ), table )
1036 							)
1037 					);
1038 				}
1039 			}
1040 		}
1041 		else {
1042 			String name = metadata.getOptions()
1043 					.getNamingStrategy()
1044 					.propertyToColumnName( attributeBinding.getAttribute().getName() );
1045 			name = quoteIdentifier( name );
1046 			Column column = attributeBinding.getContainer()
1047 									.seekEntityBinding()
1048 									.getPrimaryTable()
1049 									.locateOrCreateColumn( name );
1050 			column.setNullable( relationalValueSourceContainer.areValuesNullableByDefault() );
1051 			valueBindings.add(
1052 					new SimpleValueBinding(
1053 							column,
1054 							relationalValueSourceContainer.areValuesIncludedInInsertByDefault(),
1055 							relationalValueSourceContainer.areValuesIncludedInUpdateByDefault()
1056 					)
1057 			);
1058 		}
1059 		attributeBinding.setSimpleValueBindings( valueBindings );
1060 	}
