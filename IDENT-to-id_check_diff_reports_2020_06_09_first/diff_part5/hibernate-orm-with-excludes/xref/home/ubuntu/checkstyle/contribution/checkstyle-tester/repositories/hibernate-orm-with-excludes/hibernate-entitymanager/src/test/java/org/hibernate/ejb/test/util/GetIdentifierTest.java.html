27  import org.junit.Test;
28  import javax.persistence.EntityManager;
29  import org.hibernate.ejb.test.BaseEntityManagerFunctionalTestCase;
30  import org.hibernate.proxy.HibernateProxy;
31  import org.hibernate.testing.TestForIssue;
32  import static org.junit.Assert.assertEquals;
33  import static org.junit.Assert.assertTrue;
34  
35  /**
36   * @author Emmanuel Bernard
37   */
38  public class GetIdentifierTest extends BaseEntityManagerFunctionalTestCase {
39  	@Test
40  	public void testSimpleId() {
41  		EntityManager em = entityManagerFactory().createEntityManager();
42  		em.getTransaction().begin();
43  		Book book = new Book();
44  		em.persist( book );
45  		em.flush();
46  		assertEquals( book.getId(), em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier( book ) );
47  		em.getTransaction().rollback();
48  		em.close();
49  	}
50  
51  	@Test
52  	@TestForIssue(jiraKey = "HHH-7561")
53  	public void testProxyObject() {
54  		EntityManager em = entityManagerFactory().createEntityManager();
55  		em.getTransaction().begin();
56  		Book book = new Book();
57  		em.persist( book );
58  		em.flush();
59  		em.clear(); // Clear persistence context to receive proxy object below.
60  		Book proxy = em.getReference( Book.class, book.getId() );
61  		assertTrue( proxy instanceof HibernateProxy );
62  		assertEquals( book.getId(), em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier( proxy ) );
63  		em.getTransaction().rollback();
64  		em.close();
65  
66  		em = entityManagerFactory().createEntityManager();
67  		em.getTransaction().begin();
68  		Author author = new Author();
69  		Article article = new Article( author );
70  		em.persist( author );
71  		em.persist( article );
72  		em.flush();
73  		em.clear(); // Clear persistence context to receive proxy relation below.
74  		article = em.find( Article.class, article.getId() );
75  		assertTrue( article.getAuthor() instanceof HibernateProxy );
76  		assertEquals( author.getId(), em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier( article.getAuthor() ) );
77  		em.getTransaction().rollback();
78  		em.close();
79  	}
80  
81  	@Test
82  	public void testEmbeddedId() {
83  		EntityManager em = entityManagerFactory().createEntityManager();
84  		em.getTransaction().begin();
85  		Umbrella umbrella = new Umbrella();
86  		umbrella.setId( new Umbrella.PK() );
87  		umbrella.getId().setBrand( "Burberry" );
88  		umbrella.getId().setModel( "Red Hat" );
89  		em.persist( umbrella );
90  		em.flush();
91  		assertEquals( umbrella.getId(), em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier( umbrella ) );
92  		em.getTransaction().rollback();
93  		em.close();
94  	}
95  
96  	@Test
97  	public void testIdClass() {
98  		EntityManager em = entityManagerFactory().createEntityManager();
99  		em.getTransaction().begin();
100 		Sickness sick = new Sickness();
101 
102 		sick.setClassification( "H1N1" );
103 		sick.setType("Flu");
104 		em.persist( sick );
105 		em.flush();
106 		Sickness.PK id = new Sickness.PK();
107 		id.setClassification( sick.getClassification() );
108 		id.setType( sick.getType() );
109 		assertEquals( id, em.getEntityManagerFactory().getPersistenceUnitUtil().getIdentifier( sick ) );
110 		em.getTransaction().rollback();
111 		em.close();
112 	}
