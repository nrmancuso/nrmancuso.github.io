24  package org.hibernate.proxy.pojo.javassist;
25  import java.io.Serializable;
26  import java.lang.reflect.Method;
27  
28  import org.hibernate.HibernateException;
29  import org.hibernate.proxy.AbstractSerializableProxy;
30  import org.hibernate.proxy.HibernateProxy;
31  import org.hibernate.type.CompositeType;
32  
33  /**
34   * Serializable placeholder for Javassist proxies
35   */
36  public final class SerializableProxy extends AbstractSerializableProxy {
37  
38  	private Class persistentClass;
39  	private Class[] interfaces;
40  	private Class getIdentifierMethodClass;
41  	private Class setIdentifierMethodClass;
42  	private String getIdentifierMethodName;
43  	private String setIdentifierMethodName;
44  	private Class[] setIdentifierMethodParams;
45  	private CompositeType componentIdType;
46  
47  	public SerializableProxy() {
48  	}
49  
50  	public SerializableProxy(
51  			final String entityName,
52  			final Class persistentClass,
53  			final Class[] interfaces,
54  			final Serializable id,
55  			final Boolean readOnly,
56  			final Method getIdentifierMethod,
57  			final Method setIdentifierMethod,
58  			CompositeType componentIdType) {
59  		super( entityName, id, readOnly );
60  		this.persistentClass = persistentClass;
61  		this.interfaces = interfaces;
62  		if (getIdentifierMethod!=null) {
63  			getIdentifierMethodClass = getIdentifierMethod.getDeclaringClass();
64  			getIdentifierMethodName = getIdentifierMethod.getName();
65  		}
66  		if (setIdentifierMethod!=null) {
67  			setIdentifierMethodClass = setIdentifierMethod.getDeclaringClass();
68  			setIdentifierMethodName = setIdentifierMethod.getName();
69  			setIdentifierMethodParams = setIdentifierMethod.getParameterTypes();
70  		}
71  		this.componentIdType = componentIdType;
72  	}
73  
74  	private Object readResolve() {
75  		try {
76  			HibernateProxy proxy = JavassistLazyInitializer.getProxy(
77  				getEntityName(),
78  				persistentClass,
79  				interfaces,
80  				getIdentifierMethodName==null
81  						? null
82  						: getIdentifierMethodClass.getDeclaredMethod( getIdentifierMethodName, (Class[]) null ),
83  				setIdentifierMethodName==null
84  						? null 
85  						: setIdentifierMethodClass.getDeclaredMethod(setIdentifierMethodName, setIdentifierMethodParams),
86  				componentIdType,
87  				getId(),
88  				null
89  			);
90  			setReadOnlyBeforeAttachedToSession( ( JavassistLazyInitializer ) proxy.getHibernateLazyInitializer() );
91  			return proxy;
92  		}
93  		catch (NoSuchMethodException nsme) {
94  			throw new HibernateException("could not create proxy for entity: " + getEntityName(), nsme);
95  		}
96  	}
97  }
