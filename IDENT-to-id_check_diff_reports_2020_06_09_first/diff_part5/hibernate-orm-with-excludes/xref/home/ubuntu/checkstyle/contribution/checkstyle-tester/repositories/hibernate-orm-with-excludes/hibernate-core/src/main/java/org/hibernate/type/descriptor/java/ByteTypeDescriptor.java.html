24  package org.hibernate.type.descriptor.java;
25  import org.hibernate.type.descriptor.WrapperOptions;
26  
27  /**
28   * Descriptor for {@link Byte} handling.
29   *
30   * @author Steve Ebersole
31   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
32   */
33  public class ByteTypeDescriptor extends AbstractTypeDescriptor<Byte> {
34  	public static final ByteTypeDescriptor INSTANCE = new ByteTypeDescriptor();
35  
36  	public ByteTypeDescriptor() {
37  		super( Byte.class );
38  	}
39  
40  	public String toString(Byte value) {
41  		return value == null ? null : value.toString();
42  	}
43  
44  	public Byte fromString(String string) {
45  		return Byte.valueOf( string );
46  	}
47  
48  	@SuppressWarnings({ "unchecked" })
49  	public <X> X unwrap(Byte value, Class<X> type, WrapperOptions options) {
50  		if ( value == null ) {
51  			return null;
52  		}
53  		if ( Byte.class.isAssignableFrom( type ) ) {
54  			return (X) value;
55  		}
56  		if ( Short.class.isAssignableFrom( type ) ) {
57  			return (X) Short.valueOf( value.shortValue() );
58  		}
59  		if ( Integer.class.isAssignableFrom( type ) ) {
60  			return (X) Integer.valueOf( value.intValue() );
61  		}
62  		if ( Long.class.isAssignableFrom( type ) ) {
63  			return (X) Long.valueOf( value.longValue() );
64  		}
65  		if ( Double.class.isAssignableFrom( type ) ) {
66  			return (X) Double.valueOf( value.doubleValue() );
67  		}
68  		if ( Float.class.isAssignableFrom( type ) ) {
69  			return (X) Float.valueOf( value.floatValue() );
70  		}
71  		if ( String.class.isAssignableFrom( type ) ) {
72  			return (X) value.toString();
73  		}
74  		throw unknownUnwrap( type );
75  	}
76  
77  	@SuppressWarnings({ "UnnecessaryBoxing" })
78  	public <X> Byte wrap(X value, WrapperOptions options) {
79  		if ( value == null ) {
80  			return null;
81  		}
82  		if ( Byte.class.isInstance( value ) ) {
83  			return (Byte) value;
84  		}
85  		if ( Number.class.isInstance( value ) ) {
86  			return Byte.valueOf( ( (Number) value ).byteValue() );
87  		}
88  		if ( String.class.isInstance( value ) ) {
89  			return Byte.valueOf( ( (String) value ) );
90  		}
91  		throw unknownWrap( value.getClass() );
92  	}
93  }
