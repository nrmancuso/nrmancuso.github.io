95  	/**
96  	 * Prepares the save call using a newly generated id.
97  	 *
98  	 * @param entity The entity to be saved
99  	 * @param entityName The entity-name for the entity to be saved
100 	 * @param anything Generally cascade-specific information.
101 	 * @param source The session which is the source of this save event.
102 	 * @param requiresImmediateIdAccess does the event context require
103 	 * access to the identifier immediately after execution of this method (if
104 	 * not, post-insert style id generators may be postponed if we are outside
105 	 * a transaction).
106 	 *
107 	 * @return The id used to save the entity; may be null depending on the
108 	 *         type of id generator used and the requiresImmediateIdAccess value
109 	 */
110 	protected Serializable saveWithGeneratedId(
111 			Object entity,
112 			String entityName,
113 			Object anything,
114 			EventSource source,
115 			boolean requiresImmediateIdAccess) {
116 		EntityPersister persister = source.getEntityPersister( entityName, entity );
117 		Serializable generatedId = persister.getIdentifierGenerator().generate( source, entity );
118 		if ( generatedId == null ) {
119 			throw new IdentifierGenerationException( "null id generated for:" + entity.getClass() );
120 		}
121 		else if ( generatedId == IdentifierGeneratorHelper.SHORT_CIRCUIT_INDICATOR ) {
122 			return source.getIdentifier( entity );
123 		}
124 		else if ( generatedId == IdentifierGeneratorHelper.POST_INSERT_INDICATOR ) {
125 			return performSave( entity, null, persister, true, anything, source, requiresImmediateIdAccess );
126 		}
127 		else {
128 			// TODO: define toString()s for generators
129 			if ( LOG.isDebugEnabled() ) {
130 				LOG.debugf( "Generated identifier: %s, using strategy: %s",
131 						persister.getIdentifierType().toLoggableString( generatedId, source.getFactory() ),
132 						persister.getIdentifierGenerator().getClass().getName() );
133 			}
134 
135 			return performSave( entity, generatedId, persister, false, anything, source, true );
136 		}
137 	}

139 	/**
140 	 * Prepares the save call by checking the session caches for a pre-existing
141 	 * entity and performing any lifecycle callbacks.
142 	 *
143 	 * @param entity The entity to be saved.
144 	 * @param id The id by which to save the entity.
145 	 * @param persister The entity's persister instance.
146 	 * @param useIdentityColumn Is an identity column being used?
147 	 * @param anything Generally cascade-specific information.
148 	 * @param source The session from which the event originated.
149 	 * @param requiresImmediateIdAccess does the event context require
150 	 * access to the identifier immediately after execution of this method (if
151 	 * not, post-insert style id generators may be postponed if we are outside
152 	 * a transaction).
153 	 *
154 	 * @return The id used to save the entity; may be null depending on the
155 	 *         type of id generator used and the requiresImmediateIdAccess value
156 	 */
157 	protected Serializable performSave(
158 			Object entity,
159 			Serializable id,
160 			EntityPersister persister,
161 			boolean useIdentityColumn,
162 			Object anything,
163 			EventSource source,
164 			boolean requiresImmediateIdAccess) {
165 
166 		if ( LOG.isTraceEnabled() ) {
167 			LOG.tracev( "Saving {0}", MessageHelper.infoString( persister, id, source.getFactory() ) );
168 		}
169 
170 		final EntityKey key;
171 		if ( !useIdentityColumn ) {
172 			key = source.generateEntityKey( id, persister );
173 			Object old = source.getPersistenceContext().getEntity( key );
174 			if ( old != null ) {
175 				if ( source.getPersistenceContext().getEntry( old ).getStatus() == Status.DELETED ) {
176 					source.forceFlush( source.getPersistenceContext().getEntry( old ) );
177 				}
178 				else {
179 					throw new NonUniqueObjectException( id, persister.getEntityName() );
180 				}
181 			}
182 			persister.setIdentifier( entity, id, source );
183 		}
184 		else {
185 			key = null;
186 		}
187 
188 		if ( invokeSaveLifecycle( entity, persister, source ) ) {
189 			return id; //EARLY EXIT
190 		}
191 
192 		return performSaveOrReplicate(
193 				entity,
194 				key,
195 				persister,
196 				useIdentityColumn,
197 				anything,
198 				source,
199 				requiresImmediateIdAccess
200 		);
201 	}
202 
203 	protected boolean invokeSaveLifecycle(Object entity, EntityPersister persister, EventSource source) {
204 		// Sub-insertions should occur before containing insertion so
205 		// Try to do the callback now
206 		if ( persister.implementsLifecycle() ) {
207 			LOG.debug( "Calling onSave()" );
208 			if ( ( ( Lifecycle ) entity ).onSave( source ) ) {
209 				LOG.debug( "Insertion vetoed by onSave()" );
210 				return true;
211 			}
212 		}
213 		return false;
214 	}

216 	/**
217 	 * Performs all the actual work needed to save an entity (well to get the save moved to
218 	 * the execution queue).
219 	 *
220 	 * @param entity The entity to be saved
221 	 * @param key The id to be used for saving the entity (or null, in the case of identity columns)
222 	 * @param persister The entity's persister instance.
223 	 * @param useIdentityColumn Should an identity column be used for id generation?
224 	 * @param anything Generally cascade-specific information.
225 	 * @param source The session which is the source of the current event.
226 	 * @param requiresImmediateIdAccess Is access to the identifier required immediately
227 	 * after the completion of the save?  persist(), for example, does not require this...
228 	 *
229 	 * @return The id used to save the entity; may be null depending on the
230 	 *         type of id generator used and the requiresImmediateIdAccess value
231 	 */
232 	protected Serializable performSaveOrReplicate(
233 			Object entity,
234 			EntityKey key,
235 			EntityPersister persister,
236 			boolean useIdentityColumn,
237 			Object anything,
238 			EventSource source,
239 			boolean requiresImmediateIdAccess) {
240 
241 		Serializable id = key == null ? null : key.getIdentifier();
242 
243 		boolean inTxn = source.getTransactionCoordinator().isTransactionInProgress();
244 		boolean shouldDelayIdentityInserts = !inTxn && !requiresImmediateIdAccess;
245 
246 		// Put a placeholder in entries, so we don't recurse back and try to save() the
247 		// same object again. QUESTION: should this be done before onSave() is called?
248 		// likewise, should it be done before onUpdate()?
249 		source.getPersistenceContext().addEntry(
250 				entity,
251 				Status.SAVING,
252 				null,
253 				null,
254 				id,
255 				null,
256 				LockMode.WRITE,
257 				useIdentityColumn,
258 				persister,
259 				false,
260 				false
261 		);
262 
263 		cascadeBeforeSave( source, persister, entity, anything );
264 
265 		Object[] values = persister.getPropertyValuesToInsert( entity, getMergeMap( anything ), source );
266 		Type[] types = persister.getPropertyTypes();
267 
268 		boolean substitute = substituteValuesIfNecessary( entity, id, values, persister, source );
269 
270 		if ( persister.hasCollections() ) {
271 			substitute = substitute || visitCollectionsBeforeSave( entity, id, values, types, source );
272 		}
273 
274 		if ( substitute ) {
275 			persister.setPropertyValues( entity, values );
276 		}
277 
278 		TypeHelper.deepCopy(
279 				values,
280 				types,
281 				persister.getPropertyUpdateability(),
282 				values,
283 				source
284 		);
285 
286 		AbstractEntityInsertAction insert = addInsertAction(
287 				values, id, entity, persister, useIdentityColumn, source, shouldDelayIdentityInserts
288 		);
289 
290 		// postpone initializing id in case the insert has non-nullable transient dependencies
291 		// that are not resolved until cascadeAfterSave() is executed
292 		cascadeAfterSave( source, persister, entity, anything );
293 		if ( useIdentityColumn && insert.isEarlyInsert() ) {
294 			if ( ! EntityIdentityInsertAction.class.isInstance( insert ) ) {
295 				throw new IllegalStateException(
296 						"Insert should be using an identity column, but action is of unexpected type: " +
297 								insert.getClass().getName() );
298 			}
299 			id = ( ( EntityIdentityInsertAction ) insert ).getGeneratedId();
300 			
301 			insert.handleNaturalIdPostSaveNotifications(id);
302 		}
303 
304 		markInterceptorDirty( entity, persister, source );
305 
306 		return id;
307 	}

309 	private AbstractEntityInsertAction addInsertAction(
310 			Object[] values,
311 			Serializable id,
312 			Object entity,
313 			EntityPersister persister,
314 			boolean useIdentityColumn,
315 			EventSource source,
316 			boolean shouldDelayIdentityInserts) {
317 		if ( useIdentityColumn ) {
318 			EntityIdentityInsertAction insert = new EntityIdentityInsertAction(
319 					values, entity, persister, isVersionIncrementDisabled(), source, shouldDelayIdentityInserts
320 			);
321 			source.getActionQueue().addAction( insert );
322 			return insert;
323 		}
324 		else {
325 			Object version = Versioning.getVersion( values, persister );
326 			EntityInsertAction insert = new EntityInsertAction(
327 					id, values, entity, version, persister, isVersionIncrementDisabled(), source
328 			);
329 			source.getActionQueue().addAction( insert );
330 			return insert;
331 		}
332 	}

334 	private void markInterceptorDirty(Object entity, EntityPersister persister, EventSource source) {
335 		if ( persister.getInstrumentationMetadata().isInstrumented() ) {
336 			FieldInterceptor interceptor = persister.getInstrumentationMetadata().injectInterceptor(
337 					entity,
338 					persister.getEntityName(),
339 					null,
340 					source
341 			);
342 			interceptor.dirty();
343 		}
344 	}

361 	protected boolean visitCollectionsBeforeSave(Object entity, Serializable id, Object[] values, Type[] types, EventSource source) {
362 		WrapVisitor visitor = new WrapVisitor( source );
363 		// substitutes into values by side-effect
364 		visitor.processEntityPropertyValues( values, types );
365 		return visitor.isSubstitutionRequired();
366 	}

368 	/**
369 	 * Perform any property value substitution that is necessary
370 	 * (interceptor callback, version initialization...)
371 	 *
372 	 * @param entity The entity
373 	 * @param id The entity identifier
374 	 * @param values The snapshot entity state
375 	 * @param persister The entity persister
376 	 * @param source The originating session
377 	 *
378 	 * @return True if the snapshot state changed such that
379 	 * reinjection of the values into the entity is required.
380 	 */
381 	protected boolean substituteValuesIfNecessary(
382 			Object entity,
383 			Serializable id,
384 			Object[] values,
385 			EntityPersister persister,
386 			SessionImplementor source) {
387 		boolean substitute = source.getInterceptor().onSave(
388 				entity,
389 				id,
390 				values,
391 				persister.getPropertyNames(),
392 				persister.getPropertyTypes()
393 		);
394 
395 		//keep the existing version number in the case of replicate!
396 		if ( persister.isVersioned() ) {
397 			substitute = Versioning.seedVersion(
398 					values,
399 					persister.getVersionProperty(),
400 					persister.getVersionType(),
401 					source
402 			) || substitute;
403 		}
404 		return substitute;
405 	}

407 	/**
408 	 * Handles the calls needed to perform pre-save cascades for the given entity.
409 	 *
410 	 * @param source The session from whcih the save event originated.
411 	 * @param persister The entity's persister instance.
412 	 * @param entity The entity to be saved.
413 	 * @param anything Generally cascade-specific data
414 	 */
415 	protected void cascadeBeforeSave(
416 			EventSource source,
417 			EntityPersister persister,
418 			Object entity,
419 			Object anything) {
420 
421 		// cascade-save to many-to-one BEFORE the parent is saved
422 		source.getPersistenceContext().incrementCascadeLevel();
423 		try {
424 			new Cascade( getCascadeAction(), Cascade.BEFORE_INSERT_AFTER_DELETE, source )
425 					.cascade( persister, entity, anything );
426 		}
427 		finally {
428 			source.getPersistenceContext().decrementCascadeLevel();
429 		}
430 	}
431 
432 	/**
433 	 * Handles to calls needed to perform post-save cascades.
434 	 *
435 	 * @param source The session from which the event originated.
436 	 * @param persister The entity's persister instance.
437 	 * @param entity The entity beng saved.
438 	 * @param anything Generally cascade-specific data
439 	 */
440 	protected void cascadeAfterSave(
441 			EventSource source,
442 			EntityPersister persister,
443 			Object entity,
444 			Object anything) {
445 
446 		// cascade-save to collections AFTER the collection owner was saved
447 		source.getPersistenceContext().incrementCascadeLevel();
448 		try {
449 			new Cascade( getCascadeAction(), Cascade.AFTER_INSERT_BEFORE_DELETE, source )
450 					.cascade( persister, entity, anything );
451 		}
452 		finally {
453 			source.getPersistenceContext().decrementCascadeLevel();
454 		}
455 	}

459 	/**
460 	 * Determine whether the entity is persistent, detached, or transient
461 	 *
462 	 * @param entity The entity to check
463 	 * @param entityName The name of the entity
464 	 * @param entry The entity's entry in the persistence context
465 	 * @param source The originating session.
466 	 *
467 	 * @return The state.
468 	 */
469 	protected EntityState getEntityState(
470 			Object entity,
471 			String entityName,
472 			EntityEntry entry, //pass this as an argument only to avoid double looking
473 			SessionImplementor source) {
474 
475 		final boolean traceEnabled = LOG.isTraceEnabled();
476 		if ( entry != null ) { // the object is persistent
477 
478 			//the entity is associated with the session, so check its status
479 			if ( entry.getStatus() != Status.DELETED ) {
480 				// do nothing for persistent instances
481 				if ( traceEnabled ) {
482 					LOG.tracev( "Persistent instance of: {0}", getLoggableName( entityName, entity ) );
483 				}
484 				return EntityState.PERSISTENT;
485 			}
486 			// ie. e.status==DELETED
487 			if ( traceEnabled ) {
488 				LOG.tracev( "Deleted instance of: {0}", getLoggableName( entityName, entity ) );
489 			}
490 			return EntityState.DELETED;
491 		}
492 		// the object is transient or detached
493 
494 		// the entity is not associated with the session, so
495 		// try interceptor and unsaved-value
496 
497 		if ( ForeignKeys.isTransient( entityName, entity, getAssumedUnsaved(), source )) {
498 			if (  traceEnabled ) {
499 				LOG.tracev( "Transient instance of: {0}", getLoggableName( entityName, entity ) );
500 			}
501 			return EntityState.TRANSIENT;
502 		}
503 		if ( traceEnabled ) {
504 			LOG.tracev( "Detached instance of: {0}", getLoggableName( entityName, entity ) );
505 		}
506 		return EntityState.DETACHED;
507 	}

509 	protected String getLoggableName(String entityName, Object entity) {
510 		return entityName == null ? entity.getClass().getName() : entityName;
511 	}
