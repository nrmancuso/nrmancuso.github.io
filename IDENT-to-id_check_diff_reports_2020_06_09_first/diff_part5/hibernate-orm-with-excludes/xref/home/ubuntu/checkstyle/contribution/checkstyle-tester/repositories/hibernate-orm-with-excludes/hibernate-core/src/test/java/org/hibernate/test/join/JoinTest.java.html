54  	@Test
55  	public void testSequentialSelects() {
56  		Session s = openSession();
57  		Transaction t = s.beginTransaction();
58  		
59  		Employee mark = new Employee();
60  		mark.setName("Mark");
61  		mark.setTitle("internal sales");
62  		mark.setSex('M');
63  		mark.setAddress("buckhead");
64  		mark.setZip("30305");
65  		mark.setCountry("USA");
66  		
67  		Customer joe = new Customer();
68  		joe.setName("Joe");
69  		joe.setAddress("San Francisco");
70  		joe.setZip("XXXXX");
71  		joe.setCountry("USA");
72  		joe.setComments("Very demanding");
73  		joe.setSex('M');
74  		joe.setSalesperson(mark);
75  		
76  		Person yomomma = new Person();
77  		yomomma.setName("mum");
78  		yomomma.setSex('F');
79  		
80  		s.save(yomomma);
81  		s.save(mark);
82  		s.save(joe);		
83  		
84  		assertEquals( s.createQuery("from java.io.Serializable").list().size(), 0 );
85  		
86  		assertEquals( s.createQuery("from Person").list().size(), 3 );
87  		assertEquals( s.createQuery("from Person p where p.class is null").list().size(), 1 );
88  		assertEquals( s.createQuery("from Person p where p.class = Customer").list().size(), 1 );
89  		assertTrue(s.createQuery("from Customer c").list().size()==1);
90  		s.clear();
91  
92  		List customers = s.createQuery("from Customer c left join fetch c.salesperson").list();
93  		for ( Iterator iter = customers.iterator(); iter.hasNext(); ) {
94  			Customer c = (Customer) iter.next();
95  			assertTrue( Hibernate.isInitialized( c.getSalesperson() ) );
96  			assertEquals( c.getSalesperson().getName(), "Mark" );
97  		}
98  		assertEquals( customers.size(), 1 );
99  		s.clear();
100 		
101 		customers = s.createQuery("from Customer").list();
102 		for ( Iterator iter = customers.iterator(); iter.hasNext(); ) {
103 			Customer c = (Customer) iter.next();
104 			assertFalse( Hibernate.isInitialized( c.getSalesperson() ) );
105 			assertEquals( c.getSalesperson().getName(), "Mark" );
106 		}
107 		assertEquals( customers.size(), 1 );
108 		s.clear();
109 		
110 
111 		mark = (Employee) s.get( Employee.class, new Long( mark.getId() ) );
112 		joe = (Customer) s.get( Customer.class, new Long( joe.getId() ) );
113 		
114  		mark.setZip("30306");
115 		assertEquals( s.createQuery("from Person p where p.zip = '30306'").list().size(), 1 );
116 		s.delete(mark);
117 		s.delete(joe);
118 		s.delete(yomomma);
119 		assertTrue( s.createQuery("from Person").list().isEmpty() );
120 		t.commit();
121 		s.close();
122 	}

124 	@Test
125 	public void testSequentialSelectsOptionalData() throws Exception {
126 		Session s = openSession();
127 		Transaction t = s.beginTransaction();
128 
129 		User jesus = new User();
130 		jesus.setName("Jesus Olvera y Martinez");
131 		jesus.setSex('M');
132 
133 		s.save(jesus);
134 
135 		assertEquals( s.createQuery("from java.io.Serializable").list().size(), 0 );
136 		
137 		assertEquals( s.createQuery("from Person").list().size(), 1 );
138 		assertEquals( s.createQuery("from Person p where p.class is null").list().size(), 0 );
139 		assertEquals( s.createQuery("from Person p where p.class = User").list().size(), 1 );
140 		assertTrue(s.createQuery("from User u").list().size()==1);
141 		s.clear();
142 
143 		// Remove the optional row from the join table and requery the User obj
144 		doWork(s);
145 		s.clear();
146 
147 		jesus = (User) s.get( Person.class, new Long( jesus.getId() ) );
148 		s.clear();
149 
150 		// Cleanup the test data
151 		s.delete(jesus);
152 
153 		assertTrue( s.createQuery("from Person").list().isEmpty() );
154 		t.commit();
155 		s.close();
156 	}

158 	private void doWork(final Session s) {
159 		s.doWork(
160 				new AbstractWork() {
161 					@Override
162 					public void execute(Connection connection) throws SQLException {
163 						PreparedStatement ps = connection.prepareStatement( "delete from t_user" );
164 						ps.execute();
165 					}
166 				}
167 		);
168 	}

170 	@Test
171 	public void testCustomColumnReadAndWrite() {
172 		Session s = openSession();
173 		Transaction t = s.beginTransaction();
174 		final double HEIGHT_INCHES = 73;
175 		final double HEIGHT_CENTIMETERS = HEIGHT_INCHES * 2.54d;
176 		Person p = new Person();
177 		p.setName("Emmanuel");
178 		p.setSex('M');
179 		p.setHeightInches(HEIGHT_INCHES);
180 		s.persist(p);
181 		final double PASSWORD_EXPIRY_WEEKS = 4;
182 		final double PASSWORD_EXPIRY_DAYS = PASSWORD_EXPIRY_WEEKS * 7d;
183 		User u = new User();
184 		u.setName("Steve");
185 		u.setSex('M');
186 		u.setPasswordExpiryDays(PASSWORD_EXPIRY_DAYS);
187 		s.persist(u);
188 		s.flush();
189 		
190 		// Test value conversion during insert
191 		// Oracle returns BigDecimaal while other dialects return Double;
192 		// casting to Number so it works on all dialects
193 		Number heightViaSql = (Number)s.createSQLQuery("select height_centimeters from person where name='Emmanuel'").uniqueResult();
194 		assertEquals(HEIGHT_CENTIMETERS, heightViaSql.doubleValue(), 0.01d);
195 		Number expiryViaSql = (Number)s.createSQLQuery("select pwd_expiry_weeks from t_user where person_id=?")
196 			.setLong(0, u.getId())
197 			.uniqueResult();
198 		assertEquals(PASSWORD_EXPIRY_WEEKS, expiryViaSql.doubleValue(), 0.01d);
199 		
200 		// Test projection
201 		Double heightViaHql = (Double)s.createQuery("select p.heightInches from Person p where p.name = 'Emmanuel'").uniqueResult();
202 		assertEquals(HEIGHT_INCHES, heightViaHql, 0.01d);
203 		Double expiryViaHql = (Double)s.createQuery("select u.passwordExpiryDays from User u where u.name = 'Steve'").uniqueResult();
204 		assertEquals(PASSWORD_EXPIRY_DAYS, expiryViaHql, 0.01d);
205 		
206 		// Test restriction and entity load via criteria
207 		p = (Person)s.createCriteria(Person.class)
208 			.add(Restrictions.between("heightInches", HEIGHT_INCHES - 0.01d, HEIGHT_INCHES + 0.01d))
209 			.uniqueResult();
210 		assertEquals(HEIGHT_INCHES, p.getHeightInches(), 0.01d);
211 		u = (User)s.createCriteria(User.class)
212 			.add(Restrictions.between("passwordExpiryDays", PASSWORD_EXPIRY_DAYS - 0.01d, PASSWORD_EXPIRY_DAYS + 0.01d))
213 			.uniqueResult();
214 		assertEquals(PASSWORD_EXPIRY_DAYS, u.getPasswordExpiryDays(), 0.01d);
215 		
216 		// Test predicate and entity load via HQL
217 		p = (Person)s.createQuery("from Person p where p.heightInches between ? and ?")
218 			.setDouble(0, HEIGHT_INCHES - 0.01d)
219 			.setDouble(1, HEIGHT_INCHES + 0.01d)
220 			.uniqueResult();
221 		assertEquals(HEIGHT_INCHES, p.getHeightInches(), 0.01d);
222 		u = (User)s.createQuery("from User u where u.passwordExpiryDays between ? and ?")
223 			.setDouble(0, PASSWORD_EXPIRY_DAYS - 0.01d)
224 			.setDouble(1, PASSWORD_EXPIRY_DAYS + 0.01d)
225 			.uniqueResult();
226 		assertEquals(PASSWORD_EXPIRY_DAYS, u.getPasswordExpiryDays(), 0.01d);
227 		
228 		// Test update
229 		p.setHeightInches(1);
230 		u.setPasswordExpiryDays(7d);
231 		s.flush();
232 		heightViaSql = (Number)s.createSQLQuery("select height_centimeters from person where name='Emmanuel'").uniqueResult();
233 		assertEquals(2.54d, heightViaSql.doubleValue(), 0.01d);
234 		expiryViaSql = (Number)s.createSQLQuery("select pwd_expiry_weeks from t_user where person_id=?")
235 			.setLong(0, u.getId())
236 			.uniqueResult();
237 		assertEquals(1d, expiryViaSql.doubleValue(), 0.01d);
238 		
239 		s.delete(p);
240 		s.delete(u);
241 		assertTrue( s.createQuery("from Person").list().isEmpty() );		
242 		
243 		t.commit();
244 		s.close();
245 		
246 	}
