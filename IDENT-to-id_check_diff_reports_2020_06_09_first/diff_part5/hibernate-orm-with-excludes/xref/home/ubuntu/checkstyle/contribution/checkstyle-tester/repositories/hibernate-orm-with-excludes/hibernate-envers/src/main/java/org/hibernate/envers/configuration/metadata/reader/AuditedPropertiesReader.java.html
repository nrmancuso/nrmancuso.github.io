38  /**
39   * Reads persistent properties form a
40   * {@link org.hibernate.envers.configuration.metadata.reader.PersistentPropertiesSource}
41   * and adds the ones that are audited to a
42   * {@link org.hibernate.envers.configuration.metadata.reader.AuditedPropertiesHolder},
43   * filling all the auditing data.
44   * @author Adam Warski (adam at warski dot org)
45   * @author Erik-Berndt Scheper
46   * @author Hern&aacut;n Chanfreau
47   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
48   * @author Michal Skowronek (mskowr at o2 dot pl)
49   */

93  	public void read() {
94  		// First reading the access types for the persistent properties.
95  		readPersistentPropertiesAccess();
96  
97          // Retrieve classes and properties that are explicitly marked for auditing process by any superclass
98          // of currently mapped entity or itself.
99          XClass clazz = persistentPropertiesSource.getXClass();
100         readAuditOverrides(clazz);
101 
102         // Adding all properties from the given class.
103         addPropertiesFromClass(clazz);
104 	}

106     /**
107      * Recursively constructs sets of audited and not audited properties and classes which behavior has been overridden
108      * using {@link AuditOverride} annotation.
109      * @param clazz Class that is being processed. Currently mapped entity shall be passed during first invocation.
110      */
111     private void readAuditOverrides(XClass clazz) {
112         /* TODO: Code to remove with @Audited.auditParents - start. */
113         Audited allClassAudited = clazz.getAnnotation(Audited.class);
114         if (allClassAudited != null && allClassAudited.auditParents().length > 0) {
115             for (Class c : allClassAudited.auditParents()) {
116                 XClass parentClass = reflectionManager.toXClass(c);
117                 checkSuperclass(clazz, parentClass);
118                 if (!overriddenNotAuditedClasses.contains(parentClass)) {
119                     // If the class has not been marked as not audited by the subclass.
120                     overriddenAuditedClasses.add(parentClass);
121                 }
122             }
123         }
124         /* TODO: Code to remove with @Audited.auditParents - finish. */
125         List<AuditOverride> auditOverrides = computeAuditOverrides(clazz);
126         for (AuditOverride auditOverride : auditOverrides) {
127             if (auditOverride.forClass() != void.class) {
128                 XClass overrideClass = reflectionManager.toXClass(auditOverride.forClass());
129                 checkSuperclass(clazz, overrideClass);
130                 String propertyName = auditOverride.name();
131                 if (!StringTools.isEmpty(propertyName)) {
132                     // Override @Audited annotation on property level.
133                     XProperty property = getProperty(overrideClass, propertyName);
134                     if (auditOverride.isAudited()) {
135                         if (!overriddenNotAuditedProperties.contains(property)) {
136                             // If the property has not been marked as not audited by the subclass.
137                             overriddenAuditedProperties.add(property);
138                         }
139                     } else {
140                         if (!overriddenAuditedProperties.contains(property)) {
141                             // If the property has not been marked as audited by the subclass.
142                             overriddenNotAuditedProperties.add(property);
143                         }
144                     }
145                 } else {
146                     // Override @Audited annotation on class level.
147                     if (auditOverride.isAudited()) {
148                         if (!overriddenNotAuditedClasses.contains(overrideClass)) {
149                             // If the class has not been marked as not audited by the subclass.
150                             overriddenAuditedClasses.add(overrideClass);
151                         }
152                     } else {
153                         if (!overriddenAuditedClasses.contains(overrideClass)) {
154                             // If the class has not been marked as audited by the subclass.
155                             overriddenNotAuditedClasses.add(overrideClass);
156                         }
157                     }
158                 }
159             }
160         }
161         XClass superclass = clazz.getSuperclass();
162         if (!clazz.isInterface() && !Object.class.getName().equals(superclass.getName())) {
163             readAuditOverrides(superclass);
164         }
165     }

167     /**
168      * @param clazz Source class.
169      * @return List of @AuditOverride annotations applied at class level.
170      */
171     private List<AuditOverride> computeAuditOverrides(XClass clazz) {
172         AuditOverrides auditOverrides = clazz.getAnnotation(AuditOverrides.class);
173         AuditOverride auditOverride = clazz.getAnnotation(AuditOverride.class);
174         if (auditOverrides == null && auditOverride != null) {
175             return Arrays.asList(auditOverride);
176         } else if (auditOverrides != null && auditOverride == null) {
177             return Arrays.asList(auditOverrides.value());
178         } else if (auditOverrides != null && auditOverride != null) {
179             throw new MappingException("@AuditOverrides annotation should encapsulate all @AuditOverride declarations. " +
180                                        "Please revise Envers annotations applied to class " + clazz.getName() + ".");
181         }
182         return Collections.EMPTY_LIST;
183     }

185     /**
186      * Checks whether one class is assignable from another. If not {@link MappingException} is thrown.
187      * @param child Subclass.
188      * @param parent Superclass.
189      */
190     private void checkSuperclass(XClass child, XClass parent) {
191         if (!parent.isAssignableFrom(child)) {
192             throw new MappingException("Class " + parent.getName() + " is not assignable from " + child.getName() + ". " +
193                                        "Please revise Envers annotations applied to " + child.getName() + " type.");
194         }
195     }

197     /**
198      * Checks whether class contains property with a given name. If not {@link MappingException} is thrown.
199      * @param clazz Class.
200      * @param propertyName Property name.
201      * @return Property object.
202      */
203     private XProperty getProperty(XClass clazz, String propertyName) {
204         XProperty property = Tools.getProperty(clazz, propertyName);
205         if (property == null) {
206             throw new MappingException("Property '" + propertyName + "' not found in class " + clazz.getName() + ". " +
207                                        "Please revise Envers annotations applied to class " + persistentPropertiesSource.getXClass() + ".");
208         }
209         return property;
210     }

212 	private void readPersistentPropertiesAccess() {
213 		Iterator<Property> propertyIter = persistentPropertiesSource.getPropertyIterator();
214 		while (propertyIter.hasNext()) {
215 			Property property = (Property) propertyIter.next();
216 			addPersistentProperty(property);
217 			if ("embedded".equals(property.getPropertyAccessorName()) && property.getName().equals(property.getNodeName())) {
218 				// If property name equals node name and embedded accessor type is used, processing component
219 				// has been defined with <properties> tag. See HHH-6636 JIRA issue.
220 				createPropertiesGroupMapping(property);
221 			}
222 		}
223 	}
224 
225     private void addPersistentProperty(Property property) {
226         if ("field".equals(property.getPropertyAccessorName())) {
227             fieldAccessedPersistentProperties.add(property.getName());
228         } else {
229             propertyAccessedPersistentProperties.add(property.getName());
230         }
231     }
232 
233     private void createPropertiesGroupMapping(Property property) {
234         Component component = (Component) property.getValue();
235         Iterator<Property> componentProperties = component.getPropertyIterator();
236         while (componentProperties.hasNext()) {
237             Property componentProperty = componentProperties.next();
238             propertiesGroupMapping.put(componentProperty.getName(), component.getNodeName());
239         }
240     }

242     /**
243      * @param clazz Class which properties are currently being added.
244      * @return {@link Audited} annotation of specified class. If processed type hasn't been explicitly marked, method
245      *         checks whether given class exists in {@link AuditedPropertiesReader#overriddenAuditedClasses} collection.
246      *         In case of success, {@link Audited} configuration of currently mapped entity is returned, otherwise
247      *         {@code null}. If processed type exists in {@link AuditedPropertiesReader#overriddenNotAuditedClasses}
248      *         collection, the result is also {@code null}.
249      */
250     private Audited computeAuditConfiguration(XClass clazz) {
251         Audited allClassAudited = clazz.getAnnotation(Audited.class);
252         // If processed class is not explicitly marked with @Audited annotation, check whether auditing is
253         // forced by any of its child entities configuration (@AuditedOverride.forClass).
254         if (allClassAudited == null && overriddenAuditedClasses.contains(clazz)) {
255             // Declared audited parent copies @Audited.modStore and @Audited.targetAuditMode configuration from
256             // currently mapped entity.
257             allClassAudited = persistentPropertiesSource.getXClass().getAnnotation(Audited.class);
258             if (allClassAudited == null) {
259                 // If parent class declares @Audited on the field/property level.
260                 allClassAudited = DEFAULT_AUDITED;
261             }
262         } else if (allClassAudited != null && overriddenNotAuditedClasses.contains(clazz)) {
263             return null;
264         }
265         return allClassAudited;
266     }

268     /**
269      * Recursively adds all audited properties of entity class and its superclasses.
270      * @param clazz Currently processed class.
271      */
272 	private void addPropertiesFromClass(XClass clazz)  {
273 		Audited allClassAudited = computeAuditConfiguration(clazz);
274 
275 		//look in the class
276 		addFromProperties(clazz.getDeclaredProperties("field"), "field", fieldAccessedPersistentProperties, allClassAudited);
277 		addFromProperties(clazz.getDeclaredProperties("property"), "property", propertyAccessedPersistentProperties, allClassAudited);
278 		
279 		if(allClassAudited != null || !auditedPropertiesHolder.isEmpty()) {
280 			XClass superclazz = clazz.getSuperclass();
281 			if (!clazz.isInterface() && !"java.lang.Object".equals(superclazz.getName())) {
282 				addPropertiesFromClass(superclazz);
283 			}
284 		}
285 	}

287 	private void addFromProperties(Iterable<XProperty> properties, String accessType, Set<String> persistentProperties, Audited allClassAudited) {
288 		for (XProperty property : properties) {
289 			// If this is not a persistent property, with the same access type as currently checked,
290 			// it's not audited as well. 
291 			// If the property was already defined by the subclass, is ignored by superclasses
292 			if ((persistentProperties.contains(property.getName()) && (!auditedPropertiesHolder
293 					.contains(property.getName())))) {
294 				Value propertyValue = persistentPropertiesSource.getProperty(property.getName()).getValue();
295 				if (propertyValue instanceof Component) {
296 					this.addFromComponentProperty(property, accessType, (Component)propertyValue, allClassAudited);
297 				} else {
298 					this.addFromNotComponentProperty(property, accessType, allClassAudited);
299 				}
300 			} else if (propertiesGroupMapping.containsKey(property.getName())) {
301 				// Retrieve embedded component name based on class field.
302 				final String embeddedName = propertiesGroupMapping.get(property.getName());
303 				if (!auditedPropertiesHolder.contains(embeddedName)) {
304 					// Manage properties mapped within <properties> tag.
305 					Value propertyValue = persistentPropertiesSource.getProperty(embeddedName).getValue();
306 					this.addFromPropertiesGroup(embeddedName, property, accessType, (Component)propertyValue, allClassAudited);
307 				}
308 			}
309 		}
310 	}

312 	private void addFromPropertiesGroup(String embeddedName, XProperty property, String accessType, Component propertyValue,
313 										Audited allClassAudited) {
314 		ComponentAuditingData componentData = new ComponentAuditingData();
315 		boolean isAudited = fillPropertyData(property, componentData, accessType, allClassAudited);
316 		if (isAudited) {
317 			// EntityPersister.getPropertyNames() returns name of embedded component instead of class field.
318 			componentData.setName(embeddedName);
319 			// Marking component properties as placed directly in class (not inside another component).
320 			componentData.setBeanName(null);
321 
322 			PersistentPropertiesSource componentPropertiesSource = new ComponentPropertiesSource( reflectionManager, propertyValue );
323 			AuditedPropertiesReader audPropReader = new AuditedPropertiesReader(
324 					ModificationStore.FULL, componentPropertiesSource, componentData, globalCfg, reflectionManager,
325 					propertyNamePrefix + MappingTools.createComponentPrefix(embeddedName)
326 			);
327 			audPropReader.read();
328 
329 			auditedPropertiesHolder.addPropertyAuditingData(embeddedName, componentData);
330 		}
331 	}

333 	private void addFromComponentProperty(XProperty property, String accessType, Component propertyValue, Audited allClassAudited) {
334 		ComponentAuditingData componentData = new ComponentAuditingData();
335 		boolean isAudited = fillPropertyData( property, componentData, accessType, allClassAudited );
336 
337 		if ( propertyValue.isDynamic() ) {
338 			if ( isAudited ) {
339 				throw new MappingException(
340 						"Audited dynamic-component properties are not supported. Consider applying @NotAudited annotation to "
341 								+ propertyValue.getOwner().getEntityName() + "#" + property + "."
342 				);
343 			}
344 			return;
345 		}
346 
347 		PersistentPropertiesSource componentPropertiesSource = new ComponentPropertiesSource(
348 				reflectionManager, propertyValue
349 		);
350 
351 		ComponentAuditedPropertiesReader audPropReader = new ComponentAuditedPropertiesReader(
352 				ModificationStore.FULL, componentPropertiesSource, componentData, globalCfg, reflectionManager,
353 				propertyNamePrefix + MappingTools.createComponentPrefix( property.getName() )
354 		);
355 		audPropReader.read();
356 
357 		if ( isAudited ) {
358 			// Now we know that the property is audited
359 			auditedPropertiesHolder.addPropertyAuditingData( property.getName(), componentData );
360 		}
361 	}

363 	private void addFromNotComponentProperty(XProperty property, String accessType, Audited allClassAudited){
364 		PropertyAuditingData propertyData = new PropertyAuditingData();
365 		boolean isAudited = fillPropertyData(property, propertyData, accessType, allClassAudited);
366 
367 		if (isAudited) {
368 			// Now we know that the property is audited
369 			auditedPropertiesHolder.addPropertyAuditingData(property.getName(), propertyData);
370 		}
371 	}

374 	/**
375 	 * Checks if a property is audited and if yes, fills all of its data.
376 	 * @param property Property to check.
377 	 * @param propertyData Property data, on which to set this property's modification store.
378 	 * @param accessType Access type for the property.
379 	 * @return False if this property is not audited.
380 	 */
381 	private boolean fillPropertyData(XProperty property, PropertyAuditingData propertyData,
382 									 String accessType, Audited allClassAudited) {
383 
384 		// check if a property is declared as not audited to exclude it
385 		// useful if a class is audited but some properties should be excluded
386 		NotAudited unVer = property.getAnnotation(NotAudited.class);
387 		if ((unVer != null && !overriddenAuditedProperties.contains(property)) || overriddenNotAuditedProperties.contains(property)) {
388 			return false;
389 		} else {
390 			// if the optimistic locking field has to be unversioned and the current property
391 			// is the optimistic locking field, don't audit it
392 			if (globalCfg.isDoNotAuditOptimisticLockingField()) {
393 				Version jpaVer = property.getAnnotation(Version.class);
394 				if (jpaVer != null) {
395 					return false;
396 				}
397 			}
398 		}
399 
400 		
401 		if(!this.checkAudited(property, propertyData, allClassAudited)){
402 			return false;
403 		}
404 
405 		String propertyName = propertyNamePrefix + property.getName();
406 		propertyData.setName(propertyName);
407 		propertyData.setModifiedFlagName(
408                 MetadataTools.getModifiedFlagPropertyName(
409                         propertyName,
410                         globalCfg.getModifiedFlagSuffix()));
411 		propertyData.setBeanName(property.getName());
412 		propertyData.setAccessType(accessType);
413 
414 		addPropertyJoinTables(property, propertyData);
415 		addPropertyAuditingOverrides(property, propertyData);
416 		if (!processPropertyAuditingOverrides(property, propertyData)) {
417 			return false; // not audited due to AuditOverride annotation
418 		}
419         addPropertyMapKey(property, propertyData);
420         setPropertyAuditMappedBy(property, propertyData);
421         setPropertyRelationMappedBy(property, propertyData);
422 
423 		return true;
424 	}

427 	protected boolean checkAudited(XProperty property,
428 			PropertyAuditingData propertyData, Audited allClassAudited) {
429 		// Checking if this property is explicitly audited or if all properties are.
430 		Audited aud = (property.isAnnotationPresent(Audited.class)) ? (property.getAnnotation(Audited.class)) : allClassAudited;
431 		if (aud == null && overriddenAuditedProperties.contains(property) && !overriddenNotAuditedProperties.contains(property)) {
432 			// Assigning @Audited defaults. If anyone needs to customize those values in the future,
433 			// appropriate fields shall be added to @AuditOverride annotation.
434 			aud = DEFAULT_AUDITED;
435 		}
436 		if (aud != null) {
437 			propertyData.setStore(aud.modStore());
438 			propertyData.setRelationTargetAuditMode(aud.targetAuditMode());
439 			propertyData.setUsingModifiedFlag(checkUsingModifiedFlag(aud));
440 			return true;
441 		} else {
442 			return false;
443 		}
444 	}
445 
446 	protected boolean checkUsingModifiedFlag(Audited aud) {
447 		return globalCfg.hasSettingForUsingModifiedFlag() ?
448 				globalCfg.isGlobalWithModifiedFlag() : aud.withModifiedFlag();
449 	}

452         OneToMany oneToMany = property.getAnnotation(OneToMany.class);
453         if (oneToMany != null && !"".equals(oneToMany.mappedBy())) {
454             propertyData.setRelationMappedBy(oneToMany.mappedBy());
455         }

458 	private void setPropertyAuditMappedBy(XProperty property, PropertyAuditingData propertyData) {
459         AuditMappedBy auditMappedBy = property.getAnnotation(AuditMappedBy.class);
460         if (auditMappedBy != null) {
461 		    propertyData.setAuditMappedBy(auditMappedBy.mappedBy());
462             if (!"".equals(auditMappedBy.positionMappedBy())) {
463                 propertyData.setPositionMappedBy(auditMappedBy.positionMappedBy());
464             }
465         }

475 	private void addPropertyJoinTables(XProperty property, PropertyAuditingData propertyData) {
476 		// first set the join table based on the AuditJoinTable annotation
477 		AuditJoinTable joinTable = property.getAnnotation(AuditJoinTable.class);
478 		if (joinTable != null) {

485 	/***
486 	 * Add the {@link org.hibernate.envers.AuditOverride} annotations.
487 	 *
488 	 * @param property the property being processed
489 	 * @param propertyData the Envers auditing data for this property
490 	 */
491 	private void addPropertyAuditingOverrides(XProperty property, PropertyAuditingData propertyData) {
492 		AuditOverride annotationOverride = property.getAnnotation(AuditOverride.class);
493 		if (annotationOverride != null) {
494 			propertyData.addAuditingOverride(annotationOverride);
495 		}
496 		AuditOverrides annotationOverrides = property.getAnnotation(AuditOverrides.class);
497 		if (annotationOverrides != null) {
498 			propertyData.addAuditingOverrides(annotationOverrides);
499 		}
500 	}

502 	/**
503 	 * Process the {@link org.hibernate.envers.AuditOverride} annotations for this property.
504 	 *
505 	 * @param property
506 	 *            the property for which the {@link org.hibernate.envers.AuditOverride}
507 	 *            annotations are being processed
508 	 * @param propertyData
509 	 *            the Envers auditing data for this property
510 	 * @return {@code false} if isAudited() of the override annotation was set to
511 	 */
512 	private boolean processPropertyAuditingOverrides(XProperty property, PropertyAuditingData propertyData) {
513 		// if this property is part of a component, process all override annotations
514 		if (this.auditedPropertiesHolder instanceof ComponentAuditingData) {
515 			List<AuditOverride> overrides = ((ComponentAuditingData) this.auditedPropertiesHolder).getAuditingOverrides();
516 			for (AuditOverride override : overrides) {
517 				if (property.getName().equals(override.name())) {
518 					// the override applies to this property
519 					if (!override.isAudited()) {
520 						return false;
521 					} else {
522 						if (override.auditJoinTable() != null) {
523 							propertyData.setJoinTable(override.auditJoinTable());
524 						}
525 					}
526 				}
527 			}
528 			
529 		}
530 		return true;
531 	}

537         public boolean withModifiedFlag() { return false; }

549     public static class ComponentPropertiesSource implements PersistentPropertiesSource {
550 		private final XClass xclass;
551 		private final Component component;
552 
553 		public ComponentPropertiesSource(ReflectionManager reflectionManager, Component component) {
554 			try {
555 				this.xclass = reflectionManager.classForName(component.getComponentClassName(), this.getClass());
556 			} catch (ClassNotFoundException e) {
557 				throw new MappingException(e);
558 			}
559 
560 			this.component = component;
561 		}
562 
563 		@SuppressWarnings({"unchecked"})
564 		public Iterator<Property> getPropertyIterator() { return component.getPropertyIterator(); }
565 		public Property getProperty(String propertyName) { return component.getProperty(propertyName); }
566 		public XClass getXClass() { return xclass; }
567 	}
