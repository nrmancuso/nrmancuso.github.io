62  /**
63   * A <code>CertStore</code> that retrieves <code>Certificates</code> or
64   * <code>CRL</code>s from a URI, for example, as specified in an X.509
65   * AuthorityInformationAccess or CRLDistributionPoint extension.
66   * <p>
67   * For CRLs, this implementation retrieves a single DER encoded CRL per URI.
68   * For Certificates, this implementation retrieves a single DER encoded CRL or
69   * a collection of Certificates encoded as a PKCS#7 "certs-only" CMS message.
70   * <p>
71   * This <code>CertStore</code> also implements Certificate/CRL caching.
72   * Currently, the cache is shared between all applications in the VM and uses a
73   * hardcoded policy. The cache has a maximum size of 185 entries, which are held
74   * by SoftReferences. A request will be satisfied from the cache if we last
75   * checked for an update within CHECK_INTERVAL (last 30 seconds). Otherwise,
76   * we open an URLConnection to download the Certificate(s)/CRL using an
77   * If-Modified-Since request (HTTP) if possible. Note that both positive and
78   * negative responses are cached, i.e. if we are unable to open the connection
79   * or the Certificate(s)/CRL cannot be parsed, we remember this result and
80   * additional calls during the CHECK_INTERVAL period do not try to open another
81   * connection.
82   * <p>
83   * The URICertStore is not currently a standard CertStore type. We should
84   * consider adding a standard "URI" CertStore type.
85   *
86   * @author Andreas Sterbenz
87   * @author Sean Mullan
88   * @since 7.0
89   */
90  class URICertStore extends CertStoreSpi {
91  
92      private static final Debug debug = Debug.getInstance("certpath");
93  
94      // interval between checks for update of cached Certificates/CRLs
95      // (30 seconds)
96      private final static int CHECK_INTERVAL = 30 * 1000;
97  
98      // size of the cache (see Cache class for sizing recommendations)
99      private final static int CACHE_SIZE = 185;
100 
101     // X.509 certificate factory instance
102     private final CertificateFactory factory;
103 
104     // cached Collection of X509Certificates (may be empty, never null)
105     private Collection<X509Certificate> certs =
106         Collections.<X509Certificate>emptySet();
107 
108     // cached X509CRL (may be null)
109     private X509CRL crl;
110 
111     // time we last checked for an update
112     private long lastChecked;
113 
114     // time server returned as last modified time stamp
115     // or 0 if not available
116     private long lastModified;
117 
118     // the URI of this CertStore
119     private URI uri;
120 
121     // true if URI is ldap
122     private boolean ldap = false;
123     private CertStore ldapCertStore;
124     private String ldapPath;

126     /**
127      * Holder class to lazily load LDAPCertStoreHelper if present.
128      */
129     private static class LDAP {
130         private static final String CERT_STORE_HELPER =
131             "sun.security.provider.certpath.ldap.LDAPCertStoreHelper";
132         private static final CertStoreHelper helper =
133             AccessController.doPrivileged(
134                 new PrivilegedAction<CertStoreHelper>() {
135                     public CertStoreHelper run() {
136                         try {
137                             Class<?> c = Class.forName(CERT_STORE_HELPER, true, null);
138                             return (CertStoreHelper)c.newInstance();
139                         } catch (ClassNotFoundException cnf) {
140                             return null;
141                         } catch (InstantiationException e) {
142                             throw new AssertionError(e);
143                         } catch (IllegalAccessException e) {
144                             throw new AssertionError(e);
145                         }
146                     }});
147         static CertStoreHelper helper() {
148             return helper;
149         }
150     }

152     /**
153      * Creates a URICertStore.
154      *
155      * @param parameters specifying the URI
156      */
157     URICertStore(CertStoreParameters params)
158         throws InvalidAlgorithmParameterException, NoSuchAlgorithmException {
159         super(params);
160         if (!(params instanceof URICertStoreParameters)) {
161             throw new InvalidAlgorithmParameterException
162                 ("params must be instanceof URICertStoreParameters");
163         }
164         this.uri = ((URICertStoreParameters) params).uri;
165         // if ldap URI, use an LDAPCertStore to fetch certs and CRLs
166         if (uri.getScheme().toLowerCase(Locale.ENGLISH).equals("ldap")) {
167             if (LDAP.helper() == null)
168                 throw new NoSuchAlgorithmException("LDAP not present");
169             ldap = true;
170             ldapCertStore = LDAP.helper().getCertStore(uri);
171             ldapPath = uri.getPath();
172             // strip off leading '/'
173             if (ldapPath.charAt(0) == '/') {
174                 ldapPath = ldapPath.substring(1);
175             }
176         }
177         try {
178             factory = CertificateFactory.getInstance("X.509");
179         } catch (CertificateException e) {
180             throw new RuntimeException();
181         }
182     }

184     /**
185      * Returns a URI CertStore. This method consults a cache of
186      * CertStores (shared per JVM) using the URI as a key.
187      */
188     private static final Cache certStoreCache =
189         Cache.newSoftMemoryCache(CACHE_SIZE);
190     static synchronized CertStore getInstance(URICertStoreParameters params)
191         throws NoSuchAlgorithmException, InvalidAlgorithmParameterException {
192         if (debug != null) {
193             debug.println("CertStore URI:" + params.uri);
194         }
195         CertStore ucs = (CertStore) certStoreCache.get(params);
196         if (ucs == null) {
197             ucs = new UCS(new URICertStore(params), null, "URI", params);
198             certStoreCache.put(params, ucs);
199         } else {
200             if (debug != null) {
201                 debug.println("URICertStore.getInstance: cache hit");
202             }
203         }
204         return ucs;
205     }
206 
207     /**
208      * Creates a CertStore from information included in the AccessDescription
209      * object of a certificate's Authority Information Access Extension.
210      */
211     static CertStore getInstance(AccessDescription ad) {
212         if (!ad.getAccessMethod().equals(AccessDescription.Ad_CAISSUERS_Id)) {
213             return null;
214         }
215         GeneralNameInterface gn = ad.getAccessLocation().getName();
216         if (!(gn instanceof URIName)) {
217             return null;
218         }
219         URI uri = ((URIName) gn).getURI();
220         try {
221             return URICertStore.getInstance
222                 (new URICertStore.URICertStoreParameters(uri));
223         } catch (Exception ex) {
224             if (debug != null) {
225                 debug.println("exception creating CertStore: " + ex);
226                 ex.printStackTrace();
227             }
228             return null;
229         }
230     }

232     /**
233      * Returns a <code>Collection</code> of <code>X509Certificate</code>s that
234      * match the specified selector. If no <code>X509Certificate</code>s
235      * match the selector, an empty <code>Collection</code> will be returned.
236      *
237      * @param selector a <code>CertSelector</code> used to select which
238      *  <code>X509Certificate</code>s should be returned. Specify
239      *  <code>null</code> to return all <code>X509Certificate</code>s.
240      * @return a <code>Collection</code> of <code>X509Certificate</code>s that
241      *         match the specified selector
242      * @throws CertStoreException if an exception occurs
243      */
244     public synchronized Collection<X509Certificate> engineGetCertificates
245         (CertSelector selector) throws CertStoreException {
246 
247         // if ldap URI we wrap the CertSelector in an LDAPCertSelector to
248         // avoid LDAP DN matching issues (see LDAPCertSelector for more info)
249         if (ldap) {
250             X509CertSelector xsel = (X509CertSelector) selector;
251             try {
252                 xsel = LDAP.helper().wrap(xsel, xsel.getSubject(), ldapPath);
253             } catch (IOException ioe) {
254                 throw new CertStoreException(ioe);
255             }
256             // Fetch the certificates via LDAP. LDAPCertStore has its own
257             // caching mechanism, see the class description for more info.
258             return (Collection<X509Certificate>)
259                 ldapCertStore.getCertificates(xsel);
260         }
261 
262         // Return the Certificates for this entry. It returns the cached value
263         // if it is still current and fetches the Certificates otherwise.
264         // For the caching details, see the top of this class.
265         long time = System.currentTimeMillis();
266         if (time - lastChecked < CHECK_INTERVAL) {
267             if (debug != null) {
268                 debug.println("Returning certificates from cache");
269             }
270             return getMatchingCerts(certs, selector);
271         }
272         lastChecked = time;
273         InputStream in = null;
274         try {
275             URLConnection connection = uri.toURL().openConnection();
276             if (lastModified != 0) {
277                 connection.setIfModifiedSince(lastModified);
278             }
279             in = connection.getInputStream();
280             long oldLastModified = lastModified;
281             lastModified = connection.getLastModified();
282             if (oldLastModified != 0) {
283                 if (oldLastModified == lastModified) {
284                     if (debug != null) {
285                         debug.println("Not modified, using cached copy");
286                     }
287                     return getMatchingCerts(certs, selector);
288                 } else if (connection instanceof HttpURLConnection) {
289                     // some proxy servers omit last modified
290                     HttpURLConnection hconn = (HttpURLConnection) connection;
291                     if (hconn.getResponseCode()
292                                 == HttpURLConnection.HTTP_NOT_MODIFIED) {
293                         if (debug != null) {
294                             debug.println("Not modified, using cached copy");
295                         }
296                         return getMatchingCerts(certs, selector);
297                     }
298                 }
299             }
300             if (debug != null) {
301                 debug.println("Downloading new certificates...");
302             }
303             certs = (Collection<X509Certificate>)
304                 factory.generateCertificates(in);
305             return getMatchingCerts(certs, selector);
306         } catch (IOException e) {
307             if (debug != null) {
308                 debug.println("Exception fetching certificates:");
309                 e.printStackTrace();
310             }
311         } catch (CertificateException e) {
312             if (debug != null) {
313                 debug.println("Exception fetching certificates:");
314                 e.printStackTrace();
315             }
316         } finally {
317             if (in != null) {
318                 try {
319                     in.close();
320                 } catch (IOException e) {
321                     // ignore
322                 }
323             }
324         }
325         // exception, forget previous values
326         lastModified = 0;
327         certs = Collections.<X509Certificate>emptySet();
328         return certs;
329     }

331     /**
332      * Iterates over the specified Collection of X509Certificates and
333      * returns only those that match the criteria specified in the
334      * CertSelector.
335      */
336     private static Collection<X509Certificate> getMatchingCerts
337         (Collection<X509Certificate> certs, CertSelector selector) {
338         // if selector not specified, all certs match
339         if (selector == null) {
340             return certs;
341         }
342         List<X509Certificate> matchedCerts =
343             new ArrayList<X509Certificate>(certs.size());
344         for (X509Certificate cert : certs) {
345             if (selector.match(cert)) {
346                 matchedCerts.add(cert);
347             }
348         }
349         return matchedCerts;
350     }

352     /**
353      * Returns a <code>Collection</code> of <code>X509CRL</code>s that
354      * match the specified selector. If no <code>X509CRL</code>s
355      * match the selector, an empty <code>Collection</code> will be returned.
356      *
357      * @param selector A <code>CRLSelector</code> used to select which
358      *  <code>X509CRL</code>s should be returned. Specify <code>null</code>
359      *  to return all <code>X509CRL</code>s.
360      * @return A <code>Collection</code> of <code>X509CRL</code>s that
361      *         match the specified selector
362      * @throws CertStoreException if an exception occurs
363      */
364     public synchronized Collection<X509CRL> engineGetCRLs(CRLSelector selector)
365         throws CertStoreException {
366 
367         // if ldap URI we wrap the CRLSelector in an LDAPCRLSelector to
368         // avoid LDAP DN matching issues (see LDAPCRLSelector for more info)
369         if (ldap) {
370             X509CRLSelector xsel = (X509CRLSelector) selector;
371             try {
372                 xsel = LDAP.helper().wrap(xsel, null, ldapPath);
373             } catch (IOException ioe) {
374                 throw new CertStoreException(ioe);
375             }
376             // Fetch the CRLs via LDAP. LDAPCertStore has its own
377             // caching mechanism, see the class description for more info.
378             return (Collection<X509CRL>) ldapCertStore.getCRLs(xsel);
379         }
380 
381         // Return the CRLs for this entry. It returns the cached value
382         // if it is still current and fetches the CRLs otherwise.
383         // For the caching details, see the top of this class.
384         long time = System.currentTimeMillis();
385         if (time - lastChecked < CHECK_INTERVAL) {
386             if (debug != null) {
387                 debug.println("Returning CRL from cache");
388             }
389             return getMatchingCRLs(crl, selector);
390         }
391         lastChecked = time;
392         InputStream in = null;
393         try {
394             URLConnection connection = uri.toURL().openConnection();
395             if (lastModified != 0) {
396                 connection.setIfModifiedSince(lastModified);
397             }
398             in = connection.getInputStream();
399             long oldLastModified = lastModified;
400             lastModified = connection.getLastModified();
401             if (oldLastModified != 0) {
402                 if (oldLastModified == lastModified) {
403                     if (debug != null) {
404                         debug.println("Not modified, using cached copy");
405                     }
406                     return getMatchingCRLs(crl, selector);
407                 } else if (connection instanceof HttpURLConnection) {
408                     // some proxy servers omit last modified
409                     HttpURLConnection hconn = (HttpURLConnection) connection;
410                     if (hconn.getResponseCode()
411                                 == HttpURLConnection.HTTP_NOT_MODIFIED) {
412                         if (debug != null) {
413                             debug.println("Not modified, using cached copy");
414                         }
415                         return getMatchingCRLs(crl, selector);
416                     }
417                 }
418             }
419             if (debug != null) {
420                 debug.println("Downloading new CRL...");
421             }
422             crl = (X509CRL) factory.generateCRL(in);
423             return getMatchingCRLs(crl, selector);
424         } catch (IOException e) {
425             if (debug != null) {
426                 debug.println("Exception fetching CRL:");
427                 e.printStackTrace();
428             }
429         } catch (CRLException e) {
430             if (debug != null) {
431                 debug.println("Exception fetching CRL:");
432                 e.printStackTrace();
433             }
434         } finally {
435             if (in != null) {
436                 try {
437                     in.close();
438                 } catch (IOException e) {
439                     // ignore
440                 }
441             }
442         }
443         // exception, forget previous values
444         lastModified = 0;
445         crl = null;
446         return Collections.<X509CRL>emptyList();
447     }
448 
449     /**
450      * Checks if the specified X509CRL matches the criteria specified in the
451      * CRLSelector.
452      */
453     private static Collection<X509CRL> getMatchingCRLs
454         (X509CRL crl, CRLSelector selector) {
455         if (selector == null || (crl != null && selector.match(crl))) {
456             return Collections.<X509CRL>singletonList(crl);
457         } else {
458             return Collections.<X509CRL>emptyList();
459         }
460     }

462     /**
463      * CertStoreParameters for the URICertStore.
464      */
465     static class URICertStoreParameters implements CertStoreParameters {
466         private final URI uri;
467         private volatile int hashCode = 0;
468         URICertStoreParameters(URI uri) {
469             this.uri = uri;
470         }
471         public boolean equals(Object obj) {
472             if (!(obj instanceof URICertStoreParameters)) {
473                 return false;
474             }
475             URICertStoreParameters params = (URICertStoreParameters) obj;
476             return uri.equals(params.uri);
477         }
478         public int hashCode() {
479             if (hashCode == 0) {
480                 int result = 17;
481                 result = 37*result + uri.hashCode();
482                 hashCode = result;
483             }
484             return hashCode;
485         }
486         public Object clone() {
487             try {
488                 return super.clone();
489             } catch (CloneNotSupportedException e) {
490                 /* Cannot happen */
491                 throw new InternalError(e.toString());
492             }
493         }
494     }
