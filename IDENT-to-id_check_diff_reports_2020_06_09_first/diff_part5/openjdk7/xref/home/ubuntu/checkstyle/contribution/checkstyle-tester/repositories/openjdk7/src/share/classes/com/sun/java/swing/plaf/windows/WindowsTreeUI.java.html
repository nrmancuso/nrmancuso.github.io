64      /**
65        * Ensures that the rows identified by beginRow through endRow are
66        * visible.
67        */
68      protected void ensureRowsAreVisible(int beginRow, int endRow) {
69          if(tree != null && beginRow >= 0 && endRow < getRowCount(tree)) {
70              Rectangle visRect = tree.getVisibleRect();
71              if(beginRow == endRow) {
72                  Rectangle     scrollBounds = getPathBounds(tree, getPathForRow
73                                                             (tree, beginRow));
74  
75                  if(scrollBounds != null) {
76                      scrollBounds.x = visRect.x;
77                      scrollBounds.width = visRect.width;
78                      tree.scrollRectToVisible(scrollBounds);
79                  }
80              }
81              else {
82                  Rectangle   beginRect = getPathBounds(tree, getPathForRow
83                                                        (tree, beginRow));
84                  Rectangle   testRect = beginRect;
85                  int         beginY = beginRect.y;
86                  int         maxY = beginY + visRect.height;
87  
88                  for(int counter = beginRow + 1; counter <= endRow; counter++) {
89                      testRect = getPathBounds(tree,
90                                               getPathForRow(tree, counter));
91                      if((testRect.y + testRect.height) > maxY)
92                          counter = endRow;
93                  }
94                  tree.scrollRectToVisible(new Rectangle(visRect.x, beginY, 1,
95                                                    testRect.y + testRect.height-
96                                                    beginY));
97              }
98          }
99      }

112     /**
113      * The minus sign button icon
114      * <p>
115      * <strong>Warning:</strong>
116      * Serialized objects of this class will not be compatible with
117      * future Swing releases.  The current serialization support is appropriate
118      * for short term storage or RMI between applications running the same
119      * version of Swing.  A future release of Swing will provide support for
120      * long term persistence.
121      */
122     public static class ExpandedIcon implements Icon, Serializable {
123 
124         static public Icon createExpandedIcon() {
125             return new ExpandedIcon();
126         }
127 
128         Skin getSkin(Component c) {
129             XPStyle xp = XPStyle.getXP();
130             return (xp != null) ? xp.getSkin(c, Part.TVP_GLYPH) : null;
131         }
132 
133         public void paintIcon(Component c, Graphics g, int x, int y) {
134             Skin skin = getSkin(c);
135             if (skin != null) {
136                 skin.paintSkin(g, x, y, State.OPENED);
137                 return;
138             }
139 
140             Color     backgroundColor = c.getBackground();
141 
142             if(backgroundColor != null)
143                 g.setColor(backgroundColor);
144             else
145                 g.setColor(Color.white);
146             g.fillRect(x, y, SIZE-1, SIZE-1);
147             g.setColor(Color.gray);
148             g.drawRect(x, y, SIZE-1, SIZE-1);
149             g.setColor(Color.black);
150             g.drawLine(x + 2, y + HALF_SIZE, x + (SIZE - 3), y + HALF_SIZE);
151         }
152 
153         public int getIconWidth() {
154             Skin skin = getSkin(null);
155             return (skin != null) ? skin.getWidth() : SIZE;
156         }
157 
158         public int getIconHeight() {
159             Skin skin = getSkin(null);
160             return (skin != null) ? skin.getHeight() : SIZE;
161         }
162     }

164     /**
165      * The plus sign button icon
166      * <p>
167      * <strong>Warning:</strong>
168      * Serialized objects of this class will not be compatible with
169      * future Swing releases.  The current serialization support is appropriate
170      * for short term storage or RMI between applications running the same
171      * version of Swing.  A future release of Swing will provide support for
172      * long term persistence.
173      */
174     public static class CollapsedIcon extends ExpandedIcon {
175         static public Icon createCollapsedIcon() {
176             return new CollapsedIcon();
177         }
178 
179         public void paintIcon(Component c, Graphics g, int x, int y) {
180             Skin skin = getSkin(c);
181             if (skin != null) {
182                 skin.paintSkin(g, x, y, State.CLOSED);
183             } else {
184             super.paintIcon(c, g, x, y);
185             g.drawLine(x + HALF_SIZE, y + 2, x + HALF_SIZE, y + (SIZE - 3));
186             }
187         }
188     }

190     public class WindowsTreeCellRenderer extends DefaultTreeCellRenderer {
191 
192         /**
193          * Configures the renderer based on the passed in components.
194          * The value is set from messaging the tree with
195          * <code>convertValueToText</code>, which ultimately invokes
196          * <code>toString</code> on <code>value</code>.
197          * The foreground color is set based on the selection and the icon
198          * is set based on on leaf and expanded.
199          */
200         public Component getTreeCellRendererComponent(JTree tree, Object value,
201                                                       boolean sel,
202                                                       boolean expanded,
203                                                       boolean leaf, int row,
204                                                       boolean hasFocus) {
205             super.getTreeCellRendererComponent(tree, value, sel,
206                                                expanded, leaf, row,
207                                                hasFocus);
208             // Windows displays the open icon when the tree item selected.
209             if (!tree.isEnabled()) {
210                 setEnabled(false);
211                 if (leaf) {
212                     setDisabledIcon(getLeafIcon());
213                 } else if (sel) {
214                     setDisabledIcon(getOpenIcon());
215                 } else {
216                     setDisabledIcon(getClosedIcon());
217                 }
218             }
219             else {
220                 setEnabled(true);
221                 if (leaf) {
222                     setIcon(getLeafIcon());
223                 } else if (sel) {
224                     setIcon(getOpenIcon());
225                 } else {
226                     setIcon(getClosedIcon());
227                 }
228             }
229             return this;
230         }

232     }
