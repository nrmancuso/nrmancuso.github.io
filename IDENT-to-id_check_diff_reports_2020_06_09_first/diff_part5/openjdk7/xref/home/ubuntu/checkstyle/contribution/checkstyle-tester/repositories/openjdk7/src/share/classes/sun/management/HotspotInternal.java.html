26  package sun.management;
27  
28  import javax.management.MBeanRegistration;
29  import javax.management.MBeanServer;
30  import javax.management.ObjectName;
31  
32  /**
33   * Implementation class of HotspotInternalMBean interface.
34   *
35   * <p> This is designed for internal customer use to create
36   * this MBean dynamically from an agent which will then register
37   * all internal MBeans to the platform MBeanServer.
38   */
39  public class HotspotInternal
40      implements HotspotInternalMBean, MBeanRegistration {
41  
42      private final static String HOTSPOT_INTERNAL_MBEAN_NAME =
43          "sun.management:type=HotspotInternal";
44      private static ObjectName objName = Util.newObjectName(HOTSPOT_INTERNAL_MBEAN_NAME);
45      private MBeanServer server = null;
46  
47      /**
48       * Default constructor that registers all hotspot internal MBeans
49       * to the MBeanServer that creates this MBean.
50       */
51      public HotspotInternal() {
52      }
53  
54      public ObjectName preRegister(MBeanServer server,
55                                    ObjectName name) throws java.lang.Exception {
56          // register all internal MBeans when this MBean is instantiated
57          // and to be registered in a MBeanServer.
58          ManagementFactoryHelper.registerInternalMBeans(server);
59          this.server = server;
60          return objName;
61      }
62  
63      public void postRegister(Boolean registrationDone) {};
64  
65      public void preDeregister() throws java.lang.Exception {
66          // unregister all internal MBeans when this MBean is unregistered.
67          ManagementFactoryHelper.unregisterInternalMBeans(server);
68      }
69  
70      public void postDeregister() {};
71  
72  }
