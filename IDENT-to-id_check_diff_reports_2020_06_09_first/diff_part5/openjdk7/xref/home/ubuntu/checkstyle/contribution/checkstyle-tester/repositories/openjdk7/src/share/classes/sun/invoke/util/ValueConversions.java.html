68      // There are several levels of this unboxing conversions:
69      //   no conversions:  exactly Integer.valueOf, etc.
70      //   implicit conversions sanctioned by JLS 5.1.2, etc.
71      //   explicit conversions as allowed by explicitCastArguments

73      static int unboxInteger(Object x, boolean cast) {
74          if (x instanceof Integer)
75              return ((Integer) x).intValue();
76          return primitiveConversion(Wrapper.INT, x, cast).intValue();
77      }

152     private static final EnumMap<Wrapper, MethodHandle>[]
153             UNBOX_CONVERSIONS = newWrapperCaches(4);

155     private static MethodHandle unbox(Wrapper wrap, boolean raw, boolean cast) {
156         EnumMap<Wrapper, MethodHandle> cache = UNBOX_CONVERSIONS[(cast?1:0)+(raw?2:0)];
157         MethodHandle mh = cache.get(wrap);
158         if (mh != null) {
159             return mh;
160         }
161         // slow path
162         switch (wrap) {
163             case OBJECT:
164                 mh = IDENTITY; break;
165             case VOID:
166                 mh = raw ? ALWAYS_ZERO : IGNORE; break;
167             case INT: case LONG:
168                 // these guys don't need separate raw channels
169                 if (raw)  mh = unbox(wrap, false, cast);
170                 break;
171         }
172         if (mh != null) {
173             cache.put(wrap, mh);
174             return mh;
175         }
176         // look up the method
177         String name = "unbox" + wrap.simpleName() + (raw ? "Raw" : "");
178         MethodType type = unboxType(wrap, raw);
179         try {
180             mh = IMPL_LOOKUP.findStatic(THIS_CLASS, name, type);
181         } catch (ReflectiveOperationException ex) {
182             mh = null;
183         }
184         if (mh != null) {
185             mh = MethodHandles.insertArguments(mh, 1, cast);
186             cache.put(wrap, mh);
187             return mh;
188         }
189         throw new IllegalArgumentException("cannot find unbox adapter for " + wrap
190                 + (cast ? " (cast)" : "") + (raw ? " (raw)" : ""));
191     }

215     /// Primitive conversions
216     public static Number primitiveConversion(Wrapper wrap, Object x, boolean cast) {
217         // Maybe merge this code with Wrapper.convert/cast.
218         Number res = null;
219         if (x == null) {
220             if (!cast)  return null;
221             return ZERO_INT;
222         }
223         if (x instanceof Number) {
224             res = (Number) x;
225         } else if (x instanceof Boolean) {
226             res = ((boolean)x ? ONE_INT : ZERO_INT);
227         } else if (x instanceof Character) {
228             res = (int)(char)x;
229         } else {
230             // this will fail with the required ClassCastException:
231             res = (Number) x;
232         }
233         Wrapper xwrap = Wrapper.findWrapperType(x.getClass());
234         if (xwrap == null || !cast && !wrap.isConvertibleFrom(xwrap))
235             // this will fail with the required ClassCastException:
236             return (Number) wrap.wrapperType().cast(x);
237         return res;
238     }

319     private static MethodHandle box(Wrapper wrap, boolean exact, boolean raw) {
320         EnumMap<Wrapper, MethodHandle> cache = BOX_CONVERSIONS[(exact?1:0)+(raw?2:0)];
321         MethodHandle mh = cache.get(wrap);
322         if (mh != null) {
323             return mh;
324         }
325         // slow path
326         switch (wrap) {
327             case OBJECT:
328                 mh = IDENTITY; break;
329             case VOID:
330                 if (!raw)  mh = ZERO_OBJECT;
331                 break;
332             case INT: case LONG:
333                 // these guys don't need separate raw channels
334                 if (raw)  mh = box(wrap, exact, false);
335                 break;
336         }
337         if (mh != null) {
338             cache.put(wrap, mh);
339             return mh;
340         }
341         // look up the method
342         String name = "box" + wrap.simpleName() + (raw ? "Raw" : "");
343         MethodType type = boxType(wrap, raw);
344         if (exact) {
345             try {
346                 mh = IMPL_LOOKUP.findStatic(THIS_CLASS, name, type);
347             } catch (ReflectiveOperationException ex) {
348                 mh = null;
349             }
350         } else {
351             mh = box(wrap, !exact, raw).asType(type.erase());
352         }
353         if (mh != null) {
354             cache.put(wrap, mh);
355             return mh;
356         }
357         throw new IllegalArgumentException("cannot find box adapter for "
358                 + wrap + (exact ? " (exact)" : "") + (raw ? " (raw)" : ""));
359     }

443     /**
444      * Because we normalize primitive types to reduce the number of signatures,
445      * primitives are sometimes manipulated under an "erased" type,
446      * either int (for types other than long/double) or long (for all types).
447      * When the erased primitive value is then boxed into an Integer or Long,
448      * the final boxed primitive is sometimes required.  This transformation
449      * is called a "rebox".  It takes an Integer or Long and produces some
450      * other boxed value, typed (inexactly) as an Object
451      */
452     public static MethodHandle rebox(Wrapper wrap) {
453         EnumMap<Wrapper, MethodHandle> cache = REBOX_CONVERSIONS[0];
454         MethodHandle mh = cache.get(wrap);
455         if (mh != null) {
456             return mh;
457         }
458         // slow path
459         switch (wrap) {
460             case OBJECT:
461                 mh = IDENTITY; break;
462             case VOID:
463                 throw new IllegalArgumentException("cannot rebox a void");
464         }
465         if (mh != null) {
466             cache.put(wrap, mh);
467             return mh;
468         }
469         // look up the method
470         String name = "reboxRaw" + wrap.simpleName();
471         MethodType type = reboxType(wrap);
472         try {
473             mh = IMPL_LOOKUP.findStatic(THIS_CLASS, name, type);
474             mh = mh.asType(IDENTITY.type());
475         } catch (ReflectiveOperationException ex) {
476             mh = null;
477         }
478         if (mh != null) {
479             cache.put(wrap, mh);
480             return mh;
481         }
482         throw new IllegalArgumentException("cannot find rebox adapter for " + wrap);
483     }

539             CONSTANT_FUNCTIONS = newWrapperCaches(2);

541     public static MethodHandle zeroConstantFunction(Wrapper wrap) {
542         EnumMap<Wrapper, MethodHandle> cache = CONSTANT_FUNCTIONS[0];
543         MethodHandle mh = cache.get(wrap);
544         if (mh != null) {
545             return mh;
546         }
547         // slow path
548         MethodType type = MethodType.methodType(wrap.primitiveType());
549         switch (wrap) {
550             case VOID:
551                 mh = EMPTY;
552                 break;
553             case OBJECT:
554             case INT: case LONG: case FLOAT: case DOUBLE:
555                 try {
556                     mh = IMPL_LOOKUP.findStatic(THIS_CLASS, "zero"+wrap.simpleName(), type);
557                 } catch (ReflectiveOperationException ex) {
558                     mh = null;
559                 }
560                 break;
561         }
562         if (mh != null) {
563             cache.put(wrap, mh);
564             return mh;
565         }
566 
567         // use the raw method
568         Wrapper rawWrap = wrap.rawPrimitive();
569         if (mh == null && rawWrap != wrap) {
570             mh = MethodHandles.explicitCastArguments(zeroConstantFunction(rawWrap), type);
571         }
572         if (mh != null) {
573             cache.put(wrap, mh);
574             return mh;
575         }
576         throw new IllegalArgumentException("cannot find zero constant for " + wrap);
577     }

706     /** Return a method that casts its sole argument (an Object) to the given type
707      *  and returns it as the given type (if exact is true), or as plain Object (if erase is true).
708      */
709     public static MethodHandle cast(Class<?> type) {
710         boolean exact = false;
711         if (type.isPrimitive())  throw new IllegalArgumentException("cannot cast primitive type "+type);
712         MethodHandle mh = null;
713         Wrapper wrap = null;
714         EnumMap<Wrapper, MethodHandle> cache = null;
715         if (Wrapper.isWrapperType(type)) {
716             wrap = Wrapper.forWrapperType(type);
717             cache = WRAPPER_CASTS[exact?1:0];
718             mh = cache.get(wrap);
719             if (mh != null)  return mh;
720         }
721         if (VerifyType.isNullReferenceConversion(Object.class, type))

731         }

733             cache.put(wrap, mh);
734         return mh;

877     static MethodHandle convertFloatFunction(Wrapper wrap, boolean toFloat, boolean doubleSize) {
878         EnumMap<Wrapper, MethodHandle> cache = CONVERT_FLOAT_FUNCTIONS[(toFloat?1:0)+(doubleSize?2:0)];
879         MethodHandle mh = cache.get(wrap);
880         if (mh != null) {
881             return mh;
882         }
883         // slow path
884         Wrapper fwrap = (doubleSize ? Wrapper.DOUBLE : Wrapper.FLOAT);
885         Class<?> fix = wrap.primitiveType();
886         Class<?> flt = (doubleSize ? double.class : float.class);
887         Class<?> src = toFloat ? fix : flt;
888         Class<?> dst = toFloat ? flt : fix;
889         if (src == dst)  return identity(wrap);
890         MethodType type = MethodType.methodType(dst, src);
891         switch (wrap) {
892             case VOID:
893                 mh = toFloat ? zeroConstantFunction(fwrap) : MethodHandles.dropArguments(EMPTY, 0, flt);
894                 break;
895             case OBJECT:
896                 mh = toFloat ? unbox(flt) : box(flt);
897                 break;
898             default:
899                 try {
900                     mh = IMPL_LOOKUP.findStatic(THIS_CLASS, src.getSimpleName()+"To"+capitalize(dst.getSimpleName()), type);
901                 } catch (ReflectiveOperationException ex) {
902                     mh = null;
903                 }
904                 break;
905         }
906         if (mh != null) {
907             assert(mh.type() == type) : mh;
908             cache.put(wrap, mh);
909             return mh;
910         }
911 
912         throw new IllegalArgumentException("cannot find float conversion constant for " +
913                                            src.getSimpleName()+" -> "+dst.getSimpleName());
914     }

939     public static Object convertArrayElements(Class<?> arrayType, Object array) {
940         Class<?> src = array.getClass().getComponentType();
941         Class<?> dst = arrayType.getComponentType();
942         if (src == null || dst == null)  throw new IllegalArgumentException("not array type");
943         Wrapper sw = (src.isPrimitive() ? Wrapper.forPrimitiveType(src) : null);
944         Wrapper dw = (dst.isPrimitive() ? Wrapper.forPrimitiveType(dst) : null);
945         int length;
946         if (sw == null) {
947             Object[] a = (Object[]) array;
948             length = a.length;
949             if (dw == null)
950                 return Arrays.copyOf(a, length, arrayType.asSubclass(Object[].class));
951             Object res = dw.makeArray(length);
952             dw.copyArrayUnboxing(a, 0, res, 0, length);
953             return res;
954         }
955         length = java.lang.reflect.Array.getLength(array);
956         Object[] res;
957         if (dw == null) {
958             res = Arrays.copyOf(NO_ARGS_ARRAY, length, arrayType.asSubclass(Object[].class));
959         } else {
960             res = new Object[length];
961         }
962         sw.copyArrayBoxing(array, 0, res, 0, length);
963         if (dw == null)  return res;
964         Object a = dw.makeArray(length);
965         dw.copyArrayUnboxing(res, 0, a, 0, length);
966         return a;
967     }

1091     private static MethodHandle buildVarargsArray(MethodHandle producer, int nargs) {
1092         // Build up the result mh as a sequence of fills like this:
1093         //   producer(fill(fill(fill(newArray(23),0,x1..x10),10,x11..x20),20,x21..x23))
1094         // The various fill(_,10*I,___*[J]) are reusable.
1095         MethodHandle filler = filler(nargs);
1096         MethodHandle mh = producer;
1097         mh = MethodHandles.dropArguments(mh, 1, filler.type().parameterList());
1098         mh = MethodHandles.foldArguments(mh, filler);
1099         mh = MethodHandles.foldArguments(mh, buildNewArray(nargs));
1100         return mh;
1101     }

1107     private static final MethodHandle[] FILLERS = new MethodHandle[MAX_ARITY+1];
1108     // filler(N).invoke(a, arg0..arg[N-1]) fills a[0]..a[N-1]
1109     private static MethodHandle filler(int nargs) {
1110         MethodHandle filler = FILLERS[nargs];
1111         if (filler != null)  return filler;
1112         return FILLERS[nargs] = buildFiller(nargs);
1113     }
1114     private static MethodHandle buildFiller(int nargs) {
1115         if (nargs == 0)
1116             return MethodHandles.identity(Object[].class);
1117         final int CHUNK = (FILL_ARRAYS.length - 1);
1118         int rightLen = nargs % CHUNK;
1119         int leftLen = nargs - rightLen;
1120         if (rightLen == 0) {
1121             leftLen = nargs - (rightLen = CHUNK);
1122             if (FILLERS[leftLen] == null) {
1123                 // build some precursors from left to right
1124                 for (int j = 0; j < leftLen; j += CHUNK)  filler(j);
1125             }
1126         }
1127         MethodHandle leftFill = filler(leftLen);  // recursive fill
1128         MethodHandle rightFill = FILL_ARRAYS[rightLen];
1129         rightFill = MethodHandles.insertArguments(rightFill, 1, (int) leftLen);  // [leftLen..nargs-1]
1130 
1131         // Combine the two fills: right(left(newArray(nargs), x1..x20), x21..x23)
1132         MethodHandle mh = filler(0);  // identity function produces result
1133         mh = MethodHandles.dropArguments(mh, 1, rightFill.type().parameterList());
1134         mh = MethodHandles.foldArguments(mh, rightFill);
1135         if (leftLen > 0) {
1136             mh = MethodHandles.dropArguments(mh, 1, leftFill.type().parameterList());
1137             mh = MethodHandles.foldArguments(mh, leftFill);
1138         }
1139         return mh;
1140     }

1153      */
1154     public static MethodHandle varargsArray(Class<?> arrayType, int nargs) {
1155         Class<?> elemType = arrayType.getComponentType();
1156         if (elemType == null)  throw new IllegalArgumentException("not an array: "+arrayType);
1157         // FIXME: Need more special casing and caching here.
1158         if (elemType == Object.class)
1159             return varargsArray(nargs);
1160         // other cases:  primitive arrays, subtypes of Object[]
1161         MethodHandle cache[] = TYPED_COLLECTORS.get(elemType);
1162         MethodHandle mh = nargs < cache.length ? cache[nargs] : null;
1163         if (mh != null)  return mh;
1164         MethodHandle producer = buildArrayProducer(arrayType);
1165         mh = buildVarargsArray(producer, nargs);
1166         mh = mh.asType(MethodType.methodType(arrayType, Collections.<Class<?>>nCopies(nargs, elemType)));
1167         cache[nargs] = mh;
1168         return mh;
1169     }
