37  /*
38   *  5 types of entry in SJIS_X_0213/Unicode mapping table
39   *
40   *  (1)Single-Byte
41   *     JIS_X_0213 does not define single-byte character itself, the
42   *     JIS_X_0201 entries are added in for sjis implementation.
43   *
44   *  (2)Double-Byte SJIS <-> BMP Unicode
45   *     ex: 0x8140 U+3000    # IDEOGRAPHIC SPACE
46   *
47   *  (3)Double-Byte SJIS <-> Supplementary
48   *     ex: 0xFCF0 U+2A61A   # <cjk> [2000] [Unicode3.1]
49   *
50   *  (4)Double-Byte SJIS <-> Composite
51   *   ex: 0x83F6 U+31F7+309A # [2000]
52   *
53   *  (5)"Windows-only" special mapping entries
54   *     are handled by MS932_0213.
55   */
56  
57  public class SJIS_0213 extends Charset {
58      public SJIS_0213() {
59          super("x-SJIS_0213", ExtendedCharsets.aliasesFor("SJIS_0213"));
60      }
61  
62      public boolean contains(Charset cs) {
63          return ((cs.name().equals("US-ASCII"))
64                  || (cs instanceof SJIS)
65                  || (cs instanceof SJIS_0213));
66      }
67  
68      public CharsetDecoder newDecoder() {
69          return new Decoder(this);
70      }
71  
72      public CharsetEncoder newEncoder() {
73          return new Encoder(this);
74      }

76      static CharsetMapping mapping =
77          CharsetMapping.get(SJIS_0213.class.getResourceAsStream("sjis0213.dat"));

79      protected static class Decoder extends CharsetDecoder {
80          protected static final char UNMAPPABLE = CharsetMapping.UNMAPPABLE_DECODING;
81  
82          protected Decoder(Charset cs) {
83              super(cs, 0.5f, 1.0f);
84          }
85  
86          private CoderResult decodeArrayLoop(ByteBuffer src, CharBuffer dst) {
87              byte[] sa = src.array();
88              int sp = src.arrayOffset() + src.position();
89              int sl = src.arrayOffset() + src.limit();
90  
91              char[] da = dst.array();
92              int dp = dst.arrayOffset() + dst.position();
93              int dl = dst.arrayOffset() + dst.limit();
94  
95              try {
96                  while (sp < sl) {
97                      int b1 = sa[sp] & 0xff;
98                      char c = decodeSingle(b1);
99                      int inSize = 1, outSize = 1;
100                     char[] cc = null;
101                     if (c == UNMAPPABLE) {
102                         if (sl - sp < 2)
103                             return CoderResult.UNDERFLOW;
104                         int b2 = sa[sp + 1] & 0xff;
105                         c = decodeDouble(b1, b2);
106                         inSize++;
107                         if (c == UNMAPPABLE) {
108                             cc = decodeDoubleEx(b1, b2);
109                             if (cc == null) {
110                                 if (decodeSingle(b2) == UNMAPPABLE)
111                                     return CoderResult.unmappableForLength(2);
112                                 else
113                                     return CoderResult.unmappableForLength(1);
114                             }
115                             outSize++;
116                         }
117                     }
118                     if (dl - dp < outSize)
119                         return CoderResult.OVERFLOW;
120                     if (outSize == 2) {
121                         da[dp++] = cc[0];
122                         da[dp++] = cc[1];
123                     } else {
124                         da[dp++] = c;
125                     }
126                     sp += inSize;
127                 }
128                 return CoderResult.UNDERFLOW;
129             } finally {
130                 src.position(sp - src.arrayOffset());
131                 dst.position(dp - dst.arrayOffset());
132             }
133         }
134 
135         private CoderResult decodeBufferLoop(ByteBuffer src, CharBuffer dst) {
136             int mark = src.position();
137             try {
138                 while (src.hasRemaining()) {
139                     char[] cc = null;
140                     int b1 = src.get() & 0xff;
141                     char c = decodeSingle(b1);
142                     int inSize = 1, outSize = 1;
143                     if (c == UNMAPPABLE) {
144                         if (src.remaining() < 1)
145                             return CoderResult.UNDERFLOW;
146                         int b2 = src.get() & 0xff;
147                         inSize++;
148                         c = decodeDouble(b1, b2);
149                         if (c == UNMAPPABLE) {
150                             cc = decodeDoubleEx(b1, b2);
151                             if (cc == null) {
152                                 if (decodeSingle(b2) == UNMAPPABLE)
153                                     return CoderResult.unmappableForLength(2);
154                                 else
155                                     return CoderResult.unmappableForLength(1);
156                             }
157                             outSize++;
158                         }
159                     }
160                     if (dst.remaining() < outSize)
161                         return CoderResult.OVERFLOW;
162                     if (outSize == 2) {
163                         dst.put(cc[0]);
164                         dst.put(cc[1]);
165                     } else {
166                         dst.put(c);
167                     }
168                     mark += inSize;
169                 }
170                 return CoderResult.UNDERFLOW;
171             } finally {
172                 src.position(mark);
173             }
174         }
175 
176         protected CoderResult decodeLoop(ByteBuffer src, CharBuffer dst) {
177             if (src.hasArray() && dst.hasArray())
178                 return decodeArrayLoop(src, dst);
179             else
180                 return decodeBufferLoop(src, dst);
181         }
182 
183         protected char decodeSingle(int b) {
184             return mapping.decodeSingle(b);
185         }
186 
187         protected char decodeDouble(int b1, int b2) {
188             return mapping.decodeDouble(b1, b2);
189         }
190 
191         private char[] cc = new char[2];
192         private CharsetMapping.Entry comp = new CharsetMapping.Entry();
193         protected char[] decodeDoubleEx(int b1, int b2) {
194             int db = (b1 << 8) | b2;
195             if (mapping.decodeSurrogate(db, cc) != null)
196                 return cc;
197             comp.bs = db;
198             if (mapping.decodeComposite(comp, cc) != null)
199                 return cc;
200             return null;
201         }

204     protected static class Encoder extends CharsetEncoder {
205         protected static final int UNMAPPABLE = CharsetMapping.UNMAPPABLE_ENCODING;
206         protected static final int MAX_SINGLEBYTE = 0xff;
207 
208         protected Encoder(Charset cs) {
209             super(cs, 2.0f, 2.0f);
210         }
211 
212         public boolean canEncode(char c) {
213             return (encodeChar(c) != UNMAPPABLE);
214         }
215 
216         protected int encodeChar(char ch) {
217             return mapping.encodeChar(ch);
218         }
219 
220         protected int encodeSurrogate(char hi, char lo) {
221             return mapping.encodeSurrogate(hi, lo);
222         }
223 
224         private CharsetMapping.Entry comp = new CharsetMapping.Entry();
225         protected int encodeComposite(char base, char cc) {
226             comp.cp = base;
227             comp.cp2 = cc;
228             return mapping.encodeComposite(comp);
229         }
230 
231         protected boolean isCompositeBase(char ch) {
232             comp.cp = ch;
233             return mapping.isCompositeBase(comp);
234         }
235 
236         // Unlike surrogate pair, the base character of a base+cc composite
237         // itself is a legal codepoint in 0213, if we simply return UNDERFLOW
238         // when a base candidate is the last input char in the CharBuffer, like
239         // what we do for the surrogte pair, encoding will fail if this base
240         // character is indeed the last character of the input char sequence.
241         // Keep this base candidate in "leftoverBase" so we can flush it out
242         // at the end of the encoding circle.
243         char leftoverBase = 0;
244         protected CoderResult encodeArrayLoop(CharBuffer src, ByteBuffer dst) {
245             char[] sa = src.array();
246             int sp = src.arrayOffset() + src.position();
247             int sl = src.arrayOffset() + src.limit();
248             byte[] da = dst.array();
249             int dp = dst.arrayOffset() + dst.position();
250             int dl = dst.arrayOffset() + dst.limit();
251 
252             try {
253                 while (sp < sl) {
254                     int db;
255                     char c = sa[sp];
256                     if (leftoverBase != 0) {
257                         boolean isComp = false;
258                         db = encodeComposite(leftoverBase, c);
259                         if (db == UNMAPPABLE)
260                             db = encodeChar(leftoverBase);
261                         else
262                             isComp = true;
263                         if (dl - dp < 2)
264                             return CoderResult.OVERFLOW;
265                         da[dp++] = (byte)(db >> 8);
266                         da[dp++] = (byte)db;
267                         leftoverBase = 0;
268                         if (isComp) {
269                             sp++;
270                             continue;
271                         }
272                     }
273                     if (isCompositeBase(c)) {
274                         leftoverBase = c;
275                     } else {
276                         db = encodeChar(c);
277                         if (db <= MAX_SINGLEBYTE) {      // SingleByte
278                             if (dl <= dp)
279                                 return CoderResult.OVERFLOW;
280                             da[dp++] = (byte)db;
281                         } else if (db != UNMAPPABLE) {   // DoubleByte
282                             if (dl - dp < 2)
283                                 return CoderResult.OVERFLOW;
284                             da[dp++] = (byte)(db >> 8);
285                             da[dp++] = (byte)db;
286                         } else if (Character.isHighSurrogate(c)) {
287                             if ((sp + 1) == sl)
288                                 return CoderResult.UNDERFLOW;
289                             char c2 = sa[sp + 1];
290                             if (!Character.isLowSurrogate(c2))
291                                 return CoderResult.malformedForLength(1);
292                             db = encodeSurrogate(c, c2);
293                             if (db == UNMAPPABLE)
294                                 return CoderResult.unmappableForLength(2);
295                             if (dl - dp < 2)
296                                 return CoderResult.OVERFLOW;
297                             da[dp++] = (byte)(db >> 8);
298                             da[dp++] = (byte)db;
299                             sp++;
300                         } else if (Character.isLowSurrogate(c)) {
301                             return CoderResult.malformedForLength(1);
302                         } else {
303                             return CoderResult.unmappableForLength(1);
304                         }
305                     }
306                     sp++;
307                 }
308                 return CoderResult.UNDERFLOW;
309             } finally {
310                 src.position(sp - src.arrayOffset());
311                 dst.position(dp - dst.arrayOffset());
312             }
313         }
314 
315         protected CoderResult encodeBufferLoop(CharBuffer src, ByteBuffer dst) {
316             int mark = src.position();
317             try {
318                 while (src.hasRemaining()) {
319                     int db;
320                     char c = src.get();
321                     if (leftoverBase != 0) {
322                         boolean isComp = false;
323                         db = encodeComposite(leftoverBase, c);
324                         if (db == UNMAPPABLE)
325                             db = encodeChar(leftoverBase);
326                         else
327                             isComp = true;
328                         if (dst.remaining() < 2)
329                             return CoderResult.OVERFLOW;
330                         dst.put((byte)(db >> 8));
331                         dst.put((byte)(db));
332                         leftoverBase = 0;
333                         if (isComp) {
334                             mark++;
335                             continue;
336                         }
337                     }
338                     if (isCompositeBase(c)) {
339                         leftoverBase = c;
340                     } else {
341                         db = encodeChar(c);
342                         if (db <= MAX_SINGLEBYTE) {    // Single-byte
343                             if (dst.remaining() < 1)
344                                 return CoderResult.OVERFLOW;
345                             dst.put((byte)db);
346                         } else if (db != UNMAPPABLE) {   // DoubleByte
347                             if (dst.remaining() < 2)
348                                 return CoderResult.OVERFLOW;
349                             dst.put((byte)(db >> 8));
350                             dst.put((byte)(db));
351                         } else if (Character.isHighSurrogate(c)) {
352                             if (!src.hasRemaining())     // Surrogates
353                                 return CoderResult.UNDERFLOW;
354                             char c2 = src.get();
355                             if (!Character.isLowSurrogate(c2))
356                                 return CoderResult.malformedForLength(1);
357                             db = encodeSurrogate(c, c2);
358                             if (db == UNMAPPABLE)
359                                 return CoderResult.unmappableForLength(2);
360                             if (dst.remaining() < 2)
361                                 return CoderResult.OVERFLOW;
362                             dst.put((byte)(db >> 8));
363                             dst.put((byte)(db));
364                             mark++;
365                         } else if (Character.isLowSurrogate(c)) {
366                             return CoderResult.malformedForLength(1);
367                         } else {
368                             return CoderResult.unmappableForLength(1);
369                         }
370                     }
371                     mark++;
372                 }
373                 return CoderResult.UNDERFLOW;
374             } finally {
375                 src.position(mark);
376             }
377         }
378 
379         protected CoderResult encodeLoop(CharBuffer src, ByteBuffer dst) {
380             if (src.hasArray() && dst.hasArray())
381                 return encodeArrayLoop(src, dst);
382             else
383                 return encodeBufferLoop(src, dst);
384         }
385 
386         protected CoderResult implFlush(ByteBuffer dst) {
387             if (leftoverBase > 0) {
388                 if (dst.remaining() < 2)
389                     return CoderResult.OVERFLOW;
390                 int db = encodeChar(leftoverBase);
391                 dst.put((byte)(db >> 8));
392                 dst.put((byte)(db));
393                 leftoverBase = 0;
394             }
395             return CoderResult.UNDERFLOW;
396         }
397 
398         protected void implReset() {
399             leftoverBase = 0;
400         }
401     }
