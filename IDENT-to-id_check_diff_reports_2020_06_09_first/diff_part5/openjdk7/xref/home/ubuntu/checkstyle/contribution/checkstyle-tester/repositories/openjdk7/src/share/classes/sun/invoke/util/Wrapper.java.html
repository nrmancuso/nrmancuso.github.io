70      private static abstract class Format {
71          static final int SLOT_SHIFT = 0, SIZE_SHIFT = 2, KIND_SHIFT = 12;
72          static final int
73                  SIGNED   = (-1) << KIND_SHIFT,
74                  UNSIGNED = 0    << KIND_SHIFT,
75                  FLOATING = 1    << KIND_SHIFT;
76          static final int
77                  SLOT_MASK = ((1<<(SIZE_SHIFT-SLOT_SHIFT))-1),
78                  SIZE_MASK = ((1<<(KIND_SHIFT-SIZE_SHIFT))-1);
79          static int format(int kind, int size, int slots) {
80              assert(((kind >> KIND_SHIFT) << KIND_SHIFT) == kind);
81              assert((size & (size-1)) == 0); // power of two
82              assert((kind == SIGNED)   ? (size > 0) :
83                     (kind == UNSIGNED) ? (size > 0) :
84                     (kind == FLOATING) ? (size == 32 || size == 64)  :
85                     false);
86              assert((slots == 2) ? (size == 64) :
87                     (slots == 1) ? (size <= 32) :
88                     false);
89              return kind | (size << SIZE_SHIFT) | (slots << SLOT_SHIFT);
90          }
91          static final int
92                  INT      = SIGNED   | (32 << SIZE_SHIFT) | (1 << SLOT_SHIFT),
93                  SHORT    = SIGNED   | (16 << SIZE_SHIFT) | (1 << SLOT_SHIFT),
94                  BOOLEAN  = UNSIGNED | (1  << SIZE_SHIFT) | (1 << SLOT_SHIFT),
95                  CHAR     = UNSIGNED | (16 << SIZE_SHIFT) | (1 << SLOT_SHIFT),
96                  FLOAT    = FLOATING | (32 << SIZE_SHIFT) | (1 << SLOT_SHIFT),
97                  VOID     = UNSIGNED | (0  << SIZE_SHIFT) | (0 << SLOT_SHIFT),
98                  NUM_MASK = (-1) << SIZE_SHIFT;
99          static int signed(int size)   { return format(SIGNED,   size, (size > 32 ? 2 : 1)); }
100         static int unsigned(int size) { return format(UNSIGNED, size, (size > 32 ? 2 : 1)); }
101         static int floating(int size) { return format(FLOATING, size, (size > 32 ? 2 : 1)); }
102         static int other(int slots)   { return slots << SLOT_SHIFT; }
103     }

107     /** How many bits are in the wrapped value?  Returns 0 for OBJECT or VOID. */
108     public int     bitWidth()      { return (format >> Format.SIZE_SHIFT) & Format.SIZE_MASK; }
109     /** How many JVM stack slots occupied by the wrapped value?  Returns 0 for VOID. */
110     public int     stackSlots()    { return (format >> Format.SLOT_SHIFT) & Format.SLOT_MASK; }
111     /** Does the wrapped value occupy a single JVM stack slot? */
112     public boolean isSingleWord()  { return (format & (1 << Format.SLOT_SHIFT)) != 0; }
113     /** Does the wrapped value occupy two JVM stack slots? */
114     public boolean isDoubleWord()  { return (format & (2 << Format.SLOT_SHIFT)) != 0; }
115     /** Is the wrapped type numeric (not void or object)? */
116     public boolean isNumeric()     { return (format & Format.NUM_MASK) != 0; }
117     /** Is the wrapped type a primitive other than float, double, or void? */
118     public boolean isIntegral()    { return isNumeric() && format < Format.FLOAT; }
119     /** Is the wrapped type one of int, boolean, byte, char, or short? */
120     public boolean isSubwordOrInt() { return isIntegral() && isSingleWord(); }
121     /* Is the wrapped value a signed integral type (one of byte, short, int, or long)? */
122     public boolean isSigned()      { return format < Format.VOID; }
123     /* Is the wrapped value an unsigned integral type (one of boolean or char)? */
124     public boolean isUnsigned()    { return format >= Format.BOOLEAN && format < Format.FLOAT; }
125     /** Is the wrapped type either float or double? */
126     public boolean isFloating()    { return format >= Format.FLOAT; }
127     /** Is the wrapped type either void or a reference? */
128     public boolean isOther()       { return (format & ~Format.SLOT_MASK) == 0; }

130     /** Does the JLS 5.1.2 allow a variable of this wrapper's
131      *  primitive type to be assigned from a value of the given wrapper's primitive type?
132      *  Cases:
133      *  <ul>
134      *  <li>unboxing followed by widening primitive conversion
135      *  <li>any type converted to {@code void} (i.e., dropping a method call's value)
136      *  <li>boxing conversion followed by widening reference conversion to {@code Object}
137      *  </ul>
138      *  These are the cases allowed by MethodHandle.asType.
139      */
140     public boolean isConvertibleFrom(Wrapper source) {
141         if (this == source)  return true;
142         if (this.compareTo(source) < 0) {
143             // At best, this is a narrowing conversion.
144             return false;
145         }
146         // All conversions are allowed in the enum order between floats and signed ints.
147         // First detect non-signed non-float types (boolean, char, Object, void).
148         boolean floatOrSigned = (((this.format & source.format) & Format.SIGNED) != 0);
149         if (!floatOrSigned) {
150             if (this.isOther())  return true;
151             // can convert char to int or wider, but nothing else
152             if (source.format == Format.CHAR)  return true;
153             // no other conversions are classified as widening
154             return false;
155         }
156         // All signed and float conversions in the enum order are widening.
157         assert(this.isFloating() || this.isSigned());
158         assert(source.isFloating() || source.isSigned());
159         return true;
160     }

162     static { assert(checkConvertibleFrom()); }
163     private static boolean checkConvertibleFrom() {
164         // Check the matrix for correct classification of widening conversions.
165         for (Wrapper w : values()) {
166             assert(w.isConvertibleFrom(w));
167             assert(VOID.isConvertibleFrom(w));
168             if (w != VOID) {
169                 assert(OBJECT.isConvertibleFrom(w));
170                 assert(!w.isConvertibleFrom(VOID));
171             }
172             // check relations with unsigned integral types:
173             if (w != CHAR) {
174                 assert(!CHAR.isConvertibleFrom(w));
175                 if (!w.isConvertibleFrom(INT))
176                     assert(!w.isConvertibleFrom(CHAR));
177             }
178             if (w != BOOLEAN) {
179                 assert(!BOOLEAN.isConvertibleFrom(w));
180                 if (w != VOID && w != OBJECT)
181                     assert(!w.isConvertibleFrom(BOOLEAN));
182             }
183             // check relations with signed integral types:
184             if (w.isSigned()) {
185                 for (Wrapper x : values()) {
186                     if (w == x)  continue;
187                     if (x.isFloating())
188                         assert(!w.isConvertibleFrom(x));
189                     else if (x.isSigned()) {
190                         if (w.compareTo(x) < 0)
191                             assert(!w.isConvertibleFrom(x));
192                         else
193                             assert(w.isConvertibleFrom(x));
194                     }
195                 }
196             }
197             // check relations with floating types:
198             if (w.isFloating()) {
199                 for (Wrapper x : values()) {
200                     if (w == x)  continue;
201                     if (x.isSigned())
202                         assert(w.isConvertibleFrom(x));
203                     else if (x.isFloating()) {
204                         if (w.compareTo(x) < 0)
205                             assert(!w.isConvertibleFrom(x));
206                         else
207                             assert(w.isConvertibleFrom(x));
208                     }
209                 }
210             }
211         }
212         return true;  // i.e., assert(true)
213     }

307     // Note on perfect hashes:
308     //   for signature chars c, do (c + (c >> 1)) % 16
309     //   for primitive type names n, do (n[0] + n[2]) % 16
310     // The type name hash works for both primitive and wrapper names.
311     // You can add "java/lang/Object" to the primitive names.
312     // But you add the wrapper name Object, use (n[2] + (3*n[1])) % 16.
313     private static final Wrapper[] FROM_PRIM = new Wrapper[16];
314     private static final Wrapper[] FROM_WRAP = new Wrapper[16];
315     private static final Wrapper[] FROM_CHAR = new Wrapper[16];
316     private static int hashPrim(Class<?> x) {
317         String xn = x.getName();
318         if (xn.length() < 3)  return 0;
319         return (xn.charAt(0) + xn.charAt(2)) % 16;
320     }
321     private static int hashWrap(Class<?> x) {
322         String xn = x.getName();
323         final int offset = 10; assert(offset == "java.lang.".length());
324         if (xn.length() < offset+3)  return 0;
325         return (3*xn.charAt(offset+1) + xn.charAt(offset+2)) % 16;
326     }
327     private static int hashChar(char x) {
328         return (x + (x >> 1)) % 16;
329     }
330     static {
331         for (Wrapper w : values()) {
332             int pi = hashPrim(w.primitiveType);
333             int wi = hashWrap(w.wrapperType);
334             int ci = hashChar(w.basicTypeChar);
335             assert(FROM_PRIM[pi] == null);
336             assert(FROM_WRAP[wi] == null);
337             assert(FROM_CHAR[ci] == null);
338             FROM_PRIM[pi] = w;
339             FROM_WRAP[wi] = w;
340             FROM_CHAR[ci] = w;
341         }
342         //assert(jdk.sun.invoke.util.WrapperTest.test(false));
343     }

351     /** What is the wrapper type for this wrapper?
352      * Otherwise, the example type must be the wrapper type,
353      * or the corresponding primitive type.
354      * (For {@code OBJECT}, the example type can be any non-primitive,
355      * and is normalized to {@code Object.class}.)
356      * The resulting class type has the same type parameter.
357      */
358     public <T> Class<T> wrapperType(Class<T> exampleType) {
359         if (exampleType == wrapperType) {
360             return exampleType;
361         } else if (exampleType == primitiveType ||
362                    wrapperType == Object.class ||
363                    exampleType.isInterface()) {
364             return forceType(wrapperType, exampleType);
365         }
366         throw newClassCastException(exampleType, primitiveType);
367     }

453     private <T> T convert(Object x, Class<T> type, boolean isCast) {
454         if (this == OBJECT) {
455             // If the target wrapper is OBJECT, just do a reference cast.
456             // If the target type is an interface, perform no runtime check.
457             // (This loophole is safe, and is allowed by the JVM verifier.)
458             // If the target type is a primitive, change it to a wrapper.
459             @SuppressWarnings("unchecked")
460             T result = (T) x;  // unchecked warning is expected here
461             return result;
462         }
463         Class<T> wtype = wrapperType(type);
464         if (wtype.isInstance(x)) {
465             @SuppressWarnings("unchecked")
466             T result = (T) x;  // unchecked warning is expected here
467             return result;
468         }
469         Class<?> sourceType = x.getClass();  // throw NPE if x is null
470         if (!isCast) {
471             Wrapper source = findWrapperType(sourceType);
472             if (source == null || !this.isConvertibleFrom(source)) {
473                 throw newClassCastException(wtype, sourceType);
474             }
475         }
476         @SuppressWarnings("unchecked")
477         T result = (T) wrap(x);  // unchecked warning is expected here
478         assert result.getClass() == wtype;
479         return result;
480     }

482     /** Cast a reference type to another reference type.
483      * If the target type is an interface, perform no runtime check.
484      * (This loophole is safe, and is allowed by the JVM verifier.)
485      * If the target type is a primitive, change it to a wrapper.
486      */
487     static <T> Class<T> forceType(Class<?> type, Class<T> exampleType) {
488         boolean z = (type == exampleType ||
489                type.isPrimitive() && forPrimitiveType(type) == findWrapperType(exampleType) ||
490                exampleType.isPrimitive() && forPrimitiveType(exampleType) == findWrapperType(type) ||
491                type == Object.class && !exampleType.isPrimitive());
492         if (!z)
493             System.out.println(type+" <= "+exampleType);
494         assert(type == exampleType ||
495                type.isPrimitive() && forPrimitiveType(type) == findWrapperType(exampleType) ||
496                exampleType.isPrimitive() && forPrimitiveType(exampleType) == findWrapperType(type) ||
497                type == Object.class && !exampleType.isPrimitive());
498         @SuppressWarnings("unchecked")
499         Class<T> result = (Class<T>) type;  // unchecked warning is expected here
500         return result;
501     }

503     /** Wrap a value in this wrapper's type.
504      * Performs standard primitive conversions, including truncation and float conversions.
505      * Performs returns the unchanged reference for {@code OBJECT}.
506      * Returns null for {@code VOID}.
507      * Returns a zero value for a null input.
508      * @throws ClassCastException if this wrapper is numeric and the operand
509      *                            is not a number, character, boolean, or null
510      */
511     public Object wrap(Object x) {
512         // do non-numeric wrappers first
513         switch (basicTypeChar) {
514             case 'L': return x;
515             case 'V': return null;
516         }
517         Number xn = numberValue(x);
518         switch (basicTypeChar) {
519             case 'I': return Integer.valueOf(xn.intValue());
520             case 'J': return Long.valueOf(xn.longValue());
521             case 'F': return Float.valueOf(xn.floatValue());
522             case 'D': return Double.valueOf(xn.doubleValue());
523             case 'S': return Short.valueOf((short) xn.intValue());
524             case 'B': return Byte.valueOf((byte) xn.intValue());
525             case 'C': return Character.valueOf((char) xn.intValue());
526             case 'Z': return Boolean.valueOf(boolValue(xn.longValue()));
527         }
528         throw new InternalError("bad wrapper");
529     }

531     /** Wrap a value (an int or smaller value) in this wrapper's type.
532      * Performs standard primitive conversions, including truncation and float conversions.
533      * Produces an {@code Integer} for {@code OBJECT}, although the exact type
534      * of the operand is not known.
535      * Returns null for {@code VOID}.
536      */
537     public Object wrap(int x) {
538         if (basicTypeChar == 'L')  return (Integer)x;
539         switch (basicTypeChar) {
540             case 'L': throw newIllegalArgumentException("cannot wrap to object type");
541             case 'V': return null;
542             case 'I': return Integer.valueOf((int)x);
543             case 'J': return Long.valueOf(x);
544             case 'F': return Float.valueOf(x);
545             case 'D': return Double.valueOf(x);
546             case 'S': return Short.valueOf((short) x);
547             case 'B': return Byte.valueOf((byte) x);
548             case 'C': return Character.valueOf((char) x);
549             case 'Z': return Boolean.valueOf(boolValue(x));
550         }
551         throw new InternalError("bad wrapper");
552     }

554     /** Wrap a value (a long or smaller value) in this wrapper's type.
555      * Does not perform floating point conversion.
556      * Produces a {@code Long} for {@code OBJECT}, although the exact type
557      * of the operand is not known.
558      * Returns null for {@code VOID}.
559      */
560     public Object wrapRaw(long x) {
561         switch (basicTypeChar) {
562             case 'F':  return Float.valueOf(Float.intBitsToFloat((int)x));
563             case 'D':  return Double.valueOf(Double.longBitsToDouble(x));
564             case 'L':  // same as 'J':
565             case 'J':  return (Long) x;
566         }
567         // Other wrapping operations are just the same, given that the
568         // operand is already promoted to an int.
569         return wrap((int)x);
570     }

572     /** Produce bitwise value which encodes the given wrapped value.
573      * Does not perform floating point conversion.
574      * Returns zero for {@code VOID}.
575      */
576     public long unwrapRaw(Object x) {
577         switch (basicTypeChar) {
578             case 'F':  return Float.floatToRawIntBits((Float) x);
579             case 'D':  return Double.doubleToRawLongBits((Double) x);
580 
581             case 'L': throw newIllegalArgumentException("cannot unwrap from sobject type");
582             case 'V': return 0;
583             case 'I': return (int)(Integer) x;
584             case 'J': return (long)(Long) x;
585             case 'S': return (short)(Short) x;
586             case 'B': return (byte)(Byte) x;
587             case 'C': return (char)(Character) x;
588             case 'Z': return (boolean)(Boolean) x ? 1 : 0;
589         }
590         throw new InternalError("bad wrapper");
591     }

595         return rawPrimitive().primitiveType();

598     /** Report, as a wrapper, what primitive type holds this guy's raw value.
599      *  Returns self for INT, LONG, OBJECT; returns LONG for DOUBLE,
600      *  else returns INT.
601      */
602     public Wrapper rawPrimitive() {
603         switch (basicTypeChar) {
604             case 'S': case 'B':
605             case 'C': case 'Z':
606             case 'V':
607             case 'F':
608                 return INT;
609             case 'D':
610                 return LONG;
611         }
612         return this;
613     }

615     private static Number numberValue(Object x) {
616         if (x instanceof Number)     return (Number)x;
617         if (x instanceof Character)  return (int)(Character)x;
618         if (x instanceof Boolean)    return (Boolean)x ? 1 : 0;
619         // Remaining allowed case of void:  Must be a null reference.
620         return (Number)x;
621     }

624         bits &= 1;  // simple 31-bit zero extension
625         return (bits != 0);

642     public void copyArrayUnboxing(Object[] values, int vpos, Object a, int apos, int length) {
643         if (a.getClass() != arrayType())
644             arrayType().cast(a);  // throw NPE or CCE if bad type
645         for (int i = 0; i < length; i++) {
646             Object value = values[i+vpos];
647             value = convert(value, primitiveType);
648             java.lang.reflect.Array.set(a, i+apos, value);
649         }
650     }
651     public void copyArrayBoxing(Object a, int apos, Object[] values, int vpos, int length) {
652         if (a.getClass() != arrayType())
653             arrayType().cast(a);  // throw NPE or CCE if bad type
654         for (int i = 0; i < length; i++) {
655             Object value = java.lang.reflect.Array.get(a, i+apos);
656             //Already done: value = convert(value, primitiveType);
657             assert(value.getClass() == wrapperType);
658             values[i+vpos] = value;
659         }
