54          size = intArg(args, 0, DEFAULT_SIZE);

57                    new ConcurrentSkipListMap());
58          lockSteps(new TreeMap(reverseOrder()),
59                    new ConcurrentSkipListMap(reverseOrder()));

67      static void lockSteps(NavigableMap m1, NavigableMap m2) {
68          if (maybe(4)) m1 = serialClone(m1);
69          if (maybe(4)) m2 = serialClone(m2);
70          lockStep(m1,
71                   m2);
72          lockStep(m1.descendingMap(),
73                   m2.descendingMap());
74          lockStep(fullSubMap(m1),
75                   fullSubMap(m2));
76          lockStep(fullSubMap(m1.descendingMap()),
77                   fullSubMap(m2.descendingMap()));
78          lockStep(fullHeadMap(m1),
79                   fullHeadMap(m2));
80          lockStep(fullHeadMap(m1.descendingMap()),
81                   fullHeadMap(m2.descendingMap()));
82          lockStep(fullTailMap(m1),
83                   fullTailMap(m2));
84          lockStep(fullTailMap(m1.descendingMap()),
85                   fullTailMap(m2.descendingMap()));
86      }

89          if (maybe(4)) s1 = serialClone(s1);
90          if (maybe(4)) s2 = serialClone(s2);
91          lockStep(s1,
92                   s2);
93          lockStep(s1.descendingSet(),
94                   s2.descendingSet());
95          lockStep(fullSubSet(s1),
96                   fullSubSet(s2));
97          lockStep(fullSubSet(s1.descendingSet()),
98                   fullSubSet(s2.descendingSet()));
99          lockStep(fullHeadSet(s1),
100                  fullHeadSet(s2));
101         lockStep(fullHeadSet(s1.descendingSet()),
102                  fullHeadSet(s2.descendingSet()));
103         lockStep(fullTailSet(s1),
104                  fullTailSet(s2));
105         lockStep(fullTailSet(s1.descendingSet()),
106                  fullTailSet(s2.descendingSet()));
107     }

206     static void checkNavigableSet(final NavigableSet s) {
207         if (s.comparator() == null)
208             check(s.descendingSet().descendingSet().comparator() == null);
209         equal(s.isEmpty(), s.size() == 0);
210         equal2(s, s.descendingSet());
211         if (maybe(4) && s instanceof Serializable)
212             equal2(s, serialClone(s));
213         Comparator cmp = comparator(s);
214         if (s.isEmpty()) {
215             THROWS(NoSuchElementException.class,
216                    new Fun(){void f(){ s.first(); }},
217                    new Fun(){void f(){ s.last();  }});
218             equal(null, s.lower(1));
219             equal(null, s.floor(1));
220             equal(null, s.ceiling(1));
221             equal(null, s.higher(1));
222         } else {
223             Object a = s.first();
224             Object z = s.last();
225             equal(s.lower(a), null);
226             equal(s.higher(z), null);
227             equal2(s, s.tailSet(a));
228             equal2(s, s.headSet(z, true));
229             equal2(s, s.subSet(a, true, z, true));
230 
231             testEmptySet(s.subSet(a, true, a, false));
232             testEmptySet(s.subSet(z, true, z, false));
233             testEmptySet(s.headSet(a, false));
234             testEmptySet(s.tailSet(z, false));
235 
236             equal2(s.headSet(a, true), singleton(a));
237             equal2(s.tailSet(z, true), singleton(z));
238         }
239         Iterator[] its = new Iterator[] {
240             s.iterator(),
241             s.descendingSet().descendingSet().iterator(),
242         };
243         for (final Iterator it : its)
244             if (maybe(4))
245                 THROWS(IllegalStateException.class,
246                        new Fun(){void f(){ it.remove(); }});
247         Object prev = null;
248         for (Object e : s) {
249             check(s.contains(e));
250             for (Iterator it : its) equalNext(it, e);
251             equal(e, s.ceiling(e));
252             equal(e, s.floor(e));
253             check(s.higher(e) == null || cmp.compare(e, s.higher(e)) < 0);
254             equal(s.lower(e), prev);
255             if (prev == null) {
256             } else {
257                 check(cmp.compare(prev, e) < 0);
258             }
259             prev = e;
260         }
261         for (final Iterator it : its) {
262             if (maybe(2))
263                 check(! it.hasNext());
264             Fun fun = new Fun(){void f(){ it.next(); }};
265             THROWS(NoSuchElementException.class, fun, fun, fun);
266         }
267     }

269     static void equalIterators(final Iterator<?> it1,
270                                final Iterator<?> it2) {
271         while (it1.hasNext()) {
272             if (maybe(2))
273                 check(it2.hasNext());
274             equal(it1.next(), it2.next());
275         }
276         check(! it2.hasNext());
277     }

279     static void equalNavigableSetsLeaf(final NavigableSet s1,
280                                        final NavigableSet s2) {
281         equal2(s1,            s2);
282         equal( s1.size(),     s2.size());
283         equal( s1.isEmpty(),  s2.isEmpty());
284         equal( s1.hashCode(), s2.hashCode());
285         equal( s1.toString(), s2.toString());
286         if (! s1.isEmpty()) {
287             equal(s1.first(), s2.first());
288             equal(s1.last(),  s2.last());
289         }
290         equalIterators(s1.iterator(), s2.iterator());
291         equalIterators(s1.descendingIterator(), s2.descendingIterator());
292         checkNavigableSet(s1);
293         checkNavigableSet(s2);
294     }

296     static void equalNavigableSets(final NavigableSet s1,
297                                    final NavigableSet s2) {
298         equalNavigableSetsLeaf(s1, s2);
299         equalNavigableSetsLeaf(s1.descendingSet(), s2.descendingSet());
300         equalNavigableSetsLeaf(s1.descendingSet().descendingSet(), s2);
301         Object min = s1.isEmpty() ? Integer.MIN_VALUE : s1.first();
302         Object max = s2.isEmpty() ? Integer.MAX_VALUE : s2.last();
303         if (s1.comparator() != null &&
304             s1.comparator().compare(min, max) > 0) {
305             Object tmp = min; min = max; max = tmp;
306         }
307 
308         equalNavigableSetsLeaf(s1.subSet(min, true, max, true),
309                                s2.subSet(min, true, max, true));
310         equalNavigableSetsLeaf(s1.tailSet(min, true),
311                                s2.tailSet(min, true));
312         equalNavigableSetsLeaf(s1.headSet(max, true),
313                                s2.headSet(max, true));
314         equalNavigableSetsLeaf((NavigableSet) s1.subSet(min, max),
315                                (NavigableSet) s2.subSet(min, max));
316         equalNavigableSetsLeaf((NavigableSet) s1.tailSet(min),
317                                (NavigableSet) s2.tailSet(min));
318         equalNavigableSetsLeaf((NavigableSet) s1.headSet(max),
319                                (NavigableSet) s2.headSet(max));
320     }

323     static <T> T[] concat(T[]... arrays) {
324         int len = 0;
325         for (int i = 0; i < arrays.length; i++)
326             len += arrays[i].length;
327         T[] a = (T[])java.lang.reflect.Array
328             .newInstance(arrays[0].getClass().getComponentType(), len);
329         int k = 0;
330         for (int i = 0; i < arrays.length; i++) {
331             T[] array = arrays[i];
332             System.arraycopy(array, 0, a, k, array.length);
333             k += array.length;
334         }
335         return a;
336     }

338     static void checkNavigableMap(final NavigableMap m) {
339         if (m.comparator() == null) {
340             check(m.descendingMap().descendingMap().comparator() == null);
341             check(m.descendingKeySet().descendingSet().comparator() == null);
342         }
343         equal(m.isEmpty(), m.size() == 0);
344         equal2(m, m.descendingMap());
345         if (maybe(4))
346             equal2(m, serialClone(m));
347         equal2(m.keySet(), m.descendingKeySet());
348         Comparator cmp = comparator(m);
349         if (m.isEmpty()) {
350             THROWS(NoSuchElementException.class,
351                    new Fun(){void f(){ m.firstKey(); }},
352                    new Fun(){void f(){ m.lastKey();  }});
353             equal(null, m.firstEntry());
354             equal(null, m.lastEntry());
355             equal(null, m.pollFirstEntry());
356             equal(null, m.pollLastEntry());
357             equal(null, m.lowerKey(1));
358             equal(null, m.floorKey(1));
359             equal(null, m.ceilingKey(1));
360             equal(null, m.higherKey(1));
361             equal(null, m.lowerEntry(1));
362             equal(null, m.floorEntry(1));
363             equal(null, m.ceilingEntry(1));
364             equal(null, m.higherEntry(1));
365         } else {
366             Object a = m.firstKey();
367             Object z = m.lastKey();
368             equal(m.lowerKey(a), null);
369             equal(m.higherKey(z), null);
370             equal(a, m.firstEntry().getKey());
371             equal(z, m.lastEntry().getKey());
372             equal2(m, m.tailMap(a));
373             equal2(m, m.headMap(z, true));
374             equal2(m, m.subMap(a, true, z, true));
375 
376             testEmptyMap(m.subMap(a, true, a, false));
377             testEmptyMap(m.subMap(z, true, z, false));
378             testEmptyMap(m.headMap(a, false));
379             testEmptyMap(m.tailMap(z, false));
380 
381             equal2(m.headMap(a, true), singletonMap(a, m.get(a)));
382             equal2(m.tailMap(z, true), singletonMap(z, m.get(z)));
383         }
384 
385         Iterator[] kits = new Iterator[] {
386             m.keySet().iterator(),
387             m.descendingMap().descendingKeySet().iterator(),
388             m.descendingKeySet().descendingSet().iterator(),
389         };
390         Iterator[] vits = new Iterator[] {
391             m.values().iterator(),
392             m.descendingMap().descendingMap().values().iterator(),
393         };
394         Iterator[] eits = new Iterator[] {
395             m.entrySet().iterator(),
396             m.descendingMap().descendingMap().entrySet().iterator(),
397         };
398         Iterator[] its = concat(kits, vits, eits);
399         for (final Iterator it : its)
400             if (maybe(4))
401                 THROWS(IllegalStateException.class,
402                        new Fun(){void f(){ it.remove(); }});
403         Map.Entry prev = null;
404         for (Map.Entry e : (Set<Map.Entry>) m.entrySet()) {
405             Object k = e.getKey();
406             Object v = e.getValue();
407             check(m.containsKey(k));
408             check(m.containsValue(v));
409             for (Iterator kit : kits) equalNext(kit, k);
410             for (Iterator vit : vits) equalNext(vit, v);
411             for (Iterator eit : eits) equalNext(eit, e);
412             equal(k, m.ceilingKey(k));
413             equal(k, m.ceilingEntry(k).getKey());
414             equal(k, m.floorKey(k));
415             equal(k, m.floorEntry(k).getKey());
416             check(m.higherKey(k) == null || cmp.compare(k, m.higherKey(k)) < 0);
417             check(m.lowerKey(k)  == null || cmp.compare(k, m.lowerKey(k))  > 0);
418             equal(m.lowerEntry(k), prev);
419             if (prev == null) {
420                 equal(m.lowerKey(k), null);
421             } else {
422                 equal(m.lowerKey(k), prev.getKey());
423                 check(cmp.compare(prev.getKey(), e.getKey()) < 0);
424             }
425             prev = e;
426         }
427         for (final Iterator it : its) {
428             if (maybe(2))
429                 check(! it.hasNext());
430             Fun fun = new Fun(){void f(){ it.next(); }};
431             THROWS(NoSuchElementException.class, fun, fun, fun);
432         }
433     }

448     static void equalNavigableMaps(NavigableMap m1,
449                                    NavigableMap m2) {
450         equalNavigableMapsLeaf(m1, m2);
451         equalNavigableSetsLeaf((NavigableSet) m1.keySet(),
452                                (NavigableSet) m2.keySet());
453         equalNavigableSets(m1.navigableKeySet(),
454                            m2.navigableKeySet());
455         equalNavigableSets(m1.descendingKeySet(),
456                            m2.descendingKeySet());
457         equal2(m1.entrySet(),
458                m2.entrySet());
459         equalNavigableMapsLeaf(m1.descendingMap(),
460                                m2.descendingMap());
461         equalNavigableMapsLeaf(m1.descendingMap().descendingMap(),
462                                m2);
463         equalNavigableSetsLeaf((NavigableSet) m1.descendingMap().keySet(),
464                                (NavigableSet) m2.descendingMap().keySet());
465         equalNavigableSetsLeaf(m1.descendingMap().descendingKeySet(),
466                                m2.descendingMap().descendingKeySet());
467         equal2(m1.descendingMap().entrySet(),
468                m2.descendingMap().entrySet());
469 
470         //----------------------------------------------------------------
471         // submaps
472         //----------------------------------------------------------------
473         Object min = Integer.MIN_VALUE;
474         Object max = Integer.MAX_VALUE;
475         if (m1.comparator() != null
476             && m1.comparator().compare(min, max) > 0) {
477             Object tmp = min; min = max; max = tmp;
478         }
479         switch (rnd.nextInt(6)) {
480         case 0:
481             equalNavigableMapsLeaf(m1.subMap(min, true, max, true),
482                                    m2.subMap(min, true, max, true));
483             break;
484         case 1:
485             equalNavigableMapsLeaf(m1.tailMap(min, true),
486                                    m2.tailMap(min, true));
487             break;
488         case 2:
489             equalNavigableMapsLeaf(m1.headMap(max, true),
490                                    m2.headMap(max, true));
491             break;
492         case 3:
493             equalNavigableMapsLeaf((NavigableMap) m1.subMap(min, max),
494                                    (NavigableMap) m2.subMap(min, max));
495             break;
496         case 4:
497             equalNavigableMapsLeaf((NavigableMap) m1.tailMap(min),
498                                    (NavigableMap) m2.tailMap(min));
499             break;
500         case 5:
501             equalNavigableMapsLeaf((NavigableMap) m1.headMap(max),
502                                    (NavigableMap) m2.headMap(max));
503             break;
504         }
505     }

510     static MapFrobber randomAdder(NavigableMap m) {
511         final Integer k = unusedKey(m);
512         final MapFrobber[] randomAdders = {
513             new MapFrobber() {void frob(NavigableMap m) {
514                 equal(m.put(k, k+1), null);
515                 equal(m.get(k), k+1);
516                 if (maybe(4)) {
517                     equal(m.put(k, k+1), k+1);
518                     equal(m.get(k), k+1);}}},
519             new MapFrobber() {void frob(NavigableMap m) {
520                 m.descendingMap().put(k, k+1);
521                 equal(m.get(k), k+1);}},
522             new MapFrobber() {void frob(NavigableMap m) {
523                 m.tailMap(k,true).headMap(k,true).put(k,k+1);}},
524             new MapFrobber() {void frob(NavigableMap m) {
525                 m.tailMap(k,true).headMap(k,true).descendingMap().put(k,k+1);}}
526         };
527         return new MapFrobber() {void frob(NavigableMap m) {
528             randomAdders[rnd.nextInt(randomAdders.length)].frob(m);
529             if (maybe(2)) equal(m.get(k), k+1);
530             if (maybe(4)) {
531                 equal(m.put(k, k+1), k+1);
532                 equal(m.get(k), k+1);}}};
533     }

535     static SetFrobber randomAdder(NavigableSet s) {
536         final Integer e = unusedElt(s);
537         final SetFrobber[] randomAdders = {
538             new SetFrobber() {void frob(NavigableSet s) {
539                 check(s.add(e));}},
540             new SetFrobber() {void frob(NavigableSet s) {
541                 s.descendingSet().add(e);}},
542             new SetFrobber() {void frob(NavigableSet s) {
543                 s.tailSet(e,true).headSet(e,true).add(e);}},
544             new SetFrobber() {void frob(NavigableSet s) {
545                 s.descendingSet().tailSet(e,true).headSet(e,true).add(e);}}
546         };
547         return new SetFrobber() {void frob(NavigableSet s) {
548             if (maybe(2)) check(! s.contains(e));
549             randomAdders[rnd.nextInt(randomAdders.length)].frob(s);
550             if (maybe(2)) check(! s.add(e));
551             if (maybe(2)) check(s.contains(e));}};
552     }

586     static void checkUnusedKey(NavigableMap m, Object k) {
