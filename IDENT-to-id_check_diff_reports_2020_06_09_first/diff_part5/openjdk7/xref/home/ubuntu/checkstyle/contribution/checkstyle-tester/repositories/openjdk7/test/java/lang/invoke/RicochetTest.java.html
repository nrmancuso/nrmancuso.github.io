158                 case 1:  args2 = (int[]) coll.invokeExact(args[0]); break;

162                 case 5:  args2 = (int[]) coll.invokeExact(args[0], args[1], args[2], args[3], args[4]); break;

164             assertArrayEquals(args, args2);
165             MethodHandle mh = coll.asSpreader(int[].class, nargs);

229     @Test
230     public void testIntCollects() throws Throwable {
231         if (!startTest("testIntCollects"))  return;
232         for (MethodHandle lister : INT_LISTERS) {
233             int outputs = lister.type().parameterCount();
234             for (int collects = 0; collects <= Math.min(outputs, INT_COLLECTORS.length-1); collects++) {
235                 int inputs = outputs - 1 + collects;
236                 if (inputs < 0)  continue;
237                 for (int pos = 0; pos + collects <= inputs; pos++) {
238                     MethodHandle collector = INT_COLLECTORS[collects];
239                     int[] args = new int[inputs];
240                     int ap = 0, arg = 31;
241                     for (int i = 0; i < pos; i++)
242                         args[ap++] = arg++ + 0;
243                     for (int i = 0; i < collects; i++)
244                         args[ap++] = arg++ + 10;
245                     while (ap < args.length)
246                         args[ap++] = arg++ + 20;
247                     // calculate piecemeal:
248                     //System.out.println("testIntCollects "+Arrays.asList(lister, pos, collector)+" on "+Arrays.toString(args));
249                     int[] collargs = Arrays.copyOfRange(args, pos, pos+collects);
250                     int coll = (int) collector.asSpreader(int[].class, collargs.length).invokeExact(collargs);
251                     int[] listargs = Arrays.copyOfRange(args, 0, outputs);
252                     System.arraycopy(args, pos+collects, listargs, pos+1, outputs - (pos+1));
253                     listargs[pos] = coll;
254                     //System.out.println("  coll="+coll+" listargs="+Arrays.toString(listargs));
255                     Object expect = lister.asSpreader(int[].class, listargs.length).invokeExact(listargs);
256                     //System.out.println("  expect="+expect);
257 
258                     // now use the combined MH, and test the output:
259                     MethodHandle mh = collectArguments(lister, pos, INT_COLLECTORS[collects]);
260                     if (mh == null)  continue;  // no infix collection, yet
261                     assert(mh.type().parameterCount() == inputs);
262                     Object observe = mh.asSpreader(int[].class, args.length).invokeExact(args);
263                     assertEquals(expect, observe);
264                 }
265             }
266         }
267     }

269     private static MethodHandle collectArguments(MethodHandle lister, int pos, MethodHandle collector) {
270         int collects = collector.type().parameterCount();
271         int outputs = lister.type().parameterCount();
272         if (pos == outputs - 1)
273             return MethodHandles.filterArguments(lister, pos,
274                         collector.asSpreader(int[].class, collects))
275                             .asCollector(int[].class, collects);
276         //return MethodHandles.collectArguments(lister, pos, collector); //no such animal
277         return null;
278     }

288     @Test
289     public void testReturns() throws Throwable {
290         if (!startTest("testReturns"))  return;
291         // fault injection:
292         int faultCount = 0;  // total of 1296 tests
293         faultCount = Integer.getInteger("testReturns.faultCount", 0);
294         for (Class<?> ret : RETURN_TYPES) {
295             // make a complicated identity function and pass something through it
296             System.out.println(ret.getSimpleName());
297             Class<?> vret = (ret == void.class) ? Void.class : ret;
298             MethodHandle id = // (vret)->ret
299                 identity(vret).asType(methodType(ret, vret));
300             final int LENGTH = 4;
301             int[] index = {0};
302             Object vals = java.lang.reflect.Array.newInstance(vret, LENGTH);
303             MethodHandle indexGetter =  //()->int
304                 insertArguments(arrayElementGetter(index.getClass()), 0, index, 0);
305             MethodHandle valSelector =  // (int)->vret
306                 arrayElementGetter(vals.getClass()).bindTo(vals);
307             MethodHandle valGetter =  // ()->vret
308                 foldArguments(valSelector, indexGetter);
309             if (ret != void.class) {
310                 for (int i = 0; i < LENGTH; i++) {
311                     Object val = (i + 50);
312                     if (ret == boolean.class)  val = (i % 3 == 0);
313                     if (ret == String.class)   val = "#"+i;
314                     if (ret == char.class)     val = (char)('a'+i);
315                     if (ret == byte.class)     val = (byte)~i;
316                     if (ret == short.class)    val = (short)(1<<i);
317                     java.lang.reflect.Array.set(vals, i, val);
318                 }
319             }
320             for (int i = 0; i < LENGTH; i++) {
321                 Object val = java.lang.reflect.Array.get(vals, i);
322                 System.out.println(i+" => "+val);
323                 index[0] = i;
324                 if (--faultCount == 0)  index[0] ^= 1;
325                 Object x = valGetter.invokeWithArguments();
326                 assertEquals(val, x);
327                 // make a return-filter call:  x = id(valGetter())
328                 if (--faultCount == 0)  index[0] ^= 1;
329                 x = filterReturnValue(valGetter, id).invokeWithArguments();
330                 assertEquals(val, x);
331                 // make a filter call:  x = id(*,valGetter(),*)
332                 for (int len = 1; len <= 4; len++) {
333                     for (int pos = 0; pos < len; pos++) {
334                         MethodHandle proj = id;  // lambda(..., vret x,...){x}
335                         for (int j = 0; j < len; j++) {
336                             if (j == pos)  continue;
337                             proj = dropArguments(proj, j, Object.class);
338                         }
339                         assert(proj.type().parameterCount() == len);
340                         // proj: (Object*, pos: vret, Object*)->ret
341                         assertEquals(vret, proj.type().parameterType(pos));
342                         MethodHandle vgFilter = dropArguments(valGetter, 0, Object.class);
343                         if (--faultCount == 0)  index[0] ^= 1;
344                         x = filterArguments(proj, pos, vgFilter).invokeWithArguments(new Object[len]);
345                         assertEquals(val, x);
346                     }
347                 }
348                 // make a fold call:
349                 for (int len = 0; len <= 4; len++) {
350                     for (int fold = 0; fold <= len; fold++) {
351                         MethodHandle proj = id;  // lambda(ret x, ...){x}
352                         if (ret == void.class)  proj = constant(Object.class, null);
353                         int arg0 = (ret == void.class ? 0 : 1);
354                         for (int j = 0; j < len; j++) {
355                             proj = dropArguments(proj, arg0, Object.class);
356                         }
357                         assert(proj.type().parameterCount() == arg0 + len);
358                         // proj: (Object*, pos: vret, Object*)->ret
359                         if (arg0 != 0)  assertEquals(vret, proj.type().parameterType(0));
360                         MethodHandle vgFilter = valGetter.asType(methodType(ret));
361                         for (int j = 0; j < fold; j++) {
362                             vgFilter = dropArguments(vgFilter, j, Object.class);
363                         }
364                         x = foldArguments(proj, vgFilter).invokeWithArguments(new Object[len]);
365                         if (--faultCount == 0)  index[0] ^= 1;
366                         assertEquals(val, x);
367                     }
368                 }
369             }
370         }
371         //System.out.println("faultCount="+faultCount);
372     }

374     private static MethodHandle sequence(MethodHandle mh1, MethodHandle... mhs) {
375         MethodHandle res = mh1;
376         for (MethodHandle mh2 : mhs)
377             res = filterReturnValue(res, mh2);

380     private static void assertEqualFunction(MethodHandle x, MethodHandle y) throws Throwable {
381         assertEquals(x.type(), y.type()); //??
382         MethodType t = x.type();
383         if (t.parameterCount() == 0) {
384             assertEqualFunctionAt(null, x, y);
385             return;
386         }
387         Class<?> ptype = t.parameterType(0);
388         if (ptype == long.class || ptype == Long.class) {
389             for (long i = -10; i <= 10; i++) {
390                 assertEqualFunctionAt(i, x, y);
391             }
392         } else {
393             for (int i = -10; i <= 10; i++) {
394                 assertEqualFunctionAt(i, x, y);
395             }
396         }
397     }
398     private static void assertEqualFunctionAt(Object v, MethodHandle x, MethodHandle y) throws Throwable {
399         Object[] args = new Object[x.type().parameterCount()];
400         Arrays.fill(args, v);
401         Object xval = invokeWithCatch(x, args);
402         Object yval = invokeWithCatch(y, args);
403         String msg = "ok";
404         if (!Objects.equals(xval, yval)) {
405             msg = ("applying "+x+" & "+y+" to "+v);
406         }
407         assertEquals(msg, xval, yval);
408     }
409     private static Object invokeWithCatch(MethodHandle mh, Object... args) throws Throwable {
410         try {
411             return mh.invokeWithArguments(args);
412         } catch (Throwable ex) {
413             System.out.println("threw: "+mh+Arrays.asList(args));
414             ex.printStackTrace();
415             return ex;
416         }
417     }

434     static int getProperty(String name, int dflt) {
435         String qual = LOOKUP.lookupClass().getName();
436         String prop = System.getProperty(qual+"."+name);
437         if (prop == null)  prop = System.getProperty(name);
438         if (prop == null)  return dflt;
439         return Integer.parseInt(prop);
440     }

561     private static void verbose(int level, Object a, Object... bcd) {
562         if (level > VERBOSITY)  return;
563         String m = a.toString();
564         if (bcd != null && bcd.length > 0) {
565             List<Object> l = new ArrayList<>(bcd.length);
566             for (Object x : bcd) {
567                 if (x instanceof Object[])  x = Arrays.asList((Object[])x);
568                 if (x instanceof int[])     x = Arrays.toString((int[])x);
569                 if (x instanceof long[])    x = Arrays.toString((long[])x);
570                 l.add(x);
571             }
572             m = m+Arrays.asList(bcd);
573         }
574         System.out.println(m);
575     }
