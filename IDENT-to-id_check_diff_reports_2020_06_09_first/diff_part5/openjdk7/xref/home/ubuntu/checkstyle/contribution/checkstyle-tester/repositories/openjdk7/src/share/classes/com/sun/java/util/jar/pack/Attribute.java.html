111     // Canonicalized lists of trivial attrs (Deprecated, etc.)
112     // are used by trimToSize, in order to reduce footprint
113     // of some common cases.  (Note that Code attributes are
114     // always zero size.)
115     public static List<Attribute> getCanonList(List<Attribute> al) {
116         synchronized (canonLists) {
117             List<Attribute> cl = canonLists.get(al);
118             if (cl == null) {
119                 cl = new ArrayList<>(al.size());
120                 cl.addAll(al);
121                 cl = Collections.unmodifiableList(cl);
122                 canonLists.put(al, cl);
123             }
124             return cl;
125         }

135                 attributes.put(key, a);

209     // Metadata.
210     //
211     // We define metadata using similar layouts
212     // for all five kinds of metadata attributes.
213     //
214     // Regular annotations are a counted list of [RSHNH[RUH(1)]][...]
215     //   pack.method.attribute.RuntimeVisibleAnnotations=[NH[(1)]][RSHNH[RUH(1)]][TB...]
216     //
217     // Parameter annotations are a counted list of regular annotations.
218     //   pack.method.attribute.RuntimeVisibleParameterAnnotations=[NH[(1)]][NH[(1)]][RSHNH[RUH(1)]][TB...]
219     //
220     // RuntimeInvisible annotations are defined similarly...
221     // Non-method annotations are defined similarly...
222     //
223     // Annotation are a simple tagged value [TB...]
224     //   pack.attribute.method.AnnotationDefault=[TB...]
225     //
226     static {
227         String mdLayouts[] = {
228             Attribute.normalizeLayoutString
229             (""
230              +"\n  # parameter_annotations :="
231              +"\n  [ NB[(1)] ]     # forward call to annotations"
232              ),
233             Attribute.normalizeLayoutString
234             (""
235              +"\n  # annotations :="
236              +"\n  [ NH[(1)] ]     # forward call to annotation"
237              +"\n  "
238              +"\n  # annotation :="
239              +"\n  [RSH"
240              +"\n    NH[RUH (1)]   # forward call to value"
241              +"\n    ]"
242              ),
243             Attribute.normalizeLayoutString
244             (""
245              +"\n  # value :="
246              +"\n  [TB # Callable 2 encodes one tagged value."
247              +"\n    (\\B,\\C,\\I,\\S,\\Z)[KIH]"
248              +"\n    (\\D)[KDH]"
249              +"\n    (\\F)[KFH]"
250              +"\n    (\\J)[KJH]"
251              +"\n    (\\c)[RSH]"
252              +"\n    (\\e)[RSH RUH]"
253              +"\n    (\\s)[RUH]"
254              +"\n    (\\[)[NH[(0)]] # backward self-call to value"
255              +"\n    (\\@)[RSH NH[RUH (0)]] # backward self-call to value"
256              +"\n    ()[] ]"
257              )
258         };
259         Map<Layout, Attribute> sd = standardDefs;
260         String defaultLayout     = mdLayouts[2];
261         String annotationsLayout = mdLayouts[1] + mdLayouts[2];
262         String paramsLayout      = mdLayouts[0] + annotationsLayout;
263         for (int ctype = 0; ctype < ATTR_CONTEXT_LIMIT; ctype++) {
264             if (ctype == ATTR_CONTEXT_CODE)  continue;
265             define(sd, ctype,
266                    "RuntimeVisibleAnnotations",   annotationsLayout);
267             define(sd, ctype,
268                    "RuntimeInvisibleAnnotations", annotationsLayout);
269             if (ctype == ATTR_CONTEXT_METHOD) {
270                 define(sd, ctype,
271                        "RuntimeVisibleParameterAnnotations",   paramsLayout);
272                 define(sd, ctype,
273                        "RuntimeInvisibleParameterAnnotations", paramsLayout);
274                 define(sd, ctype,
275                        "AnnotationDefault", defaultLayout);
276             }
277         }

308         public void trimToSize() {
309             if (attributes == null) {
310                 return;
311             }
312             if (attributes.isEmpty()) {
313                 attributes = null;
314                 return;
315             }
316             if (attributes instanceof ArrayList) {
317                 ArrayList<Attribute> al = (ArrayList<Attribute>)attributes;
318                 al.trimToSize();
319                 boolean allCanon = true;
320                 for (Attribute a : al) {
321                     if (!a.isCanonical()) {
322                         allCanon = false;
323                     }
324                     if (a.fixups != null) {
325                         assert(!a.isCanonical());
326                         a.fixups = Fixups.trimToSize(a.fixups);
327                     }
328                 }
329                 if (allCanon) {
330                     // Replace private writable attribute list
331                     // with only trivial entries by public unique
332                     // immutable attribute list with the same entries.
333                     attributes = getCanonList(al);
334                 }
335             }
336         }

338         public void addAttribute(Attribute a) {
339             if (attributes == null)
340                 attributes = new ArrayList<>(3);
341             else if (!(attributes instanceof ArrayList))
342                 attributes = new ArrayList<>(attributes);  // unfreeze it
343             attributes.add(a);
344         }

347             if (attributes == null)       return null;
348             if (!attributes.contains(a))  return null;
349             if (!(attributes instanceof ArrayList))
350                 attributes = new ArrayList<>(attributes);  // unfreeze it
351             attributes.remove(a);
352             return a;
353         }

430     static final byte EK_INT  = 1;     // B H I SH etc.
431     static final byte EK_BCI  = 2;     // PH POH etc.
432     static final byte EK_BCO  = 3;     // OH etc.
433     static final byte EK_FLAG = 4;     // FH etc.

436     static final byte EK_UN   = 7;     // TB(...)[...] etc.
437     static final byte EK_CASE = 8;     // (...)[...] etc.
438     static final byte EK_CALL = 9;     // (0), (1), etc.
439     static final byte EK_CBLE = 10;    // [...][...] etc.
440     static final byte EF_SIGN  = 1<<0;   // INT is signed
441     static final byte EF_DELTA = 1<<1;   // BCI/BCI value is diff'ed w/ previous

446     /** A "class" of attributes, characterized by a context-type, name
447      *  and format.  The formats are specified in a "little language".
448      */
449     public static
450     class Layout implements Comparable {
451         int ctype;       // attribute context type, e.g., ATTR_CONTEXT_CODE
452         String name;     // name of attribute
453         boolean hasRefs; // this kind of attr contains CP refs?
454         String layout;   // layout specification
455         int bandCount;   // total number of elems
456         Element[] elems; // tokenization of layout
457         Attribute canon; // canonical instance of this layout
458 
459         public int ctype() { return ctype; }
460         public String name() { return name; }
461         public String layout() { return layout; }
462         public Attribute canonicalInstance() { return canon; }
463 
464         public Entry getNameRef() {
465             return ConstantPool.getUtf8Entry(name());
466         }
467 
468         public boolean isEmpty() {
469             return layout.isEmpty();
470         }

472         public Layout(int ctype, String name, String layout) {
473             this.ctype = ctype;
474             this.name = name.intern();
475             this.layout = layout.intern();
476             assert(ctype < ATTR_CONTEXT_LIMIT);
477             boolean hasCallables = layout.startsWith("[");
478             try {
479                 if (!hasCallables) {
480                     this.elems = tokenizeLayout(this, -1, layout);
481                 } else {
482                     String[] bodies = splitBodies(layout);
483                     // Make the callables now, so they can be linked immediately.
484                     Element[] lelems = new Element[bodies.length];
485                     this.elems = lelems;
486                     for (int i = 0; i < lelems.length; i++) {
487                         Element ce = this.new Element();
488                         ce.kind = EK_CBLE;
489                         ce.removeBand();
490                         ce.bandIndex = NO_BAND_INDEX;
491                         ce.layout = bodies[i];
492                         lelems[i] = ce;
493                     }
494                     // Next fill them in.
495                     for (int i = 0; i < lelems.length; i++) {
496                         Element ce = lelems[i];
497                         ce.body = tokenizeLayout(this, i, bodies[i]);
498                     }
499                     //System.out.println(Arrays.asList(elems));
500                 }
501             } catch (StringIndexOutOfBoundsException ee) {
502                 // simplest way to catch syntax errors...
503                 throw new RuntimeException("Bad attribute layout: "+layout, ee);
504             }
505             // Some uses do not make a fresh one for each occurrence.
506             // For example, if layout == "", we only need one attr to share.
507             canon = new Attribute(this, noBytes);
508         }
509         private Layout() {}
510         static Layout makeKey(int ctype, String name, String layout) {
511             Layout def = new Layout();
512             def.ctype = ctype;
513             def.name = name.intern();
514             def.layout = layout.intern();
515             assert(ctype < ATTR_CONTEXT_LIMIT);
516             return def;
517         }
518         static Layout makeKey(int ctype, String name) {
519             return makeKey(ctype, name, "");
520         }

562         public
563         class Element {
564             String layout;   // spelling in the little language
565             byte flags;      // EF_SIGN, etc.
566             byte kind;       // EK_UINT, etc.
567             byte len;        // scalar length of element
568             byte refKind;    // CONSTANT_String, etc.
569             int bandIndex;   // which band does this element govern?
570             int value;       // extra parameter
571             Element[] body;  // extra data (for replications, unions, calls)
572 
573             boolean flagTest(byte mask) { return (flags & mask) != 0; }
574 
575             Element() {
576                 bandIndex = bandCount++;
577             }
578 
579             void removeBand() {
580                 --bandCount;
581                 assert(bandIndex == bandCount);
582                 bandIndex = NO_BAND_INDEX;
583             }
584 
585             public boolean hasBand() {
586                 return bandIndex >= 0;
587             }
588             public String toString() {
589                 String str = layout;
590                 // If -ea, print out more informative strings!
591                 assert((str = stringForDebug()) != null);
592                 return str;
593             }
594             private String stringForDebug() {
595                 Element[] lbody = this.body;
596                 switch (kind) {
597                 case EK_CALL:
598                     lbody = null;
599                     break;
600                 case EK_CASE:
601                     if (flagTest(EF_BACK))
602                         lbody = null;
603                     break;
604                 }
605                 return layout
606                     + (!hasBand()?"":"#"+bandIndex)
607                     + "<"+ (flags==0?"":""+flags)+kind+len
608                     + (refKind==0?"":""+refKind) + ">"
609                     + (value==0?"":"("+value+")")
610                     + (lbody==null?"": ""+Arrays.asList(lbody));
611             }
612         }

614         public boolean hasCallables() {
615             return (elems.length > 0 && elems[0].kind == EK_CBLE);
616         }
617         static private final Element[] noElems = {};
618         public Element[] getCallables() {
619             if (hasCallables()) {
620                 Element[] nelems = Arrays.copyOf(elems, elems.length);
621                 return nelems;
622             } else
623                 return noElems;  // no callables at all
624         }
625         public Element[] getEntryPoint() {
626             if (hasCallables())
627                 return elems[0].body;  // body of first callable
628             else {
629                 Element[] nelems = Arrays.copyOf(elems, elems.length);
630                 return nelems;  // no callables; whole body
631             }
632         }

634         /** Return a sequence of tokens from the given attribute bytes.
635          *  Sequence elements will be 1-1 correspondent with my layout tokens.
636          */
637         public void parse(Holder holder,
638                           byte[] bytes, int pos, int len, ValueStream out) {
639             int end = parseUsing(getEntryPoint(),
640                                  holder, bytes, pos, len, out);
641             if (end != pos + len)
642                 throw new InternalError("layout parsed "+(end-pos)+" out of "+len+" bytes");
643         }
644         /** Given a sequence of tokens, return the attribute bytes.
645          *  Sequence elements must be 1-1 correspondent with my layout tokens.
646          *  The returned object is a cookie for Fixups.finishRefs, which
647          *  must be used to harden any references into integer indexes.
648          */
649         public Object unparse(ValueStream in, ByteArrayOutputStream out) {
650             Object[] fixups = { null };
651             unparseUsing(getEntryPoint(), fixups, in, out);
652             return fixups[0]; // return ref-bearing cookie, if any
653         }

721     /** Remove any informal "pretty printing" from the layout string.
722      *  Removes blanks and control chars.
723      *  Removes '#' comments (to end of line).
724      *  Replaces '\c' by the decimal code of the character c.
725      *  Replaces '0xNNN' by the decimal code of the hex number NNN.

727     static public
728     String normalizeLayoutString(String layout) {
729         StringBuilder buf = new StringBuilder();
730         for (int i = 0, len = layout.length(); i < len; ) {
731             char ch = layout.charAt(i++);
732             if (ch <= ' ') {
733                 // Skip whitespace and control chars
734                 continue;
735             } else if (ch == '#') {
736                 // Skip to end of line.
737                 int end1 = layout.indexOf('\n', i);
738                 int end2 = layout.indexOf('\r', i);
739                 if (end1 < 0)  end1 = len;
740                 if (end2 < 0)  end2 = len;
741                 i = Math.min(end1, end2);
742             } else if (ch == '\\') {
743                 // Map a character reference to its decimal code.
744                 buf.append((int) layout.charAt(i++));
745             } else if (ch == '0' && layout.startsWith("0x", i-1)) {
746                 // Map a hex numeral to its decimal code.
747                 int start = i-1;
748                 int end = start+2;
749                 while (end < len) {
750                     int dig = layout.charAt(end);
751                     if ((dig >= '0' && dig <= '9') ||
752                         (dig >= 'a' && dig <= 'f'))
753                         ++end;
754                     else
755                         break;
756                 }
757                 if (end > start) {
758                     String num = layout.substring(start, end);
759                     buf.append(Integer.decode(num));
760                     i = end;
761                 } else {
762                     buf.append(ch);
763                 }
764             } else {
765                 buf.append(ch);
766             }
767         }
768         String result = buf.toString();
769         if (false && !result.equals(layout)) {
770             Utils.log.info("Normalizing layout string");
771             Utils.log.info("    From: "+layout);
772             Utils.log.info("    To:   "+result);
773         }
774         return result;
775     }

849         boolean prevBCI = false;

852             int body;
853             Layout.Element e = self.new Element();
854             byte kind;
855             //System.out.println("at "+i+": ..."+layout.substring(i));
856             // strip a prefix
857             switch (layout.charAt(i++)) {
858             /// layout_element: integral
859             case 'B': case 'H': case 'I': case 'V': // unsigned_int
860                 kind = EK_INT;

862                 i = tokenizeUInt(e, layout, i);
863                 break;
864             case 'S': // signed_int
865                 kind = EK_INT;
866                 --i; // reparse
867                 i = tokenizeSInt(e, layout, i);
868                 break;
869             case 'P': // bc_index
870                 kind = EK_BCI;
871                 if (layout.charAt(i++) == 'O') {
872                     // bc_index: 'PO' tokenizeUInt
873                     e.flags |= EF_DELTA;
874                     // must follow P or PO:
875                     if (!prevBCI)
876                         { i = -i; continue; } // fail
877                     i++; // move forward
878                 }
879                 --i; // reparse
880                 i = tokenizeUInt(e, layout, i);
881                 break;
882             case 'O': // bc_offset
883                 kind = EK_BCO;
884                 e.flags |= EF_DELTA;
885                 // must follow P or PO:
886                 if (!prevBCI)
887                     { i = -i; continue; } // fail
888                 i = tokenizeSInt(e, layout, i);
889                 break;
890             case 'F': // flag
891                 kind = EK_FLAG;
892                 i = tokenizeUInt(e, layout, i);
893                 break;
894             case 'N': // replication: 'N' uint '[' elem ... ']'
895                 kind = EK_REPL;

897                 if (layout.charAt(i++) != '[')
898                     { i = -i; continue; } // fail
899                 i = skipBody(layout, body = i);
900                 e.body = tokenizeLayout(self, curCble,
901                                         layout.substring(body, i++));
902                 break;
903             case 'T': // union: 'T' any_int union_case* '(' ')' '[' body ']'
904                 kind = EK_UN;
905                 i = tokenizeSInt(e, layout, i);
906                 List<Layout.Element> cases = new ArrayList<>();
907                 for (;;) {
908                     // Keep parsing cases until we hit the default case.
909                     if (layout.charAt(i++) != '(')
910                         { i = -i; break; } // fail
911                     int beg = i;
912                     i = layout.indexOf(')', i);
913                     String cstr = layout.substring(beg, i++);
914                     int cstrlen = cstr.length();
915                     if (layout.charAt(i++) != '[')
916                         { i = -i; break; } // fail
917                     // Check for duplication.
918                     if (layout.charAt(i) == ']')
919                         body = i;  // missing body, which is legal here
920                     else
921                         i = skipBody(layout, body = i);
922                     Layout.Element[] cbody
923                         = tokenizeLayout(self, curCble,
924                                          layout.substring(body, i++));
925                     if (cstrlen == 0) {
926                         Layout.Element ce = self.new Element();
927                         ce.body = cbody;
928                         ce.kind = EK_CASE;
929                         ce.removeBand();
930                         cases.add(ce);
931                         break;  // done with the whole union
932                     } else {
933                         // Parse a case string.
934                         boolean firstCaseNum = true;
935                         for (int cp = 0, endp;; cp = endp+1) {
936                             // Look for multiple case tags:
937                             endp = cstr.indexOf(',', cp);

939                             String cstr1 = cstr.substring(cp, endp);

941                                 cstr1 = "empty";  // will fail parse

944                             int dash = findCaseDash(cstr1, 0);

952                             }
953                             // Add a case for each value in value0..value1
954                             for (;; value0++) {
955                                 Layout.Element ce = self.new Element();
956                                 ce.body = cbody;  // all cases share one body
957                                 ce.kind = EK_CASE;
958                                 ce.removeBand();

961                                     ce.flags |= EF_BACK;
962                                 firstCaseNum = false;

966                             }
967                             if (endp == cstrlen) {
968                                 break;  // done with this case
969                             }
970                         }
971                     }

979                         // Duplicate tag.
980                         { i = -i; break; } // fail
981                     }
982                 }
983                 break;
984             case '(': // call: '(' '-'? digit+ ')'
985                 kind = EK_CALL;
986                 e.removeBand();

988                 String cstr = layout.substring(start+1, i++);
989                 int offset = Integer.parseInt(cstr);
990                 int target = curCble + offset;

992                     self.elems == null ||

994                     target >= self.elems.length)
995                     { i = -i; continue; } // fail

997                 assert(ce.kind == EK_CBLE);
998                 e.value = target;

1000                 // Is it a (recursive) backward call?
1001                 if (offset <= 0) {

1003                     e.flags  |= EF_BACK;

1005                 }
1006                 break;
1007             case 'K':  // reference_type: constant_ref
1008                 kind = EK_REF;
1009                 switch (layout.charAt(i++)) {
1010                 case 'I': e.refKind = CONSTANT_Integer; break;
1011                 case 'J': e.refKind = CONSTANT_Long; break;
1012                 case 'F': e.refKind = CONSTANT_Float; break;
1013                 case 'D': e.refKind = CONSTANT_Double; break;
1014                 case 'S': e.refKind = CONSTANT_String; break;
1015                 case 'Q': e.refKind = CONSTANT_Literal; break;

1017                 }
1018                 break;
1019             case 'R': // schema_ref
1020                 kind = EK_REF;
1021                 switch (layout.charAt(i++)) {
1022                 case 'C': e.refKind = CONSTANT_Class; break;
1023                 case 'S': e.refKind = CONSTANT_Signature; break;
1024                 case 'D': e.refKind = CONSTANT_NameandType; break;
1025                 case 'F': e.refKind = CONSTANT_Fieldref; break;
1026                 case 'M': e.refKind = CONSTANT_Methodref; break;
1027                 case 'I': e.refKind = CONSTANT_InterfaceMethodref; break;
1028 
1029                 case 'U': e.refKind = CONSTANT_Utf8; break; //utf8_ref
1030                 case 'Q': e.refKind = CONSTANT_All; break; //untyped_ref
1031 
1032                 default: { i = -i; continue; } // fail
1033                 }
1034                 break;
1035             default: { i = -i; continue; } // fail
1036             }
1037

1039             if (kind == EK_REF) {
1040                 // reference: reference_type -><- ( 'N' )? tokenizeUInt
1041                 if (layout.charAt(i++) == 'N') {

1051

1053             e.kind = kind;
1054             e.layout = layout.substring(start, i);
1055             col.add(e);
1056         }

1115     /** Find an occurrence of hyphen '-' between two numerals. */
1116     static //private
1117     int findCaseDash(String layout, int fromIndex) {
1118         if (fromIndex <= 0)  fromIndex = 1;  // minimum dash pos
1119         int lastDash = layout.length() - 2;  // maximum dash pos
1120         for (;;) {
1121             int dash = layout.indexOf('-', fromIndex);
1122             if (dash < 0 || dash > lastDash)  return -1;
1123             if (isDigit(layout.charAt(dash-1))) {
1124                 char afterDash = layout.charAt(dash+1);
1125                 if (afterDash == '-' && dash+2 < layout.length())
1126                     afterDash = layout.charAt(dash+2);
1127                 if (isDigit(afterDash)) {
1128                     // matched /[0-9]--?[0-9]/; return position of dash
1129                     return dash;
1130                 }
1131             }
1132             fromIndex = dash+1;
1133         }

1136     int parseIntBefore(String layout, int dash) {
1137         int end = dash;
1138         int beg = end;
1139         while (beg > 0 && isDigit(layout.charAt(beg-1))) {
1140             --beg;

1144         if (beg >= 1 && layout.charAt(beg-1) == '-')  --beg;

1154         while (end < limit && isDigit(layout.charAt(end))) {

1163         int dash = findCaseDash(layout, 0);
1164         if (dash < 0)  return layout;  // no dashes (the common case)
1165         StringBuilder result = new StringBuilder(layout.length() * 3);
1166         int sofar = 0;  // how far have we processed the layout?

1178                 result.append(",");  // close off i numeral
1179             }

1182         }

1184         return result.toString();

1193     // Parse attribute bytes, putting values into bands.  Returns new pos.
1194     // Used when reading a class file (local refs resolved with local cpMap).
1195     // Also used for ad hoc scanning.
1196     static
1197     int parseUsing(Layout.Element[] elems, Holder holder,
1198                    byte[] bytes, int pos, int len, ValueStream out) {
1199         int prevBCI = 0;
1200         int prevRBCI = 0;
1201         int end = pos + len;
1202         int[] buf = { 0 };  // for calls to parseInt, holds 2nd result
1203         for (int i = 0; i < elems.length; i++) {
1204             Layout.Element e = elems[i];
1205             int bandIndex = e.bandIndex;
1206             int value;
1207             int BCI, RBCI;
1208             switch (e.kind) {
1209             case EK_INT:
1210                 pos = parseInt(e, bytes, pos, buf);
1211                 value = buf[0];
1212                 out.putInt(bandIndex, value);
1213                 break;
1214             case EK_BCI:  // PH, POH
1215                 pos = parseInt(e, bytes, pos, buf);
1216                 BCI = buf[0];
1217                 RBCI = out.encodeBCI(BCI);
1218                 if (!e.flagTest(EF_DELTA)) {
1219                     // PH:  transmit R(bci), store bci
1220                     value = RBCI;
1221                 } else {
1222                     // POH:  transmit D(R(bci)), store bci
1223                     value = RBCI - prevRBCI;
1224                 }
1225                 prevBCI = BCI;
1226                 prevRBCI = RBCI;
1227                 out.putInt(bandIndex, value);
1228                 break;
1229             case EK_BCO:  // OH
1230                 assert(e.flagTest(EF_DELTA));
1231                 // OH:  transmit D(R(bci)), store D(bci)
1232                 pos = parseInt(e, bytes, pos, buf);
1233                 BCI = prevBCI + buf[0];
1234                 RBCI = out.encodeBCI(BCI);
1235                 value = RBCI - prevRBCI;
1236                 prevBCI = BCI;
1237                 prevRBCI = RBCI;
1238                 out.putInt(bandIndex, value);
1239                 break;
1240             case EK_FLAG:
1241                 pos = parseInt(e, bytes, pos, buf);
1242                 value = buf[0];
1243                 out.putInt(bandIndex, value);
1244                 break;
1245             case EK_REPL:
1246                 pos = parseInt(e, bytes, pos, buf);
1247                 value = buf[0];
1248                 out.putInt(bandIndex, value);
1249                 for (int j = 0; j < value; j++) {
1250                     pos = parseUsing(e.body, holder, bytes, pos, end-pos, out);
1251                 }
1252                 break;  // already transmitted the scalar value
1253             case EK_UN:
1254                 pos = parseInt(e, bytes, pos, buf);
1255                 value = buf[0];
1256                 out.putInt(bandIndex, value);
1257                 Layout.Element ce = matchCase(e, value);
1258                 pos = parseUsing(ce.body, holder, bytes, pos, end-pos, out);
1259 
1260                 break;  // already transmitted the scalar value
1261             case EK_CALL:
1262                 // Adjust band offset if it is a backward call.
1263                 assert(e.body.length == 1);
1264                 assert(e.body[0].kind == EK_CBLE);
1265                 if (e.flagTest(EF_BACK))
1266                     out.noteBackCall(e.value);
1267                 pos = parseUsing(e.body[0].body, holder, bytes, pos, end-pos, out);
1268                 break;  // no additional scalar value to transmit
1269             case EK_REF:
1270                 pos = parseInt(e, bytes, pos, buf);
1271                 int localRef = buf[0];
1272                 Entry globalRef;
1273                 if (localRef == 0) {
1274                     globalRef = null;  // N.B. global null reference is -1
1275                 } else {

1310     static private
1311     int parseInt(Layout.Element e, byte[] bytes, int pos, int[] buf) {
1312         int value = 0;
1313         int loBits = e.len * 8;
1314         // Read in big-endian order:
1315         for (int bitPos = loBits; (bitPos -= 8) >= 0; ) {
1316             value += (bytes[pos++] & 0xFF) << bitPos;
1317         }
1318         if (loBits < 32 && e.flagTest(EF_SIGN)) {
1319             // sign-extend subword value
1320             int hiBits = 32 - loBits;
1321             value = (value << hiBits) >> hiBits;
1322         }
1323         buf[0] = value;
1324         return pos;

1327     // Format attribute bytes, drawing values from bands.
1328     // Used when emptying attribute bands into a package model.
1329     // (At that point CP refs. are not yet assigned indexes.)
1330     static
1331     void unparseUsing(Layout.Element[] elems, Object[] fixups,
1332                       ValueStream in, ByteArrayOutputStream out) {
1333         int prevBCI = 0;
1334         int prevRBCI = 0;
1335         for (int i = 0; i < elems.length; i++) {
1336             Layout.Element e = elems[i];
1337             int bandIndex = e.bandIndex;
1338             int value;
1339             int BCI, RBCI;  // "RBCI" is R(BCI), BCI's coded representation
1340             switch (e.kind) {
1341             case EK_INT:
1342                 value = in.getInt(bandIndex);
1343                 unparseInt(e, value, out);
1344                 break;
1345             case EK_BCI:  // PH, POH
1346                 value = in.getInt(bandIndex);
1347                 if (!e.flagTest(EF_DELTA)) {
1348                     // PH:  transmit R(bci), store bci
1349                     RBCI = value;
1350                 } else {
1351                     // POH:  transmit D(R(bci)), store bci
1352                     RBCI = prevRBCI + value;
1353                 }
1354                 assert(prevBCI == in.decodeBCI(prevRBCI));
1355                 BCI = in.decodeBCI(RBCI);
1356                 unparseInt(e, BCI, out);
1357                 prevBCI = BCI;
1358                 prevRBCI = RBCI;
1359                 break;
1360             case EK_BCO:  // OH
1361                 value = in.getInt(bandIndex);
1362                 assert(e.flagTest(EF_DELTA));
1363                 // OH:  transmit D(R(bci)), store D(bci)
1364                 assert(prevBCI == in.decodeBCI(prevRBCI));
1365                 RBCI = prevRBCI + value;
1366                 BCI = in.decodeBCI(RBCI);
1367                 unparseInt(e, BCI - prevBCI, out);
1368                 prevBCI = BCI;
1369                 prevRBCI = RBCI;
1370                 break;
1371             case EK_FLAG:
1372                 value = in.getInt(bandIndex);
1373                 unparseInt(e, value, out);
1374                 break;
1375             case EK_REPL:
1376                 value = in.getInt(bandIndex);
1377                 unparseInt(e, value, out);
1378                 for (int j = 0; j < value; j++) {
1379                     unparseUsing(e.body, fixups, in, out);
1380                 }
1381                 break;
1382             case EK_UN:
1383                 value = in.getInt(bandIndex);
1384                 unparseInt(e, value, out);
1385                 Layout.Element ce = matchCase(e, value);
1386                 unparseUsing(ce.body, fixups, in, out);
1387                 break;
1388             case EK_CALL:
1389                 assert(e.body.length == 1);
1390                 assert(e.body[0].kind == EK_CBLE);
1391                 unparseUsing(e.body[0].body, fixups, in, out);
1392                 break;
1393             case EK_REF:
1394                 Entry globalRef = in.getRef(bandIndex);
1395                 int localRef;
1396                 if (globalRef != null) {
1397                     // It's a one-element array, really an lvalue.
1398                     fixups[0] = Fixups.add(fixups[0], null, out.size(),
1399                                            Fixups.U2_FORMAT, globalRef);
1400                     localRef = 0; // placeholder for fixups
1401                 } else {
1402                     localRef = 0; // fixed null value
1403                 }
1404                 unparseInt(e, localRef, out);
1405                 break;
1406             default: assert(false); continue;
1407             }
1408         }

1412     void unparseInt(Layout.Element e, int value, ByteArrayOutputStream out) {
1413         int loBits = e.len * 8;
1414         if (loBits == 0) {
1415             // It is not stored at all ('V' layout).
1416             return;
1417         }
1418         if (loBits < 32) {
1419             int hiBits = 32 - loBits;
1420             int codedValue;
1421             if (e.flagTest(EF_SIGN))
1422                 codedValue = (value << hiBits) >> hiBits;
1423             else
1424                 codedValue = (value << hiBits) >>> hiBits;
1425             if (codedValue != value)
1426                 throw new InternalError("cannot code in "+e.len+" bytes: "+value);
1427         }
1428         // Write in big-endian order:
1429         for (int bitPos = loBits; (bitPos -= 8) >= 0; ) {
1430             out.write((byte)(value >>> bitPos));
1431         }
