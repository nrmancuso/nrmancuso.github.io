32  package j2dbench;

34  public abstract class Test extends Option.Enable {
35      private DependentLink dependencies;
36  
37      public Test(Group parent, String nodeName, String description) {
38          super(parent, nodeName, description, false);
39      }
40  
41      public void addDependency(Modifier mod) {
42          addDependency(mod, null);
43      }
44  
45      public void addDependency(Modifier mod, Modifier.Filter filter) {
46          dependencies = DependentLink.add(dependencies, mod, filter);
47      }
48  
49      public void addDependencies(Group g, boolean recursive) {
50          addDependencies(g, recursive, null);
51      }
52  
53      public void addDependencies(Group g, boolean recursive,
54                                  Modifier.Filter filter)
55      {
56          if (g instanceof Modifier) {
57              addDependency((Modifier) g, filter);
58          }
59          for (Node n = g.getFirstChild(); n != null; n = n.getNext()) {
60              if (n instanceof Modifier) {
61                  addDependency((Modifier) n, filter);
62              } else if (recursive && n instanceof Group) {
63                  addDependencies((Group) n, recursive, filter);
64              }
65          }
66      }
67  
68      public void runTest(TestEnvironment env) {
69          if (!env.isStopped() && isEnabled()) {
70              dependencies.recurseAndRun(env, this);
71          }
72      }
73  
74      public void runOneTest(TestEnvironment env) {
75          if (!env.isStopped()) {
76              Result result = new Result(this);
77              env.erase();
78              Object ctx = initTest(env, result);
79              result.setModifiers(env.getModifiers());
80              try {
81                  runTestLoop(env, result, ctx);
82              } catch (Throwable t) {
83                  result.setError(t);
84              }
85              cleanupTest(env, ctx);
86              // Skip recording results if we were interrupted before
87              // anything interesting happened...
88              if (result.getError() != null || result.getNumRuns() != 0) {
89                  if (J2DBench.printresults.isEnabled()) {
90                      result.summarize();
91                  }
92                  env.record(result);
93              }
94              ctx = null;
95              result = null;
96              env.idle();  // Also done after this method returns...
97          }
98      }
99  
100     public abstract Object initTest(TestEnvironment env, Result result);
101     public abstract void runTest(Object context, int numReps);
102     public abstract void cleanupTest(TestEnvironment env, Object context);
103 
104     public void runTestLoop(TestEnvironment env, Result result, Object ctx) {
105         // Prime the pump
106         runTest(ctx, 1);
107 
108         // Determine the number of reps
109         int numReps = env.getRepCount();
110         if (numReps == 0) {
111             numReps = calibrate(env, ctx);
112         }
113         result.setReps(numReps);
114 
115         int numRuns = env.getRunCount();
116         for (int i = 0; i < numRuns; i++) {
117             if (env.idle()) {
118                 break;
119             }
120 
121             env.sync();
122             env.startTiming();
123             runTest(ctx, numReps);
124             env.sync();
125             env.stopTiming();
126             result.addTime(env.getTimeMillis());
127 
128             env.flushToScreen();
129         }
130     }
131 
132     public int calibrate(TestEnvironment env, Object ctx) {
133         long testTime = env.getTestTime();
134         int numReps = 0;
135         int totalReps = 0;
136 
137         // First do one at a time until we get to 1 second elapsed
138         // But, if we get to 1000 reps we'll start ramping up our
139         // reps per cycle and throwing sync() calls in to make sure
140         // we aren't spinning our gears queueing up graphics calls
141         env.idle();
142         long now = System.currentTimeMillis();
143         long startTime = now;
144         while (numReps < 1000 && now < startTime + 1000) {
145             runTest(ctx, 1);
146             numReps++;
147             now = System.currentTimeMillis();
148         }
149 
150         // Time to shift gears into an exponential number of tests
151         // sync() each time in case batching at a lower level is
152         // causing us to spin our gears
153         env.sync();
154         now = System.currentTimeMillis();
155         int reps = 250;
156         while (now < startTime + 1000) {
157             runTest(ctx, reps);
158             env.sync();
159             numReps += reps;
160             reps *= 2;
161             now = System.currentTimeMillis();
162         }
163 
164         // Now keep estimating how many reps it takes to hit our target
165         // time exactly, trying it out, and guessing again.
166         while (now < startTime + testTime) {
167             int estimate = (int) (numReps * testTime / (now - startTime));
168             if (estimate <= numReps) {
169                 estimate = numReps+1;
170             }
171             runTest(ctx, estimate - numReps);
172             numReps = estimate;
173             env.sync();
174             now = System.currentTimeMillis();
175         }
176 
177         // Now make one last estimate of how many reps it takes to
178         // hit the target exactly in case we overshot.
179         int estimate = (int) (numReps * testTime / (now - startTime));
180         if (estimate < 1) {
181             estimate = 1;
182         }
183         return estimate;
184     }

186     /*
187      * Finds a new width (w2) such that
188      *     (w-2) <= w2 <= w
189      *     and w2 is not a multiple of 3 (the X step size)
190      *     and GCD(w2, h) is as small as possible
191      */
192     static int prevw;
193     public static int adjustWidth(int w, int h) {
194         int bestv = w;
195         int bestw = w;
196         boolean verbose = (prevw != w && J2DBench.verbose.isEnabled());
197         for (int i = 0; i < 3; i++) {
198             int w2 = w-i;
199             int u = w2;
200             int v = h;
201             while (u > 0) {
202                 if (u < v) { int t = u; u = v; v = t; }
203                 u -= v;
204             }
205             if (verbose) {
206                 System.out.println("w = "+w2+", h = "+h+
207                                    ", w % 3 == "+(w2 % 3)+
208                                    ", gcd(w, h) = "+v);
209             }
210             if (v < bestv && (w2 % 3) != 0) {
211                 bestv = v;
212                 bestw = w2;
213             }
214         }
215         if (verbose) {
216             System.out.println("using "+bestw+" (gcd = "+bestv+")");
217             prevw = w;
218         }
219         return bestw;
220     }

222     public String toString() {
223         return "Test("+getTreeName()+")";
224     }

226     public static class DependentLink {
227         public static DependentLink add(DependentLink d, Modifier mod,
228                                         Modifier.Filter filter)
229         {
230             DependentLink dl = new DependentLink(mod, filter);
231             if (d == null) {
232                 d = dl;
233             } else {
234                 DependentLink last = d;
235                 while (last.next != null) {
236                     last = last.next;
237                 }
238                 last.next = dl;
239             }
240             return d;
241         }
242 
243         private DependentLink next;
244         private Modifier mod;
245         private Modifier.Filter filter;
246 
247         private DependentLink(Modifier mod, Modifier.Filter filter) {
248             this.mod = mod;
249             this.filter = filter;
250         }
251 
252         public Modifier getModifier() {
253             return mod;
254         }
255 
256         public Modifier.Filter getFilter() {
257             return filter;
258         }
259 
260         public DependentLink getNext() {
261             return next;
262         }
263 
264         public void recurseAndRun(TestEnvironment env, Test test) {
265             Modifier.Iterator iter = mod.getIterator(env);
266             while (iter.hasNext()) {
267                 Object val = iter.next();
268                 if (filter == null || filter.isCompatible(val)) {
269                     mod.modifyTest(env, val);
270                     if (next == null) {
271                         test.runOneTest(env);
272                         env.idle();  // One more time outside of runOneTest()
273                     } else {
274                         next.recurseAndRun(env, test);
275                     }
276                     mod.restoreTest(env, val);
277                 }
278             }
279         }
280     }
