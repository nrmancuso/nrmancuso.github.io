25  package java.net;
26  
27  import java.io.IOException;
28  import java.io.FileDescriptor;
29  import sun.misc.SharedSecrets;
30  import sun.misc.JavaIOFileDescriptorAccess;
31  
32  /**
33   * This class defines the plain DatagramSocketImpl that is used on
34   * Windows platforms greater than or equal to Windows Vista. These
35   * platforms have a dual layer TCP/IP stack and can handle both IPv4
36   * and IPV6 through a single file descriptor.
37   * <p>
38   * Note: Multicasting on a dual layer TCP/IP stack is always done with
39   * TwoStacksPlainDatagramSocketImpl. This is to overcome the lack
40   * of behavior defined for multicasting over a dual layer socket by the RFC.
41   *
42   * @author Chris Hegarty
43   */
44  
45  class DualStackPlainDatagramSocketImpl extends AbstractPlainDatagramSocketImpl
46  {
47      static JavaIOFileDescriptorAccess fdAccess = SharedSecrets.getJavaIOFileDescriptorAccess();
48  
49      protected void datagramSocketCreate() throws SocketException {
50          if (fd == null)
51              throw new SocketException("Socket closed");
52  
53          int newfd = socketCreate(false /* v6Only */);
54  
55          fdAccess.set(fd, newfd);
56      }
57  
58      protected synchronized void bind0(int lport, InetAddress laddr)
59          throws SocketException {
60          int nativefd = checkAndReturnNativeFD();
61  
62          if (laddr == null)
63              throw new NullPointerException("argument address");
64  
65          socketBind(nativefd, laddr, lport);
66          if (lport == 0) {
67              localPort = socketLocalPort(nativefd);
68          } else {
69              localPort = lport;
70          }
71      }
72  
73      protected synchronized int peek(InetAddress address) throws IOException {
74          int nativefd = checkAndReturnNativeFD();
75  
76          if (address == null)
77              throw new NullPointerException("Null address in peek()");
78  
79          // Use peekData()
80          DatagramPacket peekPacket = new DatagramPacket(new byte[1], 1);
81          int peekPort = peekData(peekPacket);
82          address = peekPacket.getAddress();
83          return peekPort;
84      }
85  
86      protected synchronized int peekData(DatagramPacket p) throws IOException {
87          int nativefd = checkAndReturnNativeFD();
88  
89          if (p == null)
90              throw new NullPointerException("packet");
91          if (p.getData() == null)
92              throw new NullPointerException("packet buffer");
93  
94          return socketReceiveOrPeekData(nativefd, p, timeout, connected, true /*peek*/);
95      }
96  
97      protected synchronized void receive0(DatagramPacket p) throws IOException {
98          int nativefd = checkAndReturnNativeFD();
99  
100         if (p == null)
101             throw new NullPointerException("packet");
102         if (p.getData() == null)
103             throw new NullPointerException("packet buffer");
104 
105         socketReceiveOrPeekData(nativefd, p, timeout, connected, false /*receive*/);
106     }
107 
108     protected void send(DatagramPacket p) throws IOException {
109         int nativefd = checkAndReturnNativeFD();
110 
111         if (p == null)
112             throw new NullPointerException("null packet");
113 
114         if (p.getAddress() == null ||p.getData() ==null)
115             throw new NullPointerException("null address || null buffer");
116 
117         socketSend(nativefd, p.getData(), p.getOffset(), p.getLength(),
118                    p.getAddress(), p.getPort(), connected);
119     }
120 
121     protected void connect0(InetAddress address, int port) throws SocketException {
122         int nativefd = checkAndReturnNativeFD();
123 
124         if (address == null)
125             throw new NullPointerException("address");
126 
127         socketConnect(nativefd, address, port);
128     }
129 
130     protected void disconnect0(int family /*unused*/) {
131         if (fd == null || !fd.valid())
132             return;   // disconnect doesn't throw any exceptions
133 
134         socketDisconnect(fdAccess.get(fd));
135     }
136 
137     protected void datagramSocketClose() {
138         if (fd == null || !fd.valid())
139             return;   // close doesn't throw any exceptions
140 
141         socketClose(fdAccess.get(fd));
142         fdAccess.set(fd, -1);
143     }
144 
145     protected void socketSetOption(int opt, Object val) throws SocketException {
146         int nativefd = checkAndReturnNativeFD();
147 
148         int optionValue = 0;
149 
150         switch(opt) {
151             case IP_TOS :
152             case SO_RCVBUF :
153             case SO_SNDBUF :
154                 optionValue = ((Integer)val).intValue();
155                 break;
156             case SO_REUSEADDR :
157             case SO_BROADCAST :
158                 optionValue = ((Boolean)val).booleanValue() ? 1 : 0;
159                 break;
160             default: /* shouldn't get here */
161                 throw new SocketException("Option not supported");
162         }
163 
164         socketSetIntOption(nativefd, opt, optionValue);
165     }
166 
167     protected Object socketGetOption(int opt) throws SocketException {
168         int nativefd = checkAndReturnNativeFD();
169 
170          // SO_BINDADDR is not a socket option.
171         if (opt == SO_BINDADDR) {
172             return socketLocalAddress(nativefd);
173         }
174 
175         int value = socketGetIntOption(nativefd, opt);
176         Object returnValue = null;
177 
178         switch (opt) {
179             case SO_REUSEADDR :
180             case SO_BROADCAST :
181                 returnValue =  (value == 0) ? Boolean.FALSE : Boolean.TRUE;
182                 break;
183             case IP_TOS :
184             case SO_RCVBUF :
185             case SO_SNDBUF :
186                 returnValue = new Integer(value);
187                 break;
188             default: /* shouldn't get here */
189                 throw new SocketException("Option not supported");
190         }
191 
192         return returnValue;
193     }
194 
195     /* Multicast specific methods.
196      * Multicasting on a dual layer TCP/IP stack is always done with
197      * TwoStacksPlainDatagramSocketImpl. This is to overcome the lack
198      * of behavior defined for multicasting over a dual layer socket by the RFC.
199      */
200     protected void join(InetAddress inetaddr, NetworkInterface netIf)
201         throws IOException {
202         throw new IOException("Method not implemented!");
203     }
204 
205     protected void leave(InetAddress inetaddr, NetworkInterface netIf)
206         throws IOException {
207         throw new IOException("Method not implemented!");
208     }
209 
210     protected void setTimeToLive(int ttl) throws IOException {
211         throw new IOException("Method not implemented!");
212     }
213 
214     protected int getTimeToLive() throws IOException {
215         throw new IOException("Method not implemented!");
216     }
217 
218 
219     protected void setTTL(byte ttl) throws IOException {
220         throw new IOException("Method not implemented!");
221     }
222 
223     protected byte getTTL() throws IOException {
224         throw new IOException("Method not implemented!");
225     }
226     /* END Multicast specific methods */
227 
228     private int checkAndReturnNativeFD() throws SocketException {
229         if (fd == null || !fd.valid())
230             throw new SocketException("Socket closed");
231 
232         return fdAccess.get(fd);
233     }
234 
235     /* Native methods */
236 
237     private static native void initIDs();
238 
239     private static native int socketCreate(boolean v6Only);
240 
241     private static native void socketBind(int fd, InetAddress localAddress, int localport)
242         throws SocketException;
243 
244     private static native void socketConnect(int fd, InetAddress address, int port)
245         throws SocketException;
246 
247     private static native void socketDisconnect(int fd);
248 
249     private static native void socketClose(int fd);
250 
251     private static native int socketLocalPort(int fd) throws SocketException;
252 
253     private static native Object socketLocalAddress(int fd) throws SocketException;
254 
255     private static native int socketReceiveOrPeekData(int fd, DatagramPacket packet,
256         int timeout, boolean connected, boolean peek) throws IOException;
257 
258     private static native void socketSend(int fd, byte[] data, int offset, int length,
259         InetAddress address, int port, boolean connected) throws IOException;
260 
261     private static native void socketSetIntOption(int fd, int cmd,
262         int optionValue) throws SocketException;
263 
264     private static native int socketGetIntOption(int fd, int cmd) throws SocketException;
265 }
