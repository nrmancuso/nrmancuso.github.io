177     /**
178      * Creates a default singleton ObjectName for a given class.
179      * @param clazz The interface class of the MBean for which we want to obtain
180      *        a default singleton name, or its implementation class.
181      *        Give one or the other depending on what you wish to see in
182      *        the value of the key {@code type=}.
183      * @return A default singleton name for a singleton MBean class.
184      * @throws IllegalArgumentException if the name can't be created
185      *         for some unfathomable reason (e.g. an unexpected
186      *         exception was raised).
187      **/
188     public final static ObjectName makeSingletonName(Class clazz) {
189         try {
190             final Package p = clazz.getPackage();
191             final String packageName = (p==null)?null:p.getName();
192             final String className   = clazz.getSimpleName();
193             final String domain;
194             if (packageName == null || packageName.length()==0) {
195                 // We use a reference to ScanDirAgent.class to ease
196                 // to keep track of possible class renaming.
197                 domain = ScanDirAgent.class.getSimpleName();
198             } else {
199                 domain = packageName;
200             }
201             final ObjectName name = new ObjectName(domain,"type",className);
202             return name;
203         } catch (Exception x) {
204             final IllegalArgumentException iae =
205                     new IllegalArgumentException(String.valueOf(clazz),x);
206             throw iae;
207         }
208     }

336     // Applies a configuration.
337     // throws IllegalStateException if lock can't be acquired.
338     // Unregisters all existing directory scanners, the create and registers
339     // new directory scanners according to the given config.
340     // Then pushes the log config to the result log manager.
341     //
342     private void applyConfiguration(ScanManagerConfig bean)
343         throws IOException, JMException {
344         if (bean == null) return;
345         if (!sequencer.tryAcquire()) {
346             throw new IllegalStateException("Can't acquire lock");
347         }
348         try {
349             unregisterScanners();
350             final DirectoryScannerConfig[] scans = bean.getScanList();
351             if (scans == null) return;
352             for (DirectoryScannerConfig scan : scans) {
353                 addDirectoryScanner(scan);
354             }
355             log.setConfig(bean.getInitialResultLogConfig());
356         } finally {
357             sequencer.release();
358         }
359     }

387     // Creates and registers a new directory scanner.
388     // Called by applyConfiguration.
389     // throws IllegalStateException if state is not STOPPED or COMPLETED
390     // (you cannot change the config while scanning is scheduled or running).
391     //
392     private DirectoryScannerMXBean addDirectoryScanner(
393                 DirectoryScannerConfig bean)
394         throws JMException {
395         try {
396             final DirectoryScannerMXBean scanner;
397             final ObjectName scanName;
398             synchronized (this) {
399                 if (state != STOPPED && state != COMPLETED)
400                    throw new IllegalStateException(state.toString());
401                 scanner = createDirectoryScanner(bean);
402                 scanName = makeDirectoryScannerName(bean.getName());
403             }
404             LOG.fine("server: "+mbeanServer);
405             LOG.fine("scanner: "+scanner);
406             LOG.fine("scanName: "+scanName);
407             final ObjectInstance moi =
408                 mbeanServer.registerMBean(scanner,scanName);
409             final ObjectName moiName = moi.getObjectName();
410             final DirectoryScannerMXBean proxy =
411                 JMX.newMXBeanProxy(mbeanServer,moiName,
412                 DirectoryScannerMXBean.class,true);
413             scanmap.put(moiName,proxy);
414             return proxy;
415         } catch (RuntimeException x) {
416             final String msg = "Operation failed: "+x;
417             if (LOG.isLoggable(Level.FINEST))
418                 LOG.log(Level.FINEST,msg,x);
419             else LOG.fine(msg);
420             throw x;
421         } catch (JMException x) {
422             final String msg = "Operation failed: "+x;
423             if (LOG.isLoggable(Level.FINEST))
424                 LOG.log(Level.FINEST,msg,x);
425             else LOG.fine(msg);
426             throw x;
427         }
428     }

458     /**
459      * For each operation, this map stores a list of states from
460      * which the corresponding operation can be legally called.
461      * For instance, it is legal to call "stop" regardless of the
462      * application state. However, "schedule" can be called only if
463      * the application state is STOPPED, etc...
464      **/
465     private final static Map<String,EnumSet<ScanState>> allowedStates;
466     static {
467         allowedStates = newHashMap();
468         // You can always call stop
469         allowedStates.put("stop",EnumSet.allOf(ScanState.class));
470 
471         // You can only call closed when stopped
472         allowedStates.put("close",EnumSet.of(STOPPED,COMPLETED,CLOSED));
473 
474         // You can call schedule only when the current task is
475         // completed or stopped.
476         allowedStates.put("schedule",EnumSet.of(STOPPED,COMPLETED));
477 
478         // switch reserved for background task: goes from SCHEDULED to
479         //    RUNNING when it enters the run() method.
480         allowedStates.put("scan-running",EnumSet.of(SCHEDULED));
481 
482         // switch reserved for background task: goes from RUNNING to
483         //    SCHEDULED when it has completed but needs to reschedule
484         //    itself for specified interval.
485         allowedStates.put("scan-scheduled",EnumSet.of(RUNNING));
486 
487         // switch reserved for background task:
488         //     goes from RUNNING to COMPLETED upon successful completion
489         allowedStates.put("scan-done",EnumSet.of(RUNNING));
490     }

499     /**
500      * Enqueue a state changed notification for the given states.
501      **/
502     private void queueStateChangedNotification(
503                     long sequence,
504                     long time,
505                     ScanState old,
506                     ScanState current) {
507         final AttributeChangeNotification n =
508                 new AttributeChangeNotification(SCAN_MANAGER_NAME,sequence,time,
509                 "ScanManager State changed to "+current,"State",
510                 ScanState.class.getName(),old.toString(),current.toString());
511         // Queue the notification. We have created an unlimited queue, so
512         // this method should always succeed.
513         try {
514             if (!pendingNotifs.offer(n,2,TimeUnit.SECONDS)) {
515                 LOG.fine("Can't queue Notification: "+n);
516             }
517         } catch (InterruptedException x) {
518                 LOG.fine("Can't queue Notification: "+x);
519         }
520     }
521 
522     /**
523      * Send all notifications present in the queue.
524      **/
525     private void sendQueuedNotifications() {
526         Notification n;
527         while ((n = pendingNotifs.poll()) != null) {
528             broadcaster.sendNotification(n);
529         }
530     }

532     /**
533      * Checks that the current state is allowed for the given operation,
534      * and if so, switch its value to the new desired state.
535      * This operation also enqueue the appropriate state changed
536      * notification.
537      **/
538     private ScanState switchState(ScanState desired,String forOperation) {
539         return switchState(desired,allowedStates.get(forOperation));
540     }

542     /**
543      * Checks that the current state is one of the allowed states,
544      * and if so, switch its value to the new desired state.
545      * This operation also enqueue the appropriate state changed
546      * notification.
547      **/
548     private ScanState switchState(ScanState desired,EnumSet<ScanState> allowed) {
549         final ScanState old;
550         final long timestamp;
551         final long sequence;
552         synchronized(this) {
553             old = state;
554             if (!allowed.contains(state))
555                throw new IllegalStateException(state.toString());
556             state = desired;
557             timestamp = System.currentTimeMillis();
558             sequence  = getNextSeqNumber();
559         }
560         LOG.fine("switched state: "+old+" -> "+desired);
561         if (old != desired)
562             queueStateChangedNotification(sequence,timestamp,old,desired);
563         return old;
564     }

569     // (possibly right away if delay=0) by a Timer thread.

572     // The timer used by this object. Lazzy evaluation. Cleaned in
573     // postDeregister()
574     //
575     private Timer timer = null;
576 
577     // See ScanManagerMXBean
578     public void schedule(long delay, long interval) {
579         if (!sequencer.tryAcquire()) {
580             throw new IllegalStateException("Can't acquire lock");
581         }
582         try {
583             LOG.fine("scheduling new task: state="+state);
584             final ScanState old = switchState(SCHEDULED,"schedule");
585             final boolean scheduled =
586                 scheduleSession(new SessionTask(interval),delay);
587             if (scheduled)
588                 LOG.fine("new task scheduled: state="+state);
589         } finally {
590             sequencer.release();
591         }
592         sendQueuedNotifications();
593     }
594 
595     // Schedule a SessionTask. The session task may reschedule
596     // a new identical task when it eventually ends.
597     // We use this logic so that the 'interval' time is measured
598     // starting at the end of the task that finishes, rather than
599     // at its beginning. Therefore if a repeated task takes x ms,
600     // it will be repeated every x+interval ms.
601     //
602     private synchronized boolean scheduleSession(SessionTask task, long delay) {
603         if (state == STOPPED) return false;
604         if (timer == null) timer = new Timer("ScanManager");
605         tasklist.add(task);
606         timer.schedule(task,delay);
607         return true;
608     }

619     // ---------------------------------------------------------------
620     // Methods used to implement stop() -  stop() is asynchronous,
621     // and needs to notify any running background task that it needs
622     // to stop. It also needs to prevent scheduled task from being
623     // run.
624     // ---------------------------------------------------------------
625 
626     // See ScanManagerMXBean
627     public void stop() {
628         if (!sequencer.tryAcquire())
629             throw new IllegalStateException("Can't acquire lock");
630         int errcount = 0;
631         final StringBuilder b = new StringBuilder();
632 
633         try {
634             switchState(STOPPED,"stop");
635 
636             errcount += cancelSessionTasks(b);
637             errcount += stopDirectoryScanners(b);
638         } finally {
639             sequencer.release();
640         }
641 
642         sendQueuedNotifications();
643         if (errcount > 0) {
644             b.insert(0,"stop partially failed with "+errcount+" error(s):");
645             throw new RuntimeException(b.toString());
646         }
647     }

649     // See ScanManagerMXBean
650     public void close() {
651         switchState(CLOSED,"close");
652         sendQueuedNotifications();
653     }

655     // Appends exception to a StringBuilder message.
656     //
657     private void append(StringBuilder b,String prefix,Throwable t) {
658         final String first = (prefix==null)?"\n":"\n"+prefix;
659         b.append(first).append(String.valueOf(t));
660         Throwable cause = t;
661         while ((cause = cause.getCause())!=null) {
662             b.append(first).append("Caused by:").append(first);
663             b.append('\t').append(String.valueOf(cause));
664         }
665     }

667     // Cancels all scheduled session tasks
668     //
669     private int cancelSessionTasks(StringBuilder b) {
670         int errcount = 0;
671         // Stops scheduled tasks if any...
672         //
673         for (SessionTask task : tasklist) {
674             try {
675                 task.cancel();
676                 tasklist.remove(task);
677             } catch (Exception ex) {
678                 errcount++;
679                 append(b,"\t",ex);
680             }
681         }
682         return errcount;
683     }

685     // Stops all DirectoryScanners configured for this object.
686     //
687     private int stopDirectoryScanners(StringBuilder b) {
688         int errcount = 0;
689         // Stops directory scanners if any...
690         //
691         for (DirectoryScannerMXBean s : scanmap.values()) {
692             try {
693                 s.stop();
694             } catch (Exception ex) {
695                 errcount++;
696                 append(b,"\t",ex);
697             }
698         }
699         return errcount;
700     }

708     private void scanAllDirectories()
709         throws IOException, InstanceNotFoundException {
710 
711         int errcount = 0;
712         final StringBuilder b = new StringBuilder();
713         for (ObjectName key : scanmap.keySet()) {
714             final DirectoryScannerMXBean s = scanmap.get(key);
715             try {
716                 if (state == STOPPED) return;
717                 s.scan();
718             } catch (Exception ex) {
719                 LOG.log(Level.FINE,key + " failed to scan: "+ex,ex);
720                 errcount++;
721                 append(b,"\t",ex);
722             }
723         }
724         if (errcount > 0) {
725             b.insert(0,"scan partially performed with "+errcount+" error(s):");
726             throw new RuntimeException(b.toString());
727         }
728     }

730     // List of scheduled session task. Needed by stop() to cancel
731     // scheduled sessions. There's usually at most 1 session in
732     // this list (unless there's a bug somewhere ;-))
733     //
734     private final ConcurrentLinkedQueue<SessionTask> tasklist =
735             new ConcurrentLinkedQueue<SessionTask>();
736 
737     // Used to give a unique id to session task - useful for
738     // debugging.
739     //
740     private volatile static long taskcount = 0;
741 
742     /**
743      * A session task will be scheduled to run in background in a
744      * timer thread. There can be at most one session task running
745      * at a given time (this is ensured by using a timer - which is
746      * a single threaded object).
747      *
748      * If the session needs to be repeated, it will reschedule an
749      * identical session when it finishes to run. This ensure that
750      * two session runs are separated by the given interval time.
751      *
752      **/
753     private class SessionTask extends TimerTask {
754 
755         /**
756          * Delay after which the next iteration of this task will
757          * start. This delay is measured  starting at the end of
758          * the previous iteration.
759          **/
760         final long delayBeforeNext;
761 
762         /**
763          * A unique id for this task.
764          **/
765         final long taskid;
766 
767         /**
768          * Whether it's been cancelled by stop()
769          **/
770         volatile boolean cancelled=false;
771 
772         /**
773          * create a new SessionTask.
774          **/
775         SessionTask(long scheduleNext) {
776             delayBeforeNext = scheduleNext;
777             taskid = taskcount++;
778         }
779 
780         /**
781          * When run() begins, the state is switched to RUNNING.
782          * When run() ends then:
783          *      If the task is repeated, the state will be switched
784          *      to SCHEDULED (because a new task was scheduled).
785          *      Otherwise the state will be switched to either
786          *      STOPPED (if it was stopped before it could complete)
787          *      or COMPLETED (if it completed gracefully)
788          * This method is used to switch to the desired state and
789          * send the appropriate notifications.
790          * When entering the method, we check whether the state is
791          * STOPPED. If so, we return false - and the SessionTask will
792          * stop. Otherwise, we switch the state to the desired value.
793          **/
794         private boolean notifyStateChange(ScanState newState,String condition) {
795             synchronized (ScanManager.this) {
796                 if (state == STOPPED || state == CLOSED) return false;
797                 switchState(newState,condition);
798             }
799             sendQueuedNotifications();
800             return true;
801         }
802 
803         // Cancels this task.
804         public boolean cancel() {
805             cancelled=true;
806             return super.cancel();
807         }
808 
809         /**
810          * Invoke all directories scanners in sequence. At each
811          * step, checks to see whether the task should stop.
812          **/
813         private boolean execute() {
814             final String tag = "Scheduled session["+taskid+"]";
815             try {
816                 if (cancelled) {
817                     LOG.finer(tag+" cancelled: done");
818                     return false;
819                 }
820                 if (!notifyStateChange(RUNNING,"scan-running")) {
821                     LOG.finer(tag+" stopped: done");
822                     return false;
823                 }
824                 scanAllDirectories();
825             } catch (Exception x) {
826                 if (LOG.isLoggable(Level.FINEST)) {
827                     LOG.log(Level.FINEST,
828                             tag+" failed to scan: "+x,x);
829                 } else if (LOG.isLoggable(Level.FINE)) {
830                     LOG.fine(tag+" failed to scan: "+x);
831                 }
832             }
833             return true;
834         }
835 
836         /**
837          * Schedule an identical task for next iteration.
838          **/
839         private boolean scheduleNext() {
840             final String tag = "Scheduled session["+taskid+"]";
841 
842             // We need now to reschedule a new task for after 'delayBeforeNext' ms.
843             try {
844                 LOG.finer(tag+": scheduling next session for "+ delayBeforeNext + "ms");
845                 if (cancelled || !notifyStateChange(SCHEDULED,"scan-scheduled")) {
846                     LOG.finer(tag+" stopped: do not reschedule");
847                     return false;
848                 }
849                 final SessionTask nextTask = new SessionTask(delayBeforeNext);
850                 if (!scheduleSession(nextTask,delayBeforeNext)) return false;
851                 LOG.finer(tag+": next session successfully scheduled");
852             } catch (Exception x) {
853                 if (LOG.isLoggable(Level.FINEST)) {
854                     LOG.log(Level.FINEST,tag+
855                             " failed to schedule next session: "+x,x);
856                 } else if (LOG.isLoggable(Level.FINE)) {
857                     LOG.fine(tag+" failed to schedule next session: "+x);
858                 }
859             }
860             return true;
861         }
862 
863 
864         /**
865          * The run method:
866          * executes scanning logic, the schedule next iteration if needed.
867          **/
868         public void run() {
869             final String tag = "Scheduled session["+taskid+"]";
870             LOG.entering(SessionTask.class.getName(),"run");
871             LOG.finer(tag+" starting...");
872             try {
873                 if (execute()==false) return;
874 
875                 LOG.finer(tag+" terminating - state is "+state+
876                     ((delayBeforeNext >0)?(" next session is due in "+delayBeforeNext+" ms."):
877                         " no additional session scheduled"));
878 
879                 // if delayBeforeNext <= 0 we are done, either because the session was
880                 // stopped or because it successfully completed.
881                 if (delayBeforeNext <= 0) {
882                     if (!notifyStateChange(COMPLETED,"scan-done"))
883                         LOG.finer(tag+" stopped: done");
884                     else
885                         LOG.finer(tag+" completed: done");
886                     return;
887                 }
888 
889                 // we need to reschedule a new session for 'delayBeforeNext' ms.
890                 scheduleNext();
891 
892             } finally {
893                 tasklist.remove(this);
894                 LOG.finer(tag+" finished...");
895                 LOG.exiting(SessionTask.class.getName(),"run");
896             }
897         }
898     }

965     /**
966      * Allows the MBean to perform any operations it needs before being
967      * registered in the MBean server. If the name of the MBean is not
968      * specified, the MBean can provide a name for its registration. If
969      * any exception is raised, the MBean will not be registered in the
970      * MBean server.
971      * <p>In this implementation, we check that the provided name is
972      * either {@code null} or equals to {@link #SCAN_MANAGER_NAME}. If it
973      * isn't then we throw an IllegalArgumentException, otherwise we return
974      * {@link #SCAN_MANAGER_NAME}.</p>
975      * <p>This ensures that there will be a single instance of ScanManager
976      * registered in a given MBeanServer, and that it will always be
977      * registered with the singleton's {@link #SCAN_MANAGER_NAME}.</p>
978      * <p>We do not need to check whether an MBean by that name is
979      *    already registered because the MBeanServer will perform
980      *    this check just after having called preRegister().</p>
981      * @param server The MBean server in which the MBean will be registered.
982      * @param name The object name of the MBean. This name is null if the
983      * name parameter to one of the createMBean or registerMBean methods in
984      * the MBeanServer interface is null. In that case, this method must
985      * return a non-null ObjectName for the new MBean.
986      * @return The name under which the MBean is to be registered. This value
987      * must not be null. If the name parameter is not null, it will usually
988      * but not necessarily be the returned value.
989      * @throws Exception This exception will be caught by the MBean server and
990      * re-thrown as an MBeanRegistrationException.
991      */
992     public ObjectName preRegister(MBeanServer server, ObjectName name) throws Exception {
993         if (name != null) {
994             if (!SCAN_MANAGER_NAME.equals(name))
995                 throw new IllegalArgumentException(String.valueOf(name));
996         }
997         mbeanServer = server;
998         return SCAN_MANAGER_NAME;
999     }

1010     /**
1011      * Allows the MBean to perform any operations needed after having
1012      * been registered in the MBean server or after the registration has
1013      * failed.
1014      * <p>
1015      * If registration was not successful, the method returns immediately.
1016      * <p>
1017      * If registration is successful, register the {@link ResultLogManager}
1018      * and default {@link ScanDirConfigMXBean}. If registering these
1019      * MBean fails, the {@code ScanManager} state will be switched to
1020      * {@link #close CLOSED}, and postRegister ends there.
1021      * </p>
1022      * <p>Otherwise the {@code ScanManager} will ask the
1023      * {@link ScanDirConfigMXBean} to load its configuration.
1024      * If it succeeds, the configuration will be {@link
1025      * #applyConfiguration applied}. Otherwise, the method simply returns,
1026      * assuming that the user will later create/update a configuration and
1027      * apply it.
1028      * @param registrationDone Indicates whether or not the MBean has been
1029      * successfully registered in the MBean server. The value false means
1030      * that the registration has failed.
1031      */
1032     public void postRegister(Boolean registrationDone) {
1033         if (!registrationDone) return;
1034         Exception test=null;
1035         try {
1036             mbeanServer.registerMBean(log,
1037                     ResultLogManager.RESULT_LOG_MANAGER_NAME);
1038             final String defconf = getDefaultConfigurationFileName();
1039             final String conf = System.getProperty("scandir.config.file",defconf);
1040             final String confname = ScanDirConfig.guessConfigName(conf,defconf);
1041             final ObjectName defaultProfileName =
1042                     makeMBeanName(ScanDirConfigMXBean.class,confname);
1043             if (!mbeanServer.isRegistered(defaultProfileName))
1044                 mbeanServer.registerMBean(new ScanDirConfig(conf),
1045                         defaultProfileName);
1046             config = JMX.newMXBeanProxy(mbeanServer,defaultProfileName,
1047                     ScanDirConfigMXBean.class,true);
1048             configmap.put(defaultProfileName,config);
1049         } catch (Exception x) {
1050             LOG.config("Failed to populate MBeanServer: "+x);
1051             close();
1052             return;
1053         }
1054         try {
1055             config.load();
1056         } catch (Exception x) {
1057             LOG.finest("No config to load: "+x);
1058             test = x;
1059         }
1060         if (test == null) {
1061             try {
1062                 applyConfiguration(config.getConfiguration());
1063             } catch (Exception x) {
1064                 if (LOG.isLoggable(Level.FINEST))
1065                     LOG.log(Level.FINEST,"Failed to apply config: "+x,x);
1066                 LOG.config("Failed to apply config: "+x);
1067             }
1068         }
1069     }

1099      * Allows the MBean to perform any operations it needs before being
1100      * unregistered by the MBean server.
1101      * This implementation also unregisters all the MXBeans
1102      * that were created by this object.
1103      * @throws IllegalStateException if the lock can't be acquire, or if
1104      *         the MBean's state doesn't allow the MBean to be unregistered
1105      *         (e.g. because it's scheduled or running).
1106      * @throws Exception This exception will be caught by the MBean server and
1107      * re-thrown as an MBeanRegistrationException.
1108      */
1109     public void preDeregister() throws Exception {
1110         try {
1111             close();
1112             if (!sequencer.tryAcquire())
1113                 throw new IllegalStateException("can't acquire lock");
1114             try {
1115                 unregisterScanners();
1116                 unregisterConfigs();
1117                 unregisterResultLogManager();
1118             } finally {
1119                 sequencer.release();
1120             }
1121         } catch (Exception x) {
1122             LOG.log(Level.FINEST,"Failed to unregister: "+x,x);
1123             throw x;
1124         }
1125     }
