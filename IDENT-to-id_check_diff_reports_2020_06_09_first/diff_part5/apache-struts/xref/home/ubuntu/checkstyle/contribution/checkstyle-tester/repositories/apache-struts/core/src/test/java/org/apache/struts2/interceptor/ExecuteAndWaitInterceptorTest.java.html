102     public void testOneWaitWithDelay() throws Exception {
103         waitInterceptor.setDelay(200);
104         waitInterceptor.setDelaySleepInterval(100);
105 
106         ActionProxy proxy = buildProxy("action1");
107         long before = System.currentTimeMillis();
108         String result = proxy.execute();
109         long after = System.currentTimeMillis();
110         assertEquals("wait", result);
111         assertTrue("delay should be ca. 200 millis", (after - before) >= 190);
112 
113         Thread.sleep(400);
114 
115         ActionProxy proxy2 = buildProxy("action1");
116         String result2 = proxy2.execute();
117         assertEquals("success", result2);
118     }

171     public void testFromDeserializedSession() throws Exception {
172         waitInterceptor.setDelay(0);
173         waitInterceptor.setDelaySleepInterval(0);
174 
175         ActionProxy proxy = buildProxy("action1");
176         String result = proxy.execute();
177         assertEquals("wait", result);
178 
179         ByteArrayOutputStream baos = new ByteArrayOutputStream();
180         ObjectOutputStream oos = new ObjectOutputStream(baos);
181         oos.writeObject(session);//WW-4900 action1 and invocation are not serializable but we should not fail at this line
182         oos.close();
183         byte[] b = baos.toByteArray();
184         baos.close();
185 
186         ByteArrayInputStream bais = new ByteArrayInputStream(b);
187         ObjectInputStream ois = new ObjectInputStream(bais);
188         session = (Map<String, Object>) ois.readObject();
189         context = ActionContext.of(context).withSession(session).getContextMap();
190         ois.close();
191         bais.close();
192 
193         Thread.sleep(1000);
194 
195         ActionProxy proxy2 = buildProxy("action1");
196         String result2 = proxy2.execute();
197         assertEquals("wait", result2);//WW-4900 A new thread should be started when background thread missed
198 
199         Thread.sleep(1000);
200 
201         ActionProxy proxy3 = buildProxy("action1");
202         String result3 = proxy3.execute();
203         assertEquals("success", result3);
204     }

206     protected ActionProxy buildProxy(String actionName) throws Exception {
207         return actionProxyFactory.createActionProxy("", actionName, null, context);
208     }
209 
210     protected void setUp() throws Exception {
211         super.setUp();
212         loadConfigurationProviders(new WaitConfigurationProvider());
213 
214         session = new HashMap<>();
215         params = new HashMap<>();
216         context = new HashMap<>();
217 
218         request = new StrutsMockHttpServletRequest();
219         httpSession = new StrutsMockHttpSession();
220         request.setSession(httpSession);
221         request.setParameterMap(params);
222 
223         context = ActionContext.of(context)
224             .withSession(session)
225             .withParameters(HttpParameters.create().build())
226             .withServletRequest(request)
227             .getContextMap();
228 
229         container.inject(parametersInterceptor);
230     }

233         super.tearDown();
234     }

236     private class WaitConfigurationProvider implements ConfigurationProvider {
237 
238         Configuration configuration;
239 
240         public void destroy() {
241             waitInterceptor.destroy();
242         }
243 
244         public boolean needsReload() {
245             return false;
246         }
247 
248         public void init(Configuration configuration) throws ConfigurationException {
249             this.configuration = configuration;
250         }
251 
252         public void loadPackages() throws ConfigurationException {
253 
254 
255             // interceptors
256             waitInterceptor = new ExecuteAndWaitInterceptor();
257             parametersInterceptor = new ParametersInterceptor();
258             PackageConfig wait = new PackageConfig.Builder("")
259                 .addActionConfig("action1", new ActionConfig.Builder("", "action1", ExecuteAndWaitDelayAction.class.getName())
260                     .addResultConfig(new ResultConfig.Builder(Action.SUCCESS, MockResult.class.getName()).build())
261                     .addResultConfig(new ResultConfig.Builder(ExecuteAndWaitInterceptor.WAIT, MockResult.class.getName()).build())
262                     .addInterceptor(new InterceptorMapping("params", parametersInterceptor))
263                     .addInterceptor(new InterceptorMapping("execAndWait", waitInterceptor))
264                     .build())
265                 .build();
266             configuration.addPackageConfig("", wait);
267         }
268 
269         public void register(ContainerBuilder builder, LocatableProperties props) throws ConfigurationException {
270             builder.factory(ObjectFactory.class);
271             builder.factory(ActionProxyFactory.class, DefaultActionProxyFactory.class);
272             builder.factory(OgnlUtil.class, OgnlUtil.class);
273         }
