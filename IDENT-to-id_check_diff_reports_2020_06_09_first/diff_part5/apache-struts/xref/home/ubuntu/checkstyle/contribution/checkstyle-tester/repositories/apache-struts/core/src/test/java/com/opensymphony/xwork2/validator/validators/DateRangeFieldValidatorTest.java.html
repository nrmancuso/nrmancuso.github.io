19  package com.opensymphony.xwork2.validator.validators;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.TextProviderFactory;
23  import com.opensymphony.xwork2.XWorkTestCase;
24  import com.opensymphony.xwork2.util.ValueStack;
25  import com.opensymphony.xwork2.util.ValueStackFactory;
26  import com.opensymphony.xwork2.validator.DummyValidatorContext;
27  import com.opensymphony.xwork2.validator.ValidatorContext;
28  
29  import java.util.Calendar;
30  import java.util.Date;
31  import java.util.Locale;
32  
33  public class DateRangeFieldValidatorTest extends XWorkTestCase {
34  
35      private TextProviderFactory tpf;
36  
37      public void testPassValidation() throws Exception {
38          // given
39          ValidationAction action = prepareAction(createDate(2013, 6, 6));
40          ValidatorContext context = new DummyValidatorContext(action, tpf);
41          DateRangeFieldValidator validator = prepareValidator(action, context);
42  
43          // when
44          validator.validate(action);
45  
46          // then
47          assertEquals(0, context.getFieldErrors().size());
48      }
49  
50      public void testMinValidation() throws Exception {
51          // given
52          ValidationAction action = prepareAction(createDate(2012, Calendar.MARCH, 3));
53          ValidatorContext context = new DummyValidatorContext(action, tpf);
54          DateRangeFieldValidator validator = prepareValidator(action, context);
55  
56          System.out.println(ActionContext.getContext().getLocale());
57  
58          // when
59          validator.validate(action);
60  
61          // then
62          assertEquals(1, context.getFieldErrors().size());
63          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 03.03.12", context.getFieldErrors().get("dateRange").get(0));
64      }
65  
66      public void testMaxValidation() throws Exception {
67          // given
68          ValidationAction action = prepareAction(createDate(2014, Calendar.APRIL, 4));
69          ValidatorContext context = new DummyValidatorContext(action, tpf);
70          DateRangeFieldValidator validator = prepareValidator(action, context);
71  
72          // when
73          validator.validate(action);
74  
75          // then
76          assertEquals(1, context.getFieldErrors().size());
77          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 04.04.14", context.getFieldErrors().get("dateRange").get(0));
78      }
79  
80      private ValidationAction prepareAction(Date range) {
81          ValidationAction action = new ValidationAction();
82          action.setDateMinValue(createDate(2013, Calendar.JANUARY, 1));
83          action.setDateMaxValue(createDate(2013, Calendar.DECEMBER, 12));
84          action.setDateRange(range);
85          return action;
86      }
87  
88      private Date createDate(int year, int month, int day) {
89          Calendar cal = Calendar.getInstance();
90          cal.set(year, month, day);
91          return cal.getTime();
92      }
93  
94      private DateRangeFieldValidator prepareValidator(ValidationAction action, ValidatorContext context) {
95          ValueStack valueStack = container.getInstance(ValueStackFactory.class).createValueStack();
96          valueStack.getActionContext().withLocale(new Locale("de"));
97          valueStack.push(action);
98  
99          DateRangeFieldValidator validator = new DateRangeFieldValidator();
100         validator.setValueStack(valueStack);
101 
102         validator.setMaxExpression("${dateMaxValue}");
103         validator.setMinExpression("${dateMinValue}");
104         validator.setValidatorContext(context);
105         validator.setFieldName("dateRange");
106         validator.setDefaultMessage("Max is ${dateMaxValue}, min is ${dateMinValue} but value is ${dateRange}");
107 
108         return validator;
109     }
110 
111     @Override
112     public void setUp() throws Exception {
113         super.setUp();
114         tpf = container.getInstance(TextProviderFactory.class);
115     }
116 
117 }
