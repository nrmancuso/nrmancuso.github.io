95      public void testDateConversion() throws ParseException {
96          java.sql.Date sqlDate = new java.sql.Date(System.currentTimeMillis());
97          assertEquals(sqlDate, converter.convertValue(context, null, null, null, sqlDate, Date.class));
98  
99          SimpleDateFormat format = new SimpleDateFormat("MM/dd/yyyy hh:mm:ss");
100         Date date = format.parse("01/10/2001 00:00:00");
101 
102         SimpleDateFormat formatt = new SimpleDateFormat("hh:mm:ss");
103         java.sql.Time datet = new java.sql.Time(formatt.parse("10:11:12").getTime());
104 
105         String dateStr = (String) converter.convertValue(context, null, null, null, date, String.class);
106         String datetStr = (String) converter.convertValue(context, null, null, null, datet, String.class);
107 
108         Date date2 = (Date) converter.convertValue(context, null, null, null, dateStr, Date.class);
109         assertEquals(date, date2);
110         java.sql.Date date3 = (java.sql.Date) converter.convertValue(context, null, null, null, dateStr, java.sql.Date.class);
111         assertEquals(date, date3);
112         java.sql.Timestamp ts = (java.sql.Timestamp) converter.convertValue(context, null, null, null, dateStr, java.sql.Timestamp.class);
113         assertEquals(date, ts);
114         java.sql.Time time1 = (java.sql.Time) converter.convertValue(context, null, null, null, datetStr, java.sql.Time.class);
115         assertEquals(datet, time1);
116 
117         Date dateWithTime = format.parse("01/10/2001 01:02:03");
118         Date dateRfc3339 = (Date) converter.convertValue(context, null, null, null, "2001-01-10T01:02:03", Date.class);
119         assertEquals(dateWithTime, dateRfc3339);
120 
121         Date dateRfc3339DateOnly = (Date) converter.convertValue(context, null, null, null, "2001-01-10", Date.class);
122         assertEquals(date, dateRfc3339DateOnly);
123     }

186         // see XW-341
187         String dateStr = "13/01/2005"; // us date format is used in context
188         Object res = converter.convertValue(context, null, null, null, dateStr, Date.class);
189         assertEquals(res, OgnlRuntime.NoConversionPossible);
190 
191         dateStr = "02/30/2005"; // us date format is used in context
192         res = converter.convertValue(context, null, null, null, dateStr, Date.class);
193         assertEquals(res, OgnlRuntime.NoConversionPossible);
194 
195         // and test a date that is passable
196         SimpleDateFormat format = new SimpleDateFormat("MM/dd/yyyy");
197         dateStr = "12/31/2005"; // us date format
198         res = converter.convertValue(context, null, null, null, dateStr, Date.class);
199         Date date = format.parse(dateStr);
200         assertNotSame(res, OgnlRuntime.NoConversionPossible);
201         assertEquals(date, res);

465 
466         ClassLoader cl = Thread.currentThread().getContextClassLoader();
467         try {
468             Thread.currentThread().setContextClassLoader(new ClassLoader(cl) {
469                 @Override

475                                 return !done;
476                             }
477 
478                             public URL nextElement() {

480                                     throw new RuntimeException("Conversion configuration loading " +

484 
485                                 done = true;
486                                 return getClass().getResource("/com/opensymphony/xwork2/conversion/impl/test-struts-conversion.properties");
487                             }
488                         };
489                     } else {
490                         return super.getResources(name);
491                     }
492                 }
493             });
494             setUp();
495         } finally {
496             Thread.currentThread().setContextClassLoader(cl);
497         }
