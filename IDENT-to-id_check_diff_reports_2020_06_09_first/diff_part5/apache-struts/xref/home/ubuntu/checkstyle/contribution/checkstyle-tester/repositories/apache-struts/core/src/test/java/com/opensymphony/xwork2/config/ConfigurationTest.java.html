90      public void testWildcardName() {
91          RuntimeConfiguration configuration = configurationManager.getConfiguration().getRuntimeConfiguration();
92  
93          ActionConfig config = configuration.getActionConfig("", "WildCard/Simple/input");
94          
95          assertNotNull(config);
96          assertTrue("Wrong class name, "+config.getClassName(), 
97                  "com.opensymphony.xwork2.SimpleAction".equals(config.getClassName()));
98          assertTrue("Wrong method name", "input".equals(config.getMethodName()));
99          
100         Map<String, String> p = config.getParams();
101         assertTrue("Wrong parameter, "+p.get("foo"), "Simple".equals(p.get("foo")));
102         assertTrue("Wrong parameter, "+p.get("bar"), "input".equals(p.get("bar")));
103     }

208     public void testMultipleContainerProviders() throws Exception {
209         // to start from scratch
210         configurationManager.destroyConfiguration();
211          // to build basic configuration
212         configurationManager.getConfiguration();
213 
214         Mock mockContainerProvider = new Mock(ContainerProvider.class);
215         mockContainerProvider.expect("init", C.ANY_ARGS);
216         mockContainerProvider.expect("register", C.ANY_ARGS);
217         mockContainerProvider.matchAndReturn("equals", C.ANY_ARGS, false);
218         mockContainerProvider.matchAndReturn("toString", "foo");
219         mockContainerProvider.matchAndReturn("destroy", null);
220         mockContainerProvider.expectAndReturn("needsReload", true);
221         // the order of providers must be changed as just first is checked if reload is needed
222         configurationManager.addContainerProvider((ContainerProvider) mockContainerProvider.proxy());
223         XmlConfigurationProvider provider = new XmlConfigurationProvider("xwork-sample.xml");
224         container.inject(provider);
225         configurationManager.addContainerProvider(provider);
226 
227         Configuration config = null;
228         try {
229             config = configurationManager.getConfiguration();
230         } catch (ConfigurationException e) {
231             e.printStackTrace();
232             fail();
233         }
234         
235         RuntimeConfiguration configuration = config.getRuntimeConfiguration();
236 
237         // check that it has configuration from xml
238         assertNotNull(configuration.getActionConfig("/foo/bar", "Bar"));
239 
240         mockContainerProvider.verify();
241     }
