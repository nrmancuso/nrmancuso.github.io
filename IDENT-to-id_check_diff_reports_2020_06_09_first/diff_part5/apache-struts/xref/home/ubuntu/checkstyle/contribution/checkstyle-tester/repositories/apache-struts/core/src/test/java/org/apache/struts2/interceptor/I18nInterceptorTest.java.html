55      public void testNoSessionNoLocale() {
56          request.setSession(null);
57          try {
58              interceptor.intercept(mai);
59              assertTrue(true);
60          } catch (Exception ignore) {
61              fail("Shouldn't throw any exception!");
62          }
63  
64          assertFalse("should have been removed",
65                  mai.getInvocationContext().getParameters().get(I18nInterceptor.DEFAULT_PARAMETER).isDefined());

67          assertNull("should not be stored here", session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE));
68      }

226     public void testCookieCreation() throws Exception {
227 
228         prepare(I18nInterceptor.DEFAULT_COOKIE_PARAMETER, "da_DK");
229 
230         final Cookie cookie = new Cookie(I18nInterceptor.DEFAULT_COOKIE_ATTRIBUTE, "da_DK");
231 
232         HttpServletResponse response = EasyMock.createMock(HttpServletResponse.class);
233         response.addCookie(CookieMatcher.eqCookie(cookie));
234         EasyMock.replay(response);
235 
236         ac.put(StrutsStatics.HTTP_RESPONSE, response);
237         interceptor.setLocaleStorage(I18nInterceptor.Storage.COOKIE.name());
238         interceptor.intercept(mai);
239 
240         EasyMock.verify(response);
241 
242         assertNull(session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE)); // should not be stored here
243         assertNull(session.get(I18nInterceptor.DEFAULT_SESSION_ATTRIBUTE)); // should not create a locale object
244     }

246     private void prepare(String key, Serializable value) {
247         Map<String, Serializable> params = new HashMap<>();
248         params.put(key, value);
249 
250         mai.getInvocationContext().setParameters(HttpParameters.create(params).build());
251     }
252 
253     public void setUp() throws Exception {
254         interceptor = new I18nInterceptor();
255         interceptor.setLocaleProviderFactory(new DefaultLocaleProviderFactory());
256         interceptor.init();
257         session = new HashMap<>();
258 
259         ac = ActionContext.of(new HashMap<>()).bind();
260         ac.setSession(session);
261         ac.setParameters(HttpParameters.create().build());
262 
263         request = new MockHttpServletRequest();
264         request.setSession(new MockHttpSession());
265         ServletActionContext.setRequest(request);
266 
267         Action action = () -> Action.SUCCESS;
268 
269         MockActionProxy proxy = new MockActionProxy();
270         proxy.setAction(action);
271         proxy.setNamespace("i18n");
272         proxy.setActionName("anAction");
273 
274         mai = new MockActionInvocation();
275         ((MockActionInvocation) mai).setAction(action);
276         ((MockActionInvocation) mai).setInvocationContext(ac);
277         ((MockActionInvocation) mai).setProxy(proxy);
278     }

288     static class CookieMatcher implements IArgumentMatcher {
289         private Cookie expected;
290 
291         CookieMatcher(Cookie cookie) {
292             expected = cookie;
293         }
294 
295         public boolean matches(Object argument) {
296             Cookie cookie = ((Cookie) argument);
297             return
298                 (cookie.getName().equals(expected.getName()) &&
299                  cookie.getValue().equals(expected.getValue()));
300         }
301 
302         public static Cookie eqCookie(Cookie ck) {
303             EasyMock.reportMatcher(new CookieMatcher(ck));
304             return null;
305         }
306 
307         public void appendTo(StringBuffer buffer) {
308             buffer
309                 .append("Received")
310                 .append(expected.getName())
311                 .append("/")
312                 .append(expected.getValue());
313         }
314     }
