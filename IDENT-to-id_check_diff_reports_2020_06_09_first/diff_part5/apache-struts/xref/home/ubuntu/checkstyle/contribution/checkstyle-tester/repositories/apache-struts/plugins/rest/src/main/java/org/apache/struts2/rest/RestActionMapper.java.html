36  /**
37   * <!-- START SNIPPET: description -->
38   *
39   * This Restful action mapper enforces Ruby-On-Rails Rest-style mappings.  If the method 
40   * is not specified (via '!' or 'method:' prefix), the method is "guessed" at using 
41   * ReST-style conventions that examine the URL and the HTTP method.  Special care has
42   * been given to ensure this mapper works correctly with the codebehind plugin so that
43   * XML configuration is unnecessary.
44   *  
45   * <p>
46   *   This mapper supports the following parameters:
47   * </p>
48   * <ul>
49   *   <li><code>struts.mapper.idParameterName</code> - If set, this value will be the name
50   *       of the parameter under which the id is stored.  The id will then be removed
51   *       from the action name.  Whether or not the method is specified, the mapper will 
52   *       try to truncate the identifier from the url and store it as a parameter.
53   *   </li>
54   *   <li><code>struts.mapper.indexMethodName</code> - The method name to call for a GET
55   *       request with no id parameter. Defaults to 'index'.
56   *   </li>
57   *   <li><code>struts.mapper.getMethodName</code> - The method name to call for a GET
58   *       request with an id parameter. Defaults to 'show'.
59   *   </li>
60   *   <li><code>struts.mapper.postMethodName</code> - The method name to call for a POST
61   *       request with no id parameter. Defaults to 'create'.
62   *   </li>
63   *   <li><code>struts.mapper.putMethodName</code> - The method name to call for a PUT
64   *       request with an id parameter. Defaults to 'update'.
65   *   </li>
66   *   <li><code>struts.mapper.deleteMethodName</code> - The method name to call for a DELETE
67   *       request with an id parameter. Defaults to 'destroy'.
68   *   </li>
69   *   <li><code>struts.mapper.editMethodName</code> - The method name to call for a GET
70   *       request with an id parameter and the 'edit' view specified. Defaults to 'edit'.
71   *   </li>
72   *   <li><code>struts.mapper.newMethodName</code> - The method name to call for a GET
73   *       request with no id parameter and the 'new' view specified. Defaults to 'editNew'.
74   *   </li>
75   * </ul>
76   * <p>
77   * The following URL's will invoke its methods:
78   * </p>
79   * <ul> 
80   *  <li><code>GET:    /movies                =&gt; method="index"</code></li>
81   *  <li><code>GET:    /movies/Thrillers      =&gt; method="show", id="Thrillers"</code></li>
82   *  <li><code>GET:    /movies/Thrillers;edit =&gt; method="edit", id="Thrillers"</code></li>
83   *  <li><code>GET:    /movies/Thrillers/edit =&gt; method="edit", id="Thrillers"</code></li>
84   *  <li><code>GET:    /movies/new            =&gt; method="editNew"</code></li>
85   *  <li><code>POST:   /movies                =&gt; method="create"</code></li>
86   *  <li><code>PUT:    /movies/Thrillers      =&gt; method="update", id="Thrillers"</code></li>
87   *  <li><code>DELETE: /movies/Thrillers      =&gt; method="destroy", id="Thrillers"</code></li>
88   * </ul>
89   * <p>
90   * To simulate the HTTP methods PUT and DELETE, since they aren't supported by HTML,
91   * the HTTP parameter "_method" will be used.
92   * </p>
93   * <!-- END SNIPPET: description -->
94   */

188     public ActionMapping getMapping(HttpServletRequest request,
189             ConfigurationManager configManager) {
190         ActionMapping mapping = new ActionMapping();
191         String uri = RequestUtils.getUri(request);
192 
193         uri = dropExtension(uri, mapping);
194         if (uri == null) {
195             return null;
196         }
197 
198         parseNameAndNamespace(uri, mapping, configManager);
199 
200         handleSpecialParameters(request, mapping);
201 
202         if (mapping.getName() == null) {
203             return null;
204         }
205 
206         // handle "name!method" convention.
207         handleDynamicMethodInvocation(mapping, mapping.getName());
208 
209         String fullName = mapping.getName();
210         // Only try something if the action name is specified
211         if (StringUtils.isNotEmpty(fullName)) {
212 
213             // cut off any ;jsessionid= type appendix but allow the rails-like ;edit
214             int scPos = fullName.indexOf(';');
215             if (scPos > -1 && !"edit".equals(fullName.substring(scPos + 1))) {
216                 fullName = fullName.substring(0, scPos);
217             }
218 
219             int lastSlashPos = fullName.lastIndexOf('/');
220             String id = null;
221             if (lastSlashPos > -1) {
222 
223                 // fun trickery to parse 'actionName/id/methodName' in the case of 'animals/dog/edit'
224                 int prevSlashPos = fullName.lastIndexOf('/', lastSlashPos - 1);
225                 //WW-4589 do not overwrite explicit method name
226                 if (prevSlashPos > -1 && mapping.getMethod() == null) {
227                     mapping.setMethod(fullName.substring(lastSlashPos + 1));
228                     fullName = fullName.substring(0, lastSlashPos);
229                     lastSlashPos = prevSlashPos;
230                 }
231                 id = fullName.substring(lastSlashPos + 1);
232             }
233 
234 
235 
236             // If a method hasn't been explicitly named, try to guess using ReST-style patterns
237             if (mapping.getMethod() == null) {
238 
239             	if (isOptions(request)) {
240                 	mapping.setMethod(optionsMethodName);
241                 
242             	// Handle uris with no id, possibly ending in '/'
243             	} else if (lastSlashPos == -1 || lastSlashPos == fullName.length() -1) {
244 
245                     // Index e.g. foo
246                     if (isGet(request)) {
247                         mapping.setMethod(indexMethodName);
248                         
249                     // Creating a new entry on POST e.g. foo
250                     } else if (isPost(request)) {
251                     	if (isExpectContinue(request)) {
252                             mapping.setMethod(postContinueMethodName);
253                     	} else {
254                             mapping.setMethod(postMethodName);
255                     	}
256                     }
257 
258                 // Handle uris with an id at the end
259                 } else if (id != null) {
260                     
261                     // Viewing the form to edit an item e.g. foo/1;edit
262                     if (isGet(request) && id.endsWith(";edit")) {
263                         id = id.substring(0, id.length() - ";edit".length());
264                         mapping.setMethod(editMethodName);
265                         
266                     // Viewing the form to create a new item e.g. foo/new
267                     } else if (isGet(request) && "new".equals(id)) {
268                         mapping.setMethod(newMethodName);
269 
270                     // Removing an item e.g. foo/1
271                     } else if (isDelete(request)) {
272                         mapping.setMethod(deleteMethodName);
273                         
274                     // Viewing an item e.g. foo/1
275                     } else if (isGet(request)) {
276                         mapping.setMethod(getMethodName);
277                     
278                     // Updating an item e.g. foo/1    
279                     }  else if (isPut(request)) {
280                     	if (isExpectContinue(request)) {
281                             mapping.setMethod(putContinueMethodName);
282                     	} else {
283                             mapping.setMethod(putMethodName);
284                     	}
285                     }
286                 }
287             }
288 
289             // cut off the id parameter, even if a method is specified
290             if (id != null) {
291                 if (!"new".equals(id)) {
292                     if (mapping.getParams() == null) {
293                         mapping.setParams(new HashMap<String, Object>());
294                     }
295                     mapping.getParams().put(idParameterName, new String[]{id});
296                 }
297                 fullName = fullName.substring(0, lastSlashPos);
298             }
299 
300             mapping.setName(cleanupActionName(fullName));
301         }
302         return mapping;
303     }

305     private void handleDynamicMethodInvocation(ActionMapping mapping, String name) {
306         int exclamation = name.lastIndexOf("!");
307         if (exclamation != -1) {
308             String actionName = name.substring(0, exclamation);
309             String actionMethod = name.substring(exclamation + 1);
310 
311             // WW-4585
312             // add any ; appendix to name, it will be handled later in getMapping method
313             int scPos = actionMethod.indexOf(';');
314             if (scPos != -1) {
315                 actionName = actionName + actionMethod.substring(scPos);
316                 actionMethod = actionMethod.substring(0, scPos);
317             }
318 
319             mapping.setName(actionName);
320             if (allowDynamicMethodCalls) {
321                 mapping.setMethod(cleanupMethodName(actionMethod));
322             } else {
323                 mapping.setMethod(null);
324             }
325         }
326     }

328     /**
329      * Parses the name and namespace from the uri.  Uses the configured package 
330      * namespaces to determine the name and id parameter, to be parsed later.
331      *
332      * @param uri
333      *            The uri
334      * @param mapping
335      *            The action mapping to populate
336      */
337     protected void parseNameAndNamespace(String uri, ActionMapping mapping, ConfigurationManager configManager) {
338         String namespace, name;
339         int lastSlash = uri.lastIndexOf("/");
340         if (lastSlash == -1) {
341             namespace = "";
342             name = uri;
343         } else if (lastSlash == 0) {
344             // ww-1046, assume it is the root namespace, it will fallback to
345             // default
346             // namespace anyway if not found in root namespace.
347             namespace = "/";
348             name = uri.substring(lastSlash + 1);
349         } else {
350             // Try to find the namespace in those defined, defaulting to ""
351             Configuration config = configManager.getConfiguration();
352             String prefix = uri.substring(0, lastSlash);
353             namespace = "";
354             // Find the longest matching namespace, defaulting to the default
355             for (Object o : config.getPackageConfigs().values()) {
356                 String ns = ((PackageConfig) o).getNamespace();
357                 if (ns != null && prefix.startsWith(ns) && (prefix.length() == ns.length() || prefix.charAt(ns.length()) == '/')) {
358                     if (ns.length() > namespace.length()) {
359                         namespace = ns;
360                     }
361                 }
362             }
363 
364             name = uri.substring(namespace.length() + 1);
365         }
366 
367         mapping.setNamespace(cleanupNamespaceName(namespace));
368         mapping.setName(name);
369     }

371     protected boolean isGet(HttpServletRequest request) {
372         return GET.equalsIgnoreCase(request.getMethod());
373     }
374 
375     protected boolean isPost(HttpServletRequest request) {
376         return POST.equalsIgnoreCase(request.getMethod());
377     }
378 
379     protected boolean isPut(HttpServletRequest request) {
380         return PUT.equalsIgnoreCase(request.getMethod()) || isPost(request) && PUT.equalsIgnoreCase(request.getParameter(HTTP_METHOD_PARAM));
381     }
382 
383     protected boolean isDelete(HttpServletRequest request) {
384         return DELETE.equalsIgnoreCase(request.getMethod()) || DELETE.equalsIgnoreCase(request.getParameter(HTTP_METHOD_PARAM));
385     }
386 
387     protected boolean isOptions(HttpServletRequest request) {
388         return OPTIONS.equalsIgnoreCase(request.getMethod());
389     }
390     
391     protected boolean isExpectContinue(HttpServletRequest request) {
392     	String expect = request.getHeader("Expect");
393     	return (expect != null && expect.toLowerCase().contains("100-continue")); 
394     }
