19  package com.opensymphony.xwork2.ognl;
20  
21  import com.opensymphony.xwork2.TextProvider;
22  import com.opensymphony.xwork2.conversion.NullHandler;
23  import com.opensymphony.xwork2.conversion.impl.XWorkConverter;
24  import com.opensymphony.xwork2.inject.Container;
25  import com.opensymphony.xwork2.inject.Inject;
26  import com.opensymphony.xwork2.ognl.accessor.CompoundRootAccessor;
27  import com.opensymphony.xwork2.util.CompoundRoot;
28  import com.opensymphony.xwork2.util.ValueStack;
29  import com.opensymphony.xwork2.util.ValueStackFactory;
30  import ognl.MethodAccessor;
31  import ognl.OgnlRuntime;
32  import ognl.PropertyAccessor;
33  import org.apache.commons.lang3.BooleanUtils;
34  import org.apache.logging.log4j.LogManager;
35  import org.apache.logging.log4j.Logger;
36  import org.apache.struts2.StrutsConstants;
37  
38  import java.util.Map;
39  import java.util.Set;
40  
41  /**
42   * Creates an Ognl value stack
43   */
44  public class OgnlValueStackFactory implements ValueStackFactory {
45  
46      private static final Logger LOG = LogManager.getLogger(OgnlValueStackFactory.class);
47  
48      protected XWorkConverter xworkConverter;
49      protected CompoundRootAccessor compoundRootAccessor;
50      protected TextProvider textProvider;
51      protected Container container;
52  
53      @Inject
54      protected void setXWorkConverter(XWorkConverter converter) {
55          this.xworkConverter = converter;
56      }
57  
58      @Inject("system")
59      protected void setTextProvider(TextProvider textProvider) {
60          this.textProvider = textProvider;
61      }
62  
63      public ValueStack createValueStack() {
64          ValueStack stack = new OgnlValueStack(xworkConverter, compoundRootAccessor, textProvider,
65              containerAllowsStaticMethodAccess(), containerAllowsStaticFieldAccess());
66          container.inject(stack);
67          return stack.getActionContext()
68              .withContainer(container)
69              .withValueStack(stack)
70              .getValueStack();
71      }
72  
73      public ValueStack createValueStack(ValueStack stack) {
74          ValueStack result = new OgnlValueStack(stack, xworkConverter, compoundRootAccessor,
75              containerAllowsStaticMethodAccess(), containerAllowsStaticFieldAccess());
76          container.inject(result);
77          return result.getActionContext()
78              .withContainer(container)
79              .withValueStack(result)
80              .getValueStack();
81      }
82  
83      @Inject
84      protected void setContainer(Container container) throws ClassNotFoundException {
85          Set<String> names = container.getInstanceNames(PropertyAccessor.class);
86          for (String name : names) {
87              Class cls = Class.forName(name);
88              if (cls != null) {
89                  if (Map.class.isAssignableFrom(cls)) {
90                      PropertyAccessor acc = container.getInstance(PropertyAccessor.class, name);
91                  }
92                  OgnlRuntime.setPropertyAccessor(cls, container.getInstance(PropertyAccessor.class, name));
93                  if (compoundRootAccessor == null && CompoundRoot.class.isAssignableFrom(cls)) {
94                      compoundRootAccessor = (CompoundRootAccessor) container.getInstance(PropertyAccessor.class, name);
95                  }
96              }
97          }
98  
99          names = container.getInstanceNames(MethodAccessor.class);
100         for (String name : names) {
101             Class cls = Class.forName(name);
102             if (cls != null) {
103                 OgnlRuntime.setMethodAccessor(cls, container.getInstance(MethodAccessor.class, name));
104             }
105         }
106 
107         names = container.getInstanceNames(NullHandler.class);
108         for (String name : names) {
109             Class cls = Class.forName(name);
110             if (cls != null) {
111                 OgnlRuntime.setNullHandler(cls, new OgnlNullHandlerWrapper(container.getInstance(NullHandler.class, name)));
112             }
113         }
114         if (compoundRootAccessor == null) {
115             throw new IllegalStateException("Couldn't find the compound root accessor");
116         }
117         this.container = container;
118     }
119 
120     /**
121      * Retrieve allowsStaticMethodAccess state from the container (allows for lazy fetching)
122      *
123      * @return
124      */
125     protected boolean containerAllowsStaticMethodAccess() {
126         return BooleanUtils.toBoolean(container.getInstance(String.class, StrutsConstants.STRUTS_ALLOW_STATIC_METHOD_ACCESS));
127     }
128 
129     /**
130      * Retrieve allowStaticFieldAccess state from the container (allows for lazy fetching)
131      *
132      * @return
133      */
134     protected boolean containerAllowsStaticFieldAccess() {
135         return BooleanUtils.toBoolean(container.getInstance(String.class, StrutsConstants.STRUTS_ALLOW_STATIC_FIELD_ACCESS));
136     }
137 
138 }
