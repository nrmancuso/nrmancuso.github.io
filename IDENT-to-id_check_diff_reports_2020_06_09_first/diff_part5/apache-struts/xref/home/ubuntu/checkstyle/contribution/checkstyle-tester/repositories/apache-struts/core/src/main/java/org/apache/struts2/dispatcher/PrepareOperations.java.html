40  /**
41   * Contains preparation operations for a request before execution
42   */
43  public class PrepareOperations {
44  
45      private static final Logger LOG = LogManager.getLogger(PrepareOperations.class);
46  
47      /**
48       * Maintains per-request override of devMode configuration.
49       */
50      private static ThreadLocal<Boolean> devModeOverride = new InheritableThreadLocal<>();
51  
52  
53      private Dispatcher dispatcher;
54      private static final String STRUTS_ACTION_MAPPING_KEY = "struts.actionMapping";
55      public static final String CLEANUP_RECURSION_COUNTER = "__cleanup_recursion_counter";
56  
57      public PrepareOperations(Dispatcher dispatcher) {
58          this.dispatcher = dispatcher;
59      }
60  
61      /**
62       * Creates the action context and initializes the thread local
63       *
64       * @param request servlet request
65       * @param response servlet response
66       *
67       * @return the action context
68       */
69      public ActionContext createActionContext(HttpServletRequest request, HttpServletResponse response) {
70          ActionContext ctx;
71          int counter = 1;
72          Integer oldCounter = (Integer) request.getAttribute(CLEANUP_RECURSION_COUNTER);
73          if (oldCounter != null) {
74              counter = oldCounter + 1;
75          }
76          
77          ActionContext oldContext = ActionContext.getContext();
78          if (oldContext != null) {
79              // detected existing context, so we are probably in a forward
80              ctx = ActionContext.of(new HashMap<>(oldContext.getContextMap())).bind();
81          } else {
82              ctx = ServletActionContext.getActionContext(request);   //checks if we are probably in an async
83              if (ctx == null) {
84                  ValueStack stack = dispatcher.getContainer().getInstance(ValueStackFactory.class).createValueStack();
85                  stack.getContext().putAll(dispatcher.createContextMap(request, response, null));
86                  ctx = ActionContext.of(stack.getContext()).bind();
87              }
88          }
89          request.setAttribute(CLEANUP_RECURSION_COUNTER, counter);
90          return ctx;
91      }
92  
93      /**
94       * Cleans up a request of thread locals
95       *
96       * @param request servlet request
97       */
98      public void cleanupRequest(HttpServletRequest request) {
99          Integer counterVal = (Integer) request.getAttribute(CLEANUP_RECURSION_COUNTER);
100         if (counterVal != null) {
101             counterVal -= 1;
102             request.setAttribute(CLEANUP_RECURSION_COUNTER, counterVal);
103             if (counterVal > 0 ) {
104                 LOG.debug("skipping cleanup counter={}", counterVal);
105                 return;
106             }
107         }
108         // always clean up the thread request, even if an action hasn't been executed
109         try {
110             dispatcher.cleanUpRequest(request);
111         } finally {
112             ActionContext.clear();
113             Dispatcher.setInstance(null);
114             devModeOverride.remove();
115         }
116     }

118     /**
119      * Assigns the dispatcher to the dispatcher thread local
120      */
121     public void assignDispatcherToThread() {
122         Dispatcher.setInstance(dispatcher);
123     }

129      * @param response servlet response

131     public void setEncodingAndLocale(HttpServletRequest request, HttpServletResponse response) {
132         dispatcher.prepare(request, response);

135     /**
136      * Wraps the request with the Struts wrapper that handles multipart requests better
137      *
138      * @param oldRequest servlet request
139      *
140      * @return The new request, if there is one
141      * @throws ServletException on any servlet related error
142      */
143     public HttpServletRequest wrapRequest(HttpServletRequest oldRequest) throws ServletException {
144         HttpServletRequest request = oldRequest;
145         try {
146             // Wrap request first, just in case it is multipart/form-data
147             // parameters might not be accessible through before encoding (ww-1278)
148             request = dispatcher.wrapRequest(request);
149             ServletActionContext.setRequest(request);
150         } catch (IOException e) {
151             throw new ServletException("Could not wrap servlet request with MultipartRequestWrapper!", e);
152         }
153         return request;
154     }

156     /**
157      * Finds and optionally creates an {@link ActionMapping}.  It first looks in the current request to see if one
158      * has already been found, otherwise, it creates it and stores it in the request.  No mapping will be created in the
159      * case of static resource requests or unidentifiable requests for other servlets, for example.
160      *
161      * @param request servlet request
162      * @param response servlet response
163      *
164      * @return the action mapping
165      */
166     public ActionMapping findActionMapping(HttpServletRequest request, HttpServletResponse response) {
167         return findActionMapping(request, response, false);
168     }
169 
170     /**
171      * Finds and optionally creates an {@link ActionMapping}.  if forceLookup is false, it first looks in the current request to see if one
172      * has already been found, otherwise, it creates it and stores it in the request.  No mapping will be created in the
173      * case of static resource requests or unidentifiable requests for other servlets, for example.
174      * @param forceLookup if true, the action mapping will be looked up from the ActionMapper instance, ignoring if there is one
175      * in the request or not 
176      *
177      * @param request servlet request
178      * @param response servlet response
179      *
180      * @return the action mapping
181      */
182     public ActionMapping findActionMapping(HttpServletRequest request, HttpServletResponse response, boolean forceLookup) {
183         ActionMapping mapping = (ActionMapping) request.getAttribute(STRUTS_ACTION_MAPPING_KEY);
184         if (mapping == null || forceLookup) {
185             try {
186                 mapping = dispatcher.getContainer().getInstance(ActionMapper.class).getMapping(request, dispatcher.getConfigurationManager());
187                 if (mapping != null) {
188                     request.setAttribute(STRUTS_ACTION_MAPPING_KEY, mapping);
189                 }
190             } catch (Exception ex) {
191                 if (dispatcher.isHandleException() || dispatcher.isDevMode()) {
192                     dispatcher.sendError(request, response, HttpServletResponse.SC_INTERNAL_SERVER_ERROR, ex);
193                 }
194             }
195         }
196 
197         return mapping;
198     }
199 
200     /**
201      * Cleans up the dispatcher instance
202      */
203     public void cleanupDispatcher() {
204         if (dispatcher == null) {
205             throw new StrutsException("Something is seriously wrong, Dispatcher is not initialized (null) ");
206         } else {
207             try {
208                 dispatcher.cleanup();
209             } finally {
210                 ActionContext.clear();
211             }
212         }
213     }
214 
215     /**
216      * Check whether the request matches a list of exclude patterns.
217      *
218      * @param request          The request to check patterns against
219      * @param excludedPatterns list of patterns for exclusion
220      *
221      * @return <tt>true</tt> if the request URI matches one of the given patterns
222      */
223     public boolean isUrlExcluded( HttpServletRequest request, List<Pattern> excludedPatterns ) {
224         if (excludedPatterns != null) {
225             String uri = RequestUtils.getUri(request);
226             for ( Pattern pattern : excludedPatterns ) {
227                 if (pattern.matcher(uri).matches()) {
228                     return true;
229                 }
230             }
231         }
232         return false;
233     }

235     /**
236      * Set an override of the static devMode value.  Do not set this via a
237      * request parameter or any other unprotected method.  Using a signed
238      * cookie is one safe way to turn it on per request.
239      *
240      * @param devMode   the override value
241      */
242     public static void overrideDevMode(boolean devMode) {
243         devModeOverride.set(devMode);
244     }
245 
246     /**
247      * @return Boolean override value, or null if no override
248      */
249     public static Boolean getDevModeOverride()
250     {
251         return devModeOverride.get();
252     }
