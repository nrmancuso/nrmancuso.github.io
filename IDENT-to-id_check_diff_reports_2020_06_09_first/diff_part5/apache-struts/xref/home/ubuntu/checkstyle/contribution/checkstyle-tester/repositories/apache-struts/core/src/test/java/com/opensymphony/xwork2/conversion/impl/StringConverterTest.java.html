44      public void testDoubleToStringConversionPL() throws Exception {
45          // given
46          StringConverter converter = new StringConverter();
47          Map<String, Object> context = createContextWithLocale(new Locale("pl", "PL"));
48  
49          // when has max fraction digits
50          Object value = converter.convertValue(context, null, null, null, Double.MIN_VALUE, null);
51  
52          // then does not lose fraction digits
53          assertEquals("0," + StringUtils.repeat('0', 323) + "49", value);
54  
55          // when has max integer digits
56          value = converter.convertValue(context, null, null, null, Double.MAX_VALUE, null);
57  
58          // then does not lose integer digits
59          assertEquals("17976931348623157" + StringUtils.repeat('0', 292), value);
60  
61          // when cannot be represented exactly with a finite binary number
62          value = converter.convertValue(context, null, null, null, 0.1d, null);
63  
64          // then produce the shortest decimal representation that can unambiguously identify the true value of the floating-point number
65          assertEquals("0,1", value);
66      }

68      public void testFloatToStringConversionPL() throws Exception {
69          // given
70          StringConverter converter = new StringConverter();
71          Map<String, Object> context = createContextWithLocale(new Locale("pl", "PL"));
72  
73          // when has max fraction digits
74          Object value = converter.convertValue(context, null, null, null, Float.MIN_VALUE, null);
75  
76          // then does not lose fraction digits
77          assertEquals("0," + StringUtils.repeat('0', 44) + "14", value);
78  
79          // when has max integer digits
80          value = converter.convertValue(context, null, null, null, Float.MAX_VALUE, null);
81  
82          // then does not lose integer digits
83          assertEquals("34028235" + StringUtils.repeat('0', 31), value);
84  
85          // when cannot be represented exactly with a finite binary number
86          value = converter.convertValue(context, null, null, null, 0.1f, null);
87  
88          // then produce the shortest decimal representation that can unambiguously identify the true value of the floating-point number
89          assertEquals("0,1", value);
90      }

92      public void testBigDecimalToStringConversionPL() throws Exception {
93          // given
94          StringConverter converter = new StringConverter();
95          Map<String, Object> context = createContextWithLocale(new Locale("pl", "PL"));
96  
97          // when a bit bigger than double
98          String aBitBiggerThanDouble = "17976931348623157" + StringUtils.repeat('0', 291) + "1."
99                  + StringUtils.repeat('0', 324) + "49";
100         Object value = converter.convertValue(context, null, null, null,
101                 new BigDecimal(aBitBiggerThanDouble), null);
102 
103         // then does not lose integer and fraction digits
104         assertEquals(aBitBiggerThanDouble.substring(0, 309) + "," + aBitBiggerThanDouble.substring(310), value);
105     }
