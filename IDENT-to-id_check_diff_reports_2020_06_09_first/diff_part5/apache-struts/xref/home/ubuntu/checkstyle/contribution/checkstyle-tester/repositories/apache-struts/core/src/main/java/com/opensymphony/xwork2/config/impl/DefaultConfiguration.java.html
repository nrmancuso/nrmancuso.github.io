117     public void addPackageConfig(String name, PackageConfig packageContext) {
118         PackageConfig check = packageContexts.get(name);
119         if (check != null) {
120             if (check.getLocation() != null && packageContext.getLocation() != null
121                     && check.getLocation().equals(packageContext.getLocation())) {
122                 LOG.debug("The package name '{}' is already been loaded by the same location and could be removed: {}",
123                         name, packageContext.getLocation());
124             } else {
125                 throw new ConfigurationException("The package name '" + name
126                         + "' at location "+packageContext.getLocation()
127                         + " is already been used by another package at location " + check.getLocation(),
128                         packageContext);
129             }
130         }
131         packageContexts.put(name, packageContext);
132     }

150     /**
151      * Calls the ConfigurationProviderFactory.getConfig() to tell it to reload the configuration and then calls
152      * buildRuntimeConfiguration().
153      *
154      * @return list of package providers
155      *
156      * @throws ConfigurationException in case of any configuration errors
157      */
158     public synchronized List<PackageProvider> reloadContainer(List<ContainerProvider> providers) throws ConfigurationException {
159         packageContexts.clear();
160         loadedFileNames.clear();
161         List<PackageProvider> packageProviders = new ArrayList<>();
162 
163         ContainerProperties props = new ContainerProperties();
164         ContainerBuilder builder = new ContainerBuilder();
165         Container bootstrap = createBootstrapContainer(providers);
166         for (final ContainerProvider containerProvider : providers)
167         {
168             bootstrap.inject(containerProvider);
169             containerProvider.init(this);
170             containerProvider.register(builder, props);
171         }
172         props.setConstants(builder);
173 
174         builder.factory(Configuration.class, new Factory<Configuration>() {
175             public Configuration create(Context context) throws Exception {
176                 return DefaultConfiguration.this;
177             }
178 
179             @Override
180             public Class<? extends Configuration> type() {
181                 return DefaultConfiguration.this.getClass();
182             }
183         });
184 
185         ActionContext oldContext = ActionContext.getContext();
186         try {
187             // Set the bootstrap container for the purposes of factory creation
188 
189             setContext(bootstrap);
190             container = builder.create(false);
191             setContext(container);
192             objectFactory = container.getInstance(ObjectFactory.class);
193 
194             // Process the configuration providers first
195             for (final ContainerProvider containerProvider : providers)
196             {
197                 if (containerProvider instanceof PackageProvider) {
198                     container.inject(containerProvider);
199                     ((PackageProvider)containerProvider).loadPackages();
200                     packageProviders.add((PackageProvider)containerProvider);
201                 }
202             }
203 
204             // Then process any package providers from the plugins
205             Set<String> packageProviderNames = container.getInstanceNames(PackageProvider.class);
206             for (String name : packageProviderNames) {
207                 PackageProvider provider = container.getInstance(PackageProvider.class, name);
208                 provider.init(this);
209                 provider.loadPackages();
210                 packageProviders.add(provider);
211             }
212 
213             rebuildRuntimeConfiguration();
214         } finally {
215             if (oldContext == null) {
216                 ActionContext.clear();
217             }
218         }
219         return packageProviders;
220     }

222     protected ActionContext setContext(Container cont) {
223         ActionContext context = ActionContext.getContext();
224         if (context == null) {
225             ValueStack vs = cont.getInstance(ValueStackFactory.class).createValueStack();
226             context = ActionContext.of(vs.getContext()).bind();
227         }
228         return context;
229     }

298     /**
299      * <p>
300      * This builds the internal runtime configuration used by Xwork for finding and configuring Actions from the
301      * programmatic configuration data structures. All of the old runtime configuration will be discarded and rebuilt.
302      * </p>
303      *
304      * <p>
305      * It basically flattens the data structures to make the information easier to access.  It will take
306      * an {@link ActionConfig} and combine its data with all inherited dast.  For example, if the {@link ActionConfig}
307      * is in a package that contains a global result and it also contains a result, the resulting {@link ActionConfig}
308      * will have two results.
309      * </p>
310      *
311      * @return runtime configuration
312      * @throws ConfigurationException in case of any configuration errors
313      */
314     protected synchronized RuntimeConfiguration buildRuntimeConfiguration() throws ConfigurationException {
315         Map<String, Map<String, ActionConfig>> namespaceActionConfigs = new LinkedHashMap<>();
316         Map<String, String> namespaceConfigs = new LinkedHashMap<>();
317 
318         for (PackageConfig packageConfig : packageContexts.values()) {
319 
320             if (!packageConfig.isAbstract()) {
321                 String namespace = packageConfig.getNamespace();
322                 Map<String, ActionConfig> configs = namespaceActionConfigs.get(namespace);
323 
324                 if (configs == null) {
325                     configs = new LinkedHashMap<>();
326                 }
327 
328                 Map<String, ActionConfig> actionConfigs = packageConfig.getAllActionConfigs();
329 
330                 for (Object o : actionConfigs.keySet()) {
331                     String actionName = (String) o;
332                     ActionConfig baseConfig = actionConfigs.get(actionName);
333                     configs.put(actionName, buildFullActionConfig(packageConfig, baseConfig));
334                 }
335 
336                 namespaceActionConfigs.put(namespace, configs);
337                 if (packageConfig.getFullDefaultActionRef() != null) {
338                     namespaceConfigs.put(namespace, packageConfig.getFullDefaultActionRef());
339                 }
340             }
341         }
342 
343         PatternMatcher<int[]> matcher = container.getInstance(PatternMatcher.class);
344         boolean appendNamedParameters = Boolean.parseBoolean(
345                 container.getInstance(String.class, StrutsConstants.STRUTS_MATCHER_APPEND_NAMED_PARAMETERS)
346         );
347 
348         return new RuntimeConfigurationImpl(Collections.unmodifiableMap(namespaceActionConfigs),
349                 Collections.unmodifiableMap(namespaceConfigs), matcher, appendNamedParameters);
350     }

352     private void setDefaultResults(Map<String, ResultConfig> results, PackageConfig packageContext) {
353         String defaultResult = packageContext.getFullDefaultResultType();
354 
355         for (Map.Entry<String, ResultConfig> entry : results.entrySet()) {
356 
357             if (entry.getValue() == null) {
358                 ResultTypeConfig resultTypeConfig = packageContext.getAllResultTypeConfigs().get(defaultResult);
359                 entry.setValue(new ResultConfig.Builder(null, resultTypeConfig.getClassName()).build());
360             }
361         }
362     }

364     /**
365      * Builds the full runtime actionconfig with all of the defaults and inheritance
366      *
367      * @param packageContext the PackageConfig which holds the base config we're building from
368      * @param baseConfig     the ActionConfig which holds only the configuration specific to itself, without the defaults
369      *                       and inheritance
370      * @return a full ActionConfig for runtime configuration with all of the inherited and default params
371      * @throws com.opensymphony.xwork2.config.ConfigurationException
372      *
373      */
374     private ActionConfig buildFullActionConfig(PackageConfig packageContext, ActionConfig baseConfig) throws ConfigurationException {
375         Map<String, String> params = new TreeMap<>(baseConfig.getParams());
376         Map<String, ResultConfig> results = new TreeMap<>();
377 
378         if (!baseConfig.getPackageName().equals(packageContext.getName()) && packageContexts.containsKey(baseConfig.getPackageName())) {
379             results.putAll(packageContexts.get(baseConfig.getPackageName()).getAllGlobalResults());
380         } else {
381             results.putAll(packageContext.getAllGlobalResults());
382         }
383 
384        	results.putAll(baseConfig.getResults());
385 
386         setDefaultResults(results, packageContext);
387 
388         List<InterceptorMapping> interceptors = new ArrayList<>(baseConfig.getInterceptors());
389 
390         if (interceptors.size() <= 0) {
391             String defaultInterceptorRefName = packageContext.getFullDefaultInterceptorRef();
392 
393             if (defaultInterceptorRefName != null) {
394                 interceptors.addAll(InterceptorBuilder.constructInterceptorReference(new PackageConfig.Builder(packageContext), defaultInterceptorRefName,
395                         new LinkedHashMap<String, String>(), packageContext.getLocation(), objectFactory));
396             }
397         }
398 
399         String methodRegex = container.getInstance(String.class, StrutsConstants.STRUTS_SMI_METHOD_REGEX);
400         if (methodRegex == null) {
401             methodRegex = ActionConfig.DEFAULT_METHOD_REGEX;
402         }
403 
404         LOG.debug("Using pattern [{}] to match allowed methods when SMI is disabled!", methodRegex);
405 
406         return new ActionConfig.Builder(baseConfig)
407             .addParams(params)
408             .addResultConfigs(results)
409             .defaultClassName(packageContext.getDefaultClassRef())  // fill in default if non class has been provided
410             .interceptors(interceptors)
411             .setStrictMethodInvocation(packageContext.isStrictMethodInvocation())
412             .setDefaultMethodRegex(methodRegex)
413             .addExceptionMappings(packageContext.getAllExceptionMappingConfigs())
414             .build();
415     }

418     private static class RuntimeConfigurationImpl implements RuntimeConfiguration {
419 
420         private Map<String, Map<String, ActionConfig>> namespaceActionConfigs;
421         private Map<String, ActionConfigMatcher> namespaceActionConfigMatchers;
422         private NamespaceMatcher namespaceMatcher;
423         private Map<String, String> namespaceConfigs;
424 
425         public RuntimeConfigurationImpl(Map<String, Map<String, ActionConfig>> namespaceActionConfigs,
426                                         Map<String, String> namespaceConfigs,
427                                         PatternMatcher<int[]> matcher,
428                                         boolean appendNamedParameters)
429         {
430             this.namespaceActionConfigs = namespaceActionConfigs;
431             this.namespaceConfigs = namespaceConfigs;
432 
433             this.namespaceActionConfigMatchers = new LinkedHashMap<>();
434             this.namespaceMatcher = new NamespaceMatcher(matcher, namespaceActionConfigs.keySet(), appendNamedParameters);
435 
436             for (Map.Entry<String, Map<String, ActionConfig>> entry : namespaceActionConfigs.entrySet()) {
437                 ActionConfigMatcher configMatcher = new ActionConfigMatcher(matcher, entry.getValue(), true, appendNamedParameters);
438                 namespaceActionConfigMatchers.put(entry.getKey(), configMatcher);
439             }
440         }
441 
442 
443         /**
444          * Gets the configuration information for an action name, or returns null if the
445          * name is not recognized.
446          *
447          * @param name      the name of the action
448          * @param namespace the namespace for the action or null for the empty namespace, ""
449          * @return the configuration information for action requested
450          */
451         public ActionConfig getActionConfig(String namespace, String name) {
452             ActionConfig config = findActionConfigInNamespace(namespace, name);
453 
454             // try wildcarded namespaces
455             if (config == null) {
456                 NamespaceMatch match = namespaceMatcher.match(namespace);
457                 if (match != null) {
458                     config = findActionConfigInNamespace(match.getPattern(), name);
459 
460                     // If config found, place all the matches found in the namespace processing in the action's parameters
461                     if (config != null) {
462                         config = new ActionConfig.Builder(config)
463                                 .addParams(match.getVariables())
464                                 .build();
465                     }
466                 }
467             }
468 
469             // fail over to empty namespace
470             if (config == null && StringUtils.isNotBlank(namespace)) {
471                 config = findActionConfigInNamespace("", name);
472             }
473 
474 
475             return config;
476         }
477 
478         private ActionConfig findActionConfigInNamespace(String namespace, String name) {
479             ActionConfig config = null;
480             if (namespace == null) {
481                 namespace = "";
482             }
483             Map<String, ActionConfig> actions = namespaceActionConfigs.get(namespace);
484             if (actions != null) {
485                 config = actions.get(name);
486                 // Check wildcards
487                 if (config == null) {
488                     config = namespaceActionConfigMatchers.get(namespace).match(name);
489                     // fail over to default action
490                     if (config == null) {
491                         String defaultActionRef = namespaceConfigs.get(namespace);
492                         if (defaultActionRef != null) {
493                             config = actions.get(defaultActionRef);
494                         }
495                     }
496                 }
497             }
498             return config;
499         }
500 
501         /**
502          * Gets the configuration settings for every action.
503          *
504          * @return a Map of namespace - > Map of ActionConfig objects, with the key being the action name
505          */
506         public Map<String, Map<String, ActionConfig>>  getActionConfigs() {
507             return namespaceActionConfigs;
508         }
509 
510         @Override
511         public String toString() {
512             StringBuilder buff = new StringBuilder("RuntimeConfiguration - actions are\n");
513 
514             for (Map.Entry<String, Map<String, ActionConfig>> entry : namespaceActionConfigs.entrySet()) {
515                 Map<String, ActionConfig> actionConfigs = entry.getValue();
516 
517                 for (String s : actionConfigs.keySet()) {
518                     buff.append(entry.getKey()).append("/").append(s).append("\n");
519                 }
520             }
521 
522             return buff.toString();
523         }
524     }

526     class ContainerProperties extends LocatableProperties {
527         private static final long serialVersionUID = -7320625750836896089L;
528 
529         @Override
530         public Object setProperty(String key, String value) {
531             String oldValue = getProperty(key);
532             if (LOG.isInfoEnabled() && oldValue != null && !oldValue.equals(value) && !defaultFrameworkBeanName.equals(oldValue)) {
533                 LOG.info("Overriding property {} - old value: {} new value: {}", key, oldValue, value);
534             }
535             return super.setProperty(key, value);
536         }
537 
538         public void setConstants(ContainerBuilder builder) {
539             for (Object keyobj : keySet()) {
540                 String key = (String)keyobj;
541                 builder.factory(String.class, key, new LocatableConstantFactory<>(getProperty(key), getPropertyLocation(key)));
542             }
543         }
