19  package com.opensymphony.xwork2.validator.annotations;
20  
21  import java.lang.annotation.ElementType;
22  import java.lang.annotation.Retention;
23  import java.lang.annotation.RetentionPolicy;
24  import java.lang.annotation.Target;
25  
26  /**
27   * <!-- START SNIPPET: description -->
28   * <p>
29   * The validator allows you to forward validator to object properties of your action
30   * using the objects own validator files. This allows you to use the ModelDriven development
31   * pattern and manage your validations for your models in one place, where they belong, next to
32   * your model classes.
33   * </p>
34   *
35   * <p>
36   * The ConditionalVisitorFieldValidator can handle either simple Object properties, Collections of Objects, or Arrays.
37   * The error message for the ConditionalVisitorFieldValidator will be appended in front of validator messages added
38   * by the validations for the Object message.
39   * </p>
40   *
41   * <!-- END SNIPPET: description -->
42   *
43   * <p><u>Annotation usage:</u></p>
44   *
45   * <!-- START SNIPPET: usage -->
46   * <p>The annotation must be applied at method level.</p>
47   * <!-- END SNIPPET: usage -->
48   *
49   * <p><u>Annotation parameters:</u></p>
50   *
51   * <!-- START SNIPPET: parameters -->
52   * <table class='confluenceTable' summary=''>
53   * <tr>
54   * <th class='confluenceTh'> Parameter </th>
55   * <th class='confluenceTh'> Required </th>
56   * <th class='confluenceTh'> Default </th>
57   * <th class='confluenceTh'> Notes </th>
58   * </tr>
59   * <tr>
60   * <td class='confluenceTd'>expression</td>
61   * <td class='confluenceTd'>yes</td>
62   * <td class='confluenceTd'>&nbsp;</td>
63   * <td class='confluenceTd'>Boolean conditional expression</td>
64   * </tr>
65   * <tr>
66   * <td class='confluenceTd'>message</td>
67   * <td class='confluenceTd'>yes</td>
68   * <td class='confluenceTd'>&nbsp;</td>
69   * <td class='confluenceTd'>field error message</td>
70   * </tr>
71   * <tr>
72   * <td class='confluenceTd'>key</td>
73   * <td class='confluenceTd'>no</td>
74   * <td class='confluenceTd'>&nbsp;</td>
75   * <td class='confluenceTd'>i18n key from language specific properties file.</td>
76   * </tr>
77   * <tr>
78   * <td class='confluenceTd'>messageParams</td>
79   * <td class='confluenceTd'>no</td>
80   * <td class='confluenceTd'>&nbsp;</td>
81   * <td class='confluenceTd'>Additional params to be used to customize message - will be evaluated against the Value Stack</td>
82   * </tr>
83   * <tr>
84   * <td class='confluenceTd'>fieldName</td>
85   * <td class='confluenceTd'>no</td>
86   * <td class='confluenceTd'>&nbsp;</td>
87   * <td class='confluenceTd'>&nbsp;</td>
88   * </tr>
89   * <tr>
90   * <td class='confluenceTd'>shortCircuit</td>
91   * <td class='confluenceTd'>no</td>
92   * <td class='confluenceTd'>false</td>
93   * <td class='confluenceTd'>If this validator should be used as shortCircuit.</td>
94   * </tr>
95   * <tr>
96   * <td class='confluenceTd'> context </td>
97   * <td class='confluenceTd'> no </td>
98   * <td class='confluenceTd'> action alias </td>
99   * <td class='confluenceTd'> Determines the context to use for validating the Object property. If not defined, the context of the Action validation is propagated to the Object property validation.  In the case of Action validation, this context is the Action alias.  </td>
100  * </tr>
101  * <tr>
102  * <td class='confluenceTd'> appendPrefix </td>
103  * <td class='confluenceTd'> no </td>
104  * <td class='confluenceTd'> true </td>
105  * <td class='confluenceTd'> Determines whether the field name of this field validator should be prepended to the field name of the visited field to determine the full field name when an error occurs.  For example, suppose that the bean being validated has a "name" property.  If <em>appendPrefix</em> is true, then the field error will be stored under the field "bean.name".  If <em>appendPrefix</em> is false, then the field error will be stored under the field "name".  <br> <img class="emoticon" src="/images/icons/emoticons/warning.gif" height="16" width="16" alt="" style="border: 0px; align: middle;"> If you are using the VisitorFieldValidator to validate the model from a ModelDriven Action, you should set <em>appendPrefix</em> to false unless you are using "model.name" to reference the properties on your model. </td>
106  * </tr>
107  * </table>
108  * <!-- END SNIPPET: parameters -->
109  *
110  * <p><u>Example code:</u></p>
111  *
112  * <pre>
113  * <!-- START SNIPPET: example -->
114  * &#64;ConditionalVisitorFieldValidator(expression="app.appid &gt; 100",  message = "Default message", key = "i18n.key", shortCircuit = true, context = "action alias", appendPrefix = true)
115  * <!-- END SNIPPET: example -->
116  * </pre>
117  *
118  * @author Matt Raible
119  */
120 @Target({ElementType.METHOD})
121 @Retention(RetentionPolicy.RUNTIME)
122 public @interface ConditionalVisitorFieldValidator {
123 
124     /**
125      * Determines the context to use for validating the Object property.
126      * If not defined, the context of the Action validator is propagated to the Object property validator.
127      * In the case of Action validator, this context is the Action alias.
128      *
129      * @return the context
130      */
131     String context() default "";
132 
133     /**
134      * Determines whether the field name of this field validator should be prepended to the field name of
135      * the visited field to determine the full field name when an error occurs. For example, suppose that
136      * the bean being validated has a 'name' property.
137      *
138      * If appendPrefix is true, then the field error will be stored under the field 'bean.name'.
139      * If appendPrefix is false, then the field error will be stored under the field 'name'.
140      *
141      * If you are using the ConditionalVisitorFieldValidator to validate the model from a ModelDriven Action,
142      * you should set appendPrefix to false unless you are using "model.name" to reference the properties
143      * on your model.
144      *
145      * @return if append prefix
146      */
147     boolean appendPrefix() default true;
148 
149     /**
150      * @return The conditional expression.
151      */
152     String expression();
153 
154     /**
155      * @return The default error message for this validator.
156      * NOTE: It is required to set a message, if you are not using the message key for 18n lookup!
157      */
158     String message() default "";
159 
160     /**
161      * @return The message key to lookup for i18n.
162      */
163     String key() default "";
164 
165     /**
166      * @return Additional params to be used to customize message - will be evaluated against the Value Stack
167      */
168     String[] messageParams() default {};
169 
170     /**
171      * @return The optional fieldName for SIMPLE validator types.
172      */
173     String fieldName() default "";
174 
175     /**
176      * If this is activated, the validator will be used as short-circuit.
177      *
178      * Adds the short-circuit='true' attribute value if <tt>true</tt>.
179      *
180      * @return true if validator will be used as short-circuit. Default is false.
181      */
182     boolean shortCircuit() default false;
183     
184 }
