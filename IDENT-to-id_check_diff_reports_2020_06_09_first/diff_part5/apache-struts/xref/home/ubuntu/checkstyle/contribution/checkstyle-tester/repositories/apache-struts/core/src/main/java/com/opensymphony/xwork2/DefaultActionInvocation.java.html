135     /**
136      * If the DefaultActionInvocation has been executed before and the Result is an instance of ActionChainResult, this method
137      * will walk down the chain of ActionChainResults until it finds a non-chain result, which will be returned. If the
138      * DefaultActionInvocation's result has not been executed before, the Result instance will be created and populated with
139      * the result params.
140      *
141      * @return a Result instance
142      * @throws Exception in case of any error
143      */
144     public Result getResult() throws Exception {
145         Result returnResult = result;
146 
147         // If we've chained to other Actions, we need to find the last result
148         while (returnResult instanceof ActionChainResult) {
149             ActionProxy aProxy = ((ActionChainResult) returnResult).getProxy();
150 
151             if (aProxy != null) {
152                 Result proxyResult = aProxy.getInvocation().getResult();
153 
154                 if ((proxyResult != null) && (aProxy.getExecuteResult())) {
155                     returnResult = proxyResult;
156                 } else {
157                     break;
158                 }
159             } else {
160                 break;
161             }
162         }
163 
164         return returnResult;
165     }

183     /**
184      * Register a com.opensymphony.xwork2.interceptor.PreResultListener to be notified after the Action is executed and before the
185      * Result is executed. The ActionInvocation implementation must guarantee that listeners will be called in the order
186      * in which they are registered. Listener registration and execution does not need to be thread-safe.
187      *
188      * @param listener to register
189      */
190     public void addPreResultListener(PreResultListener listener) {
191         if (preResultListeners == null) {
192             preResultListeners = new ArrayList<>(1);
193         }
194 
195         preResultListeners.add(listener);
196     }

198     public Result createResult() throws Exception {
199         LOG.trace("Creating result related to resultCode [{}]", resultCode);
200 
201         if (explicitResult != null) {
202             Result ret = explicitResult;
203             explicitResult = null;
204 
205             return ret;
206         }
207         ActionConfig config = proxy.getConfig();
208         Map<String, ResultConfig> results = config.getResults();
209 
210         ResultConfig resultConfig = null;
211 
212         try {
213             resultConfig = results.get(resultCode);
214         } catch (NullPointerException e) {
215             LOG.debug("Got NPE trying to read result configuration for resultCode [{}]", resultCode);
216         }
217         
218         if (resultConfig == null) {
219             // If no result is found for the given resultCode, try to get a wildcard '*' match.
220             resultConfig = results.get("*");
221         }
222 
223         if (resultConfig != null) {
224             try {
225                 return objectFactory.buildResult(resultConfig, invocationContext.getContextMap());
226             } catch (Exception e) {
227                 LOG.error("There was an exception while instantiating the result of type {}", resultConfig.getClassName(), e);
228                 throw new StrutsException(e, resultConfig);
229             }
230         } else if (resultCode != null && !Action.NONE.equals(resultCode) && unknownHandlerManager.hasUnknownHandlers()) {
231             return unknownHandlerManager.handleUnknownResult(invocationContext, proxy.getActionName(), proxy.getConfig(), resultCode);
232         }
233         return null;
234     }

236     /**
237      * @throws ConfigurationException If no result can be found with the returned code
238      */
239     public String invoke() throws Exception {
240         if (executed) {
241             throw new IllegalStateException("Action has already executed");
242         }
243 
244         if (asyncManager == null || !asyncManager.hasAsyncActionResult()) {
245             if (interceptors.hasNext()) {
246                 final InterceptorMapping interceptorMapping = interceptors.next();
247                 Interceptor interceptor = interceptorMapping.getInterceptor();
248                 if (interceptor instanceof WithLazyParams) {
249                     interceptor = lazyParamInjector.injectParams(interceptor, interceptorMapping.getParams(), invocationContext);
250                 }
251                 resultCode = interceptor.intercept(DefaultActionInvocation.this);
252             } else {
253                 resultCode = invokeActionOnly();
254             }
255         } else {
256             Object asyncActionResult = asyncManager.getAsyncActionResult();
257             if (asyncActionResult instanceof Throwable) {
258                 throw new Exception((Throwable) asyncActionResult);
259             }
260             asyncAction = null;
261             resultCode = saveResult(proxy.getConfig(), asyncActionResult);
262         }
263 
264         if (asyncManager == null || asyncAction == null) {
265             // this is needed because the result will be executed, then control will return to the Interceptor, which will
266             // return above and flow through again
267             if (!executed) {
268                 if (preResultListeners != null) {
269                     LOG.trace("Executing PreResultListeners for result [{}]", result);
270 
271                     for (Object preResultListener : preResultListeners) {
272                         PreResultListener listener = (PreResultListener) preResultListener;
273 
274                         listener.beforeResult(this, resultCode);
275                     }
276                 }
277 
278                 // now execute the result, if we're supposed to
279                 if (proxy.getExecuteResult()) {
280                     executeResult();
281                 }
282 
283                 executed = true;
284             }
285         } else {
286             asyncManager.invokeAsyncAction(asyncAction);
287         }
288 
289         return resultCode;
290     }

292     public String invokeActionOnly() throws Exception {
293         return invokeAction(getAction(), proxy.getConfig());
294     }

296     protected void createAction(Map<String, Object> contextMap) {
297         // load action
298         try {
299             action = objectFactory.buildAction(proxy.getActionName(), proxy.getNamespace(), proxy.getConfig(), contextMap);
300         } catch (InstantiationException e) {
301             throw new StrutsException("Unable to instantiate Action!", e, proxy.getConfig());
302         } catch (IllegalAccessException e) {
303             throw new StrutsException("Illegal access to constructor, is it public?", e, proxy.getConfig());
304         } catch (Exception e) {
305             String gripe;
306 
307             if (proxy == null) {
308                 gripe = "Whoa!  No ActionProxy instance found in current ActionInvocation.  This is bad ... very bad";
309             } else if (proxy.getConfig() == null) {
310                 gripe = "Sheesh.  Where'd that ActionProxy get to?  I can't find it in the current ActionInvocation!?";
311             } else if (proxy.getConfig().getClassName() == null) {
312                 gripe = "No Action defined for '" + proxy.getActionName() + "' in namespace '" + proxy.getNamespace() + "'";
313             } else {
314                 gripe = "Unable to instantiate Action, " + proxy.getConfig().getClassName() + ",  defined for '" + proxy.getActionName() + "' in namespace '" + proxy.getNamespace() + "'";
315             }
316 
317             gripe += (((" -- " + e.getMessage()) != null) ? e.getMessage() : " [no message in exception]");
318             throw new StrutsException(gripe, e, proxy.getConfig());
319         }
320 
321         if (actionEventListener != null) {
322             action = actionEventListener.prepare(action, stack);
323         }
324     }

326     protected Map<String, Object> createContextMap() {
327         ActionContext actionContext;
328 
329         if (ActionContext.containsValueStack(extraContext)) {
330             // In case the ValueStack was passed in
331             stack = ActionContext.of(extraContext).getValueStack();
332 
333             if (stack == null) {
334                 throw new IllegalStateException("There was a null Stack set into the extra params.");
335             }
336 
337             actionContext = stack.getActionContext();
338         } else {
339             // create the value stack
340             // this also adds the ValueStack to its context
341             stack = valueStackFactory.createValueStack();
342 
343             // create the action context
344             actionContext = stack.getActionContext();
345         }
346 
347         return actionContext
348             .withExtraContext(extraContext)
349             .withActionInvocation(this)
350             .withContainer(container)
351             .getContextMap();
352     }

354     /**
355      * Uses getResult to get the final Result and executes it
356      *
357      * @throws ConfigurationException If not result can be found with the returned code
358      */
359     private void executeResult() throws Exception {
360         result = createResult();
361 
362         if (result != null) {
363             result.execute(this);
364         } else if (resultCode != null && !Action.NONE.equals(resultCode)) {
365             throw new ConfigurationException("No result defined for action " + getAction().getClass().getName()
366                     + " and result " + getResultCode(), proxy.getConfig());
367         } else {
368             if (LOG.isDebugEnabled()) {
369                 LOG.debug("No result returned for action {} at {}", getAction().getClass().getName(), proxy.getConfig().getLocation());
370             }
371         }
372     }

374     public void init(ActionProxy proxy) {
375         this.proxy = proxy;
376         Map<String, Object> contextMap = createContextMap();
377 
378         // Setting this so that other classes, like object factories, can use the ActionProxy and other
379         // contextual information to operate
380         ActionContext actionContext = ActionContext.getContext();
381 
382         if (actionContext != null) {
383             actionContext.withActionInvocation(this);
384         }
385 
386         createAction(contextMap);
387 
388         if (pushAction) {
389             stack.push(action);
390             contextMap.put("action", action);
391         }
392 
393         invocationContext = ActionContext.of(contextMap)
394             .withActionName(proxy.getActionName());
395 
396         createInterceptors(proxy);
397 
398         prepareLazyParamInjector(invocationContext.getValueStack());
399     }
400 
401     protected void prepareLazyParamInjector(ValueStack valueStack) {
402         lazyParamInjector = new WithLazyParams.LazyParamInjector(valueStack);
403         container.inject(lazyParamInjector);
404     }

406     protected void createInterceptors(ActionProxy proxy) {
407         // Get a new List so we don't get problems with the iterator if someone changes the original list
408         List<InterceptorMapping> interceptorList = new ArrayList<>(proxy.getConfig().getInterceptors());
409         interceptors = interceptorList.iterator();
410     }

412     protected String invokeAction(Object action, ActionConfig actionConfig) throws Exception {
413         String methodName = proxy.getMethod();
414 
415         LOG.debug("Executing action method = {}", methodName);
416 
417         try {
418             Object methodResult;
419             try {
420                 methodResult = ognlUtil.callMethod(methodName + "()", getStack().getContext(), action);
421             } catch (MethodFailedException e) {
422                 // if reason is missing method,  try checking UnknownHandlers
423                 if (e.getReason() instanceof NoSuchMethodException) {
424                     if (unknownHandlerManager.hasUnknownHandlers()) {
425                         try {
426                             methodResult = unknownHandlerManager.handleUnknownMethod(action, methodName);
427                         } catch (NoSuchMethodException ignore) {
428                             // throw the original one
429                             throw e;
430                         }
431                     } else {
432                         // throw the original one
433                         throw e;
434                     }
435                     // throw the original exception as UnknownHandlers weren't able to handle invocation as well
436                     if (methodResult == null) {
437                         throw e;
438                     }
439                 } else {
440                     // exception isn't related to missing action method, throw it
441                     throw e;
442                 }
443             }
444             return saveResult(actionConfig, methodResult);
445         } catch (NoSuchPropertyException e) {
446             throw new IllegalArgumentException("The " + methodName + "() is not defined in action " + getAction().getClass() + "");
447         } catch (MethodFailedException e) {
448             // We try to return the source exception.
449             Throwable t = e.getCause();
450 
451             if (actionEventListener != null) {
452                 String result = actionEventListener.handleException(t, getStack());
453                 if (result != null) {
454                     return result;
455                 }
456             }
457             if (t instanceof Exception) {
458                 throw (Exception) t;
459             } else {
460                 throw e;
461             }
462         }
463     }

465     /**
466      * Save the result to be used later.
467      * @param actionConfig current ActionConfig
468      * @param methodResult the result of the action.
469      * @return the result code to process.
470      */
471     protected String saveResult(ActionConfig actionConfig, Object methodResult) {
472         if (methodResult instanceof Result) {
473             this.explicitResult = (Result) methodResult;
474 
475             // Wire the result automatically
476             container.inject(explicitResult);
477             return null;
478         } else if (methodResult instanceof Callable) {
479             asyncAction = (Callable) methodResult;
480             return null;
481         } else {
482             return (String) methodResult;
483         }
484     }
