65      public void testValidationAnnotation() throws Exception {
66          // given
67          AnnotationActionValidatorManager manager = createValidationManager(AnnotationValidationAction.class, Locale.US);
68  
69          // when
70          List<Validator> validators = manager.getValidators(AnnotationValidationAction.class, null);
71  
72          // then

74          for (Validator validator : validators) {
75              validate(validator);
76          }

79      public void testValidationAnnotationExpParams() throws Exception {
80          // given
81          AnnotationActionValidatorManager manager = createValidationManager(AnnotationValidationExpAction.class, Locale.US);
82  
83          // when
84          List<Validator> validators = manager.getValidators(AnnotationValidationExpAction.class, null);
85  
86          // then
87          ValueStack valueStack = container.getInstance(ValueStackFactory.class).createValueStack();
88          valueStack.push(new AnnotationValidationExpAction());

97      private void validate(Validator validator) throws Exception {
98          if (validator.getValidatorType().equals("regex")) {
99              validateRegexValidator((RegexFieldValidator) validator);
100         } else if (validator.getValidatorType().equals("conditionalvisitor")) {
101             validateConditionalFieldVisitorValidator((ConditionalVisitorFieldValidator) validator);
102         } else if (validator.getValidatorType().equals("conversion")) {
103             validateConversionFieldErrorVisitorValidator((ConversionErrorFieldValidator) validator);
104         } else if (validator.getValidatorType().equals("myValidator")) {
105             validateMyValidator((MyValidator) validator);
106         } else if (validator.getValidatorType().equals("date")) {
107             validateDateRangeFieldValidator((DateRangeFieldValidator) validator);
108         } else if (validator.getValidatorType().equals("double")) {
109             validateDoubleRangeFieldValidator((DoubleRangeFieldValidator) validator);
110         } else if (validator.getValidatorType().equals("email")) {
111             validateEmailValidator((EmailValidator) validator);
112         } else if (validator.getValidatorType().equals("creditcard")) {
113             validateCreditCardValidator((CreditCardValidator) validator);
114         } else if (validator.getValidatorType().equals("expression")) {
115             validateExpressionValidator((ExpressionValidator) validator);
116         } else if (validator.getValidatorType().equals("fieldexpression")) {
117             validateFieldExpressionValidator((FieldExpressionValidator) validator);
118         } else if (validator.getValidatorType().equals("int")) {
119             validateIntRangeFieldValidator((IntRangeFieldValidator) validator);
120         } else if (validator.getValidatorType().equals("long")) {
121             validateLongRangeFieldValidator((LongRangeFieldValidator) validator);
122         } else if (validator.getValidatorType().equals("required")) {
123             validateRequiredFieldValidator((RequiredFieldValidator) validator);
124         } else if (validator.getValidatorType().equals("requiredstring")) {
125             validateRequiredStringValidator((RequiredStringValidator) validator);
126         } else if (validator.getValidatorType().equals("short")) {
127             validateShortRangeFieldValidator((ShortRangeFieldValidator) validator);
128         } else if (validator.getValidatorType().equals("stringlength")) {
129             validateStringLengthFieldValidator((StringLengthFieldValidator) validator);
130         } else if (validator.getValidatorType().equals("url")) {
131             validateUrlValidator((URLValidator) validator);
132         } else if (validator.getValidatorType().equals("visitor")) {
133             validateVisitorFieldValidator((VisitorFieldValidator) validator);
134         }
135     }

313     private AnnotationActionValidatorManager createValidationManager(final Class<? extends ActionSupport> actionClass, Locale locale) throws Exception {
314         loadConfigurationProviders(new ConfigurationProvider() {
315             public void destroy() {
316 
317             }
318 
319             public void init(Configuration configuration) throws ConfigurationException {
320                 configuration.addPackageConfig("default", new PackageConfig.Builder("default")
321                         .addActionConfig("annotation", new ActionConfig.Builder("", "annotation", actionClass.getName()).build())
322                         .build());
323             }
324 
325             public boolean needsReload() {
326                 return false;
327             }
328 
329             public void loadPackages() throws ConfigurationException {
330 
331             }
332 
333             public void register(ContainerBuilder builder, LocatableProperties props) throws ConfigurationException {
334                 builder.constant(StrutsConstants.STRUTS_DEVMODE, true);
335             }
336         });
337 
338         // ActionContext is destroyed during rebuilding configuration
339         ActionContext.getContext().withLocale(locale);
340 
341         ActionInvocation invocation = new DefaultActionInvocation(ActionContext.getContext().getContextMap(), true);
342         container.inject(invocation);
343         invocation.init(actionProxyFactory.createActionProxy("", "annotation", null, ActionContext.getContext().getContextMap()));
344 
345         AnnotationActionValidatorManager manager = new AnnotationActionValidatorManager();
346         container.inject(manager);
347 
348         ValidatorFactory vf = container.getInstance(ValidatorFactory.class);
349         vf.registerValidator("myValidator", MyValidator.class.getName());
350 
351         return manager;
352     }
