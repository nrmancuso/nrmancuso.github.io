99      PageInfo(BeanRepository beanRepository, String jspFile) {
100 
101         this.jspFile = jspFile;
102         this.beanRepository = beanRepository;
103         this.taglibsMap = new HashMap();
104         this.jspPrefixMapper = new HashMap();
105         this.xmlPrefixMapper = new HashMap();
106         this.nonCustomTagPrefixMap = new HashMap();
107         this.imports = new Vector();
108         this.dependants = new Vector();
109         this.includePrelude = new Vector();
110         this.includeCoda = new Vector();
111         this.pluginDcls = new Vector();
112         this.prefixes = new HashSet();
113 
114         // Enter standard imports
115         for(int i = 0; i < Constants.STANDARD_IMPORTS.length; i++)
116             imports.add(Constants.STANDARD_IMPORTS[i]);
117     }

313     /*
314      * Pushes the given URI onto the stack of URIs to which the given prefix
315      * is mapped.
316      *
317      * @param prefix The prefix whose stack of URIs is to be pushed
318      * @param uri The URI to be pushed onto the stack
319      */
320     public void pushPrefixMapping(String prefix, String uri) {
321         LinkedList stack = (LinkedList) xmlPrefixMapper.get(prefix);
322         if (stack == null) {
323             stack = new LinkedList();
324             xmlPrefixMapper.put(prefix, stack);
325         }
326         stack.addFirst(uri);
327     }
328 
329     /*
330      * Removes the URI at the top of the stack of URIs to which the given
331      * prefix is mapped.
332      *
333      * @param prefix The prefix whose stack of URIs is to be popped
334      */
335     public void popPrefixMapping(String prefix) {
336         LinkedList stack = (LinkedList) xmlPrefixMapper.get(prefix);
337         if (stack == null || stack.size() == 0) {
338             // XXX throw new Exception("XXX");
339         }
340         stack.removeFirst();
341     }
342 
343     /*
344      * Returns the URI to which the given prefix maps.
345      *
346      * @param prefix The prefix whose URI is sought
347      *
348      * @return The URI to which the given prefix maps
349      */
350     public String getURI(String prefix) {
351 
352         String uri = null;
353 
354         LinkedList stack = (LinkedList) xmlPrefixMapper.get(prefix);
355         if (stack == null || stack.size() == 0) {
356             uri = (String) jspPrefixMapper.get(prefix);
357         } else {
358             uri = (String) stack.getFirst();
359         }
360 
361         return uri;
362     }

367     /*
368      * language
369      */
370     public void setLanguage(String value, Node n, ErrorDispatcher err,
371                 boolean pagedir)
372         throws JasperException {
373 
374         if (!"java".equalsIgnoreCase(value)) {
375             if (pagedir)
376                 err.jspError(n, "jsp.error.page.language.nonjava");
377             else
378                 err.jspError(n, "jsp.error.tag.language.nonjava");
379         }
380 
381         language = value;
382     }

386     }

389         return getLanguage(true);

393     /*
394      * extends
395      */
396     public void setExtends(String value, Node.PageDirective n) {
397 
398         xtends = value;
399 
400         /*
401          * If page superclass is top level class (i.e. not in a package)
402          * explicitly import it. If this is not done, the compiler will assume
403          * the extended class is in the same pkg as the generated servlet.
404          */
405         if (value.indexOf('.') < 0)
406             n.addImport(value);
407     }
408

448     /*
449      * buffer
450      */
451     public void setBufferValue(String value, Node n, ErrorDispatcher err)
452         throws JasperException {
453 
454         if ("none".equalsIgnoreCase(value))
455             buffer = 0;
456         else {
457             if (value == null || !value.endsWith("kb"))
458                 err.jspError(n, "jsp.error.page.invalid.buffer");
459             try {
460                 Integer k = new Integer(value.substring(0, value.length()-2));
461                 buffer = k.intValue() * 1024;
462             } catch (NumberFormatException e) {
463                 err.jspError(n, "jsp.error.page.invalid.buffer");
464             }
465         }
466 
467         bufferValue = value;
468     }
469 
470     public String getBufferValue() {
471         return bufferValue;
472     }
473 
474     public int getBuffer() {
475         return buffer;
476     }
