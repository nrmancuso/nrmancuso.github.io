19  package org.apache.struts2.views.util;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.ActionInvocation;
23  import com.opensymphony.xwork2.util.ValueStack;
24  import org.apache.struts2.StrutsConstants;
25  import org.apache.struts2.util.StrutsUtil;
26  import org.apache.struts2.views.jsp.ui.OgnlTool;
27  
28  import javax.servlet.http.HttpServletRequest;
29  import javax.servlet.http.HttpServletResponse;
30  import java.util.HashMap;
31  import java.util.Map;
32  
33  /**
34   * Value Stack's Context related Utilities.
35   */
36  public class ContextUtil {
37      public static final String REQUEST = "request";
38      public static final String RESPONSE = "response";
39      public static final String SESSION = "session";
40      public static final String BASE = "base";
41      public static final String STACK = "stack";
42      public static final String OGNL = "ognl";
43      public static final String STRUTS = "struts";
44      public static final String ACTION = "action";
45  
46      public static Map<String, Object> getStandardContext(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
47          HashMap<String, Object> map = new HashMap<>();
48          map.put(REQUEST, req);
49          map.put(RESPONSE, res);
50          map.put(SESSION, req.getSession(false));
51          map.put(BASE, req.getContextPath());
52          map.put(STACK, stack);
53          map.put(OGNL, stack.getActionContext().getContainer().getInstance(OgnlTool.class));
54          map.put(STRUTS, new StrutsUtil(stack, req, res));
55  
56          ActionInvocation invocation = stack.getActionContext().getActionInvocation();
57          if (invocation != null) {
58              map.put(ACTION, invocation.getAction());
59          }
60          return map;
61      }
62  
63      /**
64       * Return true if either Configuration's altSyntax is on or the stack context's useAltSyntax is on
65       *
66       * @param context stack's context
67       * @return boolean
68       */
69      public static boolean isUseAltSyntax(Map<String, Object> context) {
70          // We didn't make altSyntax static cause, if so, struts.configuration.xml.reload will not work
71          // plus the Configuration implementation should cache the properties, which the framework's
72          // configuration implementation does
73          String tagAltSytnax = ActionContext.of(context).getContainer().getInstance(String.class, StrutsConstants.STRUTS_TAG_ALTSYNTAX);
74  
75          return "true".equals(tagAltSytnax) || (
76              (context.containsKey("useAltSyntax") &&
77                  context.get("useAltSyntax") != null &&
78                  "true".equals(context.get("useAltSyntax").toString()))
79          );
80      }
81  
82      /**
83       * Returns a String for overriding the default templateSuffix if templateSuffix is on the stack
84       *
85       * @param context stack's context
86       * @return String
87       */
88      public static String getTemplateSuffix(Map<String, Object> context) {
89          return context.containsKey("templateSuffix") ? (String) context.get("templateSuffix") : null;
90      }
