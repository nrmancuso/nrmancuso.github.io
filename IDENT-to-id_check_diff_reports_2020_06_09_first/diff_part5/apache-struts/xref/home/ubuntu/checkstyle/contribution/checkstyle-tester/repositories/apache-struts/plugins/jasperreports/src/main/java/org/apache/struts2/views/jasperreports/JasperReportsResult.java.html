57  /**
58   * <!-- START SNIPPET: description -->
59   * <p>
60   * Generates a JasperReports report using the specified format or PDF if no
61   * format is specified.
62   * </p>
63   * <!-- END SNIPPET: description -->
64   * <p>
65   * <b>This result type takes the following parameters:</b>
66   * </p>
67   * <!-- START SNIPPET: params -->
68   *
69   * <ul>
70   *
71   * <li><b>location (default)</b> - the location where the compiled jasper report
72   * definition is (foo.jasper), relative from current URL.</li>
73   * <li><b>dataSource (required)</b> - the EL expression used to retrieve the
74   * datasource from the value stack (usually a List).</li>
75   * <li><b>parse</b> - true by default. If set to false, the location param will
76   * not be parsed for EL expressions.</li>
77   * <li><b>format</b> - the format in which the report should be generated. Valid
78   * values can be found in {@link JasperReportConstants}. If no format is
79   * specified, PDF will be used.</li>
80   * <li><b>contentDisposition</b> - disposition (defaults to "inline", values are
81   * typically <i>filename="document.pdf"</i>).</li>
82   * <li><b>documentName</b> - name of the document (will generate the http header
83   * <code>Content-disposition = X; filename=X.[format]</code>).</li>
84   * <li><b>delimiter</b> - the delimiter used when generating CSV reports. By
85   * default, the character used is ",".</li>
86   * <li><b>imageServletUrl</b> - name of the url that, when prefixed with the
87   * context page, can return report images.</li>
88   * <li>
89   * <b>reportParameters</b> - (2.1.2+) OGNL expression used to retrieve a map of
90   * report parameters from the value stack. The parameters may be accessed
91   * in the report via the usual JR mechanism and might include data not
92   * part of the dataSource, such as the user name of the report creator, etc.
93   * </li>
94   * <li>
95   * <b>exportParameters</b> - (2.1.2+) OGNL expression used to retrieve a map of
96   * JR exporter parameters from the value stack. The export parameters are
97   * used to customize the JR export. For example, a PDF export might enable
98   * encryption and set the user password to a string known to the report creator.
99   * </li>
100  * <li>
101  * <b>connection</b> - (2.1.7+) JDBC Connection which can be passed to the
102  * report instead of dataSource
103  * </li>
104  * <li><b>wrapField</b> - (2.3.18+) defines if fields should warp with ValueStackDataSource
105  * see https://issues.apache.org/jira/browse/WW-3698 for more details
106  * </li>
107  * </ul>
108  * <p>
109  * This result follows the same rules from {@link StrutsResultSupport}.
110  * Specifically, all parameters will be parsed if the "parse" parameter
111  * is not set to false.
112  * </p>
113  * <!-- END SNIPPET: params -->
114  * <p><b>Example:</b></p>
115  * <pre>
116  * <!-- START SNIPPET: example1 -->
117  * &lt;result name="success" type="jasper"&gt;
118  *   &lt;param name="location"&gt;foo.jasper&lt;/param&gt;
119  *   &lt;param name="dataSource"&gt;mySource&lt;/param&gt;
120  *   &lt;param name="format"&gt;CSV&lt;/param&gt;
121  * &lt;/result&gt;
122  * <!-- END SNIPPET: example1 -->
123  * </pre>
124  * <p>
125  * or for pdf
126  *
127  * <pre>
128  * <!-- START SNIPPET: example2 -->
129  * &lt;result name="success" type="jasper"&gt;
130  *   &lt;param name="location"&gt;foo.jasper&lt;/param&gt;
131  *   &lt;param name="dataSource"&gt;mySource&lt;/param&gt;
132  * &lt;/result&gt;
133  * <!-- END SNIPPET: example2 -->
134  * </pre>
135  */

248     protected void doExecute(String finalLocation, ActionInvocation invocation) throws Exception {
249         // Will throw a runtime exception if no "datasource" property. TODO Best place for that is...?
250         initializeProperties(invocation);
251 
252         LOG.debug("Creating JasperReport for dataSource = {}, format = {}", dataSource, format);
253 
254         HttpServletRequest request = invocation.getInvocationContext().getServletRequest();
255         HttpServletResponse response = invocation.getInvocationContext().getServletResponse();
256 
257         // Handle IE special case: it sends a "contype" request first.
258         // TODO Set content type to config settings?
259         if ("contype".equals(request.getHeader("User-Agent"))) {
260             try (OutputStream outputStream = response.getOutputStream()) {
261                 response.setContentType("application/pdf");
262                 response.setContentLength(0);
263             } catch (IOException e) {
264                 LOG.error("Error writing report output", e);
265                 throw new ServletException(e.getMessage(), e);
266             }
267             return;
268         }
269 
270         // Construct the data source for the report.
271         ValueStack stack = invocation.getStack();
272         ValueStackDataSource stackDataSource = null;
273 
274         Connection conn = (Connection) stack.findValue(connection);
275         if (conn == null) {
276             stackDataSource = new ValueStackDataSource(stack, dataSource, wrapField);
277         }
278 
279         if ("https".equalsIgnoreCase(request.getScheme())) {
280             // set the the HTTP Header to work around IE SSL weirdness
281             response.setHeader("CACHE-CONTROL", "PRIVATE");
282             response.setHeader("Cache-Control", "maxage=3600");
283             response.setHeader("Pragma", "public");
284             response.setHeader("Accept-Ranges", "none");
285         }
286 
287         // Determine the directory that the report file is in and set the reportDirectory parameter
288         // For WW 2.1.7:
289         //  ServletContext servletContext = ((ServletConfig) invocation.getInvocationContext().get(ServletActionContext.SERVLET_CONFIG)).getServletContext();
290         ServletContext servletContext = invocation.getInvocationContext().getServletContext();
291         String systemId = servletContext.getRealPath(finalLocation);
292         Map<String, Object> parameters = new ValueStackShadowMap(stack);
293         File directory = new File(systemId.substring(0, systemId.lastIndexOf(File.separator)));
294         parameters.put("reportDirectory", directory);
295         parameters.put(JRParameter.REPORT_LOCALE, invocation.getInvocationContext().getLocale());
296 
297         // put timezone in jasper report parameter
298         if (timeZone != null) {
299             timeZone = conditionalParse(timeZone, invocation);
300             final TimeZone tz = TimeZone.getTimeZone(timeZone);
301             if (tz != null) {
302                 // put the report time zone
303                 parameters.put(JRParameter.REPORT_TIME_ZONE, tz);
304             }
305         }
306 
307         // Add any report parameters from action to param map.
308         Map reportParams = (Map) stack.findValue(reportParameters);
309         if (reportParams != null) {
310             LOG.debug("Found report parameters; adding to parameters...");
311             parameters.putAll(reportParams);
312         }
313 
314         ByteArrayOutputStream output;
315         JasperPrint jasperPrint;
316 
317         // Fill the report and produce a print object
318         try {
319             JasperReport jasperReport = (JasperReport) JRLoader.loadObject(new File(systemId));
320             if (conn == null) {
321                 jasperPrint = JasperFillManager.fillReport(jasperReport, parameters, stackDataSource);
322             } else {
323                 jasperPrint = JasperFillManager.fillReport(jasperReport, parameters, conn);
324             }
325         } catch (JRException e) {
326             LOG.error("Error building report for uri {}", systemId, e);
327             throw new ServletException(e.getMessage(), e);
328         }
329 
330         // Export the print object to the desired output format
331         try {
332             if (contentDisposition != null || documentName != null) {
333                 final StringBuilder tmp = new StringBuilder();
334                 tmp.append((contentDisposition == null) ? "inline" : contentDisposition);
335 
336                 if (documentName != null) {
337                     tmp.append("; filename=");
338                     tmp.append(documentName);
339                     tmp.append(".");
340                     tmp.append(format.toLowerCase());
341                 }
342 
343                 response.setHeader("Content-disposition", tmp.toString());
344             }
345 
346             //TODO: replace deprecated logic
347             JRExporter exporter;
348 
349             switch (format) {
350                 case FORMAT_PDF:
351                     response.setContentType("application/pdf");
352                     exporter = new JRPdfExporter();
353                     break;
354                 case FORMAT_CSV:
355                     response.setContentType("text/csv");
356                     exporter = new JRCsvExporter();
357                     break;
358                 case FORMAT_HTML:
359                     response.setContentType("text/html");
360 
361                     // IMAGES_MAPS seems to be only supported as "backward compatible" from JasperReports 1.1.0
362 
363                     Map imagesMap = new HashMap();
364                     request.getSession(true).setAttribute("IMAGES_MAP", imagesMap);
365 
366                     exporter = new HtmlExporter();
367                     exporter.setParameter(JRHtmlExporterParameter.IMAGES_MAP, imagesMap);
368                     exporter.setParameter(JRHtmlExporterParameter.IMAGES_URI, request.getContextPath() + imageServletUrl);
369 
370                     // Needed to support chart images:
371                     exporter.setParameter(JRExporterParameter.JASPER_PRINT, jasperPrint);
372                     request.getSession().setAttribute("net.sf.jasperreports.j2ee.jasper_print", jasperPrint);
373                     break;
374                 case FORMAT_XLS:
375                     response.setContentType("application/vnd.ms-excel");
376                     exporter = new JRXlsExporter();
377                     break;
378                 case FORMAT_XML:
379                     response.setContentType("text/xml");
380                     exporter = new JRXmlExporter();
381                     break;
382                 case FORMAT_RTF:
383                     response.setContentType("application/rtf");
384                     exporter = new JRRtfExporter();
385                     break;
386                 default:
387                     throw new ServletException("Unknown report format: " + format);
388             }
389 
390             Map exportParams = (Map) stack.findValue(exportParameters);
391             if (exportParams != null) {
392                 LOG.debug("Found export parameters; adding to exporter parameters...");
393                 exporter.getParameters().putAll(exportParams);
394             }
395 
396             output = exportReportToBytes(jasperPrint, exporter);
397         } catch (JRException e) {
398             LOG.error("Error producing {} report for uri {}", format, systemId, e);
399             throw new ServletException(e.getMessage(), e);
400         } finally {
401             try {
402                 if (conn != null) {
403                     // avoid NPE if connection was not used for the report
404                     conn.close();
405                 }
406             } catch (Exception e) {
407                 LOG.warn("Could not close db connection properly", e);
408             }
409         }
410 
411         response.setContentLength(output.size());
412         // Will throw ServletException on IOException.
413         writeReport(response, output);
414     }
415 
416     /**
417      * Writes report bytes to response output stream.
418      *
419      * @param response Current response.
420      * @param output   Report bytes to write.
421      * @throws ServletException on stream IOException.
422      */
423     private void writeReport(HttpServletResponse response, ByteArrayOutputStream output) throws ServletException {
424         try (OutputStream outputStream = response.getOutputStream()) {
425             output.writeTo(outputStream);
426             outputStream.flush();
427         } catch (IOException e) {
428             LOG.error("Error writing report output", e);
429             throw new ServletException(e.getMessage(), e);
430         }
431     }

433     /**
434      * Sets up result properties, parsing etc.
435      *
436      * @param invocation Current invocation.
437      * @throws Exception on initialization error.
438      */
439     private void initializeProperties(ActionInvocation invocation) {
440         if (dataSource == null && connection == null) {
441             String message = "No dataSource specified...";
442             LOG.error(message);
443             throw new RuntimeException(message);
444         }
445         if (dataSource != null) {
446             dataSource = conditionalParse(dataSource, invocation);
447         }
448 
449         format = conditionalParse(format, invocation);
450         if (StringUtils.isEmpty(format)) {
451             format = FORMAT_PDF;
452         }
453 
454         if (contentDisposition != null) {
455             contentDisposition = conditionalParse(contentDisposition, invocation);
456         }
457 
458         if (documentName != null) {
459             documentName = conditionalParse(documentName, invocation);
460         }
461 
462         reportParameters = conditionalParse(reportParameters, invocation);
463         exportParameters = conditionalParse(exportParameters, invocation);
464     }

466     /**
467      * Run a Jasper report to CSV format and put the results in a byte array
468      *
469      * @param jasperPrint The Print object to render as CSV
470      * @param exporter    The exporter to use to export the report
471      * @return A CSV formatted report
472      * @throws net.sf.jasperreports.engine.JRException If there is a problem running the report
473      */
474     private ByteArrayOutputStream exportReportToBytes(JasperPrint jasperPrint, JRExporter exporter) throws JRException {
475         ByteArrayOutputStream baos = new ByteArrayOutputStream();
476 
477         exporter.setParameter(JRExporterParameter.JASPER_PRINT, jasperPrint);
478         exporter.setParameter(JRExporterParameter.OUTPUT_STREAM, baos);
479         if (delimiter != null) {
480             exporter.setParameter(JRCsvExporterParameter.FIELD_DELIMITER, delimiter);
481         }
482 
483         exporter.exportReport();
484         return baos;
485     }
