384         loadConfigurationProviders(new DefaultPropertiesProvider());
385 
386         Object defaultMaxLengthFromConfiguration = container.getInstance(String.class, StrutsConstants.STRUTS_OGNL_EXPRESSION_MAX_LENGTH);
387         if (defaultMaxLengthFromConfiguration != null) {
388             assertTrue("non-null defaultMaxLengthFromConfiguration not a String ?", defaultMaxLengthFromConfiguration instanceof String);
389             assertTrue("non-null defaultMaxLengthFromConfiguration not empty string by default ?", ((String) defaultMaxLengthFromConfiguration).length() == 0);
390         } else {
391             assertNull("defaultMaxLengthFromConfiguration not null ?", defaultMaxLengthFromConfiguration);
392         }
393         // Original test logic was to confirm failure of exceeding the default value.  Now the feature should be disabled by default,
394         // so this test's expectations are now changed.
395         Integer repeat = Integer.valueOf(256);  // Since maxlength is disabled by default, just choose an arbitrary value for test
396 
397         OgnlValueStack vs = createValueStack();
398         try {
399             vs.findValue(StringUtils.repeat('.', repeat + 1), true);
400             fail("findValue did not throw any exception (should either fail as invalid expression syntax or security exception) ?");
401         } catch (Exception ex) {
402             // If STRUTS_OGNL_EXPRESSION_MAX_LENGTH feature is disabled (default), the parse should fail due to a reason of invalid expression syntax
403             // with ParseException.  Previously when it was enabled the reason for the failure would have been SecurityException.
404             assertTrue(ex.getCause() instanceof OgnlException);
405             assertTrue(((OgnlException) ex.getCause()).getReason() instanceof ParseException);
406         }
