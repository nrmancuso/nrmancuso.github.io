84          Number lowerBound, upperBound;
85  
86          @Override
87          public void initMethod(Method method) {
88              lowerBound = upperBound = null;
89          }

91          @Override
92          public void sawOpcode(int seen) {
93              if(seen == Const.INVOKESTATIC && getClassConstantOperand().equals("java/lang/Math") && (getMethodDescriptorOperand().getName().equals("max")
94                      || getMethodDescriptorOperand().getName().equals("min"))) {
95                  Object const1 = stack.getStackItem(0).getConstant();
96                  Object const2 = stack.getStackItem(1).getConstant();
97                  Number n = null;
98                  if(const1 != null ^ const2 != null) {
99                      n = (const1 instanceof Number) ? (Number)const1 : (Number)const2;
100                     if(getMethodDescriptorOperand().getName().equals("min")) {
101                         upperBound = n;
102                     } else {
103                         lowerBound = n;
104                     }
105                 } else {
106                     upperBound = lowerBound = null;
107                 }
108                 XMethod rvo1 = stack.getStackItem(0).getReturnValueOf();
109                 XMethod rvo2 = stack.getStackItem(1).getReturnValueOf();
110                 if(rvo1 != null ^ rvo2 != null) {
111                     XMethod rvo = rvo1 == null ? rvo2 : rvo1;
112                     if (lowerBound instanceof Comparable && upperBound != null && upperBound.getClass() == lowerBound.getClass()
113                             && rvo.getClassDescriptor().getClassName().equals("java/lang/Math")
114                             && (rvo.getName().equals("max") || rvo.getName().equals("min"))) {
115                         @SuppressWarnings("unchecked")
116                         int result = ((Comparable<Number>)lowerBound).compareTo(upperBound);
117                         if(result > 0) {
118                             accumulator.accumulateBug(
119                                     new BugInstance("DM_INVALID_MIN_MAX", HIGH_PRIORITY).addClassAndMethod(DumbMethods.this)
120                                     .addString(String.valueOf(n)),
121                                     DumbMethods.this);
122                         }
123                     }
124                 }
125             }

222         private void checkRange(Item item, Object minValue, Object maxValue, String pattern) {
223             if(!(item.getConstant() instanceof Number)) {
224                 return;
225             }
226             int value = ((Number)item.getConstant()).intValue();
227             int intMin = Integer.MIN_VALUE;
228             int intMax = Integer.MAX_VALUE;
229             if(minValue instanceof Number) {
230                 intMin = ((Number)minValue).intValue();
231             }
232             if(maxValue instanceof Number) {
233                 intMax = ((Number)maxValue).intValue();
234             } else if(maxValue instanceof String) {
235                 intMax = ((String)maxValue).length()-1;
236             } else if (maxValue instanceof OpcodeStack.Item){
237                 OpcodeStack.Item maxItem = (OpcodeStack.Item ) maxValue;
238                 if (maxItem.getSignature().charAt(0) == '[' && maxItem.getConstant() instanceof Integer) {
239                     intMax = ((Integer)maxItem.getConstant())-1;
240 
241                 }
242             }
243 
244             if(value < intMin || value > intMax) {
245                 BugInstance bug = new BugInstance(pattern, NORMAL_PRIORITY ).addClassAndMethod(DumbMethods.this).addSourceLine(DumbMethods.this)
246                         .addInt(value).describe(IntAnnotation.INT_VALUE);
247 
248                 if (intMin <= intMax) {
249                     if (value < intMin) {
250                         bug.addInt(intMin).describe(IntAnnotation.INT_MIN_VALUE);
251                     }
252                     if (value > intMax) {
253                         bug.addInt(intMax) .describe(IntAnnotation.INT_MAX_VALUE);
254                     }
255                 }
256 
257 
258                 if (isMethodCall()) {
259                     bug.addCalledMethod(DumbMethods.this);
260                 }
261 
262 
263 
264                 accumulator.accumulateBug(bug, DumbMethods.this);
265             }
266         }

269         @Override
270         public void sawOpcode(int seen) {
271             // System.out.printf("%4d %s%n", getPC(), OPCODE_NAMES[seen]);
272             switch(seen) {
273             case Const.IALOAD:
274             case Const.AALOAD:
275             case Const.SALOAD:
276             case Const.CALOAD:
277             case Const.BALOAD:
278             case Const.LALOAD:
279             case Const.DALOAD:
280             case Const.FALOAD: {
281                 checkRange(stack.getStackItem(0), 0, stack.getStackItem(1), "RANGE_ARRAY_INDEX");
282                 break;
283             }
284             case Const.IASTORE:
285             case Const.AASTORE:
286             case Const.SASTORE:
287             case Const.CASTORE:
288             case Const.BASTORE:
289             case Const.LASTORE:
290             case Const.DASTORE:
291             case Const.FASTORE: {
292 
293                 checkRange(stack.getStackItem(1), 0,  stack.getStackItem(2), "RANGE_ARRAY_INDEX");
294                 break;
295             }
296             case Const.INVOKESTATIC: {
297                 MethodDescriptor m = getMethodDescriptorOperand();
298                 if(m.getSlashedClassName().equals("java/lang/System") && m.getName().equals("arraycopy")) {
299                     // void arraycopy(Object src, int srcPos, Object dest, int destPos, int length)
300                     Item length = stack.getStackItem(0);
301                     Object constantLength = length.getConstant();
302                     //                    if (constantLength instanceof Number && constantLength.equals(0)) {
303                     //                        break;
304                     //                    }
305                     Item srcPos = stack.getStackItem(3);
306                     Item src = stack.getStackItem(4);
307                     checkRange(srcPos, 0, src, "RANGE_ARRAY_OFFSET");
308                     Item dest =  stack.getStackItem(2);
309                     Item destPos = stack.getStackItem(1);
310                     checkRange(destPos, 0, dest, "RANGE_ARRAY_OFFSET");
311 
312                     if(constantLength instanceof Number) {
313                         int length1 = Integer.MAX_VALUE;
314                         if(src.getConstant() instanceof Integer) {
315                             length1 = (int) src.getConstant();
316                         }
317                         if(srcPos.getConstant() instanceof Integer) {
318                             length1 -= (int) srcPos.getConstant();
319                         }
320                         int length2 = Integer.MAX_VALUE;
321                         if(dest.getConstant() instanceof Integer) {
322                             length2 = (int) stack.getStackItem(2).getConstant();
323                         }
324                         if(destPos.getConstant() instanceof Integer) {
325                             length2 -= (int) stack.getStackItem(1).getConstant();
326                         }
327                         checkRange(length, 0, Math.min(length1, length2), "RANGE_ARRAY_LENGTH");
328                     }
329                 }
330                 break;
331             }
332             case Const.INVOKEVIRTUAL:
333             case Const.INVOKESPECIAL: {
334                 MethodDescriptor m = getMethodDescriptorOperand();
335                 if(m.getSlashedClassName().equals("java/lang/String")) {
336                     if((m.getName().equals("charAt") || m.getName().equals("codePointAt")) && m.getSignature().startsWith("(I)")) {
337                         checkRange(stack.getStackItem(0), 0, stack.getStackItem(1).getConstant(), "RANGE_STRING_INDEX");
338                     }
339                     if(m.getName().equals("substring") || m.getName().equals("subSequence")) {
340                         int nArgs = getNumberArguments(m.getSignature());
341                         Item thisArg = stack.getStackItem(nArgs);
342                         Item firstArg = stack.getStackItem(nArgs-1);
343                         Object thisConstantValue = thisArg.getConstant();
344                         int maxLength = thisConstantValue instanceof String ? ((String)thisConstantValue).length() : Integer.MAX_VALUE;
345                         checkRange(firstArg, 0,maxLength, "RANGE_STRING_INDEX");
346                         if(nArgs == 2) {
347                             Item secondArg = stack.getStackItem(0);
348                             checkRange(secondArg, firstArg.getConstant() == null ? 0 : firstArg.getConstant(),
349                                     maxLength,
350                                     "RANGE_STRING_INDEX");
351                         }
352                     }
353                 }
354                 if ((m.getSignature().startsWith("([BII)") || m.getSignature().startsWith("([CII)") || m.getSignature().startsWith("([III)"))
355                         && (((m.getName().equals("write") || m.getName().equals("read")) && m.getSlashedClassName().startsWith(
356                                 "java/io/")) || (m.getName().equals(Const.CONSTRUCTOR_NAME) && m.getSlashedClassName().equals("java/lang/String")))) {
357                     Item arrayArg = stack.getStackItem(2);
358                     Item offsetArg = stack.getStackItem(1);
359                     Item lengthArg = stack.getStackItem(0);
360                     int length = Integer.MAX_VALUE;
361                     if(arrayArg.getConstant() instanceof Integer) {
362                         length = (int) arrayArg.getConstant();
363                     }
364                     if(offsetArg.getConstant() instanceof Integer) {
365                         checkRange(offsetArg, 0, saturatingIncrement(length), "RANGE_ARRAY_OFFSET");
366                         length -= (int) offsetArg.getConstant();
367                     }
368                     checkRange(lengthArg, 0, saturatingIncrement(length), "RANGE_ARRAY_LENGTH");
369                 }
370                 break;
371             }
372             default:
373                 break;
374             }
375         }

394     private class VacuousComparisonSubDetector extends SubDetector {
395         @Override
396         public void sawOpcode(int seen) {
397             boolean foundVacuousComparison = false;
398             if (seen == Const.IF_ICMPGT || seen == Const.IF_ICMPLE) {
399                 OpcodeStack.Item rhs = stack.getStackItem(0);
400                 Object rhsConstant = rhs.getConstant();
401                 if (rhsConstant instanceof Integer && ((Integer) rhsConstant).intValue() == Integer.MAX_VALUE) {
402                     foundVacuousComparison = true;
403                 }
404                 OpcodeStack.Item lhs = stack.getStackItem(1);
405                 Object lhsConstant = lhs.getConstant();
406                 if (lhsConstant instanceof Integer && ((Integer) lhsConstant).intValue() == Integer.MIN_VALUE) {
407                     foundVacuousComparison = true;
408                 }
409 
410             }
411             if (seen == Const.IF_ICMPLT || seen == Const.IF_ICMPGE) {
412                 OpcodeStack.Item rhs = stack.getStackItem(0);
413                 Object rhsConstant = rhs.getConstant();
414                 if (rhsConstant instanceof Integer && ((Integer) rhsConstant).intValue() == Integer.MIN_VALUE) {
415                     foundVacuousComparison = true;
416                 }
417                 OpcodeStack.Item lhs = stack.getStackItem(1);
418                 Object lhsConstant = lhs.getConstant();
419                 if (lhsConstant instanceof Integer && ((Integer) lhsConstant).intValue() == Integer.MAX_VALUE) {
420                     foundVacuousComparison = true;
421                 }
422 
423             }
424             if (foundVacuousComparison) {
425                 accumulator.accumulateBug(new BugInstance(DumbMethods.this, "INT_VACUOUS_COMPARISON", getBranchOffset() < 0 ? HIGH_PRIORITY
426                         : NORMAL_PRIORITY).addClassAndMethod(DumbMethods.this), DumbMethods.this);
427             }
428         }

450                     OpcodeStack.Item item = stack.getStackItem(0);
451                     if (item.getRegisterNumber() == 1) {
452                         sawInstanceofCheck = true;
453                     }
454                 } else if (seen == Const.INSTANCEOF || seen == Const.INVOKEVIRTUAL && "getClass".equals(getNameConstantOperand())
455                         && "()Ljava/lang/Class;".equals(getSigConstantOperand())) {
456                     OpcodeStack.Item item = stack.getStackItem(0);
457                     if (item.getRegisterNumber() == 1) {
458                         sawInstanceofCheck = true;
459                     }
460                 } else if (seen == Const.INVOKESPECIAL && "equals".equals(getNameConstantOperand())
461                         && "(Ljava/lang/Object;)Z".equals(getSigConstantOperand())) {
462                     OpcodeStack.Item item0 = stack.getStackItem(0);
463                     OpcodeStack.Item item1 = stack.getStackItem(1);
464                     if (item1.getRegisterNumber() + item0.getRegisterNumber() == 1) {
465                         sawInstanceofCheck = true;
466                     }
467                 } else if (seen == Const.CHECKCAST && !sawInstanceofCheck) {
468                     OpcodeStack.Item item = stack.getStackItem(0);
469                     if (item.getRegisterNumber() == 1) {
470                         if (getSizeOfSurroundingTryBlock(getPC()) == Integer.MAX_VALUE) {
471                             accumulator.accumulateBug(new BugInstance(DumbMethods.this, "BC_EQUALS_METHOD_SHOULD_WORK_FOR_ALL_OBJECTS",
472                                     NORMAL_PRIORITY).addClassAndMethod(DumbMethods.this), DumbMethods.this);
473                         }
474 
475                         reportedBadCastInEquals = true;
476                     }
477                 }
478             }
479         }
