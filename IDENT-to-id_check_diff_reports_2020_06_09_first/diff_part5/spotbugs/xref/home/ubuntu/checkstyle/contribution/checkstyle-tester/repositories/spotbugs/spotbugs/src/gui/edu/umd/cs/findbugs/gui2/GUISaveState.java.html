142     private static String[] generateSorterKeys(int numSorters) {
143         String[] result = new String[numSorters];
144         for (int x = 0; x < result.length; x++) {
145             result[x] = "Sorter" + x;
146         }
147         return result;
148     }

157     public static void loadInstance() {
158         GUISaveState newInstance = new GUISaveState();
159         newInstance.recentFiles = new ArrayList<>();
160         Preferences p = Preferences.userNodeForPackage(GUISaveState.class);
161 
162         newInstance.tabSize = p.getInt(TAB_SIZE, 4);
163 
164         newInstance.fontSize = p.getFloat(FONT_SIZE, 12.0f);
165 
166         newInstance.starterDirectoryForLoadBugs = new File(p.get(GUISaveState.STARTERDIRECTORY,
167                 SystemProperties.getProperty("user.dir")));
168 
169         int prevCommentsSize = p.getInt(GUISaveState.PREVCOMMENTSSIZE, 0);
170 
171         for (int x = 0; x < prevCommentsSize; x++) {
172             String comment = p.get(GUISaveState.COMMENTKEYS[x], "");
173             newInstance.previousComments.add(comment);
174         }
175 
176         int size = Math.min(MAXNUMRECENTPROJECTS, p.getInt(GUISaveState.NUMPROJECTS, 0));
177         for (int x = 0; x < size; x++) {
178             newInstance.addRecentFile(new File(p.get(GUISaveState.RECENTPROJECTKEYS[x], "")));
179         }
180 
181         int sorterSize = p.getInt(GUISaveState.SORTERTABLELENGTH, -1);
182         if (sorterSize != -1) {
183             ArrayList<Sortables> sortColumns = new ArrayList<>();
184             String[] sortKeys = GUISaveState.generateSorterKeys(sorterSize);
185             for (int x = 0; x < sorterSize; x++) {
186                 Sortables s = Sortables.getSortableByPrettyName(p.get(sortKeys[x], "*none*"));
187 
188                 if (s == null) {
189                     if (MainFrame.GUI2_DEBUG) {
190                         System.err.println("Sort order was corrupted, using default sort order");
191                     }
192                     newInstance.useDefault = true;
193                     break;
194                 }
195                 sortColumns.add(s);
196             }
197             if (!newInstance.useDefault) {
198                 // add in default columns
199                 Set<Sortables> missingSortColumns = new HashSet<>(Arrays.asList(DEFAULT_COLUMN_HEADERS));
200                 missingSortColumns.removeAll(sortColumns);
201                 sortColumns.addAll(missingSortColumns);
202                 newInstance.sortColumns = sortColumns.toArray(new Sortables[sortColumns.size()]);
203             }
204         } else {
205             newInstance.useDefault = true;
206         }
207 
208         newInstance.dockingLayout = p.getByteArray(DOCKINGLAYOUT, new byte[0]);
209 
210         String boundsString = p.get(FRAME_BOUNDS, null);
211         Rectangle r = new Rectangle(0, 0, 800, 650);
212         if (boundsString != null) {
213             String[] a = boundsString.split(",", 4);
214             if (a.length > 0) {
215                 try {
216                     r.x = Math.max(0, Integer.parseInt(a[0]));
217                 } catch (NumberFormatException nfe) {
218                     assert true;
219                 }
220             }
221             if (a.length > 1) {
222                 try {
223                     r.y = Math.max(0, Integer.parseInt(a[1]));
224                 } catch (NumberFormatException nfe) {
225                     assert true;
226                 }
227             }
228             if (a.length > 2) {
229                 try {
230                     r.width = Math.max(40, Integer.parseInt(a[2]));
231                 } catch (NumberFormatException nfe) {
232                     assert true;
233                 }
234             }
235             if (a.length > 3) {
236                 try {
237                     r.height = Math.max(40, Integer.parseInt(a[3]));
238                 } catch (NumberFormatException nfe) {
239                     assert true;
240                 }
241             }
242         }
243         newInstance.frameBounds = r;
244         newInstance.extendedWindowState = p.getInt(EXTENDED_WINDOW_STATE, Frame.NORMAL);
245 
246         newInstance.splitMain = p.getInt(SPLIT_MAIN, 400);
247         newInstance.splitSummary = p.getInt(SPLIT_SUMMARY_NEW, 400);
248         newInstance.splitTop = p.getInt(SPLIT_TOP, -1);
249         newInstance.splitTreeComments = p.getInt(SPLIT_TREE_COMMENTS, 250);
250         newInstance.packagePrefixSegments = p.getInt(PACKAGE_PREFIX_SEGEMENTS, 3);
251 
252         String plugins = p.get(CUSTOM_PLUGINS, "");
253         if (plugins.length() > 0) {
254             for (String s : plugins.split(" ")) {
255                 try {
256                     URI u = new URI(s);
257                     Plugin.addCustomPlugin(u);
258                     newInstance.customPlugins.add(u);
259                 } catch (PluginException e) {
260                     assert true;
261                 } catch (URISyntaxException e) {
262                     assert true;
263                 }
264             }
265         }
266 
267         String enabledPluginsString = p.get(ENABLED_PLUGINS, "");
268         String disabledPluginsString = p.get(DISABLED_PLUGINS, "");
269         newInstance.enabledPlugins = new ArrayList<>(Arrays.asList(enabledPluginsString.split(",")));
270         newInstance.disabledPlugins = new ArrayList<>(Arrays.asList(disabledPluginsString.split(",")));
271 
272         instance = newInstance;
273     }

276         Preferences p = Preferences.userNodeForPackage(GUISaveState.class);
277         try {
278             p.clear();
279         } catch (BackingStoreException e) {
280             Debug.println(e);
281         }
282         instance = new GUISaveState();
283     }

314     /**
315      * This should be the method called to add a reused file for the recent
316      * menu.
317      */
318     public void fileReused(File f) {
319         if (!recentFiles.contains(f)) {
320             throw new IllegalStateException("Selected a recent project that doesn't exist?");
321         } else {
322             recentFiles.remove(f);
323             recentFiles.add(f);
324         }
325     }

378     public void save() {
379         Preferences p = Preferences.userNodeForPackage(GUISaveState.class);
380 
381         p.putInt(TAB_SIZE, tabSize);
382 
383         p.putFloat(FONT_SIZE, fontSize);
384 
385         try {
386             p.put(STARTERDIRECTORY, starterDirectoryForLoadBugs.getCanonicalPath());
387         } catch (IOException e) {
388             Debug.println(e);
389         }
390         int sorterLength = MainFrame.getInstance().getSorter().getColumnCount();
391         ArrayList<Sortables> sortables = MainFrame.getInstance().getSorter().getOrder();
392         p.putInt(GUISaveState.SORTERTABLELENGTH, sorterLength);
393 
394         String[] sorterKeys = GUISaveState.generateSorterKeys(sorterLength);
395         for (int x = 0; x < sorterKeys.length; x++) {
396             p.put(sorterKeys[x], sortables.get(x).prettyName);
397         }
398 
399         p.putInt(GUISaveState.PREVCOMMENTSSIZE, previousComments.size());
400 
401         for (int x = 0; x < previousComments.size(); x++) {
402             String comment = previousComments.get(x);
403             p.put(GUISaveState.COMMENTKEYS[x], comment);
404         }
405 
406         int size = recentFiles.size();
407         while (recentFiles.size() > MAXNUMRECENTPROJECTS) {
408             recentFiles.remove(0);
409         }
410 
411         p.putInt(GUISaveState.NUMPROJECTS, Math.min(size, MAXNUMRECENTPROJECTS));
412         for (int x = 0; x < Math.min(size, MAXNUMRECENTPROJECTS); x++) {
413             File file = recentFiles.get(x);
414             p.put(GUISaveState.RECENTPROJECTKEYS[x], file.getAbsolutePath());
415         }
416 
417         p.putByteArray(DOCKINGLAYOUT, dockingLayout);
418 
419         p.put(FRAME_BOUNDS, frameBounds.x + "," + frameBounds.y + "," + frameBounds.width + "," + frameBounds.height);
420         p.putInt(EXTENDED_WINDOW_STATE, extendedWindowState);
421 
422         p.putInt(SPLIT_MAIN, splitMain);
423         p.putDouble(SPLIT_SUMMARY_NEW, splitSummary);
424         p.putInt(SPLIT_TOP, splitTop);
425         p.putInt(SPLIT_TREE_COMMENTS, splitTreeComments);
426         p.putInt(PACKAGE_PREFIX_SEGEMENTS, packagePrefixSegments);
427 
428         p.put(ENABLED_PLUGINS, StringUtils.join(enabledPlugins, ','));
429         p.put(DISABLED_PLUGINS, StringUtils.join(disabledPlugins, ','));
430         p.put(CUSTOM_PLUGINS, StringUtils.join(customPlugins, ' '));
431     }

565     SorterTableColumnModel getStarterTable() {
566         if (starterTable != null) {
567             return starterTable;
568         }
569 
570         if (useDefault || sortColumns == null) {
571             starterTable = new SorterTableColumnModel(GUISaveState.DEFAULT_COLUMN_HEADERS);
572         } else {
573             starterTable = new SorterTableColumnModel(sortColumns);
574         }
575 
576         return starterTable;
