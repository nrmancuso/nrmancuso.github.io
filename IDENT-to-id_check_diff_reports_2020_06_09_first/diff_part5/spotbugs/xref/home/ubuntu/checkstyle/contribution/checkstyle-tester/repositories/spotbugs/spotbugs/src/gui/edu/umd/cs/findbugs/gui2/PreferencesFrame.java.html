184     private void handleWindowClose() {
185         TreeModel bt = (MainFrame.getInstance().getTree().getModel());
186         if (bt instanceof BugTreeModel) {
187             ((BugTreeModel) bt).checkSorter();
188         }
189         Project project = getCurrentProject();
190 
191         boolean changed = pluginsAdded;
192         pluginsAdded = false;
193         List<String> enabledPlugins = new ArrayList<>();
194         List<String> disabledPlugins = new ArrayList<>();
195         for (Map.Entry<Plugin, EnabledSettings> entry : pluginEnabledStatus.entrySet()) {
196             Plugin plugin = entry.getKey();
197             EnabledSettings enabled = entry.getValue();
198             if (project != null) {
199                 Boolean newSetting = enabled.project;
200                 Boolean existingSetting = project.getPluginStatus(plugin);
201                 boolean sameSettings = Objects.equals(existingSetting, newSetting);
202                 if (!sameSettings) {
203                     project.setPluginStatusTrinary(plugin.getPluginId(), newSetting);
204                     changed = true;
205                 }
206             }
207             if (enabled.global) {
208                 enabledPlugins.add(plugin.getPluginId());
209             } else {
210                 disabledPlugins.add(plugin.getPluginId());
211             }
212             if (plugin.isGloballyEnabled() != enabled.global) {
213                 plugin.setGloballyEnabled(enabled.global);
214                 changed = true;
215             }
216         }
217 
218         if (changed && project != null) {
219             MainFrame.getInstance().updateBugTree();
220             MainFrame.getInstance().setProjectChanged(true);
221         }
222         if (project == null) {
223             GUISaveState.getInstance().setPluginsEnabled(enabledPlugins, disabledPlugins);
224             GUISaveState.getInstance().save();
225         }
226     }

233     private JPanel createPluginPane() {
234         final JPanel pluginPanel = new JPanel();
235         pluginPanel.setLayout(new BorderLayout());
236         pluginPanelCenter = new JPanel();
237 
238         pluginPanel.add(pluginPanelCenter, BorderLayout.CENTER);
239 
240         pluginPanelCenter.setBorder(new EmptyBorder(10,10,10,10));
241         pluginPanelCenter.setLayout(new GridBagLayout());
242 
243         JButton addButton = new JButton("Install new plugin...");
244         JPanel south = new JPanel();
245 
246         south.add(addButton);
247         addButton.addActionListener(e -> {
248             JFileChooser chooser = new JFileChooser();
249             chooser.addChoosableFileFilter(new FileFilter() {
250 
251                 @Override
252                 public String getDescription() {
253                     return "SpotBugs Plugin (*.jar)";
254                 }
255 
256                 @Override
257                 public boolean accept(File f) {
258                     if (f.isDirectory()) {
259                         return true;
260                     }
261                     if (!f.canRead()) {
262                         return false;
263                     }
264                     if (f.getName().endsWith(".jar")) {
265                         return true;
266                     }
267                     return false;
268                 }
269             });
270             chooser.setDialogTitle("Select a SpotBugs plugin jar");
271             int retvalue = chooser.showDialog(PreferencesFrame.this, "Install");
272 
273             if (retvalue == JFileChooser.APPROVE_OPTION) {
274                 File f = chooser.getSelectedFile();
275                 try {
276                     // load and enable for project (if loaded)
277                     Plugin plugin = Plugin.loadCustomPlugin(f, PreferencesFrame.this.getCurrentProject());
278 
279                     GUISaveState guiSaveState = GUISaveState.getInstance();
280                     URL url = f.toURI().toURL();
281                     // add to FBGUI custom plugins list
282                     guiSaveState.addCustomPlugin(url);
283                     // add to list of enabled plugins
284                     guiSaveState.setPluginEnabled(plugin.getPluginId());
285                     plugin.setGloballyEnabled(true);
286                     guiSaveState.save();
287                     pluginsAdded = true;
288                     rebuildPluginCheckboxes();
289 
290                 } catch (PluginException | MalformedURLException e1) {
291                     LOGGER.log(Level.WARNING, "Could not load " + f.getPath(), e1);
292                     JOptionPane.showMessageDialog(PreferencesFrame.this, "Could not load " + f.getPath()
293                     + "\n\n"
294                     + e1.getClass().getSimpleName() + ": " + e1.getMessage(),
295                     "Error Loading Plugin", JOptionPane.ERROR_MESSAGE);
296                 }
297             }
298         });
299 
300         pluginPanel.add(south, BorderLayout.SOUTH);
301 
302         return pluginPanel;
303     }

309     private void rebuildPluginCheckboxes() {
310         Project currentProject = getCurrentProject();
311         pluginPanelCenter.removeAll();
312         if (currentProject != null) {
313             GridBagConstraints g = new GridBagConstraints();
314             g.fill = GridBagConstraints.NONE;
315             g.insets = new Insets(5,5,5,5);
316             g.gridy = 0;
317             //        g.anchor = GridBagConstraints.WEST;
318             //        g.gridx = 1;
319             //        pluginPanelCenter.add(new JLabel("Global Setting"), g);
320             g.anchor = GridBagConstraints.CENTER;
321             g.gridx = 2;
322             pluginPanelCenter.add(new JLabel("Project Setting"), g);
323         }
324         Collection<Plugin> plugins = Plugin.getAllPlugins();
325         int added = 0;
326         for (final Plugin plugin : plugins) {
327             if (plugin.isCorePlugin()) {
328                 continue;
329             }
330             String text = plugin.getShortDescription();
331             String id = plugin.getPluginId();
332             if (text == null) {
333                 text = id;
334             }
335             final URL url = plugin.getPluginLoader().getURL();
336             String pluginUrlStr = url.toExternalForm();
337             if ("file".equals(url.getProtocol())) {
338                 try {
339                     pluginUrlStr = new File(URLDecoder.decode(url.getPath(), "UTF-8")).getAbsolutePath();
340                 } catch (UnsupportedEncodingException e) {
341 
342                 }
343             }
344             text = String.format("<html>%s<br><font style='font-weight:normal;font-style:italic'>%s",
345                     text, pluginUrlStr);
346 
347             EnabledSettings enabled = isEnabled(currentProject, plugin);
348             final JCheckBox checkGlobal = new JCheckBox(text, enabled.global);
349             final boolean cannotDisable = plugin.isEnabledByDefault() && plugin.cannotDisable();
350             if (cannotDisable) {
351                 if (!enabled.global) {
352                     throw new IllegalStateException(
353                             plugin.getPluginId() + " is enabled by default and cannot be disabled, but is disabled");
354                 }
355                 checkGlobal.setEnabled(false);
356             } else {
357                 checkGlobal.addMouseListener(new MouseAdapter() {
358 
359                     @Override
360                     public void mousePressed(MouseEvent e) {
361                         if (SwingUtilities.isRightMouseButton(e)) {
362                             JPopupMenu menu = new JPopupMenu();
363                             JMenuItem item = new JMenuItem("Uninstall " + plugin.getShortDescription() + "...");
364                             item.addActionListener(new UninstallClickListener(plugin, url));
365                             menu.add(item);
366                             menu.show(checkGlobal, e.getX(), e.getY());
367                         }
368                     }
369                 });
370                 checkGlobal.addActionListener(e -> pluginEnabledStatus.get(plugin).global = checkGlobal.isSelected());
371             }
372             checkGlobal.setVerticalTextPosition(SwingConstants.TOP);
373             String longText = plugin.getDetailedDescription();
374             if (longText != null) {
375                 checkGlobal.setToolTipText("<html>" + longText + "</html>");
376             }
377             pluginEnabledStatus.put(plugin, enabled);
378 
379             GridBagConstraints gbc = new GridBagConstraints();
380             gbc.fill = GridBagConstraints.BOTH;
381             gbc.weightx = 1;
382             gbc.gridx = 1;
383             gbc.insets = new Insets(0,5,0,5);
384             gbc.gridy = added+1;
385             gbc.anchor = GridBagConstraints.NORTHWEST;
386             pluginPanelCenter.add(checkGlobal, gbc);
387 
388             if (currentProject != null && !cannotDisable) {
389                 final JComboBox<String> combo = new WideComboBox<>(new String[]{"DEFAULT", "DISABLED", "ENABLED"});
390                 if (enabled.project == null) {
391                     combo.setSelectedIndex(0);
392                 } else {
393                     combo.setSelectedIndex(enabled.project ? 2 : 1);
394                 }
395                 combo.setRenderer(new DefaultListCellRenderer() {
396                     @Override
397                     public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
398                         if (index == -1) {
399                             if (value.equals("DEFAULT")) {
400                                 value = "Default";
401                             }
402                             if (value.equals("DISABLED")) {
403                                 value = "Disabled";
404                             }
405                             if (value.equals("ENABLED")) {
406                                 value = "Enabled";
407                             }
408                         } else {
409                             if (value.equals("DEFAULT")) {
410                                 value = "Default (use global setting)";
411                             }
412                             if (value.equals("DISABLED")) {
413                                 value = "Disabled for this project";
414                             }
415                             if (value.equals("ENABLED")) {
416                                 value = "Enabled for this project";
417                             }
418                         }
419                         return super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
420                     }
421                 });
422                 combo.addActionListener(e -> {
423                     Boolean[] array = { null, false, true };
424                     int i = combo.getSelectedIndex();
425                     if (i < 0 || i > 2)
426                     {
427                         return; // ??
428                     }
429                     pluginEnabledStatus.get(plugin).project = array[i];
430                 });
431                 gbc.gridx = 2;
432                 gbc.fill = GridBagConstraints.NONE;
433                 gbc.weightx = 0;
434                 pluginPanelCenter.add(combo, gbc);
435             }
436             added++;
437         }
438         if (added == 0) {
439             JLabel label = new JLabel("<html>No plugins are loaded.<br> " +
440                     "Try installing <u><font color=blue>fb-contrib</font></u> - or write your own<br>" +
441                     "plugin for your project's needs!");
442             label.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
443             label.addMouseListener(new MouseAdapter() {
444                 @Override
445                 public void mouseClicked(MouseEvent e) {
446                     try {
447                         LaunchBrowser.showDocument(new URL("https://sourceforge.net/projects/fb-contrib/"));
448                     } catch (MalformedURLException e1) {
449                         throw new IllegalStateException(e1);
450                     }
451                 }
452             });
453             label.setBorder(new EmptyBorder(10,10,10,10));
454             pluginPanelCenter.add(label);
455         }
456         PreferencesFrame.this.pack();
457     }

515     private void changeTabSize() {
516         int tabSize;
517 
518         try {
519             tabSize = Integer.decode(tabTextField.getText());
520         } catch (NumberFormatException exc) {
521             JOptionPane.showMessageDialog(instance, "Error in tab size field.", "Tab Size Error",
522                     JOptionPane.INFORMATION_MESSAGE);
523             return;
524         }
525 
526         if (tabSize < TAB_MIN || tabSize > TAB_MAX) {
527             JOptionPane.showMessageDialog(instance, "Tab size exceedes range (" + TAB_MIN + " - " + TAB_MAX + ").",
528                     "Tab Size Excedes Range", JOptionPane.INFORMATION_MESSAGE);
529             return;
530         }
531 
532         if (tabSize != GUISaveState.getInstance().getTabSize()) {
533             GUISaveState.getInstance().setTabSize(tabSize);
534             MainFrame.getInstance().getSourceCodeDisplayer().clearCache();
535             // This causes the GUI to redisplay the current code
536             MainFrame.getInstance().syncBugInformation();
537         }
538     }

540     private void changeFontSize() {
541         float fontSize;
542 
543         try {
544             fontSize = Float.parseFloat(fontTextField.getText());
545         } catch (NumberFormatException exc) {
546             JOptionPane.showMessageDialog(instance, "Error in font size field.", "Font Size Error",
547                     JOptionPane.INFORMATION_MESSAGE);
548             return;
549         }
550 
551         if (fontSize < FONT_MIN || fontSize > FONT_MAX) {
552             JOptionPane.showMessageDialog(instance, "Font size exceedes range (" + FONT_MIN + " - " + FONT_MAX + ").",
553                     "Font Size Exceedes Range", JOptionPane.INFORMATION_MESSAGE);
554             return;
555         }
556 
557         if (fontSize != GUISaveState.getInstance().getFontSize()) {
558             GUISaveState.getInstance().setFontSize(fontSize);
559             JOptionPane.showMessageDialog(instance, "To implement the new font size please restart SpotBugs.",
560                     "Changing Font",
561                     JOptionPane.INFORMATION_MESSAGE);
562         }
563     }

565     private void changePackagePrefixLength() {
566         int value;
567 
568         try {
569             value = Integer.parseInt(packagePrefixLengthTextField.getText());
570         } catch (NumberFormatException exc) {
571             JOptionPane.showMessageDialog(instance, "Error in package prefix length field.", "Package Prefix Length Error",
572                     JOptionPane.INFORMATION_MESSAGE);
573             return;
574         }
575 
576         if (value < 1 || value > 12) {
577             JOptionPane.showMessageDialog(instance, "package prefix length exceedes range (" + 1 + " - " + 12 + ").",
578                     "package prefix lengthe exceedes range", JOptionPane.INFORMATION_MESSAGE);
579             return;
580         }
581 
582         if (value != GUISaveState.getInstance().getPackagePrefixSegments()) {
583             GUISaveState.getInstance().setPackagePrefixSegments(value);
584             BugTreeModel bt = (BugTreeModel) (MainFrame.getInstance().getTree().getModel());
585             bt.needToRebuild();
586             bt.checkSorter();
587         }
588

600     private JPanel createFilterPane() {
601         JButton addButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.add_dot_btn", "Add..."));
602         JButton removeButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.remove_btn", "Remove"));
603         JButton removeAllButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.remove_all_btn", "Remove All"));
604         JPanel filterPanel = new JPanel();
605         filterPanel.setLayout(new GridBagLayout());
606         GridBagConstraints gbc = new GridBagConstraints();
607 
608         gbc.gridheight = 1;
609         gbc.gridwidth = 1;
610         gbc.gridx = 0;
611         gbc.gridy = 0;
612         gbc.fill = GridBagConstraints.BOTH;
613         gbc.weightx = 1;
614         gbc.weighty = 1;
615         filterPanel.add(new JLabel("<HTML>These rules control which bugs are shown and which are hidden in this project"), gbc);
616 
617         gbc.gridheight = 4;
618         gbc.gridwidth = 1;
619         gbc.gridx = 0;
620         gbc.gridy = 1;
621         gbc.fill = GridBagConstraints.BOTH;
622         gbc.weightx = 1;
623         gbc.weighty = 1;
624         filterPanel.add(new JScrollPane(filterCheckBoxList), gbc);
625         updateFilterPanel();
626 
627         gbc.gridheight = 1;
628         gbc.gridx = 1;
629         gbc.gridy = 1;
630         gbc.fill = GridBagConstraints.HORIZONTAL;
631         gbc.weightx = 0;
632         gbc.weighty = 0;
633         filterPanel.add(addButton, gbc);
634         addButton.addActionListener(evt -> NewFilterFrame.open());
635 
636         gbc.gridy = 2;
637         gbc.insets = new Insets(5, 0, 0, 0);
638         filterPanel.add(removeButton, gbc);
639         removeButton.addActionListener(evt -> {
640             Object[] selected = filterCheckBoxList.getSelectedValues();
641             if (selected.length == 0) {
642                 return;
643             }
644             for (Object o : selected) {
645                 MatchBox box = (MatchBox) o;
646                 MainFrame.getInstance().getProject().getSuppressionFilter().removeChild(box.getMatcher());
647             }
648             FilterActivity.notifyListeners(FilterListener.Action.UNFILTERING, null);
649             MainFrame.getInstance().setProjectChanged(true);
650             updateFilterPanel();
651         });
652         gbc.gridy = 3;
653         gbc.weighty = 0;
654         gbc.insets = new Insets(5, 0, 0, 0);
655         filterPanel.add(removeAllButton, gbc);
656         removeAllButton.addActionListener(evt -> {
657             boolean needsRebuild = false;
658             Filter suppressionFilter = MainFrame.getInstance().getProject().getSuppressionFilter();
659             if (!suppressionFilter.isEmpty()) {
660                 needsRebuild = true;
661             }
662             suppressionFilter.clear();
663 
664             if (needsRebuild) {
665                 // filters being cleared were disabled
666                 FilterActivity.notifyListeners(FilterListener.Action.UNFILTERING, null);
667             }
668             MainFrame.getInstance().setProjectChanged(true);
669             updateFilterPanel();
670         });
671         gbc.gridy = 4;
672         gbc.weighty = 1;
673         gbc.insets = new Insets(0, 0, 0, 0);
674         filterPanel.add(Box.createGlue(), gbc);
675 
676         return filterPanel;
677     }

679     private static class MatchBox extends JCheckBox {
680         Matcher m;
681 
682         MatchBox(String text, Matcher m) {
683             super(text);
684             this.m = m;
685         }
686 
687         Matcher getMatcher() {
688             return m;
689         }
690     }
691 
692     void updateFilterPanel() {
693         ArrayList<MatchBox> boxes = new ArrayList<>();
694         final Filter f = MainFrame.getInstance().getProject().getSuppressionFilter();
695 
696         for (final Matcher m : f.getChildren()) {
697             MatchBox box = new MatchBox(m.toString(), m);
698             box.addItemListener(evt -> {
699                 boolean isSelected = ((JCheckBox) evt.getSource()).isSelected();
700                 boolean wasSelected = f.isEnabled(m);
701                 if (isSelected == wasSelected) {
702                     return;
703                 }
704                 f.setEnabled(m, isSelected);
705                 updateFilters(isSelected);
706 
707             });
708             box.setSelected(f.isEnabled(m));
709             boxes.add(box);
710         }
711 
712         filterCheckBoxList.setListData(boxes.toArray(new MatchBox[boxes.size()]));
713     }

719         MainFrame.getInstance().setProjectChanged(true);

722     private class UninstallClickListener implements ActionListener {
723         private final Plugin plugin;
724         private final URL url;
725 
726         public UninstallClickListener(Plugin plugin, URL url) {
727             this.plugin = plugin;
728             this.url = url;
729         }
730 
731         @Override
732         public void actionPerformed(ActionEvent e) {
733             int result = JOptionPane.showOptionDialog(PreferencesFrame.this,
734                     "Are you sure you want to uninstall " + plugin.getShortDescription() + "?" +
735                             "\n\nNo files will be deleted from your computer.", "",
736                             JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null,
737                             new Object[]{"Uninstall", "Cancel"}, "Cancel");
738             if (result == 0) {
739                 if (!GUISaveState.getInstance().removeCustomPlugin(url)) {
740                     if ("file".equals(url.getProtocol())) {
741                         String path = url.toExternalForm();
742                         try {
743                             path = new File(URLDecoder.decode(url.getPath(), "UTF-8")).getAbsolutePath();
744                         } catch (UnsupportedEncodingException ex) {
745                         }
746                         try {
747                             StringSelection contents = new StringSelection(path);
748                             Toolkit.getDefaultToolkit().getSystemClipboard().setContents(contents, contents);
749                         } catch (Exception e1) {
750                         }
751                         JOptionPane.showMessageDialog(PreferencesFrame.this,
752                                 "The plugin could not be uninstalled automatically.\n\n" +
753                                         "You can try to delete this plugin manually: \n"
754                                         + path + "\n\n(This path has been copied to your clipboard)",
755                                         "Error", JOptionPane.ERROR_MESSAGE);
756                     } else {
757                         JOptionPane.showMessageDialog(PreferencesFrame.this,
758                                 "This plugin is not actually in the list of plugins...\n" +
759                                         "Not sure what to do...\n "
760                                         + url.toExternalForm()
761                                         + "\n\nPlugin URL's:\n" +
762                                         GUISaveState.getInstance().getCustomPlugins(),
763                                         "Error", JOptionPane.ERROR_MESSAGE);
764                     }
765                 } else {
766                     JOptionPane.showMessageDialog(PreferencesFrame.this,
767                             "Changes will take effect after you restart SpotBugs.");
768                 }
769             }
770         }
771     }
