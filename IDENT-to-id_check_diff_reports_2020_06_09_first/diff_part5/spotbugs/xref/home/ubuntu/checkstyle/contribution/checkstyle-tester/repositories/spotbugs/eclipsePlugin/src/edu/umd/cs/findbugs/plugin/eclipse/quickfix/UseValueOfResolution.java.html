49  /**
50   * The <CODE>UseValueOfResolution</CODE> replace the inefficient creation of an
51   * instance, by the static <CODE>valueOf(...)</CODE> method.
52   *
53   * @see <a
54   *      href="http://findbugs.sourceforge.net/bugDescriptions.html#DM_BOOLEAN_CTOR">DM_BOOLEAN_CTOR</a>
55   * @see <a
56   *      href="http://findbugs.sourceforge.net/bugDescriptions.html#DM_FP_NUMBER_CTOR">DM_FP_NUMBER_CTOR</a>
57   * @see <a
58   *      href="http://findbugs.sourceforge.net/bugDescriptions.html#DM_NUMBER_CTOR">DM_NUMBER_CTOR</a>
59   * @author <a href="mailto:twyss@hsr.ch">Thierry Wyss</a>
60   * @author <a href="mailto:mbusarel@hsr.ch">Marco Busarello</a>
61   * @version 1.0
62   */
63  public class UseValueOfResolution extends BugResolution {
64  
65      private static final String VALUE_OF_METHOD_NAME = "valueOf";
66  
67      private static final Set<String> primitiveWrapperClasses = new HashSet<>();
68  
69      static {
70          primitiveWrapperClasses.add("java.lang.Double");
71          primitiveWrapperClasses.add("java.lang.Integer");
72          primitiveWrapperClasses.add("java.lang.Boolean");
73          primitiveWrapperClasses.add("java.lang.Float");
74      }
75  
76      private boolean isDouble;
77  
78      private boolean isFloatingPoint;
79  
80      public boolean isStaticImport() {
81          // This can be changed to true to add the static import and base the fix on that.
82          return false;
83      }
84  
85      @Override
86      protected void repairBug(ASTRewrite rewrite, CompilationUnit workingUnit, BugInstance bug) throws BugResolutionException {
87          Assert.isNotNull(rewrite);
88          Assert.isNotNull(workingUnit);
89  
90          ClassInstanceCreation primitiveTypeCreation = findPrimitiveTypeCreation(getASTNode(workingUnit,
91                  bug.getPrimarySourceLineAnnotation()));
92          if (primitiveTypeCreation == null) {
93              throw new BugResolutionException("Primitive type creation not found.");
94          }
95          MethodInvocation valueOfInvocation = createValueOfInvocation(rewrite, workingUnit, primitiveTypeCreation);
96          rewrite.replace(primitiveTypeCreation, valueOfInvocation, null);
97      }
98  
99      @CheckForNull
100     protected ClassInstanceCreation findPrimitiveTypeCreation(ASTNode node) {
101         PrimitiveTypeCreationFinder visitor = new PrimitiveTypeCreationFinder();
102         node.accept(visitor);
103         return visitor.getPrimitiveTypeCreation();
104     }
105 
106     protected MethodInvocation createValueOfInvocation(ASTRewrite rewrite, CompilationUnit compilationUnit,
107             ClassInstanceCreation primitiveTypeCreation) {
108         Assert.isNotNull(rewrite);
109         Assert.isNotNull(primitiveTypeCreation);
110 
111         final AST ast = rewrite.getAST();
112         MethodInvocation valueOfInvocation = ast.newMethodInvocation();
113         valueOfInvocation.setName(ast.newSimpleName(VALUE_OF_METHOD_NAME));
114 
115         ITypeBinding binding = primitiveTypeCreation.getType().resolveBinding();
116         if (isStaticImport()) {
117             addStaticImports(rewrite, compilationUnit, binding.getQualifiedName() + "." + VALUE_OF_METHOD_NAME);
118         } else {
119             valueOfInvocation.setExpression(ast.newSimpleName(binding.getName()));
120         }
121 
122         List<?> arguments = primitiveTypeCreation.arguments();
123         List<Expression> newArguments = valueOfInvocation.arguments();
124         for (Object argument : arguments) {
125             Expression expression = (Expression) rewrite.createCopyTarget((ASTNode) argument);
126             newArguments.add(expression);
127         }
128 
129         return valueOfInvocation;
130     }
131 
132     @Override
133     protected boolean resolveBindings() {
134         return true;
135     }
136 
137     @Override
138     protected ASTVisitor getCustomLabelVisitor() {
139         return new PrimitiveTypeCreationFinder();
140     }
141 
142     @Override
143     public void setOptions(@Nonnull Map<String, String> options) {
144         // This setup (having two separate plugin.xml entries) is done to show off the
145         // ApplicabilityVisitor, although it could be done without, just by having a
146         // slightly fancier (and uglier) getLabelReplacement()
147         isFloatingPoint = Boolean.parseBoolean(options.get("isFloatingPoint"));
148         isDouble = Boolean.parseBoolean(options.get("isDouble"));
149     }
150 
151     @Override
152     protected ASTVisitor getApplicabilityVisitor() {
153         return new PrimitiveTypeCreationFinder();
154     }
155 
156     protected class PrimitiveTypeCreationFinder extends ASTVisitor implements CustomLabelVisitor, ApplicabilityVisitor {
157 
158         private ClassInstanceCreation primitiveTypeCreation = null;
159 
160         @Override
161         public boolean visit(ClassInstanceCreation node) {
162             if (primitiveTypeCreation == null) {
163                 if (!isPrimitiveTypeCreation(node)) {
164                     return true;
165                 }
166                 this.primitiveTypeCreation = node;
167             }
168             return false;
169         }
170 
171         public ClassInstanceCreation getPrimitiveTypeCreation() {
172             return primitiveTypeCreation;
173         }
174 
175         private boolean isPrimitiveTypeCreation(ClassInstanceCreation node) {
176             ITypeBinding typeBinding = node.resolveTypeBinding();
177             return primitiveWrapperClasses.contains(typeBinding.getQualifiedName());
178         }
179 
180         @Override
181         public String getLabelReplacement() {
182             // returns what is in the constructor arguments
183             if (primitiveTypeCreation == null || primitiveTypeCreation.arguments().isEmpty()) {
184                 return "...";   //safe return value
185             }
186             // can safely use toString() here because it's user facing, not being used as actual code
187             return primitiveTypeCreation.arguments().get(0).toString();
188         }
189 
190         @Override
191         public boolean isApplicable() {
192             // we must have found a primativeTypeCreation.  If we are not dealing with
193             // DM_FP_NUMBER_CTOR, it's automatically applicable.  Otherwise,
194             // we must match the isDouble argument with actually resolving a double
195             return primitiveTypeCreation != null
196                     && (!isFloatingPoint || (isDouble == "java.lang.Double".equals(primitiveTypeCreation.resolveTypeBinding()
197                             .getQualifiedName())));
198         }
199 
200     }
