22  /**
23   * Represents a function with 4 arguments.
24   *
25   * @param <T1> argument 1 of the function
26   * @param <T2> argument 2 of the function
27   * @param <T3> argument 3 of the function
28   * @param <T4> argument 4 of the function
29   * @param <R> return type of the function
30   * @author Daniel Dietrich
31   */
32  @FunctionalInterface
33  public interface Function4<T1, T2, T3, T4, R> extends Lambda<R> {
34  
35      /**
36       * The <a href="https://docs.oracle.com/javase/8/docs/api/index.html">serial version uid</a>.
37       */
38      long serialVersionUID = 1L;
39  
40      /**
41       * Creates a {@code Function4} based on
42       * <ul>
43       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">method reference</a></li>
44       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/lambdaexpressions.html#syntax">lambda expression</a></li>
45       * </ul>
46       *
47       * Examples (w.l.o.g. referring to Function1):
48       * <pre><code>// using a lambda expression
49       * Function1&lt;Integer, Integer&gt; add1 = Function1.of(i -&gt; i + 1);
50       *
51       * // using a method reference (, e.g. Integer method(Integer i) { return i + 1; })
52       * Function1&lt;Integer, Integer&gt; add2 = Function1.of(this::method);
53       *
54       * // using a lambda reference
55       * Function1&lt;Integer, Integer&gt; add3 = Function1.of(add1::apply);
56       * </code></pre>
57       * <p>
58       * <strong>Caution:</strong> Reflection loses type information of lambda references.
59       * <pre><code>// type of a lambda expression
60       * Type&lt;?, ?&gt; type1 = add1.getType(); // (Integer) -&gt; Integer
61       *
62       * // type of a method reference
63       * Type&lt;?, ?&gt; type2 = add2.getType(); // (Integer) -&gt; Integer
64       *
65       * // type of a lambda reference
66       * Type&lt;?, ?&gt; type3 = add3.getType(); // (Object) -&gt; Object
67       * </code></pre>
68       *
69       * @param methodReference (typically) a method reference, e.g. {@code Type::method}
70       * @param <R> return type
71       * @param <T1> 1st argument
72       * @param <T2> 2nd argument
73       * @param <T3> 3rd argument
74       * @param <T4> 4th argument
75       * @return a {@code Function4}
76       */
77      static <T1, T2, T3, T4, R> Function4<T1, T2, T3, T4, R> of(Function4<T1, T2, T3, T4, R> methodReference) {
78          return methodReference;
79      }
80  
81      /**
82       * Lifts the given {@code partialFunction} into a total function that returns an {@code Option} result.
83       *
84       * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
85       * @param <R> return type
86       * @param <T1> 1st argument
87       * @param <T2> 2nd argument
88       * @param <T3> 3rd argument
89       * @param <T4> 4th argument
90       * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Some(result)}
91       *         if the function is defined for the given arguments, and {@code None} otherwise.
92       */
93      @SuppressWarnings("RedundantTypeArguments")
94      static <T1, T2, T3, T4, R> Function4<T1, T2, T3, T4, Option<R>> lift(Function4<? super T1, ? super T2, ? super T3, ? super T4, ? extends R> partialFunction) {
95          return (t1, t2, t3, t4) -> Try.<R>of(() -> partialFunction.apply(t1, t2, t3, t4)).toOption();
96      }
97  
98      /**
99       * Lifts the given {@code partialFunction} into a total function that returns an {@code Try} result.
100      *
101      * @param partialFunction a function that is not defined for all values of the domain (e.g. by throwing)
102      * @param <R> return type
103      * @param <T1> 1st argument
104      * @param <T2> 2nd argument
105      * @param <T3> 3rd argument
106      * @param <T4> 4th argument
107      * @return a function that applies arguments to the given {@code partialFunction} and returns {@code Success(result)}
108      *         if the function is defined for the given arguments, and {@code Failure(throwable)} otherwise.
109      */
110     static <T1, T2, T3, T4, R> Function4<T1, T2, T3, T4, Try<R>> liftTry(Function4<? super T1, ? super T2, ? super T3, ? super T4, ? extends R> partialFunction) {
111         return (t1, t2, t3, t4) -> Try.of(() -> partialFunction.apply(t1, t2, t3, t4));
112     }
113 
114     /**
115      * Narrows the given {@code Function4<? super T1, ? super T2, ? super T3, ? super T4, ? extends R>} to {@code Function4<T1, T2, T3, T4, R>}
116      *
117      * @param f A {@code Function4}
118      * @param <R> return type
119      * @param <T1> 1st argument
120      * @param <T2> 2nd argument
121      * @param <T3> 3rd argument
122      * @param <T4> 4th argument
123      * @return the given {@code f} instance as narrowed type {@code Function4<T1, T2, T3, T4, R>}
124      */
125     @SuppressWarnings("unchecked")
126     static <T1, T2, T3, T4, R> Function4<T1, T2, T3, T4, R> narrow(Function4<? super T1, ? super T2, ? super T3, ? super T4, ? extends R> f) {
127         return (Function4<T1, T2, T3, T4, R>) f;
128     }
129 
130     /**
131      * Applies this function to 4 arguments and returns the result.
132      *
133      * @param t1 argument 1
134      * @param t2 argument 2
135      * @param t3 argument 3
136      * @param t4 argument 4
137      * @return the result of function application
138      * 
139      */
140     R apply(T1 t1, T2 t2, T3 t3, T4 t4);
141 
142     /**
143      * Applies this function partially to one argument.
144      *
145      * @param t1 argument 1
146      * @return a partial application of this function
147      */
148     default Function3<T2, T3, T4, R> apply(T1 t1) {
149         return (T2 t2, T3 t3, T4 t4) -> apply(t1, t2, t3, t4);
150     }
151 
152     /**
153      * Applies this function partially to two arguments.
154      *
155      * @param t1 argument 1
156      * @param t2 argument 2
157      * @return a partial application of this function
158      */
159     default Function2<T3, T4, R> apply(T1 t1, T2 t2) {
160         return (T3 t3, T4 t4) -> apply(t1, t2, t3, t4);
161     }
162 
163     /**
164      * Applies this function partially to three arguments.
165      *
166      * @param t1 argument 1
167      * @param t2 argument 2
168      * @param t3 argument 3
169      * @return a partial application of this function
170      */
171     default Function1<T4, R> apply(T1 t1, T2 t2, T3 t3) {
172         return (T4 t4) -> apply(t1, t2, t3, t4);
173     }
174 
175     @Override
176     default int arity() {
177         return 4;
178     }
179 
180     /**
181      * Returns a function that always returns the constant
182      * value that you give in parameter.
183      *
184      * @param <T1> generic parameter type 1 of the resulting function
185      * @param <T2> generic parameter type 2 of the resulting function
186      * @param <T3> generic parameter type 3 of the resulting function
187      * @param <T4> generic parameter type 4 of the resulting function
188      * @param <R> the result type
189      * @param value the value to be returned
190      * @return a function always returning the given value
191      */
192     static <T1, T2, T3, T4, R> Function4<T1, T2, T3, T4, R> constant(R value) {
193         return (t1, t2, t3, t4) -> value;
194     }
195 
196     @Override
197     default Function1<T1, Function1<T2, Function1<T3, Function1<T4, R>>>> curried() {
198         return t1 -> t2 -> t3 -> t4 -> apply(t1, t2, t3, t4);
199     }
200 
201     @Override
202     default Function1<Tuple4<T1, T2, T3, T4>, R> tupled() {
203         return t -> apply(t._1, t._2, t._3, t._4);
204     }
205 
206     @Override
207     default Function4<T4, T3, T2, T1, R> reversed() {
208         return (t4, t3, t2, t1) -> apply(t1, t2, t3, t4);
209     }
210 
211     @Override
212     default Function4<T1, T2, T3, T4, R> memoized() {
213         if (isMemoized()) {
214             return this;
215         } else {
216             final Map<Tuple4<T1, T2, T3, T4>, R> cache = new HashMap<>();
217             return (Function4<T1, T2, T3, T4, R> & Memoized) (t1, t2, t3, t4)
218                     -> Memoized.of(cache, Tuple.of(t1, t2, t3, t4), tupled());
219         }
220     }
221 
222     /**
223      * Returns a composed function that first applies this Function4 to the given argument and then applies
224      * {@linkplain Function} {@code after} to the result.
225      *
226      * @param <V> return type of after
227      * @param after the function applied after this
228      * @return a function composed of this and after
229      * @throws NullPointerException if after is null
230      */
231     default <V> Function4<T1, T2, T3, T4, V> andThen(Function<? super R, ? extends V> after) {
232         Objects.requireNonNull(after, "after is null");
233         return (t1, t2, t3, t4) -> after.apply(apply(t1, t2, t3, t4));
234     }
235 
236 }

238 interface Function4Module {
239 
240     // DEV-NOTE: we do not plan to expose this as public API
241     @SuppressWarnings("unchecked")
242     static <T extends Throwable, R> R sneakyThrow(Throwable t) throws T {
243         throw (T) t;
244     }
