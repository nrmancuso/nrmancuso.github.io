9   /**
10   * This exception is temporarily used during development in order to indicate that an implementation is missing.
11   * <p>
12   * The idiomatic way is to use one of {@link API#TODO()} and {@link API#TODO(String)}.
13   */
14  public class NotImplementedError extends Error {
15  
16      private static final long serialVersionUID = 1L;
17  
18      /**
19       * Creates a {@code NotImplementedError} containing the message "an implementation is missing".
20       */
21      public NotImplementedError() {
22          super("An implementation is missing.");
23      }
24  
25      /**
26       * Creates a {@code NotImplementedError} containing the given {@code message}.
27       *
28       * @param message A text that describes the error
29       */
30      public NotImplementedError(String message) {
31          super(message);
32      }
33  }
