43      /**
44       * Creates a {@code CheckedFunction7} based on
45       * <ul>
46       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/methodreferences.html">method reference</a></li>
47       * <li><a href="https://docs.oracle.com/javase/tutorial/java/javaOO/lambdaexpressions.html#syntax">lambda expression</a></li>
48       * </ul>
49       *
50       * Examples (w.l.o.g. referring to Function1):
51       * <pre><code>// using a lambda expression
52       * Function1&lt;Integer, Integer&gt; add1 = Function1.of(i -&gt; i + 1);
53       *
54       * // using a method reference (, e.g. Integer method(Integer i) { return i + 1; })
55       * Function1&lt;Integer, Integer&gt; add2 = Function1.of(this::method);
56       *
57       * // using a lambda reference
58       * Function1&lt;Integer, Integer&gt; add3 = Function1.of(add1::apply);
59       * </code></pre>
60       * <p>
61       * <strong>Caution:</strong> Reflection loses type information of lambda references.
62       * <pre><code>// type of a lambda expression
63       * Type&lt;?, ?&gt; type1 = add1.getType(); // (Integer) -&gt; Integer
64       *
65       * // type of a method reference
66       * Type&lt;?, ?&gt; type2 = add2.getType(); // (Integer) -&gt; Integer
67       *
68       * // type of a lambda reference
69       * Type&lt;?, ?&gt; type3 = add3.getType(); // (Object) -&gt; Object
70       * </code></pre>
71       *
72       * @param methodReference (typically) a method reference, e.g. {@code Type::method}
73       * @param <R> return type
74       * @param <T1> 1st argument
75       * @param <T2> 2nd argument
76       * @param <T3> 3rd argument
77       * @param <T4> 4th argument
78       * @param <T5> 5th argument
79       * @param <T6> 6th argument
80       * @param <T7> 7th argument
81       * @return a {@code CheckedFunction7}
82       */
83      static <T1, T2, T3, T4, T5, T6, T7, R> CheckedFunction7<T1, T2, T3, T4, T5, T6, T7, R> of(CheckedFunction7<T1, T2, T3, T4, T5, T6, T7, R> methodReference) {
84          return methodReference;
85      }

280             return (CheckedFunction7<T1, T2, T3, T4, T5, T6, T7, R> & Memoized) (t1, t2, t3, t4, t5, t6, t7)
281                     -> Memoized.of(cache, Tuple.of(t1, t2, t3, t4, t5, t6, t7), t -> Try.of(() -> apply(t1, t2, t3, t4, t5, t6, t7)).get());

285     /**
286      * Return a composed function that first applies this CheckedFunction7 to the given arguments and in case of throwable
287      * try to get value from {@code recover} function with same arguments and throwable information.
288      *
289      * @param recover the function applied in case of throwable
290      * @return a function composed of this and recover
291      * @throws NullPointerException if recover is null
292      */
293     default Function7<T1, T2, T3, T4, T5, T6, T7, R> recover(Function<? super Throwable, ? extends Function7<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? super T7, ? extends R>> recover) {
294         Objects.requireNonNull(recover, "recover is null");
295         return (t1, t2, t3, t4, t5, t6, t7) -> {
296             try {
297                 return this.apply(t1, t2, t3, t4, t5, t6, t7);
298             } catch (Throwable throwable) {
299                 final Function7<? super T1, ? super T2, ? super T3, ? super T4, ? super T5, ? super T6, ? super T7, ? extends R> func = recover.apply(throwable);
300                 Objects.requireNonNull(func, () -> "recover return null for " + throwable.getClass() + ": " + throwable.getMessage());
301                 return func.apply(t1, t2, t3, t4, t5, t6, t7);
302             }
303         };
304     }
