43  	@Test
44  	public void testShadowInterfaceMethodsCanOverrideObjectMethods() {
45  		//contract: Interface (made by reflection) method equals overrides Object#equals
46  		Factory f = new Launcher().getFactory();
47  		CtType<?> iface = f.Interface().get(Comparator.class);
48  		assertTrue(iface.isShadow());
49  		CtMethod<?> comparatorEquals = iface.getMethodsByName("equals").get(0);
50  		
51  		CtType<?> object = f.Class().get(Object.class);
52  		assertTrue(object.isShadow());
53  		CtMethod<?> objectEquals = object.getMethodsByName("equals").get(0);
54  		
55  		assertTrue(comparatorEquals.isOverriding(objectEquals));
56  	}

58  	@Test
59  	public void testInterfaceMethodsCanOverrideObjectMethods() throws Exception {
60  		//contract: Interface (made from sources) method equals overrides Object#equals
61  		Launcher launcher = new Launcher();
62  		Factory f = launcher.getFactory();
63  		SpoonModelBuilder comp = launcher.createCompiler();
64  		comp.addInputSources(SpoonResourceHelper.resources("./src/test/java/spoon/test/method_overriding/testclasses2/ObjectInterface.java"));
65  		comp.build();
66  		CtType<?> iface = f.Interface().get(ObjectInterface.class);
67  		assertFalse(iface.isShadow());
68  		CtMethod<?> ifaceEquals = iface.getMethodsByName("equals").get(0);
69  		
70  		CtType<?> object = iface.getFactory().Class().get(Object.class);
71  		assertTrue(object.isShadow());
72  		CtMethod<?> objectEquals = object.getMethodsByName("equals").get(0);
73  		
74  		assertTrue(ifaceEquals.isOverriding(objectEquals));
75  	}

78  	public void testMethodOverride() {
79  		checkMethodOverride((m1, m2)->m1.isOverriding(m2));
80  	}

83  	public void testMethodOverrideByReference() {
84  		checkMethodOverride((m1, m2)->m1.getReference().isOverriding(m2.getReference()));
85  	}

87  	private void checkMethodOverride(BiFunction<CtMethod<?>, CtMethod<?>, Boolean> isOverriding) {
88  		Factory factory = ModelUtils.build(new File("src/test/java/spoon/test/method_overriding/testclasses").listFiles());
89  		Map<String, List<CtMethod>> methodsByName = new HashMap<>();
90  		factory.getModel().filterChildren(new TypeFilter<>(CtMethod.class)).forEach((CtMethod m)->{
91  			List<CtMethod> methods = methodsByName.get(m.getSimpleName());
92  			if(methods==null) {
93  				methods = new ArrayList<>();
94  				methodsByName.put(m.getSimpleName(), methods);
95  			}
96  			methods.add(m);
97  		});
98  		assertFalse(methodsByName.isEmpty());
99  		for (Map.Entry<String, List<CtMethod>> e : methodsByName.entrySet()) {
100 			combine(e.getValue(), 0, isOverriding);
101 		}
102 	}
103 
104 	private void combine(List<CtMethod> value, int start, BiFunction<CtMethod<?>, CtMethod<?>, Boolean> isOverriding) {
105 		CtMethod m1 = value.get(start);
106 		if(start+1 < value.size()) {
107 			for (CtMethod m2 : value.subList(start+1, value.size())) {
108 				if(m1.getDeclaringType().isSubtypeOf(m2.getDeclaringType().getReference())) {
109 					checkOverride(m1, m2, isOverriding);
110 				} else if(m2.getDeclaringType().isSubtypeOf(m1.getDeclaringType().getReference())) {
111 					checkOverride(m2, m1, isOverriding);
112 				} else {
113 					checkNotOverride(m1, m2, isOverriding);
114 				}
115 			}
116 			combine(value, start+1, isOverriding);
117 		}
118 	}
119 
120 	private void checkOverride(CtMethod m1, CtMethod m2, BiFunction<CtMethod<?>, CtMethod<?>, Boolean> isOverriding) {
121 		assertTrue(descr(m1)+" overriding "+descr(m2), 		isOverriding.apply(m1, m2));
122 		assertFalse(descr(m2)+" NOT overriding "+descr(m1), isOverriding.apply(m2, m1));
123 	}
124 
125 	private void checkNotOverride(CtMethod m1, CtMethod m2, BiFunction<CtMethod<?>, CtMethod<?>, Boolean> isOverriding) {
126 		assertFalse(descr(m1)+" NOT overriding "+descr(m2), isOverriding.apply(m1, m2));
127 		assertFalse(descr(m2)+" NOT overriding "+descr(m1), isOverriding.apply(m2, m1));
128 	}
129 
130 	private String descr(CtMethod m) {
131 		return m.getDeclaringType().getSimpleName()+"#"+m.getSimpleName();
132 	}
