6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtCodeSnippetStatement;
10  import spoon.reflect.code.CtStatement;
11  import spoon.reflect.declaration.CtCodeSnippet;
12  import spoon.reflect.visitor.CtVisitor;
13  import spoon.support.compiler.SnippetCompilationError;
14  import spoon.support.compiler.SnippetCompilationHelper;
15  
16  import static spoon.reflect.path.CtRole.SNIPPET;
17  
18  public class CtCodeSnippetStatementImpl extends CtStatementImpl implements CtCodeSnippetStatement {
19  
20  	private static final long serialVersionUID = 1L;
21  
22  	@Override
23  	public void accept(CtVisitor visitor) {
24  		visitor.visitCtCodeSnippetStatement(this);
25  	}
26  
27  	@MetamodelPropertyField(role = SNIPPET)
28  	String value;
29  
30  	@Override
31  	public String getValue() {
32  		return value;
33  	}
34  
35  	@Override
36  	public <C extends CtCodeSnippet> C setValue(String value) {
37  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, SNIPPET, value, this.value);
38  		this.value = value;
39  		return (C) this;
40  	}
41  
42  	@Override
43  	@SuppressWarnings("unchecked")
44  	public <S extends CtStatement> S compile() throws SnippetCompilationError {
45  		return (S) SnippetCompilationHelper.compileStatement(this);
46  	}
47  
48  	@Override
49  	public CtCodeSnippetStatement clone() {
50  		return (CtCodeSnippetStatement) super.clone();
51  	}
52  }
