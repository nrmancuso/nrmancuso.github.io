6   package spoon.support.visitor.java.internal;
7   
8   import spoon.reflect.declaration.CtAnnotation;
9   import spoon.reflect.declaration.CtPackage;
10  import spoon.reflect.declaration.CtType;
11  
12  import java.lang.annotation.Annotation;
13  
14  public class PackageRuntimeBuilderContext extends AbstractRuntimeBuilderContext {
15  	private CtPackage ctPackage;
16  
17  	public PackageRuntimeBuilderContext(CtPackage ctPackage) {
18  		super(ctPackage);
19  		this.ctPackage = ctPackage;
20  	}
21  
22  	@Override
23  	public void addType(CtType<?> aType) {
24  		ctPackage.addType(aType);
25  	}
26  
27  	@Override
28  	public void addAnnotation(CtAnnotation<Annotation> ctAnnotation) {
29  		ctPackage.addAnnotation(ctAnnotation);
30  	}
31  }
