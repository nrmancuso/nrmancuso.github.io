17  package spoon.test.factory;
18  
19  import org.junit.Test;
20  import spoon.reflect.code.CtLiteral;
21  import spoon.reflect.code.CtNewArray;
22  import spoon.reflect.declaration.CtAnnotation;
23  import spoon.reflect.declaration.CtClass;
24  import spoon.reflect.factory.AnnotationFactory;
25  
26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertTrue;
28  import static spoon.testing.utils.ModelUtils.build;
29  
30  public class AnnotationFactoryTest {
31  
32  	@Test
33  	public void testAnnotate() throws Exception {
34  
35  		CtClass<?> type = build("spoon.test.testclasses", "SampleClass");
36  
37  		AnnotationFactory af = type.getFactory().Annotation();
38  		af.annotate(type, SampleAnnotation.class, "names", new String[]{"foo", "bar"});
39  
40  		final CtAnnotation<SampleAnnotation> annotation = type.getAnnotation(type.getFactory().Annotation().createReference(SampleAnnotation.class));
41  		assertTrue(annotation.getValue("names") instanceof CtNewArray);
42  		final CtNewArray names = annotation.getValue("names");
43  		assertEquals(2, names.getElements().size());
44  		assertEquals("foo", ((CtLiteral) names.getElements().get(0)).getValue());
45  		assertEquals("bar", ((CtLiteral) names.getElements().get(1)).getValue());
46  	}
47  }
48  
49  @interface SampleAnnotation {
50  	String[] names();
51  }
