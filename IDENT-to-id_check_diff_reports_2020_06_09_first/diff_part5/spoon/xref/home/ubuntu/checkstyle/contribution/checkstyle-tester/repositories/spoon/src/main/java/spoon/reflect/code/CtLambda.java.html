6   package spoon.reflect.code;
7   
8   import spoon.reflect.declaration.CtExecutable;
9   import spoon.reflect.declaration.CtMethod;
10  import spoon.reflect.reference.CtTypeReference;
11  import spoon.support.DerivedProperty;
12  import spoon.reflect.annotations.PropertyGetter;
13  import spoon.reflect.annotations.PropertySetter;
14  import spoon.support.UnsettableProperty;
15  
16  import java.util.Set;
17  
18  import static spoon.reflect.path.CtRole.EXPRESSION;
19  
20  /**
21   * This code element represents the creation of a lambda. A lambda
22   * can have two sorts of body : an simple expression or a block of
23   * statements. The usage of this concept in this class is:
24   *
25   * <pre>
26   *     java.util.List l = new java.util.ArrayList();
27   *     l.stream().map(
28   *       x -&gt; { return x.toString(); } // a lambda
29   *     );
30   * </pre>
31   *
32   * <ul>
33   * <li>
34   * If your lambda has an expression, getBody method will
35   * return null and getExpression method will return a
36   * CtExpression.
37   * </li>
38   * <li>
39   * If your lambda has a block of statement, getExpression
40   * method will return null and getBody will returns a CtBlock
41   * with all statements.
42   * </li>
43   * </ul>
44   *
45   * So keep this in mind when you would like the body of a CtLambda.
46   *
47   * @param <T>
48   * 		created type
49   */
50  public interface CtLambda<T> extends CtExpression<T>, CtExecutable<T> {
51  	/**
52  	 * Gets the expression in the body. Null if the body is a list
53  	 * of statements.
54  	 */
55  	@PropertyGetter(role = EXPRESSION)
56  	CtExpression<T> getExpression();
57  
58  	/**
59  	 * @return the method that this lambda expression implements.
60  	 * Must be defined as a non-default method in an interface, e.g. Consumer.accept().
61  	 */
62  	@DerivedProperty
63  	<R> CtMethod<R> getOverriddenMethod();
64  
65  	/**
66  	 * Sets the expression in the body of the lambda. Nothing will change
67  	 * if the lambda already has a value in the body attribute.
68  	 */
69  	@PropertySetter(role = EXPRESSION)
70  	<C extends CtLambda<T>> C setExpression(CtExpression<T> expression);
71  
72  	@Override
73  	CtLambda<T> clone();
74  
75  	@Override
76  	@UnsettableProperty
77  	<T1 extends CtExecutable<T>> T1 setThrownTypes(Set<CtTypeReference<? extends Throwable>> thrownTypes);
78  }
