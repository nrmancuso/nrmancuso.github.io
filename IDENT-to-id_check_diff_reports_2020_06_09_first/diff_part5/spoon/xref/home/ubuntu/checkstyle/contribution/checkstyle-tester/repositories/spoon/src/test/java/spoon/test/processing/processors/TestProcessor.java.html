17  package spoon.test.processing.processors;
18  
19  import java.util.Date;
20  
21  import org.apache.log4j.Level;
22  import spoon.processing.AbstractProcessor;
23  import spoon.reflect.cu.CompilationUnit;
24  import spoon.reflect.declaration.CtClass;
25  import spoon.reflect.declaration.CtElement;
26  import spoon.reflect.declaration.CtPackage;
27  import spoon.reflect.declaration.CtTypedElement;
28  import spoon.template.Substitution;
29  import spoon.test.template.testclasses.constructors.TemplateWithConstructor;
30  
31  public class TestProcessor extends AbstractProcessor<CtElement> {
32  
33  	@Override
34  	public void init() {
35  		super.init();
36  		getEnvironment().debugMessage("MAIN METHODS: " + getFactory().Method().getMainMethods());
37  	}
38  
39  	public void process(CtElement element) {
40  		if ((!(element instanceof CtPackage)) && !element.isParentInitialized()) {
41  			getEnvironment().report(this, Level.ERROR, element,
42  					"Element's parent is null (" + element + ")");
43  			throw new RuntimeException("uninitialized parent detected");
44  		}
45  		if (element instanceof CtTypedElement) {
46  			if (((CtTypedElement<?>) element).getType() == null) {
47  				getEnvironment().report(this, Level.WARN, element,
48  						"Element's type is null (" + element + ")");
49  			}
50  		}
51  		if (element instanceof CtClass) {
52  			CtClass<?> c = (CtClass<?>) element;
53  			if ("Secondary".equals(c.getSimpleName())) {
54  				@SuppressWarnings("unused")
55  				CompilationUnit cu = c.getPosition().getCompilationUnit();
56  			}
57  			if ("C1".equals(c.getSimpleName())) {
58  				Substitution.insertAll(c, new TemplateWithConstructor(
59  						getFactory().Type().createReference(Date.class)));
60  			}
61  		}
62  	}
63  
64  }
