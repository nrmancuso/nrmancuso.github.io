6   package spoon.pattern;
7   
8   import spoon.SpoonException;
9   import spoon.pattern.internal.node.RootNode;
10  
11  /**
12   * Defines what happens when a {@link RootNode} has to be replaced by another {@link RootNode}, default in {@link #FAIL}.
13   */
14  public enum ConflictResolutionMode {
15  	/**
16  	 * Throw {@link SpoonException} if a conflict happens, it is the default in most cases. But there are some standard Pattern builder algorithms (mainly these which deals with legacy Templates), which are using the other modes.
17  	 */
18  	FAIL,
19  	/**
20  	 * Get rid of old {@link RootNode} and use new {@link RootNode} instead
21  	 */
22  	USE_NEW_NODE,
23  	/**
24  	 * Keep old {@link RootNode} and ignore requests to add new {@link RootNode}
25  	 */
26  	KEEP_OLD_NODE,
27  	/**
28  	 * Add new {@link RootNode} after existing nodes
29  	 */
30  	APPEND
31  }
