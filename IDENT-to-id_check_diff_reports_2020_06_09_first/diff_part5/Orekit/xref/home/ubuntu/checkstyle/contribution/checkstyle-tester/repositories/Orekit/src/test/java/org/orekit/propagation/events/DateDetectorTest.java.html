51      @Test
52      public void testSimpleTimer() throws OrekitException {
53          DateDetector dateDetector = new DateDetector(maxCheck, threshold, iniDate.shiftedBy(2.0*dt));
54      	Assert.assertEquals(2 * dt, dateDetector.getDate().durationFrom(iniDate), 1.0e-10);
55          propagator.addEventDetector(dateDetector);
56          final SpacecraftState finalState = propagator.propagate(iniDate.shiftedBy(100.*dt));
57  
58          Assert.assertEquals(2.0*dt, finalState.getDate().durationFrom(iniDate), threshold);
59      }

61      @Test
62      public void testEmbeddedTimer() throws OrekitException {
63      	dateDetector = new DateDetector(maxCheck, threshold);
64          Assert.assertNull(dateDetector.getDate());
65      	EventDetector nodeDetector = new NodeDetector(iniOrbit, iniOrbit.getFrame()).
66      	        withHandler(new ContinueOnEvent<NodeDetector>() {
67                      private static final long serialVersionUID = 1L;
68      	            public Action eventOccurred(SpacecraftState s, NodeDetector nd, boolean increasing)
69      	                throws OrekitException {
70      	                if (increasing) {
71      	                    nodeDate = s.getDate();
72      	                    dateDetector.addEventDate(nodeDate.shiftedBy(dt));
73      	                }
74      	                return Action.CONTINUE;
75      	            }
76      	        });
77  
78          propagator.addEventDetector(nodeDetector);
79          propagator.addEventDetector(dateDetector);
80          final SpacecraftState finalState = propagator.propagate(iniDate.shiftedBy(100.*dt));
81  
82          Assert.assertEquals(dt, finalState.getDate().durationFrom(nodeDate), threshold);
83      }

85      @Test
86      public void testAutoEmbeddedTimer() throws OrekitException {
87          dateDetector = new DateDetector(maxCheck, threshold, iniDate.shiftedBy(-dt)).
88                  withHandler(new ContinueOnEvent<DateDetector>() {
89                      private static final long serialVersionUID = 1L;
90                      public Action eventOccurred(SpacecraftState s, DateDetector dd,  boolean increasing)
91                              throws OrekitException {
92                          AbsoluteDate nextDate = s.getDate().shiftedBy(-dt);
93                          dd.addEventDate(nextDate);
94                          ++evtno;
95                          return Action.CONTINUE;
96                      }
97                  });
98          propagator.addEventDetector(dateDetector);
99          propagator.propagate(iniDate.shiftedBy(-100.*dt));
100 
101         Assert.assertEquals(100, evtno);
102     }

104     @Test(expected=IllegalArgumentException.class)
105     public void testExceptionTimer() throws OrekitException {
106         dateDetector = new DateDetector(maxCheck, threshold, iniDate.shiftedBy(dt)).
107                 withHandler(new ContinueOnEvent<DateDetector>() {
108                     private static final long serialVersionUID = 1L;
109                     public Action eventOccurred(SpacecraftState s, DateDetector dd, boolean increasing)
110                         throws OrekitException {
111                         double step = (evtno % 2 == 0) ? 2.*maxCheck : maxCheck/2.;
112                         AbsoluteDate nextDate = s.getDate().shiftedBy(step);
113                         dd.addEventDate(nextDate);
114                         ++evtno;
115                         return Action.CONTINUE;
116                     }
117                 });
118         propagator.addEventDetector(dateDetector);
119         propagator.propagate(iniDate.shiftedBy(100.*dt));
120     }

122     /**
123      * Check that a generic event handler can be used with an event detector.
124      *
125      * @throws OrekitException on error.
126      */
127     @Test
128     public void testGenericHandler() throws OrekitException {
129         //setup
130         dateDetector = new DateDetector(maxCheck, threshold, iniDate.shiftedBy(dt));
131         // generic event handler that works with all detectors.
132         EventHandler<EventDetector> handler = new EventHandler<EventDetector>() {
133             @Override
134             public Action eventOccurred(SpacecraftState s,
135                                         EventDetector detector,
136                                         boolean increasing)
137                     throws OrekitException {
138                 Assert.assertSame(dateDetector, detector);
139                 return Action.STOP;
140             }
141 
142             @Override
143             public SpacecraftState resetState(EventDetector detector,
144                                               SpacecraftState oldState)
145                     throws OrekitException {
146                 throw new RuntimeException("Should not be called");
147             }
148         };
149 
150         //action
151         dateDetector = dateDetector.withHandler(handler);
152         propagator.addEventDetector(dateDetector);
153         SpacecraftState finalState = propagator.propagate(iniDate.shiftedBy(100 * dt));
154 
155         //verify
156         Assert.assertEquals(dt, finalState.getDate().durationFrom(iniDate), threshold);
157     }

159     @Before
160     public void setUp() {
161         try {
162             Utils.setDataRoot("regular-data");
163             final double mu = 3.9860047e14;
164             final Vector3D position  = new Vector3D(-6142438.668, 3492467.560, -25767.25680);
165             final Vector3D velocity  = new Vector3D(505.8479685, 942.7809215, 7435.922231);
166             iniDate  = new AbsoluteDate(1969, 7, 28, 4, 0, 0.0, TimeScalesFactory.getTT());
167             iniOrbit = new EquinoctialOrbit(new PVCoordinates(position, velocity),
168                                             FramesFactory.getEME2000(), iniDate, mu);
169             SpacecraftState initialState = new SpacecraftState(iniOrbit);
170             double[] absTolerance = {
171                 0.001, 1.0e-9, 1.0e-9, 1.0e-6, 1.0e-6, 1.0e-6, 0.001
172             };
173             double[] relTolerance = {
174                 1.0e-7, 1.0e-4, 1.0e-4, 1.0e-7, 1.0e-7, 1.0e-7, 1.0e-7
175             };
176             AdaptiveStepsizeIntegrator integrator =
177                 new DormandPrince853Integrator(0.001, 1000, absTolerance, relTolerance);
178             integrator.setInitialStepSize(60);
179             propagator = new NumericalPropagator(integrator);
180             propagator.setInitialState(initialState);
181             dt = 60.;
182             maxCheck  = 10.;
183             threshold = 10.e-10;
184             evtno = 0;
185         } catch (OrekitException oe) {
186             Assert.fail(oe.getLocalizedMessage());
187         }
188     }
