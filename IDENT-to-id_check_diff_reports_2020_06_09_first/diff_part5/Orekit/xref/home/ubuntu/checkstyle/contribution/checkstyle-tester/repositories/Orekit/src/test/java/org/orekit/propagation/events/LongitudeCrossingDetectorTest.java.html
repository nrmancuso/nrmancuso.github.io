48      @Test
49      public void testRegularCrossing() throws OrekitException {
50  
51          final OneAxisEllipsoid earth = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
52                                                              Constants.WGS84_EARTH_FLATTENING,
53                                                              FramesFactory.getITRF(IERSConventions.IERS_2010, true));
54  
55          LongitudeCrossingDetector d =
56                  new LongitudeCrossingDetector(earth, FastMath.toRadians(10.0)).
57                  withMaxCheck(60).
58                  withThreshold(1.e-6).
59                  withHandler(new ContinueOnEvent<LongitudeCrossingDetector>());
60  
61          Assert.assertEquals(60.0, d.getMaxCheckInterval(), 1.0e-15);
62          Assert.assertEquals(1.0e-6, d.getThreshold(), 1.0e-15);
63          Assert.assertEquals(10.0, FastMath.toDegrees(d.getLongitude()), 1.0e-14);
64          Assert.assertEquals(AbstractDetector.DEFAULT_MAX_ITER, d.getMaxIterationCount());
65          Assert.assertSame(earth, d.getBody());
66  
67          final TimeScale utc = TimeScalesFactory.getUTC();
68          final Vector3D position = new Vector3D(-6142438.668, 3492467.56, -25767.257);
69          final Vector3D velocity = new Vector3D(505.848, 942.781, 7435.922);
70          final AbsoluteDate date = new AbsoluteDate(2003, 9, 16, utc);
71          final Orbit orbit = new EquinoctialOrbit(new PVCoordinates(position,  velocity),
72                                                   FramesFactory.getEME2000(), date,
73                                                   Constants.EIGEN5C_EARTH_MU);
74  
75          Propagator propagator =
76              new EcksteinHechlerPropagator(orbit,
77                                            Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
78                                            Constants.EIGEN5C_EARTH_MU,
79                                            Constants.EIGEN5C_EARTH_C20,
80                                            Constants.EIGEN5C_EARTH_C30,
81                                            Constants.EIGEN5C_EARTH_C40,
82                                            Constants.EIGEN5C_EARTH_C50,
83                                            Constants.EIGEN5C_EARTH_C60);
84  
85          EventsLogger logger = new EventsLogger();
86          propagator.addEventDetector(logger.monitorDetector(d));
87  
88          propagator.propagate(date.shiftedBy(Constants.JULIAN_DAY));
89          AbsoluteDate previous = null;
90          for (LoggedEvent e : logger.getLoggedEvents()) {
91              SpacecraftState state = e.getState();
92              double longitude = earth.transform(state.getPVCoordinates(earth.getBodyFrame()).getPosition(),
93                                                earth.getBodyFrame(), null).getLongitude();
94              Assert.assertEquals(10.0, FastMath.toDegrees(longitude), 1.6e-7);
95              if (previous != null) {
96                  // same time interval regardless of increasing/decreasing,
97                  // as increasing/decreasing flag is irrelevant for this detector
98                   Assert.assertEquals(4954.70, state.getDate().durationFrom(previous), 1e10);
99              }
100             previous = state.getDate();
101         }
102         Assert.assertEquals(16, logger.getLoggedEvents().size());
103 
104     }

106     @Test
107     public void testZigZag() throws OrekitException {
108 
109         final OneAxisEllipsoid earth = new OneAxisEllipsoid(Constants.WGS84_EARTH_EQUATORIAL_RADIUS,
110                                                             Constants.WGS84_EARTH_FLATTENING,
111                                                             FramesFactory.getITRF(IERSConventions.IERS_2010, true));
112 
113         LongitudeCrossingDetector d =
114                 new LongitudeCrossingDetector(600.0, 1.e-6, earth, FastMath.toRadians(-100.0)).
115                 withHandler(new ContinueOnEvent<LongitudeCrossingDetector>());
116 
117         Assert.assertEquals(600.0, d.getMaxCheckInterval(), 1.0e-15);
118         Assert.assertEquals(1.0e-6, d.getThreshold(), 1.0e-15);
119         Assert.assertEquals(-100.0, FastMath.toDegrees(d.getLongitude()), 1.0e-14);
120         Assert.assertEquals(AbstractDetector.DEFAULT_MAX_ITER, d.getMaxIterationCount());
121 
122         KeplerianOrbit orbit =
123                         new KeplerianOrbit(24464560.0, 0.7311, 0.122138, 3.10686, 1.00681,
124                                            0.048363, PositionAngle.MEAN,
125                                            FramesFactory.getEME2000(),
126                                            AbsoluteDate.J2000_EPOCH,
127                                            Constants.EIGEN5C_EARTH_MU);
128 
129 
130         Propagator propagator = new KeplerianPropagator(orbit);
131 
132         EventsLogger logger = new EventsLogger();
133         propagator.addEventDetector(logger.monitorDetector(d));
134 
135         propagator.propagate(orbit.getDate().shiftedBy(Constants.JULIAN_DAY));
136         double[] expectedLatitudes = new double[] { -6.5394381901, -0.4918760372, +6.5916016832 };
137         Assert.assertEquals(3, logger.getLoggedEvents().size());
138         for (int i = 0; i < 3; ++i) {
139             SpacecraftState state = logger.getLoggedEvents().get(i).getState();
140             GeodeticPoint gp = earth.transform(state.getPVCoordinates(earth.getBodyFrame()).getPosition(),
141                                                earth.getBodyFrame(), null);
142             Assert.assertEquals(expectedLatitudes[i], FastMath.toDegrees(gp.getLatitude()),  1.0e-10);
143             Assert.assertEquals(-100.0,               FastMath.toDegrees(gp.getLongitude()), 1.2e-9);
144         }
145 
146     }
