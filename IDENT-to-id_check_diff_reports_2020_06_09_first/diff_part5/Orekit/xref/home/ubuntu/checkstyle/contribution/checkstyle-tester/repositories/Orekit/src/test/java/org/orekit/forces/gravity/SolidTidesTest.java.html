50      @Test
51      public void testDefaultInterpolation() throws OrekitException {
52  
53          IERSConventions conventions = IERSConventions.IERS_2010;
54          Frame eme2000 = FramesFactory.getEME2000();
55          Frame itrf    = FramesFactory.getITRF(conventions, true);
56          TimeScale utc = TimeScalesFactory.getUTC();
57          UT1Scale  ut1 = TimeScalesFactory.getUT1(conventions, true);
58          NormalizedSphericalHarmonicsProvider gravityField =
59                  GravityFieldFactory.getConstantNormalizedProvider(5, 5);
60  
61          // initialization
62          AbsoluteDate date = new AbsoluteDate(1970, 07, 01, 13, 59, 27.816, utc);
63          Orbit orbit = new KeplerianOrbit(7201009.7124401, 1e-3, FastMath.toRadians(98.7),
64                                           FastMath.toRadians(93.0), FastMath.toRadians(15.0 * 22.5),
65                                           0, PositionAngle.MEAN, eme2000, date,
66                                           gravityField.getMu());
67  
68          AbsoluteDate target = date.shiftedBy(7 * Constants.JULIAN_DAY);
69          ForceModel hf = new HolmesFeatherstoneAttractionModel(itrf, gravityField);
70          SpacecraftState raw = propagate(orbit, target, hf,
71                                          new SolidTides(itrf, gravityField.getAe(), gravityField.getMu(),
72                                                         gravityField.getTideSystem(), true, Double.NaN, -1,
73                                                         conventions, ut1,
74                                                         CelestialBodyFactory.getSun(),
75                                                         CelestialBodyFactory.getMoon()));
76          SpacecraftState interpolated = propagate(orbit, target, hf,
77                                                   new SolidTides(itrf, gravityField.getAe(), gravityField.getMu(),
78                                                                  gravityField.getTideSystem(),
79                                                                  conventions, ut1,
80                                                                  CelestialBodyFactory.getSun(),
81                                                                  CelestialBodyFactory.getMoon()));
82          Assert.assertEquals(0.0,
83                              Vector3D.distance(raw.getPVCoordinates().getPosition(),
84                                                interpolated.getPVCoordinates().getPosition()),
85                              2.0e-5); // threshold would be 1.2e-3 for 30 days propagation
86  
87      }

149     private void doTestTideEffect(Orbit orbit, IERSConventions conventions, double delta1, double delta2)
150         throws OrekitException {
151 
152         Frame itrf    = FramesFactory.getITRF(conventions, true);
153         UT1Scale  ut1 = TimeScalesFactory.getUT1(conventions, true);
154         NormalizedSphericalHarmonicsProvider gravityField =
155                 GravityFieldFactory.getConstantNormalizedProvider(5, 5);
156 
157         // initialization
158 
159         AbsoluteDate target = orbit.getDate().shiftedBy(7 * Constants.JULIAN_DAY);
160         ForceModel hf = new HolmesFeatherstoneAttractionModel(itrf, gravityField);
161         SpacecraftState noTides              = propagate(orbit, target, hf);
162         SpacecraftState solidTidesNoPoleTide = propagate(orbit, target, hf,
163                                                          new SolidTides(itrf, gravityField.getAe(), gravityField.getMu(),
164                                                                         gravityField.getTideSystem(), false,
165                                                                         SolidTides.DEFAULT_STEP, SolidTides.DEFAULT_POINTS,
166                                                                         conventions, ut1,
167                                                                         CelestialBodyFactory.getSun(),
168                                                                         CelestialBodyFactory.getMoon()));
169         SpacecraftState solidTidesPoleTide = propagate(orbit, target, hf,
170                                                        new SolidTides(itrf, gravityField.getAe(), gravityField.getMu(),
171                                                                       gravityField.getTideSystem(), true,
172                                                                       SolidTides.DEFAULT_STEP, SolidTides.DEFAULT_POINTS,
173                                                                       conventions, ut1,
174                                                                       CelestialBodyFactory.getSun(),
175                                                                       CelestialBodyFactory.getMoon()));
176         Assert.assertEquals(delta1,
177                             Vector3D.distance(noTides.getPVCoordinates().getPosition(),
178                                               solidTidesNoPoleTide.getPVCoordinates().getPosition()),
179                             0.01);
180         Assert.assertEquals(delta2,
181                             Vector3D.distance(solidTidesNoPoleTide.getPVCoordinates().getPosition(),
182                                               solidTidesPoleTide.getPVCoordinates().getPosition()),
183                             0.01);
184 
185     }
186 
187     private SpacecraftState propagate(Orbit orbit, AbsoluteDate target, ForceModel ... forceModels)
188         throws OrekitException {
189         double[][] tolerances = NumericalPropagator.tolerances(10, orbit, OrbitType.KEPLERIAN);
190         AbstractIntegrator integrator = new DormandPrince853Integrator(1.0e-3, 300, tolerances[0], tolerances[1]);
191         NumericalPropagator propagator = new NumericalPropagator(integrator);
192         for (ForceModel forceModel : forceModels) {
193             propagator.addForceModel(forceModel);
194         }
195         propagator.setInitialState(new SpacecraftState(orbit));
196         return propagator.propagate(target);
197     }
