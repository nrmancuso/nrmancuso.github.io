24  package org.hibernate.internal.util.jndi;
25  
26  import java.util.Hashtable;
27  import java.util.Map;
28  import java.util.Properties;
29  import java.util.Set;
30  import javax.naming.Context;
31  import javax.naming.InitialContext;
32  import javax.naming.Name;
33  import javax.naming.NameNotFoundException;
34  import javax.naming.NamingException;
35  
36  import org.hibernate.cfg.Environment;
37  
38  /**
39   * Helper for dealing with JNDI.
40   *
41   * @deprecated As JNDI access should get routed through {@link org.hibernate.service.jndi.spi.JndiService}
42   */
43  @Deprecated
44  public final class JndiHelper {
45  	private JndiHelper() {
46  	}
47  
48  	/**
49  	 * Given a hodgepodge of properties, extract out the ones relevant for JNDI interaction.
50  	 *
51  	 * @param configurationValues The map of config values
52  	 *
53  	 * @return The extracted JNDI specific properties.
54  	 */
55  	@SuppressWarnings({ "unchecked" })
56  	public static Properties extractJndiProperties(Map configurationValues) {
57  		final Properties jndiProperties = new Properties();
58  
59  		for ( Map.Entry entry : (Set<Map.Entry>) configurationValues.entrySet() ) {
60  			if ( !String.class.isInstance( entry.getKey() ) ) {
61  				continue;
62  			}
63  			final String propertyName = (String) entry.getKey();
64  			final Object propertyValue = entry.getValue();
65  			if ( propertyName.startsWith( Environment.JNDI_PREFIX ) ) {
66  				// write the IntialContextFactory class and provider url to the result only if they are
67  				// non-null; this allows the environmental defaults (if any) to remain in effect
68  				if ( Environment.JNDI_CLASS.equals( propertyName ) ) {
69  					if ( propertyValue != null ) {
70  						jndiProperties.put( Context.INITIAL_CONTEXT_FACTORY, propertyValue );
71  					}
72  				}
73  				else if ( Environment.JNDI_URL.equals( propertyName ) ) {
74  					if ( propertyValue != null ) {
75  						jndiProperties.put( Context.PROVIDER_URL, propertyValue );
76  					}
77  				}
78  				else {
79  					final String passThruPropertyname = propertyName.substring( Environment.JNDI_PREFIX.length() + 1 );
80  					jndiProperties.put( passThruPropertyname, propertyValue );
81  				}
82  			}
83  		}
84  
85  		return jndiProperties;
86  	}
87  
88  	public static InitialContext getInitialContext(Properties props) throws NamingException {
89  		Hashtable hash = extractJndiProperties(props);
90  		return hash.size()==0 ?
91  				new InitialContext() :
92  				new InitialContext(hash);
93  	}
94  
95  	/**
96  	 * Bind val to name in ctx, and make sure that all intermediate contexts exist.
97  	 *
98  	 * @param ctx the root context
99  	 * @param name the name as a string
100 	 * @param val the object to be bound
101 	 *
102 	 * @throws NamingException Indicates a problem performing the bind.
103 	 */
104 	public static void bind(Context ctx, String name, Object val) throws NamingException {
105 		try {
106 			ctx.rebind(name, val);
107 		}
108 		catch (Exception e) {
109 			Name n = ctx.getNameParser("").parse(name);
110 			while ( n.size() > 1 ) {
111 				String ctxName = n.get(0);
112 
113 				Context subctx=null;
114 				try {
115 					subctx = (Context) ctx.lookup(ctxName);
116 				}
117 				catch (NameNotFoundException ignore) {
118 				}
119 
120 				if (subctx!=null) {
121 					ctx = subctx;
122 				}
123 				else {
124 					ctx = ctx.createSubcontext(ctxName);
125 				}
126 				n = n.getSuffix(1);
127 			}
128 			ctx.rebind(n, val);
129 		}
130 	}
131 }
