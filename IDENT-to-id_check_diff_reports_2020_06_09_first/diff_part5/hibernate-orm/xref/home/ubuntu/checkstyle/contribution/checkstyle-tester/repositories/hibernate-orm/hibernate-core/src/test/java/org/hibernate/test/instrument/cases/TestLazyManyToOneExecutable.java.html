1   package org.hibernate.test.instrument.cases;
2   import junit.framework.Assert;
3   
4   import org.hibernate.Hibernate;
5   import org.hibernate.Session;
6   import org.hibernate.Transaction;
7   import org.hibernate.test.instrument.domain.Document;
8   import org.hibernate.test.instrument.domain.Folder;
9   import org.hibernate.test.instrument.domain.Owner;
10  
11  /**
12   * @author Steve Ebersole
13   */
14  public class TestLazyManyToOneExecutable extends AbstractExecutable {
15  	public void execute() {
16  		Session s = getFactory().openSession();
17  		Transaction t = s.beginTransaction();
18  		Owner gavin = new Owner();
19  		Document hia = new Document();
20  		Folder fol = new Folder();
21  		gavin.setName("gavin");
22  		hia.setName("Hibernate in Action");
23  		hia.setSummary("blah");
24  		hia.updateText("blah blah");
25  		fol.setName("books");
26  		hia.setOwner(gavin);
27  		hia.setFolder(fol);
28  		fol.getDocuments().add(hia);
29  		s.persist(gavin);
30  		s.persist(fol);
31  		t.commit();
32  		s.close();
33  
34  		s = getFactory().openSession();
35  		t = s.beginTransaction();
36  		hia = (Document) s.createCriteria(Document.class).uniqueResult();
37  		Assert.assertEquals( hia.getFolder().getClass(), Folder.class);
38  		fol = hia.getFolder();
39  		Assert.assertTrue( Hibernate.isInitialized(fol) );
40  		t.commit();
41  		s.close();
42  
43  		s = getFactory().openSession();
44  		t = s.beginTransaction();
45  		hia = (Document) s.createCriteria(Document.class).uniqueResult();
46  		Assert.assertSame( hia.getFolder(), s.load(Folder.class, fol.getId()) );
47  		Assert.assertTrue( Hibernate.isInitialized( hia.getFolder() ) );
48  		t.commit();
49  		s.close();
50  
51  		s = getFactory().openSession();
52  		t = s.beginTransaction();
53  		fol = (Folder) s.get(Folder.class, fol.getId());
54  		hia = (Document) s.createCriteria(Document.class).uniqueResult();
55  		Assert.assertSame( fol, hia.getFolder() );
56  		fol = hia.getFolder();
57  		Assert.assertTrue( Hibernate.isInitialized(fol) );
58  		t.commit();
59  		s.close();
60  
61  		s = getFactory().openSession();
62  		t = s.beginTransaction();
63  		fol = (Folder) s.load(Folder.class, fol.getId());
64  		hia = (Document) s.createCriteria(Document.class).uniqueResult();
65  		Assert.assertNotSame( fol, hia.getFolder() );
66  		fol = hia.getFolder();
67  		Assert.assertTrue( Hibernate.isInitialized(fol) );
68  		s.delete(hia.getFolder());
69  		s.delete(hia.getOwner());
70  		t.commit();
71  		s.close();
72  	}
73  }
