24  package org.hibernate.envers.entities.mapper.relation.query;
25  
26  import org.hibernate.envers.configuration.AuditEntitiesConfiguration;
27  import org.hibernate.envers.configuration.GlobalConfiguration;
28  import org.hibernate.envers.entities.mapper.relation.MiddleIdData;
29  import org.hibernate.envers.tools.query.Parameters;
30  import org.hibernate.envers.tools.query.QueryBuilder;
31  import org.hibernate.envers.strategy.AuditStrategy;
32  
33  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.DEL_REVISION_TYPE_PARAMETER;
34  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REFERENCED_ENTITY_ALIAS;
35  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REFERENCED_ENTITY_ALIAS_DEF_AUD_STR;
36  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REVISION_PARAMETER;
37  
38  /**
39   * Selects data from an audit entity.
40   *
41   * @author Adam Warski (adam at warski dot org)
42   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
43   */
44  public final class OneAuditEntityQueryGenerator extends AbstractRelationQueryGenerator {
45  	private final String queryString;
46  	private final String queryRemovedString;
47  
48  	public OneAuditEntityQueryGenerator(GlobalConfiguration globalCfg, AuditEntitiesConfiguration verEntCfg,
49  										AuditStrategy auditStrategy, MiddleIdData referencingIdData,
50  										String referencedEntityName, MiddleIdData referencedIdData, boolean revisionTypeInId) {
51  		super( verEntCfg, referencingIdData, revisionTypeInId );
52  
53  		/*
54  		 * The valid query that we need to create:
55  		 *   SELECT e FROM versionsReferencedEntity e
56  		 *   WHERE
57  		 * (only entities referenced by the association; id_ref_ing = id of the referencing entity)
58  		 *     e.id_ref_ing = :id_ref_ing AND
59  		 * (selecting e entities at revision :revision)
60  		 *   --> for DefaultAuditStrategy:
61  		 *     e.revision = (SELECT max(e2.revision) FROM versionsReferencedEntity e2
62  		 *       WHERE e2.revision <= :revision AND e2.id = e.id)
63  		 *
64  		 *   --> for ValidityAuditStrategy:
65  		 *     e.revision <= :revision and (e.endRevision > :revision or e.endRevision is null)
66  		 *
67  		 *     AND
68  		 * (only non-deleted entities)
69  		 *     e.revision_type != DEL
70  		 */
71  		final QueryBuilder commonPart = commonQueryPart( verEntCfg.getAuditEntityName( referencedEntityName ) );
72  		final QueryBuilder validQuery = commonPart.deepCopy();
73  		final QueryBuilder removedQuery = commonPart.deepCopy();
74  		createValidDataRestrictions(
75  				globalCfg, auditStrategy, referencedIdData, validQuery, validQuery.getRootParameters()
76  		);
77  		createValidAndRemovedDataRestrictions( globalCfg, auditStrategy, referencedIdData, removedQuery );
78  
79  		queryString = queryToString( validQuery );
80  		queryRemovedString = queryToString( removedQuery );
81  	}
82  
83  	/**
84  	 * Compute common part for both queries.
85  	 */
86  	private QueryBuilder commonQueryPart(String versionsReferencedEntityName) {
87  		// SELECT e FROM versionsEntity e
88  		final QueryBuilder qb = new QueryBuilder( versionsReferencedEntityName, REFERENCED_ENTITY_ALIAS );
89  		qb.addProjection( null, REFERENCED_ENTITY_ALIAS, false, false );
90  		// WHERE
91  		// e.id_ref_ed = :id_ref_ed
92  		referencingIdData.getPrefixedMapper().addNamedIdEqualsToQuery( qb.getRootParameters(), null, true );
93  		return qb;
94  	}
95  
96  	/**
97  	 * Creates query restrictions used to retrieve only actual data.
98  	 */
99  	private void createValidDataRestrictions(GlobalConfiguration globalCfg, AuditStrategy auditStrategy,
100 											 MiddleIdData referencedIdData, QueryBuilder qb, Parameters rootParameters) {
101 		final String revisionPropertyPath = verEntCfg.getRevisionNumberPath();
102 		// (selecting e entities at revision :revision)
103 		// --> based on auditStrategy (see above)
104 		auditStrategy.addEntityAtRevisionRestriction(
105 				globalCfg, qb, rootParameters, revisionPropertyPath,
106 				verEntCfg.getRevisionEndFieldName(), true, referencedIdData, revisionPropertyPath,
107 				verEntCfg.getOriginalIdPropName(), REFERENCED_ENTITY_ALIAS, REFERENCED_ENTITY_ALIAS_DEF_AUD_STR,
108 				true
109 		);
110 		// e.revision_type != DEL
111 		rootParameters.addWhereWithNamedParam( getRevisionTypePath(), false, "!=", DEL_REVISION_TYPE_PARAMETER );
112 	}
113 
114 	/**
115 	 * Create query restrictions used to retrieve actual data and deletions that took place at exactly given revision.
116 	 */
117 	private void createValidAndRemovedDataRestrictions(GlobalConfiguration globalCfg, AuditStrategy auditStrategy,
118 													   MiddleIdData referencedIdData, QueryBuilder remQb) {
119 		final Parameters disjoint = remQb.getRootParameters().addSubParameters( "or" );
120 		final Parameters valid = disjoint.addSubParameters( "and" ); // Restrictions to match all valid rows.
121 		final Parameters removed = disjoint.addSubParameters( "and" ); // Restrictions to match all rows deleted at exactly given revision.
122 		// Excluding current revision, because we need to match data valid at the previous one.
123 		createValidDataRestrictions( globalCfg, auditStrategy, referencedIdData, remQb, valid );
124 		// e.revision = :revision
125 		removed.addWhereWithNamedParam( verEntCfg.getRevisionNumberPath(), false, "=", REVISION_PARAMETER );
126 		// e.revision_type = DEL
127 		removed.addWhereWithNamedParam( getRevisionTypePath(), false, "=", DEL_REVISION_TYPE_PARAMETER );
128 	}
129 
130 	@Override
131 	protected String getQueryString() {
132 		return queryString;
133 	}
134 
135 	@Override
136 	protected String getQueryRemovedString() {
137 		return queryRemovedString;
138 	}
139 }
