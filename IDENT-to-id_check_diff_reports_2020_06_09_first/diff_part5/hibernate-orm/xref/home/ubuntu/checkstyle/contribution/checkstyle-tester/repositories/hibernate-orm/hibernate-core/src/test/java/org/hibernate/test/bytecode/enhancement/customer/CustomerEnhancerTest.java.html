125 	private void testFor(Class entityClassToEnhance) throws Exception {
126 		Enhancer enhancer = new Enhancer( enhancementContext );
127 		CtClass entityCtClass = generateCtClassForAnEntity( entityClassToEnhance );
128 		byte[] original = entityCtClass.toBytecode();
129 		//byte[] enhanced = enhancer.enhance( entityCtClass.getName(), original );
130         byte[] enhanced = enhancer.enhance( entityCtClass.getName(), original );
131 		assertFalse( "entity was not enhanced", Arrays.equals( original, enhanced ) );
132 
133 		ClassLoader cl = new ClassLoader() { };
134 		ClassPool cp = new ClassPool( false );
135 		cp.appendClassPath( new LoaderClassPath( cl ) );
136 		CtClass enhancedCtClass = cp.makeClass( new ByteArrayInputStream( enhanced ) );
137         enhancedCtClass.debugWriteFile("/tmp");
138         //just for debugging
139         Class addressClass = null;
140         Class customerInventoryClass = null;
141         Class supplierComponentPKClass = null;
142 
143         CtClass addressCtClass = generateCtClassForAnEntity( org.hibernate.test.bytecode.enhancement.customer.Address.class );
144         byte[] enhancedAddress = enhancer.enhanceComposite(Address.class.getName(), addressCtClass.toBytecode());
145         CtClass enhancedCtClassAddress = cp.makeClass( new ByteArrayInputStream( enhancedAddress ) );
146         enhancedCtClassAddress.debugWriteFile("/tmp");
147         addressClass = enhancedCtClassAddress.toClass( cl, this.getClass().getProtectionDomain() );
148 
149         CtClass customerInventoryCtClass = generateCtClassForAnEntity( CustomerInventory.class );
150         byte[] enhancedCustomerInventory = enhancer.enhance(CustomerInventory.class.getName(), customerInventoryCtClass.toBytecode());
151         CtClass enhancedCtClassCustomerInventory = cp.makeClass( new ByteArrayInputStream( enhancedCustomerInventory ) );
152         enhancedCtClassCustomerInventory.debugWriteFile("/tmp");
153         customerInventoryClass = enhancedCtClassCustomerInventory.toClass( cl, this.getClass().getProtectionDomain() );
154 
155 
156         CtClass supplierComponentPKCtClass = generateCtClassForAnEntity( SupplierComponentPK.class );
157         byte[] enhancedSupplierComponentPK = enhancer.enhanceComposite(SupplierComponentPK.class.getName(), supplierComponentPKCtClass.toBytecode());
158         CtClass enhancedCtClassSupplierComponentPK = cp.makeClass( new ByteArrayInputStream( enhancedSupplierComponentPK ) );
159         enhancedCtClassSupplierComponentPK.debugWriteFile("/tmp");
160         supplierComponentPKClass = enhancedCtClassSupplierComponentPK.toClass( cl, this.getClass().getProtectionDomain() );
161 
162 		Class entityClass = enhancedCtClass.toClass( cl, this.getClass().getProtectionDomain() );
163 		Object entityInstance = entityClass.newInstance();
164 
165 		assertTyping( ManagedEntity.class, entityInstance );
166 
167 		// call the new methods
168 		//
169 		Method setter = entityClass.getMethod( EnhancerConstants.ENTITY_ENTRY_SETTER_NAME, EntityEntry.class );
170 		Method getter = entityClass.getMethod( EnhancerConstants.ENTITY_ENTRY_GETTER_NAME );
171 		assertNull( getter.invoke( entityInstance ) );
172 		setter.invoke( entityInstance, makeEntityEntry() );
173 		assertNotNull( getter.invoke( entityInstance ) );
174 		setter.invoke( entityInstance, new Object[] {null} );
175 		assertNull( getter.invoke( entityInstance ) );
176 
177 		Method entityInstanceGetter = entityClass.getMethod( EnhancerConstants.ENTITY_INSTANCE_GETTER_NAME );
178 		assertSame( entityInstance, entityInstanceGetter.invoke( entityInstance ) );
179 
180 		Method previousGetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_GETTER_NAME );
181 		Method previousSetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_SETTER_NAME, ManagedEntity.class );
182 		previousSetter.invoke( entityInstance, entityInstance );
183 		assertSame( entityInstance, previousGetter.invoke( entityInstance ) );
184 
185 		Method nextGetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_GETTER_NAME );
186 		Method nextSetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_SETTER_NAME, ManagedEntity.class );
187 		nextSetter.invoke( entityInstance, entityInstance );
188 		assertSame( entityInstance, nextGetter.invoke( entityInstance ) );
189 
190 		// add an attribute interceptor...
191 		Method interceptorGetter = entityClass.getMethod( EnhancerConstants.INTERCEPTOR_GETTER_NAME );
192 		Method interceptorSetter = entityClass.getMethod( EnhancerConstants.INTERCEPTOR_SETTER_NAME, PersistentAttributeInterceptor.class );
193 
194 		assertNull( interceptorGetter.invoke( entityInstance ) );
195 		entityClass.getMethod( "getId" ).invoke( entityInstance );
196 
197 		interceptorSetter.invoke( entityInstance, new LocalPersistentAttributeInterceptor() );
198 		assertNotNull(interceptorGetter.invoke(entityInstance));
199 
200 		// dirty checking is unfortunately just printlns for now... just verify the test output
201 		entityClass.getMethod( "getId" ).invoke( entityInstance );
202 		entityClass.getMethod( "setId", Integer.class ).invoke( entityInstance, entityClass.getMethod( "getId" ).invoke( entityInstance ) );
203 		entityClass.getMethod( "setId", Integer.class ).invoke( entityInstance, 1 );
204         assertTrue((Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance));
205 
206 		entityClass.getMethod( "setFirstName", String.class ).invoke(entityInstance, "Erik");
207 
208 		entityClass.getMethod( "setLastName", String.class ).invoke(entityInstance, "Mykland");
209         assertEquals(3, ((Set) entityClass.getMethod("$$_hibernate_getDirtyAttributes").invoke(entityInstance)).size());
210 
211         entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
212 
213         //testing composite object
214         assert addressClass != null;
215         Object address =  addressClass.newInstance();
216 
217         assert customerInventoryClass != null;
218         Object customerInventory =  customerInventoryClass.newInstance();
219 
220         //Method adrInterceptorGetter = addressClass.getMethod( EnhancerConstants.INTERCEPTOR_GETTER_NAME );
221         //Method adrInterceptorSetter = addressClass.getMethod( EnhancerConstants.INTERCEPTOR_SETTER_NAME, PersistentAttributeInterceptor.class );
222         //adrInterceptorSetter.invoke( address, new LocalPersistentAttributeInterceptor() );
223 
224         entityClass.getMethod("setAddress", addressClass).invoke(entityInstance, address);
225 
226         addressClass.getMethod("setCity", String.class).invoke(address, "Arendal");
227 
228         Set tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
229         assertEquals(2, tracked.size());
230         Iterator iter = tracked.iterator();
231         assertEquals("address", iter.next());
232         assertEquals("address.city", iter.next());
233 
234         entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
235 
236         //make sure that new composite instances are cleared
237         Object address2 =  addressClass.newInstance();
238 
239         entityClass.getMethod("setAddress", addressClass).invoke(entityInstance, address2);
240         addressClass.getMethod("setStreet1", String.class).invoke(address, "Heggedalveien");
241 
242         tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
243         assertEquals(1, tracked.size());
244 
245     }

247 	private CtClass generateCtClassForAnEntity(Class entityClassToEnhance) throws Exception {
248 		ClassPool cp = new ClassPool( false );
249 		return cp.makeClass(
250 				getClass().getClassLoader().getResourceAsStream(
251 						entityClassToEnhance.getName().replace( '.', '/' ) + ".class"
252 				)
253 		);
254 	}
