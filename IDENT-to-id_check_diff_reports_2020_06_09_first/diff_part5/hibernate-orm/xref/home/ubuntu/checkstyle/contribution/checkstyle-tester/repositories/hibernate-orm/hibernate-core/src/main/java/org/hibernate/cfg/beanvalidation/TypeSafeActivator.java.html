77  	 * Used to validate a supplied ValidatorFactory instance as being castable to ValidatorFactory.
78  	 *
79  	 * @param object The supplied ValidatorFactory instance.
80  	 */
81  	@SuppressWarnings( {"UnusedDeclaration"})
82  	public static void validateSuppliedFactory(Object object) {
83  		if ( ! ValidatorFactory.class.isInstance( object ) ) {
84  			throw new IntegrationException(
85  					"Given object was not an instance of " + ValidatorFactory.class.getName()
86  							+ "[" + object.getClass().getName() + "]"
87  			);
88  		}
89  	}

91  	@SuppressWarnings("UnusedDeclaration")
92  	public static void activate(ActivationContext activationContext) {
93  		final Properties properties = activationContext.getConfiguration().getProperties();
94  		final ValidatorFactory factory;
95  		try {
96  			factory = getValidatorFactory( properties );
97  		}
98  		catch (IntegrationException e) {
99  			if ( activationContext.getValidationModes().contains( ValidationMode.CALLBACK ) ) {
100 				throw new IntegrationException( "Bean Validation provider was not available, but 'callback' validation was requested", e );
101 			}
102 			if ( activationContext.getValidationModes().contains( ValidationMode.DDL ) ) {
103 				throw new IntegrationException( "Bean Validation provider was not available, but 'ddl' validation was requested", e );
104 			}
105 
106 			LOG.debug( "Unable to acquire Bean Validation ValidatorFactory, skipping activation" );
107 			return;
108 		}
109 
110 		applyRelationalConstraints( factory, activationContext );
111 
112 		applyCallbackListeners( factory, activationContext );
113 	}

115 	@SuppressWarnings( {"UnusedDeclaration"})
116 	public static void applyCallbackListeners(ValidatorFactory validatorFactory, ActivationContext activationContext) {
117 		final Set<ValidationMode> modes = activationContext.getValidationModes();
118 		if ( ! ( modes.contains( ValidationMode.CALLBACK ) || modes.contains( ValidationMode.AUTO ) ) ) {
119 			return;
120 		}
121 
122 		// de-activate not-null tracking at the core level when Bean Validation is present unless the user explicitly
123 		// asks for it
124 		if ( activationContext.getConfiguration().getProperty( Environment.CHECK_NULLABILITY ) == null ) {
125 			activationContext.getSessionFactory().getSettings().setCheckNullability( false );
126 		}
127 
128 		final BeanValidationEventListener listener = new BeanValidationEventListener(
129 				validatorFactory,
130 				activationContext.getConfiguration().getProperties()
131 		);
132 
133 		final EventListenerRegistry listenerRegistry = activationContext.getServiceRegistry()
134 				.getService( EventListenerRegistry.class );
135 
136 		listenerRegistry.addDuplicationStrategy( DuplicationStrategyImpl.INSTANCE );
137 
138 		listenerRegistry.appendListeners( EventType.PRE_INSERT, listener );
139 		listenerRegistry.appendListeners( EventType.PRE_UPDATE, listener );
140 		listenerRegistry.appendListeners( EventType.PRE_DELETE, listener );
141 
142 		listener.initialize( activationContext.getConfiguration() );
143 	}

145 	@SuppressWarnings({"unchecked", "UnusedParameters"})
146 	private static void applyRelationalConstraints(ValidatorFactory factory, ActivationContext activationContext) {
147 		final Properties properties = activationContext.getConfiguration().getProperties();
148 		if ( ! ConfigurationHelper.getBoolean( BeanValidationIntegrator.APPLY_CONSTRAINTS, properties, true ) ){
149 			LOG.debug( "Skipping application of relational constraints from legacy Hibernate Validator" );
150 			return;
151 		}
152 
153 		final Set<ValidationMode> modes = activationContext.getValidationModes();
154 		if ( ! ( modes.contains( ValidationMode.DDL ) || modes.contains( ValidationMode.AUTO ) ) ) {
155 			return;
156 		}
157 
158 		applyRelationalConstraints(
159 				activationContext.getConfiguration().createMappings().getClasses().values(),
160 				properties,
161 				activationContext.getServiceRegistry().getService( JdbcServices.class ).getDialect()
162 		);
163 	}

165 	@SuppressWarnings( {"UnusedDeclaration"})
166 	public static void applyRelationalConstraints(Collection<PersistentClass> persistentClasses, Properties properties, Dialect dialect) {
167 		ValidatorFactory factory = getValidatorFactory( properties );
168 		Class<?>[] groupsArray = new GroupsPerOperation( properties ).get( GroupsPerOperation.Operation.DDL );
169 		Set<Class<?>> groups = new HashSet<Class<?>>( Arrays.asList( groupsArray ) );
170 
171 		for ( PersistentClass persistentClass : persistentClasses ) {
172 			final String className = persistentClass.getClassName();
173 
174 			if ( className == null || className.length() == 0 ) {
175 				continue;
176 			}
177 			Class<?> clazz;
178 			try {
179 				clazz = ReflectHelper.classForName( className, TypeSafeActivator.class );
180 			}
181 			catch ( ClassNotFoundException e ) {
182 				throw new AssertionFailure( "Entity class not found", e );
183 			}
184 
185 			try {
186 				applyDDL( "", persistentClass, clazz, factory, groups, true, dialect );
187 			}
188 			catch (Exception e) {
189 				LOG.unableToApplyConstraints( className, e );
190 			}
191 		}
192 	}

194 	private static void applyDDL(
195 			String prefix,
196 			PersistentClass persistentClass,
197 			Class<?> clazz,
198 			ValidatorFactory factory,
199 			Set<Class<?>> groups,
200 			boolean activateNotNull,
201 			Dialect dialect) {
202 		final BeanDescriptor descriptor = factory.getValidator().getConstraintsForClass( clazz );
203 		//no bean level constraints can be applied, go to the properties
204 
205 		for ( PropertyDescriptor propertyDesc : descriptor.getConstrainedProperties() ) {
206 			Property property = findPropertyByName( persistentClass, prefix + propertyDesc.getPropertyName() );
207 			boolean hasNotNull;
208 			if ( property != null ) {
209 				hasNotNull = applyConstraints(
210 						propertyDesc.getConstraintDescriptors(), property, propertyDesc, groups, activateNotNull, dialect
211 				);
212 				if ( property.isComposite() && propertyDesc.isCascaded() ) {
213 					Class<?> componentClass = ( (Component) property.getValue() ).getComponentClass();
214 
215 					/*
216 					 * we can apply not null if the upper component let's us activate not null
217 					 * and if the property is not null.
218 					 * Otherwise, all sub columns should be left nullable
219 					 */
220 					final boolean canSetNotNullOnColumns = activateNotNull && hasNotNull;
221 					applyDDL(
222 							prefix + propertyDesc.getPropertyName() + ".",
223 							persistentClass, componentClass, factory, groups,
224 							canSetNotNullOnColumns,
225                             dialect
226 					);
227 				}
228 				//FIXME add collection of components
229 			}
230 		}
231 	}

233 	private static boolean applyConstraints(
234 			Set<ConstraintDescriptor<?>> constraintDescriptors,
235 			Property property,
236 			PropertyDescriptor propertyDesc,
237 			Set<Class<?>> groups,
238 			boolean canApplyNotNull,
239 			Dialect dialect) {
240 		boolean hasNotNull = false;
241 		for ( ConstraintDescriptor<?> descriptor : constraintDescriptors ) {
242 			if ( groups != null && Collections.disjoint( descriptor.getGroups(), groups ) ) {
243 				continue;
244 			}
245 
246 			if ( canApplyNotNull ) {
247 				hasNotNull = hasNotNull || applyNotNull( property, descriptor );
248 			}
249 
250 			// apply bean validation specific constraints
251 			applyDigits( property, descriptor );
252 			applySize( property, descriptor, propertyDesc );
253 			applyMin( property, descriptor, dialect );
254 			applyMax( property, descriptor, dialect );
255 
256 			// apply hibernate validator specific constraints - we cannot import any HV specific classes though!
257 			// no need to check explicitly for @Range. @Range is a composed constraint using @Min and @Max which
258 			// will be taken care later
259 			applyLength( property, descriptor, propertyDesc );
260 
261 			// pass an empty set as composing constraints inherit the main constraint and thus are matching already
262 			hasNotNull = hasNotNull || applyConstraints(
263 					descriptor.getComposingConstraints(),
264 					property, propertyDesc, null,
265 					canApplyNotNull,
266                     dialect
267 			);
268 		}
269 		return hasNotNull;
270 	}

272 	private static void applyMin(Property property, ConstraintDescriptor<?> descriptor, Dialect dialect) {
273 		if ( Min.class.equals( descriptor.getAnnotation().annotationType() ) ) {
274 			@SuppressWarnings("unchecked")
275 			ConstraintDescriptor<Min> minConstraint = (ConstraintDescriptor<Min>) descriptor;
276 			long min = minConstraint.getAnnotation().value();
277 
278 			Column col = (Column) property.getColumnIterator().next();
279 			String checkConstraint = col.getQuotedName(dialect) + ">=" + min;
280 			applySQLCheck( col, checkConstraint );
281 		}
282 	}
283 
284 	private static void applyMax(Property property, ConstraintDescriptor<?> descriptor, Dialect dialect) {
285 		if ( Max.class.equals( descriptor.getAnnotation().annotationType() ) ) {
286 			@SuppressWarnings("unchecked")
287 			ConstraintDescriptor<Max> maxConstraint = (ConstraintDescriptor<Max>) descriptor;
288 			long max = maxConstraint.getAnnotation().value();
289 			Column col = (Column) property.getColumnIterator().next();
290 			String checkConstraint = col.getQuotedName(dialect) + "<=" + max;
291 			applySQLCheck( col, checkConstraint );
292 		}
293 	}

295 	private static void applySQLCheck(Column col, String checkConstraint) {
296 		String existingCheck = col.getCheckConstraint();
297 		// need to check whether the new check is already part of the existing check, because applyDDL can be called
298 		// multiple times
299 		if ( StringHelper.isNotEmpty( existingCheck ) && !existingCheck.contains( checkConstraint ) ) {
300 			checkConstraint = col.getCheckConstraint() + " AND " + checkConstraint;
301 		}
302 		col.setCheckConstraint( checkConstraint );
303 	}

305 	private static boolean applyNotNull(Property property, ConstraintDescriptor<?> descriptor) {
306 		boolean hasNotNull = false;
307 		if ( NotNull.class.equals( descriptor.getAnnotation().annotationType() ) ) {
308 			if ( !( property.getPersistentClass() instanceof SingleTableSubclass ) ) {
309 				//single table should not be forced to null
310 				if ( !property.isComposite() ) { //composite should not add not-null on all columns
311 					@SuppressWarnings( "unchecked" )
312 					Iterator<Column> iter = property.getColumnIterator();
313 					while ( iter.hasNext() ) {
314 						iter.next().setNullable( false );
315 						hasNotNull = true;
316 					}
317 				}
318 			}
319 			hasNotNull = true;
320 		}
321 		return hasNotNull;
322 	}

324 	private static void applyDigits(Property property, ConstraintDescriptor<?> descriptor) {
325 		if ( Digits.class.equals( descriptor.getAnnotation().annotationType() ) ) {
326 			@SuppressWarnings("unchecked")
327 			ConstraintDescriptor<Digits> digitsConstraint = (ConstraintDescriptor<Digits>) descriptor;
328 			int integerDigits = digitsConstraint.getAnnotation().integer();
329 			int fractionalDigits = digitsConstraint.getAnnotation().fraction();
330 			Column col = (Column) property.getColumnIterator().next();
331 			col.setPrecision( integerDigits + fractionalDigits );
332 			col.setScale( fractionalDigits );
333 		}
334 	}
335 
336 	private static void applySize(Property property, ConstraintDescriptor<?> descriptor, PropertyDescriptor propertyDescriptor) {
337 		if ( Size.class.equals( descriptor.getAnnotation().annotationType() )
338 				&& String.class.equals( propertyDescriptor.getElementClass() ) ) {
339 			@SuppressWarnings("unchecked")
340 			ConstraintDescriptor<Size> sizeConstraint = (ConstraintDescriptor<Size>) descriptor;
341 			int max = sizeConstraint.getAnnotation().max();
342 			Column col = (Column) property.getColumnIterator().next();
343 			if ( max < Integer.MAX_VALUE ) {
344 				col.setLength( max );
345 			}
346 		}
347 	}
348 
349 	private static void applyLength(Property property, ConstraintDescriptor<?> descriptor, PropertyDescriptor propertyDescriptor) {
350 		if ( "org.hibernate.validator.constraints.Length".equals(
351 				descriptor.getAnnotation().annotationType().getName()
352 		)
353 				&& String.class.equals( propertyDescriptor.getElementClass() ) ) {
354 			@SuppressWarnings("unchecked")
355 			int max = (Integer) descriptor.getAttributes().get( "max" );
356 			Column col = (Column) property.getColumnIterator().next();
357 			if ( max < Integer.MAX_VALUE ) {
358 				col.setLength( max );
359 			}
360 		}
361 	}

363 	/**
364 	 * @param associatedClass
365 	 * @param propertyName
366      * @return the property by path in a recursive way, including IdentifierProperty in the loop if propertyName is
367      * <code>null</code>.  If propertyName is <code>null</code> or empty, the IdentifierProperty is returned
368 	 */
369 	private static Property findPropertyByName(PersistentClass associatedClass, String propertyName) {
370 		Property property = null;
371 		Property idProperty = associatedClass.getIdentifierProperty();
372 		String idName = idProperty != null ? idProperty.getName() : null;
373 		try {
374 			if ( propertyName == null
375 					|| propertyName.length() == 0
376 					|| propertyName.equals( idName ) ) {
377 				//default to id
378 				property = idProperty;
379 			}
380 			else {
381 				if ( propertyName.indexOf( idName + "." ) == 0 ) {
382 					property = idProperty;
383 					propertyName = propertyName.substring( idName.length() + 1 );
384 				}
385 				StringTokenizer st = new StringTokenizer( propertyName, ".", false );
386 				while ( st.hasMoreElements() ) {
387 					String element = (String) st.nextElement();
388 					if ( property == null ) {
389 						property = associatedClass.getProperty( element );
390 					}
391 					else {
392 						if ( !property.isComposite() ) {
393 							return null;
394 						}
395 						property = ( (Component) property.getValue() ).getProperty( element );
396 					}
397 				}
398 			}
399 		}
400 		catch ( MappingException e ) {
401 			try {
402 				//if we do not find it try to check the identifier mapper
403 				if ( associatedClass.getIdentifierMapper() == null ) {
404 					return null;
405 				}
406 				StringTokenizer st = new StringTokenizer( propertyName, ".", false );
407 				while ( st.hasMoreElements() ) {
408 					String element = (String) st.nextElement();
409 					if ( property == null ) {
410 						property = associatedClass.getIdentifierMapper().getProperty( element );
411 					}
412 					else {
413 						if ( !property.isComposite() ) {
414 							return null;
415 						}
416 						property = ( (Component) property.getValue() ).getProperty( element );
417 					}
418 				}
419 			}
420 			catch ( MappingException ee ) {
421 				return null;
422 			}
423 		}
424 		return property;
425 	}

427 	private static ValidatorFactory getValidatorFactory(Map<Object, Object> properties) {
428 		ValidatorFactory factory = null;
429 		if ( properties != null ) {
430 			Object unsafeProperty = properties.get( FACTORY_PROPERTY );
431 			if ( unsafeProperty != null ) {
432 				try {
433 					factory = ValidatorFactory.class.cast( unsafeProperty );
434 				}
435 				catch ( ClassCastException e ) {
436 					throw new IntegrationException(
437 							"Property " + FACTORY_PROPERTY
438 									+ " should contain an object of type " + ValidatorFactory.class.getName()
439 					);
440 				}
441 			}
442 		}
443 		if ( factory == null ) {
444 			try {
445 				factory = Validation.buildDefaultValidatorFactory();
446 			}
447 			catch ( Exception e ) {
448 				throw new IntegrationException( "Unable to build the default ValidatorFactory", e );
449 			}
450 		}
451 		return factory;
452 	}
