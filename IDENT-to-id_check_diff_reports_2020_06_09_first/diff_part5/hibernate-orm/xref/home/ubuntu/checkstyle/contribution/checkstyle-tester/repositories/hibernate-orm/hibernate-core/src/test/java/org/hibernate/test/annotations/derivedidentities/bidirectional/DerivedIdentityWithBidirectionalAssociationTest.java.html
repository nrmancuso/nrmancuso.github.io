24  package org.hibernate.test.annotations.derivedidentities.bidirectional;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.test.util.SchemaUtil;
30  import org.hibernate.testing.Skip;
31  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
32  
33  import static org.junit.Assert.assertEquals;
34  import static org.junit.Assert.assertTrue;
35  
36  /**
37   * @author Hardy Ferentschik
38   */
39  @Skip( condition = Skip.AlwaysSkip.class,message = "sdf")
40  public class DerivedIdentityWithBidirectionalAssociationTest extends BaseCoreFunctionalTestCase {
41  	@Test
42  	public void testBidirectionalAssociation() throws Exception {
43  		assertTrue( SchemaUtil.isColumnPresent( "Dependent", "emp_empId", configuration() ) );
44  		assertTrue( !SchemaUtil.isColumnPresent( "Dependent", "empPK", configuration() ) );
45  		Employee e = new Employee();
46  		e.empId = 1;
47  		e.empName = "Emmanuel";
48  
49  		Session s = openSession();
50  		s.getTransaction().begin();
51  		s.persist( e );
52  		Dependent d = new Dependent();
53  		d.emp = e;
54  		s.persist( d );
55  		s.flush();
56  		s.clear();
57  		d = getDerivedClassById( e, s, Dependent.class );
58  		assertEquals( e.empId, d.emp.empId );
59  		s.getTransaction().rollback();
60  		s.close();
61  	}
62  
63  	private <T> T getDerivedClassById(Employee e, Session s, Class<T> clazz) {
64  		return clazz.cast( s.createQuery( "from " + clazz.getName() + " d where d.emp.empId = :empId" )
65  						.setParameter( "empId", e.empId ).uniqueResult() );
66  	}
67  
68  	@Override
69  	protected Class<?>[] getAnnotatedClasses() {
70  		return new Class<?>[] {
71  				Dependent.class,
72  				Employee.class
73  		};
74  	}
