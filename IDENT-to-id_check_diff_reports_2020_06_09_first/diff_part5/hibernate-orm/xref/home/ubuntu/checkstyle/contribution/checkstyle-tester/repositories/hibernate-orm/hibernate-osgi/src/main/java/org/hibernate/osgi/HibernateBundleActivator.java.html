24  package org.hibernate.osgi;
25  
26  import java.util.Dictionary;
27  import java.util.Hashtable;
28  
29  import javax.persistence.spi.PersistenceProvider;
30  
31  import org.hibernate.Session;
32  import org.hibernate.SessionFactory;
33  import org.hibernate.ejb.HibernatePersistence;
34  import org.hibernate.internal.util.ClassLoaderHelper;
35  import org.osgi.framework.BundleActivator;
36  import org.osgi.framework.BundleContext;
37  import org.osgi.framework.FrameworkUtil;
38  import org.osgi.framework.ServiceRegistration;
39  
40  /**
41   * This BundleActivator provides three different uses of Hibernate in OSGi
42   * environments:
43   * 
44   * 1.) Enterprise OSGi JPA: The OSGi container/runtime is in charge of managing
45   *     all of the client bundles' persistence units.  The container/runtime is
46   *     also in charge of creating and managing the EntityManagerFactory through a
47   *     registered PersistenceProvider (this).
48   * 2.) Un-managed OSGI JPA: Same as #1, but the container does not manage
49   *     the persistence units.  Client bundles identify a typical
50   *     PersistenceProvider, registered by this activator.
51   * 3.) Client bundles create and manage their own SessionFactory.  A
52   *     SessionFactory is registered as an OSGi ServiceFactory -- each requesting
53   *     bundle gets its own instance of a SessionFactory.  The use of services,
54   *     rather than direct use of Configuration, is necessary to shield users
55   *     from ClassLoader issues.  See {@link #OsgiSessionFactoryService} for more
56   *     information.
57   * 
58   * @author Brett Meyer
59   * @author Tim Ward
60   */
61  public class HibernateBundleActivator implements BundleActivator {
62  	
63  	private ServiceRegistration<?> persistenceProviderService;
64  	private ServiceRegistration<?> sessionFactoryService;
65  	
66  	@Override
67  	public void start(BundleContext context) throws Exception {
68  
69  		OsgiClassLoader osgiClassLoader = new OsgiClassLoader();
70  		osgiClassLoader.addBundle( FrameworkUtil.getBundle( Session.class ) );
71  		osgiClassLoader.addBundle( FrameworkUtil.getBundle( HibernatePersistence.class ) );
72  		ClassLoaderHelper.overridenClassLoader = osgiClassLoader;
73  
74  		OsgiJtaPlatform osgiJtaPlatform = new OsgiJtaPlatform( context );
75  
76  		Dictionary properties = new Hashtable();
77  		// In order to support existing persistence.xml files, register
78  		// using the legacy provider name.
79  		properties.put( "javax.persistence.provider", HibernatePersistence.class.getName() );
80  		persistenceProviderService = context.registerService( PersistenceProvider.class.getName(), 
81  				new OsgiPersistenceProviderService( osgiClassLoader, osgiJtaPlatform, context ), properties );
82  		
83  		sessionFactoryService = context.registerService( SessionFactory.class.getName(),
84  				new OsgiSessionFactoryService( osgiClassLoader, osgiJtaPlatform, context ), new Hashtable());
85  	}
86  
87  	@Override
88  	public void stop(BundleContext context) throws Exception {
89  		persistenceProviderService.unregister();
90  		persistenceProviderService = null;
91  		sessionFactoryService.unregister();
92  		sessionFactoryService = null;
93  
94  		ClassLoaderHelper.overridenClassLoader = null;
95  	}
96  }
