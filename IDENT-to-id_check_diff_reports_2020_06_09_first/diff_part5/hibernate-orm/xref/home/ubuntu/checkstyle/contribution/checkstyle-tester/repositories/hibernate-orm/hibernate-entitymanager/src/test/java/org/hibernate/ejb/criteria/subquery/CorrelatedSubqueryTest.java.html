48  public class CorrelatedSubqueryTest extends AbstractMetamodelSpecificTest {
49  	@Test
50  	public void testBasicCorrelation() {
51  		CriteriaBuilder builder = entityManagerFactory().getCriteriaBuilder();
52  		EntityManager em = getOrCreateEntityManager();
53  		em.getTransaction().begin();
54  
55  		CriteriaQuery<Customer> criteria = builder.createQuery( Customer.class );
56  		Root<Customer> customer = criteria.from( Customer.class );
57  		criteria.select( customer );
58  		Subquery<Order> orderSubquery = criteria.subquery( Order.class );
59  		Root<Customer> customerCorrelationRoot = orderSubquery.correlate( customer );
60  		Join<Customer, Order> customerOrderCorrelationJoin = customerCorrelationRoot.join( Customer_.orders );
61  		orderSubquery.select( customerOrderCorrelationJoin );
62  		criteria.where( builder.not( builder.exists( orderSubquery ) ) );
63  		em.createQuery( criteria ).getResultList();
64  
65  		em.getTransaction().commit();
66  		em.close();
67  	}

69  	@Test
70  	public void testRestrictedCorrelation() {
71  		CriteriaBuilder builder = entityManagerFactory().getCriteriaBuilder();
72  		EntityManager em = getOrCreateEntityManager();
73  		em.getTransaction().begin();
74  
75  		CriteriaQuery<Order> criteria = builder.createQuery( Order.class );
76  		Root<Order> orderRoot = criteria.from( Order.class );
77  		criteria.select( orderRoot );
78  		// create correlated subquery
79  		Subquery<Customer> customerSubquery = criteria.subquery( Customer.class );
80  		Root<Order> orderRootCorrelation = customerSubquery.correlate( orderRoot );
81  		Join<Order, Customer> orderCustomerJoin = orderRootCorrelation.join( Order_.customer );
82  		customerSubquery.where( builder.like( orderCustomerJoin.get( Customer_.name ), "%Caruso" ) )
83  				.select( orderCustomerJoin );
84  		criteria.where( builder.exists( customerSubquery ) );
85  		em.createQuery( criteria ).getResultList();
86  
87  		em.getTransaction().commit();
88  		em.close();
89  	}

91  	@Test
92  	@SkipForDialect(value=SybaseASE15Dialect.class, jiraKey="HHH-3032")
93  	public void testCorrelationExplicitSelectionCorrelation() {
94  		CriteriaBuilder builder = entityManagerFactory().getCriteriaBuilder();
95  		EntityManager em = getOrCreateEntityManager();
96  		em.getTransaction().begin();
97  
98  		CriteriaQuery<Customer> customerCriteria = builder.createQuery( Customer.class );
99  		Root<Customer> customer = customerCriteria.from( Customer.class );
100 		Join<Customer, Order> o = customer.join( Customer_.orders );
101 		Subquery<Order> sq = customerCriteria.subquery(Order.class);
102 		Join<Customer, Order> sqo = sq.correlate(o);
103 		Join<Order, LineItem> sql = sqo.join(Order_.lineItems);
104 		sq.where( builder.gt(sql.get( LineItem_.quantity), 3) );
105 		// use the correlation itself as the subquery selection (initially caused problems wrt aliases)
106 		sq.select(sqo);
107 		customerCriteria.select(customer).distinct(true);
108 		customerCriteria.where(builder.exists(sq));
109 		em.createQuery( customerCriteria ).getResultList();
110 
111 		em.getTransaction().commit();
112 		em.close();
113 	}
