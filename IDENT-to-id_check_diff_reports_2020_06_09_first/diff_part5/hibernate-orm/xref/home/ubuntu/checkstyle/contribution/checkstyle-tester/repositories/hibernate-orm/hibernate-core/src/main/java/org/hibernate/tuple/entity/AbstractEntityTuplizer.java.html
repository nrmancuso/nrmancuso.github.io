171 	/**
172 	 * Constructs a new AbstractEntityTuplizer instance.
173 	 *
174 	 * @param entityMetamodel The "interpreted" information relating to the mapped entity.
175 	 * @param mappingInfo The parsed "raw" mapping data relating to the given entity.
176 	 */
177 	public AbstractEntityTuplizer(EntityMetamodel entityMetamodel, PersistentClass mappingInfo) {
178 		this.entityMetamodel = entityMetamodel;
179 
180 		if ( !entityMetamodel.getIdentifierProperty().isVirtual() ) {
181 			idGetter = buildPropertyGetter( mappingInfo.getIdentifierProperty(), mappingInfo );
182 			idSetter = buildPropertySetter( mappingInfo.getIdentifierProperty(), mappingInfo );
183 		}
184 		else {
185 			idGetter = null;
186 			idSetter = null;
187 		}
188 
189 		propertySpan = entityMetamodel.getPropertySpan();
190 
191         getters = new Getter[propertySpan];
192 		setters = new Setter[propertySpan];
193 
194 		Iterator itr = mappingInfo.getPropertyClosureIterator();
195 		boolean foundCustomAccessor=false;
196 		int i=0;
197 		while ( itr.hasNext() ) {
198 			//TODO: redesign how PropertyAccessors are acquired...
199 			Property property = (Property) itr.next();
200 			getters[i] = buildPropertyGetter(property, mappingInfo);
201 			setters[i] = buildPropertySetter(property, mappingInfo);
202 			if ( !property.isBasicPropertyAccessor() ) {
203 				foundCustomAccessor = true;
204 			}
205 			i++;
206 		}
207 		hasCustomAccessors = foundCustomAccessor;
208 
209         instantiator = buildInstantiator( mappingInfo );
210 
211 		if ( entityMetamodel.isLazy() ) {
212 			proxyFactory = buildProxyFactory( mappingInfo, idGetter, idSetter );
213 			if (proxyFactory == null) {
214 				entityMetamodel.setLazy( false );
215 			}
216 		}
217 		else {
218 			proxyFactory = null;
219 		}
220 
221 		Component mapper = mappingInfo.getIdentifierMapper();
222 		if ( mapper == null ) {
223 			identifierMapperType = null;
224 			mappedIdentifierValueMarshaller = null;
225 		}
226 		else {
227 			identifierMapperType = (CompositeType) mapper.getType();
228 			mappedIdentifierValueMarshaller = buildMappedIdentifierValueMarshaller(
229 					(ComponentType) entityMetamodel.getIdentifierProperty().getType(),
230 					(ComponentType) identifierMapperType
231 			);
232 		}
233 	}

235 	/**
236 	 * Constructs a new AbstractEntityTuplizer instance.
237 	 *
238 	 * @param entityMetamodel The "interpreted" information relating to the mapped entity.
239 	 * @param mappingInfo The parsed "raw" mapping data relating to the given entity.
240 	 */
241 	public AbstractEntityTuplizer(EntityMetamodel entityMetamodel, EntityBinding mappingInfo) {
242 		this.entityMetamodel = entityMetamodel;
243 
244 		if ( !entityMetamodel.getIdentifierProperty().isVirtual() ) {
245 			idGetter = buildPropertyGetter( mappingInfo.getHierarchyDetails().getEntityIdentifier().getValueBinding() );
246 			idSetter = buildPropertySetter( mappingInfo.getHierarchyDetails().getEntityIdentifier().getValueBinding() );
247 		}
248 		else {
249 			idGetter = null;
250 			idSetter = null;
251 		}
252 
253 		propertySpan = entityMetamodel.getPropertySpan();
254 
255 		getters = new Getter[ propertySpan ];
256 		setters = new Setter[ propertySpan ];
257 
258 		boolean foundCustomAccessor = false;
259 		int i = 0;
260 		for ( AttributeBinding property : mappingInfo.getAttributeBindingClosure() ) {
261 			if ( property == mappingInfo.getHierarchyDetails().getEntityIdentifier().getValueBinding() ) {
262 				continue; // ID binding processed above
263 			}
264 
265 			//TODO: redesign how PropertyAccessors are acquired...
266 			getters[ i ] = buildPropertyGetter( property );
267 			setters[ i ] = buildPropertySetter( property );
268 			if ( ! property.isBasicPropertyAccessor() ) {
269 				foundCustomAccessor = true;
270 			}
271 			i++;
272 		}
273 		hasCustomAccessors = foundCustomAccessor;
274 
275 		instantiator = buildInstantiator( mappingInfo );
276 
277 		if ( entityMetamodel.isLazy() ) {
278 			proxyFactory = buildProxyFactory( mappingInfo, idGetter, idSetter );
279 			if ( proxyFactory == null ) {
280 				entityMetamodel.setLazy( false );
281 			}
282 		}
283 		else {
284 			proxyFactory = null;
285 		}
286 
287 
288 		// TODO: Fix this when components are working (HHH-6173)
289 		//Component mapper = mappingInfo.getEntityIdentifier().getIdentifierMapper();
290 		Component mapper = null;
291 		if ( mapper == null ) {
292 			identifierMapperType = null;
293 			mappedIdentifierValueMarshaller = null;
294 		}
295 		else {
296 			identifierMapperType = ( CompositeType ) mapper.getType();
297 			mappedIdentifierValueMarshaller = buildMappedIdentifierValueMarshaller(
298 					( ComponentType ) entityMetamodel.getIdentifierProperty().getType(),
299 					( ComponentType ) identifierMapperType
300 			);
301 		}
302 	}

326 	public Serializable getIdentifier(Object entity, SessionImplementor session) {
327 		final Object id;
328 		if ( entityMetamodel.getIdentifierProperty().isEmbedded() ) {
329 			id = entity;
330 		}
331 		else if ( HibernateProxy.class.isInstance( entity ) ) {
332 			id = ( (HibernateProxy) entity ).getHibernateLazyInitializer().getIdentifier();
333 		}
334 		else {
335 			if ( idGetter == null ) {
336 				if (identifierMapperType==null) {
337 					throw new HibernateException( "The class has no identifier property: " + getEntityName() );
338 				}
339 				else {
340 					id = mappedIdentifierValueMarshaller.getIdentifier( entity, getEntityMode(), session );
341 				}
342 			}
343 			else {
344                 id = idGetter.get( entity );
345             }
346         }
347 
348 		try {
349 			return (Serializable) id;
350 		}
351 		catch ( ClassCastException cce ) {
352 			StringBuilder msg = new StringBuilder( "Identifier classes must be serializable. " );
353 			if ( id != null ) {
354 				msg.append( id.getClass().getName() ).append( " is not serializable. " );
355 			}
356 			if ( cce.getMessage() != null ) {
357 				msg.append( cce.getMessage() );
358 			}
359 			throw new ClassCastException( msg.toString() );
360 		}
361 	}

384 			idSetter.set( entity, id, getFactory() );
385 		}
386 		else if ( identifierMapperType != null ) {
387 			mappedIdentifierValueMarshaller.setIdentifier( entity, id, getEntityMode(), session );
388 		}

396 	private final MappedIdentifierValueMarshaller mappedIdentifierValueMarshaller;
397 
398 	private static MappedIdentifierValueMarshaller buildMappedIdentifierValueMarshaller(
399 			ComponentType mappedIdClassComponentType,
400 			ComponentType virtualIdComponent) {
401 		// so basically at this point we know we have a "mapped" composite identifier
402 		// which is an awful way to say that the identifier is represented differently
403 		// in the entity and in the identifier value.  The incoming value should
404 		// be an instance of the mapped identifier class (@IdClass) while the incoming entity
405 		// should be an instance of the entity class as defined by metamodel.
406 		//
407 		// However, even within that we have 2 potential scenarios:
408 		//		1) @IdClass types and entity @Id property types match
409 		//			- return a NormalMappedIdentifierValueMarshaller
410 		//		2) They do not match
411 		//			- return a IncrediblySillyJpaMapsIdMappedIdentifierValueMarshaller
412 		boolean wereAllEquivalent = true;
413 		// the sizes being off is a much bigger problem that should have been caught already...
414 		for ( int i = 0; i < virtualIdComponent.getSubtypes().length; i++ ) {
415 			if ( virtualIdComponent.getSubtypes()[i].isEntityType()
416 					&& ! mappedIdClassComponentType.getSubtypes()[i].isEntityType() ) {
417 				wereAllEquivalent = false;
418 				break;
419 			}
420 		}
421 
422 		return wereAllEquivalent
423 				? (MappedIdentifierValueMarshaller) new NormalMappedIdentifierValueMarshaller( virtualIdComponent, mappedIdClassComponentType )
424 				: (MappedIdentifierValueMarshaller) new IncrediblySillyJpaMapsIdMappedIdentifierValueMarshaller( virtualIdComponent, mappedIdClassComponentType );
425 	}

427 	private static class NormalMappedIdentifierValueMarshaller implements MappedIdentifierValueMarshaller {
428 		private final ComponentType virtualIdComponent;
429 		private final ComponentType mappedIdentifierType;
430 
431 		private NormalMappedIdentifierValueMarshaller(ComponentType virtualIdComponent, ComponentType mappedIdentifierType) {
432 			this.virtualIdComponent = virtualIdComponent;
433 			this.mappedIdentifierType = mappedIdentifierType;
434 		}

436 		public Object getIdentifier(Object entity, EntityMode entityMode, SessionImplementor session) {
437 			Object id = mappedIdentifierType.instantiate( entityMode );
438 			final Object[] propertyValues = virtualIdComponent.getPropertyValues( entity, entityMode );
439 			mappedIdentifierType.setPropertyValues( id, propertyValues, entityMode );
440 			return id;
441 		}
442 
443 		public void setIdentifier(Object entity, Serializable id, EntityMode entityMode, SessionImplementor session) {
444 			virtualIdComponent.setPropertyValues(
445 					entity,
446 					mappedIdentifierType.getPropertyValues( id, session ),
447 					entityMode
448 			);
449 		}
450 	}

452 	private static class IncrediblySillyJpaMapsIdMappedIdentifierValueMarshaller implements MappedIdentifierValueMarshaller {
453 		private final ComponentType virtualIdComponent;
454 		private final ComponentType mappedIdentifierType;
455 
456 		private IncrediblySillyJpaMapsIdMappedIdentifierValueMarshaller(ComponentType virtualIdComponent, ComponentType mappedIdentifierType) {
457 			this.virtualIdComponent = virtualIdComponent;
458 			this.mappedIdentifierType = mappedIdentifierType;
459 		}
460 
461 		public Object getIdentifier(Object entity, EntityMode entityMode, SessionImplementor session) {
462 			final Object id = mappedIdentifierType.instantiate( entityMode );
463 			final Object[] propertyValues = virtualIdComponent.getPropertyValues( entity, entityMode );
464 			final Type[] subTypes = virtualIdComponent.getSubtypes();
465 			final Type[] copierSubTypes = mappedIdentifierType.getSubtypes();
466 			final Iterable<PersistEventListener> persistEventListeners = persistEventListeners( session );
467 			final PersistenceContext persistenceContext = session.getPersistenceContext();
468 			final int length = subTypes.length;
469 			for ( int i = 0 ; i < length; i++ ) {
470 				if ( propertyValues[i] == null ) {
471 					throw new HibernateException( "No part of a composite identifier may be null" );
472 				}
473 				//JPA 2 @MapsId + @IdClass points to the pk of the entity
474 				if ( subTypes[i].isAssociationType() && ! copierSubTypes[i].isAssociationType() ) {
475 					// we need a session to handle this use case
476 					if ( session == null ) {
477 						throw new AssertionError(
478 								"Deprecated version of getIdentifier (no session) was used but session was required"
479 						);
480 					}
481 					final Object subId;
482 					if ( HibernateProxy.class.isInstance( propertyValues[i] ) ) {
483 						subId = ( (HibernateProxy) propertyValues[i] ).getHibernateLazyInitializer().getIdentifier();
484 					}
485 					else {
486 						EntityEntry pcEntry = session.getPersistenceContext().getEntry( propertyValues[i] );
487 						if ( pcEntry != null ) {
488 							subId = pcEntry.getId();
489 						}
490 						else {
491 							LOG.debug( "Performing implicit derived identity cascade" );
492 							final PersistEvent event = new PersistEvent( null, propertyValues[i], (EventSource) session );
493 							for ( PersistEventListener listener : persistEventListeners ) {
494 								listener.onPersist( event );
495 							}
496 							pcEntry = persistenceContext.getEntry( propertyValues[i] );
497 							if ( pcEntry == null || pcEntry.getId() == null ) {
498 								throw new HibernateException( "Unable to process implicit derived identity cascade" );
499 							}
500 							else {
501 								subId = pcEntry.getId();
502 							}
503 						}
504 					}
505 					propertyValues[i] = subId;
506 				}
507 			}
508 			mappedIdentifierType.setPropertyValues( id, propertyValues, entityMode );
509 			return id;
510 		}
511 
512 		public void setIdentifier(Object entity, Serializable id, EntityMode entityMode, SessionImplementor session) {
513 			final Object[] extractedValues = mappedIdentifierType.getPropertyValues( id, entityMode );
514 			final Object[] injectionValues = new Object[ extractedValues.length ];
515 			final PersistenceContext persistenceContext = session.getPersistenceContext();
516 			for ( int i = 0; i < virtualIdComponent.getSubtypes().length; i++ ) {
517 				final Type virtualPropertyType = virtualIdComponent.getSubtypes()[i];
518 				final Type idClassPropertyType = mappedIdentifierType.getSubtypes()[i];
519 				if ( virtualPropertyType.isEntityType() && ! idClassPropertyType.isEntityType() ) {
520 					if ( session == null ) {
521 						throw new AssertionError(
522 								"Deprecated version of getIdentifier (no session) was used but session was required"
523 						);
524 					}
525 					final String associatedEntityName = ( (EntityType) virtualPropertyType ).getAssociatedEntityName();
526 					final EntityKey entityKey = session.generateEntityKey(
527 							(Serializable) extractedValues[i],
528 							session.getFactory().getEntityPersister( associatedEntityName )
529 					);
530 					// it is conceivable there is a proxy, so check that first
531 					Object association = persistenceContext.getProxy( entityKey );
532 					if ( association == null ) {
533 						// otherwise look for an initialized version
534 						association = persistenceContext.getEntity( entityKey );
535 					}
536 					injectionValues[i] = association;
537 				}
538 				else {
539 					injectionValues[i] = extractedValues[i];
540 				}
541 			}
542 			virtualIdComponent.setPropertyValues( entity, injectionValues, entityMode );
543 		}
544 	}

601 	public Object[] getPropertyValues(Object entity) throws HibernateException {
602 		boolean getAll = shouldGetAllProperties( entity );
603 		final int span = entityMetamodel.getPropertySpan();
604 		final Object[] result = new Object[span];
605 
606 		for ( int j = 0; j < span; j++ ) {
607 			StandardProperty property = entityMetamodel.getProperties()[j];
608 			if ( getAll || !property.isLazy() ) {
609 				result[j] = getters[j].get( entity );
610 			}
611 			else {
612 				result[j] = LazyPropertyInitializer.UNFETCHED_PROPERTY;
613 			}
614 		}
615 		return result;
616 	}

618 	public Object[] getPropertyValuesToInsert(Object entity, Map mergeMap, SessionImplementor session)
619 	throws HibernateException {
620 		final int span = entityMetamodel.getPropertySpan();
621 		final Object[] result = new Object[span];
622 
623 		for ( int j = 0; j < span; j++ ) {

633 	public Object getPropertyValue(Object entity, String propertyPath) throws HibernateException {
634 		int loc = propertyPath.indexOf('.');
635 		String basePropertyName = loc > 0
636 				? propertyPath.substring( 0, loc )
637 				: propertyPath;
638 		//final int index = entityMetamodel.getPropertyIndexOrNull( basePropertyName );
639 		Integer index = entityMetamodel.getPropertyIndexOrNull( basePropertyName );
640 		if (index == null) {
641 			propertyPath = "_identifierMapper." + propertyPath;
642 			loc = propertyPath.indexOf('.');
643 			basePropertyName = loc > 0
644 				? propertyPath.substring( 0, loc )
645 				: propertyPath;
646 		}
647 		index = entityMetamodel.getPropertyIndexOrNull( basePropertyName );
648 		final Object baseValue = getPropertyValue( entity, index.intValue() );
649 		if ( loc > 0 ) {
650 			if ( baseValue == null ) {
651 				return null;
652 			}
653 			return getComponentValue(
654 					(ComponentType) entityMetamodel.getPropertyTypes()[index.intValue()],
655 					baseValue,
656 					propertyPath.substring(loc+1)
657 			);
658 		}
659 		else {
660 			return baseValue;
661 		}
662 	}

664 	/**
665 	 * Extract a component property value.
666 	 *
667 	 * @param type The component property types.
668 	 * @param component The component instance itself.
669 	 * @param propertyPath The property path for the property to be extracted.
670 	 * @return The property value extracted.
671 	 */
672 	protected Object getComponentValue(ComponentType type, Object component, String propertyPath) {
673 		final int loc = propertyPath.indexOf( '.' );
674 		final String basePropertyName = loc > 0
675 				? propertyPath.substring( 0, loc )
676 				: propertyPath;
677 		final int index = findSubPropertyIndex( type, basePropertyName );
678 		final Object baseValue = type.getPropertyValue( component, index );
679 		if ( loc > 0 ) {
680 			if ( baseValue == null ) {
681 				return null;
682 			}
683 			return getComponentValue(
684 					(ComponentType) type.getSubtypes()[index],
685 					baseValue,
686 					propertyPath.substring(loc+1)
687 			);
688 		}
689 		else {
690 			return baseValue;
691 		}
692 
693 	}

695 	private int findSubPropertyIndex(ComponentType type, String subPropertyName) {
696 		final String[] propertyNames = type.getPropertyNames();
697 		for ( int index = 0; index<propertyNames.length; index++ ) {
698 			if ( subPropertyName.equals( propertyNames[index] ) ) {
699 				return index;
700 			}
701 		}
702 		throw new MappingException( "component property not found: " + subPropertyName );
703 	}
