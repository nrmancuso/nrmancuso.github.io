34  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.DEL_REVISION_TYPE_PARAMETER;
35  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.MIDDLE_ENTITY_ALIAS;
36  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REFERENCED_ENTITY_ALIAS;
37  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REFERENCED_ENTITY_ALIAS_DEF_AUD_STR;
38  import static org.hibernate.envers.entities.mapper.relation.query.QueryConstants.REVISION_PARAMETER;

40  /**
41   * Selects data from a relation middle-table and a related versions entity.
42   *
43   * @author Adam Warski (adam at warski dot org)
44   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
45   */
46  public final class TwoEntityQueryGenerator extends AbstractRelationQueryGenerator {
47  	private final String queryString;
48  	private final String queryRemovedString;
49  
50  	public TwoEntityQueryGenerator(GlobalConfiguration globalCfg, AuditEntitiesConfiguration verEntCfg,
51  								   AuditStrategy auditStrategy, String versionsMiddleEntityName,
52  								   MiddleIdData referencingIdData, MiddleIdData referencedIdData,
53  								   boolean revisionTypeInId, MiddleComponentData... componentData) {
54  		super( verEntCfg, referencingIdData, revisionTypeInId );
55  
56  		/*
57  		 * The valid query that we need to create:
58  		 *   SELECT new list(ee, e) FROM versionsReferencedEntity e, middleEntity ee
59  		 *   WHERE
60  		 * (entities referenced by the middle table; id_ref_ed = id of the referenced entity)
61  		 *     ee.id_ref_ed = e.id_ref_ed AND
62  		 * (only entities referenced by the association; id_ref_ing = id of the referencing entity)
63  		 *     ee.id_ref_ing = :id_ref_ing AND
64  		 *
65  		 * (selecting e entities at revision :revision)
66  		 *   --> for DefaultAuditStrategy:
67  		 *     e.revision = (SELECT max(e2.revision) FROM versionsReferencedEntity e2
68  		 *       WHERE e2.revision <= :revision AND e2.id = e.id)
69  		 *
70  		 *   --> for ValidityAuditStrategy:
71  		 *     e.revision <= :revision and (e.endRevision > :revision or e.endRevision is null)
72  		 *
73  		 *     AND
74  		 *
75  		 * (the association at revision :revision)
76  		 *   --> for DefaultAuditStrategy:
77  		 *     ee.revision = (SELECT max(ee2.revision) FROM middleEntity ee2
78  		 *       WHERE ee2.revision <= :revision AND ee2.originalId.* = ee.originalId.*)
79  		 *
80  		 *   --> for ValidityAuditStrategy:
81  		 *     ee.revision <= :revision and (ee.endRevision > :revision or ee.endRevision is null)
82  		 *
83  		 * (only non-deleted entities and associations)
84  		 *     ee.revision_type != DEL AND
85  		 *     e.revision_type != DEL
86  		 */
87  		final QueryBuilder commonPart = commonQueryPart( referencedIdData, versionsMiddleEntityName, verEntCfg.getOriginalIdPropName() );
88  		final QueryBuilder validQuery = commonPart.deepCopy();
89  		final QueryBuilder removedQuery = commonPart.deepCopy();
90  		createValidDataRestrictions(
91  				globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, validQuery,
92  				validQuery.getRootParameters(), true, componentData
93  		);
94  		createValidAndRemovedDataRestrictions(
95  				globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, removedQuery, componentData
96  		);
97  
98  		queryString = queryToString( validQuery );
99  		queryRemovedString = queryToString( removedQuery );
100 	}
101 
102 	/**
103 	 * Compute common part for both queries.
104 	 */
105 	private QueryBuilder commonQueryPart(MiddleIdData referencedIdData, String versionsMiddleEntityName,
106 										 String originalIdPropertyName) {
107 		final String eeOriginalIdPropertyPath = MIDDLE_ENTITY_ALIAS + "." + originalIdPropertyName;
108 		// SELECT new list(ee) FROM middleEntity ee
109 		QueryBuilder qb = new QueryBuilder( versionsMiddleEntityName, MIDDLE_ENTITY_ALIAS );
110 		qb.addFrom( referencedIdData.getAuditEntityName(), REFERENCED_ENTITY_ALIAS );
111 		qb.addProjection( "new list", MIDDLE_ENTITY_ALIAS + ", " + REFERENCED_ENTITY_ALIAS, false, false );
112 		// WHERE
113 		final Parameters rootParameters = qb.getRootParameters();
114 		// ee.id_ref_ed = e.id_ref_ed
115 		referencedIdData.getPrefixedMapper().addIdsEqualToQuery(
116 				rootParameters, eeOriginalIdPropertyPath, referencedIdData.getOriginalMapper(),
117 				REFERENCED_ENTITY_ALIAS + "." + originalIdPropertyName
118 		);
119 		// ee.originalId.id_ref_ing = :id_ref_ing
120 		referencingIdData.getPrefixedMapper().addNamedIdEqualsToQuery( rootParameters, originalIdPropertyName, true );
121 		return qb;
122 	}
123 
124 	/**
125 	 * Creates query restrictions used to retrieve only actual data.
126 	 */
127 	private void createValidDataRestrictions(GlobalConfiguration globalCfg, AuditStrategy auditStrategy, MiddleIdData referencedIdData,
128 											 String versionsMiddleEntityName, QueryBuilder qb, Parameters rootParameters,
129 											 boolean inclusive, MiddleComponentData... componentData) {
130 		final String revisionPropertyPath = verEntCfg.getRevisionNumberPath();
131 		final String originalIdPropertyName = verEntCfg.getOriginalIdPropName();
132 		final String eeOriginalIdPropertyPath = MIDDLE_ENTITY_ALIAS + "." + originalIdPropertyName;
133 		final String revisionTypePropName = getRevisionTypePath();
134 		// (selecting e entities at revision :revision)
135 		// --> based on auditStrategy (see above)
136 		auditStrategy.addEntityAtRevisionRestriction(
137 				globalCfg, qb, rootParameters, REFERENCED_ENTITY_ALIAS + "." + revisionPropertyPath,
138 				REFERENCED_ENTITY_ALIAS + "." + verEntCfg.getRevisionEndFieldName(), false, referencedIdData, revisionPropertyPath,
139 				originalIdPropertyName, REFERENCED_ENTITY_ALIAS, REFERENCED_ENTITY_ALIAS_DEF_AUD_STR, true
140 		);
141 		// (with ee association at revision :revision)
142 		// --> based on auditStrategy (see above)
143 		auditStrategy.addAssociationAtRevisionRestriction( qb, rootParameters, revisionPropertyPath,
144 				verEntCfg.getRevisionEndFieldName(), true, referencingIdData, versionsMiddleEntityName,
145 				eeOriginalIdPropertyPath, revisionPropertyPath, originalIdPropertyName, MIDDLE_ENTITY_ALIAS,
146 				inclusive, componentData
147 		);
148 		// ee.revision_type != DEL
149 		rootParameters.addWhereWithNamedParam( revisionTypePropName, "!=", DEL_REVISION_TYPE_PARAMETER );
150 		// e.revision_type != DEL
151 		rootParameters.addWhereWithNamedParam( REFERENCED_ENTITY_ALIAS + "." + revisionTypePropName, false, "!=", DEL_REVISION_TYPE_PARAMETER );
152 	}
153 
154 	/**
155 	 * Create query restrictions used to retrieve actual data and deletions that took place at exactly given revision.
156 	 */
157 	private void createValidAndRemovedDataRestrictions(GlobalConfiguration globalCfg, AuditStrategy auditStrategy,
158 													   MiddleIdData referencedIdData, String versionsMiddleEntityName,
159 													   QueryBuilder remQb, MiddleComponentData... componentData) {
160 		final Parameters disjoint = remQb.getRootParameters().addSubParameters( "or" );
161 		final Parameters valid = disjoint.addSubParameters( "and" ); // Restrictions to match all valid rows.
162 		final Parameters removed = disjoint.addSubParameters( "and" ); // Restrictions to match all rows deleted at exactly given revision.
163 		final String revisionPropertyPath = verEntCfg.getRevisionNumberPath();
164 		final String revisionTypePropName = getRevisionTypePath();
165 		// Excluding current revision, because we need to match data valid at the previous one.
166 		createValidDataRestrictions( globalCfg, auditStrategy, referencedIdData, versionsMiddleEntityName, remQb, valid, false, componentData );
167 		// ee.revision = :revision
168 		removed.addWhereWithNamedParam( revisionPropertyPath, "=", REVISION_PARAMETER );
169 		// e.revision = :revision
170 		removed.addWhereWithNamedParam( REFERENCED_ENTITY_ALIAS + "." + revisionPropertyPath, false, "=", REVISION_PARAMETER );
171 		// ee.revision_type = DEL
172 		removed.addWhereWithNamedParam( revisionTypePropName, "=", DEL_REVISION_TYPE_PARAMETER );
173 		// e.revision_type = DEL
174 		removed.addWhereWithNamedParam( REFERENCED_ENTITY_ALIAS + "." + revisionTypePropName, false, "=", DEL_REVISION_TYPE_PARAMETER );
175 	}
176 
177 	@Override
178 	protected String getQueryString() {
179 		return queryString;
180 	}
181 
182 	@Override
183 	protected String getQueryRemovedString() {
184 		return queryRemovedString;
185 	}
