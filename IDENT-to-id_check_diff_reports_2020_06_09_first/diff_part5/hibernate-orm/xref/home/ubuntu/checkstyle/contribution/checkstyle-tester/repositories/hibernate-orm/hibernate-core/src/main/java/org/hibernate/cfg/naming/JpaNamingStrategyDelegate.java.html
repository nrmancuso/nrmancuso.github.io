24  package org.hibernate.cfg.naming;
25  
26  import org.hibernate.AssertionFailure;
27  import org.hibernate.internal.util.StringHelper;
28  
29  /**
30   * @author Gail Badner
31   */
32  public class JpaNamingStrategyDelegate extends NamingStrategyDelegateAdapter {
33  
34  	@Override
35  	public String determineImplicitPrimaryTableName(String entityName, String jpaEntityName) {
36  		return StringHelper.unqualify( determineEntityNameToUse( entityName, jpaEntityName ) );
37  	}
38  
39  	@Override
40  	public String determineImplicitElementCollectionTableName(
41  			String ownerEntityName,
42  			String ownerJpaEntityName,
43  			String ownerEntityTable,
44  			String propertyPath) {
45  		// JPA states we should use the following as default:
46  		//      "The concatenation of the name of the containing entity and the name of the
47  		//       collection attribute, separated by an underscore.
48  		// aka:
49  		//     if owning entity has a JPA entity name: {OWNER JPA ENTITY NAME}_{COLLECTION ATTRIBUTE NAME}
50  		//     otherwise: {OWNER ENTITY NAME}_{COLLECTION ATTRIBUTE NAME}
51  		return determineEntityNameToUse( ownerEntityName, ownerJpaEntityName )
52  				+ '_'
53  				+ StringHelper.unqualify( propertyPath );
54  	}
55  
56  	@Override
57  	public String determineImplicitElementCollectionJoinColumnName(
58  			String ownerEntityName, String ownerJpaEntityName, String ownerEntityTable, String referencedColumnName, String propertyPath) {
59  		// JPA states we should use the following as default:
60  		//     "The concatenation of the following: the name of the entity; "_"; the name of the
61  		//      referenced primary key column"
62  		return determineEntityNameToUse( ownerEntityName, ownerJpaEntityName )
63  				+ '_'
64  				+ referencedColumnName;
65  	}
66  
67  	@Override
68  	public String determineImplicitEntityAssociationJoinTableName(
69  			String ownerEntityName,
70  			String ownerJpaEntityName,
71  			String ownerEntityTable,
72  			String associatedEntityName,
73  			String associatedJpaEntityName,
74  			String associatedEntityTable,
75  			String propertyPath) {
76  		// JPA states we should use the following as default:
77  		//		"The concatenated names of the two associated primary entity tables (owning side
78  		//		first), separated by an underscore."
79  		// aka:
80  		// 		{OWNING SIDE PRIMARY TABLE NAME}_{NON-OWNING SIDE PRIMARY TABLE NAME}
81  
82  		return  ownerEntityTable
83  				+ '_'
84  				+ associatedEntityTable;
85  	}
86  
87  	@Override
88  	public String determineImplicitEntityAssociationJoinColumnName(
89  			String propertyEntityName, String propertyJpaEntityName, String propertyTableName, String referencedColumnName, String referencingPropertyName) {
90  		// JPA states we should use the following as default:
91  		//      "The concatenation of the following: the name of the referencing relationship
92  		//      property or field of the referencing entity or embeddable class; "_"; the name
93  		//      of the referenced primary key column. If there is no such referencing relationship
94  		//      property or field in the entity, or if the join is for an element collection, the
95  		//      join column name is formed as the concatenation of the following: the name of the
96  		//      entity; "_"; the name of the referenced primary key column
97  		// The part referring to an entity collection can be disregarded here since, determination of
98  		// an element collection foreign key column name is covered  by #entityAssociationJoinTableName().
99  		//
100 		// For a unidirectional association:
101 		//      {PROPERTY_ENTITY_NAME}_{REFERENCED_COLUMN_NAME}
102 		// For a bidirectional association:
103 		//      {REFERENCING_PROPERTY_NAME}_{REFERENCED_COLUMN_NAME}
104 		final String header;
105 		if ( referencingPropertyName == null ) {
106 			// This is a unidirectional association.
107 			header = determineEntityNameToUse( propertyEntityName, propertyJpaEntityName );
108 		}
109 		else {
110 			// This is a bidirectional association.
111 			header = StringHelper.unqualify( referencingPropertyName );
112 		}
113 		if ( header == null ) {
114 			throw new AssertionFailure( "propertyJpaEntityName and referencingPropertyName cannot both be empty." );
115 		}
116 		return toPhysicalColumnName( header + "_" + referencedColumnName );
117 	}
118 
119 	@Override
120 	public String determineLogicalElementCollectionTableName(
121 			String tableName,
122 			String ownerEntityName,
123 			String ownerJpaEntityName,
124 			String ownerEntityTable,
125 			String propertyName) {
126 		if ( tableName != null ) {
127 			return tableName;
128 		}
129 		else {
130 			return determineImplicitElementCollectionTableName(
131 					ownerEntityName,
132 					ownerJpaEntityName,
133 					ownerEntityTable,
134 					propertyName
135 			);
136 		}
137 	}
138 
139 	@Override
140 	public String determineLogicalEntityAssociationJoinTableName(
141 			String tableName,
142 			String ownerEntityName,
143 			String ownerJpaEntityName,
144 			String ownerEntityTable,
145 			String associatedEntityName,
146 			String associatedJpaEntityName,
147 			String associatedEntityTable,
148 			String propertyName) {
149 		if ( tableName != null ) {
150 			return tableName;
151 		}
152 		else {
153 			return determineImplicitEntityAssociationJoinTableName(
154 					ownerEntityName,
155 					ownerJpaEntityName,
156 					ownerEntityTable,
157 					associatedEntityName,
158 					associatedJpaEntityName,
159 					associatedEntityTable,
160 					propertyName
161 			);
162 		}
163 	}
164 
165 	private String determineEntityNameToUse(String entityName, String jpaEntityName) {
166 		if ( StringHelper.isNotEmpty( jpaEntityName ) ) {
167 			// prefer the JPA entity name, if specified...
168 			return jpaEntityName;
169 		}
170 		else {
171 			// otherwise, use the Hibernate entity name
172 			return StringHelper.unqualifyEntityName( entityName );
173 		}
174 	}
175 }
