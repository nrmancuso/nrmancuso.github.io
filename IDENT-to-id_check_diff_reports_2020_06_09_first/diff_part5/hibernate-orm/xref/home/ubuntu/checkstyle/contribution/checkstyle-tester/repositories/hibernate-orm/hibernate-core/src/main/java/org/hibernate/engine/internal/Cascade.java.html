52  /**
53   * Delegate responsible for, in conjunction with the various
54   * {@link org.hibernate.engine.spi.CascadingAction actions}, implementing cascade processing.
55   *
56   * @author Gavin King
57   * @see org.hibernate.engine.spi.CascadingAction
58   */
59  public final class Cascade {
60  
61  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, Cascade.class.getName());
62  
63  	/**
64  	 * A cascade point that occurs just after the insertion of the parent entity and
65  	 * just before deletion
66  	 */
67  	public static final int AFTER_INSERT_BEFORE_DELETE = 1;
68  	/**
69  	 * A cascade point that occurs just before the insertion of the parent entity and
70  	 * just after deletion
71  	 */
72  	public static final int BEFORE_INSERT_AFTER_DELETE = 2;
73  	/**
74  	 * A cascade point that occurs just after the insertion of the parent entity and
75  	 * just before deletion, inside a collection
76  	 */
77  	public static final int AFTER_INSERT_BEFORE_DELETE_VIA_COLLECTION = 3;
78  	/**
79  	 * A cascade point that occurs just after update of the parent entity
80  	 */
81  	public static final int AFTER_UPDATE = 0;
82  	/**
83  	 * A cascade point that occurs just before the session is flushed
84  	 */
85  	public static final int BEFORE_FLUSH = 0;
86  	/**
87  	 * A cascade point that occurs just after eviction of the parent entity from the
88  	 * session cache
89  	 */
90  	public static final int AFTER_EVICT = 0;
91  	/**
92  	 * A cascade point that occurs just after locking a transient parent entity into the
93  	 * session cache
94  	 */
95  	public static final int BEFORE_REFRESH = 0;
96  	/**
97  	 * A cascade point that occurs just after refreshing a parent entity
98  	 */
99  	public static final int AFTER_LOCK = 0;
100 	/**
101 	 * A cascade point that occurs just before merging from a transient parent entity into
102 	 * the object in the session cache
103 	 */
104 	public static final int BEFORE_MERGE = 0;

106 	private int componentPathStackDepth = 0;
107 	private int cascadeTo;
108 	private EventSource eventSource;
109 	private CascadingAction action;

111 	public Cascade(final CascadingAction action, final int cascadeTo, final EventSource eventSource) {
112 		this.cascadeTo = cascadeTo;
113 		this.eventSource = eventSource;
114 		this.action = action;
115 	}

121 	/**
122 	 * Cascade an action from the parent entity instance to all its children.
123 	 *
124 	 * @param persister The parent's entity persister
125 	 * @param parent The parent reference.
126 	 * @throws HibernateException
127 	 */
128 	public void cascade(final EntityPersister persister, final Object parent)
129 	throws HibernateException {
130 		cascade( persister, parent, null );
131 	}
132 
133 	/**
134 	 * Cascade an action from the parent entity instance to all its children.  This
135 	 * form is typically called from within cascade actions.
136 	 *
137 	 * @param persister The parent's entity persister
138 	 * @param parent The parent reference.
139 	 * @param anything Anything ;)   Typically some form of cascade-local cache
140 	 * which is specific to each CascadingAction type
141 	 * @throws HibernateException
142 	 */
143 	public void cascade(final EntityPersister persister, final Object parent, final Object anything)
144 			throws HibernateException {
145 
146 		if ( persister.hasCascades() || action.requiresNoCascadeChecking() ) { // performance opt
147 			final boolean traceEnabled = LOG.isTraceEnabled();
148 			if ( traceEnabled ) {
149 				LOG.tracev( "Processing cascade {0} for: {1}", action, persister.getEntityName() );
150 			}
151 
152 			Type[] types = persister.getPropertyTypes();
153 			CascadeStyle[] cascadeStyles = persister.getPropertyCascadeStyles();
154 			boolean hasUninitializedLazyProperties = persister.hasUninitializedLazyProperties( parent );
155 			for ( int i=0; i<types.length; i++) {
156 				final CascadeStyle style = cascadeStyles[i];
157 				final String propertyName = persister.getPropertyNames()[i];
158 				if ( hasUninitializedLazyProperties && persister.getPropertyLaziness()[i] && ! action.performOnLazyProperty() ) {
159 					//do nothing to avoid a lazy property initialization
160 					continue;
161 				}
162 
163 				if ( style.doCascade( action ) ) {
164 					cascadeProperty(
165 						    parent,
166 					        persister.getPropertyValue( parent, i ),
167 					        types[i],
168 					        style,
169 							propertyName,
170 					        anything,
171 					        false
172 					);
173 				}
174 				else if ( action.requiresNoCascadeChecking() ) {
175 					action.noCascade(
176 							eventSource,
177 							persister.getPropertyValue( parent, i ),
178 							parent,
179 							persister,
180 							i
181 					);
182 				}
183 			}
184 
185 			if ( traceEnabled ) {
186 				LOG.tracev( "Done processing cascade {0} for: {1}", action, persister.getEntityName() );
187 			}
188 		}
189 	}

191 	/**
192 	 * Cascade an action to the child or children
193 	 */
194 	private void cascadeProperty(
195 			final Object parent,
196 			final Object child,
197 			final Type type,
198 			final CascadeStyle style,
199 			final String propertyName,
200 			final Object anything,
201 			final boolean isCascadeDeleteEnabled) throws HibernateException {
202 		
203 		if ( child != null ) {
204 			if ( type.isAssociationType() ) {
205 				AssociationType associationType = (AssociationType) type;
206 				if ( cascadeAssociationNow( associationType ) ) {
207 					cascadeAssociation(
208 							parent,
209 							child,
210 							type,
211 							style,
212 							anything,
213 							isCascadeDeleteEnabled
214 						);
215 				}
216 			}
217 			else if ( type.isComponentType() ) {
218 				cascadeComponent( parent, child, (CompositeType) type, propertyName, anything );
219 			}
220 		}
221 		
222 		// potentially we need to handle orphan deletes for one-to-ones here...
223 		if ( isLogicalOneToOne( type ) ) {
224 			// We have a physical or logical one-to-one.  See if the attribute cascade settings and action-type require
225 			// orphan checking
226 			if ( style.hasOrphanDelete() && action.deleteOrphans() ) {
227 				// value is orphaned if loaded state for this property shows not null
228 				// because it is currently null.
229 				final EntityEntry entry = eventSource.getPersistenceContext().getEntry( parent );
230 				if ( entry != null && entry.getStatus() != Status.SAVING ) {
231 					final Object loadedValue;
232 					if ( componentPathStackDepth == 0 ) {
233 						// association defined on entity
234 						loadedValue = entry.getLoadedValue( propertyName );
235 					}
236 					else {
237 						// association defined on component
238 						// 		todo : this is currently unsupported because of the fact that
239 						//		we do not know the loaded state of this value properly
240 						//		and doing so would be very difficult given how components and
241 						//		entities are loaded (and how 'loaded state' is put into the
242 						//		EntityEntry).  Solutions here are to either:
243 						//			1) properly account for components as a 2-phase load construct
244 						//			2) just assume the association was just now orphaned and
245 						// 				issue the orphan delete.  This would require a special
246 						//				set of SQL statements though since we do not know the
247 						//				orphaned value, something a delete with a subquery to
248 						// 				match the owner.
249 //							final EntityType entityType = (EntityType) type;
250 //							final String getPropertyPath = composePropertyPath( entityType.getPropertyName() );
251 						loadedValue = null;
252 					}
253 					
254 					// orphaned if the association was nulled (child == null) or receives a new value while the
255 					// entity is managed (without first nulling and manually flushing).
256 					if ( child == null || ( loadedValue != null && child != loadedValue ) ) {
257 						final EntityEntry valueEntry = eventSource
258 								.getPersistenceContext().getEntry( 
259 										loadedValue );
260 						// Need to check this in case the context has
261 						// already been flushed.  See HHH-7829.
262 						if ( valueEntry != null ) {
263 							final String entityName = valueEntry.getPersister().getEntityName();
264 							if ( LOG.isTraceEnabled() ) {
265 								final Serializable id = valueEntry.getPersister().getIdentifier( loadedValue, eventSource );
266 								final String description = MessageHelper.infoString( entityName, id );
267 								LOG.tracev( "Deleting orphaned entity instance: {0}", description );
268 							}
269 							
270 							if (type.isAssociationType() && ((AssociationType)type).getForeignKeyDirection().equals(
271 											ForeignKeyDirection.FOREIGN_KEY_TO_PARENT )) {
272 								// If FK direction is to-parent, we must remove the orphan *before* the queued update(s)
273 								// occur.  Otherwise, replacing the association on a managed entity, without manually
274 								// nulling and flushing, causes FK constraint violations.
275 								eventSource.removeOrphanBeforeUpdates( entityName, loadedValue );
276 							}
277 							else {
278 								// Else, we must delete after the updates.
279 								eventSource.delete( entityName, loadedValue, isCascadeDeleteEnabled, new HashSet() );
280 							}
281 						}
282 					}
283 				}
284 			}
285 		}
286 	}

288 	/**
289 	 * Check if the association is a one to one in the logical model (either a shared-pk
290 	 * or unique fk).
291 	 *
292 	 * @param type The type representing the attribute metadata
293 	 *
294 	 * @return True if the attribute represents a logical one to one association
295 	 */
296 	private boolean isLogicalOneToOne(Type type) {
297 		return type.isEntityType() && ( (EntityType) type ).isLogicalOneToOne();
298 	}
299 
300 	private boolean cascadeAssociationNow(AssociationType associationType) {
301 		return associationType.getForeignKeyDirection().cascadeNow(cascadeTo);
302 	}
303 
304 	private void cascadeComponent(
305 			final Object parent,
306 			final Object child,
307 			final CompositeType componentType,
308 			final String componentPropertyName,
309 			final Object anything) {
310 		componentPathStackDepth++;
311 		Object[] children = componentType.getPropertyValues( child, eventSource );
312 		Type[] types = componentType.getSubtypes();
313 		for ( int i=0; i<types.length; i++ ) {
314 			final CascadeStyle componentPropertyStyle = componentType.getCascadeStyle(i);
315 			final String subPropertyName = componentType.getPropertyNames()[i];
316 			if ( componentPropertyStyle.doCascade(action) ) {
317 				cascadeProperty(
318 						parent,
319 						children[i],
320 						types[i],
321 						componentPropertyStyle,
322 						subPropertyName,
323 						anything,
324 						false
325 					);
326 			}
327 		}
328 		componentPathStackDepth--;
329 	}
330 
331 	private void cascadeAssociation(
332 			final Object parent,
333 			final Object child,
334 			final Type type,
335 			final CascadeStyle style,
336 			final Object anything,
337 			final boolean isCascadeDeleteEnabled) {
338 		if ( type.isEntityType() || type.isAnyType() ) {
339 			cascadeToOne( parent, child, type, style, anything, isCascadeDeleteEnabled );
340 		}
341 		else if ( type.isCollectionType() ) {
342 			cascadeCollection( parent, child, style, anything, (CollectionType) type );
343 		}
344 	}

346 	/**
347 	 * Cascade an action to a collection
348 	 */
349 	private void cascadeCollection(
350 			final Object parent,
351 			final Object child,
352 			final CascadeStyle style,
353 			final Object anything,
354 			final CollectionType type) {
355 		CollectionPersister persister = eventSource.getFactory()
356 				.getCollectionPersister( type.getRole() );
357 		Type elemType = persister.getElementType();
358 
359 		final int oldCascadeTo = cascadeTo;
360 		if ( cascadeTo==AFTER_INSERT_BEFORE_DELETE) {
361 			cascadeTo = AFTER_INSERT_BEFORE_DELETE_VIA_COLLECTION;
362 		}
363 
364 		//cascade to current collection elements
365 		if ( elemType.isEntityType() || elemType.isAnyType() || elemType.isComponentType() ) {
366 			cascadeCollectionElements(
367 				parent,
368 				child,
369 				type,
370 				style,
371 				elemType,
372 				anything,
373 				persister.isCascadeDeleteEnabled()
374 			);
375 		}
376 
377 		cascadeTo = oldCascadeTo;
378 	}

380 	/**
381 	 * Cascade an action to a to-one association or any type
382 	 */
383 	private void cascadeToOne(
384 			final Object parent,
385 			final Object child,
386 			final Type type,
387 			final CascadeStyle style,
388 			final Object anything,
389 			final boolean isCascadeDeleteEnabled) {
390 		final String entityName = type.isEntityType()
391 				? ( (EntityType) type ).getAssociatedEntityName()
392 				: null;
393 		if ( style.reallyDoCascade(action) ) { //not really necessary, but good for consistency...
394 			eventSource.getPersistenceContext().addChildParent(child, parent);
395 			try {
396 				action.cascade(eventSource, child, entityName, anything, isCascadeDeleteEnabled);
397 			}
398 			finally {
399 				eventSource.getPersistenceContext().removeChildParent(child);
400 			}
401 		}
402 	}

404 	/**
405 	 * Cascade to the collection elements
406 	 */
407 	private void cascadeCollectionElements(
408 			final Object parent,
409 			final Object child,
410 			final CollectionType collectionType,
411 			final CascadeStyle style,
412 			final Type elemType,
413 			final Object anything,
414 			final boolean isCascadeDeleteEnabled) throws HibernateException {
415 
416 		boolean reallyDoCascade = style.reallyDoCascade(action) && child!=CollectionType.UNFETCHED_COLLECTION;
417 
418 		if ( reallyDoCascade ) {
419             final boolean traceEnabled = LOG.isTraceEnabled();
420 			if ( traceEnabled ) {
421 				LOG.tracev( "Cascade {0} for collection: {1}", action, collectionType.getRole() );
422 			}
423 
424 			Iterator iter = action.getCascadableChildrenIterator(eventSource, collectionType, child);
425 			while ( iter.hasNext() ) {
426 				cascadeProperty(
427 						parent,
428 						iter.next(),
429 						elemType,
430 						style,
431 						null,
432 						anything,
433 						isCascadeDeleteEnabled
434 					);
435 			}
436 
437 			if ( traceEnabled ) {
438 				LOG.tracev( "Done cascade {0} for collection: {1}", action, collectionType.getRole() );
439 			}
440 		}
441 
442 		final boolean deleteOrphans = style.hasOrphanDelete() &&
443 				action.deleteOrphans() &&
444 				elemType.isEntityType() &&
445 				child instanceof PersistentCollection; //a newly instantiated collection can't have orphans
446 
447 		if ( deleteOrphans ) { // handle orphaned entities!!
448 			final boolean traceEnabled = LOG.isTraceEnabled();
449 			if ( traceEnabled ) {
450 				LOG.tracev( "Deleting orphans for collection: {0}", collectionType.getRole() );
451 			}
452 			// we can do the cast since orphan-delete does not apply to:
453 			// 1. newly instantiated collections
454 			// 2. arrays (we can't track orphans for detached arrays)
455 			final String entityName = collectionType.getAssociatedEntityName( eventSource.getFactory() );
456 			deleteOrphans( entityName, (PersistentCollection) child );
457 
458 			if ( traceEnabled ) {
459 				LOG.tracev( "Done deleting orphans for collection: {0}", collectionType.getRole() );
460 			}
461 		}
462 	}

464 	/**
465 	 * Delete any entities that were removed from the collection
466 	 */
467 	private void deleteOrphans(String entityName, PersistentCollection pc) throws HibernateException {
468 		//TODO: suck this logic into the collection!
469 		final Collection orphans;
470 		if ( pc.wasInitialized() ) {
471 			CollectionEntry ce = eventSource.getPersistenceContext().getCollectionEntry(pc);
472 			orphans = ce==null ?
473 					java.util.Collections.EMPTY_LIST :
474 					ce.getOrphans(entityName, pc);
475 		}
476 		else {
477 			orphans = pc.getQueuedOrphans(entityName);
478 		}
479 
480 		final Iterator orphanIter = orphans.iterator();
481 		while ( orphanIter.hasNext() ) {
482 			Object orphan = orphanIter.next();
483 			if (orphan!=null) {
484 				LOG.tracev( "Deleting orphaned entity instance: {0}", entityName );
485 				eventSource.delete( entityName, orphan, false, new HashSet() );
486 			}
487 		}
488 	}
