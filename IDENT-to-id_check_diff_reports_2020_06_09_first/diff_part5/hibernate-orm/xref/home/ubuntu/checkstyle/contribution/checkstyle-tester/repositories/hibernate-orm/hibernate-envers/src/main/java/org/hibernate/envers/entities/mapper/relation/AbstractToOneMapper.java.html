1   package org.hibernate.envers.entities.mapper.relation;
2   
3   import java.io.Serializable;
4   import java.util.List;
5   import java.util.Map;
6   
7   import org.hibernate.collection.spi.PersistentCollection;
8   import org.hibernate.engine.spi.SessionImplementor;
9   import org.hibernate.envers.configuration.AuditConfiguration;
10  import org.hibernate.envers.entities.EntityConfiguration;
11  import org.hibernate.envers.entities.PropertyData;
12  import org.hibernate.envers.entities.mapper.PersistentCollectionChangeData;
13  import org.hibernate.envers.entities.mapper.PropertyMapper;
14  import org.hibernate.envers.exception.AuditException;
15  import org.hibernate.envers.reader.AuditReaderImplementor;
16  import org.hibernate.envers.tools.reflection.ReflectionTools;
17  import org.hibernate.internal.util.ReflectHelper;
18  import org.hibernate.property.Setter;
19  
20  /**
21   * Base class for property mappers that manage to-one relation.
22   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
23   */
24  public abstract class AbstractToOneMapper implements PropertyMapper {
25      private final PropertyData propertyData;
26  
27      protected AbstractToOneMapper(PropertyData propertyData) {
28          this.propertyData = propertyData;
29      }
30  
31      @Override
32      public boolean mapToMapFromEntity(SessionImplementor session, Map<String, Object> data, Object newObj, Object oldObj) {
33          return false;
34      }
35  
36      @Override
37      public void mapToEntityFromMap(AuditConfiguration verCfg, Object obj, Map data, Object primaryKey,
38                                     AuditReaderImplementor versionsReader, Number revision) {
39          if (obj != null) {
40              nullSafeMapToEntityFromMap(verCfg, obj, data, primaryKey, versionsReader, revision);
41          }
42      }
43  
44      @Override
45      public List<PersistentCollectionChangeData> mapCollectionChanges(SessionImplementor session, String referencingPropertyName,
46                                                                       PersistentCollection newColl, Serializable oldColl,
47  																	 Serializable id) {
48          return null;
49      }
50  
51      /**
52       * @param verCfg Audit configuration.
53       * @param entityName Entity name.
54       * @return Entity class, name and information whether it is audited or not.
55       */
56      protected EntityInfo getEntityInfo(AuditConfiguration verCfg, String entityName) {
57          EntityConfiguration entCfg = verCfg.getEntCfg().get(entityName);
58          boolean isRelationAudited = true;
59          if (entCfg == null) {
60              // a relation marked as RelationTargetAuditMode.NOT_AUDITED
61              entCfg = verCfg.getEntCfg().getNotVersionEntityConfiguration(entityName);
62              isRelationAudited = false;
63          }
64          Class entityClass;
65          try {
66  			entityClass = ReflectHelper.classForName(entCfg.getEntityClassName());
67  		}
68  		catch ( ClassNotFoundException e ) {
69  			throw new AuditException( e );
70  		}
71          return new EntityInfo(entityClass, entityName, isRelationAudited);
72      }
73  
74      protected void setPropertyValue(Object targetObject, Object value) {
75          Setter setter = ReflectionTools.getSetter(targetObject.getClass(), propertyData);
76          setter.set(targetObject, value, null);
77      }
78  
79      /**
80       * @return Bean property that represents the relation.
81       */
82      protected PropertyData getPropertyData() {
83          return propertyData;
84      }
85  
86      /**
87       * Parameter {@code obj} is never {@code null}.
88       * @see PropertyMapper#mapToEntityFromMap(AuditConfiguration, Object, Map, Object, AuditReaderImplementor, Number)
89       */
90      public abstract void nullSafeMapToEntityFromMap(AuditConfiguration verCfg, Object obj, Map data, Object primaryKey,
91                                                      AuditReaderImplementor versionsReader, Number revision);
92  
93      /**
94       * Simple descriptor of an entity.
95       */
96      protected class EntityInfo {
97          private final Class entityClass;
98          private final String entityName;
99          private final boolean audited;
100 
101         public EntityInfo(Class entityClass, String entityName, boolean audited) {
102             this.entityClass = entityClass;
103             this.entityName = entityName;
104             this.audited = audited;
105         }
106 
107         public Class getEntityClass() { return entityClass; }
108         public String getEntityName() { return entityName; }
109         public boolean isAudited() { return audited; }
110     }
111 }
