161 	private void validateBind() {
162 		if ( property.isAnnotationPresent( Immutable.class ) ) {
163 			throw new AnnotationException(
164 					"@Immutable on property not allowed. " +
165 							"Only allowed on entity level or on a collection."
166 			);
167 		}
168 		if ( !declaringClassSet ) {
169 			throw new AssertionFailure( "declaringClass has not been set before a bind" );
170 		}
171 	}

177 	private Property makePropertyAndValue() {
178 		validateBind();
179 		LOG.debugf( "MetadataSourceProcessor property %s with lazy=%s", name, lazy );
180 		String containerClassName = holder == null ?
181 				null :
182 				holder.getClassName();
183 		simpleValueBinder = new SimpleValueBinder();
184 		simpleValueBinder.setMappings( mappings );
185 		simpleValueBinder.setPropertyName( name );
186 		simpleValueBinder.setReturnedClassName( returnedClassName );
187 		simpleValueBinder.setColumns( columns );
188 		simpleValueBinder.setPersistentClassName( containerClassName );
189 		simpleValueBinder.setType( property, returnedClass, containerClassName );
190 		simpleValueBinder.setMappings( mappings );
191 		simpleValueBinder.setReferencedEntityName( referencedEntityName );
192 		simpleValueBinder.setAccessType( accessType );
193 		SimpleValue propertyValue = simpleValueBinder.make();
194 		setValue( propertyValue );
195 		return makeProperty();
196 	}

212 	private Property bind(Property prop) {
213 		if (isId) {
214 			final RootClass rootClass = ( RootClass ) holder.getPersistentClass();
215 			//if an xToMany, it as to be wrapped today.
216 			//FIXME this pose a problem as the PK is the class instead of the associated class which is not really compliant with the spec
217 			if ( isXToMany || entityBinder.wrapIdsInEmbeddedComponents() ) {
218 				Component identifier = (Component) rootClass.getIdentifier();
219 				if (identifier == null) {
220 					identifier = AnnotationBinder.createComponent( holder, new PropertyPreloadedData(null, null, null), true, false, mappings );
221 					rootClass.setIdentifier( identifier );
222 					identifier.setNullValue( "undefined" );
223 					rootClass.setEmbeddedIdentifier( true );
224 					rootClass.setIdentifierMapper( identifier );
225 				}
226 				//FIXME is it good enough?
227 				identifier.addProperty( prop );
228 			}
229 			else {
230 				rootClass.setIdentifier( ( KeyValue ) getValue() );
231 				if (embedded) {
232 					rootClass.setEmbeddedIdentifier( true );
233 				}
234 				else {
235 					rootClass.setIdentifierProperty( prop );
236 					final org.hibernate.mapping.MappedSuperclass superclass = BinderHelper.getMappedSuperclassOrNull(
237 							declaringClass,
238 							inheritanceStatePerClass,
239 							mappings
240 					);
241 					if (superclass != null) {
242 						superclass.setDeclaredIdentifierProperty(prop);
243 					}
244 					else {
245 						//we know the property is on the actual entity
246 						rootClass.setDeclaredIdentifierProperty( prop );
247 					}
248 				}
249 			}
250 		}
251 		else {
252 			holder.addProperty( prop, columns, declaringClass );
253 		}
254 		return prop;
255 	}

257 	//used when the value is provided and the binding is done elsewhere
258 	public Property makeProperty() {
259 		validateMake();
260 		LOG.debugf( "Building property %s", name );
261 		Property prop = new Property();
262 		prop.setName( name );
263 		prop.setNodeName( name );
264 		prop.setValue( value );
265 		prop.setLazy( lazy );
266 		prop.setCascade( cascade );
267 		prop.setPropertyAccessorName( accessType.getType() );
268 		
269 		Generated ann = property != null ?
270 				property.getAnnotation( Generated.class ) :
271 				null;
272 		GenerationTime generated = ann != null ?
273 				ann.value() :
274 				null;
275 		if ( generated != null ) {
276 			if ( !GenerationTime.NEVER.equals( generated ) ) {
277 				if ( property.isAnnotationPresent( javax.persistence.Version.class )
278 						&& GenerationTime.INSERT.equals( generated ) ) {
279 					throw new AnnotationException(
280 							"@Generated(INSERT) on a @Version property not allowed, use ALWAYS: "
281 									+ StringHelper.qualify( holder.getPath(), name )
282 					);
283 				}
284 				insertable = false;
285 				if ( GenerationTime.ALWAYS.equals( generated ) ) {
286 					updatable = false;
287 				}
288 				prop.setGeneration( PropertyGeneration.parse( generated.toString().toLowerCase() ) );
289 			}
290 		}
291 		
292 		NaturalId naturalId = property != null ? property.getAnnotation( NaturalId.class ) : null;
293 		if ( naturalId != null ) {
294 			if ( ! entityBinder.isRootEntity() ) {
295 				throw new AnnotationException( "@NaturalId only valid on root entity (or its @MappedSuperclasses)" );
296 			}
297 			if ( ! naturalId.mutable() ) {
298 				updatable = false;
299 			}
300 			prop.setNaturalIdentifier( true );
301 		}
302 		
303 		// HHH-4635 -- needed for dialect-specific property ordering
304 		Lob lob = property != null ? property.getAnnotation( Lob.class ) : null;
305 		prop.setLob( lob != null );
306 		
307 		prop.setInsertable( insertable );
308 		prop.setUpdateable( updatable );
309 
310 		// this is already handled for collections in CollectionBinder...
311 		if ( Collection.class.isInstance( value ) ) {
312 			prop.setOptimisticLocked( ( (Collection) value ).isOptimisticLocked() );
313 		}
314 		else {
315 			final OptimisticLock lockAnn = property != null
316 					? property.getAnnotation( OptimisticLock.class )
317 					: null;
318 			if ( lockAnn != null ) {
319 				//TODO this should go to the core as a mapping validation checking
320 				if ( lockAnn.excluded() && (
321 						property.isAnnotationPresent( javax.persistence.Version.class )
322 								|| property.isAnnotationPresent( Id.class )
323 								|| property.isAnnotationPresent( EmbeddedId.class ) ) ) {
324 					throw new AnnotationException(
325 							"@OptimisticLock.exclude=true incompatible with @Id, @EmbeddedId and @Version: "
326 									+ StringHelper.qualify( holder.getPath(), name )
327 					);
328 				}
329 			}
330 			final boolean isOwnedValue = !isToOneValue( value ) || insertable; // && updatable as well???
331 			final boolean includeInOptimisticLockChecks = ( lockAnn != null )
332 					? ! lockAnn.excluded()
333 					: isOwnedValue;
334 			prop.setOptimisticLocked( includeInOptimisticLockChecks );
335 		}
336 
337 		LOG.tracev( "Cascading {0} with {1}", name, cascade );
338 		this.mappingProperty = prop;
339 		return prop;
340 	}
