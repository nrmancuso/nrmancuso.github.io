59          String authority, path;

62          Client (String authority, String path, boolean allowerror) {
63              super("Thread-" + path);
64              this.authority = authority;
65              this.path = path;
66              this.allowerror = allowerror;
67          }
68  
69          public void run () {
70              try {
71                  URI u = new URI ("http", authority, path, null, null);
72                  URL url = u.toURL();
73                  URLConnection urlc = url.openConnection();
74                  InputStream is = urlc.getInputStream();
75                  read (is);
76                  is.close();
77              } catch (URISyntaxException  e) {
78                  System.out.println (e);
79                  error = true;
80              } catch (IOException e) {
81                  if (!allowerror) {
82                      System.out.println (Thread.currentThread().getName()
83                              + " " + e);
84                      e.printStackTrace();
85                      error = true;
86                  }
87              }
88          }
89      }

359     static class AuthenticationHandler {
360         static void errorReply(HttpExchange exchange, String reply)
361                 throws IOException
362         {
363             exchange.getResponseHeaders().add("Connection", "close");
364             exchange.getResponseHeaders().add("WWW-Authenticate", reply);
365             exchange.sendResponseHeaders(401, 0);
366             exchange.close();
367         }
368 
369         static void proxyReply (HttpExchange exchange, String reply)
370                 throws IOException
371         {
372             exchange.getResponseHeaders().add("Proxy-Authenticate", reply);
373             exchange.sendResponseHeaders(407, 0);
374         }
375 
376         static void okReply (HttpExchange exchange) throws IOException {
377             exchange.getResponseHeaders().add("Connection", "close");
378             String response = "Hello .";
379             exchange.sendResponseHeaders(200, response.getBytes().length);
380             OutputStream os = exchange.getResponseBody();
381             os.write(response.getBytes());
382             os.close();
383             exchange.close();
384         }
385     }

399     static void doServerTests (String authority, Server server) throws Exception
400     {
401         System.out.println ("Doing Server tests");
402         System.out.println ("T1");
403         c1 = new Client (authority, "/test/realm1/t1a", false);
404         c2 = new Client (authority, "/test/realm2/t1b", false);
405         c3 = new Client (authority, "/test/realm1/t1c", false);
406         c4 = new Client (authority, "/test/realm2/t1d", false);
407         t1cond1latch = new CountDownLatch(1);
408         t1cond2latch = new CountDownLatch(1);
409         c1.start(); c2.start();
410         t1cond1latch.await();
411         c3.start(); c4.start();
412         c1.join(); c2.join(); c3.join(); c4.join();
413 
414         int f = auth.getCount();
415         if (f != 2) {
416             except ("Authenticator was called "+f+" times. Should be 2",
417                     server);
418         }
419         if (error) {
420             except ("error occurred", server);
421         }
422 
423         auth.resetCount();
424         System.out.println ("T2");
425 
426         c5 = new Client (authority, "/test/realm3/t2a", true);
427         c6 = new Client (authority, "/test/realm3/t2b", false);
428         t2condlatch = new CountDownLatch(1);
429         c5.start ();
430         t2condlatch.await();
431         c6.start ();
432         c5.join(); c6.join();
433 
434         f = auth.getCount();
435         if (f != redirects+1) {
436             except ("Authenticator was called "+f+" times. Should be: "
437                     + redirects+1, server);
438         }
439         if (error) {
440             except ("error occurred", server);
441         }
442     }

471     public void runTest(boolean proxy) throws Exception {
472         System.setProperty ("http.maxRedirects", Integer.toString (redirects));
473         System.setProperty ("http.auth.serializeRequests", "true");
474         Authenticator.setDefault (auth);
475         try (Server server = new Server()) {
476             server.startServer();
477             System.out.println ("Server: listening on port: "
478                     + server.getPort());
479             if (proxy) {
480                 System.setProperty ("http.proxyHost", "localhost");
481                 System.setProperty ("http.proxyPort",
482                         Integer.toString(server.getPort()));
483                 doProxyTests ("www.foo.com", server);
484             } else {
485                 doServerTests ("localhost:"+server.getPort(), server);
486             }
487         }
488
