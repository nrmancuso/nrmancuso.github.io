711     // get the ssl context
712     private static SSLContext getSSLContext(String trusedCertStr,
713             String keyCertStr, byte[] modulus,
714             byte[] privateExponent, char[] passphrase) throws Exception {
715 
716         // generate certificate from cert string
717         CertificateFactory cf = CertificateFactory.getInstance("X.509");
718 
719         ByteArrayInputStream is =
720                     new ByteArrayInputStream(trusedCertStr.getBytes());
721         Certificate trusedCert = cf.generateCertificate(is);
722         is.close();
723 
724         // create a key store
725         KeyStore ks = KeyStore.getInstance("JKS");
726         ks.load(null, null);
727 
728         // import the trused cert
729         ks.setCertificateEntry("RSA Export Signer", trusedCert);
730 
731         if (keyCertStr != null) {
732             // generate the private key.
733             RSAPrivateKeySpec priKeySpec = new RSAPrivateKeySpec(
734                                             new BigInteger(modulus),
735                                             new BigInteger(privateExponent));
736             KeyFactory kf = KeyFactory.getInstance("RSA");
737             RSAPrivateKey priKey =
738                     (RSAPrivateKey)kf.generatePrivate(priKeySpec);
739 
740             // generate certificate chain
741             is = new ByteArrayInputStream(keyCertStr.getBytes());
742             Certificate keyCert = cf.generateCertificate(is);
743             is.close();
744 
745             Certificate[] chain = new Certificate[2];
746             chain[0] = keyCert;
747             chain[1] = trusedCert;
748 
749             // import the key entry.
750             ks.setKeyEntry("Whatever", priKey, passphrase, chain);
751         }
752 
753         // create SSL context
754         TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
755         tmf.init(ks);
756 
757         TrustManager tms[] = tmf.getTrustManagers();
758         if (tms == null || tms.length == 0) {
759             throw new Exception("unexpected trust manager implementation");
760         } else {
761            if (!(tms[0] instanceof X509ExtendedTrustManager)) {
762                throw new Exception("unexpected trust manager implementation: "
763                                 + tms[0].getClass().getCanonicalName());
764            }
765         }
766 
767 
768         SSLContext ctx = SSLContext.getInstance("TLS");
769 
770         if (keyCertStr != null) {
771             KeyManagerFactory kmf = KeyManagerFactory.getInstance("SunX509");
772             kmf.init(ks, passphrase);
773 
774             ctx.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
775         } else {
776             ctx.init(null, tmf.getTrustManagers(), null);
777         }
778 
779         return ctx;
780     }

806      * Primary constructor, used to drive remainder of the test.
807      *
808      * Fork off the other side, then do your work.
809      */
810     PKIXExtendedTM() throws Exception {
811         if (separateServerThread) {
812             startServer(true);
813             startClient(false);
814         } else {
815             startClient(true);
816             startServer(false);
817         }
818 
819         /*
820          * Wait for other side to close down.
821          */
822         if (separateServerThread) {
823             serverThread.join();
824         } else {
825             clientThread.join();
826         }
827 
828         /*
829          * When we get here, the test is pretty much over.
830          *
831          * If the main thread excepted, that propagates back
832          * immediately.  If the other thread threw an exception, we
833          * should report back.
834          */
835         if (serverException != null)
836             throw serverException;
837         if (clientException != null)
838             throw clientException;
839     }
