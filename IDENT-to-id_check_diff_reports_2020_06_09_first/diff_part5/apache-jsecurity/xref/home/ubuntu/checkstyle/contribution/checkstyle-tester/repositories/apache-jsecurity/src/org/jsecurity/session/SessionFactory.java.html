19  package org.jsecurity.session;
20  
21  import org.jsecurity.authz.AuthorizationException;
22  import org.jsecurity.authz.HostUnauthorizedException;
23  
24  import java.io.Serializable;
25  import java.net.InetAddress;
26  
27  /**
28   * A <tt>SessionFactory</tt> is responsible for starting new {@link Session Session}s and
29   * acquiring existing {@link Session Session}s.
30   *
31   * @author Les Hazlewood
32   * @since 0.1
33   */
34  public interface SessionFactory {
35  
36      /**
37       * Starts a new session within the system for the host with the specified
38       * originating IP address.
39       *
40       * <p>An implementation of this interface may be configured to allow a <tt>null</tt> argument,
41       * thereby indicating the originating IP is either unknown or has been
42       * explicitly omitted by the caller.  However, if the implementation is configured to require
43       * a valid <tt>hostAddress</tt> and the argument is <tt>null</tt>, an
44       * {@link IllegalArgumentException IllegalArgumentException} will be thrown.
45       *
46       * <p>In web-based systems, this InetAddress can be inferred from the
47       * {@link javax.servlet.ServletRequest#getRemoteAddr() javax.servlet.ServletRequest.getRemoteAddr()}
48       * method, or in socket-based systems, it can be obtained via inspecting the socket
49       * initiator's host IP.
50       *
51       * <p>Most secure environments <em>should</em> require that a valid, non-<tt>null</tt>
52       * <tt>hostAddress</tt> be specified, since knowing the <tt>hostAddress</tt> allows for more
53       * flexibility when securing a system: by requiring an InetAddress, access control policies
54       * can also ensure access is restricted to specific client <em>locations</em> in
55       * addition to user principals, if so desired.
56       *
57       * <p><b>Caveat</b> - if clients to your system are on a
58       * public network (as would be the case for a public web site), odds are high the clients can be
59       * behind a NAT (Network Address Translation) router or HTTP proxy server.  If so, all clients
60       * accessing your system behind that router or proxy will have the same originating IP address.
61       * If your system is configured to allow only one session per IP, then the next request from a
62       * different NAT or proxy client will fail and access will be deny for that client.  Just be
63       * aware that ip-based security policies are best utilized in LAN or private WAN environments
64       * when you can be ensure clients will not share IPs or be behind such NAT routers or
65       * proxy servers.
66       *
67       * @param hostAddress the originating host InetAddress of the external party
68       *                    (user, 3rd party product, etc) that is attempting to interact with the system.
69       * @return a handle to the newly created session.
70       * @throws HostUnauthorizedException if the system access control policy restricts access based
71       *                                   on client location/IP and the specified hostAddress hasn't been enabled.
72       * @throws IllegalArgumentException  if the system is configured to require a valid,
73       *                                   non-<tt>null</tt> argument and the specified <tt>hostAddress</tt> is null.
74       */
75      Session start(InetAddress hostAddress) throws HostUnauthorizedException, IllegalArgumentException;
76  
77      /**
78       * Acquires a handle to the session identified by the specified <tt>sessionId</tt>.
79       *
80       * <p><b>Although simple, this method finally enables behavior absent in Java for years:</b>
81       *
82       * <p>the
83       * ability to participate in a server-side session across clients of different mediums,
84       * such as web appliations, Java applets, standalone C# clients over XMLRPC and/or SOAP, and
85       * many others.  This is a <em>huge</em> benefit in heterogeneous enterprise applications.
86       *
87       * <p>To maintain session integrity across client mediums, the sessionId must be transmitted
88       * to all client mediums securely (e.g. over SSL) to prevent man-in-the-middle attacks.  This
89       * is nothing new - all web applications are susceptible to the same problem when transmitting
90       * {@link javax.servlet.http.Cookie Cookie}s or when using URL rewriting.  As long as the
91       * <tt>sessionId</tt> is transmitted securely, session integrity can be maintained.
92       *
93       * @param sessionId the id of the session to acquire.
94       * @return a handle to the session identified by <tt>sessionId</tt>
95       * @throws InvalidSessionException if the session identified by <tt>sessionId</tt> has
96       *                                 been stopped, expired, or doesn't exist.
97       * @throws AuthorizationException  if the executor of this method is not allowed to acquire
98       *                                 (i.e. join) the session identified by <tt>sessionId</tt>.  The reason for the exception
99       *                                 is implementation specific and could be for any number of reasons.  A common reason in many
100      *                                 systems would be if one host tried to acquire/join a session that originated on an entirely
101      *                                 different host (although it is not a JSecurity requirement this scenario is disallowed -
102      *                                 its just an example that <em>may</em> throw an Exception in many systems).
103      * @see HostUnauthorizedException
104      */
105     Session getSession(Serializable sessionId) throws InvalidSessionException, AuthorizationException;
106 
107 }
