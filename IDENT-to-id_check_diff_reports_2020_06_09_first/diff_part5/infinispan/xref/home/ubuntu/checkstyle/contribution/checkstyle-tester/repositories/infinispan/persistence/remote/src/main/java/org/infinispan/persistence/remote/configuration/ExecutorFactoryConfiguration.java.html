1   package org.infinispan.persistence.remote.configuration;
2   
3   import org.infinispan.commons.configuration.AbstractTypedPropertiesConfiguration;
4   import org.infinispan.commons.configuration.attributes.AttributeDefinition;
5   import org.infinispan.commons.configuration.attributes.AttributeInitializer;
6   import org.infinispan.commons.configuration.attributes.AttributeSet;
7   import org.infinispan.commons.executors.ExecutorFactory;
8   import org.infinispan.executors.DefaultExecutorFactory;
9   
10  public class ExecutorFactoryConfiguration extends AbstractTypedPropertiesConfiguration {
11     static final AttributeDefinition<ExecutorFactory> EXECUTOR_FACTORY = AttributeDefinition.builder("executorFactory", null, ExecutorFactory.class)
12           .initializer(new AttributeInitializer<ExecutorFactory>() {
13  
14              @Override
15              public ExecutorFactory initialize() {
16                 return new DefaultExecutorFactory();
17              }
18           }).immutable().build();
19  
20     public static AttributeSet attributeSet() {
21        return new AttributeSet(ExecutorFactoryConfiguration.class, AbstractTypedPropertiesConfiguration.attributeSet(), EXECUTOR_FACTORY);
22     };
23  
24     ExecutorFactoryConfiguration(AttributeSet attributes) {
25        super(attributes);
26     }
27  
28     public ExecutorFactory factory() {
29        return attributes.attribute(EXECUTOR_FACTORY).get();
30     }
31  
32     @Override
33     public String toString() {
34        return "ExecutorFactoryConfiguration [attributes=" + attributes + "]";
35     }
36  
37     AttributeSet attributes() {
38        return attributes;
39     }
40  
41  }
