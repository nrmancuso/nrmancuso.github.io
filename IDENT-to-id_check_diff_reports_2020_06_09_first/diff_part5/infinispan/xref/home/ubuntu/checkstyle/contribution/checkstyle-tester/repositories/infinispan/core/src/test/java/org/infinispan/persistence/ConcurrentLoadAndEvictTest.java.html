32  /**
33   * Tests a thread going past the cache loader interceptor and the interceptor deciding that loading is not necessary,
34   * then another thread rushing ahead and evicting the entry from memory.
35   *
36   * @author Manik Surtani
37   */
38  @Test(groups = "functional", testName = "persistence.ConcurrentLoadAndEvictTest")
39  public class ConcurrentLoadAndEvictTest extends SingleCacheManagerTest {
40     SlowDownInterceptor sdi;
41  
42     @Override
43     protected EmbeddedCacheManager createCacheManager() throws Exception {
44        sdi = new SlowDownInterceptor();
45        // we need a loader and a custom interceptor to intercept get() calls
46        // after the CLI, to slow it down so an evict goes through first
47        ConfigurationBuilder config = new ConfigurationBuilder();
48        config
49           .persistence()
50              .addStore(DummyInMemoryStoreConfigurationBuilder.class)
51           .customInterceptors()
52              .addInterceptor()
53                 .interceptor(sdi).after(InvocationContextInterceptor.class)
54           .transaction().transactionMode(TransactionMode.NON_TRANSACTIONAL);
55  
56        return TestCacheManagerFactory.createCacheManager(config);
57     }
58  
59     public void testEvictBeforeRead() throws PersistenceException, ExecutionException, InterruptedException {
60        cache = cacheManager.getCache();
61        cache.put("a", "b");
62        assert cache.get("a").equals("b");
63        CacheLoader cl = TestingUtil.getCacheLoader(cache);
64        assert cl != null;
65        MarshalledEntry se = cl.load("a");
66        assert se != null;
67        assert se.getValue().equals("b");
68  
69        // clear the cache
70        cache.getAdvancedCache().withFlags(SKIP_CACHE_STORE).clear();
71  
72        se = cl.load("a");
73        assert se != null;
74        assert se.getValue().equals("b");
75  
76        // now attempt a concurrent get and evict.
77        ExecutorService e = Executors.newFixedThreadPool(1);
78        sdi.enabled = true;
79  
80        log.info("test::doing the get");
81  
82        // call the get
83        Future<String> future = e.submit(new Callable<String>() {
84           @Override
85           public String call() throws Exception {
86              return (String) cache.get("a");
87           }
88        });
89  
90        // now run the evict.
91        log.info("test::before the evict");
92        cache.evict("a");
93        log.info("test::after the evict");
94  
95        // make sure the get call, which would have gone past the cache loader interceptor first, gets the correct value.
96        assert future.get().equals("b");
97  
98        // disable the SlowDownInterceptor
99        sdi.enabled = false;
100 
101       // and check that the key actually has been evicted
102       assert !TestingUtil.extractComponent(cache, DataContainer.class).containsKey("a");
103 
104       e.shutdownNow();
105    }

107    public static class SlowDownInterceptor extends CommandInterceptor implements Cloneable{
108 
109       private static final long serialVersionUID = 8790944676490291484L;
110 
111       volatile boolean enabled = false;
112       transient CountDownLatch getLatch = new CountDownLatch(1);
113       transient CountDownLatch evictLatch = new CountDownLatch(1);
114 
115       @Override
116       public Object visitGetKeyValueCommand(InvocationContext ctx, GetKeyValueCommand command) throws Throwable {
117          if (enabled) {
118             getLog().trace("Wait for evict to give go ahead...");
119             if (!evictLatch.await(60000, TimeUnit.MILLISECONDS))
120                throw new TimeoutException("Didn't see get after 60 seconds!");
121          }
122          try {
123             return invokeNextInterceptor(ctx, command);
124          } finally {
125             getLog().trace("After get, now let evict go through");
126             if (enabled) getLatch.countDown();
127          }
128       }
129 
130       @Override
131       public Object visitEvictCommand(InvocationContext ctx, EvictCommand command) throws Throwable {
132          if (enabled) {
133             evictLatch.countDown();
134             getLog().trace("Wait for get to finish...");
135             if (!getLatch.await(60000, TimeUnit.MILLISECONDS))
136                throw new TimeoutException("Didn't see evict after 60 seconds!");
137          }
138          return invokeNextInterceptor(ctx, command);
139       }
140       @Override
141       public SlowDownInterceptor clone(){
142          try {
143             return (SlowDownInterceptor) super.clone();
144          } catch (CloneNotSupportedException e) {
145             throw new RuntimeException("Should not happen", e);
146          }
147       }
148    }
