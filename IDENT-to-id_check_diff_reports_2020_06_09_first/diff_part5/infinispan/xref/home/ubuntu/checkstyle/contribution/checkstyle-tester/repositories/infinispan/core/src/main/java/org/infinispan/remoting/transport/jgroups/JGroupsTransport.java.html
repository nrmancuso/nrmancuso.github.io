78  /**
79   * An encapsulation of a JGroups transport. JGroups transports can be configured using a variety of
80   * methods, usually by passing in one of the following properties:
81   * <ul>
82   * <li><tt>configurationString</tt> - a JGroups configuration String</li>
83   * <li><tt>configurationXml</tt> - JGroups configuration XML as a String</li>
84   * <li><tt>configurationFile</tt> - String pointing to a JGroups XML configuration file</li>
85   * <li><tt>channelLookup</tt> - Fully qualified class name of a
86   * {@link org.infinispan.remoting.transport.jgroups.JGroupsChannelLookup} instance</li>
87   * </ul>
88   * These are normally passed in as Properties in
89   * {@link org.infinispan.configuration.global.TransportConfigurationBuilder#withProperties(java.util.Properties)} or
90   * in the Infinispan XML configuration file.

95   */

194    protected void startJGroupsChannelIfNeeded() {
195       String clusterName = configuration.transport().clusterName();
196       if (connectChannel) {
197          try {
198             channel.connect(clusterName);
199          } catch (Exception e) {
200             throw new CacheException("Unable to start JGroups Channel", e);
201          }
202 
203          try {
204             // Normally this would be done by CacheManagerJmxRegistration but
205             // the channel is not started when the cache manager starts but
206             // when first cache starts, so it's safer to do it here.
207             globalStatsEnabled = configuration.globalJmxStatistics().enabled();
208             if (globalStatsEnabled) {
209                String groupName = String.format("type=channel,cluster=%s", ObjectName.quote(clusterName));
210                mbeanServer = JmxUtil.lookupMBeanServer(configuration);
211                domain = JmxUtil.buildJmxDomain(configuration, mbeanServer, groupName);
212                JmxConfigurator.registerChannel((JChannel) channel, mbeanServer, domain, clusterName, true);
213             }
214          } catch (Exception e) {
215             throw new CacheException("Channel connected, but unable to register MBeans", e);
216          }
217       }
218       address = fromJGroupsAddress(channel.getAddress());
219       if (!connectChannel) {
220          // the channel was already started externally, we need to initialize our member list
221          viewAccepted(channel.getView());
222       }
223       if (log.isInfoEnabled())
224          log.localAndPhysicalAddress(clusterName, getAddress(), getPhysicalAddresses());
225    }

234    @Override
235    public void waitForView(int viewId) throws InterruptedException {
236       if (channel == null)
237          return;
238       log.tracef("Waiting on view %d being accepted", viewId);
239       viewUpdateLock.lock();
240       try {
241          while (channel != null && getViewId() < viewId) {
242             viewUpdateCondition.await();
243          }
244       } finally {
245          viewUpdateLock.unlock();
246       }
247    }

249    @Override
250    public void stop() {
251       String clusterName = configuration.transport().clusterName();
252       try {
253          if (disconnectChannel && channel != null && channel.isConnected()) {
254             log.disconnectJGroups(clusterName);
255 
256             // Unregistering before disconnecting/closing because
257             // after that the cluster name is null
258             if (globalStatsEnabled) {
259                JmxConfigurator.unregisterChannel((JChannel) channel, mbeanServer, domain, channel.getClusterName());
260             }
261 
262             channel.disconnect();
263          }
264          if (closeChannel && channel != null && channel.isOpen()) {
265             channel.close();
266          }
267       } catch (Exception toLog) {
268          log.problemClosingChannel(toLog, clusterName);
269       }
270 
271       if (dispatcher != null) {
272          log.stoppingRpcDispatcher(clusterName);
273          dispatcher.stop();
274          if (channel != null) {
275             // Remove reference to up_handler
276             UpHandler handler = channel.getUpHandler();
277             if (handler instanceof Muxer<?>) {
278                @SuppressWarnings("unchecked")
279                Muxer<UpHandler> mux = (Muxer<UpHandler>) handler;
280                mux.setDefaultHandler(null);
281             } else {
282                channel.setUpHandler(null);
283             }
284          }
285       }
286 
287       channel = null;
288       viewId = -1;
289       members = InfinispanCollections.emptyList();
290       coordinator = null;
291       isCoordinator = false;
292       dispatcher = null;
293 
294       // Wake up any view waiters
295       viewUpdateLock.lock();
296       try {
297          viewUpdateCondition.signalAll();
298       } finally {
299          viewUpdateLock.unlock();
300       }
301 
302    }

304    protected void initChannel() {
305       final TransportConfiguration transportCfg = configuration.transport();
306       if (channel == null) {
307          buildChannel();
308          String transportNodeName = transportCfg.nodeName();
309          if (transportNodeName != null && transportNodeName.length() > 0) {
310             long range = Short.MAX_VALUE * 2;
311             long randomInRange = (long) ((Math.random() * range) % range) + 1;
312             transportNodeName = transportNodeName + "-" + randomInRange;
313             channel.setName(transportNodeName);
314          }
315       }
316 
317       // Channel.LOCAL *must* be set to false so we don't see our own messages - otherwise
318       // invalidations targeted at remote instances will be received by self.
319       // NOTE: total order needs to deliver own messages. the invokeRemotely method has a total order boolean
320       //       that when it is false, it discard our own messages, maintaining the property needed
321       channel.setDiscardOwnMessages(false);
322 
323       // if we have a TopologyAwareConsistentHash, we need to set our own address generator in JGroups
324       if (transportCfg.hasTopologyInfo()) {
325          // We can do this only if the channel hasn't been started already
326          if (connectChannel) {
327             ((JChannel) channel).setAddressGenerator(new AddressGenerator() {
328                @Override
329                public org.jgroups.Address generateAddress() {
330                   return TopologyUUID.randomUUID(channel.getName(),
331                         transportCfg.siteId(), transportCfg.rackId(),
332                         transportCfg.machineId());
333                }
334             });
335          } else {
336             if (channel.getAddress() instanceof TopologyUUID) {
337                TopologyUUID topologyAddress = (TopologyUUID) channel.getAddress();
338                if (!transportCfg.siteId().equals(topologyAddress.getSiteId())
339                      || !transportCfg.rackId().equals(topologyAddress.getRackId())
340                      || !transportCfg.machineId().equals(topologyAddress.getMachineId())) {
341                   throw new CacheException("Topology information does not match the one set by the provided JGroups channel");
342                }
343             } else {
344                throw new CacheException("JGroups address does not contain topology coordinates");
345             }
346          }
347       }
348    }

363    // This is per CM, so the CL in use should be the CM CL
364    private void buildChannel() {
365 	  FileLookup fileLookup = FileLookupFactory.newInstance();
366 
367       // in order of preference - we first look for an external JGroups file, then a set of XML
368       // properties, and
369       // finally the legacy JGroups String properties.
370       String cfg;
371       if (props != null) {
372          if (props.containsKey(CHANNEL_LOOKUP)) {
373             String channelLookupClassName = props.getProperty(CHANNEL_LOOKUP);
374 
375             try {
376                JGroupsChannelLookup lookup = Util.getInstance(channelLookupClassName, configuration.classLoader());
377                channel = lookup.getJGroupsChannel(props);
378                connectChannel = lookup.shouldConnect();
379                disconnectChannel = lookup.shouldDisconnect();
380                closeChannel = lookup.shouldClose();
381             } catch (ClassCastException e) {
382                log.wrongTypeForJGroupsChannelLookup(channelLookupClassName, e);
383                throw new CacheException(e);
384             } catch (Exception e) {
385                log.errorInstantiatingJGroupsChannelLookup(channelLookupClassName, e);
386                throw new CacheException(e);
387             }
388          }
389 
390          if (channel == null && props.containsKey(CONFIGURATION_FILE)) {
391             cfg = props.getProperty(CONFIGURATION_FILE);
392             Collection<URL> confs = null;
393             try {
394                confs = fileLookup.lookupFileLocations(cfg, configuration.classLoader());
395             } catch (IOException io) {
396                //ignore, we check confs later for various states
397             }
398             if (confs.isEmpty()) {
399                throw log.jgroupsConfigurationNotFound(cfg);
400             } else if (confs.size() > 1) {
401                log.ambiguousConfigurationFiles(Util.toStr(confs));
402             }
403             try {
404                channel = new JChannel(confs.iterator().next());
405             } catch (Exception e) {
406                throw log.errorCreatingChannelFromConfigFile(cfg, e);
407             }
408          }
409 
410          if (channel == null && props.containsKey(CONFIGURATION_XML)) {
411             cfg = props.getProperty(CONFIGURATION_XML);
412             try {
413                channel = new JChannel(XmlConfigHelper.stringToElement(cfg));
414             } catch (Exception e) {
415                throw log.errorCreatingChannelFromXML(cfg, e);
416             }
417          }
418 
419          if (channel == null && props.containsKey(CONFIGURATION_STRING)) {
420             cfg = props.getProperty(CONFIGURATION_STRING);
421             try {
422                channel = new JChannel(cfg);
423             } catch (Exception e) {
424                throw log.errorCreatingChannelFromConfigString(cfg, e);
425 
426             }
427          }
428       }
429 
430       if (channel == null) {
431          log.unableToUseJGroupsPropertiesProvided(props);
432          try {
433             channel = new JChannel(fileLookup.lookupFileLocation(DEFAULT_JGROUPS_CONFIGURATION_FILE, configuration.classLoader()));
434          } catch (Exception e) {
435             throw log.errorCreatingChannelFromConfigFile(DEFAULT_JGROUPS_CONFIGURATION_FILE, e);
436          }
437       }
438    }

500                                                 ResponseMode mode, long timeout, boolean usePriorityQueue, ResponseFilter responseFilter,

502       DeliverOrder deliverOrder = DeliverOrder.PER_SENDER;

504          deliverOrder = DeliverOrder.TOTAL;

511    @Override
512    public Map<Address, Response> invokeRemotely(Collection<Address> recipients, ReplicableCommand rpcCommand,
513                                                 ResponseMode mode, long timeout, ResponseFilter responseFilter,
514                                                 DeliverOrder deliverOrder, boolean anycast) throws Exception {
515       if (recipients != null && recipients.isEmpty()) {
516          // don't send if recipients list is empty
517          log.trace("Destination list is empty: no need to send message");
518          return InfinispanCollections.emptyMap();
519       }
520       boolean totalOrder = deliverOrder == DeliverOrder.TOTAL;
521 
522       if (trace)
523          log.tracef("dests=%s, command=%s, mode=%s, timeout=%s", recipients, rpcCommand, mode, timeout);
524       Address self = getAddress();
525       boolean ignoreLeavers = mode == ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS || mode == ResponseMode.WAIT_FOR_VALID_RESPONSE;
526       if (mode.isSynchronous() && recipients != null && !getMembers().containsAll(recipients)) {
527          if (!ignoreLeavers) { // SYNCHRONOUS
528             throw new SuspectException("One or more nodes have left the cluster while replicating command " + rpcCommand);
529          }
530       }
531       boolean asyncMarshalling = mode == ResponseMode.ASYNCHRONOUS;
532 
533       List<org.jgroups.Address> jgAddressList = toJGroupsAddressListExcludingSelf(recipients, totalOrder);
534       int membersSize = members.size();
535       boolean broadcast = jgAddressList == null || recipients.size() == membersSize;
536       if (membersSize < 3 || (jgAddressList != null && jgAddressList.size() < 2)) broadcast = false;
537       RspList<Object> rsps = null;
538       Response singleResponse = null;
539       org.jgroups.Address singleJGAddress = null;
540 
541       if (broadcast) {
542          rsps = dispatcher.broadcastRemoteCommands(rpcCommand, toJGroupsMode(mode), timeout,
543                                                    toJGroupsFilter(responseFilter), deliverOrder, asyncMarshalling, ignoreLeavers);
544       } else if (totalOrder) {
545          rsps = dispatcher.invokeRemoteCommands(jgAddressList, rpcCommand, toJGroupsMode(mode), timeout,
546                                                 toJGroupsFilter(responseFilter), deliverOrder, asyncMarshalling, ignoreLeavers);
547       } else {
548          if (jgAddressList == null || !jgAddressList.isEmpty()) {
549             boolean singleRecipient = !ignoreLeavers && jgAddressList != null && jgAddressList.size() == 1;
550             boolean skipRpc = false;
551             if (jgAddressList == null) {
552                ArrayList<Address> others = new ArrayList<>(members);
553                others.remove(self);
554                skipRpc = others.isEmpty();
555                singleRecipient = !ignoreLeavers && others.size() == 1;
556                if (singleRecipient) singleJGAddress = toJGroupsAddress(others.get(0));
557             }
558             if (!skipRpc) {
559                if (singleRecipient) {
560                   if (singleJGAddress == null) singleJGAddress = jgAddressList.get(0);
561                   singleResponse = dispatcher.invokeRemoteCommand(singleJGAddress, rpcCommand, toJGroupsMode(mode), timeout,
562                                                                   deliverOrder, asyncMarshalling);
563                } else {
564                   rsps = dispatcher.invokeRemoteCommands(jgAddressList, rpcCommand, toJGroupsMode(mode), timeout,
565                                                          toJGroupsFilter(responseFilter), deliverOrder, asyncMarshalling, ignoreLeavers);
566                }
567             }
568          }
569       }
570 
571       if (mode.isAsynchronous())
572          return InfinispanCollections.emptyMap();// async case
573 
574       Map<Address, Response> responses;
575       if (rsps == null) {
576          if (singleJGAddress == null || (singleResponse == null && rpcCommand instanceof ClusteredGetCommand)) {
577             responses = InfinispanCollections.emptyMap();
578          } else {
579             responses = Collections.singletonMap(fromJGroupsAddress(singleJGAddress), singleResponse);
580          }
581       } else {
582          Map<Address, Response> retval = new HashMap<>(rsps.size());
583 
584          boolean noValidResponses = true;
585          for (Rsp<Object> rsp : rsps.values()) {
586             noValidResponses &= parseResponseAndAddToResponseList(rsp.getValue(), rsp.getException(), retval, rsp.wasSuspected(), rsp.wasReceived(), fromJGroupsAddress(rsp.getSender()),
587                                                                   responseFilter != null, ignoreLeavers);
588          }
589 
590          if (noValidResponses)
591             throw new TimeoutException("Timed out waiting for valid responses!");
592          responses = retval;
593       }
594       return responses;
595    }

597    @Override
598    public Map<Address, Response> invokeRemotely(Map<Address, ReplicableCommand> rpcCommands, ResponseMode mode,
599          long timeout, boolean usePriorityQueue, ResponseFilter responseFilter, boolean totalOrder, boolean anycast)
600          throws Exception {
601       if (rpcCommands == null || rpcCommands.isEmpty()) {
602          // don't send if recipients list is empty
603          log.trace("Destination list is empty: no need to send message");
604          return InfinispanCollections.emptyMap();
605       }
606 
607       if (trace)
608          log.tracef("commands=%s, mode=%s, timeout=%s", rpcCommands, mode, timeout);
609       boolean ignoreLeavers = mode == ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS || mode == ResponseMode.WAIT_FOR_VALID_RESPONSE;
610       List<Address> members = getMembers();
611 
612       Map<org.jgroups.Address, ReplicableCommand> forJGroups = new HashMap<>();
613       for (Map.Entry<Address, ReplicableCommand> entry : rpcCommands.entrySet()) {
614          if (!members.contains(entry.getKey())) {
615             if (mode.isSynchronous() && ignoreLeavers) {
616                throw new SuspectException("One or more nodes have left the cluster while replicating commands " + rpcCommands);
617             }
618          } else {
619             forJGroups.put(toJGroupsAddress(entry.getKey()), entry.getValue());
620          }
621       }
622       boolean asyncMarshalling = mode == ResponseMode.ASYNCHRONOUS;
623       if (!usePriorityQueue && (ResponseMode.SYNCHRONOUS == mode || ResponseMode.SYNCHRONOUS_IGNORE_LEAVERS == mode))
624          usePriorityQueue = true;
625 
626       RspList<Object> rsps = dispatcher.invokeRemoteCommands(forJGroups, toJGroupsMode(mode), timeout,
627                usePriorityQueue, asyncMarshalling, ignoreLeavers);
628 
629       if (mode.isAsynchronous() || rsps == null)
630          return InfinispanCollections.emptyMap();// async case
631 
632       Set<Address> suspectedAddresses = new HashSet<>();
633       Map<Address, Response> retval = new HashMap<Address, Response>(rsps.size());
634       boolean noValidResponses = true;
635       for (Entry<org.jgroups.Address, Rsp<Object>> entry : rsps.entrySet()) {
636          Rsp<Object> rsp = entry.getValue();
637          Object value = rsp.getValue();
638          if (value instanceof ExceptionResponse) {
639             ExceptionResponse exceptionResponse = (ExceptionResponse) value;
640             if (exceptionResponse.getException() instanceof SuspectedException) {
641                suspectedAddresses.add(fromJGroupsAddress(entry.getKey()));
642                continue;
643             }
644          }
645          noValidResponses &= parseResponseAndAddToResponseList(value,
646                rsp.getException(), retval, rsp.wasSuspected(), rsp.wasReceived(),
647                fromJGroupsAddress(rsp.getSender()), responseFilter != null, ignoreLeavers);
648       }
649 
650       if (noValidResponses) {
651          boolean throwException;
652          // It is possible for their to be no valid response if we ignored leavers and
653          // all of our targets left
654          if (ignoreLeavers) {
655             // If all the targets were suspected we didn't have a timeout
656             throwException = !suspectedAddresses.containsAll(rpcCommands.keySet());
657          } else {
658             throwException = true;
659          }
660          if (throwException)
661             throw new TimeoutException("Timed out waiting for valid responses!");
662       }
663       return retval;
664    }

666    @Override
667    public BackupResponse backupRemotely(Collection<XSiteBackup> backups, XSiteReplicateCommand rpcCommand) throws Exception {
668       log.tracef("About to send to backups %s, command %s", backups, rpcCommand);
669       Buffer buf = CommandAwareRpcDispatcher.marshallCall(dispatcher.getMarshaller(), rpcCommand);
670       Map<XSiteBackup, Future<Object>> syncBackupCalls = new HashMap<>(backups.size());
671       for (XSiteBackup xsb : backups) {
672          SiteMaster recipient = new SiteMaster(xsb.getSiteName());
673          if (xsb.isSync()) {
674             RequestOptions sync = new RequestOptions(org.jgroups.blocks.ResponseMode.GET_ALL, xsb.getTimeout());
675             syncBackupCalls.put(xsb, dispatcher.sendMessageWithFuture(CommandAwareRpcDispatcher.constructMessage(buf, recipient, true, org.jgroups.blocks.ResponseMode.GET_ALL, false, false), sync));
676          } else {
677             RequestOptions async = new RequestOptions(org.jgroups.blocks.ResponseMode.GET_NONE, xsb.getTimeout());
678             dispatcher.sendMessage(CommandAwareRpcDispatcher.constructMessage(buf, recipient, false, org.jgroups.blocks.ResponseMode.GET_NONE, false, false), async);
679          }
680       }
681       return new JGroupsBackupResponse(syncBackupCalls, timeService);
682    }

684    private static org.jgroups.blocks.ResponseMode toJGroupsMode(ResponseMode mode) {
685       switch (mode) {
686          case ASYNCHRONOUS:
687          case ASYNCHRONOUS_WITH_SYNC_MARSHALLING:
688             return org.jgroups.blocks.ResponseMode.GET_NONE;
689          case SYNCHRONOUS:

695       throw new CacheException("Unknown response mode " + mode);

719       @Override

729          List<List<Address>> l = new ArrayList<List<Address>>(subviews.size());

732          return l;

736    @Override
737    public void viewAccepted(View newView) {
738       log.debugf("New view accepted: %s", newView);
739       List<org.jgroups.Address> newMembers = newView.getMembers();
740       if (newMembers == null || newMembers.isEmpty()) {
741          log.debugf("Received null or empty member list from JGroups channel: " + newView);
742          return;
743       }
744 
745       List<Address> oldMembers = members;
746 
747       // Update every view-related field while holding the lock so that waitForView only returns
748       // after everything was updated.
749       viewUpdateLock.lock();
750       try {
751          viewId = (int) newView.getViewId().getId();
752 
753          // we need a defensive copy anyway
754          members = fromJGroupsAddressList(newMembers);
755 
756          // Delta view debug log for large cluster
757          if (log.isDebugEnabled() && oldMembers != null) {
758             List<Address> joined = new ArrayList<Address>(members);
759             joined.removeAll(oldMembers);
760             List<Address> left = new ArrayList<Address>(oldMembers);
761             left.removeAll(members);
762             log.debugf("Joined: %s, Left: %s", joined, left);
763          }
764 
765          // Now that we have a view, figure out if we are the isCoordinator
766          coordinator = fromJGroupsAddress(newView.getCreator());
767          isCoordinator = coordinator != null && coordinator.equals(getAddress());
768 
769          // Wake up any threads that are waiting to know about who the isCoordinator is
770          // do it before the notifications, so if a listener throws an exception we can still start
771          viewUpdateCondition.signalAll();
772       } finally {
773          viewUpdateLock.unlock();
774       }
775 
776       // now notify listeners - *after* updating the isCoordinator. - JBCACHE-662
777       boolean hasNotifier = notifier != null;
778       if (hasNotifier) {
779          String clusterName = configuration.transport().clusterName();
780          Notify n;
781          if (newView instanceof MergeView) {
782             log.receivedMergedView(clusterName, newView);
783             n = new NotifyMerge();
784          } else {
785             log.receivedClusterView(clusterName, newView);
786             n = new NotifyViewChange();
787          }
788 
789          n.emitNotification(oldMembers, newView);
790       }
791 
792       JGroupsAddressCache.pruneAddressCache();
793    }

822    private List<org.jgroups.Address> toJGroupsAddressListExcludingSelf(Collection<Address> list, boolean totalOrder) {
823       if (list == null)
824          return null;
825       if (list.isEmpty())
826          return InfinispanCollections.emptyList();
827 
828       List<org.jgroups.Address> retval = new ArrayList<org.jgroups.Address>(list.size());
829       boolean ignoreSelf = !totalOrder; //in total order, we need to send the message to ourselves!
830       Address self = getAddress();
831       for (Address a : list) {
832          if (!ignoreSelf || !a.equals(self)) {
833             retval.add(toJGroupsAddress(a));
834          } else {
835             ignoreSelf = false; // short circuit address equality for future iterations
836          }
837       }
838 
839       return retval;
840    }

842    private static List<Address> fromJGroupsAddressList(List<org.jgroups.Address> list) {
843       if (list == null || list.isEmpty())
844          return InfinispanCollections.emptyList();
845 
846       List<Address> retval = new ArrayList<Address>(list.size());
847       for (org.jgroups.Address a : list)
848          retval.add(fromJGroupsAddress(a));
849       return Collections.unmodifiableList(retval);
850    }
