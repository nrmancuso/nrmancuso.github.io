20  @Test(groups = "functional", testName = "distribution.DistAsyncTxFuncTest")
21  public class DistAsyncTxFuncTest extends DistSyncTxFuncTest {
22  
23     ReplListener r1, r2, r3, r4;
24     ReplListener[] r;
25     Map<Cache<?, ?>, ReplListener> listenerLookup;
26     List<Address> listenerCaches;
27  
28     public DistAsyncTxFuncTest() {
29        sync = false;
30        tx = true;
31        testRetVals = true;
32        cleanup = CleanupPhase.AFTER_METHOD; // ensure any stale TXs are wiped
33     }
34  
35     @Override
36     protected void createCacheManagers() throws Throwable {
37        super.createCacheManagers();
38        r1 = new ReplListener(c1, true, true);
39        r2 = new ReplListener(c2, true, true);
40        r3 = new ReplListener(c3, true, true);
41        r4 = new ReplListener(c4, true, true);
42        r = new ReplListener[]{r1, r2, r3, r4};
43        listenerLookup = new HashMap<Cache<?, ?>, ReplListener>();
44        listenerCaches = new ArrayList<Address>();
45        for (ReplListener rl : r) {
46           listenerCaches.add(addressOf(rl.getCache()));
47           listenerLookup.put(rl.getCache(), rl);
48        }
49     }
50    
51     @Override
52     protected void asyncWait(Object key, Class<? extends VisitableCommand> command, Cache<?, ?>... cachesOnWhichKeyShouldInval) {
53        if (cachesOnWhichKeyShouldInval == null) cachesOnWhichKeyShouldInval = new Cache[0];
54        List<Cache<?, ?>> cachesOnWhichKeyShouldInvalList = new ArrayList(Arrays.asList(cachesOnWhichKeyShouldInval));
55        if (key == null) {
56           // test all caches.
57           for (ReplListener rl : r) rl.expect(command);
58           for (ReplListener rl : r) rl.waitForRpc();
59        } else {
60           for (Cache<?, ?> c : getOwners(key)) {
61              log.info("Analysing cache " + addressOf(c) + ".  Listeners are avbl for caches " + listenerCaches);
62              if (cachesOnWhichKeyShouldInvalList.remove(c)) {
63                 listenerLookup.get(c).expect(command, InvalidateL1Command.class);
64              } else {
65                 listenerLookup.get(c).expect(command);
66              }
67              listenerLookup.get(c).waitForRpc();
68           }
69  
70           for (Cache<?, ?> c : cachesOnWhichKeyShouldInvalList) {
71              listenerLookup.get(c).expect(InvalidateL1Command.class);
72              listenerLookup.get(c).waitForRpc();
73           }
74        }
75     }
76  
77     @Override
78     protected void asyncTxWait(Object... keys) {
79        // Wait for a tx completion event
80        if (keys != null) {
81  
82           Set<Cache<?, ?>> cachesInTx = new HashSet<Cache<?, ?>>();
83           for (Object k : keys) {
84              cachesInTx.addAll(Arrays.asList(getOwners(k)));
85           }
86  
87           log.warn("In asyncTxWait, waiting for repl events on caches " + cachesInTx + " on keys " + Arrays.toString(keys));
88  
89           for (Cache<?, ?> c : cachesInTx) {
90              listenerLookup.get(c).expectAnyWithTx();
91              listenerLookup.get(c).waitForRpc(240, TimeUnit.SECONDS);
92           }
93  
94           // This sucks but for async transactions we still need this!!
95           TestingUtil.sleepThread(500 * keys.length);
96        }
97     }
98  }
