279 	private List<AuditJoinTableInfo> getAuditJoinTableRows(String middleEntityName, String joinColumnIdName,
280 														   String joinColumnIdProp, String inverseJoinColumnIdName,
281 														   String inverseJoinColumnIdProp, String revProp, String revIdProp,
282 														   String revTypeProp) throws Exception {
283 		StringBuilder qryBuilder = new StringBuilder( "select " );
284 		qryBuilder.append( "aud " );
285 		qryBuilder.append( ", " ).append( joinColumnIdProp ).append( " as joinColumnId" );
286 		qryBuilder.append( ", " ).append( inverseJoinColumnIdProp ).append( " as inverseJoinColumnId" );
287 		qryBuilder.append( ", " ).append( revProp ).append( " as rev" );
288 		qryBuilder.append( ", " ).append( revIdProp ).append( " as revId" );
289 		qryBuilder.append( ", " ).append( revTypeProp ).append( " as revType" );
290 		qryBuilder.append( " from " ).append( middleEntityName ).append( " aud " );
291 		qryBuilder.append( " order by joinColumnId asc, inverseJoinColumnId asc, revId asc" );
292 
293 		String query = qryBuilder.toString();
294 
295 		EntityManager em = createIsolatedEntityManager();
296 		Query qry = em.createQuery( query );
297 
298 		@SuppressWarnings("unchecked")
299 		List<Object[]> auditJoinTableRows = qry.getResultList();
300 		List<AuditJoinTableInfo> result = new ArrayList<AuditJoinTableInfo>( auditJoinTableRows.size() );
301 
302 		for ( Object[] auditJoinTableRow : auditJoinTableRows ) {
303 			Long joinColumnId = (Long) auditJoinTableRow[1];
304 			Long inverseJoinColumnId = (Long) auditJoinTableRow[2];
305 			SequenceIdRevisionEntity rev = (SequenceIdRevisionEntity) auditJoinTableRow[3];
306 			RevisionType revType = (RevisionType) auditJoinTableRow[5];
307 
308 			AuditJoinTableInfo info = new AuditJoinTableInfo(
309 					middleEntityName, rev, revType, joinColumnIdName, joinColumnId,
310 					inverseJoinColumnIdName, inverseJoinColumnId
311 			);
312 			result.add( info );
313 		}
314 
315 		return result;
316 	}

318 	private static class AuditJoinTableInfo {
319 		private final String name;
320 		private final Integer revId;
321 		private final RevisionType revType;
322 		private final String joinColumnName;
323 		private final Long joinColumnId;
324 		private final String inverseJoinColumnName;
325 		private final Long inverseJoinColumnId;
326 
327 		private AuditJoinTableInfo(String name, SequenceIdRevisionEntity rev,
328 								   RevisionType revType, String joinColumnName, Long joinColumnId,
329 								   String inverseJoinColumnName, Long inverseJoinColumnId) {
330 			this.name = name;
331 			this.revId = rev.getId();
332 			this.revType = revType;
333 			this.joinColumnName = joinColumnName;
334 			this.joinColumnId = joinColumnId;
335 			this.inverseJoinColumnName = inverseJoinColumnName;
336 			this.inverseJoinColumnId = inverseJoinColumnId;
337 		}
338 
339 		@Override
340 		public String toString() {
341 			return "AuditJoinTableInfo [name=" + name + ", revId=" + revId
342 					+ ", revType=" + revType + ", " + joinColumnName + "="
343 					+ joinColumnId + ", " + inverseJoinColumnName + "="
344 					+ inverseJoinColumnId + "]";
345 		}
346 
347 		@Override
348 		public boolean equals(Object o) {
349 			if ( this == o ) {
350 				return true;
351 			}
352 			if ( !( o instanceof AuditJoinTableInfo ) ) {
353 				return false;
354 			}
355 
356 			AuditJoinTableInfo that = (AuditJoinTableInfo) o;
357 
358 			if ( inverseJoinColumnId != null ? !inverseJoinColumnId.equals( that.inverseJoinColumnId ) : that.inverseJoinColumnId != null ) {
359 				return false;
360 			}
361 			if ( joinColumnId != null ? !joinColumnId.equals( that.joinColumnId ) : that.joinColumnId != null ) {
362 				return false;
363 			}
364 			if ( name != null ? !name.equals( that.name ) : that.name != null ) {
365 				return false;
366 			}
367 			if ( revId != null ? !revId.equals( that.revId ) : that.revId != null ) {
368 				return false;
369 			}
370 			if ( revType != that.revType ) {
371 				return false;
372 			}
373 
374 			return true;
375 		}
376 
377 		@Override
378 		public int hashCode() {
379 			int result = name != null ? name.hashCode() : 0;
380 			result = 31 * result + ( revId != null ? revId.hashCode() : 0 );
381 			result = 31 * result + ( revType != null ? revType.hashCode() : 0 );
382 			result = 31 * result + ( joinColumnId != null ? joinColumnId.hashCode() : 0 );
383 			result = 31 * result + ( inverseJoinColumnId != null ? inverseJoinColumnId.hashCode() : 0 );
384 			return result;
385 		}
386 	}

388 	public static void tryCommit(TransactionManager tm) throws Exception {
389 		if ( tm.getStatus() == Status.STATUS_MARKED_ROLLBACK ) {
390 			tm.rollback();
391 		}
392 		else {
393 			tm.commit();
394 		}
