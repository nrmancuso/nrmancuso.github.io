134 	 * Render the identifier select, but in a 'scalar' context (i.e. generate the column alias).
135 	 *
136 	 * @param i the sequence of the returned type
137 	 * @return the identifier select with the column alias.
138 	 */
139 	String renderScalarIdentifierSelect(int i) {
140 		checkInitialized();
141 		String[] cols = getPropertyMapping( EntityPersister.ENTITY_ID ).toColumns( getTableAlias(), EntityPersister.ENTITY_ID );
142 		StringBuilder buf = new StringBuilder();
143 		// For property references generate <tablealias>.<columnname> as <projectionalias>
144 		for ( int j = 0; j < cols.length; j++ ) {
145 			String column = cols[j];
146 			if ( j > 0 ) {
147 				buf.append( ", " );
148 			}
149 			buf.append( column ).append( " as " ).append( NameGenerator.scalarName( i, j ) );
150 		}
151 		return buf.toString();
152 	}

154 	/**
155 	 * Returns the identifier select SQL fragment.
156 	 *
157 	 * @param size The total number of returned types.
158 	 * @param k    The sequence of the current returned type.
159 	 * @return the identifier select SQL fragment.
160 	 */
161 	String renderIdentifierSelect(int size, int k) {
162 		checkInitialized();
163 		// Render the identifier select fragment using the table alias.
164 		if ( fromElement.getFromClause().isSubQuery() ) {
165 			// TODO: Replace this with a more elegant solution.
166 			String[] idColumnNames = ( persister != null ) ?
167 					( ( Queryable ) persister ).getIdentifierColumnNames() : new String[0];
168 			StringBuilder buf = new StringBuilder();
169 			for ( int i = 0; i < idColumnNames.length; i++ ) {
170 				buf.append( fromElement.getTableAlias() ).append( '.' ).append( idColumnNames[i] );
171 				if ( i != idColumnNames.length - 1 ) buf.append( ", " );
172 			}
173 			return buf.toString();
174 		}
175 		else {
176 			if (persister==null) {
177 				throw new QueryException( "not an entity" );
178 			}
179 			String fragment = ( ( Queryable ) persister ).identifierSelectFragment( getTableAlias(), getSuffix( size, k ) );
180 			return trimLeadingCommaAndSpaces( fragment );
181 		}
182 	}

204 		checkInitialized();

244 	/**
245 	 * This accounts for a quirk in Queryable, where it sometimes generates ',  ' in front of the
246 	 * SQL fragment.  :-P
247 	 *
248 	 * @param fragment An SQL fragment.
249 	 * @return The fragment, without the leading comma and spaces.
250 	 */
251 	private static String trimLeadingCommaAndSpaces(String fragment) {
252 		if ( fragment.length() > 0 && fragment.charAt( 0 ) == ',' ) {
253 			fragment = fragment.substring( 1 );
254 		}
255 		fragment = fragment.trim();
256 		return fragment.trim();
257 	}

263 	public JoinSequence getJoinSequence() {
264 		if ( joinSequence != null ) {
265 			return joinSequence;
266 		}
267 
268 		// Class names in the FROM clause result in a JoinSequence (the old FromParser does this).
269 		if ( persister instanceof Joinable ) {
270 			Joinable joinable = ( Joinable ) persister;
271 			return fromElement.getSessionFactoryHelper().createJoinSequence().setRoot( joinable, getTableAlias() );
272 		}
273 		else {
274 			return null;	// TODO: Should this really return null?  If not, figure out something better to do here.
275 		}

279 		if ( this.queryableCollection != null ) {
280 			throw new IllegalStateException( "QueryableCollection is already defined for " + this + "!" );
281 		}
282 		this.queryableCollection = queryableCollection;
283 		if ( !queryableCollection.isOneToMany() ) {
284 			// For many-to-many joins, use the tablename from the queryable collection for the default text.
285 			fromElement.setText( queryableCollection.getTableName() + " " + getTableAlias() );
286 		}

293 	/**
294 	 * Returns the type of a property, given it's name (the last part) and the full path.
295 	 *
296 	 * @param propertyName The last part of the full path to the property.
297 	 * @return The type.
298 	 * @0param propertyPath The full property path.
299 	 */
300 	public Type getPropertyType(String propertyName, String propertyPath) {
301 		checkInitialized();
302 		Type type = null;
303 		// If this is an entity and the property is the identifier property, then use getIdentifierType().
304 		//      Note that the propertyName.equals( propertyPath ) checks whether we have a component
305 		//      key reference, where the component class property name is the same as the
306 		//      entity id property name; if the two are not equal, this is the case and
307 		//      we'd need to "fall through" to using the property mapping.
308 		if ( persister != null && propertyName.equals( propertyPath ) && propertyName.equals( persister.getIdentifierPropertyName() ) ) {
309 			type = persister.getIdentifierType();
310 		}
311 		else {	// Otherwise, use the property mapping.
312 			PropertyMapping mapping = getPropertyMapping( propertyName );
313 			type = mapping.toType( propertyPath );
314 		}
315 		if ( type == null ) {
316 			throw new MappingException( "Property " + propertyName + " does not exist in " +
317 					( ( queryableCollection == null ) ? "class" : "collection" ) + " "
318 					+ ( ( queryableCollection == null ) ? fromElement.getClassName() : queryableCollection.getRole() ) );
319 		}
320 		return type;
321 	}

327 	String[] toColumns(String tableAlias, String path, boolean inSelect, boolean forceAlias) {
328 		checkInitialized();
329 		PropertyMapping propertyMapping = getPropertyMapping( path );
330 
331 		if ( !inSelect && queryableCollection != null && CollectionProperties.isCollectionProperty( path ) ) {
332 			// If this from element is a collection and the path is a collection property (maxIndex, etc.)
333 			// requiring a sub-query then generate a sub-query.
334 			//h
335 			// Unless we are in the select clause, because some dialects do not support
336 			// Note however, that some dialects do not  However, in the case of this being a collection property reference being in the select, not generating the subquery
337 			// will not generally work.  The specific cases I am thinking about are the minIndex, maxIndex
338 			// (most likely minElement, maxElement as well) cases.
339 			//	todo : if ^^ is the case we should thrown an exception here rather than waiting for the sql error
340 			//		if the dialect supports select-clause subqueries we could go ahead and generate the subquery also
341 			Map enabledFilters = fromElement.getWalker().getEnabledFilters();
342 			String subquery = CollectionSubqueryFactory.createCollectionSubquery(
343 					joinSequence.copy().setUseThetaStyle( true ),
344 					enabledFilters,
345 					propertyMapping.toColumns( tableAlias, path )
346 			);
347 			LOG.debugf( "toColumns(%s,%s) : subquery = %s", tableAlias, path, subquery );
348 			return new String[]{"(" + subquery + ")"};
349 		}
350 
351         if (forceAlias) {
352             return propertyMapping.toColumns(tableAlias, path);
353         }
354 
355 		if (fromElement.getWalker().getStatementType() == HqlSqlTokenTypes.SELECT) {
356             return propertyMapping.toColumns(tableAlias, path);
357         }
358 
359 		if (fromElement.getWalker().isSubQuery()) {
360             // for a subquery, the alias to use depends on a few things (we
361             // already know this is not an overall SELECT):
362             // 1) if this FROM_ELEMENT represents a correlation to the
363             // outer-most query
364             // A) if the outer query represents a multi-table
365             // persister, we need to use the given alias
366             // in anticipation of one of the multi-table
367             // executors being used (as this subquery will
368             // actually be used in the "id select" phase
369             // of that multi-table executor)
370             // B) otherwise, we need to use the persister's
371             // table name as the column qualification
372             // 2) otherwise (not correlated), use the given alias
373             if (isCorrelation()) {
374                 if (isMultiTable()) {
375 					return propertyMapping.toColumns(tableAlias, path);
376 				}
377                 return propertyMapping.toColumns(extractTableName(), path);
378 			}
379             return propertyMapping.toColumns(tableAlias, path);
380         }
381 
382 		if (fromElement.getWalker().getCurrentTopLevelClauseType() == HqlSqlTokenTypes.SELECT) {
383             return propertyMapping.toColumns(tableAlias, path);
384         }
385 
386 		if ( isManipulationQuery() && isMultiTable() && inWhereClause() ) {
387 			// the actual where-clause will end up being ripped out the update/delete and used in
388 			// a select to populate the temp table, so its ok to use the table alias to qualify the table refs
389 			// and safer to do so to protect from same-named columns
390 			return propertyMapping.toColumns( tableAlias, path );
391 		}
392 
393 		String[] columns = propertyMapping.toColumns( path );
394 		LOG.tracev( "Using non-qualified column reference [{0} -> ({1})]", path, ArrayHelper.toString( columns ) );
395 		return columns;
396 	}

398 	private boolean isCorrelation() {
399 		FromClause top = fromElement.getWalker().getFinalFromClause();
400 		return fromElement.getFromClause() != fromElement.getWalker().getCurrentFromClause() &&
401 	           fromElement.getFromClause() == top;
402 	}

415 	private boolean isManipulationQuery() {
416 		return fromElement.getWalker().getStatementType() == HqlSqlTokenTypes.UPDATE
417 				|| fromElement.getWalker().getStatementType() == HqlSqlTokenTypes.DELETE;
418 	}

425 			CollectionPropertyNames.COLLECTION_INDEX,
426 			CollectionPropertyNames.COLLECTION_MIN_INDEX,
427 			CollectionPropertyNames.COLLECTION_MAX_INDEX
428 	);

430 	PropertyMapping getPropertyMapping(String propertyName) {
431 		checkInitialized();
432 		if ( queryableCollection == null ) {		// Not a collection?
433 			return ( PropertyMapping ) persister;	// Return the entity property mapping.
434 		}
435 
436 		// indexed, many-to-many collections must be treated specially here if the property to
437 		// be mapped touches on the index as we must adjust the alias to use the alias from
438 		// the association table (which i different than the one passed in
439 		if ( queryableCollection.isManyToMany()
440 				&& queryableCollection.hasIndex()
441 				&& SPECIAL_MANY2MANY_TREATMENT_FUNCTION_NAMES.contains( propertyName ) ) {
442 			return new SpecialManyToManyCollectionPropertyMapping();
443 		}
444 
445 		// If the property is a special collection property name, return a CollectionPropertyMapping.
446 		if ( CollectionProperties.isCollectionProperty( propertyName ) ) {
447 			if ( collectionPropertyMapping == null ) {
448 				collectionPropertyMapping = new CollectionPropertyMapping( queryableCollection );
449 			}
450 			return collectionPropertyMapping;
451 		}
452 
453 		if ( queryableCollection.getElementType().isAnyType() ) {
454 			// collection of <many-to-any/> mappings...
455 			// used to circumvent the component-collection check below...
456 			return queryableCollection;
457 
458 		}
459 
460 		if ( queryableCollection.getElementType().isComponentType() ) {
461 			// Collection of components.
462 			if ( propertyName.equals( EntityPersister.ENTITY_ID ) ) {
463 				return ( PropertyMapping ) queryableCollection.getOwnerEntityPersister();
464 			}
465 		}
466 		return queryableCollection;
467 	}

487 	private class SpecialManyToManyCollectionPropertyMapping implements PropertyMapping {
488 		/**
489 		 * {@inheritDoc}
490 		 */
491 		public Type getType() {
492 			return queryableCollection.getCollectionType();
493 		}
494 
495 		private void validate(String propertyName) {
496 			if ( ! ( CollectionPropertyNames.COLLECTION_INDEX.equals( propertyName )
497 					|| CollectionPropertyNames.COLLECTION_MAX_INDEX.equals( propertyName )
498 					|| CollectionPropertyNames.COLLECTION_MIN_INDEX.equals( propertyName ) ) ) {
499 				throw new IllegalArgumentException( "Expecting index-related function call" );
500 			}
501 		}
502 
503 		/**
504 		 * {@inheritDoc}
505 		 */
506 		public Type toType(String propertyName) throws QueryException {
507 			validate( propertyName );
508 			return queryableCollection.getIndexType();
509 		}
510 
511 		/**
512 		 * {@inheritDoc}
513 		 */
514 		public String[] toColumns(String alias, String propertyName) throws QueryException {
515 			validate( propertyName );
516 			final String joinTableAlias = joinSequence.getFirstJoin().getAlias();
517 			if ( CollectionPropertyNames.COLLECTION_INDEX.equals( propertyName ) ) {
518 				return queryableCollection.toColumns( joinTableAlias, propertyName );
519 			}
520 
521 			final String[] cols = queryableCollection.getIndexColumnNames( joinTableAlias );
522 			if ( CollectionPropertyNames.COLLECTION_MIN_INDEX.equals( propertyName ) ) {
523 				if ( cols.length != 1 ) {
524 					throw new QueryException( "composite collection index in minIndex()" );
525 				}
526 				return new String[] { "min(" + cols[0] + ')' };
527 			}
528 			else {
529 				if ( cols.length != 1 ) {
530 					throw new QueryException( "composite collection index in maxIndex()" );
531 				}
532 				return new String[] { "max(" + cols[0] + ')' };
533 			}
534 		}
535 
536 		/**
537 		 * {@inheritDoc}
538 		 */
539 		public String[] toColumns(String propertyName) throws QueryException, UnsupportedOperationException {
540 			validate( propertyName );
541 			return queryableCollection.toColumns( propertyName );
542 		}
543 	}
