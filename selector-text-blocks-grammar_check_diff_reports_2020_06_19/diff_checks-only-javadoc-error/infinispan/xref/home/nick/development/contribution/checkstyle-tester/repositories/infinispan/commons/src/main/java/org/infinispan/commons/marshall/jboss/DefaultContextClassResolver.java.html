1   package org.infinispan.commons.marshall.jboss;
2   
3   import org.jboss.marshalling.ContextClassResolver;
4   
5   import java.lang.ref.WeakReference;
6   
7   /**
8    * This class refines <code>ContextClassLoader</code> to add a default class loader.
9    * The context class loader is only used when the default is <code>null</code>.
10   *
11   * @author Dan Berindei <dberinde@redhat.com>
12   * @since 4.2
13   */
14  public class DefaultContextClassResolver extends ContextClassResolver {
15  
16     private WeakReference<ClassLoader> defaultClassLoader;
17  
18     public DefaultContextClassResolver(ClassLoader defaultClassLoader) {
19        this.defaultClassLoader = new WeakReference<ClassLoader>(defaultClassLoader);
20     }
21  
22     @Override
23     protected ClassLoader getClassLoader() {
24        ClassLoader defaultLoader = this.defaultClassLoader.get();
25        return defaultLoader != null ? defaultLoader : super.getClassLoader();
26     }
27  }
