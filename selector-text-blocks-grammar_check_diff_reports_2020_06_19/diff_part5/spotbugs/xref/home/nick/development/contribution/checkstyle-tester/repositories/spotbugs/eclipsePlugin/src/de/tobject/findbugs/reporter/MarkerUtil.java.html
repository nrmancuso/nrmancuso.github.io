112     /**
113      * Create an Eclipse marker for given BugInstance.
114      *
115      * @param javaProject
116      *            the project
117      * @param monitor
118      */
119     public static void createMarkers(final IJavaProject javaProject, final SortedBugCollection theCollection, final ISchedulingRule rule, IProgressMonitor monitor) {
120         if (monitor.isCanceled()) {
121             return;
122         }
123         final List<MarkerParameter> bugParameters = createBugParameters(javaProject, theCollection, monitor);
124         if (monitor.isCanceled()) {
125             return;
126         }
127         WorkspaceJob wsJob = new WorkspaceJob("Creating SpotBugs markers") {
128 
129             @Override
130             public IStatus runInWorkspace(IProgressMonitor monitor1) throws CoreException {
131                 IProject project = javaProject.getProject();
132                 try {
133                     new MarkerReporter(bugParameters, theCollection, project).run(monitor1);
134                 } catch (CoreException e) {
135                     FindbugsPlugin.getDefault().logException(e, "Core exception on add marker");
136                     return e.getStatus();
137                 }
138                 return monitor1.isCanceled()? Status.CANCEL_STATUS : Status.OK_STATUS;
139             }
140         };
141         wsJob.setRule(rule);
142         wsJob.setSystem(true);
143         wsJob.setUser(false);
144         wsJob.schedule();
145     }

147     /**
148      * As a side-effect this method updates missing line information for some
149      * bugs stored in the given bug collection
150      *
151      * @param project
152      * @param theCollection
153      * @return never null
154      */
155     public static List<MarkerParameter> createBugParameters(IJavaProject project, BugCollection theCollection,
156             IProgressMonitor monitor) {
157         List<MarkerParameter> bugParameters = new ArrayList<>();
158         if (project == null) {
159             FindbugsPlugin.getDefault().logException(new NullPointerException("project is null"), "project is null");
160             return bugParameters;
161         }
162         Iterator<BugInstance> iterator = theCollection.iterator();
163         while (iterator.hasNext() && !monitor.isCanceled()) {
164             BugInstance bug = iterator.next();
165             DetectorFactory detectorFactory = bug.getDetectorFactory();
166             if(detectorFactory != null && !detectorFactory.getPlugin().isGloballyEnabled()){
167                 continue;
168             }
169             MarkerParameter mp = createMarkerParameter(project, bug);
170             if (mp != null) {
171                 bugParameters.add(mp);
172             }
173         }
174         return bugParameters;
175     }

177     private static MarkerParameter createMarkerParameter(IJavaProject project, BugInstance bug) {
178         IJavaElement type = null;
179         WorkItem resource = null;
180         try {
181             type = getJavaElement(bug, project);
182             if (type != null) {
183                 resource = new WorkItem(type);
184             }
185         } catch (JavaModelException e1) {
186             FindbugsPlugin.getDefault().logException(e1, "Could not find Java type for FindBugs warning");
187         }
188         if (resource == null) {
189             if (Reporter.DEBUG) {
190                 reportNoResourceFound(bug);
191             }
192             return null;
193         }
194 
195         // default - first class line
196         int primaryLine = bug.getPrimarySourceLineAnnotation().getStartLine();
197 
198         // FindBugs needs originally generated primary line in order to find the
199         // bug again.
200         // Sometimes this primary line is <= 0, which causes Eclipse editor to
201         // ignore it
202         // So we check if we can replace the "wrong" primary line with a
203         // "better" start line
204         // If not, we just provide two values - one for Eclipse, another for
205         // FindBugs itself.
206 
207         int startLine = DONT_KNOW_LINE;
208 
209         // only update if we don't already tried it
210         if(primaryLine != DONT_KNOW_LINE) {
211             // XXX "first line of a file" is too simplistic. What if we have inner types?
212             if (primaryLine <= 1 && type instanceof IType) {
213                 IType iType = (IType) type;
214                 try {
215                     startLine = getLineStart(iType);
216                     if(startLine > 0) {
217                         if (Reporter.DEBUG) {
218                             System.out.println("4. Fixed start line to: " + startLine + " on " + type.getElementName());
219                         }
220                     }
221                 } catch (JavaModelException e1) {
222                     FindbugsPlugin.getDefault().logException(e1, "Could not find source line for Java type " + type
223                             + "for SpotBugs warning: " + bug);
224                 }
225             }
226         }
227 
228         if (primaryLine <= 0 && startLine <= 0) {
229             // We have to provide line number, otherwise editor wouldn't show it
230             startLine = 1;
231             if (Reporter.DEBUG) {
232                 System.out.println("5. Fixed start line to *default* (1) on " + type);
233             }
234         }
235 
236         MarkerParameter parameter;
237         if (startLine > 0) {
238             parameter = new MarkerParameter(bug, resource, startLine, primaryLine);
239         } else {
240             parameter = new MarkerParameter(bug, resource, primaryLine, primaryLine);
241         }
242 //        if (Reporter.DEBUG) {
243 //            System.out
244 //            .println("Creating marker for " + resource.getPath() + ": line " + parameter.primaryLine + bug.getMessage());
245 //        }
246         return parameter;
247     }

249     private static void reportNoResourceFound(BugInstance bug) {
250         String className = null;
251         String packageName = null;
252         ClassAnnotation primaryClass = bug.getPrimaryClass();
253         if (primaryClass != null) {
254             className = primaryClass.getClassName();
255             packageName = primaryClass.getPackageName();
256         }
257         if (Reporter.DEBUG) {
258             System.out.println("NOT found resource for a BUG in BUG in class: " //$NON-NLS-1$
259                     + packageName + "." //$NON-NLS-1$
260                     + className + ": \n\t" //$NON-NLS-1$
261                     + bug.getMessage() + " / Annotation: " //$NON-NLS-1$
262                     + bug.getPrimarySourceLineAnnotation());
263         }
264     }

266     /**
267      * Get the underlying resource (Java class) for given BugInstance.
268      *
269      * @param bug
270      *            the BugInstance
271      * @param project
272      *            the project
273      * @return the IResource representing the Java class
274      */
275     private static @CheckForNull
276     IJavaElement getJavaElement(BugInstance bug, IJavaProject project) throws JavaModelException {
277 
278         SourceLineAnnotation primarySourceLineAnnotation = bug.getPrimarySourceLineAnnotation();
279         String qualifiedClassName = primarySourceLineAnnotation.getClassName();
280 
281 //        if (Reporter.DEBUG) {
282 //            System.out.println("Looking up class: " + packageName + ", " + qualifiedClassName);
283 //        }
284 
285         Matcher m = fullName.matcher(qualifiedClassName);
286         IType type;
287         String innerName = null;
288         if (m.matches() && m.group(2).length() > 0) {
289 
290             String outerQualifiedClassName = m.group(1).replace('$', '.');
291             innerName = m.group(2).substring(1);
292             // second argument is required to find also secondary types
293             type = project.findType(outerQualifiedClassName, (IProgressMonitor) null);
294 
295             /*
296              * code below only points to the first line of inner class even if
297              * this is not a class bug but field bug
298              */
299             if (type != null && !hasLineInfo(primarySourceLineAnnotation)) {
300                 completeInnerClassInfo(qualifiedClassName, innerName, type, bug);
301             }
302         } else {
303             // second argument is required to find also secondary types
304             type = project.findType(qualifiedClassName.replace('$', '.'), (IProgressMonitor) null);
305 
306             // for inner classes, some detectors does not properly report source
307             // lines:
308             // instead of reporting the first line of inner class, they report
309             // first line of parent class
310             // in this case we will try to fix this here and point to the right
311             // start line
312             if (type != null && type.isMember()) {
313                 if (!hasLineInfo(primarySourceLineAnnotation)) {
314                     completeInnerClassInfo(qualifiedClassName, type.getElementName(), type, bug);
315                 }
316             }
317         }
318 
319         // reassign it as it may be changed for inner classes
320         primarySourceLineAnnotation = bug.getPrimarySourceLineAnnotation();
321 
322         /*
323          * Eclipse can help us find the line number for fields => we trying to
324          * add line info for fields here
325          */
326         int startLine = primarySourceLineAnnotation.getStartLine();
327         // TODO don't use "1", use "0" ?
328         if (startLine <= 1 && type != null) {
329             FieldAnnotation primaryField = bug.getPrimaryField();
330             if(primaryField != null) {
331                 completeFieldInfo(qualifiedClassName, type, bug, primaryField);
332             }
333         }
334 
335         return type;
336     }

338     private static boolean hasLineInfo(SourceLineAnnotation annotation) {
339         // XXX don't use "1", use "0" ?
340         return annotation != null && annotation.getStartLine() > 1;
341     }

343     private static void completeFieldInfo(String qualifiedClassName,
344             @Nonnull IType type, @Nonnull BugInstance bug, @Nonnull FieldAnnotation field) {
345 
346         IField ifield = type.getField(field.getFieldName());
347         ISourceRange sourceRange = null;
348         JavaModelException ex = null;
349         try {
350             sourceRange = ifield.getNameRange();
351         } catch (JavaModelException e) {
352             ex = e;
353         }
354         IScanner scanner = null;
355         try {
356             // second try...
357             if (sourceRange == null) {
358                 sourceRange = ifield.getSourceRange();
359             }
360             scanner = initScanner(type, sourceRange);
361         } catch (JavaModelException e) {
362             String message = "Can not complete field annotation " + field + " for the field: " + ifield + " in class: "
363                     + qualifiedClassName + ", type " + type + ", bug " + bug;
364             if (ex != null) {
365                 // report only first one
366                 e = ex;
367             }
368             FindbugsPlugin.getDefault().logMessage(IStatus.WARNING, message, e);
369         }
370         if (scanner == null || sourceRange == null) {
371             return;
372         }
373         int lineNbr = scanner.getLineNumber(sourceRange.getOffset());
374         if(lineNbr > 0){
375             String sourceFileStr = getSourceFileHint(type, qualifiedClassName);
376             field.setSourceLines(new SourceLineAnnotation(qualifiedClassName, sourceFileStr, lineNbr, lineNbr, 0, 0));
377             if (Reporter.DEBUG) {
378                 System.out.println("2. Fixed start line to: " + lineNbr + " on " + qualifiedClassName);
379             }
380         }
381     }

384         String sourceFileStr = "";
385         IJavaElement primaryElement = type.getPrimaryElement();
386         if (primaryElement != null) {
387             return primaryElement.getElementName() + ".java";
388         }
389         return sourceFileStr;

392     private static void completeInnerClassInfo(String qualifiedClassName, String innerName, @Nonnull IType type, BugInstance bug) throws JavaModelException {
393         int lineNbr = findChildSourceLine(type, innerName, bug);
394         if(lineNbr > 0){
395             String sourceFileStr = getSourceFileHint(type, qualifiedClassName);
396             if (sourceFileStr != null && sourceFileStr.length() > 0) {
397                 bug.addSourceLine(new SourceLineAnnotation(qualifiedClassName, sourceFileStr, lineNbr, lineNbr, 0, 0));
398                 if (Reporter.DEBUG) {
399                     System.out.println("1. Fixed start line to: " + lineNbr + " on " + qualifiedClassName + "$" + innerName);
400                 }
401             }
402         }
403     }

405     /**
406      * @return start line of given type, or {@link #DONT_KNOW_LINE} if line could not be found
407      */
408     private static int getLineStart(IType source) throws JavaModelException {
409         ISourceRange range = source.getNameRange();
410         if (range == null) {
411             range = source.getSourceRange();
412         }
413         IScanner scanner = initScanner(source, range);
414         if (scanner != null && range != null) {
415             return scanner.getLineNumber(range.getOffset());
416         }
417         return DONT_KNOW_LINE;
418     }

420     /**
421      * @param source
422      *            must be not null
423      * @param range
424      *            can be null
425      * @return may return null, otherwise an initialized scanner which may
426      *         answer which source offset index belongs to which source line
427      * @throws JavaModelException
428      */
429     private static IScanner initScanner(IType source, ISourceRange range) throws JavaModelException {
430         if (range == null) {
431             return null;
432         }
433         char[] charContent = getContent(source);
434         if (charContent == null) {
435             return null;
436         }
437         IScanner scanner = ToolFactory.createScanner(false, false, false, true);
438         scanner.setSource(charContent);
439         int offset = range.getOffset();
440         try {
441             while (scanner.getNextToken() != ITerminalSymbols.TokenNameEOF) {
442                 // do nothing, just wait for the end of stream
443                 if (offset <= scanner.getCurrentTokenEndPosition()) {
444                     break;
445                 }
446             }
447         } catch (InvalidInputException e) {
448             FindbugsPlugin.getDefault().logException(e, "Could not init scanner for type: " + source);
449         }
450         return scanner;
451     }

453     @SuppressWarnings("restriction")
454     private static char[] getContent(IType source) throws JavaModelException {
455         char[] charContent = null;
456         IOpenable op = source.getOpenable();
457         if (op instanceof CompilationUnit) {
458             charContent = ((CompilationUnit) (op)).getContents();
459         }
460         if (charContent == null) {
461             String content = source.getSource();
462             if (content != null) {
463                 charContent = content.toCharArray();
464             }
465         }
466         return charContent;
467     }

469     private static int findChildSourceLine(IType parentType, String name, BugInstance bug) throws JavaModelException {
470         if (parentType == null) {
471             return -1;
472         }
473         char firstChar = name.charAt(0);
474         boolean firstIsDigit = Character.isDigit(firstChar);
475         if (!firstIsDigit) {
476             return findInnerClassSourceLine(parentType, name);
477         }
478         boolean innerFromMember = firstIsDigit && name.length() > 1 && !Character.isDigit(name.charAt(1));
479         if (innerFromMember) {
480             return findInnerClassSourceLine(parentType, name.substring(1));
481         }
482         return findInnerAnonymousClassSourceLine(parentType, name);
483     }

485     private static int findInnerAnonymousClassSourceLine(IJavaElement parentType, String innerName) throws JavaModelException {
486         IType anon = JdtUtils.findAnonymous((IType) parentType, innerName);
487         if (anon != null) {
488             return getLineStart(anon);
489         }
490         return DONT_KNOW_LINE;
491     }

493     private static int findInnerClassSourceLine(IJavaElement type, String name) throws JavaModelException {
494         String elemName = type.getElementName();
495         if (name.equals(elemName)) {
496             if (type instanceof IType) {
497                 return getLineStart((IType) type);
498             }
499         }
500         if (type instanceof IParent) {
501             IJavaElement[] children = ((IParent) type).getChildren();
502             for (int i = 0; i < children.length; i++) {
503                 // recursive call
504                 int line = findInnerClassSourceLine(children[i], name);
505                 if (line > 0) {
506                     return line;
507                 }
508             }
509         }
510         return DONT_KNOW_LINE;
511     }

566                 createMarkers(javaProject, bugs, project, monitor);
567             }
568         };
569         job.setRule(project);
570         job.scheduleInteractive();
571     }

649     public static Set<IMarker> findMarkerForJavaElement(IJavaElement elt, IMarker[] possibleCandidates, boolean recursive) {
650         String id = elt.getHandleIdentifier();
651         Set<IMarker> markers = new HashSet<>();
652         for (IMarker marker : possibleCandidates) {
653             try {
654                 Object elementId = marker.getAttribute(FindBugsMarker.UNIQUE_JAVA_ID);
655                 // UNIQUE_JAVA_ID exists first since 1.3.9 as FB attribute
656                 if (!(elementId instanceof String)) {
657                     continue;
658                 }
659                 String stringId = (String) elementId;
660                 if (!recursive) {
661                     if (stringId.equals(id)) {
662                         // exact match
663                         markers.add(marker);
664                     } else if (isDirectChild(id, stringId)) {
665                         // direct child: class in the package, but not in the
666                         // sub-package
667                         markers.add(marker);
668                     }
669                 } else if (stringId.startsWith(id)) {
670                     markers.add(marker);
671                 }
672             } catch (CoreException e) {
673                 FindbugsPlugin.getDefault().logException(e, "Marker does not contain valid java element id");
674                 continue;
675             }
676         }
677         return markers;
678     }

680     /**
681      *
682      * @param parentId
683      *            java element id of a parent element
684      * @param childId
685      *            java element id of possible child
686      * @return true if the second string represents a java element which is a
687      *         direct child of the parent element.
688      */
689     @SuppressWarnings("restriction")
690     private static boolean isDirectChild(String parentId, String childId) {
691         return childId.startsWith(parentId) && (childId.length() > (parentId.length() + 1))
692                 // if there is NOT a class file separator, then it's not a direct child
693                 && childId.charAt(parentId.length()) == JavaElement.JEM_CLASSFILE;
694     }

743     /**
744      * Find the BugCollectionAndInstance associated with given FindBugs marker.
745      *
746      * @param marker
747      *            a FindBugs marker
748      * @return the BugInstance associated with the marker, or null if we can't
749      *         find the BugInstance
750      */
751     public static @CheckForNull
752     BugCollectionAndInstance findBugCollectionAndInstanceForMarker(IMarker marker) {
753 
754         IResource resource = marker.getResource();
755         IProject project = resource.getProject();
756         if (project == null) {
757             // Also shouldn't happen.
758             FindbugsPlugin.getDefault().logError("No project for warning marker");
759             return null;
760         }
761         if (!isFindBugsMarker(marker)) {
762             // log disabled because otherwise each selection in problems view
763             // generates
764             // 6 new errors (we need refactor all bug views to get rid of this).
765             // FindbugsPlugin.getDefault().logError("Selected marker is not a FindBugs marker");
766             // FindbugsPlugin.getDefault().logError(marker.getType());
767             // FindbugsPlugin.getDefault().logError(FindBugsMarker.NAME);
768             return null;
769         }
770 
771         // We have a FindBugs marker. Get the corresponding BugInstance.
772         String bugId = marker.getAttribute(FindBugsMarker.UNIQUE_ID, null);
773         if (bugId == null) {
774             FindbugsPlugin.getDefault().logError("Marker does not contain unique id for warning");
775             return null;
776         }
777 
778         try {
779             BugCollection bugCollection = FindbugsPlugin.getBugCollection(project, null);
780             if (bugCollection == null) {
781                 FindbugsPlugin.getDefault().logError("Could not get BugCollection for SpotBugs marker");
782                 return null;
783             }
784 
785             String bugType = (String) marker.getAttribute(FindBugsMarker.BUG_TYPE);
786             Integer primaryLineNumber = (Integer) marker.getAttribute(FindBugsMarker.PRIMARY_LINE);
787 
788             // compatibility
789             if (primaryLineNumber == null) {
790                 primaryLineNumber = Integer.valueOf(getEditorLine(marker));
791             }
792 
793             if (bugType == null) {
794                 FindbugsPlugin.getDefault().logError(
795                         "Could not get find attributes for marker " + marker + ": (" + bugId + ", " + primaryLineNumber + ")");
796                 return null;
797             }
798             BugInstance bug = bugCollection.findBug(bugId, bugType, primaryLineNumber.intValue());
799             if(bug == null) {
800                 FindbugsPlugin.getDefault().logError(
801                         "Could not get find bug for marker on " + resource + ": (" + bugId + ", " + primaryLineNumber + ")");
802                 return null;
803             }
804             return new BugCollectionAndInstance(bugCollection, bug);
805         } catch (CoreException e) {
806             FindbugsPlugin.getDefault().logException(e, "Could not get BugInstance for SpotBugs marker");
807             return null;
808         }
809     }

815     /**
816      * Fish an IMarker out of given selection.
817      *
818      * @param selection
819      *            the selection
820      * @return the selected IMarker, or null if we can't find an IMarker in the
821      *         selection
822      */
823     public static Set<IMarker> getMarkerFromSelection(ISelection selection) {
824         Set<IMarker> markers = new HashSet<>();
825         if (!(selection instanceof IStructuredSelection)) {
826             return markers;
827         }
828         IStructuredSelection sSelection = (IStructuredSelection) selection;
829         for (Iterator<?> iter = sSelection.iterator(); iter.hasNext();) {
830             Object next = iter.next();
831             markers.addAll(getMarkers(next));
832         }
833         return markers;
834     }

836     public static Set<IMarker> getMarkers(Object obj) {
837         Set<IMarker> markers = new HashSet<>();
838         if (obj instanceof IMarker) {
839             IMarker marker = (IMarker) obj;
840             if (isFindBugsMarker(marker)) {
841                 markers.add(marker);
842             }
843         } else if (obj instanceof BugGroup) {
844             BugGroup group = (BugGroup) obj;
845             markers.addAll(group.getAllMarkers());
846         } else if (obj instanceof IResource) {
847             IResource res = (IResource) obj;
848             IMarker[] markers2 = MarkerUtil.getAllMarkers(res);
849             for (IMarker marker : markers2) {
850                 markers.add(marker);
851             }
852         } else if (obj instanceof IJavaElement) {
853             markers.addAll(new WorkItem((IJavaElement) obj).getMarkers(true));
854         } else if (obj instanceof IAdaptable) {
855             IAdaptable adapter = (IAdaptable) obj;
856             IMarker marker = (IMarker) adapter.getAdapter(IMarker.class);
857             if (marker == null) {
858                 IResource resource = (IResource) adapter.getAdapter(IResource.class);
859                 if (resource == null) {
860                     return markers;
861                 }
862                 IMarker[] markers2 = getMarkers(resource, IResource.DEPTH_INFINITE);
863                 markers.addAll(Arrays.asList(markers2));
864             } else if (isFindBugsMarker(marker)) {
865                 markers.add(marker);
866             }
867         }
868         return markers;
869     }

871     /**
872      * Tries to retrieve right bug marker for given selection. If there are many
873      * markers for given editor, and text selection doesn't match any of them,
874      * return null. If there is only one marker for given editor, returns this
875      * marker in any case.
876      *
877      * @param selection
878      * @param editor
879      * @return may return null
880      */
881     public static IMarker getMarkerFromEditor(ITextSelection selection, IEditorPart editor) {
882         IResource resource = (IResource) editor.getEditorInput().getAdapter(IFile.class);
883         IMarker[] allMarkers;
884         if (resource != null) {
885             allMarkers = getMarkers(resource, IResource.DEPTH_ZERO);
886         } else {
887             IClassFile classFile = (IClassFile) editor.getEditorInput().getAdapter(IClassFile.class);
888             if (classFile == null) {
889                 return null;
890             }
891             Set<IMarker> markers = getMarkers(classFile.getType());
892             allMarkers = markers.toArray(new IMarker[markers.size()]);
893         }
894         // if editor contains only one FB marker, do some cheating and always
895         // return it.
896         if (allMarkers.length == 1) {
897             return allMarkers[0];
898         }
899         // +1 because it counts real lines, but editor shows lines + 1
900         int startLine = selection.getStartLine() + 1;
901         for (IMarker marker : allMarkers) {
902             int line = getEditorLine(marker);
903             if (startLine == line) {
904                 return marker;
905             }
906         }
907         return null;
908     }

910     public static IMarker getMarkerFromSingleSelection(ISelection selection) {
911         if (selection instanceof ITextSelection) {
912             IEditorPart editor = FindbugsPlugin.getActiveWorkbenchWindow().getActivePage().getActiveEditor();
913             if (!(editor instanceof ITextEditor)) {
914                 return null;
915             }
916             IMarker marker = MarkerUtil.getMarkerFromEditor((ITextSelection) selection, (ITextEditor) editor);
917             if (marker != null) {
918                 selection = new StructuredSelection(marker);
919             } else {
920                 selection = new StructuredSelection();
921             }
922         }
923 
924         if (!(selection instanceof IStructuredSelection)) {
925             return null;
926         }
927         IStructuredSelection sSelection = (IStructuredSelection) selection;
928         if (sSelection.size() != 1) {
929             return null;
930         }
931 
932         Object next = sSelection.getFirstElement();
933         if (next instanceof IMarker) {
934             IMarker marker = (IMarker) next;
935             if (!isFindBugsMarker(marker)) {
936                 return null;
937             }
938             return marker;
939         } else if (next instanceof BugGroup) {
940             return null;
941         } else if (next instanceof IResource) {
942             return null;
943         } else if (next instanceof IAdaptable) {
944             IAdaptable adapter = (IAdaptable) next;
945             IMarker marker = (IMarker) adapter.getAdapter(IMarker.class);
946             if (!isFindBugsMarker(marker)) {
947                 return null;
948             }
949             return marker;
950         }
951         return null;
952     }

960             FindbugsPlugin.getDefault().logException(e, "Exception while checking SpotBugs type on marker.");
961         }
962         return false;

1008     /**
1009      * @param marker
1010      *            might be null
1011      * @param bugIdToFilter
1012      *            might be null
1013      * @return true if marker should be filtered
1014      */
1015     public static boolean isFiltered(IMarker marker, Set<String> bugIdToFilter) {
1016         if (marker == null) {
1017             return true;
1018         }
1019         if (bugIdToFilter == null) {
1020             return false;
1021         }
1022         String pattern = marker.getAttribute(FindBugsMarker.BUG_TYPE, "not found");
1023         String patternType = marker.getAttribute(FindBugsMarker.PATTERN_TYPE, "not found");
1024         for (String badId : bugIdToFilter) {
1025             if (badId.equals(patternType) || badId.equals(pattern)) {
1026                 return true;
1027             }
1028         }
1029         return false;
1030     }
