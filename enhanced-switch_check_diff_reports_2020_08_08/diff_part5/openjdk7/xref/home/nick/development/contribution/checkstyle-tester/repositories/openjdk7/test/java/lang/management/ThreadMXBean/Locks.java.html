38      private static Object objA = new Object();
39      private static Object objB = new Object();
40      private static Object objC = new Object();

45      private static String getLockName(Object lock) {
46          if (lock == null) return null;
47  
48          return lock.getClass().getName() + '@' +
49              Integer.toHexString(System.identityHashCode(lock));
50      }
51  
52      private static void checkBlockedObject(Thread t, Object lock, Thread owner,
53                                             Thread.State expectedState) {
54          ThreadInfo info = tm.getThreadInfo(t.getId());
55          String result = info.getLockName();
56          String expectedLock = (lock != null ? getLockName(lock) : null);
57          String expectedOwner = (owner != null ? owner.getName() : null);
58  
59          if (lock != null) {
60              if (expectedState ==Thread.State.BLOCKED) {
61                  int retryCount=0;
62                  while(info.getThreadState() != Thread.State.BLOCKED) {
63                      if (retryCount++ > 500) {
64                          throw new RuntimeException("Thread " + t.getName() +
65                                    " is expected to block on " + expectedLock +
66                                    " but got " + result +
67                                    " Thread.State = " + info.getThreadState());
68                      }
69                      goSleep(100);
70                  }
71              }
72              if (expectedState == Thread.State.WAITING &&
73                  info.getThreadState() != Thread.State.WAITING) {
74                  throw new RuntimeException("Thread " + t.getName() +
75                      " is expected to wait on " + expectedLock +
76                      " but got " + result +
77                      " Thread.State = " + info.getThreadState());
78              }
79          }
80  
81          if ((result != null && !result.equals(expectedLock)) ||
82              (result == null && expectedLock != null)) {
83              throw new RuntimeException("Thread " + t.getName() + " is blocked on " +
84                  expectedLock + " but got " + result);
85          }
86          result = info.getLockOwnerName();
87          if ((result != null && !result.equals(expectedOwner)) ||
88              (result == null && expectedOwner != null)) {
89              throw new RuntimeException("Owner of " + lock + " should be " +
90                  expectedOwner + " but got " + result);
91          }
92      }

94      private static void goSleep(long ms) {
95          try {
96              Thread.sleep(ms);
97          } catch (InterruptedException e) {
98              e.printStackTrace();
99              testFailed = true;
100         }
101     }

103     static ThreadExecutionSynchronizer thrsync = new ThreadExecutionSynchronizer();
104     static ThreadExecutionSynchronizer thrsync1 = new ThreadExecutionSynchronizer();
105 
106     static class LockAThread extends Thread {
107         public LockAThread() {
108             super("LockAThread");
109         }
110         public void run() {
111             synchronized(objA) {
112                // stop here  for LockBThread to hold objB
113                thrsync.waitForSignal();
114 
115                System.out.println("LockAThread about to block on objB");
116                synchronized(objB) {};
117             }
118             System.out.println("LockAThread about to exit");
119             // The state could be anything. The expected state value
120             // passed with this method is not verified.
121             checkBlockedObject(this, null, null, Thread.State.TERMINATED);
122         }
123     }
124 
125     static class LockBThread extends Thread {
126         public LockBThread() {
127             super("LockBThread");
128         }
129         public void run() {
130             synchronized(objB) {
131                // signal waiting LockAThread.
132                thrsync.signal();
133 
134                System.out.println("LockBThread about to block on objC");
135                // Signal main thread about to block on objC
136                thrsync1.signal();
137                synchronized(objC) {};
138             }
139             System.out.println("LockBThread about to exit");
140             // The state could be anything. The expected state value
141             // passed with this method is not verified.
142             checkBlockedObject(this, null, null, Thread.State.TERMINATED);
143         }
144 
145         public void aboutToLockC() {
146             // Stop here till LockBThread about to blocked
147             // for lock objC.
148             thrsync1.waitForSignal();
149             goSleep(500);
150         }
151     }

153     private static WaitingThread waiter;
154     private static Object ready = new Object();
155     private static CheckerThread checker;
156     static class WaitingThread extends Thread {
157         public WaitingThread() {
158             super("WaitingThread");
159         }
160         public void run() {
161             synchronized(objC) {
162                System.out.println("WaitingThread about to wait on objC");
163                try {
164                    // Signal checker thread, about to wait on objC.
165                    thrsync.signal();
166                    objC.wait();
167                } catch (InterruptedException e) {
168                    e.printStackTrace();
169                    testFailed = true;
170                }
171 
172                // block until CheckerThread finishes checking
173                System.out.println("WaitingThread about to block on ready");
174                // signal checker thread that it is about acquire
175                // object ready.
176                thrsync.signal();
177                synchronized(ready) {};
178             }
179             synchronized(objC) {
180                 try {
181                     // signal checker thread, about to wait on objC
182                     thrsync.signal();
183                     objC.wait();
184                 } catch (InterruptedException e) {
185                     e.printStackTrace();
186                     testFailed = true;
187                 }
188             }
189             System.out.println("WaitingThread about to exit waiting on objC 2");
190         }
191     }
192     static class CheckerThread extends Thread {
193         public CheckerThread() {
194             super("CheckerThread");
195         }
196         public void run() {
197             synchronized (ready) {
198                 // wait until WaitingThread about to wait for objC
199                 thrsync.waitForSignal();
200 
201                 int retryCount = 0;
202                 while (waiter.getState() != Thread.State.WAITING
203                        && retryCount++ < 500) {
204                    goSleep(100);
205                 }
206                 checkBlockedObject(waiter, objC, null, Thread.State.WAITING);
207 
208                 synchronized (objC) {
209                     objC.notify();
210                 }
211 
212                 // wait for waiter thread to about to enter
213                 // synchronized object ready.
214                 thrsync.waitForSignal();
215                 // give chance for waiter thread to get blocked on
216                 // object ready.
217                 goSleep(50);
218                 checkBlockedObject(waiter, ready, this, Thread.State.BLOCKED);
219             }
220 
221             // wait for signal from waiting thread that it is about
222             // wait for objC.
223             thrsync.waitForSignal();
224             synchronized(objC) {
225                 checkBlockedObject(waiter, objC, Thread.currentThread(), Thread.State.WAITING);
226                 objC.notify();
227             }
228 
229         }
230     }

232     public static void main(String args[]) throws Exception {
233         Thread mainThread = Thread.currentThread();
234 
235         // Test uncontested case
236         LockAThread t1;
237         LockBThread t2;
238 
239         synchronized(objC) {
240             // The state could be anything. The expected state value
241             // passed with this method is not verified.
242             checkBlockedObject(mainThread, null, null, Thread.State.RUNNABLE);
243 
244             // Test deadlock case
245             // t1 holds lockA and attempts to lock B
246             // t2 holds lockB and attempts to lock C
247             t1 = new LockAThread();
248             t1.start();
249 
250             t2 = new LockBThread();
251             t2.start();
252 
253             t2.aboutToLockC();
254 
255             checkBlockedObject(t1, objB, t2, Thread.State.BLOCKED);
256             checkBlockedObject(t2, objC, mainThread, Thread.State.BLOCKED);
257 
258             long[] expectedThreads = new long[3];
259             expectedThreads[0] = t1.getId(); // blocked on lockB
260             expectedThreads[1] = t2.getId(); // owner of lockB blocking on lockC
261             expectedThreads[2] = mainThread.getId(); // owner of lockC
262             findThreadsBlockedOn(objB, expectedThreads);
263         }
264         goSleep(100);
265 
266         // Test Object.wait() case
267         waiter = new WaitingThread();
268         waiter.start();
269 
270         checker = new CheckerThread();
271         checker.start();
272 
273         try {
274             waiter.join();
275             checker.join();
276         } catch (InterruptedException e) {
277             e.printStackTrace();
278             testFailed = true;
279         }
280 
281         if (testFailed) {
282             throw new RuntimeException("TEST FAILED.");
283         }
284         System.out.println("Test passed.");
285     }

287     private static ThreadInfo findOwnerInfo(ThreadInfo[] infos, String lock)
288         throws Exception {
289         ThreadInfo ownerInfo = null;
290         for (int i = 0; i < infos.length; i++) {
291             String blockedLock = infos[i].getLockName();
292             if (lock.equals(blockedLock)) {
293                 long threadId = infos[i].getLockOwnerId();
294                 if (threadId == -1) {
295                     throw new RuntimeException("TEST FAILED: " +
296                         lock + " expected to have owner");
297                 }
298                 for (int j = 0; j < infos.length; j++) {
299                     if (infos[j].getThreadId() == threadId) {
300                         ownerInfo = infos[j];
301                         break;
302                     }
303                 }
304             }
305         }
306         return ownerInfo;
307     }
308     private static void findThreadsBlockedOn(Object o, long[] expectedThreads)
309         throws Exception {
310         String lock = getLockName(o);
311         // Check with ThreadInfo with no stack trace (i.e. no safepoint)
312         ThreadInfo[] infos = tm.getThreadInfo(tm.getAllThreadIds());
313         doCheck(infos, lock, expectedThreads);
314 
315         // Check with ThreadInfo with stack trace
316         infos = tm.getThreadInfo(tm.getAllThreadIds(), 1);
317         doCheck(infos, lock, expectedThreads);
318     }
319 
320     private static void doCheck(ThreadInfo[] infos, String lock, long[] expectedThreads)
321         throws Exception {
322         ThreadInfo ownerInfo = null;
323         // Find the thread who is blocking on lock
324         for (int i = 0; i < infos.length;  i++) {
325             String blockedLock = infos[i].getLockName();
326             if (lock.equals(blockedLock)) {
327                 System.out.print(infos[i].getThreadName() +
328                     " blocked on " + blockedLock);
329                 ownerInfo = infos[i];
330             }
331         }
332 
333         long[] threads = new long[10];
334         int count = 0;
335         threads[count++] = ownerInfo.getThreadId();
336         while (ownerInfo != null && ownerInfo.getThreadState() == Thread.State.BLOCKED) {
337             ownerInfo = findOwnerInfo(infos, lock);
338             threads[count++] = ownerInfo.getThreadId();
339             System.out.println(" Owner = " + ownerInfo.getThreadName() +
340                 " id = " + ownerInfo.getThreadId());
341             lock = ownerInfo.getLockName();
342             System.out.print(ownerInfo.getThreadName() + " Id = " +
343                     ownerInfo.getThreadId() +
344                     " blocked on " + lock);
345         }
346         System.out.println();
347 
348         if (count != expectedThreads.length) {
349             throw new RuntimeException("TEST FAILED: " +
350                 "Expected chain of threads not matched; current count =" + count);
351         }
352         for (int i = 0; i < count; i++) {
353             if (threads[i] != expectedThreads[i]) {
354                 System.out.println("TEST FAILED: " +
355                     "Unexpected thread in the chain " + threads[i] +
356                     " expected to be " + expectedThreads[i]);
357             }
358         }
359     }
