26  package sun.io;
27  
28  /**
29   * A algorithmic conversion from ASCII to Unicode
30   *
31   * @author Limin Shi
32   */
33  public class ByteToCharASCII extends ByteToCharConverter {
34  
35      // Return the character set id
36      public String getCharacterEncoding()
37      {
38          return "ASCII";
39      }
40  
41      public int flush(char[] output, int outStart, int outEnd) {
42          // This converter will not buffer any data.
43          byteOff = charOff = 0;
44          return 0;
45      }
46  
47      /**
48       * Algorithmic character conversion
49       */
50      public int convert(byte[] input, int inOff, int inEnd,
51                         char[] output, int outOff, int outEnd)
52          throws ConversionBufferFullException, UnknownCharacterException
53      {
54          byte    inputByte;
55  
56          charOff = outOff;
57          byteOff = inOff;
58  
59          // Loop until we hit the end of the input
60          while(byteOff < inEnd)
61          {
62              // If we don't have room for the output, throw an exception
63              if (charOff >= outEnd)
64                  throw new ConversionBufferFullException();
65  
66              // Convert the input byte
67              inputByte = input[byteOff++];
68  
69              if (inputByte >= 0)
70                  output[charOff++] = (char)inputByte;
71              else {
72                  if (subMode)
73                      output[charOff++] = '\uFFFD';       // Replace Char
74                  else {
75                      badInputLength = 1;
76                      throw new UnknownCharacterException();
77                  }
78              }
79          }
80  
81          // Return the length written to the output buffer
82          return charOff-outOff;
83      }
84  
85      /*
86       *   Reset after finding bad input
87       */
88      public void reset() {
89          byteOff = charOff = 0;
90      }
91  
92  }
