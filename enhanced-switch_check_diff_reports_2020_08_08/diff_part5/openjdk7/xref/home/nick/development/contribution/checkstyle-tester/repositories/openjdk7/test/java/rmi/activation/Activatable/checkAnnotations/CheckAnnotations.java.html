140     /**
141      * check to make sure that the output from a spawned vm is
142      * formatted/annotated properly.
143      */
144     public static boolean checkAnnotations(int iteration)
145         throws IOException
146     {
147         try {
148             Thread.sleep(5000);
149         } catch(Exception e) {
150             System.err.println(e.getMessage());
151         }
152 
153         /**
154          * cause the spawned vm to generate output that will
155          * be checked for proper annotation.  printOut is
156          * actually being called on an activated implementation.
157          */
158         myRMI.printOut("out" + iteration);
159         myRMI.printErr("err" + iteration);
160         myRMI.printOut("out" + iteration);
161         myRMI.printErr("err" + iteration);
162 
163         /* we have to wait for output to filter down
164          * from children so we can read it before we
165          * kill rmid.
166          */
167 
168         String outString = null;
169         String errString = null;
170 
171         for (int i = 0 ; i < 5 ; i ++ ) {
172             // have to give output from rmid time to trickle down to
173             // this process
174             try {
175                 Thread.sleep(4000);
176             } catch(InterruptedException e) {
177             }
178 
179             outString = rmidOut.toString();
180             errString = rmidErr.toString();
181 
182             if ((!outString.equals("")) &&
183                 (!errString.equals("")))
184             {
185                 System.err.println("obtained annotations");
186                 break;
187             }
188             System.err.println("rmid output not yet received, retrying...");
189         }
190 
191         rmidOut.reset();
192         rmidErr.reset();
193 
194         // only test when we are annotating..., first run does not annotate
195         if (iteration >= 0) {
196             System.err.println("Checking annotations...");
197             System.err.println(outString);
198             System.err.println(errString);
199 
200             StringTokenizer stOut = new StringTokenizer(outString, ":");
201             StringTokenizer stErr = new StringTokenizer(errString, ":");
202 
203             String execErr = null;
204             String execOut = null;
205             String destOut = null;
206             String destErr = null;
207             String outTmp  = null;
208             String errTmp  = null;
209 
210             while (stOut.hasMoreTokens()) {
211                 execOut = outTmp;
212                 outTmp  = destOut;
213                 destOut = stOut.nextToken();
214             }
215             while (stErr.hasMoreTokens()) {
216                 execErr = errTmp;
217                 errTmp  = destErr;
218                 destErr = stErr.nextToken();
219             }
220 
221             if ((execErr == null)||(errTmp == null)||
222                 (destErr == null)) {
223                 return false;
224             }
225             if ((execOut == null)||(outTmp == null)||
226                 (destOut == null)) {
227                 return false;
228             }
229 
230             // just make sure that last two strings are what we expect.
231             if (execOut.equals("ExecGroup-" + iteration)
232                 && (new String(destOut.substring(0,4)).equals("out" +
233                                                               iteration))
234                 && (execErr.equals("ExecGroup-"+iteration))
235                 && (new String(destErr.substring(0,4)).equals("err" +
236                                                               iteration)) ) {
237                 return true;
238             } else {
239                 return false;
240             }
241         }
242         return true;
243     }
