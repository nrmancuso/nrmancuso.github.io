165 /**
166  * HMaster is the "master server" for HBase. An HBase cluster has one active
167  * master.  If many masters are started, all compete.  Whichever wins goes on to
168  * run the cluster.  All others park themselves in their constructor until
169  * master or cluster shutdown or until the active master loses its lease in
170  * zookeeper.  Thereafter, all running master jostle to take over master role.
171  *
172  * <p>The Master can be asked shutdown the cluster. See {@link #shutdown()}.  In
173  * this case it will tell all regionservers to go down and then wait on them
174  * all reporting in that they are down.  This master will then shut itself down.
175  *
176  * <p>You can also shutdown just this master.  Call {@link #stopMaster()}.
177  *
178  * @see org.apache.zookeeper.Watcher
179  */
180 @InterfaceAudience.LimitedPrivate(HBaseInterfaceAudience.TOOLS)
181 @SuppressWarnings("deprecation")
182 public class HMaster extends HRegionServer implements MasterServices, Server {
183   private static final Log LOG = LogFactory.getLog(HMaster.class.getName());

185   /**
186    * Protection against zombie master. Started once Master accepts active responsibility and
187    * starts taking over responsibilities. Allows a finite time window before giving up ownership.
188    */
189   private static class InitializationMonitor extends HasThread {
190     /** The amount of time in milliseconds to sleep before checking initialization status. */
191     public static final String TIMEOUT_KEY = "hbase.master.initializationmonitor.timeout";
192     public static final long TIMEOUT_DEFAULT = TimeUnit.MILLISECONDS.convert(15, TimeUnit.MINUTES);
193 
194     /**
195      * When timeout expired and initialization has not complete, call {@link System#exit(int)} when
196      * true, do nothing otherwise.
197      */
198     public static final String HALT_KEY = "hbase.master.initializationmonitor.haltontimeout";
199     public static final boolean HALT_DEFAULT = false;
200 
201     private final HMaster master;
202     private final long timeout;
203     private final boolean haltOnTimeout;
204 
205     /** Creates a Thread that monitors the {@link #isInitialized()} state. */
206     InitializationMonitor(HMaster master) {
207       super("MasterInitializationMonitor");
208       this.master = master;
209       this.timeout = master.getConfiguration().getLong(TIMEOUT_KEY, TIMEOUT_DEFAULT);
210       this.haltOnTimeout = master.getConfiguration().getBoolean(HALT_KEY, HALT_DEFAULT);
211       this.setDaemon(true);
212     }
213 
214     @Override
215     public void run() {
216       try {
217         while (!master.isStopped() && master.isActiveMaster()) {
218           Thread.sleep(timeout);
219           if (master.isInitialized()) {
220             LOG.debug("Initialization completed within allotted tolerance. Monitor exiting.");
221           } else {
222             LOG.error("Master failed to complete initialization after " + timeout + "ms. Please"
223                 + " consider submitting a bug report including a thread dump of this process.");
224             if (haltOnTimeout) {
225               LOG.error("Zombie Master exiting. Thread dump to stdout");
226               Threads.printThreadInfo(System.out, "Zombie HMaster");
227               System.exit(-1);
228             }
229           }
230         }
231       } catch (InterruptedException ie) {
232         LOG.trace("InitMonitor thread interrupted. Existing.");
233       }
234     }
235   }

346   /**
347    * Initializes the HMaster. The steps are as follows:
348    * <p>
349    * <ol>
350    * <li>Initialize the local HRegionServer
351    * <li>Start the ActiveMasterManager.
352    * </ol>
353    * <p>
354    * Remaining steps of initialization occur in
355    * #finishActiveMasterInitialization(MonitoredTask) after
356    * the master becomes the active one.
357    *
358    * @throws InterruptedException
359    * @throws KeeperException
360    * @throws IOException
361    */
362   public HMaster(final Configuration conf, CoordinatedStateManager csm)
363       throws IOException, KeeperException, InterruptedException {
364     super(conf, csm);
365     this.rsFatals = new MemoryBoundedLogMessageBuffer(
366       conf.getLong("hbase.master.buffer.for.rs.fatals", 1*1024*1024));
367 
368     LOG.info("hbase.rootdir=" + FSUtils.getRootDir(this.conf) +
369         ", hbase.cluster.distributed=" + this.conf.getBoolean(HConstants.CLUSTER_DISTRIBUTED, false));
370 
371     // Disable usage of meta replicas in the master
372     this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
373 
374     Replication.decorateMasterConfiguration(this.conf);
375 
376     // Hack! Maps DFSClient => Master for logs.  HDFS made this
377     // config param for task trackers, but we can piggyback off of it.
378     if (this.conf.get("mapreduce.task.attempt.id") == null) {
379       this.conf.set("mapreduce.task.attempt.id", "hb_m_" + this.serverName.toString());
380     }
381 
382     // should we check the compression codec type at master side, default true, HBASE-6370
383     this.masterCheckCompression = conf.getBoolean("hbase.master.check.compression", true);
384 
385     // should we check encryption settings at master side, default true
386     this.masterCheckEncryption = conf.getBoolean("hbase.master.check.encryption", true);
387 
388     this.metricsMaster = new MetricsMaster( new MetricsMasterWrapperImpl(this));
389 
390     // Check configuration to see whether procedure is disabled (not execute at all),
391     // unused (not used to execute DDL, but executor starts to complete unfinished operations
392     // in procedure store, or enabled (default behavior).
393     String procedureConfString = conf.get("hbase.master.procedure.tableddl", "enabled");
394     if (procedureConfString.equalsIgnoreCase("disabled")) {
395       LOG.info("Master will use handler for new table DDL"
396         + " and all unfinished table DDLs in procedure store will be disgarded.");
397       this.procedureConf = ProcedureConf.PROCEDURE_FULLY_DISABLED;
398     } else if (procedureConfString.equalsIgnoreCase("unused")) {
399       LOG.info("Master will use handler for new table DDL"
400         + " and all unfinished table DDLs in procedure store will continue to execute.");
401       this.procedureConf = ProcedureConf.HANDLER_USED;
402     } else {
403       this.procedureConf = ProcedureConf.PROCEDURE_ENABLED;
404     }
405     // preload table descriptor at startup
406     this.preLoadTableDescriptors = conf.getBoolean("hbase.master.preload.tabledescriptors", true);
407 
408     // Do we publish the status?
409 
410     boolean shouldPublish = conf.getBoolean(HConstants.STATUS_PUBLISHED,
411         HConstants.STATUS_PUBLISHED_DEFAULT);
412     Class<? extends ClusterStatusPublisher.Publisher> publisherClass =
413         conf.getClass(ClusterStatusPublisher.STATUS_PUBLISHER_CLASS,
414             ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS,
415             ClusterStatusPublisher.Publisher.class);
416 
417     if (shouldPublish) {
418       if (publisherClass == null) {
419         LOG.warn(HConstants.STATUS_PUBLISHED + " is true, but " +
420             ClusterStatusPublisher.DEFAULT_STATUS_PUBLISHER_CLASS +
421             " is not set - not publishing status");
422       } else {
423         clusterStatusPublisherChore = new ClusterStatusPublisher(this, conf, publisherClass);
424         getChoreService().scheduleChore(clusterStatusPublisherChore);
425       }
426     }
427 
428     // Some unit tests don't need a cluster, so no zookeeper at all
429     if (!conf.getBoolean("hbase.testing.nocluster", false)) {
430       activeMasterManager = new ActiveMasterManager(zooKeeper, this.serverName, this);
431       int infoPort = putUpJettyServer();
432       startActiveMasterManager(infoPort);
433     } else {
434       activeMasterManager = null;
435     }
436   }

438   // return the actual infoPort, -1 means disable info server.
439   private int putUpJettyServer() throws IOException {
440     if (!conf.getBoolean("hbase.master.infoserver.redirect", true)) {
441       return -1;
442     }
443     int infoPort = conf.getInt("hbase.master.info.port.orig",
444       HConstants.DEFAULT_MASTER_INFOPORT);
445     // -1 is for disabling info server, so no redirecting
446     if (infoPort < 0 || infoServer == null) {
447       return -1;
448     }
449     String addr = conf.get("hbase.master.info.bindAddress", "0.0.0.0");
450     if (!Addressing.isLocalAddress(InetAddress.getByName(addr))) {
451       String msg =
452           "Failed to start redirecting jetty server. Address " + addr
453               + " does not belong to this host. Correct configuration parameter: "
454               + "hbase.master.info.bindAddress";
455       LOG.error(msg);
456       throw new IOException(msg);
457     }
458 
459     RedirectServlet.regionServerInfoPort = infoServer.getPort();
460     if(RedirectServlet.regionServerInfoPort == infoPort) {
461       return infoPort;
462     }
463     masterJettyServer = new org.mortbay.jetty.Server();
464     Connector connector = new SelectChannelConnector();
465     connector.setHost(addr);
466     connector.setPort(infoPort);
467     masterJettyServer.addConnector(connector);
468     masterJettyServer.setStopAtShutdown(true);
469     Context context = new Context(masterJettyServer, "/", Context.NO_SESSIONS);
470     context.addServlet(RedirectServlet.class, "/*");
471     try {
472       masterJettyServer.start();
473     } catch (Exception e) {
474       throw new IOException("Failed to start redirecting jetty server", e);
475     }
476     return connector.getLocalPort();
477   }

612   /**
613    * Finish initialization of HMaster after becoming the primary master.
614    *
615    * <ol>
616    * <li>Initialize master components - file system manager, server manager,
617    *     assignment manager, region server tracker, etc</li>
618    * <li>Start necessary service threads - balancer, catalog janior,
619    *     executor services, etc</li>
620    * <li>Set cluster as UP in ZooKeeper</li>
621    * <li>Wait for RegionServers to check-in</li>
622    * <li>Split logs and perform data recovery, if necessary</li>
623    * <li>Ensure assignment of meta/namespace regions<li>
624    * <li>Handle either fresh cluster start or master failover</li>
625    * </ol>
626    *
627    * @throws IOException
628    * @throws InterruptedException
629    * @throws KeeperException
630    * @throws CoordinatedStateException
631    */
632   private void finishActiveMasterInitialization(MonitoredTask status)
633       throws IOException, InterruptedException, KeeperException, CoordinatedStateException {
634 
635     isActiveMaster = true;
636     Thread zombieDetector = new Thread(new InitializationMonitor(this));
637     zombieDetector.start();
638 
639     /*
640      * We are active master now... go initialize components we need to run.
641      * Note, there may be dross in zk from previous runs; it'll get addressed
642      * below after we determine if cluster startup or failover.
643      */
644 
645     status.setStatus("Initializing Master file system");
646 
647     this.masterActiveTime = System.currentTimeMillis();
648     // TODO: Do this using Dependency Injection, using PicoContainer, Guice or Spring.
649     this.fileSystemManager = new MasterFileSystem(this, this);
650 
651     // enable table descriptors cache
652     this.tableDescriptors.setCacheOn();
653     // set the META's descriptor to the correct replication
654     this.tableDescriptors.get(TableName.META_TABLE_NAME).setRegionReplication(
655         conf.getInt(HConstants.META_REPLICAS_NUM, HConstants.DEFAULT_META_REPLICA_NUM));
656     // warm-up HTDs cache on master initialization
657     if (preLoadTableDescriptors) {
658       status.setStatus("Pre-loading table descriptors");
659       this.tableDescriptors.getAll();
660     }
661 
662     // publish cluster ID
663     status.setStatus("Publishing Cluster ID in ZooKeeper");
664     ZKClusterId.setClusterId(this.zooKeeper, fileSystemManager.getClusterId());
665     this.serverManager = createServerManager(this, this);
666 
667     setupClusterConnection();
668 
669     // Invalidate all write locks held previously
670     this.tableLockManager.reapWriteLocks();
671 
672     status.setStatus("Initializing ZK system trackers");
673     initializeZKBasedSystemTrackers();
674 
675     // initialize master side coprocessors before we start handling requests
676     status.setStatus("Initializing master coprocessors");
677     this.cpHost = new MasterCoprocessorHost(this, this.conf);
678 
679     // start up all service threads.
680     status.setStatus("Initializing master service threads");
681     startServiceThreads();
682 
683     // Wake up this server to check in
684     sleeper.skipSleepCycle();
685 
686     // Wait for region servers to report in
687     this.serverManager.waitForRegionServers(status);
688     // Check zk for region servers that are up but didn't register
689     for (ServerName sn: this.regionServerTracker.getOnlineServers()) {
690       // The isServerOnline check is opportunistic, correctness is handled inside
691       if (!this.serverManager.isServerOnline(sn)
692           && serverManager.checkAndRecordNewServer(sn, ServerLoad.EMPTY_SERVERLOAD)) {
693         LOG.info("Registered server found up in zk but who has not yet reported in: " + sn);
694       }
695     }
696 
697     // get a list for previously failed RS which need log splitting work
698     // we recover hbase:meta region servers inside master initialization and
699     // handle other failed servers in SSH in order to start up master node ASAP
700     Set<ServerName> previouslyFailedServers = this.fileSystemManager
701         .getFailedServersFromLogFolders();
702 
703     // remove stale recovering regions from previous run
704     this.fileSystemManager.removeStaleRecoveringRegionsFromZK(previouslyFailedServers);
705 
706     // log splitting for hbase:meta server
707     ServerName oldMetaServerLocation = metaTableLocator.getMetaRegionLocation(this.getZooKeeper());
708     if (oldMetaServerLocation != null && previouslyFailedServers.contains(oldMetaServerLocation)) {
709       splitMetaLogBeforeAssignment(oldMetaServerLocation);
710       // Note: we can't remove oldMetaServerLocation from previousFailedServers list because it
711       // may also host user regions
712     }
713     Set<ServerName> previouslyFailedMetaRSs = getPreviouselyFailedMetaServersFromZK();
714     // need to use union of previouslyFailedMetaRSs recorded in ZK and previouslyFailedServers
715     // instead of previouslyFailedMetaRSs alone to address the following two situations:
716     // 1) the chained failure situation(recovery failed multiple times in a row).
717     // 2) master get killed right before it could delete the recovering hbase:meta from ZK while the
718     // same server still has non-meta wals to be replayed so that
719     // removeStaleRecoveringRegionsFromZK can't delete the stale hbase:meta region
720     // Passing more servers into splitMetaLog is all right. If a server doesn't have hbase:meta wal,
721     // there is no op for the server.
722     previouslyFailedMetaRSs.addAll(previouslyFailedServers);
723 
724     this.initializationBeforeMetaAssignment = true;
725 
726     // Wait for regionserver to finish initialization.
727     if (BaseLoadBalancer.tablesOnMaster(conf)) {
728       waitForServerOnline();
729     }
730 
731     //initialize load balancer
732     this.balancer.setClusterStatus(getClusterStatus());
733     this.balancer.setMasterServices(this);
734     this.balancer.initialize();
735 
736     // Check if master is shutting down because of some issue
737     // in initializing the regionserver or the balancer.
738     if(isStopped()) return;
739 
740     // Make sure meta assigned before proceeding.
741     status.setStatus("Assigning Meta Region");
742     assignMeta(status, previouslyFailedMetaRSs, HRegionInfo.DEFAULT_REPLICA_ID);
743     // check if master is shutting down because above assignMeta could return even hbase:meta isn't
744     // assigned when master is shutting down
745     if(isStopped()) return;
746 
747     status.setStatus("Submitting log splitting work for previously failed region servers");
748     // Master has recovered hbase:meta region server and we put
749     // other failed region servers in a queue to be handled later by SSH
750     for (ServerName tmpServer : previouslyFailedServers) {
751       this.serverManager.processDeadServer(tmpServer, true);
752     }
753 
754     // Update meta with new PB serialization if required. i.e migrate all HRI to PB serialization
755     // in meta. This must happen before we assign all user regions or else the assignment will
756     // fail.
757     if (this.conf.getBoolean("hbase.MetaMigrationConvertingToPB", true)) {
758       MetaMigrationConvertingToPB.updateMetaIfNecessary(this);
759     }
760 
761     // Fix up assignment manager status
762     status.setStatus("Starting assignment manager");
763     this.assignmentManager.joinCluster();
764 
765     //set cluster status again after user regions are assigned
766     this.balancer.setClusterStatus(getClusterStatus());
767 
768     // Start balancer and meta catalog janitor after meta and regions have
769     // been assigned.
770     status.setStatus("Starting balancer and catalog janitor");
771     this.clusterStatusChore = new ClusterStatusChore(this, balancer);
772     getChoreService().scheduleChore(clusterStatusChore);
773     this.balancerChore = new BalancerChore(this);
774     getChoreService().scheduleChore(balancerChore);
775     this.catalogJanitorChore = new CatalogJanitor(this, this);
776     getChoreService().scheduleChore(catalogJanitorChore);
777 
778     status.setStatus("Starting namespace manager");
779     initNamespace();
780     
781     if (this.cpHost != null) {
782       try {
783         this.cpHost.preMasterInitialization();
784       } catch (IOException e) {
785         LOG.error("Coprocessor preMasterInitialization() hook failed", e);
786       }
787     }
788 
789     status.markComplete("Initialization successful");
790     LOG.info("Master has completed initialization");
791     configurationManager.registerObserver(this.balancer);
792     initialized = true;
793     
794     status.setStatus("Starting quota manager");
795     initQuotaManager();
796 
797     // assign the meta replicas
798     Set<ServerName> EMPTY_SET = new HashSet<ServerName>();
799     int numReplicas = conf.getInt(HConstants.META_REPLICAS_NUM,
800            HConstants.DEFAULT_META_REPLICA_NUM);
801     for (int i = 1; i < numReplicas; i++) {
802       assignMeta(status, EMPTY_SET, i);
803     }
804     unassignExcessMetaReplica(zooKeeper, numReplicas);
805 
806     // clear the dead servers with same host name and port of online server because we are not
807     // removing dead server with same hostname and port of rs which is trying to check in before
808     // master initialization. See HBASE-5916.
809     this.serverManager.clearDeadServersWithSameHostNameAndPortOfOnlineServer();
810 
811     if (this.cpHost != null) {
812       // don't let cp initialization errors kill the master
813       try {
814         this.cpHost.postStartMaster();
815       } catch (IOException ioe) {
816         LOG.error("Coprocessor postStartMaster() hook failed", ioe);
817       }
818     }
819 
820     zombieDetector.interrupt();
821   }

845   private void unassignExcessMetaReplica(ZooKeeperWatcher zkw, int numMetaReplicasConfigured) {
846     // unassign the unneeded replicas (for e.g., if the previous master was configured
847     // with a replication of 3 and now it is 2, we need to unassign the 1 unneeded replica)
848     try {
849       List<String> metaReplicaZnodes = zooKeeper.getMetaReplicaNodes();
850       for (String metaReplicaZnode : metaReplicaZnodes) {
851         int replicaId = zooKeeper.getMetaReplicaIdFromZnode(metaReplicaZnode);
852         if (replicaId >= numMetaReplicasConfigured) {
853           RegionState r = MetaTableLocator.getMetaRegionState(zkw, replicaId);
854           LOG.info("Closing excess replica of meta region " + r.getRegion());
855           // send a close and wait for a max of 30 seconds
856           ServerManager.closeRegionSilentlyAndWait(getConnection(), r.getServerName(),
857               r.getRegion(), 30000);
858           ZKUtil.deleteNode(zkw, zkw.getZNodeForReplica(replicaId));
859         }
860       }
861     } catch (Exception ex) {
862       // ignore the exception since we don't want the master to be wedged due to potential
863       // issues in the cleanup of the extra regions. We can do that cleanup via hbck or manually
864       LOG.warn("Ignoring exception " + ex);
865     }
866   }

868   /**
869    * Check <code>hbase:meta</code> is assigned. If not, assign it.
870    * @param status MonitoredTask
871    * @param previouslyFailedMetaRSs
872    * @param replicaId
873    * @throws InterruptedException
874    * @throws IOException
875    * @throws KeeperException
876    */
877   void assignMeta(MonitoredTask status, Set<ServerName> previouslyFailedMetaRSs, int replicaId)
878       throws InterruptedException, IOException, KeeperException {
879     // Work on meta region
880     int assigned = 0;
881     long timeout = this.conf.getLong("hbase.catalog.verification.timeout", 1000);
882     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
883       status.setStatus("Assigning hbase:meta region");
884     } else {
885       status.setStatus("Assigning hbase:meta region, replicaId " + replicaId);
886     }
887     // Get current meta state from zk.
888     RegionStates regionStates = assignmentManager.getRegionStates();
889     RegionState metaState = MetaTableLocator.getMetaRegionState(getZooKeeper(), replicaId);
890     HRegionInfo hri = RegionReplicaUtil.getRegionInfoForReplica(HRegionInfo.FIRST_META_REGIONINFO,
891         replicaId);
892     ServerName currentMetaServer = metaState.getServerName();
893     if (!ConfigUtil.useZKForAssignment(conf)) {
894       regionStates.createRegionState(hri, metaState.getState(),
895         currentMetaServer, null);
896     } else {
897       regionStates.createRegionState(hri);
898     }
899     boolean rit = this.assignmentManager.
900       processRegionInTransitionAndBlockUntilAssigned(hri);
901     boolean metaRegionLocation = metaTableLocator.verifyMetaRegionLocation(
902       this.getConnection(), this.getZooKeeper(), timeout, replicaId);
903     if (!metaRegionLocation || !metaState.isOpened()) {
904       // Meta location is not verified. It should be in transition, or offline.
905       // We will wait for it to be assigned in enableSSHandWaitForMeta below.
906       assigned++;
907       if (!ConfigUtil.useZKForAssignment(conf)) {
908         assignMetaZkLess(regionStates, metaState, timeout, previouslyFailedMetaRSs);
909       } else if (!rit) {
910         // Assign meta since not already in transition
911         if (currentMetaServer != null) {
912           // If the meta server is not known to be dead or online,
913           // just split the meta log, and don't expire it since this
914           // could be a full cluster restart. Otherwise, we will think
915           // this is a failover and lose previous region locations.
916           // If it is really a failover case, AM will find out in rebuilding
917           // user regions. Otherwise, we are good since all logs are split
918           // or known to be replayed before user regions are assigned.
919           if (serverManager.isServerOnline(currentMetaServer)) {
920             LOG.info("Forcing expire of " + currentMetaServer);
921             serverManager.expireServer(currentMetaServer);
922           }
923           if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) {
924             splitMetaLogBeforeAssignment(currentMetaServer);
925             previouslyFailedMetaRSs.add(currentMetaServer);
926           }
927         }
928         assignmentManager.assignMeta(hri);
929       }
930     } else {
931       // Region already assigned. We didn't assign it. Add to in-memory state.
932       regionStates.updateRegionState(
933         HRegionInfo.FIRST_META_REGIONINFO, State.OPEN, currentMetaServer);
934       this.assignmentManager.regionOnline(
935         HRegionInfo.FIRST_META_REGIONINFO, currentMetaServer);
936     }
937 
938     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableMeta(TableName.META_TABLE_NAME);
939 
940     if ((RecoveryMode.LOG_REPLAY == this.getMasterFileSystem().getLogRecoveryMode())
941         && (!previouslyFailedMetaRSs.isEmpty())) {
942       // replay WAL edits mode need new hbase:meta RS is assigned firstly
943       status.setStatus("replaying log for Meta Region");
944       this.fileSystemManager.splitMetaLog(previouslyFailedMetaRSs);
945     }
946 
947     // Make sure a hbase:meta location is set. We need to enable SSH here since
948     // if the meta region server is died at this time, we need it to be re-assigned
949     // by SSH so that system tables can be assigned.
950     // No need to wait for meta is assigned = 0 when meta is just verified.
951     if (replicaId == HRegionInfo.DEFAULT_REPLICA_ID) enableServerShutdownHandler(assigned != 0);
952     LOG.info("hbase:meta with replicaId " + replicaId + " assigned=" + assigned + ", rit=" + rit +
953       ", location=" + metaTableLocator.getMetaRegionLocation(this.getZooKeeper(), replicaId));
954     status.setStatus("META assigned.");
955   }

957   private void assignMetaZkLess(RegionStates regionStates, RegionState regionState, long timeout,
958       Set<ServerName> previouslyFailedRs) throws IOException, KeeperException {
959     ServerName currentServer = regionState.getServerName();
960     if (serverManager.isServerOnline(currentServer)) {
961       LOG.info("Meta was in transition on " + currentServer);
962       assignmentManager.processRegionInTransitionZkLess();
963     } else {
964       if (currentServer != null) {
965         if (regionState.getRegion().getReplicaId() == HRegionInfo.DEFAULT_REPLICA_ID) {
966           splitMetaLogBeforeAssignment(currentServer);
967           regionStates.logSplit(HRegionInfo.FIRST_META_REGIONINFO);
968           previouslyFailedRs.add(currentServer);
969         }
970       }
971       LOG.info("Re-assigning hbase:meta, it was on " + currentServer);
972       regionStates.updateRegionState(regionState.getRegion(), State.OFFLINE);
973       assignmentManager.assignMeta(regionState.getRegion());
974     }
975   }

1062    * Start up all services. If any of these threads gets an unhandled exception
1063    * then they just die with a logged message.  This should be fine because
1064    * in general, we do not expect the master to get such unhandled exceptions
1065    *  as OOMEs; it should be lightly loaded. See what HRegionServer does if

1068   private void startServiceThreads() throws IOException{
1069    // Start the executor service pools
1070    this.service.startExecutorService(ExecutorType.MASTER_OPEN_REGION,
1071       conf.getInt("hbase.master.executor.openregion.threads", 5));
1072    this.service.startExecutorService(ExecutorType.MASTER_CLOSE_REGION,
1073       conf.getInt("hbase.master.executor.closeregion.threads", 5));
1074    this.service.startExecutorService(ExecutorType.MASTER_SERVER_OPERATIONS,
1075       conf.getInt("hbase.master.executor.serverops.threads", 5));
1076    this.service.startExecutorService(ExecutorType.MASTER_META_SERVER_OPERATIONS,
1077       conf.getInt("hbase.master.executor.serverops.threads", 5));
1078    this.service.startExecutorService(ExecutorType.M_LOG_REPLAY_OPS,
1079       conf.getInt("hbase.master.executor.logreplayops.threads", 10));
1080 
1081    // We depend on there being only one instance of this executor running
1082    // at a time.  To do concurrency, would need fencing of enable/disable of
1083    // tables.
1084    // Any time changing this maxThreads to > 1, pls see the comment at
1085    // AccessController#postCreateTableHandler
1086    this.service.startExecutorService(ExecutorType.MASTER_TABLE_OPERATIONS, 1);
1087    startProcedureExecutor();
1088 
1089    // Start log cleaner thread
1090    int cleanerInterval = conf.getInt("hbase.master.cleaner.interval", 60 * 1000);
1091    this.logCleaner =
1092       new LogCleaner(cleanerInterval,
1093          this, conf, getMasterFileSystem().getFileSystem(),
1094          getMasterFileSystem().getOldLogDir());
1095     getChoreService().scheduleChore(logCleaner);
1096 
1097    //start the hfile archive cleaner thread
1098     Path archiveDir = HFileArchiveUtil.getArchivePath(conf);
1099     this.hfileCleaner = new HFileCleaner(cleanerInterval, this, conf, getMasterFileSystem()
1100         .getFileSystem(), archiveDir);
1101     getChoreService().scheduleChore(hfileCleaner);
1102     serviceStarted = true;
1103     if (LOG.isTraceEnabled()) {
1104       LOG.trace("Started service threads");
1105     }
1106   }

1254   public boolean balance() throws IOException {
1255     // if master not initialized, don't run balancer.
1256     if (!this.initialized) {
1257       LOG.debug("Master has not been initialized, don't run balancer.");
1258       return false;
1259     }
1260     // Do this call outside of synchronized block.
1261     int maximumBalanceTime = getBalancerCutoffTime();
1262     synchronized (this.balancer) {
1263       // If balance not true, don't run balancer.
1264       if (!this.loadBalancerTracker.isBalancerOn()) return false;
1265       // Only allow one balance run at at time.
1266       if (this.assignmentManager.getRegionStates().isRegionsInTransition()) {
1267         Map<String, RegionState> regionsInTransition =
1268           this.assignmentManager.getRegionStates().getRegionsInTransition();
1269         LOG.debug("Not running balancer because " + regionsInTransition.size() +
1270           " region(s) in transition: " + org.apache.commons.lang.StringUtils.
1271             abbreviate(regionsInTransition.toString(), 256));
1272         return false;
1273       }
1274       if (this.serverManager.areDeadServersInProgress()) {
1275         LOG.debug("Not running balancer because processing dead regionserver(s): " +
1276           this.serverManager.getDeadServers());
1277         return false;
1278       }
1279 
1280       if (this.cpHost != null) {
1281         try {
1282           if (this.cpHost.preBalance()) {
1283             LOG.debug("Coprocessor bypassing balancer request");
1284             return false;
1285           }
1286         } catch (IOException ioe) {
1287           LOG.error("Error invoking master coprocessor preBalance()", ioe);
1288           return false;
1289         }
1290       }
1291 
1292       Map<TableName, Map<ServerName, List<HRegionInfo>>> assignmentsByTable =
1293         this.assignmentManager.getRegionStates().getAssignmentsByTable();
1294 
1295       List<RegionPlan> plans = new ArrayList<RegionPlan>();
1296       //Give the balancer the current cluster state.
1297       this.balancer.setClusterStatus(getClusterStatus());
1298       for (Map<ServerName, List<HRegionInfo>> assignments : assignmentsByTable.values()) {
1299         List<RegionPlan> partialPlans = this.balancer.balanceCluster(assignments);
1300         if (partialPlans != null) plans.addAll(partialPlans);
1301       }
1302       long cutoffTime = System.currentTimeMillis() + maximumBalanceTime;
1303       int rpCount = 0;  // number of RegionPlans balanced so far
1304       long totalRegPlanExecTime = 0;
1305       if (plans != null && !plans.isEmpty()) {
1306         for (RegionPlan plan: plans) {
1307           LOG.info("balance " + plan);
1308           long balStartTime = System.currentTimeMillis();
1309           //TODO: bulk assign
1310           this.assignmentManager.balance(plan);
1311           totalRegPlanExecTime += System.currentTimeMillis()-balStartTime;
1312           rpCount++;
1313           if (rpCount < plans.size() &&
1314               // if performing next balance exceeds cutoff time, exit the loop
1315               (System.currentTimeMillis() + (totalRegPlanExecTime / rpCount)) > cutoffTime) {
1316             //TODO: After balance, there should not be a cutoff time (keeping it as a security net for now)
1317             LOG.debug("No more balancing till next balance run; maximumBalanceTime=" +
1318               maximumBalanceTime);
1319             break;
1320           }
1321         }
1322       }
1323       if (this.cpHost != null) {
1324         try {
1325           this.cpHost.postBalance(rpCount < plans.size() ? plans.subList(0, rpCount) : plans);
1326         } catch (IOException ioe) {
1327           // balancing already succeeded so don't change the result
1328           LOG.error("Error invoking master coprocessor postBalance()", ioe);
1329         }
1330       }
1331     }
1332     // If LoadBalancer did not generate any plans, it means the cluster is already balanced.
1333     // Return true indicating a success.
1334     return true;
1335   }

1362   void move(final byte[] encodedRegionName,
1363       final byte[] destServerName) throws HBaseIOException {
1364     RegionState regionState = assignmentManager.getRegionStates().
1365       getRegionState(Bytes.toString(encodedRegionName));
1366     if (regionState == null) {
1367       throw new UnknownRegionException(Bytes.toStringBinary(encodedRegionName));
1368     }
1369 
1370     HRegionInfo hri = regionState.getRegion();
1371     ServerName dest;
1372     if (destServerName == null || destServerName.length == 0) {
1373       LOG.info("Passed destination servername is null/empty so " +
1374         "choosing a server at random");
1375       final List<ServerName> destServers = this.serverManager.createDestinationServersList(
1376         regionState.getServerName());
1377       dest = balancer.randomAssignment(hri, destServers);
1378       if (dest == null) {
1379         LOG.debug("Unable to determine a plan to assign " + hri);
1380         return;
1381       }
1382     } else {
1383       dest = ServerName.valueOf(Bytes.toString(destServerName));
1384       if (dest.equals(serverName) && balancer instanceof BaseLoadBalancer
1385           && !((BaseLoadBalancer)balancer).shouldBeOnMaster(hri)) {
1386         // To avoid unnecessary region moving later by balancer. Don't put user
1387         // regions on master. Regions on master could be put on other region
1388         // server intentionally by test however.
1389         LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1390           + " to avoid unnecessary region moving later by load balancer,"
1391           + " because it should not be on master");
1392         return;
1393       }
1394     }
1395 
1396     if (dest.equals(regionState.getServerName())) {
1397       LOG.debug("Skipping move of region " + hri.getRegionNameAsString()
1398         + " because region already assigned to the same server " + dest + ".");
1399       return;
1400     }
1401 
1402     // Now we can do the move
1403     RegionPlan rp = new RegionPlan(hri, regionState.getServerName(), dest);
1404 
1405     try {
1406       checkInitialized();
1407       if (this.cpHost != null) {
1408         if (this.cpHost.preMove(hri, rp.getSource(), rp.getDestination())) {
1409           return;
1410         }
1411       }
1412       // warmup the region on the destination before initiating the move. this call
1413       // is synchronous and takes some time. doing it before the source region gets
1414       // closed
1415       serverManager.sendRegionWarmup(rp.getDestination(), hri);
1416 
1417       LOG.info(getClientIdAuditPrefix() + " move " + rp + ", running balancer");
1418       this.assignmentManager.balance(rp);
1419       if (this.cpHost != null) {
1420         this.cpHost.postMove(hri, rp.getSource(), rp.getDestination());
1421       }
1422     } catch (IOException ioe) {
1423       if (ioe instanceof HBaseIOException) {
1424         throw (HBaseIOException)ioe;
1425       }
1426       throw new HBaseIOException(ioe);
1427     }
1428   }

1477   /**
1478    * Checks whether the table conforms to some sane limits, and configured
1479    * values (compression, etc) work. Throws an exception if something is wrong.
1480    * @throws IOException
1481    */
1482   private void sanityCheckTableDescriptor(final HTableDescriptor htd) throws IOException {
1483     final String CONF_KEY = "hbase.table.sanity.checks";
1484     boolean logWarn = false;
1485     if (!conf.getBoolean(CONF_KEY, true)) {
1486       logWarn = true;
1487     }
1488     String tableVal = htd.getConfigurationValue(CONF_KEY);
1489     if (tableVal != null && !Boolean.valueOf(tableVal)) {
1490       logWarn = true;
1491     }
1492 
1493     // check max file size
1494     long maxFileSizeLowerLimit = 2 * 1024 * 1024L; // 2M is the default lower limit
1495     long maxFileSize = htd.getMaxFileSize();
1496     if (maxFileSize < 0) {
1497       maxFileSize = conf.getLong(HConstants.HREGION_MAX_FILESIZE, maxFileSizeLowerLimit);
1498     }
1499     if (maxFileSize < conf.getLong("hbase.hregion.max.filesize.limit", maxFileSizeLowerLimit)) {
1500       String message = "MAX_FILESIZE for table descriptor or "
1501           + "\"hbase.hregion.max.filesize\" (" + maxFileSize
1502           + ") is too small, which might cause over splitting into unmanageable "
1503           + "number of regions.";
1504       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1505     }
1506 
1507     // check flush size
1508     long flushSizeLowerLimit = 1024 * 1024L; // 1M is the default lower limit
1509     long flushSize = htd.getMemStoreFlushSize();
1510     if (flushSize < 0) {
1511       flushSize = conf.getLong(HConstants.HREGION_MEMSTORE_FLUSH_SIZE, flushSizeLowerLimit);
1512     }
1513     if (flushSize < conf.getLong("hbase.hregion.memstore.flush.size.limit", flushSizeLowerLimit)) {
1514       String message = "MEMSTORE_FLUSHSIZE for table descriptor or "
1515           + "\"hbase.hregion.memstore.flush.size\" ("+flushSize+") is too small, which might cause"
1516           + " very frequent flushing.";
1517       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1518     }
1519 
1520     // check that coprocessors and other specified plugin classes can be loaded
1521     try {
1522       checkClassLoading(conf, htd);
1523     } catch (Exception ex) {
1524       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, ex.getMessage(), null);
1525     }
1526 
1527     // check compression can be loaded
1528     try {
1529       checkCompression(htd);
1530     } catch (IOException e) {
1531       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1532     }
1533 
1534     // check encryption can be loaded
1535     try {
1536       checkEncryption(conf, htd);
1537     } catch (IOException e) {
1538       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, e.getMessage(), e);
1539     }
1540 
1541     // check that we have at least 1 CF
1542     if (htd.getColumnFamilies().length == 0) {
1543       String message = "Table should have at least one column family.";
1544       warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1545     }
1546 
1547     for (HColumnDescriptor hcd : htd.getColumnFamilies()) {
1548       if (hcd.getTimeToLive() <= 0) {
1549         String message = "TTL for column family " + hcd.getNameAsString() + " must be positive.";
1550         warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1551       }
1552 
1553       // check blockSize
1554       if (hcd.getBlocksize() < 1024 || hcd.getBlocksize() > 16 * 1024 * 1024) {
1555         String message = "Block size for column family " + hcd.getNameAsString()
1556             + "  must be between 1K and 16MB.";
1557         warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1558       }
1559 
1560       // check versions
1561       if (hcd.getMinVersions() < 0) {
1562         String message = "Min versions for column family " + hcd.getNameAsString()
1563           + "  must be positive.";
1564         warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1565       }
1566       // max versions already being checked
1567 
1568       // check replication scope
1569       if (hcd.getScope() < 0) {
1570         String message = "Replication scope for column family "
1571           + hcd.getNameAsString() + "  must be positive.";
1572         warnOrThrowExceptionForFailure(logWarn, CONF_KEY, message, null);
1573       }
1574 
1575       // TODO: should we check coprocessors and encryption ?
1576     }
1577   }

1589   private void startActiveMasterManager(int infoPort) throws KeeperException {
1590     String backupZNode = ZKUtil.joinZNode(
1591       zooKeeper.backupMasterAddressesZNode, serverName.toString());
1592     /*
1593     * Add a ZNode for ourselves in the backup master directory since we
1594     * may not become the active master. If so, we want the actual active
1595     * master to know we are backup masters, so that it won't assign
1596     * regions to us if so configured.
1597     *
1598     * If we become the active master later, ActiveMasterManager will delete
1599     * this node explicitly.  If we crash before then, ZooKeeper will delete
1600     * this node for us since it is ephemeral.
1601     */
1602     LOG.info("Adding backup master ZNode " + backupZNode);
1603     if (!MasterAddressTracker.setMasterAddress(zooKeeper, backupZNode,
1604         serverName, infoPort)) {
1605       LOG.warn("Failed create of " + backupZNode + " by " + serverName);
1606     }
1607 
1608     activeMasterManager.setInfoPort(infoPort);
1609     // Start a thread to try to become the active master, so we won't block here
1610     Threads.setDaemonThreadRunning(new Thread(new Runnable() {
1611       @Override
1612       public void run() {
1613         int timeout = conf.getInt(HConstants.ZK_SESSION_TIMEOUT,
1614           HConstants.DEFAULT_ZK_SESSION_TIMEOUT);
1615         // If we're a backup master, stall until a primary to writes his address
1616         if (conf.getBoolean(HConstants.MASTER_TYPE_BACKUP,
1617             HConstants.DEFAULT_MASTER_TYPE_BACKUP)) {
1618           LOG.debug("HMaster started in backup mode. "
1619             + "Stalling until master znode is written.");
1620           // This will only be a minute or so while the cluster starts up,
1621           // so don't worry about setting watches on the parent znode
1622           while (!activeMasterManager.hasActiveMaster()) {
1623             LOG.debug("Waiting for master address ZNode to be written "
1624               + "(Also watching cluster state node)");
1625             Threads.sleep(timeout);
1626           }
1627         }
1628         MonitoredTask status = TaskMonitor.get().createStatus("Master startup");
1629         status.setDescription("Master startup");
1630         try {
1631           if (activeMasterManager.blockUntilBecomingActiveMaster(timeout, status)) {
1632             finishActiveMasterInitialization(status);
1633           }
1634         } catch (Throwable t) {
1635           status.setStatus("Failed to become active: " + t.getMessage());
1636           LOG.fatal("Failed to become active master", t);
1637           // HBASE-5680: Likely hadoop23 vs hadoop 20.x/1.x incompatibility
1638           if (t instanceof NoClassDefFoundError &&
1639               t.getMessage().contains("org/apache/hadoop/hdfs/protocol/FSConstants$SafeModeAction")) {
1640             // improved error message for this special case
1641             abort("HBase is having a problem with its Hadoop jars.  You may need to "
1642               + "recompile HBase against Hadoop version "
1643               +  org.apache.hadoop.util.VersionInfo.getVersion()
1644               + " or change your hadoop jars to start properly", t);
1645           } else {
1646             abort("Unhandled exception. Starting shutdown.", t);
1647           }
1648         } finally {
1649           status.cleanup();
1650         }
1651       }
1652     }, getServerName().toShortString() + ".activeMasterManager"));
1653   }

1971     List<String> backupMasterStrings;
1972     try {
1973       backupMasterStrings = ZKUtil.listChildrenNoWatch(this.zooKeeper,
1974         this.zooKeeper.backupMasterAddressesZNode);

1987             bytes = ZKUtil.getData(this.zooKeeper, ZKUtil.joinZNode(

2528   /**
2529    * Removes the table descriptors that don't match the pattern.
2530    * @param descriptors list of table descriptors to filter
2531    * @param pattern the regex to use
2532    */
2533   private static void filterTablesByRegex(final Collection<HTableDescriptor> descriptors,
2534       final Pattern pattern) {
2535     final String defaultNS = NamespaceDescriptor.DEFAULT_NAMESPACE_NAME_STR;
2536     Iterator<HTableDescriptor> itr = descriptors.iterator();
2537     while (itr.hasNext()) {
2538       HTableDescriptor htd = itr.next();
2539       String tableName = htd.getTableName().getNameAsString();
2540       boolean matched = pattern.matcher(tableName).matches();
2541       if (!matched && htd.getTableName().getNamespaceAsString().equals(defaultNS)) {
2542         matched = pattern.matcher(defaultNS + TableName.NAMESPACE_DELIM + tableName).matches();
2543       }
2544       if (!matched) {
2545         itr.remove();
2546       }
2547     }
2548   }
