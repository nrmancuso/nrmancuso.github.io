53  /**
54   * Information about a region. A region is a range of keys in the whole keyspace of a table, an
55   * identifier (a timestamp) for differentiating between subset ranges (after region split)
56   * and a replicaId for differentiating the instance for the same range and some status information
57   * about the region.
58   *
59   * The region has a unique name which consists of the following fields:
60   * <li> tableName   : The name of the table </li>
61   * <li> startKey    : The startKey for the region. </li>
62   * <li> regionId    : A timestamp when the region is created. </li>
63   * <li> replicaId   : An id starting from 0 to differentiate replicas of the same region range
64   * but hosted in separated servers. The same region range can be hosted in multiple locations.</li>
65   * <li> encodedName : An MD5 encoded string for the region name.</li>
66   *
67   * <br> Other than the fields in the region name, region info contains:
68   * <li> endKey      : the endKey for the region (exclusive) </li>
69   * <li> split       : Whether the region is split </li>
70   * <li> offline     : Whether the region is offline </li>
71   *
72   * In 0.98 or before, a list of table's regions would fully cover the total keyspace, and at any
73   * point in time, a row key always belongs to a single region, which is hosted in a single server.
74   * In 0.99+, a region can have multiple instances (called replicas), and thus a range (or row) can
75   * correspond to multiple HRegionInfo's. These HRI's share the same fields however except the
76   * replicaId field. If the replicaId is not set, it defaults to 0, which is compatible with the
77   * previous behavior of a range corresponding to 1 region.
78   */
79  @InterfaceAudience.Public
80  @InterfaceStability.Evolving
81  public class HRegionInfo implements Comparable<HRegionInfo> {
82    /*
83     * There are two versions associated with HRegionInfo: HRegionInfo.VERSION and
84     * HConstants.META_VERSION. HRegionInfo.VERSION indicates the data structure's versioning
85     * while HConstants.META_VERSION indicates the versioning of the serialized HRIs stored in
86     * the hbase:meta table.
87     *
88     * Pre-0.92:
89     *   HRI.VERSION == 0 and HConstants.META_VERSION does not exist
90      *  (is not stored at hbase:meta table)
91     *   HRegionInfo had an HTableDescriptor reference inside it.
92     *   HRegionInfo is serialized as Writable to hbase:meta table.
93     * For 0.92.x and 0.94.x:
94     *   HRI.VERSION == 1 and HConstants.META_VERSION == 0
95     *   HRI no longer has HTableDescriptor in it.
96     *   HRI is serialized as Writable to hbase:meta table.
97     * For 0.96.x:
98     *   HRI.VERSION == 1 and HConstants.META_VERSION == 1
99     *   HRI data structure is the same as 0.92 and 0.94
100    *   HRI is serialized as PB to hbase:meta table.
101    *
102    * Versioning of HRegionInfo is deprecated. HRegionInfo does protobuf
103    * serialization using RegionInfo class, which has it's own versioning.
104    */

109   /**
110    * The new format for a region name contains its encodedName at the end.
111    * The encoded name also serves as the directory name for the region
112    * in the filesystem.
113    *
114    * New region name format:
115    *    &lt;tablename>,,&lt;startkey>,&lt;regionIdTimestamp>.&lt;encodedName>.
116    * where,
117    *    &lt;encodedName> is a hex version of the MD5 hash of
118    *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
119    *
120    * The old region name format:
121    *    &lt;tablename>,&lt;startkey>,&lt;regionIdTimestamp>
122    * For region names in the old format, the encoded name is a 32-bit
123    * JenkinsHash integer value (in its decimal notation, string form).
124    *<p>
125    * **NOTE**
126    *
127    * The first hbase:meta region, and regions created by an older
128    * version of HBase (0.20 or prior) will continue to use the
129    * old region name format.
130    */
131 
132   /** Separator used to demarcate the encodedName in a region name
133    * in the new format. See description on new format above.
134    */
135   private static final int ENC_SEPARATOR = '.';
136   public  static final int MD5_HEX_LENGTH   = 32;

149   /**
150    * Does region name contain its encoded name?
151    * @param regionName region name
152    * @return boolean indicating if this a new format region
153    *         name which contains its encoded name.
154    */
155   private static boolean hasEncodedName(final byte[] regionName) {
156     // check if region name ends in ENC_SEPARATOR
157     if ((regionName.length >= 1)
158         && (regionName[regionName.length - 1] == ENC_SEPARATOR)) {
159       // region name is new format. it contains the encoded name.
160       return true;
161     }
162     return false;
163   }

165   /**
166    * @param regionName
167    * @return the encodedName
168    */
169   public static String encodeRegionName(final byte [] regionName) {
170     String encodedName;
171     if (hasEncodedName(regionName)) {
172       // region is in new format:
173       // <tableName>,<startKey>,<regionIdTimeStamp>/encodedName/
174       encodedName = Bytes.toString(regionName,
175           regionName.length - MD5_HEX_LENGTH - 1,
176           MD5_HEX_LENGTH);
177     } else {
178       // old format region name. First hbase:meta region also
179       // use this format.EncodedName is the JenkinsHash value.
180       int hashVal = Math.abs(JenkinsHash.getInstance().hash(regionName,
181         regionName.length, 0));
182       encodedName = String.valueOf(hashVal);
183     }
184     return encodedName;
185   }

342     this.offLine = false;
343     this.regionId = regionid;

346       throw new IllegalArgumentException("ReplicaId cannot be greater than" + MAX_REPLICA_ID);

351     this.split = split;
352     this.endKey = endKey == null? HConstants.EMPTY_END_ROW: endKey.clone();
353     this.startKey = startKey == null?
354       HConstants.EMPTY_START_ROW: startKey.clone();
355     this.tableName = tableName;
356     setHashCode();
357   }

441   /**
442    * Make a region name of passed parameters.
443    * @param tableName
444    * @param startKey Can be null
445    * @param id Region id (Usually timestamp from when region was created).
446    * @param replicaId
447    * @param newFormat should we create the region name in the new format
448    * @return Region name made of passed tableName, startKey, id and replicaId
449    */
450   public static byte [] createRegionName(final TableName tableName,
451       final byte [] startKey, final byte [] id, final int replicaId, boolean newFormat) {
452     int len = tableName.getName().length + 2 + id.length +
453         (startKey == null? 0: startKey.length);
454     if (newFormat) {
455       len += MD5_HEX_LENGTH + 2;
456     }
457     byte[] replicaIdBytes = null;
458     // Special casing: replicaId is only appended if replicaId is greater than
459     // 0. This is because all regions in meta would have to be migrated to the new
460     // name otherwise
461     if (replicaId > 0) {
462       // use string representation for replica id
463       replicaIdBytes = Bytes.toBytes(String.format(REPLICA_ID_FORMAT, replicaId));
464       len += 1 + replicaIdBytes.length;
465     }
466 
467     byte [] b = new byte [len];
468 
469     int offset = tableName.getName().length;
470     System.arraycopy(tableName.getName(), 0, b, 0, offset);
471     b[offset++] = HConstants.DELIMITER;
472     if (startKey != null && startKey.length > 0) {
473       System.arraycopy(startKey, 0, b, offset, startKey.length);
474       offset += startKey.length;
475     }
476     b[offset++] = HConstants.DELIMITER;
477     System.arraycopy(id, 0, b, offset, id.length);
478     offset += id.length;
479 
480     if (replicaIdBytes != null) {
481       b[offset++] = REPLICA_ID_DELIMITER;
482       System.arraycopy(replicaIdBytes, 0, b, offset, replicaIdBytes.length);
483       offset += replicaIdBytes.length;
484     }
485 
486     if (newFormat) {
487       //
488       // Encoded name should be built into the region name.
489       //
490       // Use the region name thus far (namely, <tablename>,<startKey>,<id>_<replicaId>)
491       // to compute a MD5 hash to be used as the encoded name, and append
492       // it to the byte buffer.
493       //
494       String md5Hash = MD5Hash.getMD5AsHex(b, 0, offset);
495       byte [] md5HashBytes = Bytes.toBytes(md5Hash);
496 
497       if (md5HashBytes.length != MD5_HEX_LENGTH) {
498         LOG.error("MD5-hash length mismatch: Expected=" + MD5_HEX_LENGTH +
499                   "; Got=" + md5HashBytes.length);
500       }
501 
502       // now append the bytes '.<encodedName>.' to the end
503       b[offset++] = ENC_SEPARATOR;
504       System.arraycopy(md5HashBytes, 0, b, offset, MD5_HEX_LENGTH);
505       offset += MD5_HEX_LENGTH;
506       b[offset++] = ENC_SEPARATOR;
507     }
508 
509     return b;
510   }

555   /**
556    * Separate elements of a regionName.
557    * @param regionName
558    * @return Array of byte[] containing tableName, startKey and id
559    * @throws IOException
560    */
561   public static byte [][] parseRegionName(final byte [] regionName)
562   throws IOException {
563     // Region name is of the format:
564     // tablename,startkey,regionIdTimestamp[_replicaId][.encodedName.]
565     // startkey can contain the delimiter (',') so we parse from the start and end
566 
567     // parse from start
568     int offset = -1;
569     for (int i = 0; i < regionName.length; i++) {
570       if (regionName[i] == HConstants.DELIMITER) {
571         offset = i;
572         break;
573       }
574     }
575     if (offset == -1) {
576       throw new IOException("Invalid regionName format: " + Bytes.toStringBinary(regionName));
577     }
578     byte[] tableName = new byte[offset];
579     System.arraycopy(regionName, 0, tableName, 0, offset);
580     offset = -1;
581 
582     int endOffset = regionName.length;
583     // check whether regionName contains encodedName
584     if (regionName.length > MD5_HEX_LENGTH + 2
585         && regionName[regionName.length-1] == ENC_SEPARATOR
586         && regionName[regionName.length-MD5_HEX_LENGTH-2] == ENC_SEPARATOR) {
587       endOffset = endOffset - MD5_HEX_LENGTH - 2;
588     }
589 
590     // parse from end
591     byte[] replicaId = null;
592     int idEndOffset = endOffset;
593     for (int i = endOffset - 1; i > 0; i--) {
594       if (regionName[i] == REPLICA_ID_DELIMITER) { //replicaId may or may not be present
595         replicaId = new byte[endOffset - i - 1];
596         System.arraycopy(regionName, i + 1, replicaId, 0,
597           endOffset - i - 1);
598         idEndOffset = i;
599         // do not break, continue to search for id
600       }
601       if (regionName[i] == HConstants.DELIMITER) {
602         offset = i;
603         break;
604       }
605     }
606     if (offset == -1) {
607       throw new IOException("Invalid regionName format: " + Bytes.toStringBinary(regionName));
608     }
609     byte [] startKey = HConstants.EMPTY_BYTE_ARRAY;
610     if(offset != tableName.length + 1) {
611       startKey = new byte[offset - tableName.length - 1];
612       System.arraycopy(regionName, tableName.length + 1, startKey, 0,
613           offset - tableName.length - 1);
614     }
615     byte [] id = new byte[idEndOffset - offset - 1];
616     System.arraycopy(regionName, offset + 1, id, 0,
617       idEndOffset - offset - 1);
618     byte [][] elements = new byte[replicaId == null ? 3 : 4][];
619     elements[0] = tableName;
620     elements[1] = startKey;
621     elements[2] = id;
622     if (replicaId != null) {
623       elements[3] = replicaId;
624     }
625 
626     return elements;
627   }

928   // Comparable
929   //
930 
931   @Override
932   public int compareTo(HRegionInfo o) {
933     if (o == null) {
934       return 1;
935     }
936 
937     // Are regions of same table?
938     int result = this.tableName.compareTo(o.tableName);
939     if (result != 0) {
940       return result;
941     }
942 
943     // Compare start keys.
944     result = Bytes.compareTo(this.startKey, o.startKey);
945     if (result != 0) {
946       return result;
947     }
948 
949     // Compare end keys.
950     result = Bytes.compareTo(this.endKey, o.endKey);
951 
952     if (result != 0) {
953       if (this.getStartKey().length != 0
954               && this.getEndKey().length == 0) {
955           return 1; // this is last region
956       }
957       if (o.getStartKey().length != 0
958               && o.getEndKey().length == 0) {
959           return -1; // o is the last region
960       }
961       return result;
962     }
963 
964     // regionId is usually milli timestamp -- this defines older stamps
965     // to be "smaller" than newer stamps in sort order.
966     if (this.regionId > o.regionId) {
967       return 1;
968     } else if (this.regionId < o.regionId) {
969       return -1;
970     }
971 
972     int replicaDiff = this.getReplicaId() - o.getReplicaId();
973     if (replicaDiff != 0) return replicaDiff;
974 
975     if (this.offLine == o.offLine)
976       return 0;
977     if (this.offLine == true) return -1;
978 
979     return 1;
980   }

1022   /**
1023    * Convert a RegionInfo to a HRegionInfo
1024    *
1025    * @param proto the RegionInfo to convert
1026    * @return the converted HRegionInfho
1027    */
1028   public static HRegionInfo convert(final RegionInfo proto) {
1029     if (proto == null) return null;
1030     TableName tableName =
1031         ProtobufUtil.toTableName(proto.getTableName());
1032     if (tableName.equals(TableName.META_TABLE_NAME)) {
1033       return RegionReplicaUtil.getRegionInfoForReplica(FIRST_META_REGIONINFO,
1034           proto.getReplicaId());
1035     }
1036     long regionId = proto.getRegionId();
1037     int replicaId = proto.hasReplicaId() ? proto.getReplicaId() : DEFAULT_REPLICA_ID;
1038     byte[] startKey = null;
1039     byte[] endKey = null;
1040     if (proto.hasStartKey()) {
1041       startKey = proto.getStartKey().toByteArray();
1042     }
1043     if (proto.hasEndKey()) {
1044       endKey = proto.getEndKey().toByteArray();
1045     }
1046     boolean split = false;
1047     if (proto.hasSplit()) {
1048       split = proto.getSplit();
1049     }
1050     HRegionInfo hri = new HRegionInfo(
1051         tableName,
1052         startKey,
1053         endKey, split, regionId, replicaId);
1054     if (proto.hasOffline()) {
1055       hri.setOffline(proto.getOffline());
1056     }
1057     return hri;
1058   }

1260   /**
1261    * Parses an HRegionInfo instance from the passed in stream.  Presumes the HRegionInfo was
1262    * serialized to the stream with {@link #toDelimitedByteArray()}
1263    * @param in
1264    * @return An instance of HRegionInfo.
1265    * @throws IOException
1266    */
1267   public static HRegionInfo parseFrom(final DataInputStream in) throws IOException {
1268     // I need to be able to move back in the stream if this is not a pb serialization so I can
1269     // do the Writable decoding instead.
1270     int pblen = ProtobufUtil.lengthOfPBMagic();
1271     byte [] pbuf = new byte[pblen];
1272     if (in.markSupported()) { //read it with mark()
1273       in.mark(pblen);
1274     }
1275 
1276     //assumption: if Writable serialization, it should be longer than pblen.
1277     int read = in.read(pbuf);
1278     if (read != pblen) throw new IOException("read=" + read + ", wanted=" + pblen);
1279     if (ProtobufUtil.isPBMagicPrefix(pbuf)) {
1280       return convert(HBaseProtos.RegionInfo.parseDelimitedFrom(in));
1281     } else {
1282         // Presume Writables.  Need to reset the stream since it didn't start w/ pb.
1283       if (in.markSupported()) {
1284         in.reset();
1285         HRegionInfo hri = new HRegionInfo();
1286         hri.readFields(in);
1287         return hri;
1288       } else {
1289         //we cannot use BufferedInputStream, it consumes more than we read from the underlying IS
1290         ByteArrayInputStream bais = new ByteArrayInputStream(pbuf);
1291         SequenceInputStream sis = new SequenceInputStream(bais, in); //concatenate input streams
1292         HRegionInfo hri = new HRegionInfo();
1293         hri.readFields(new DataInputStream(sis));
1294         return hri;
1295       }
1296     }
1297   }

1299   /**
1300    * Serializes given HRegionInfo's as a byte array. Use this instead of {@link #toByteArray()} when
1301    * writing to a stream and you want to use the pb mergeDelimitedFrom (w/o the delimiter, pb reads
1302    * to EOF which may not be what you want). {@link #parseDelimitedFrom(byte[], int, int)} can
1303    * be used to read back the instances.
1304    * @param infos HRegionInfo objects to serialize
1305    * @return This instance serialized as a delimited protobuf w/ a magic pb prefix.
1306    * @throws IOException
1307    * @see #toByteArray()
1308    */
1309   public static byte[] toDelimitedByteArray(HRegionInfo... infos) throws IOException {
1310     byte[][] bytes = new byte[infos.length][];
1311     int size = 0;
1312     for (int i = 0; i < infos.length; i++) {
1313       bytes[i] = infos[i].toDelimitedByteArray();
1314       size += bytes[i].length;
1315     }
1316 
1317     byte[] result = new byte[size];
1318     int offset = 0;
1319     for (byte[] b : bytes) {
1320       System.arraycopy(b, 0, result, offset, b.length);
1321       offset += b.length;
1322     }
1323     return result;
1324   }

1326   /**
1327    * Parses all the HRegionInfo instances from the passed in stream until EOF. Presumes the
1328    * HRegionInfo's were serialized to the stream with {@link #toDelimitedByteArray()}
1329    * @param bytes serialized bytes
1330    * @param offset the start offset into the byte[] buffer
1331    * @param length how far we should read into the byte[] buffer
1332    * @return All the hregioninfos that are in the byte array. Keeps reading till we hit the end.
1333    */
1334   public static List<HRegionInfo> parseDelimitedFrom(final byte[] bytes, final int offset,
1335       final int length) throws IOException {
1336     if (bytes == null) {
1337       throw new IllegalArgumentException("Can't build an object with empty bytes array");
1338     }
1339     DataInputBuffer in = new DataInputBuffer();
1340     List<HRegionInfo> hris = new ArrayList<HRegionInfo>();
1341     try {
1342       in.reset(bytes, offset, length);
1343       while (in.available() > 0) {
1344         HRegionInfo hri = parseFrom(in);
1345         hris.add(hri);
1346       }
1347     } finally {
1348       in.close();
1349     }
1350     return hris;
1351   }

1353   /**
1354    * Check whether two regions are adjacent
1355    * @param regionA
1356    * @param regionB
1357    * @return true if two regions are adjacent
1358    */
1359   public static boolean areAdjacent(HRegionInfo regionA, HRegionInfo regionB) {
1360     if (regionA == null || regionB == null) {
1361       throw new IllegalArgumentException(
1362           "Can't check whether adjacent for null region");
1363     }
1364     HRegionInfo a = regionA;
1365     HRegionInfo b = regionB;
1366     if (Bytes.compareTo(a.getStartKey(), b.getStartKey()) > 0) {
1367       a = regionB;
1368       b = regionA;
1369     }
1370     if (Bytes.compareTo(a.getEndKey(), b.getStartKey()) == 0) {
1371       return true;
1372     }
1373     return false;
1374   }
