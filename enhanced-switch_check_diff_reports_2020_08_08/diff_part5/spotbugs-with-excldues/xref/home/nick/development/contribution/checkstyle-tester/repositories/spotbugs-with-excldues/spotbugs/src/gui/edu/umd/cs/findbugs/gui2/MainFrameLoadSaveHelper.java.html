54      void importFilter() {
55          filterOpenFileChooser.setDialogTitle(L10N.getLocalString("dlg.importFilter_ttl", "Import and merge filter..."));
56  
57          boolean retry = true;
58          File f;
59          while (retry) {
60              retry = false;
61  
62              int value = filterOpenFileChooser.showOpenDialog(mainFrame);
63  
64              if (value != JFileChooser.APPROVE_OPTION) {
65                  return;
66              }
67  
68              f = filterOpenFileChooser.getSelectedFile();
69  
70              if (!f.exists()) {
71                  JOptionPane.showMessageDialog(filterOpenFileChooser, "No such file", "Invalid File", JOptionPane.WARNING_MESSAGE);
72                  retry = true;
73                  continue;
74              }
75              Filter filter;
76              try {
77                  filter = Filter.parseFilter(f.getPath());
78              } catch (IOException e) {
79                  JOptionPane.showMessageDialog(filterOpenFileChooser, "Could not load filter.");
80                  retry = true;
81                  continue;
82              }
83              mainFrame.setProjectChanged(true);
84              Filter suppressionFilter = mainFrame.getProject().getSuppressionFilter();
85  
86              for (Matcher m : filter.getChildren()) {
87                  suppressionFilter.addChild(m);
88              }
89  
90              PreferencesFrame.getInstance().updateFilterPanel();
91          }
92

98      void open() {
99          if (askToSave()) {
100             return;
101         }
102 
103         boolean loading = true;
104         SaveType fileType;
105         tryAgain: while (loading) {
106             int value = saveOpenFileChooser.showOpenDialog(mainFrame);
107             if (value != JFileChooser.APPROVE_OPTION) {
108                 return;
109             }
110 
111             loading = false;
112             fileType = convertFilterToType(saveOpenFileChooser.getFileFilter());
113             final File f = saveOpenFileChooser.getSelectedFile();
114 
115             if (!fileType.isValid(f)) {
116                 JOptionPane.showMessageDialog(saveOpenFileChooser, "That file is not compatible with the chosen file type",
117                         "Invalid File", JOptionPane.WARNING_MESSAGE);
118                 loading = true;
119                 continue;
120             }
121 
122             switch (fileType) {
123             case XML_ANALYSIS:
124                 if (!f.getName().endsWith(".xml")) {
125                     JOptionPane.showMessageDialog(saveOpenFileChooser,
126                             L10N.getLocalString("dlg.not_xml_data_lbl", "This is not a saved bug XML data file."));
127                     loading = true;
128                     continue tryAgain;
129                 }
130 
131                 if (!mainFrame.openAnalysis(f, fileType)) {
132                     // TODO: Deal if something happens when loading analysis
133                     JOptionPane.showMessageDialog(saveOpenFileChooser, "An error occurred while trying to load the analysis.");
134                     loading = true;
135                     continue tryAgain;
136                 }
137                 break;
138             case FBP_FILE:
139                 if (!openFBPFile(f)) {
140                     // TODO: Deal if something happens when loading analysis
141                     JOptionPane.showMessageDialog(saveOpenFileChooser, "An error occurred while trying to load the analysis.");
142                     loading = true;
143                     continue tryAgain;
144                 }
145                 break;
146             case FBA_FILE:
147                 if (!openFBAFile(f)) {
148                     // TODO: Deal if something happens when loading analysis
149                     JOptionPane.showMessageDialog(saveOpenFileChooser, "An error occurred while trying to load the analysis.");
150                     loading = true;
151                     continue tryAgain;
152                 }
153                 break;
154             default:
155                 assert false;
156             }
157         }
158     }

160     /** Returns true if cancelled */
161     private boolean askToSave() {
162         if (mainFrame.isProjectChanged()) {
163             int response = JOptionPane.showConfirmDialog(mainFrame, L10N.getLocalString("dlg.save_current_changes",
164                     "The current project has been changed, Save current changes?"), L10N.getLocalString("dlg.save_changes",
165                             "Save Changes?"), JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE);
166 
167             if (response == JOptionPane.YES_OPTION) {
168                 if (mainFrame.getSaveFile() != null) {
169                     save();
170                 } else {
171                     saveAs();
172                 }
173             } else if (response == JOptionPane.CANCEL_OPTION)
174             {
175                 return true;
176                 // IF no, do nothing.
177             }
178         }
179         return false;
180     }

194 
195         return true;

198     boolean exportFilter() {
199         filterOpenFileChooser.setDialogTitle(L10N.getLocalString("dlg.exportFilter_ttl", "Export filter..."));
200 
201         boolean retry = true;
202         boolean alreadyExists;
203         File f;
204         while (retry) {
205             retry = false;
206 
207             int value = filterOpenFileChooser.showSaveDialog(mainFrame);
208 
209             if (value != JFileChooser.APPROVE_OPTION) {
210                 return false;
211             }
212 
213             f = filterOpenFileChooser.getSelectedFile();
214 
215             alreadyExists = f.exists();
216             if (alreadyExists) {
217                 int response = JOptionPane.showConfirmDialog(filterOpenFileChooser,
218                         L10N.getLocalString("dlg.file_exists_lbl", "This file already exists.\nReplace it?"),
219                         L10N.getLocalString("dlg.warning_ttl", "Warning!"), JOptionPane.OK_CANCEL_OPTION,
220                         JOptionPane.WARNING_MESSAGE);
221 
222                 if (response == JOptionPane.OK_OPTION) {
223                     retry = false;
224                 }
225                 if (response == JOptionPane.CANCEL_OPTION) {
226                     retry = true;
227                     continue;
228                 }
229 
230             }
231 
232             Filter suppressionFilter = mainFrame.getProject().getSuppressionFilter();
233             try {
234                 suppressionFilter.writeEnabledMatchersAsXML(new FileOutputStream(f));
235             } catch (IOException e) {
236                 JOptionPane.showMessageDialog(mainFrame,
237                         L10N.getLocalString("dlg.saving_error_lbl", "An error occurred in saving."));
238                 return false;
239             }
240         }
241 
242         return true;
243     }

245     boolean saveAs() {
246         saveOpenFileChooser.setDialogTitle(L10N.getLocalString("dlg.saveas_ttl", "Save as..."));
247 
248         if (mainFrame.getCurrentProject() == null) {
249             JOptionPane.showMessageDialog(mainFrame, L10N.getLocalString("dlg.no_proj_save_lbl", "There is no project to save"));
250             return false;
251         }
252 
253         boolean retry = true;
254         SaveType fileType = SaveType.NOT_KNOWN;
255         boolean alreadyExists;
256         File f = null;
257         while (retry) {
258             retry = false;
259 
260             int value = saveOpenFileChooser.showSaveDialog(mainFrame);
261 
262             if (value != JFileChooser.APPROVE_OPTION) {
263                 return false;
264             }
265 
266             fileType = convertFilterToType(saveOpenFileChooser.getFileFilter());
267             if (fileType == SaveType.NOT_KNOWN) {
268                 Debug.println("Error! fileType == SaveType.NOT_KNOWN");
269                 // This should never happen b/c saveOpenFileChooser can only
270                 // display filters
271                 // given it.
272                 retry = true;
273                 continue;
274             }
275 
276             f = saveOpenFileChooser.getSelectedFile();
277 
278             f = convertFile(f, fileType);
279 
280             if (!fileType.isValid(f)) {
281                 JOptionPane.showMessageDialog(saveOpenFileChooser, "That file is not compatible with the chosen file type",
282                         "Invalid File", JOptionPane.WARNING_MESSAGE);
283                 retry = true;
284                 continue;
285             }
286 
287             alreadyExists = fileAlreadyExists(f);
288             if (alreadyExists) {
289                 int response = -1;
290 
291                 switch (fileType) {
292                 case HTML_OUTPUT:
293                     response = JOptionPane.showConfirmDialog(saveOpenFileChooser,
294                             L10N.getLocalString("dlg.analysis_exists_lbl", "This html output already exists.\nReplace it?"),
295                             L10N.getLocalString("dlg.warning_ttl", "Warning!"), JOptionPane.OK_CANCEL_OPTION,
296                             JOptionPane.WARNING_MESSAGE);
297                     break;case XML_ANALYSIS:
298                         response = JOptionPane.showConfirmDialog(saveOpenFileChooser,
299                                 L10N.getLocalString("dlg.analysis_exists_lbl", "This analysis already exists.\nReplace it?"),
300                                 L10N.getLocalString("dlg.warning_ttl", "Warning!"), JOptionPane.OK_CANCEL_OPTION,
301                                 JOptionPane.WARNING_MESSAGE);
302                         break;
303                     case FBP_FILE:
304                         response = JOptionPane.showConfirmDialog(saveOpenFileChooser,
305                                 L10N.getLocalString("FB Project File already exists",
306                                         "This FB project file already exists.\nDo you want to replace it?"), L10N.getLocalString(
307                                                 "dlg.warning_ttl", "Warning!"), JOptionPane.OK_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE);
308                         break;
309                     case FBA_FILE:
310                         response = JOptionPane.showConfirmDialog(saveOpenFileChooser, L10N.getLocalString(
311                                 "FB Analysis File already exists",
312                                 "This FB analysis file already exists.\nDo you want to replace it?"), L10N.getLocalString(
313                                         "dlg.warning_ttl", "Warning!"), JOptionPane.OK_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE);
314                         break;
315                     default:
316                         assert false;
317                 }
318 
319                 if (response == JOptionPane.OK_OPTION) {
320                     retry = false;
321                 }
322                 if (response == JOptionPane.CANCEL_OPTION) {
323                     retry = true;
324                     continue;
325                 }
326 
327             }
328 
329             SaveReturn successful = SaveReturn.SAVE_ERROR;
330 
331             switch (fileType) {
332 
333             case HTML_OUTPUT:
334                 successful = printHtml(f);
335                 break;
336             case XML_ANALYSIS:
337                 successful = saveAnalysis(f);
338                 break;
339             case FBA_FILE:
340                 successful = saveFBAFile(f);
341                 break;
342             case FBP_FILE:
343                 successful = saveFBPFile(f);
344                 break;
345             default:
346                 JOptionPane.showMessageDialog(mainFrame, "Unknown save file type");
347                 return false;
348             }
349 
350             if (successful != SaveReturn.SAVE_SUCCESSFUL) {
351                 JOptionPane.showMessageDialog(mainFrame,
352                         L10N.getLocalString("dlg.saving_error_lbl", "An error occurred in saving."));
353                 return false;
354             }
355         }
356         assert f != null;
357 
358         // saveProjectMenuItem.setEnabled(false);
359         mainFrame.getSaveMenuItem().setEnabled(false);
360         mainFrame.setSaveType(fileType);
361         mainFrame.setSaveFile(f);
362         File xmlFile = f;
363 
364         mainFrame.addFileToRecent(xmlFile);
365 
366         return true;
367     }

369     SaveType convertFilterToType(FileFilter f) {
370         if (f instanceof FindBugsFileFilter) {
371             return ((FindBugsFileFilter) f).getSaveType();
372         }
373 
374         return SaveType.NOT_KNOWN;
375     }

382     File convertFile(File f, SaveType fileType) {
383         // Checks that it has the correct file extension, makes a new file if it
384         // doesn't.
385         if (!f.getName().endsWith(fileType.getFileExtension())) {
386             f = new File(f.getAbsolutePath() + fileType.getFileExtension());
387         }
388 
389         return f;
390     }

392     void save() {
393         File sFile = mainFrame.getSaveFile();
394         assert sFile != null;
395 
396         SaveReturn result = SaveReturn.SAVE_ERROR;
397 
398         switch (mainFrame.getSaveType()) {
399 
400         case XML_ANALYSIS:
401             result = saveAnalysis(sFile);
402             break;
403         case FBA_FILE:
404             result = saveFBAFile(sFile);
405             break;
406         case FBP_FILE:
407             result = saveFBPFile(sFile);
408             break;
409         default:
410             JOptionPane.showMessageDialog(mainFrame, "Unknown save file type");
411             return;
412 
413         }
414 
415         if (result != SaveReturn.SAVE_SUCCESSFUL) {
416             JOptionPane.showMessageDialog(mainFrame, L10N.getLocalString("dlg.saving_error_lbl", "An error occurred in saving."));
417         }
418     }

437     SaveReturn printHtml(final File f) {
438 
439         Future<Object> waiter = mainFrame.getBackgroundExecutor().submit(() -> {
440             HTMLBugReporter reporter = new HTMLBugReporter( mainFrame.getProject(), "default.xsl");
441             reporter.setIsRelaxed(true);
442             reporter.setOutputStream(UTF8.printStream(new FileOutputStream(f)));
443             for(BugInstance bug : mainFrame.getBugCollection().getCollection()) {
444                 try {
445                     if (mainFrame.getViewFilter().show(bug)) {
446                         reporter.reportBug(bug);
447                     }
448                 } catch (Exception e) {
449                     e.printStackTrace();
450                 }
451             }
452             reporter.finish();
453             return null;
454         });
455         try {
456             waiter.get();
457         } catch (InterruptedException e) {
458             return SaveReturn.SAVE_ERROR;
459         } catch (ExecutionException e) {
460             return SaveReturn.SAVE_ERROR;
461         }
462 
463         return SaveReturn.SAVE_SUCCESSFUL;
464     }

466     /**
467      * Save current analysis as file passed in. Return SAVE_SUCCESSFUL if save
468      * successful. Method doesn't do much. This method is more if need to do
469      * other things in the future for saving analysis. And to keep saving naming
470      * convention.
471      */
472     SaveReturn saveAnalysis(final File f) {
473 
474         Future<Object> waiter = mainFrame.getBackgroundExecutor().submit(() -> {
475             BugSaver.saveBugs(f, mainFrame.getBugCollection(), mainFrame.getProject());
476             return null;
477         });
478         try {
479             waiter.get();
480         } catch (InterruptedException e) {
481             return SaveReturn.SAVE_ERROR;
482         } catch (ExecutionException e) {
483             return SaveReturn.SAVE_ERROR;
484         }
485 
486         mainFrame.setProjectChanged(false);
487 
488         return SaveReturn.SAVE_SUCCESSFUL;
489     }

519     void loadAnalysis(final File file) {
520 
521         Runnable runnable = () -> {
522             mainFrame.acquireDisplayWait();
523             try {
524                 Project project = new Project();
525                 project.setGuiCallback(mainFrame.getGuiCallback());
526                 project.setCurrentWorkingDirectory(file.getParentFile());
527                 BugLoader.loadBugs(mainFrame, project, file);
528                 project.getSourceFinder(); // force source finder to be
529                 // initialized
530                 mainFrame.updateBugTree();
531             } finally {
532                 mainFrame.releaseDisplayWait();
533             }
534         };
535         if (EventQueue.isDispatchThread()) {
536             new Thread(runnable, "Analysis loading thread").start();
537         } else {
538             runnable.run();
539         }
540     }

564     void loadProjectFromFile(final File f) {
565 
566         Runnable runnable = () -> {
567             final Project project = BugLoader.loadProject(mainFrame, f);
568             final BugCollection bc = project == null ? null : BugLoader.doAnalysis(project);
569             mainFrame.updateProjectAndBugCollection(bc);
570             mainFrame.setProjectAndBugCollectionInSwingThread(project, bc);
571         };
572         if (EventQueue.isDispatchThread()) {
573             new Thread(runnable).start();
574         } else {
575             runnable.run();
576         }
577     }
