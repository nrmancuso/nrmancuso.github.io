102     public void setPollInterval(String interval) throws Exception {
103         long l = RetryIntervalls.parseIntervalOrNever(interval);
104         if (l != pollInterval) {
105             this.pollInterval = l;
106             if (getState() == STARTED) {
107                 scheduler.stopScheduler(timerIDForwardPolling, schedulerID, this);
108                 schedulerID = scheduler.startScheduler(timerIDForwardPolling,
109                         pollInterval, this);

111         }

144     private void checkSQL(String sql) throws SQLException {
145         sqlIsValid = false;
146         String sqlUC = sql.toUpperCase();
147         if ( sqlUC.indexOf("DELETE ") != -1 ) {
148             lastCheckResult = "DELETE is not allowed in this SQL statement!";
149             throw new IllegalArgumentException(lastCheckResult);
150         }
151         if ( sqlUC.indexOf("UPDATE ") != -1 ) {
152             lastCheckResult = "UPDATE is not allowed in this SQL statement!";
153             throw new IllegalArgumentException(lastCheckResult);
154         }
155         try {
156             QueryForwardCmd cmd = QueryForwardCmd.getInstance(sql, this.limit > 0 ? 1 : 0, 1);
157             cmd.setUpdateDatabaseMaxRetries(1);
158             Map<String, List<String>> chk = cmd.getSeriesIUIDs(sql.indexOf('?') != -1 ? System.currentTimeMillis() : null, 0l);
159             log.debug("CheckSQL: QueryForwardCmd.getSeriesIUIDs done with result:"+chk);
160             if ( chk != null && !chk.isEmpty()) {
161                 if ( chk.keySet().iterator().next().indexOf('.') != -1 ) {
162                     lastCheckResult = "Wrong result! RetrieveAET contains '.'! Query must return series.series_iuid, series.retrieve_aets as first columns!";
163                     throw new IllegalArgumentException(lastCheckResult);
164                 }
165                 List<String> l = chk.values().iterator().next();
166                 if ( !l.isEmpty() && l.iterator().next().indexOf('.') == -1) {
167                     lastCheckResult = "Wrong result! Series Instance UID does NOT contain a '.'! Query must return series.series_iuid, series.retrieve_aets as first columns!";
168                     throw new IllegalArgumentException(lastCheckResult);
169                 }
170             }
171         } catch (SQLException x) {
172             lastCheckResult = x.getCause() == null ? x.toString() : x.getCause().toString();
173             throw x;
174         }
175         lastCheckResult="OK";
176         sqlIsValid = true;
177     }
178 
179     private void updateCmd() throws SQLException {
180         if (log.isDebugEnabled())
181             log.debug("update QueryForwardCmd when state = 3. state:"+getState()+" limit:"+limit+"\nSQL:"+sql);
182         if ( this.getState() == STARTED) {
183             if ( sqlCmd != null ) {
184                 sqlCmd.close();
185             }
186             sqlCmd = QueryForwardCmd.getInstance(sql, limit, fetchSize);
187         }
188     }

250     public void handleNotification(Notification notification, Object handback) {
251         synchronized(this) {
252             if (isRunning) {
253                 log.info("FileCopyByQuery is already running!");
254                 return;
255             }
256             isRunning = true;
257         }
258         new Thread(new Runnable(){
259             public void run() {
260                 try {
261                     doCheckForward();
262                 } catch (Exception e) {
263                     log.error("Check for Pending Series Stored failed:", e);
264                 } finally {
265                     isRunning = false;
266                 }
267             }}).start();
268     }

270     public int checkForward() {
271         synchronized(this) {
272             if (isRunning) {
273                 log.info("ForwardByQuery is already running!");
274                 return -1;
275             }
276             isRunning = true;
277         }
278         return doCheckForward();
279     }
280     
281     public int doCheckForward() {
282         if (lastCheckResult == null) {
283             try {
284                 checkSQL(sql);
285                 if (sqlCmd != null) { 
286                     sqlCmd.close();
287                 }
288                 sqlCmd = QueryForwardCmd.getInstance(sql, limit, fetchSize);
289             } catch ( Throwable t) {
290                 log.error("Check of SQL statement failed!",t);
291             }
292         }
293         if (!sqlIsValid) {
294             log.warn("SQL is not marked to be valid! checkForward is disabled!");
295             isRunning = false;
296             return -1;
297         }
298         int nrOfSeries = 0;
299         try {
300             long scheduledTime = 0L;
301             Map<String,List<String>> orders = sqlCmd.getSeriesIUIDs(delay < 0 ? null : new Long(System.currentTimeMillis()-delay), lastSeriesPk);
302             log.info("Found "+orders.size()+" MoveOrder! lastSeriesPk:"+lastSeriesPk);
303             lastSeriesPk = orders.isEmpty() ? 0 : sqlCmd.getLastSeriesPk();
304             MoveOrder order;
305             List<String> series;
306             for ( Map.Entry<String, List<String>> entry : orders.entrySet() ) {
307                 series = entry.getValue();
308                 nrOfSeries += series.size();
309                 for (int i = 0, len = series.size() ; i < len ; i++) {
310                     order = new MoveOrder(entry.getKey(), calledAET, forwardPriority, null, null, series.get(i), null);
311                     this.scheduleMove(order, scheduledTime);
312                 }
313             }
314             return nrOfSeries;
315         } catch (Exception e) {
316             log.error("Failed to check for forward:", e);
317             return 0;
318         } finally {
319             isRunning = false;
320         }
321     }
322     
323     public String showSQL() {
324         if (lastCheckResult == null) {
325             try {
326                 checkSQL(sql);
327                 if (sqlCmd != null) { 
328                     sqlCmd.close();
329                 }
330                 sqlCmd = QueryForwardCmd.getInstance(sql, limit, fetchSize);
331             } catch ( Throwable t) {
332                 return "Check of SQL statement failed!";
333             }
334         }
335         return !sqlIsValid ? "SQL is not marked to be valid!" : sqlCmd.formatSql();
336     }

338     protected void scheduleMove(MoveOrder order, long scheduledTime) {
339         try {
340            server.invoke(moveScuServiceName, "scheduleMoveOrder", new Object[] {
341                     order, new Long(scheduledTime) },
342                     new String[] { MoveOrder.class.getName(), long.class.getName() });
343         } catch (Exception e) {
344             log.error("Schedule Move failed:", e);
345         }
346     }
