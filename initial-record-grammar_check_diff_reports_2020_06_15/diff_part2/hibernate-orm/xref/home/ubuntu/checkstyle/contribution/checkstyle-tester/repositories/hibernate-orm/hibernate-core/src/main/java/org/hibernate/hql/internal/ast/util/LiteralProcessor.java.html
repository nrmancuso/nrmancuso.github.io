54  /**
55   * A delegate that handles literals and constants for HqlSqlWalker, performing the token replacement functions and
56   * classifying literals.
57   *
58   * @author josh
59   */
60  public class LiteralProcessor implements HqlSqlTokenTypes {
61  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, LiteralProcessor.class.getName());

63  	/**
64  	 * In what format should Float and Double literal values be sent to the database?
65  	 */
66  	public static DecimalLiteralFormat DECIMAL_LITERAL_FORMAT = DecimalLiteralFormat.EXACT;

68  	private HqlSqlWalker walker;

70  	public LiteralProcessor(HqlSqlWalker hqlSqlWalker) {
71  		this.walker = hqlSqlWalker;

74  	public boolean isAlias(String alias) {
75  		FromClause from = walker.getCurrentFromClause();
76  		while ( from.isSubQuery() ) {
77  			if ( from.containsClassAlias(alias) ) {
78  				return true;
79  			}
80  			from = from.getParentFromClause();
81  		}
82  		return from.containsClassAlias(alias);
83  	}

85  	public void processConstant(AST constant, boolean resolveIdent) throws SemanticException {
86  		// If the constant is an IDENT, figure out what it means...
87  		boolean isIdent = ( constant.getType() == IDENT || constant.getType() == WEIRD_IDENT );
88  		if ( resolveIdent && isIdent && isAlias( constant.getText() ) ) { // IDENT is a class alias in the FROM.
89  			IdentNode ident = ( IdentNode ) constant;
90  			// Resolve to an identity column.
91  			ident.resolve(false, true);
92  		}
93  		else {	// IDENT might be the name of a class.
94  			Queryable queryable = walker.getSessionFactoryHelper().findQueryableUsingImports( constant.getText() );
95  			if ( isIdent && queryable != null ) {
96  				constant.setText( queryable.getDiscriminatorSQLValue() );
97  			}
98  			// Otherwise, it's a literal.
99  			else {
100 				processLiteral( constant );
101 			}
102 		}
103 	}

105 	public void lookupConstant(DotNode node) throws SemanticException {
106 		String text = ASTUtil.getPathText( node );
107 		Queryable persister = walker.getSessionFactoryHelper().findQueryableUsingImports( text );
108 		if ( persister != null ) {
109 			// the name of an entity class
110 			final String discrim = persister.getDiscriminatorSQLValue();
111 			node.setDataType( persister.getDiscriminatorType() );
112             if (InFragment.NULL.equals(discrim) || InFragment.NOT_NULL.equals(discrim)) throw new InvalidPathException(
113                                                                                                                        "subclass test not allowed for null or not null discriminator: '"
114                                                                                                                        + text + "'");
115             setSQLValue(node, text, discrim); // the class discriminator value
116 		}
117 		else {
118 			Object value = ReflectHelper.getConstantValue( text );
119             if (value == null) throw new InvalidPathException("Invalid path: '" + text + "'");
120             setConstantValue(node, text, value);
121 		}
122 	}
123 
124 	private void setSQLValue(DotNode node, String text, String value) {
125 		LOG.debugf( "setSQLValue() %s -> %s", text, value );
126 		node.setFirstChild( null );	// Chop off the rest of the tree.
127 		node.setType( SqlTokenTypes.SQL_TOKEN );
128 		node.setText(value);
129 		node.setResolvedConstant( text );
130 	}
131 
132 	private void setConstantValue(DotNode node, String text, Object value) {
133 		if ( LOG.isDebugEnabled() ) {
134 			LOG.debugf( "setConstantValue() %s -> %s %s", text, value, value.getClass().getName() );
135 		}
136 		node.setFirstChild( null );	// Chop off the rest of the tree.
137 		if ( value instanceof String ) {
138 			node.setType( SqlTokenTypes.QUOTED_STRING );
139 		}
140 		else if ( value instanceof Character ) {
141 			node.setType( SqlTokenTypes.QUOTED_STRING );
142 		}
143 		else if ( value instanceof Byte ) {
144 			node.setType( SqlTokenTypes.NUM_INT );
145 		}
146 		else if ( value instanceof Short ) {
147 			node.setType( SqlTokenTypes.NUM_INT );
148 		}
149 		else if ( value instanceof Integer ) {
150 			node.setType( SqlTokenTypes.NUM_INT );
151 		}
152 		else if ( value instanceof Long ) {
153 			node.setType( SqlTokenTypes.NUM_LONG );
154 		}
155 		else if ( value instanceof Double ) {
156 			node.setType( SqlTokenTypes.NUM_DOUBLE );
157 		}
158 		else if ( value instanceof Float ) {
159 			node.setType( SqlTokenTypes.NUM_FLOAT );
160 		}
161 		else {
162 			node.setType( SqlTokenTypes.CONSTANT );
163 		}
164 		Type type;
165 		try {
166 			type = walker.getSessionFactoryHelper().getFactory().getTypeResolver().heuristicType( value.getClass().getName() );
167 		}
168 		catch ( MappingException me ) {
169 			throw new QueryException( me );
170 		}
171 		if ( type == null ) {
172 			throw new QueryException( QueryTranslator.ERROR_CANNOT_DETERMINE_TYPE + node.getText() );
173 		}
174 		try {
175 			LiteralType literalType = ( LiteralType ) type;
176 			Dialect dialect = walker.getSessionFactoryHelper().getFactory().getDialect();
177 			//noinspection unchecked
178 			node.setText( literalType.objectToSQLString( value, dialect ) );
179 		}
180 		catch ( Exception e ) {
181 			throw new QueryException( QueryTranslator.ERROR_CANNOT_FORMAT_LITERAL + node.getText(), e );
182 		}
183 		node.setDataType( type );
184 		node.setResolvedConstant( text );
185 	}

187 	public void processBoolean(AST constant) {
188 		// TODO: something much better - look at the type of the other expression!
189 		// TODO: Have comparisonExpression and/or arithmeticExpression rules complete the resolution of boolean nodes.
190 		String replacement = ( String ) walker.getTokenReplacements().get( constant.getText() );
191 		if ( replacement != null ) {
192 			constant.setText( replacement );
193 		}
194 		else {
195 			boolean bool = "true".equals( constant.getText().toLowerCase() );
196 			Dialect dialect = walker.getSessionFactoryHelper().getFactory().getDialect();
197 			constant.setText( dialect.toBooleanValueString(bool) );
198 		}
199 	}
200 
201 	private void processLiteral(AST constant) {
202 		String replacement = ( String ) walker.getTokenReplacements().get( constant.getText() );
203 		if ( replacement != null ) {
204 			if ( LOG.isDebugEnabled() ) {
205 				LOG.debugf("processConstant() : Replacing '%s' with '%s'", constant.getText(), replacement);
206 			}
207 			constant.setText( replacement );
208 		}
209 	}

211 	public void processNumeric(AST literal) {
212 		if ( literal.getType() == NUM_INT
213 				|| literal.getType() == NUM_LONG
214 				|| literal.getType() == NUM_BIG_INTEGER ) {
215 			literal.setText( determineIntegerRepresentation( literal.getText(), literal.getType() ) );
216         } else if (literal.getType() == NUM_FLOAT
217 				|| literal.getType() == NUM_DOUBLE
218 				|| literal.getType() == NUM_BIG_DECIMAL ) {
219 			literal.setText( determineDecimalRepresentation( literal.getText(), literal.getType() ) );
220         } else LOG.unexpectedLiteralTokenType(literal.getType());
221 	}
222 
223 	private String determineIntegerRepresentation(String text, int type) {
224 		try {
225 			if ( type == NUM_BIG_INTEGER ) {
226 				String literalValue = text;
227 				if ( literalValue.endsWith( "bi" ) || literalValue.endsWith( "BI" ) ) {
228 					literalValue = literalValue.substring( 0, literalValue.length() - 2 );
229 				}
230 				return new BigInteger( literalValue ).toString();
231 			}
232 			if ( type == NUM_INT ) {
233 				try {
234 					return Integer.valueOf( text ).toString();
235 				}
236 				catch( NumberFormatException e ) {
237 					LOG.tracev(
238 							"Could not format incoming text [{0}] as a NUM_INT; assuming numeric overflow and attempting as NUM_LONG",
239 							text );
240 				}
241 			}
242 			String literalValue = text;
243 			if ( literalValue.endsWith( "l" ) || literalValue.endsWith( "L" ) ) {
244 				literalValue = literalValue.substring( 0, literalValue.length() - 1 );
245 			}
246 			return Long.valueOf( literalValue ).toString();
247 		}
248 		catch( Throwable t ) {
249 			throw new HibernateException( "Could not parse literal [" + text + "] as integer", t );
250 		}
251 	}
252 
253 	public String determineDecimalRepresentation(String text, int type) {
254 		String literalValue = text;
255 		if ( type == NUM_FLOAT ) {
256 			if ( literalValue.endsWith( "f" ) || literalValue.endsWith( "F" ) ) {
257 				literalValue = literalValue.substring( 0, literalValue.length() - 1 );
258 			}
259 		}
260 		else if ( type == NUM_DOUBLE ) {
261 			if ( literalValue.endsWith( "d" ) || literalValue.endsWith( "D" ) ) {
262 				literalValue = literalValue.substring( 0, literalValue.length() - 1 );
263 			}
264 		}
265 		else if ( type == NUM_BIG_DECIMAL ) {
266 			if ( literalValue.endsWith( "bd" ) || literalValue.endsWith( "BD" ) ) {
267 				literalValue = literalValue.substring( 0, literalValue.length() - 2 );
268 			}
269 		}
270 
271 		final BigDecimal number;
272 		try {
273 			number = new BigDecimal( literalValue );
274 		}
275 		catch( Throwable t ) {
276 			throw new HibernateException( "Could not parse literal [" + text + "] as big-decimal", t );
277 		}
278 
279 		return DECIMAL_LITERAL_FORMAT.getFormatter().format( number );
280 	}

283 	private static interface DecimalFormatter {
284 		String format(BigDecimal number);
285 	}
286 
287 	private static class ExactDecimalFormatter implements DecimalFormatter {
288 		public static final ExactDecimalFormatter INSTANCE = new ExactDecimalFormatter();
289 
290 		public String format(BigDecimal number) {
291 			return number.toString();
292 		}
293 	}
294 
295 	private static class ApproximateDecimalFormatter implements DecimalFormatter {
296 		public static final ApproximateDecimalFormatter INSTANCE = new ApproximateDecimalFormatter();
297 
298 		private static final String FORMAT_STRING = "#0.0E0";
299 
300 		public String format(BigDecimal number) {
301 			try {
302 				// TODO : what amount of significant digits need to be supported here?
303 				//      - from the DecimalFormat docs:
304 				//          [significant digits] = [minimum integer digits] + [maximum fraction digits]
305 				DecimalFormat jdkFormatter = new DecimalFormat( FORMAT_STRING );
306 				jdkFormatter.setMinimumIntegerDigits( 1 );
307 				jdkFormatter.setMaximumFractionDigits( Integer.MAX_VALUE );
308 				return jdkFormatter.format( number );
309 			}
310 			catch( Throwable t ) {
311 				throw new HibernateException( "Unable to format decimal literal in approximate format [" + number.toString() + "]", t );
312 			}
313 		}
314 	}
315 
316 	public static enum DecimalLiteralFormat {
317 		/**
318 		 * Indicates that Float and Double literal values should
319 		 * be treated using the SQL "exact" format (i.e., '.001')
320 		 */
321 		EXACT {
322 			@Override
323 			public DecimalFormatter getFormatter() {
324 				return ExactDecimalFormatter.INSTANCE;
325 			}
326 		},
327 		/**
328 		 * Indicates that Float and Double literal values should
329 		 * be treated using the SQL "approximate" format (i.e., '1E-3')
330 		 */
331 		@SuppressWarnings( {"UnusedDeclaration"})
332 		APPROXIMATE {
333 			@Override
334 			public DecimalFormatter getFormatter() {
335 				return ApproximateDecimalFormatter.INSTANCE;
336 			}
337 		};
338 
339 		public abstract DecimalFormatter getFormatter();
340 	}
