24  package org.hibernate.test.collection.ordered.joinedInheritence;
25  
26  import static org.junit.Assert.assertEquals;
27  import static org.junit.Assert.assertNotNull;
28  import static org.junit.Assert.assertTrue;
29  
30  import org.hibernate.Session;
31  
32  import org.junit.Test;
33  
34  import org.hibernate.internal.util.collections.CollectionHelper;
35  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
36  
37  /**
38   * @author Steve Ebersole
39   * @author Brett Meyer
40   */
41  public class OrderCollectionOfJoinedHierarchyTest extends BaseCoreFunctionalTestCase {
42  	
43  	@Override
44  	protected Class<?>[] getAnnotatedClasses() {
45  		return new Class[] { Animal.class, Lion.class, Tiger.class, Zoo.class };
46  	}
47  
48  	@Test
49  	public void testQuerySyntaxCheck() {
50  		Session session = openSession();
51  		session.beginTransaction();
52  		session.get( Zoo.class, 1L );
53  		session.getTransaction().commit();
54  		session.close();
55  	}
56  	
57  	@Test
58  	public void testOrdering() {
59  		Zoo zoo = new Zoo();
60  		Lion lion1 = new Lion();
61  		lion1.setWeight( 2 );
62  		Lion lion2 = new Lion();
63  		lion2.setWeight( 1 );
64  		zoo.getLions().add( lion1 );
65  		zoo.getLions().add( lion2 );
66  		zoo.getAnimalsById().add( lion1 );
67  		zoo.getAnimalsById().add( lion2 );
68  		
69  		Session session = openSession();
70  		session.beginTransaction();
71  		session.persist( lion1 );
72  		session.persist( lion2 );
73  		session.persist( zoo );
74  		session.getTransaction().commit();
75  		session.clear();
76  		
77  		session.beginTransaction();
78  		zoo = (Zoo) session.get( Zoo.class, zoo.getId() );
79  		zoo.getLions().size();
80  		zoo.getLions().size();
81  		zoo.getAnimalsById().size();
82  		session.getTransaction().commit();
83  		session.close();
84  		
85  		assertNotNull( zoo );
86  		assertTrue( CollectionHelper.isNotEmpty( zoo.getLions() ) && zoo.getLions().size() == 2 );
87  		assertTrue( CollectionHelper.isNotEmpty( zoo.getAnimalsById() ) && zoo.getAnimalsById().size() == 2 );
88  		assertEquals( zoo.getLions().iterator().next().getId(), lion2.getId() );
89  		assertEquals( zoo.getAnimalsById().iterator().next().getId(), lion1.getId() );
90  	}
91  }
