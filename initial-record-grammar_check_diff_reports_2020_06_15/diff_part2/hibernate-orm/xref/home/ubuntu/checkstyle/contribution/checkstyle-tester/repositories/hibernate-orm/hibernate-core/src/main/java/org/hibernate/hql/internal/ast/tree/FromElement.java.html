52  /**
53   * Represents a single mapped class mentioned in an HQL FROM clause.  Each
54   * class reference will have the following symbols:
55   * <ul>
56   * <li>A class name - This is the name of the Java class that is mapped by Hibernate.</li>
57   * <li>[optional] an HQL alias for the mapped class.</li>
58   * <li>A table name - The name of the table that is mapped to the Java class.</li>
59   * <li>A table alias - The alias for the table that will be used in the resulting SQL.</li>
60   * </ul>
61   *
62   * @author josh
63   */

145 	private void doInitialize(FromClause fromClause, String tableAlias, String className, String classAlias,
146 							  EntityPersister persister, EntityType type) {
147 		if ( initialized ) {
148 			throw new IllegalStateException( "Already initialized!!" );
149 		}
150 		this.fromClause = fromClause;
151 		this.tableAlias = tableAlias;
152 		this.className = className;
153 		this.classAlias = classAlias;
154 		this.elementType = new FromElementType( this, persister, type );
155 		// Register the FromElement with the FROM clause, now that we have the names and aliases.
156 		fromClause.registerFromElement( this );
157 		LOG.debugf( "%s : %s (%s) -> %s", fromClause, className, classAlias == null ? "<no alias>" : classAlias, tableAlias );
158 	}

268 	protected void appendDisplayText(StringBuilder buf) {
269 		buf.append( isImplied() ? (
270 				isImpliedInFromClause() ? "implied in FROM clause" : "implied" )
271 				: "explicit" );
272 		buf.append( "," ).append( isCollectionJoin() ? "collection join" : "not a collection join" );
273 		buf.append( "," ).append( fetch ? "fetch join" : "not a fetch join" );
274 		buf.append( "," ).append( isAllPropertyFetch ? "fetch all properties" : "fetch non-lazy properties" );
275 		buf.append( ",classAlias=" ).append( getClassAlias() );
276 		buf.append( ",role=" ).append( role );
277 		buf.append( ",tableName=" ).append( getTableName() );
278 		buf.append( ",tableAlias=" ).append( getTableAlias() );
279 		FromElement origin = getRealOrigin();
280 		buf.append( ",origin=" ).append( origin == null ? "null" : origin.getText() );
281 		buf.append( ",columns={" );
282 		if ( columns != null ) {
283 			for ( int i = 0; i < columns.length; i++ ) {
284 				buf.append( columns[i] );
285 				if ( i < columns.length ) {
286 					buf.append( " " );
287 				}
288 			}
289 		}
290 		buf.append( ",className=" ).append( className );
291 		buf.append( "}" );
292 	}

337 	public String[] getIdentityColumns() {
338 		checkInitialized();
339 		final String table = getTableAlias();
340 		if ( table == null ) {
341 			throw new IllegalStateException( "No table alias for node " + this );
342 		}
343 
344 		final String propertyName;
345 		if ( getEntityPersister() != null && getEntityPersister().getEntityMetamodel() != null
346 				&& getEntityPersister().getEntityMetamodel().hasNonIdentifierPropertyNamedId() ) {
347 			propertyName = getEntityPersister().getIdentifierPropertyName();
348 		}
349 		else {
350 			propertyName = EntityPersister.ENTITY_ID;
351 		}
352 
353 		if ( getWalker().getStatementType() == HqlSqlTokenTypes.SELECT ) {
354 			return getPropertyMapping( propertyName ).toColumns( table, propertyName );
355 		}
356 		else {
357 			return getPropertyMapping( propertyName ).toColumns( propertyName );
358 		}
359 	}

385 	public void setOrigin(FromElement origin, boolean manyToMany) {
386 		this.origin = origin;
387 		this.manyToMany = manyToMany;
388 		origin.addDestination( this );
389 		if ( origin.getFromClause() == this.getFromClause() ) {
390 			// TODO: Figure out a better way to get the FROM elements in a proper tree structure.
391 			// If this is not the destination of a many-to-many, add it as a child of the origin.
392 			if ( manyToMany ) {
393 				ASTUtil.appendSibling( origin, this );
394 			}
395 			else {
396 				if ( !getWalker().isInFrom() && !getWalker().isInSelect() ) {
397 					getFromClause().addChild( this );
398 				}
399 				else {
400 					origin.addChild( this );
401 				}
402 			}
403 		}
404 		else if ( !getWalker().isInFrom() ) {
405 			// HHH-276 : implied joins in a subselect where clause - The destination needs to be added
406 			// to the destination's from clause.
407 			getFromClause().addChild( this );	// Not sure if this is will fix everything, but it works.
408 		}
409 		else {
410 			// Otherwise, the destination node was implied by the FROM clause and the FROM clause processor
411 			// will automatically add it in the right place.
412 		}
413 	}

473 	private TypeDiscriminatorMetadata buildTypeDiscriminatorMetadata() {
474 		final String aliasToUse = getTableAlias();
475 		Queryable queryable = getQueryable();
476 		if ( queryable == null ) {
477 			QueryableCollection collection = getQueryableCollection();
478 			if ( ! collection.getElementType().isEntityType() ) {
479 				throw new QueryException( "type discrimination cannot be applied to value collection [" + collection.getRole() + "]" );
480 			}
481 			queryable = (Queryable) collection.getElementPersister();
482 		}
483 
484 		handlePropertyBeingDereferenced( getDataType(), DISCRIMINATOR_PROPERTY_NAME );
485 
486 		return new TypeDiscriminatorMetadataImpl( queryable.getTypeDiscriminatorMetadata(), aliasToUse );
487 	}

614 	public void handlePropertyBeingDereferenced(Type propertySource, String propertyName) {
615 		if ( getQueryableCollection() != null && CollectionProperties.isCollectionProperty( propertyName ) ) {
616 			// propertyName refers to something like collection.size...
617 			return;
618 		}
619 		if ( propertySource.isComponentType() ) {
620 			// property name is a sub-path of a component...
621 			return;
622 		}
623 
624 		Queryable persister = getQueryable();
625 		if ( persister != null ) {
626 			try {
627 				Queryable.Declarer propertyDeclarer = persister.getSubclassPropertyDeclarer( propertyName );
628 				if ( LOG.isTraceEnabled() ) {
629 					LOG.tracev( "Handling property dereference [{0} ({1}) -> {2} ({3})]",
630 							persister.getEntityName(), getClassAlias(), propertyName, propertyDeclarer );
631 				}
632 				if ( propertyDeclarer == Queryable.Declarer.SUBCLASS ) {
633 					dereferencedBySubclassProperty = true;
634 					includeSubclasses = true;
635 				}
636 				else if ( propertyDeclarer == Queryable.Declarer.SUPERCLASS ) {
637 					dereferencedBySuperclassProperty = true;
638 				}
639 			}
640 			catch( QueryException ignore ) {
641 				// ignore it; the incoming property could not be found so we
642 				// cannot be sure what to do here.  At the very least, the
643 				// safest is to simply not apply any dereference toggling...
644 
645 			}
646 		}
647 	}
