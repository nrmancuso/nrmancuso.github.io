43  	public static Element addNativelyGeneratedId(Element parent, String name, String type,
44                                                   boolean useRevisionEntityWithNativeId) {
45          Element id_mapping = parent.addElement("id");
46          id_mapping.addAttribute("name", name).addAttribute("type", type);
47  
48          Element generator_mapping = id_mapping.addElement("generator");
49          if (useRevisionEntityWithNativeId) {
50              generator_mapping.addAttribute("class", "native");
51          } else {
52              generator_mapping.addAttribute("class", "org.hibernate.envers.enhanced.OrderedSequenceGenerator");
53              generator_mapping.addElement("param").addAttribute("name", "sequence_name").setText("REVISION_GENERATOR");
54              generator_mapping.addElement("param").addAttribute("name", "table_name").setText("REVISION_GENERATOR");
55              generator_mapping.addElement("param").addAttribute("name", "initial_value").setText("1");
56              generator_mapping.addElement("param").addAttribute("name", "increment_size").setText("1");
57          }
58  //        generator_mapping.addAttribute("class", "sequence");
59  //        generator_mapping.addElement("param").addAttribute("name", "sequence").setText("custom");
60  
61          return id_mapping;
62      }

64      public static Element addProperty(Element parent, String name, String type, boolean insertable, boolean updateable, boolean key) {
65          Element prop_mapping;
66          if (key) {
67              prop_mapping = parent.addElement("key-property");
68          } else {
69              prop_mapping = parent.addElement("property");
70          }
71  
72          prop_mapping.addAttribute("name", name);
73          prop_mapping.addAttribute("insert", Boolean.toString(insertable));
74          prop_mapping.addAttribute("update", Boolean.toString(updateable));
75  
76          if (type != null) {
77              prop_mapping.addAttribute("type", type);
78          }
79  
80          return prop_mapping;
81      }

83      public static Element addProperty(Element parent, String name, String type, boolean insertable, boolean key) {
84          return addProperty(parent, name, type, insertable, false, key);
85      }

95  	private static void addOrModifyAttribute(Element parent, String name, String value) {
96          Attribute attribute = parent.attribute(name);
97          if (attribute == null) {
98              parent.addAttribute(name, value);
99          } else {
100             attribute.setValue(value);
101         }
102     }

104     /**
105      * Column name shall be wrapped with '`' signs if quotation required.
106      */
107     public static Element addOrModifyColumn(Element parent, String name) {
108         Element column_mapping = parent.element("column");
109 
110         if (column_mapping == null) {
111             return addColumn(parent, name, null, null, null, null, null, null);
112         }
113 
114         if (!StringTools.isEmpty(name)) {
115             addOrModifyAttribute(column_mapping, "name", name);
116         }
117 
118         return column_mapping;
119     }

121     /**
122      * Adds new <code>column</code> element. Method assumes that the value of <code>name</code> attribute is already
123      * wrapped with '`' signs if quotation required. It shall be invoked when column name is taken directly from configuration
124      * file and not from {@link org.hibernate.mapping.PersistentClass} descriptor.
125      */
126     public static Element addColumn(Element parent, String name, Integer length, Integer scale, Integer precision,
127 									String sqlType, String customRead, String customWrite) {
128         return addColumn(parent, name, length, scale, precision, sqlType, customRead, customWrite, false);
129     }

131     public static Element addColumn(Element parent, String name, Integer length, Integer scale, Integer precision,
132 									String sqlType, String customRead, String customWrite, boolean quoted) {
133         Element column_mapping = parent.addElement("column");
134 
135         column_mapping.addAttribute("name", quoted ? "`" + name + "`" : name);
136         if (length != null) {
137             column_mapping.addAttribute("length", length.toString());
138         }
139 		if (scale != null) {
140 			column_mapping.addAttribute("scale", Integer.toString(scale));
141 		}
142 		if (precision != null) {
143 			column_mapping.addAttribute("precision", Integer.toString(precision));
144 		}
145 		if (!StringTools.isEmpty(sqlType)) {
146             column_mapping.addAttribute("sql-type", sqlType);
147         }
148 
149         if (!StringTools.isEmpty(customRead)) {
150             column_mapping.addAttribute("read", customRead);
151         }
152         if (!StringTools.isEmpty(customWrite)) {
153             column_mapping.addAttribute("write", customWrite);
154         }
155 
156         return column_mapping;
157     }

159     private static Element createEntityCommon(Document document, String type, AuditTableData auditTableData,
160                                               String discriminatorValue, Boolean isAbstract) {
161         Element hibernate_mapping = document.addElement("hibernate-mapping");
162         hibernate_mapping.addAttribute("auto-import", "false");
163 
164         Element class_mapping = hibernate_mapping.addElement(type);
165 
166         if (auditTableData.getAuditEntityName() != null) {
167             class_mapping.addAttribute("entity-name", auditTableData.getAuditEntityName());
168         }
169 
170         if (discriminatorValue != null) {
171             class_mapping.addAttribute("discriminator-value", discriminatorValue);
172         }
173 
174         if (!StringTools.isEmpty(auditTableData.getAuditTableName())) {
175             class_mapping.addAttribute("table", auditTableData.getAuditTableName());
176         }
177 
178         if (!StringTools.isEmpty(auditTableData.getSchema())) {
179             class_mapping.addAttribute("schema", auditTableData.getSchema());
180         }
181 
182         if (!StringTools.isEmpty(auditTableData.getCatalog())) {
183             class_mapping.addAttribute("catalog", auditTableData.getCatalog());
184         }
185 
186         if (isAbstract != null) {
187             class_mapping.addAttribute("abstract", isAbstract.toString());
188         }
189 
190         return class_mapping;
191     }

193     public static Element createEntity(Document document, AuditTableData auditTableData, String discriminatorValue,

195         return createEntityCommon(document, "class", auditTableData, discriminatorValue, isAbstract);

208                                      String schema, String catalog) {

224     public static void addColumns(Element any_mapping, Iterator<Column> columns) {
225         while (columns.hasNext()) {
226             addColumn(any_mapping, columns.next());
227         }

232      * <code>length</code>, <code>scale</code>, <code>precision</code>, <code>sql-type</code>, <code>read</code>

238         addColumn(any_mapping, column.getName(), column.getLength(), column.getScale(), column.getPrecision(),
239                   column.getSqlType(), column.getCustomRead(), column.getCustomWrite(), column.isQuoted());

242     @SuppressWarnings({"unchecked"})
243     private static void changeNamesInColumnElement(Element element, ColumnNameIterator columnNameIterator) {
244         Iterator<Element> properties = element.elementIterator();
245         while (properties.hasNext()) {
246             Element property = properties.next();
247 
248             if ("column".equals(property.getName())) {
249                 Attribute nameAttr = property.attribute("name");
250                 if (nameAttr != null) {
251                     nameAttr.setText(columnNameIterator.next());
252                 }
253             }
254         }
255     }
256 
257     @SuppressWarnings({"unchecked"})
258     public static void prefixNamesInPropertyElement(Element element, String prefix, ColumnNameIterator columnNameIterator,
259                                                     boolean changeToKey, boolean insertable) {
260         Iterator<Element> properties = element.elementIterator();
261         while (properties.hasNext()) {
262             Element property = properties.next();
263 
264             if ("property".equals(property.getName()) || "many-to-one".equals(property.getName())) {
265                 Attribute nameAttr = property.attribute("name");
266                 if (nameAttr != null) {
267                     nameAttr.setText(prefix + nameAttr.getText());
268                 }
269 
270                 changeNamesInColumnElement(property, columnNameIterator);
271 
272                 if (changeToKey) {
273                     property.setName("key-" + property.getName());
274                 }
275 
276                 if ("property".equals(property.getName())) {
277                     Attribute insert = property.attribute("insert");
278                     insert.setText(Boolean.toString(insertable));
279                 }
280             }
281         }
282     }

293     /**
294      * Adds all <code>column</code> or <code>formula</code> elements.
295      * @param element Parent element.
296      * @param columnIterator Iterator pointing at {@link org.hibernate.mapping.Column} and/or
297      *                       {@link org.hibernate.mapping.Formula} objects.
298      */
299     public static void addColumnsOrFormulas(Element element, Iterator columnIterator) {
300         while (columnIterator.hasNext()) {
301             Object o = columnIterator.next();
302             if (o instanceof Column) {
303                 addColumn(element, (Column) o);
304             } else if (o instanceof Formula) {
305                 addFormula(element, (Formula) o);
306             }
307         }
308     }

310     /**
311      * An iterator over column names.
312      */
313     public static abstract class ColumnNameIterator implements Iterator<String> { }
314 
315     public static ColumnNameIterator getColumnNameIterator(final Iterator<Column> columnIterator) {
316         return new ColumnNameIterator() {
317             public boolean hasNext() { return columnIterator.hasNext(); }
318             public String next() { return columnIterator.next().getName(); }
319             public void remove() { columnIterator.remove(); }
320         };
321     }
322 
323     public static ColumnNameIterator getColumnNameIterator(final JoinColumn[] joinColumns) {
324         return new ColumnNameIterator() {
325             int counter = 0;
326             public boolean hasNext() { return counter < joinColumns.length; }
327             public String next() { return joinColumns[counter++].name(); }
328             public void remove() { throw new UnsupportedOperationException(); }
329         };
330     }
