1   package org.hibernate.sql;
2   
3   import org.hibernate.HibernateException;
4   
5   /**
6    * @author Strong Liu
7    */
8   
9   public enum JoinType {
10  	NONE( -666 ),
11  	INNER_JOIN( 0 ),
12  	LEFT_OUTER_JOIN( 1 ),
13  	RIGHT_OUTER_JOIN( 2 ),
14  	FULL_JOIN( 4 );
15  	private int joinTypeValue;
16  
17  	JoinType(int joinTypeValue) {
18  		this.joinTypeValue = joinTypeValue;
19  	}
20  
21  	public int getJoinTypeValue() {
22  		return joinTypeValue;
23  	}
24  
25  	public static JoinType parse(int joinType){
26  		if(joinType<0){
27  			return NONE;
28  		}
29  		switch ( joinType ){
30  			case 0:
31  				return INNER_JOIN;
32  			case 1:
33  				return LEFT_OUTER_JOIN;
34  			case 2:
35  				return RIGHT_OUTER_JOIN;
36  			case 4:
37  				return FULL_JOIN;
38  			default:
39  				throw new HibernateException( "unknown join type: "+joinType );
40  		}
41  	}
42  }
