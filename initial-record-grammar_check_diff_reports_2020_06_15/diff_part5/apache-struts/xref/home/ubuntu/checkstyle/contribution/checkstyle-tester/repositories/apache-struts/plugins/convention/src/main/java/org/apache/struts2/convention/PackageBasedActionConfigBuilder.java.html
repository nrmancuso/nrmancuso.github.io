95      /**
96       * Constructs actions based on a list of packages.
97       *
98       * @param configuration         The XWork configuration that the new package configs and action configs
99       *                              are added to.
100      * @param container             Xwork Container
101      * @param objectFactory         The ObjectFactory used to create the actions and such.
102      * @param redirectToSlash       A boolean parameter that controls whether or not this will create an
103      *                              action for indexes. If this is set to true, index actions are not created because
104      *                              the unknown handler will redirect from /foo to /foo/. The only action that is created
105      *                              is to the empty action in the namespace (e.g. the namespace /foo and the action "").
106      * @param enableSmiInheritance  A boolean parameter which determines if a newly created package config inherits the SMI value of its parent package config
107      * @param defaultParentPackage  The default parent package for all the configuration.
108      */
109     @Inject
110     public PackageBasedActionConfigBuilder(Configuration configuration, Container container, ObjectFactory objectFactory,
111                                            @Inject(ConventionConstants.CONVENTION_REDIRECT_TO_SLASH) String redirectToSlash,
112                                            @Inject(ConventionConstants.CONVENTION_DEFAULT_PARENT_PACKAGE) String defaultParentPackage,
113                                            @Inject(ConventionConstants.CONVENTION_ENABLE_SMI_INHERITANCE) String enableSmiInheritance) {
114 
115         // Validate that the parameters are okay
116         this.configuration = configuration;
117         this.actionNameBuilder = container.getInstance(ActionNameBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_ACTION_NAME_BUILDER));
118         this.resultMapBuilder = container.getInstance(ResultMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_RESULT_MAP_BUILDER));
119         this.interceptorMapBuilder = container.getInstance(InterceptorMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_INTERCEPTOR_MAP_BUILDER));
120         this.objectFactory = objectFactory;
121         this.redirectToSlash = Boolean.parseBoolean(redirectToSlash);
122         this.enableSmiInheritance = Boolean.parseBoolean(enableSmiInheritance);
123 
124         if (LOG.isTraceEnabled()) {
125             LOG.trace("Setting action default parent package to [{}]", defaultParentPackage);
126         }
127 
128         this.defaultParentPackage = defaultParentPackage;
129     }

303     /**
304      * Builds the action configurations by loading all classes in the packages specified by the
305      * property <b>struts.convention.action.packages</b> and then figuring out which classes implement Action
306      * or have Action in their name. Next, if this class is in a Java package that hasn't been
307      * inspected a new PackageConfig (XWork) is created for that Java package using the Java package
308      * name. This will contain all the ActionConfigs for all the Action classes that are discovered
309      * within that Java package. Next, each class is inspected for the {@link ParentPackage}
310      * annotation which is used to control the parent package for a specific action. Lastly, the
311      * {@link ResultMapBuilder} is used to create ResultConfig instances of the action.
312      */
313     public void buildActionConfigs() {
314         //setup reload class loader based on dev settings
315         initReloadClassLoader();
316 
317         if (!disableActionScanning) {
318             if (actionPackages == null && packageLocators == null) {
319                 throw new ConfigurationException("At least a list of action packages or action package locators " +
320                         "must be given using one of the properties [struts.convention.action.packages] or " +
321                         "[struts.convention.package.locators]");
322             }
323 
324             if (LOG.isTraceEnabled()) {
325                 LOG.trace("Loading action configurations");
326                 if (actionPackages != null) {
327                     LOG.trace("Actions being loaded from action packages: {}", actionPackages);
328                 }
329                 if (packageLocators != null) {
330                     LOG.trace("Actions being loaded using package locator's: {}", packageLocators);
331                 }
332                 if (excludePackages != null) {
333                     LOG.trace("Excluding actions from packages: {}", excludePackages);
334                 }
335             }
336 
337             Set<Class> classes = findActions();
338             buildConfiguration(classes);
339         }
340     }

342     protected ClassLoaderInterface getClassLoaderInterface() {
343         if (isReloadEnabled()) {
344             return new ClassLoaderInterfaceDelegate(this.reloadingClassLoader);
345         }
346         else {
347             /*
348             if there is a ClassLoaderInterface in the context, use it, otherwise
349             default to the default ClassLoaderInterface (a wrapper around the current
350             thread classloader)
351             using this, other plugins (like OSGi) can plugin their own classloader for a while
352             and it will be used by Convention (it cannot be a bean, as Convention is likely to be
353             called multiple times, and it needs to use the default ClassLoaderInterface during normal startup)
354             */
355             ClassLoaderInterface classLoaderInterface = null;
356             ActionContext ctx = ActionContext.getContext();
357             if (ctx != null) {
358                 classLoaderInterface = (ClassLoaderInterface) ctx.get(ClassLoaderInterface.CLASS_LOADER_INTERFACE);
359             }
360             return ObjectUtils.defaultIfNull(classLoaderInterface, new ClassLoaderInterfaceDelegate(getClassLoader()));
361         }
362     }

368     @SuppressWarnings("unchecked")
369     protected Set<Class> findActions() {
370         Set<Class> classes = new HashSet<>();
371         try {
372             if (actionPackages != null || (packageLocators != null && !disablePackageLocatorsScanning)) {
373 
374                 // By default, ClassFinder scans EVERY class in the specified
375                 // url set, which can produce spurious warnings for non-action
376                 // classes that can't be loaded. We pass a package filter that
377                 // only considers classes that match the action packages
378                 // specified by the user
379                 Test<String> classPackageTest = getClassPackageTest();
380                 List<URL> urls = readUrls();
381                 ClassFinder finder = buildClassFinder(classPackageTest, urls);
382 
383                 Test<ClassFinder.ClassInfo> test = getActionClassTest();
384                 classes.addAll(finder.findClasses(test));
385             }
386         } catch (Exception ex) {
387             LOG.error("Unable to scan named packages", ex);
388         }
389 
390         return classes;
391     }

393     protected ClassFinder buildClassFinder(Test<String> classPackageTest, List<URL> urls) {
394         if (classFinderFactory != null) {
395             LOG.trace("Using ClassFinderFactory to create instance of ClassFinder!");
396             return classFinderFactory.buildClassFinder(getClassLoaderInterface(), urls, EXTRACT_BASE_INTERFACES, fileProtocols, classPackageTest);
397         } else {
398             LOG.trace("ClassFinderFactory not defined, fallback to default ClassFinder implementation");
399             return new DefaultClassFinder(getClassLoaderInterface(), urls, EXTRACT_BASE_INTERFACES, fileProtocols, classPackageTest);
400         }
401     }

403     private List<URL> readUrls() throws IOException {
404         List<URL> resourceUrls = new ArrayList<>();
405         // Usually the "classes" dir.
406         ArrayList<URL> classesList = Collections.list(getClassLoaderInterface().getResources(""));
407         for (URL url : classesList) {
408             resourceUrls.addAll(fileManager.getAllPhysicalUrls(url));
409         }
410         return buildUrlSet(resourceUrls).getUrls();
411     }
412 
413     private UrlSet buildUrlSet(List<URL> resourceUrls) throws IOException {
414         ClassLoaderInterface classLoaderInterface = getClassLoaderInterface();
415         UrlSet urlSet = new UrlSet(resourceUrls);
416         urlSet = urlSet.include(new UrlSet(classLoaderInterface, this.fileProtocols));
417 
418         //excluding the urls found by the parent class loader is desired, but fails in JBoss (all urls are removed)
419         if (excludeParentClassLoader) {
420             //exclude parent of classloaders
421             ClassLoaderInterface parent = classLoaderInterface.getParent();
422             //if reload is enabled, we need to step up one level, otherwise the UrlSet will be empty
423             //this happens because the parent of the reloading class loader is the web app classloader
424             if (parent != null && isReloadEnabled())
425                 parent = parent.getParent();
426 
427             if (parent != null)
428                 urlSet = urlSet.exclude(parent);
429 
430             try {
431                 // This may fail in some sandboxes, ie GAE
432                 ClassLoader systemClassLoader = ClassLoader.getSystemClassLoader();
433                 urlSet = urlSet.exclude(new ClassLoaderInterfaceDelegate(systemClassLoader.getParent()));
434 
435             } catch (SecurityException e) {
436                 LOG.warn("Could not get the system classloader due to security constraints, there may be improper urls left to scan");
437             }
438         }
439 
440         //try to find classes dirs inside war files
441         urlSet = urlSet.includeClassesUrl(classLoaderInterface, new UrlSet.FileProtocolNormalizer() {
442             public URL normalizeToFileProtocol(URL url) {
443                 return fileManager.normalizeToFileProtocol(url);
444             }
445         });
446 
447 
448         urlSet = urlSet.excludeJavaExtDirs()
449                     .excludeJavaEndorsedDirs()
450                     .excludeUserExtensionsDir();
451 
452         try {
453         	urlSet = urlSet.excludeJavaHome();
454         } catch (NullPointerException e) {
455         	// This happens in GAE since the sandbox contains no java.home directory
456       	    LOG.warn("Could not exclude JAVA_HOME, is this a sandbox jvm?");
457         }
458         urlSet = urlSet.excludePaths(System.getProperty("sun.boot.class.path", ""));
459         urlSet = urlSet.exclude(".*/JavaVM.framework/.*");
460 
461         if (includeJars == null) {
462             urlSet = urlSet.exclude(".*?\\.jar(!/|/)?");
463         } else {
464             if(LOG.isDebugEnabled()) {
465                 LOG.debug("jar urls regexes were specified: {}", Arrays.asList(includeJars));
466             }
467             List<URL> rawIncludedUrls = urlSet.getUrls();
468             Set<URL> includeUrls = new HashSet<>();
469             boolean[] patternUsed = new boolean[includeJars.length];
470 
471             for (URL url : rawIncludedUrls) {
472                 if (fileProtocols.contains(url.getProtocol())) {
473                     //it is a jar file, make sure it matches at least a url regex
474                     for (int i = 0; i < includeJars.length; i++) {
475                         String includeJar = includeJars[i];
476                         if (Pattern.matches(includeJar, url.toExternalForm())) {
477                             includeUrls.add(url);
478                             patternUsed[i] = true;
479                             break;
480                         }
481                     }
482                 } else {
483                     LOG.debug("It is not a jar [{}]", url);
484                     includeUrls.add(url);
485                 }
486             }
487 
488             if (LOG.isWarnEnabled()) {
489                 for (int i = 0; i < patternUsed.length; i++) {
490                     if (!patternUsed[i]) {
491                         LOG.warn("The includeJars pattern [{}] did not match any jars in the classpath", includeJars[i]);
492                     }
493                 }
494             }
495             return new UrlSet(includeUrls);
496         }
497 
498         return urlSet;
499     }

501     /**
502      * Note that we can't include the test for {@link #actionSuffix} here
503      * because a class is included if its name ends in {@link #actionSuffix} OR
504      * it implements {@link com.opensymphony.xwork2.Action}. Since the whole
505      * goal is to avoid loading the class if we don't have to, the (actionSuffix
506      * || implements Action) test will have to remain until later. See
507      * {@link #getActionClassTest()} for the test performed on the loaded
508      * {@link ClassFinder.ClassInfo} structure.
509      *
510      * @param className the name of the class to test
511      * @return true if the specified class should be included in the
512      *         package-based action scan
513      */
514     protected boolean includeClassNameInActionScan(String className) {
515         String classPackageName = StringUtils.substringBeforeLast(className, ".");
516         return (checkActionPackages(classPackageName) || checkPackageLocators(classPackageName)) && checkExcludePackages(classPackageName);
517     }

519     /**
520      * Checks if provided class package is on the exclude list
521      *
522      * @param classPackageName name of class package
523      * @return false if class package is on the {@link #excludePackages} list
524      */
525     protected boolean checkExcludePackages(String classPackageName) {
526         if(excludePackages != null && excludePackages.length > 0) {
527             WildcardHelper wildcardHelper = new WildcardHelper();
528 
529             //we really don't care about the results, just the boolean
530             Map<String, String> matchMap = new HashMap<>();
531 
532             for(String packageExclude : excludePackages) {
533                 int[] packagePattern = wildcardHelper.compilePattern(packageExclude);
534                 if(wildcardHelper.match(matchMap, classPackageName, packagePattern)) {
535                     return false;
536                 }
537             }
538         }
539         return true;
540     }

542     /**
543      * Checks if class package match provided list of action packages
544      *
545      * @param classPackageName name of class package
546      * @return true if class package is on the {@link #actionPackages} list
547      */
548     protected boolean checkActionPackages(String classPackageName) {
549         if (actionPackages != null) {
550             for (String packageName : actionPackages) {
551                 String strictPackageName = packageName + ".";
552                 if (classPackageName.equals(packageName) || classPackageName.startsWith(strictPackageName)) {
553                     return true;
554                 }
555             }
556         }
557         return false;
558     }

560     /**
561      * Checks if class package match provided list of package locators
562      *
563      * @param classPackageName name of class package
564      * @return true if class package is on the {@link #packageLocators} list
565      */
566     protected boolean checkPackageLocators(String classPackageName) {
567         if (packageLocators != null && !disablePackageLocatorsScanning && classPackageName.length() > 0
568                 && (packageLocatorsBasePackage == null || classPackageName
569                         .startsWith(packageLocatorsBasePackage))) {
570             for (String packageLocator : packageLocators) {
571                 String[] splitted = classPackageName.split("\\.");
572 
573                 if (StringTools.contains(splitted, packageLocator, false)) {
574                     return true;
575                 }
576             }
577         }
578         return false;
579     }

584      * Note that the goal is to avoid loading the class, so the test should only

599     /**
600      * Construct a {@link Test} Object that determines if a specified class
601      * should be included in the package scan based on the full {@link ClassFinder.ClassInfo}
602      * of the class. At this point, the class has been loaded, so it's ok to
603      * perform tests such as checking annotations or looking at interfaces or
604      * super-classes of the specified class.
605      *
606      * @return a {@link Test} object that returns true if the specified class
607      *         should be included in the package scan
608      */
609     protected Test<ClassFinder.ClassInfo> getActionClassTest() {
610         return new Test<ClassFinder.ClassInfo>() {
611             public boolean test(ClassFinder.ClassInfo classInfo) {
612 
613                 // Why do we call includeClassNameInActionScan here, when it's
614                 // already been called to in the initial call to ClassFinder?
615                 // When some action class passes our package filter in that step,
616                 // ClassFinder automatically includes parent classes of that action,
617                 // such as com.opensymphony.xwork2.ActionSupport.  We repeat the
618                 // package filter here to filter out such results.
619                 boolean inPackage = includeClassNameInActionScan(classInfo.getName());
620                 boolean nameMatches = matchesSuffix(classInfo.getName());
621 
622                 try {
623                     return inPackage && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
624                 } catch (ClassNotFoundException ex) {
625                     LOG.error("Unable to load class [{}]", classInfo.getName(), ex);
626                     return false;
627                 }
628             }

631                 for (String suffix : actionSuffix) {
632                     if (name.endsWith(suffix)) {
633                         return true;
634                     }
635                 }
636                 return false;
637             }
638         };

641     @SuppressWarnings("unchecked")
642     protected void buildConfiguration(Set<Class> classes) {
643         Map<String, PackageConfig.Builder> packageConfigs = new HashMap<>();
644 
645         for (Class<?> actionClass : classes) {
646             Actions actionsAnnotation = actionClass.getAnnotation(Actions.class);
647             Action actionAnnotation = actionClass.getAnnotation(Action.class);
648 
649             // Skip classes that can't be instantiated
650             if (cannotInstantiate(actionClass)) {
651                 LOG.trace("Class [{}] did not pass the instantiation test and will be ignored", actionClass.getName());
652                 continue;
653             }
654 
655             if (eagerLoading) {
656                 // Tell the ObjectFactory about this class
657                 try {
658                     objectFactory.getClassInstance(actionClass.getName());
659                 } catch (ClassNotFoundException e) {
660                     LOG.error("Object Factory was unable to load class [{}]", actionClass.getName(), e);
661                     throw new StrutsException("Object Factory was unable to load class " + actionClass.getName(), e);
662                 }
663             }
664 
665             // Determine the action package
666             String actionPackage = actionClass.getPackage().getName();
667             LOG.debug("Processing class [{}] in package [{}]", actionClass.getName(), actionPackage);
668 
669             Set<String> allowedMethods = getAllowedMethods(actionClass);
670 
671             // Determine the default namespace and action name
672             List<String> namespaces = determineActionNamespace(actionClass);
673             for (String namespace : namespaces) {
674                 String defaultActionName = determineActionName(actionClass);
675                 PackageConfig.Builder defaultPackageConfig = getPackageConfig(packageConfigs, namespace,
676                         actionPackage, actionClass, null);
677 
678                 // Verify that the annotations have no errors and also determine if the default action
679                 // configuration should still be built or not.
680                 Map<String, List<Action>> map = getActionAnnotations(actionClass);
681                 Set<String> actionNames = new HashSet<>();
682                 boolean hasDefaultMethod = ReflectionTools.containsMethod(actionClass, DEFAULT_METHOD);
683                 if (!map.containsKey(DEFAULT_METHOD)
684                         && hasDefaultMethod
685                         && actionAnnotation == null && actionsAnnotation == null
686                         && (alwaysMapExecute || map.isEmpty())) {
687                     boolean found = false;
688                     for (List<Action> actions : map.values()) {
689                         for (Action action : actions) {
690 
691                             // Check if there are duplicate action names in the annotations.
692                             String actionName = action.value().equals(Action.DEFAULT_VALUE) ? defaultActionName : action.value();
693                             if (actionNames.contains(actionName)) {
694                                 throw new ConfigurationException("The action class [" + actionClass +
695                                         "] contains two methods with an action name annotation whose value " +
696                                         "is the same (they both might be empty as well).");
697                             } else {
698                                 actionNames.add(actionName);
699                             }
700 
701                             // Check this annotation is the default action
702                             if (action.value().equals(Action.DEFAULT_VALUE)) {
703                                 found = true;
704                             }
705                         }
706                     }
707 
708                     // Build the default
709                     if (!found) {
710                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, DEFAULT_METHOD, null, allowedMethods);
711                     }
712                 }
713 
714                 // Build the actions for the annotations
715                 for (Map.Entry<String, List<Action>> entry : map.entrySet()) {
716                     String method = entry.getKey();
717                     List<Action> actions = entry.getValue();
718                     for (Action action : actions) {
719                         PackageConfig.Builder pkgCfg = defaultPackageConfig;
720                         if (action.value().contains("/") && !slashesInActionNames) {
721                             pkgCfg = getPackageConfig(packageConfigs, namespace, actionPackage,
722                                     actionClass, action);
723                         }
724 
725                         createActionConfig(pkgCfg, actionClass, defaultActionName, method, action, allowedMethods);
726                     }
727                 }
728 
729                 // some actions will not have any @Action or a default method, like the rest actions
730                 // where the action mapper is the one that finds the right method at runtime
731                 if (map.isEmpty() && mapAllMatches && actionAnnotation == null && actionsAnnotation == null) {
732                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, null, actionAnnotation, allowedMethods);
733                 }
734 
735                 //if there are @Actions or @Action at the class level, create the mappings for them
736                 String methodName = hasDefaultMethod ? DEFAULT_METHOD : null;
737                 if (actionsAnnotation != null) {
738                     List<Action> actionAnnotations = checkActionsAnnotation(actionsAnnotation);
739                     for (Action actionAnnotation2 : actionAnnotations)
740                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation2, allowedMethods);
741                 } else if (actionAnnotation != null)
742                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation, allowedMethods);
743             }
744         }
745 
746         buildIndexActions(packageConfigs);
747 
748         // Add the new actions to the configuration
749         Set<String> packageNames = packageConfigs.keySet();
750         for (String packageName : packageNames) {
751             configuration.addPackageConfig(packageName, packageConfigs.get(packageName).build());
752         }
753     }

755     private Set<String> getAllowedMethods(Class<?> actionClass) {
756         List<AllowedMethods> annotations = AnnotationUtils.findAnnotations(actionClass, AllowedMethods.class);
757         if (annotations == null || annotations.isEmpty()) {
758             return Collections.emptySet();
759         } else {
760             Set<String> methods = new HashSet<>();
761             for (AllowedMethods allowedMethods : annotations) {
762                 methods.addAll(Arrays.asList(allowedMethods.value()));
763             }
764             return methods;
765         }

778     /**
779      * Determines the namespace(s) for the action based on the action class. If there is a {@link Namespace}
780      * annotation on the class (including parent classes) or on the package that the class is in, than
781      * it is used. Otherwise, the Java package name that the class is in is used in conjunction with
782      * either the <b>struts.convention.action.packages</b> or <b>struts.convention.package.locators</b>
783      * configuration values. These are used to determine which part of the Java package name should
784      * be converted into the namespace for the XWork PackageConfig.
785      *
786      * @param actionClass The action class.
787      * @return The namespace or an empty string.
788      */
789     protected List<String> determineActionNamespace(Class<?> actionClass) {
790         List<String> namespaces = new ArrayList<>();
791 
792         // Check if there is a class or package level annotation for the namespace
793         //single namespace
794         Namespace namespaceAnnotation = AnnotationUtils.findAnnotation(actionClass, Namespace.class);
795         if (namespaceAnnotation != null) {
796             LOG.trace("Using non-default action namespace from Namespace annotation of [{}]", namespaceAnnotation.value());
797 
798             namespaces.add(namespaceAnnotation.value());
799         }
800 
801         //multiple annotations
802         Namespaces namespacesAnnotation = AnnotationUtils.findAnnotation(actionClass, Namespaces.class);
803         if (namespacesAnnotation != null) {
804             if (LOG.isTraceEnabled()) {
805                 StringBuilder sb = new StringBuilder();
806                 for (Namespace namespace : namespacesAnnotation.value())
807                     sb.append(namespace.value()).append(",");
808                 sb.deleteCharAt(sb.length() - 1);
809                 LOG.trace("Using non-default action namespaces from Namespaces annotation of [{}]", sb.toString());
810             }
811 
812             for (Namespace namespace : namespacesAnnotation.value()) {
813                 namespaces.add(namespace.value());
814             }
815         }
816 
817         //don't use default if there are annotations
818         if (!namespaces.isEmpty())
819             return namespaces;
820 
821         String pkg = actionClass.getPackage().getName();
822         String pkgPart = null;
823         if (actionPackages != null) {
824             for (String actionPackage : actionPackages) {
825                 if (pkg.startsWith(actionPackage)) {
826                     pkgPart = actionClass.getName().substring(actionPackage.length() + 1);
827                 }
828             }
829         }
830 
831         if (pkgPart == null && packageLocators != null) {
832             for (String packageLocator : packageLocators) {
833                 // check subpackage and not a part of package name, eg. actions -> my.actions.transactions - WW-3803
834                 int index = pkg.lastIndexOf("." + packageLocator + ".");
835 
836                 // This ensures that the match is at the end, beginning or has a dot on each side of it
837                 if (index >= 0 && (index + packageLocator.length() == pkg.length() || index == 0 ||
838                         (pkg.charAt(index) == '.' && pkg.charAt(index + 1 + packageLocator.length()) == '.'))) {
839                     pkgPart = actionClass.getName().substring(index + packageLocator.length() + 2);
840                 }
841             }
842         }
843 
844         if (pkgPart != null) {
845             final int indexOfDot = pkgPart.lastIndexOf('.');
846             if (indexOfDot >= 0) {
847                 String convertedNamespace = actionNameBuilder.build(pkgPart.substring(0, indexOfDot));
848                 namespaces.add("/" + convertedNamespace.replace('.', '/'));
849                 return namespaces;
850             }
851         }
852 
853         namespaces.add("");
854         return namespaces;
855     }

870     /**
871      * Locates all of the {@link Actions} and {@link Action} annotations on methods within the Action
872      * class and its parent classes.
873      *
874      * @param actionClass The action class.
875      * @return The list of annotations or an empty list if there are none.
876      */
877     protected Map<String, List<Action>> getActionAnnotations(Class<?> actionClass) {
878         Method[] methods = actionClass.getMethods();
879         Map<String, List<Action>> map = new HashMap<>();
880         for (Method method : methods) {
881             Actions actionsAnnotation = method.getAnnotation(Actions.class);
882             if (actionsAnnotation != null) {
883                 List<Action> actions = checkActionsAnnotation(actionsAnnotation);
884                 map.put(method.getName(), actions);
885             } else {
886                 Action ann = method.getAnnotation(Action.class);
887                 if (ann != null) {
888                     map.put(method.getName(), Arrays.asList(ann));
889                 }
890             }
891         }
892 
893         return map;
894     }

896     /**
897      *  Builds a list of actions from an @Actions annotation, and check that they are not all empty
898      * @param actionsAnnotation Actions annotation
899      * @return a list   of Actions
900      */
901     protected List<Action> checkActionsAnnotation(Actions actionsAnnotation) {
902         Action[] actionArray = actionsAnnotation.value();
903         boolean valuelessSeen = false;
904         List<Action> actions = new ArrayList<>();
905         for (Action ann : actionArray) {
906             if (ann.value().equals(Action.DEFAULT_VALUE) && !valuelessSeen) {
907                 valuelessSeen = true;
908             } else if (ann.value().equals(Action.DEFAULT_VALUE)) {
909                 throw new ConfigurationException("You may only add a single Action annotation that has no value parameter.");
910             }
911 
912             actions.add(ann);
913         }
914         return actions;
915     }

917     /**
918      * Creates a single ActionConfig object.
919      *
920      * @param pkgCfg       The package the action configuration instance will belong to.
921      * @param actionClass  The action class.
922      * @param actionName   The name of the action.
923      * @param actionMethod The method that the annotation was on (if the annotation is not null) or
924      *                     the default method (execute).
925      * @param annotation   The ActionName annotation that might override the action name and possibly
926      */
927     protected void createActionConfig(PackageConfig.Builder pkgCfg, Class<?> actionClass, String actionName,
928                                       String actionMethod, Action annotation, Set<String> allowedMethods) {
929     	String className = actionClass.getName();
930         if (annotation != null) {
931             actionName = annotation.value().equals(Action.DEFAULT_VALUE) ? actionName : annotation.value();
932             actionName = StringUtils.contains(actionName, "/") && !slashesInActionNames ? StringUtils.substringAfterLast(actionName, "/") : actionName;
933             if(!Action.DEFAULT_VALUE.equals(annotation.className())){
934             	className = annotation.className();
935             }
936         }
937 
938         ActionConfig.Builder actionConfig = new ActionConfig.Builder(pkgCfg.getName(), actionName, className);
939         actionConfig.methodName(actionMethod);
940 
941         if (pkgCfg.isStrictMethodInvocation()) {
942             actionConfig.addAllowedMethod(actionMethod);
943             actionConfig.addAllowedMethod(allowedMethods);
944             actionConfig.addAllowedMethod(pkgCfg.getGlobalAllowedMethods());
945         } else {
946             actionConfig.addAllowedMethod(ActionConfig.WILDCARD);
947         }
948 
949         if (LOG.isDebugEnabled()) {
950             LOG.debug("Creating action config for class [{}], name [{}] and package name [{}] in namespace [{}]",
951                     actionClass.toString(), actionName, pkgCfg.getName(), pkgCfg.getNamespace());
952         }
953 
954         //build interceptors
955         List<InterceptorMapping> interceptors = interceptorMapBuilder.build(actionClass, pkgCfg, actionName, annotation);
956         actionConfig.addInterceptors(interceptors);
957 
958         //build results
959         Map<String, ResultConfig> results = resultMapBuilder.build(actionClass, annotation, actionName, pkgCfg.build());
960         actionConfig.addResultConfigs(results);
961 
962         //add params
963         if (annotation != null)
964             actionConfig.addParams(StringTools.createParameterMap(annotation.params()));
965 
966         //add exception mappings from annotation
967         if (annotation != null)
968             actionConfig.addExceptionMappings(buildExceptionMappings(annotation.exceptionMappings(), actionName));
969 
970         //add exception mapping from class
971         ExceptionMappings exceptionMappings = actionClass.getAnnotation(ExceptionMappings.class);
972         if (exceptionMappings != null)
973             actionConfig.addExceptionMappings(buildExceptionMappings(exceptionMappings.value(), actionName));
974 
975         //add
976         pkgCfg.addActionConfig(actionName, actionConfig.build());
977 
978         //check if an action with the same name exists on that package (from XML config probably)
979         PackageConfig existingPkg = configuration.getPackageConfig(pkgCfg.getName());
980         if (existingPkg != null) {
981             // there is a package already with that name, check action
982             ActionConfig existingActionConfig = existingPkg.getActionConfigs().get(actionName);
983             if (existingActionConfig != null && LOG.isWarnEnabled()) {
984                 LOG.warn("Duplicated action definition in package [{}] with name [{}].", pkgCfg.getName(), actionName);
985             }
986         }
987 
988         //watch class file
989         if (isReloadEnabled()) {
990             URL classFile = actionClass.getResource(actionClass.getSimpleName() + ".class");
991             fileManager.monitorFile(classFile);
992             loadedFileUrls.add(classFile.toString());
993         }
994     }

996     protected List<ExceptionMappingConfig> buildExceptionMappings(ExceptionMapping[] exceptions, String actionName) {
997         List<ExceptionMappingConfig> exceptionMappings = new ArrayList<>();
998 
999         for (ExceptionMapping exceptionMapping : exceptions) {
1000             LOG.trace("Mapping exception [{}] to result [{}] for action [{}]", exceptionMapping.exception(),
1001                         exceptionMapping.result(), actionName);
1002             ExceptionMappingConfig.Builder builder = new ExceptionMappingConfig.Builder(null, exceptionMapping
1003                     .exception(), exceptionMapping.result());
1004             builder.addParams(StringTools.createParameterMap(exceptionMapping.params()));
1005             exceptionMappings.add(builder.build());
1006         }
1007 
1008         return exceptionMappings;

1011     protected PackageConfig.Builder getPackageConfig(final Map<String, PackageConfig.Builder> packageConfigs,
1012                                                    String actionNamespace, final String actionPackage, final Class<?> actionClass,
1013                                                    Action action) {
1014         if (action != null && !action.value().equals(Action.DEFAULT_VALUE)) {
1015             LOG.trace("Using non-default action namespace from the Action annotation of [{}]", action.value());
1016             String actionName = action.value();
1017             actionNamespace = StringUtils.contains(actionName, "/") ? StringUtils.substringBeforeLast(actionName, "/") : StringUtils.EMPTY;
1018         }
1019 
1020         // Next grab the parent annotation from the class
1021         ParentPackage parent = AnnotationUtils.findAnnotation(actionClass, ParentPackage.class);
1022         String parentName = null;
1023         if (parent != null) {
1024             LOG.trace("Using non-default parent package from annotation of [{}]", parent.value());
1025             parentName = parent.value();
1026         }
1027 
1028         // Finally use the default
1029         if (parentName == null) {
1030             parentName = defaultParentPackage;
1031         }
1032 
1033         if (parentName == null) {
1034             throw new ConfigurationException("Unable to determine the parent XWork package for the action class [" +
1035                     actionClass.getName() + "]");
1036         }
1037 
1038         PackageConfig parentPkg = configuration.getPackageConfig(parentName);
1039         if (parentPkg == null) {
1040             throw new ConfigurationException("Unable to locate parent package [" + parentName + "] for [" + actionClass + "]");
1041         }
1042 
1043         // Grab based on package-namespace and if it exists, we need to ensure the existing one has
1044         // the correct parent package. If not, we need to create a new package config
1045         String name = actionPackage + "#" + parentPkg.getName() + "#" + actionNamespace;
1046         PackageConfig.Builder pkgConfig = packageConfigs.get(name);
1047         if (pkgConfig == null) {
1048             pkgConfig = new PackageConfig.Builder(name).namespace(actionNamespace).addParent(parentPkg);
1049             if (enableSmiInheritance) {
1050                 pkgConfig.strictMethodInvocation(parentPkg.isStrictMethodInvocation());
1051             }
1052             packageConfigs.put(name, pkgConfig);
1053 
1054             //check for @DefaultInterceptorRef in the package
1055             DefaultInterceptorRef defaultInterceptorRef = AnnotationUtils.findAnnotation(actionClass, DefaultInterceptorRef.class);
1056             if (defaultInterceptorRef != null) {
1057                 pkgConfig.defaultInterceptorRef(defaultInterceptorRef.value());
1058 
1059                 LOG.trace("Setting [{}] as the default interceptor ref for [{}]", defaultInterceptorRef.value(), pkgConfig.getName());
1060             }
1061         }
1062 
1063         LOG.trace("Created package config named [{}] with a namespace [{}]", name, actionNamespace);
1064 
1065         return pkgConfig;
1066     }

1068     /**
1069      * Determine all the index handling actions and results based on this logic:
1070      *
1071      * 1. Loop over all the namespaces such as /foo and see if it has an action named index
1072      * 2. If an action doesn't exists in the parent namespace of the same name, create an action
1073      * in the parent namespace of the same name as the namespace that points to the index
1074      * action in the namespace. e.g. /foo -&gt; /foo/index
1075      * 3. Create the action in the namespace for empty string if it doesn't exist. e.g. /foo/
1076      * the action is "" and the namespace is /foo
1077      *
1078      * @param packageConfigs Used to store the actions.
1079      */
1080     protected void buildIndexActions(Map<String, PackageConfig.Builder> packageConfigs) {
1081         Map<String, PackageConfig.Builder> byNamespace = new HashMap<>();
1082         Collection<PackageConfig.Builder> values = packageConfigs.values();
1083         for (PackageConfig.Builder packageConfig : values) {
1084             byNamespace.put(packageConfig.getNamespace(), packageConfig);
1085         }
1086 
1087         // Step #1
1088         Set<String> namespaces = byNamespace.keySet();
1089         for (String namespace : namespaces) {
1090             // First see if the namespace has an index action
1091             PackageConfig.Builder pkgConfig = byNamespace.get(namespace);
1092             ActionConfig indexActionConfig = pkgConfig.build().getAllActionConfigs().get("index");
1093             if (indexActionConfig == null) {
1094                 continue;
1095             }
1096 
1097             // Step #2
1098             if (!redirectToSlash) {
1099                 int lastSlash = namespace.lastIndexOf('/');
1100                 if (lastSlash >= 0) {
1101                     String parentAction = namespace.substring(lastSlash + 1);
1102                     String parentNamespace = namespace.substring(0, lastSlash);
1103                     PackageConfig.Builder parent = byNamespace.get(parentNamespace);
1104                     if (parent == null || parent.build().getAllActionConfigs().get(parentAction) == null) {
1105                         if (parent == null) {
1106                             parent = new PackageConfig.Builder(parentNamespace).namespace(parentNamespace).
1107                                     addParents(pkgConfig.build().getParents());
1108                             packageConfigs.put(parentNamespace, parent);
1109                         }
1110 
1111                         if (parent.build().getAllActionConfigs().get(parentAction) == null) {
1112                             parent.addActionConfig(parentAction, indexActionConfig);
1113                         }
1114                     } else {
1115                         LOG.trace("The parent namespace [{}] already contains an action [{}]", parentNamespace, parentAction);
1116                     }
1117                 }
1118             }
1119 
1120             // Step #3
1121             if (pkgConfig.build().getAllActionConfigs().get("") == null) {
1122                 LOG.trace("Creating index ActionConfig with an action name of [] for the action class [{}]", indexActionConfig.getClassName());
1123                 pkgConfig.addActionConfig("", indexActionConfig);
1124             }
1125         }
1126     }
