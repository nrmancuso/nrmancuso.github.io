58  /**
59   * MapReduceTask is a distributed task allowing a large scale computation to be transparently
60   * parallelized across Infinispan cluster nodes.
61   * <p>
62   *
63   * Users should instantiate MapReduceTask with a reference to a cache whose data is used as input for this
64   * task. Infinispan execution environment will migrate and execute instances of provided {@link Mapper}
65   * and {@link Reducer} seamlessly across Infinispan nodes.
66   * <p>
67   *
68   * Unless otherwise specified using {@link MapReduceTask#onKeys(Object...)} filter all available
69   * key/value pairs of a specified cache will be used as input data for this task.
70   *
71   * For example, MapReduceTask that counts number of word occurrences in a particular cache where
72   * keys and values are String instances could be written as follows:
73   *
74   * <pre>
75   * MapReduceTask&lt;String, String, String, Integer&gt; task = new MapReduceTask&lt;String, String, String, Integer&gt;(cache);
76   * task.mappedWith(new WordCountMapper()).reducedWith(new WordCountReducer());
77   * Map&lt;String, Integer&gt; results = task.execute();
78   * </pre>
79   *
80   * The final result is a map where key is a word and value is a word count for that particular word.
81   * <p>
82   *
83   * Accompanying {@link Mapper} and {@link Reducer} are defined as follows:
84   *
85   * <pre>
86   *    private static class WordCountMapper implements Mapper&lt;String, String, String,Integer&gt; {
87   *
88   *     public void map(String key, String value, Collector&lt;String, Integer&gt; collector) {
89   *        StringTokenizer tokens = new StringTokenizer(value);
90   *       while (tokens.hasMoreElements()) {
91   *           String s = (String) tokens.nextElement();
92   *           collector.emit(s, 1);
93   *        }
94   *     }
95   *  }
96   *
97   *   private static class WordCountReducer implements Reducer&lt;String, Integer&gt; {
98   *
99   *      public Integer reduce(String key, Iterator&lt;Integer&gt; iter) {
100  *         int sum = 0;
101  *         while (iter.hasNext()) {
102  *            Integer i = (Integer) iter.next();
103  *            sum += i;
104  *        }
105  *         return sum;
106  *      }
107  *   }
108  * </pre>
109  *
110  * <p>
111  *
112  * Finally, as of Infinispan 5.2 release, MapReduceTask can also specify a Combiner function. The Combiner
113  * is executed on each node after the Mapper and before the global reduce phase. The Combiner receives input from
114  * the Mapper's output and the output from the Combiner is then sent to the reducers. It is useful to think
115  * of the Combiner as a node local reduce phase before global reduce phase is executed.
116  * <p>
117  *
118  * Combiners are especially useful when reduce function is both commutative and associative! In such cases
119  * we can use the Reducer itself as the Combiner; all one needs to do is to specify the Combiner:
120  * <pre>
121  * MapReduceTask&lt;String, String, String, Integer&gt; task = new MapReduceTask&lt;String, String, String, Integer&gt;(cache);
122  * task.mappedWith(new WordCountMapper()).reducedWith(new WordCountReducer()).combineWith(new WordCountReducer());
123  * Map&lt;String, Integer&gt; results = task.execute();
124  * </pre>
125  *
126  * Note that {@link Mapper} and {@link Reducer} should not be specified as inner classes. Inner classes
127  * declared in non-static contexts contain implicit non-transient references to enclosing class instances,
128  * serializing such an inner class instance will result in serialization of its associated outer class instance as well.
129  *
130  * <p>
131  *
132  * If you are not familiar with concept of map reduce distributed execution model
133  * start with Google's MapReduce research <a href="http://labs.google.com/papers/mapreduce.html">paper</a>.
134  *
135  *
136  * @author Manik Surtani
137  * @author Vladimir Blagojevic
138  * @author Sanne Grinovero
139  *
140  * @since 5.0
141  */
142 public class MapReduceTask<KIn, VIn, KOut, VOut> {

199    /**
200     * Create a new MapReduceTask given a master cache node. All distributed task executions will be
201     * initiated from this cache node.
202     *
203     * @param masterCacheNode
204     *           cache node initiating map reduce task
205     * @param distributeReducePhase
206     *           if true this task will use distributed reduce phase execution
207     * @param useIntermediateSharedCache
208     *           if true this tasks will share intermediate value cache with other executing
209     *           MapReduceTasks on the grid. Otherwise, if false, this task will use its own
210     *           dedicated cache for intermediate values
211     *
212     */
213    public MapReduceTask(Cache<KIn, VIn> masterCacheNode, boolean distributeReducePhase, boolean useIntermediateSharedCache) {
214       if (masterCacheNode == null)
215          throw new IllegalArgumentException("Can not use null cache for MapReduceTask");
216       ensureAccessPermissions(masterCacheNode.getAdvancedCache());
217       ensureProperCacheState(masterCacheNode.getAdvancedCache());
218       this.cache = masterCacheNode.getAdvancedCache();
219       this.keys = new LinkedList<KIn>();
220       ComponentRegistry componentRegistry = SecurityActions.getCacheComponentRegistry(cache);
221       this.marshaller = componentRegistry.getComponent(StreamingMarshaller.class, CACHE_MARSHALLER);
222       this.mapReduceManager = componentRegistry.getComponent(MapReduceManager.class);
223       this.cancellationService = componentRegistry.getComponent(CancellationService.class);
224       this.stateTransferManager = componentRegistry.getComponent(StateTransferManager.class);
225       this.taskId = UUID.randomUUID();
226       if (useIntermediateSharedCache) {
227          this.customIntermediateCacheName = DEFAULT_TMP_CACHE_CONFIGURATION_NAME;
228       } else {
229          this.customIntermediateCacheName = taskId.toString();
230       }
231       this.distributeReducePhase = distributeReducePhase;
232       this.useIntermediateSharedCache = useIntermediateSharedCache;
233       this.cancellableTasks = Collections.synchronizedList(new ArrayList<CancellableTaskPart>());
234       this.clusteringDependentLogic = componentRegistry.getComponent(ClusteringDependentLogic.class);
235       this.isLocalOnly = SecurityActions.getCacheRpcManager(cache) == null;
236       this.rpcOptionsBuilder = isLocalOnly ? null : new RpcOptionsBuilder(SecurityActions.getCacheRpcManager(cache).getDefaultRpcOptions(true));
237       if (!isLocalOnly) {
238          this.rpcOptionsBuilder.timeout(0, TimeUnit.MILLISECONDS);
239       }
240    }

458    protected Map<KOut, VOut> executeHelper(String resultCache) throws NullPointerException, MapReduceException {
459       ensureAccessPermissions(cache);
460       if (mapper == null)
461          throw new NullPointerException("A valid reference of Mapper is not set " + mapper);
462 
463       if (reducer == null)
464          throw new NullPointerException("A valid reference of Reducer is not set " + reducer);
465 
466       Map<KOut,VOut> result = null;
467       if(!isLocalOnly && distributeReducePhase()){
468          // init and create tmp caches
469          executeTaskInit(getIntermediateCacheName());
470          Set<KOut> allMapPhasesResponses = null;
471          try {
472             // map
473             allMapPhasesResponses = executeMapPhase();
474 
475             // reduce
476             result = executeReducePhase(resultCache, allMapPhasesResponses, useIntermediateSharedCache());
477          } catch (InterruptedException ie) {
478             Thread.currentThread().interrupt();
479             throw new MapReduceException(ie);
480          }
481          finally {
482             // cleanup tmp caches across cluster
483             EmbeddedCacheManager cm = cache.getCacheManager();
484             String intermediateCache = getIntermediateCacheName();
485             if (useIntermediatePerTaskCache()) {
486                cm.removeCache(intermediateCache);
487             } else {
488                //let's make sure shared cache is not destroyed and that we have keys to remove
489                Cache<KOut, VOut> sharedTmpCache = cm.getCache(intermediateCache);
490                if (sharedTmpCache != null && allMapPhasesResponses != null) {
491                   for (KOut k : allMapPhasesResponses) {
492                      sharedTmpCache.removeAsync(new IntermediateKey<KOut>(taskId.toString(), k));
493                   }
494                }
495             }
496          }
497       } else {
498          try {
499             if (resultCache == null || resultCache.isEmpty()) {
500                result = new HashMap<KOut, VOut>();
501                executeMapPhaseWithLocalReduction(result);
502             } else {
503                EmbeddedCacheManager cm = cache.getCacheManager();
504                Cache<KOut, VOut> c = cm.getCache(resultCache);
505                executeMapPhaseWithLocalReduction(c);
506             }
507          } catch (InterruptedException ie) {
508             Thread.currentThread().interrupt();
509             throw new MapReduceException(ie);
510          }
511       }
512       return result;
513    }

532    protected void executeTaskInit(String tmpCacheName) throws MapReduceException {
533       RpcManager rpc = cache.getRpcManager();
534       CommandsFactory factory = cache.getComponentRegistry().getComponent(CommandsFactory.class);
535 
536       //first create tmp caches on all nodes
537       final CreateCacheCommand ccc = factory.buildCreateCacheCommand(tmpCacheName, intermediateCacheConfigurationName, true, rpc.getMembers().size());
538 
539       log.debugf("Invoking %s across members %s ", ccc, cache.getRpcManager().getMembers());
540       Future<Object> future = mapReduceManager.getExecutorService().submit(new Callable<Object>() {
541          @Override
542          public Object call() throws Exception {
543             //locally
544             ccc.init(cache.getCacheManager(), stateTransferManager);
545             try {
546                return ccc.perform(null);
547             } catch (Throwable e) {
548                throw new MapReduceException("Could not initialize temporary caches for MapReduce task on remote nodes ", e);
549             }
550          }
551       });
552       try {
553          future.get();
554       } catch (Exception e) {
555          throw new MapReduceException(e);
556       }      
557       Map<Address, Response> map = rpc.invokeRemotely(cache.getRpcManager().getMembers(), ccc, rpcOptionsBuilder.build());
558       for (Entry<Address, Response> e : map.entrySet()) {
559          if (!e.getValue().isSuccessful()) {
560             throw new MapReduceException("Could not initialize tmp cache " + tmpCacheName + " at " + e.getKey()
561                   + " for  " + this);
562          }
563       }
564    }

566    protected Set<KOut> executeMapPhase() throws MapReduceException, InterruptedException {
567       RpcManager rpc = cache.getRpcManager();
568       MapCombineCommand<KIn, VIn, KOut, VOut> cmd = null;
569       Set<KOut> mapPhasesResult = new HashSet<KOut>();
570       List<MapTaskPart<Set<KOut>>> futures = new ArrayList<MapTaskPart<Set<KOut>>>();
571       if (inputTaskKeysEmpty()) {
572          for (Address target : rpc.getMembers()) {
573             if (target.equals(rpc.getAddress())) {
574                cmd = buildMapCombineCommand(taskId.toString(), clone(mapper), clone(combiner),
575                      getIntermediateCacheName(), null, true, useIntermediateSharedCache());
576             } else {
577                cmd = buildMapCombineCommand(taskId.toString(), mapper, combiner, getIntermediateCacheName(), null,
578                      true, useIntermediateSharedCache());
579             }
580             MapTaskPart<Set<KOut>> part = createTaskMapPart(cmd, target, true);
581             part.execute();
582             futures.add(part);
583          }
584       } else {
585          Map<Address, ? extends Collection<KIn>> keysToNodes = mapKeysToNodes(keys);
586          for (Entry<Address, ? extends Collection<KIn>> e : keysToNodes.entrySet()) {
587             Address address = e.getKey();
588             Collection<KIn> keys = e.getValue();
589             if (address.equals(rpc.getAddress())) {
590                cmd = buildMapCombineCommand(taskId.toString(), clone(mapper), clone(combiner),
591                      getIntermediateCacheName(), keys, true, useIntermediateSharedCache());
592             } else {
593                cmd = buildMapCombineCommand(taskId.toString(), mapper, combiner, getIntermediateCacheName(), keys,
594                      true, useIntermediateSharedCache());
595             }
596             MapTaskPart<Set<KOut>> part = createTaskMapPart(cmd, address, true);
597             part.execute();
598             futures.add(part);
599          }
600       }
601       try {
602          for (MapTaskPart<Set<KOut>> mapTaskPart : futures) {
603             Set<KOut> result = null;
604             try {
605                result = mapTaskPart.get();
606             } catch (ExecutionException ee) {
607                throw new MapReduceException("Map phase failed ", ee.getCause());
608             }
609             mapPhasesResult.addAll(result);
610          }
611       } finally {
612          cancellableTasks.clear();
613       }
614       return mapPhasesResult;

617    protected void executeMapPhaseWithLocalReduction(Map<KOut, VOut> reducedResult) throws MapReduceException, InterruptedException {
618       RpcManager rpc = SecurityActions.getCacheRpcManager(cache);
619       MapCombineCommand<KIn, VIn, KOut, VOut> cmd = null;
620       Map<KOut, List<VOut>> mapPhasesResult = new HashMap<KOut, List<VOut>>();
621       List<MapTaskPart<Map<KOut, List<VOut>>>> futures = new ArrayList<MapTaskPart<Map<KOut, List<VOut>>>>();
622       Address localAddress = clusteringDependentLogic.getAddress();
623       if (inputTaskKeysEmpty()) {
624          List<Address> targets;
625          if (isLocalOnly) {
626             targets = Collections.singletonList(localAddress);
627          } else {
628             targets = rpc.getMembers();
629          }
630          for (Address target : targets) {
631             if (target.equals(localAddress)) {
632                cmd = buildMapCombineCommand(taskId.toString(), clone(mapper), clone(combiner),
633                      getIntermediateCacheName(), null, false, false);
634             } else {
635                cmd = buildMapCombineCommand(taskId.toString(), mapper, combiner, getIntermediateCacheName(), null,
636                      false, false);
637             }
638             MapTaskPart<Map<KOut, List<VOut>>> part = createTaskMapPart(cmd, target, false);
639             part.execute();
640             futures.add(part);
641          }
642       } else {
643          Map<Address, ? extends Collection<KIn>> keysToNodes = mapKeysToNodes(keys);
644          for (Entry<Address, ? extends Collection<KIn>> e : keysToNodes.entrySet()) {
645             Address address = e.getKey();
646             Collection<KIn> keys = e.getValue();
647             if (address.equals(localAddress)) {
648                cmd = buildMapCombineCommand(taskId.toString(), clone(mapper), clone(combiner),
649                      getIntermediateCacheName(), keys, false, false);
650             } else {
651                cmd = buildMapCombineCommand(taskId.toString(), mapper, combiner, getIntermediateCacheName(), keys,
652                      false, false);
653             }
654             MapTaskPart<Map<KOut, List<VOut>>> part = createTaskMapPart(cmd, address, false);
655             part.execute();
656             futures.add(part);
657          }
658       }
659       try {
660          for (MapTaskPart<Map<KOut, List<VOut>>> mapTaskPart : futures) {
661             Map<KOut, List<VOut>> result = null;
662             try {
663                result = mapTaskPart.get();
664             } catch (ExecutionException ee) {
665                throw new MapReduceException("Map phase failed ", ee.getCause());
666             }
667             mergeResponse(mapPhasesResult, result);
668          }
669       } finally {
670          cancellableTasks.clear();
671       }

674       MapReduceTaskLifecycleService taskLifecycleService = MapReduceTaskLifecycleService
675                .getInstance();

677       try {
678          taskLifecycleService.onPreExecute(reducer, cache);
679          for (Entry<KOut, List<VOut>> e : mapPhasesResult.entrySet()) {
680             // TODO in parallel with futures
681             reducedResult.put(e.getKey(), reducer.reduce(e.getKey(), e.getValue().iterator()));
682          }
683       } finally {
684          taskLifecycleService.onPostExecute(reducer);
685       }
686    }

695    protected Map<KOut, VOut> executeReducePhase(String resultCache, Set<KOut> allMapPhasesResponses,
696             boolean useIntermediateSharedCache) throws MapReduceException, InterruptedException {
697       RpcManager rpc = cache.getRpcManager();
698       String destCache = getIntermediateCacheName();
699 
700       Cache<Object, Object> dstCache = cache.getCacheManager().getCache(destCache);
701       Map<Address, ? extends Collection<KOut>> keysToNodes = mapKeysToNodes(dstCache.getAdvancedCache()
702                .getDistributionManager(), allMapPhasesResponses, useIntermediateSharedCache);
703       Map<KOut, VOut> reduceResult = new HashMap<KOut, VOut>();
704       List<ReduceTaskPart<Map<KOut, VOut>>> reduceTasks = new ArrayList<ReduceTaskPart<Map<KOut, VOut>>>();

716          ReduceTaskPart<Map<KOut, VOut>> part = createReducePart(reduceCommand, address, destCache);

724                result = reduceTaskPart.get();

729          }
730       } finally {
731          cancellableTasks.clear();
732       }

743    private <K, V> void mergeResponse(Map<K, List<V>> result, Map<K, List<V>> m) {
744       for (Entry<K, List<V>> entry : m.entrySet()) {
745          synchronized (result) {

747             if (list != null) {

749             } else {

752             }
753             result.put(entry.getKey(), list);
754          }
755       }
