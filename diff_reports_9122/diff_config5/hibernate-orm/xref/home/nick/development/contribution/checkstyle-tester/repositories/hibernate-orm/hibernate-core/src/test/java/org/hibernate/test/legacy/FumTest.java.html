866 	private Session spoofSerialization(Session session) throws IOException {
867 		try {
868 			// Serialize the incoming out to memory
869 			ByteArrayOutputStream serBaOut = new ByteArrayOutputStream();
870 			ObjectOutputStream serOut = new ObjectOutputStream(serBaOut);
871 
872 			serOut.writeObject(session);
873 
874 			// Now, re-constitute the model from memory
875 			ByteArrayInputStream serBaIn =
876 			        new ByteArrayInputStream(serBaOut.toByteArray());
877 			ObjectInputStream serIn = new ObjectInputStream(serBaIn);
878 
879 			Session outgoing = (Session) serIn.readObject();
880 
881 			return outgoing;
882 		}
883 		catch (ClassNotFoundException cnfe) {
884 			throw new IOException("Unable to locate class on reconstruction");
885 		}
886 	}
