71  	@Test
72  	@SuppressWarnings("unchecked")
73  	public void testSftpInboundFlow() {
74  		QueueChannel out = new QueueChannel();
75  		IntegrationFlow flow = IntegrationFlows
76  				.from(Sftp.inboundAdapter(sessionFactory())
77  								.preserveTimestamp(true)
78  								.remoteDirectory("sftpSource")
79  								.regexFilter(".*\\.txt$")
80  								.localFilenameExpression("#this.toUpperCase() + '.a'")
81  								.localDirectory(getTargetLocalDirectory())
82  						.remoteComparator(Comparator.naturalOrder()),
83  						e -> e.id("sftpInboundAdapter").poller(Pollers.fixedDelay(100)))
84  				.channel(out)
85  				.get();
86  		IntegrationFlowRegistration registration = this.flowContext.registration(flow).register();
87  		Message<?> message = out.receive(10_000);
88  		assertThat(message).isNotNull();
89  		Object payload = message.getPayload();
90  		assertThat(payload).isInstanceOf(File.class);
91  		File file = (File) payload;
92  		assertThat(file.getName()).isEqualTo(" SFTPSOURCE1.TXT.a");
93  		assertThat(file.getAbsolutePath()).contains("localTarget");
94  
95  		message = out.receive(10_000);
96  		assertThat(message).isNotNull();
97  		file = (File) message.getPayload();
98  		assertThat(file.getName()).isIn("SFTPSOURCE2.TXT.a");
99  		assertThat(file.getAbsolutePath()).contains("localTarget");
100 
101 		registration.destroy();
102 	}

104 	@Test
105 	public void testSftpInboundStreamFlow() throws Exception {
106 		QueueChannel out = new QueueChannel();
107 		StandardIntegrationFlow flow = IntegrationFlows.from(
108 				Sftp.inboundStreamingAdapter(new SftpRemoteFileTemplate(sessionFactory()))
109 						.remoteDirectory("sftpSource")
110 						.regexFilter(".*\\.txt$"),
111 				e -> e.id("sftpInboundAdapter").poller(Pollers.fixedDelay(100)))
112 				.channel(out)
113 				.get();
114 		IntegrationFlowRegistration registration = this.flowContext.registration(flow).register();
115 		Message<?> message = out.receive(10_000);
116 		assertThat(message).isNotNull();
117 		assertThat(message.getPayload()).isInstanceOf(InputStream.class);
118 		assertThat(message.getHeaders().get(FileHeaders.REMOTE_FILE)).isIn(" sftpSource1.txt", "sftpSource2.txt");
119 		((InputStream) message.getPayload()).close();
120 		new IntegrationMessageHeaderAccessor(message).getCloseableResource().close();
121 
122 		message = out.receive(10_000);
123 		assertThat(message).isNotNull();
124 		assertThat(message.getPayload()).isInstanceOf(InputStream.class);
125 		assertThat(message.getHeaders().get(FileHeaders.REMOTE_FILE)).isIn(" sftpSource1.txt", "sftpSource2.txt");
126 		assertThat(message.getHeaders().get(FileHeaders.REMOTE_HOST_PORT, String.class)).contains("localhost:");
127 		((InputStream) message.getPayload()).close();
128 		new IntegrationMessageHeaderAccessor(message).getCloseableResource().close();
129 
130 		registration.destroy();
131 	}

200 	@Test
201 	@DisabledOnOs(OS.WINDOWS)
202 	public void testSftpOutboundFlowWithChmod() {
203 		IntegrationFlow flow = f -> f.handle(Sftp.outboundAdapter(sessionFactory(), FileExistsMode.FAIL)
204 				.useTemporaryFileName(false)
205 				.fileNameExpression("headers['" + FileHeaders.FILENAME + "']")
206 				.chmod(0644)
207 				.remoteDirectory("sftpTarget"));
208 		IntegrationFlowRegistration registration = this.flowContext.registration(flow).register();
209 		String fileName = "foo.file";
210 		registration.getInputChannel().send(MessageBuilder.withPayload("foo")
211 				.setHeader(FileHeaders.FILENAME, fileName)
212 				.build());
213 
214 		RemoteFileTemplate<ChannelSftp.LsEntry> template = new RemoteFileTemplate<>(sessionFactory());
215 		ChannelSftp.LsEntry[] files = template.execute(session ->
216 				session.list(getTargetRemoteDirectory().getName() + "/" + fileName));
217 		assertThat(files.length).isEqualTo(1);
218 		assertThat(files[0].getAttrs().getSize()).isEqualTo(3);
219 		String[] permissions = files[0].getAttrs().getPermissionsString().substring(1).replaceAll("--", "-").split("-");
220 		assertThat(permissions[0]).isEqualTo("rw");
221 		assertThat(permissions[1]).isEqualTo("r");
222 		assertThat(permissions[2]).isEqualTo("r");
223 
224 		registration.destroy();
225 	}

227 	@Test
228 	@SuppressWarnings("unchecked")
229 	public void testSftpMgetFlow() {
230 		QueueChannel out = new QueueChannel();
231 		IntegrationFlow flow = f -> f
232 				.handle(Sftp.outboundGateway(sessionFactory(), AbstractRemoteFileOutboundGateway.Command.MGET,
233 						"payload")
234 						.options(AbstractRemoteFileOutboundGateway.Option.RECURSIVE)
235 						.regexFileNameFilter("(subSftpSource|.*1.txt)")
236 						.localDirectoryExpression("'" + getTargetLocalDirectoryName() + "' + #remoteDirectory")
237 						.localFilenameExpression("#remoteFileName.replaceFirst('sftpSource', 'localTarget')"))
238 				.channel(out);
239 		String dir = "sftpSource/";
240 		IntegrationFlowRegistration registration = this.flowContext.registration(flow).register();
241 		registration.getInputChannel().send(new GenericMessage<>(dir + "*"));
242 		Message<?> result = out.receive(10_000);
243 		assertThat(result).isNotNull();
244 		List<File> localFiles = (List<File>) result.getPayload();
245 		// should have filtered sftpSource2.txt
246 		assertThat(localFiles.size()).isEqualTo(2);
247 
248 		for (File file : localFiles) {
249 			assertThat(file.getPath().replaceAll(Matcher.quoteReplacement(File.separator), "/")).contains(dir);
250 		}
251 		assertThat(localFiles.get(1).getPath().replaceAll(Matcher.quoteReplacement(File.separator), "/"))
252 				.contains(dir + "subSftpSource");
253 
254 		registration.destroy();
255 	}
