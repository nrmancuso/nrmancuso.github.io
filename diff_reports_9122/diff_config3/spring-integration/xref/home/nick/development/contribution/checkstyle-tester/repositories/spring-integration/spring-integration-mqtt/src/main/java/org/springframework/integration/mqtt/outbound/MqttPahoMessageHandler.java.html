41  /**
42   * Eclipse Paho Implementation. When consuming {@link org.springframework.integration.mqtt.event.MqttIntegrationEvent}s
43   * published by this component use {@code MqttPahoComponent handler = event.getSourceAsType()} to get a
44   * reference, allowing you to obtain the bean name and {@link MqttConnectOptions}. This
45   * technique allows consumption of events from both inbound and outbound endpoints in the
46   * same event listener.
47   *
48   * @author Gary Russell
49   * @author Artem Bilan
50   *
51   * @since 4.0
52   *
53   */
54  public class MqttPahoMessageHandler extends AbstractMqttMessageHandler
55  		implements MqttCallback, MqttPahoComponent, ApplicationEventPublisherAware {
56  
57  	/**
58  	 * The default completion timeout in milliseconds.
59  	 */
60  	public static final long DEFAULT_COMPLETION_TIMEOUT = 30_000L;
61  
62  	/**
63  	 * The default disconnect completion timeout in milliseconds.
64  	 */
65  	public static final long DISCONNECT_COMPLETION_TIMEOUT = 5_000L;
66  
67  	private long completionTimeout = DEFAULT_COMPLETION_TIMEOUT;
68  
69  	private long disconnectCompletionTimeout = DISCONNECT_COMPLETION_TIMEOUT;
70  
71  	private final MqttPahoClientFactory clientFactory;
72  
73  	private boolean async;
74  
75  	private boolean asyncEvents;
76  
77  	private ApplicationEventPublisher applicationEventPublisher;
78  
79  	private volatile IMqttAsyncClient client;

81  	/**
82  	 * Use this constructor for a single url (although it may be overridden if the server
83  	 * URI(s) are provided by the {@link MqttConnectOptions#getServerURIs()} provided by

115 	/**
116 	 * Set to true if you don't want to block when sending messages. Default false.
117 	 * When true, message sent/delivered events will be published for reception

119 	 * inbound-channel-adapter.
120 	 * @param async true for async.

124 		this.async = async;
125 	}
126 
127 	/**
128 	 * When {@link #setAsync(boolean)} is true, setting this to true enables
129 	 * publication of {@link MqttMessageSentEvent} and {@link MqttMessageDeliveredEvent}

131 	 * @param asyncEvents the asyncEvents.
132 	 * @since 4.1
133 	 */
134 	public void setAsyncEvents(boolean asyncEvents) {
135 		this.asyncEvents = asyncEvents;
136 	}

138 	/**
139 	 * Set the completion timeout for async operations. Not settable using the namespace.
140 	 * Default {@value #DEFAULT_COMPLETION_TIMEOUT} milliseconds.
141 	 * @param completionTimeout The timeout.
142 	 * @since 4.1
143 	 */
144 	public void setCompletionTimeout(long completionTimeout) {
145 		this.completionTimeout = completionTimeout; // NOSONAR (sync)
146 	}

148 	/**
149 	 * Set the completion timeout when disconnecting. Not settable using the namespace.
150 	 * Default {@value #DISCONNECT_COMPLETION_TIMEOUT} milliseconds.
151 	 * @param completionTimeout The timeout.
152 	 * @since 5.1.10
153 	 */
154 	public void setDisconnectCompletionTimeout(long completionTimeout) {
155 		this.disconnectCompletionTimeout = completionTimeout; // NOSONAR (sync)
156 	}

158 	@Override
159 	public void setApplicationEventPublisher(ApplicationEventPublisher applicationEventPublisher) {
160 		this.applicationEventPublisher = applicationEventPublisher;
161 	}

163 	@Override
164 	public MqttConnectOptions getConnectionInfo() {
165 		MqttConnectOptions options = this.clientFactory.getConnectionOptions();
166 		if (options.getServerURIs() == null) {
167 			String url = getUrl();
168 			if (url != null) {
169 				options = MqttUtils.cloneConnectOptions(options);
170 				options.setServerURIs(new String[]{ url });
171 			}
172 		}
173 		return options;
174 	}

176 	@Override
177 	protected void onInit() {
178 		super.onInit();
179 		Assert.state(getConverter() instanceof MqttMessageConverter,
180 				"MessageConverter must be an MqttMessageConverter");
181 	}
182 
183 	@Override
184 	protected void doStart() {
185 	}
186 
187 	@Override
188 	protected void doStop() {
189 		try {
190 			IMqttAsyncClient theClient = this.client;
191 			if (theClient != null) {
192 				theClient.disconnect().waitForCompletion(this.disconnectCompletionTimeout);
193 				theClient.close();
194 				this.client = null;
195 			}
196 		}
197 		catch (MqttException ex) {
198 			logger.error(ex, "Failed to disconnect");
199 		}
200 	}
201 
202 	private synchronized IMqttAsyncClient checkConnection() throws MqttException {
203 		if (this.client != null && !this.client.isConnected()) {
204 			this.client.setCallback(null);
205 			this.client.close();
206 			this.client = null;
207 		}
208 		if (this.client == null) {
209 			try {
210 				MqttConnectOptions connectionOptions = this.clientFactory.getConnectionOptions();
211 				Assert.state(this.getUrl() != null || connectionOptions.getServerURIs() != null,
212 						"If no 'url' provided, connectionOptions.getServerURIs() must not be null");
213 				this.client = this.clientFactory.getAsyncClientInstance(this.getUrl(), this.getClientId());
214 				incrementClientInstance();
215 				this.client.setCallback(this);
216 				this.client.connect(connectionOptions).waitForCompletion(this.completionTimeout);
217 				logger.debug("Client connected");
218 			}
219 			catch (MqttException e) {
220 				if (this.client != null) {
221 					this.client.close();
222 					this.client = null;
223 				}
224 				if (this.applicationEventPublisher != null) {
225 					this.applicationEventPublisher.publishEvent(new MqttConnectionFailedEvent(this, e));
226 				}
227 				throw new MessagingException("Failed to connect", e);
228 			}
229 		}
230 		return this.client;
231 	}
232 
233 	@Override
234 	protected void publish(String topic, Object mqttMessage, Message<?> message) {
235 		Assert.isInstanceOf(MqttMessage.class, mqttMessage, "The 'mqttMessage' must be an instance of 'MqttMessage'");
236 		try {
237 			IMqttDeliveryToken token = checkConnection()
238 					.publish(topic, (MqttMessage) mqttMessage);
239 			if (!this.async) {
240 				token.waitForCompletion(this.completionTimeout); // NOSONAR (sync)
241 			}
242 			else if (this.asyncEvents && this.applicationEventPublisher != null) {
243 				this.applicationEventPublisher.publishEvent(
244 						new MqttMessageSentEvent(this, message, topic, token.getMessageId(), getClientId(),
245 								getClientInstance()));
246 			}
247 		}
248 		catch (MqttException e) {
249 			throw new MessageHandlingException(message, "Failed to publish to MQTT in the [" + this + ']', e);
250 		}
251 	}
252 
253 	private void sendDeliveryComplete(IMqttDeliveryToken token) {
254 		if (this.async && this.asyncEvents && this.applicationEventPublisher != null) {
255 			this.applicationEventPublisher.publishEvent(
256 					new MqttMessageDeliveredEvent(this, token.getMessageId(), getClientId(),
257 							getClientInstance()));
258 		}
259 	}
260 
261 	@Override
262 	public synchronized void connectionLost(Throwable cause) {
263 		logger.error("Lost connection; will attempt reconnect on next request");
264 		if (this.client != null) {
265 			try {
266 				this.client.setCallback(null);
267 				this.client.close();
268 			}
269 			catch (MqttException e) {
270 				// NOSONAR
271 			}
272 			this.client = null;
273 			if (this.applicationEventPublisher != null) {
274 				this.applicationEventPublisher.publishEvent(new MqttConnectionFailedEvent(this, cause));
275 			}
276 		}
277 	}
278 
279 	@Override
280 	public void messageArrived(String topic, MqttMessage message) {
281 
282 	}
283 
284 	@Override
285 	public void deliveryComplete(IMqttDeliveryToken token) {
286 		sendDeliveryComplete(token);
287 	}
