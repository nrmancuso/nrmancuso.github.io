83      private final static void test(Provider p, String pub1s, String priv1s, String pub2s, String priv2s, String secrets) throws Exception {
84          KeyFactory kf = KeyFactory.getInstance("EC", p);
85          PublicKey pub1 = kf.generatePublic(new X509EncodedKeySpec(parse(pub1s)));
86          System.out.println("Testing using parameters " + ((ECPublicKey)pub1).getParams() + "...");
87  
88          PrivateKey priv1 = kf.generatePrivate(new PKCS8EncodedKeySpec(parse(priv1s)));
89          PublicKey pub2 = kf.generatePublic(new X509EncodedKeySpec(parse(pub2s)));
90          PrivateKey priv2 = kf.generatePrivate(new PKCS8EncodedKeySpec(parse(priv2s)));
91          byte[] secret = parse(secrets);
92  
93          KeyAgreement ka1 = KeyAgreement.getInstance("ECDH", p);
94          ka1.init(priv1);
95          ka1.doPhase(pub2, true);
96          byte[] s1 = ka1.generateSecret();
97          if (Arrays.equals(secret, s1) == false) {
98              System.out.println("expected: " + toString(secret));
99              System.out.println("actual:   " + toString(s1));
100             throw new Exception("Secret 1 does not match");
101         }
102 
103         KeyAgreement ka2 = KeyAgreement.getInstance("ECDH", p);
104         ka2.init(priv2);
105         ka2.doPhase(pub1, true);
106         byte[] s2 = ka2.generateSecret();
107         if (Arrays.equals(secret, s2) == false) {
108             System.out.println("expected: " + toString(secret));
109             System.out.println("actual:   " + toString(s2));
110             throw new Exception("Secret 2 does not match");
111         }
112     }
