225 	 *
226 	 * @return EntityStatistics object
227 	 */
228 	public EntityStatistics getEntityStatistics(String entityName) {
229 		ConcurrentEntityStatisticsImpl es = (ConcurrentEntityStatisticsImpl) entityStatistics.get( entityName );
230 		if ( es == null ) {
231 			es = new ConcurrentEntityStatisticsImpl( entityName );
232 			ConcurrentEntityStatisticsImpl previous;
233 			if ( ( previous = (ConcurrentEntityStatisticsImpl) entityStatistics.putIfAbsent(
234 					entityName, es
235 			) ) != null ) {
236 				es = previous;
237 			}
238 		}
239 		return es;
240 	}

307 	@Override
308 	public NaturalIdCacheStatistics getNaturalIdCacheStatistics(String regionName) {
309 		ConcurrentNaturalIdCacheStatisticsImpl nics =
310 				(ConcurrentNaturalIdCacheStatisticsImpl) naturalIdCacheStatistics.get( regionName );
311 		
312 		if ( nics == null ) {
313 			if ( sessionFactory == null ) {
314 				return null;
315 			}
316 			Region region = sessionFactory.getNaturalIdCacheRegion( regionName );
317 			if ( region == null ) {
318 				return null;
319 			}
320 			nics = new ConcurrentNaturalIdCacheStatisticsImpl( region );
321 			ConcurrentNaturalIdCacheStatisticsImpl previous;
322 			if ( ( previous = (ConcurrentNaturalIdCacheStatisticsImpl) naturalIdCacheStatistics.putIfAbsent(
323 					regionName, nics
324 			) ) != null ) {
325 				nics = previous;
326 			}
327 		}
328 		return nics;
329 	}

393 	@Override
394 	public void naturalIdQueryExecuted(String regionName, long time) {
395 		naturalIdQueryExecutionCount.getAndIncrement();
396 		boolean isLongestQuery = false;
397 		for ( long old = naturalIdQueryExecutionMaxTime.get();
398 			  ( isLongestQuery = time > old ) && ( !naturalIdQueryExecutionMaxTime.compareAndSet( old, time ) );
399 			  old = naturalIdQueryExecutionMaxTime.get() ) {
400 			// nothing to do here given the odd loop structure...
401 		}
402 		if ( isLongestQuery && regionName != null ) {
403 			naturalIdQueryExecutionMaxTimeRegion = regionName;
404 		}
405 		if ( regionName != null ) {
406 			( (ConcurrentNaturalIdCacheStatisticsImpl) getNaturalIdCacheStatistics( regionName ) ).queryExecuted( time );
407 		}
408 	}

410 	@SuppressWarnings({ "UnnecessaryBoxing" })
411 	public void queryExecuted(String hql, int rows, long time) {
412         LOG.hql(hql, Long.valueOf(time), Long.valueOf(rows));
413 		queryExecutionCount.getAndIncrement();
414 		boolean isLongestQuery = false;
415 		for ( long old = queryExecutionMaxTime.get();
416 			  ( isLongestQuery = time > old ) && ( !queryExecutionMaxTime.compareAndSet( old, time ) );
417 			  old = queryExecutionMaxTime.get() ) {
418 			// nothing to do here given the odd loop structure...
419 		}
420 		if ( isLongestQuery ) {
421 			queryExecutionMaxTimeQueryString = hql;
422 		}
423 		if ( hql != null ) {
424 			ConcurrentQueryStatisticsImpl qs = (ConcurrentQueryStatisticsImpl) getQueryStatistics( hql );
425 			qs.executed( rows, time );
426 		}
427 	}
