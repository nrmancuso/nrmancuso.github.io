255 		Master m2 = (Master) s.createCriteria(Master.class)

260 		m2 = (Master) s.createCriteria(Master.class)

887 	protected boolean isSerializableIsolationEnforced() throws Exception {
888 		Connection conn = null;
889 		try {
890 			conn = sessionFactory().getConnectionProvider().getConnection();
891 			return conn.getTransactionIsolation() >= Connection.TRANSACTION_SERIALIZABLE;
892 		}
893 		finally {
894 			if ( conn != null ) {
895 				try {
896 					sessionFactory().getConnectionProvider().closeConnection( conn );
897 				}
898 				catch ( Throwable ignore ) {
899 					// ignore...
900 				}
901 			}
902 		}
903 	}
