41  public class DerivedIdentitySimpleParentSimpleDepTest extends BaseCoreFunctionalTestCase {
42  	@Test
43  	public void testOneToOneExplicitJoinColumn() throws Exception {
44  		assertTrue( SchemaUtil.isColumnPresent( "MedicalHistory", "FK", configuration() ) );
45  		assertTrue( ! SchemaUtil.isColumnPresent( "MedicalHistory", "id", configuration() ) );
46  
47  		Session s = openSession();
48  		s.getTransaction().begin();
49  		Person person = new Person( "aaa" );
50  		s.persist( person );
51  		MedicalHistory medicalHistory = new MedicalHistory( person );
52  		s.persist( medicalHistory );
53  		s.getTransaction().commit();
54  		s.close();
55  
56  		s = openSession();
57  		s.getTransaction().begin();
58  		medicalHistory = (MedicalHistory) s.get( MedicalHistory.class, "aaa" );
59  		assertEquals( person.ssn, medicalHistory.patient.ssn );
60  		medicalHistory.lastupdate = new Date();
61  		s.getTransaction().commit();
62  		s.close();
63  
64  		s = openSession();
65  		s.getTransaction().begin();
66  		medicalHistory = (MedicalHistory) s.get( MedicalHistory.class, "aaa" );
67  		assertNotNull( medicalHistory.lastupdate );
68  		s.delete( medicalHistory );
69  		s.delete( medicalHistory.patient );
70  		s.getTransaction().commit();
71  		s.close();
72  	}
