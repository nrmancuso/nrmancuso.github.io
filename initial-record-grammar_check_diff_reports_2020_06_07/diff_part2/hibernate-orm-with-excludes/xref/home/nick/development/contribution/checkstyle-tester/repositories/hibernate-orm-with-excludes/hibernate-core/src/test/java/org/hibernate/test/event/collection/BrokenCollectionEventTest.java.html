58  	@Override
59  	protected void cleanupTest() {
60  		ParentWithCollection dummyParent = createParent( "dummyParent" );
61  		dummyParent.setChildren( createCollection() );
62  		Child dummyChild = dummyParent.addChild( "dummyChild" );
63  		Session s = openSession();
64  		Transaction tx = s.beginTransaction();
65  		List children = s.createCriteria( dummyChild.getClass() ).list();
66  		List parents = s.createCriteria( dummyParent.getClass() ).list();
67  		for ( Iterator it = parents.iterator(); it.hasNext(); ) {
68  			ParentWithCollection parent = ( ParentWithCollection ) it.next();
69  			parent.clearChildren();
70  			s.delete( parent );
71  		}
72  		for ( Iterator it = children.iterator(); it.hasNext(); ) {
73  			s.delete( it.next() );
74  		}
75  		tx.commit();
76  		s.close();
77  	}

297 	protected void checkResult(CollectionListeners listeners,
298 							 CollectionListeners.Listener listenerExpected,
299 							 ParentWithCollection parent,
300 							 int index) {
301 		checkResult( listeners, listenerExpected, parent, parent.getChildren(), index );
302 	}
303 	protected void checkResult(CollectionListeners listeners,
304 							 CollectionListeners.Listener listenerExpected,
305 							 ChildWithBidirectionalManyToMany child,
306 							 int index) {
307 		checkResult( listeners, listenerExpected, child, child.getParents(), index );
308 	}
309 
310 	protected void checkResult(CollectionListeners listeners,
311 							 CollectionListeners.Listener listenerExpected,
312 							 Entity ownerExpected,
313 							 Collection collExpected,
314 							 int index) {
315 		assertSame( listenerExpected, listeners.getListenersCalled().get( index ) );
316 		assertSame(
317 				ownerExpected,
318 				( ( AbstractCollectionEvent ) listeners.getEvents().get( index ) ).getAffectedOwnerOrNull()
319 		);
320 		assertEquals(
321 				ownerExpected.getId(),
322 				( ( AbstractCollectionEvent ) listeners.getEvents().get( index ) ).getAffectedOwnerIdOrNull()
323 		);
324 		assertEquals(
325 				ownerExpected.getClass().getName(),
326 				( (AbstractCollectionEvent) listeners.getEvents().get( index ) ).getAffectedOwnerEntityName()
327 		);
328 		assertSame(
329 				collExpected, ( ( AbstractCollectionEvent ) listeners.getEvents().get( index ) ).getCollection()
330 		);
331 	}
332 
333 	private void checkNumberOfResults(CollectionListeners listeners, int nEventsExpected) {
334 		assertEquals( nEventsExpected, listeners.getListenersCalled().size() );
335 		assertEquals( nEventsExpected, listeners.getEvents().size() );
336 	}
