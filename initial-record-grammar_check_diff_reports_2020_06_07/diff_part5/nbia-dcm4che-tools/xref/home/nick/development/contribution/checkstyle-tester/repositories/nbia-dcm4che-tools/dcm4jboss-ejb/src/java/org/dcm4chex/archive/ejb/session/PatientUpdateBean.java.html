101     public void setSessionContext(SessionContext arg0) throws EJBException,
102             RemoteException {
103         Context jndiCtx = null;
104         try {
105             jndiCtx = new InitialContext();
106             patHome = (PatientLocalHome) jndiCtx
107                     .lookup("java:comp/env/ejb/Patient");
108         } catch (NamingException e) {
109             throw new EJBException(e);

111             if (jndiCtx != null) {
112                 try {
113                     jndiCtx.close();
114                 } catch (NamingException ignore) {
115                 }
116             }
117         }
118     }

125      * @throws RemoveException 
126      * @throws EJBException 
127      * @ejb.interface-method
128      */
129     public void changePatientIdentifierList(Dataset correct, Dataset prior,
130             PatientMatching matching, boolean keepPrior) throws CreateException, FinderException,
131                     PatientAlreadyExistsException, EJBException, RemoveException {
132         LOG.info("Change PID for " + correct.getString(Tags.PatientName)
133                 + " from " + toPidAndIssuer(prior)
134                 + " to " + toPidAndIssuer(correct));
135         try {
136             patHome.selectPatient(correct, matching, false);
137             String prompt = "Patient with PID "
138                 + toPidAndIssuer(correct) + " already exists";
139             LOG.warn(prompt);
140             throw new PatientAlreadyExistsException(prompt);
141         } catch (ObjectNotFoundException e) {}
142         try {
143             PatientLocal priorPat = patHome.selectPatient(prior, matching, true);
144             Dataset newAttrs = priorPat.getAttributes(true);
145             newAttrs.putLO(Tags.PatientID, correct.getString(Tags.PatientID));
146             newAttrs.putLO(Tags.IssuerOfPatientID, correct.getString(Tags.IssuerOfPatientID));
147             if (keepPrior) {
148                 PatientLocal correctPat = patHome.create(newAttrs);
149                 merge(correctPat, priorPat, keepPrior);
150             } else {
151                 priorPat.setAttributes(newAttrs);
152             }
153         } catch (ObjectNotFoundException x) {
154             LOG.warn("Prior patient of ChangePatientIdentifierList request not found! Ignored!");
155         }
156     }

158     /**
159      * @throws RemoveException 
160      * @throws EJBException 
161      * @ejb.interface-method
162      */
163     public void mergePatient(Dataset dominant, Dataset prior,
164             PatientMatching matching, boolean keepPrior) throws CreateException, FinderException, EJBException, RemoveException {
165         LOG.info("Merge " + prior.getString(Tags.PatientName)
166                 + " with PID " + toPidAndIssuer(prior)
167                 + " to " + dominant.getString(Tags.PatientName)
168                 + " with PID " + toPidAndIssuer(dominant));
169         PatientLocal dominantPat = updateOrCreate(dominant, matching);
170         PatientLocal priorPat= updateOrCreate(prior, matching);
171         LOG.debug("Merge patient.pk "+priorPat.getPk()+" to patient.pk "+dominantPat.getPk());
172         merge(dominantPat, priorPat, keepPrior);
173     }
174 
175     private void merge(PatientLocal dominantPat, PatientLocal priorPat, boolean keepPrior) throws EJBException, RemoveException {
176         if (dominantPat.equals(priorPat)) {
177             throw new IllegalArgumentException("Prior and dominant patients are the same! Can't merge patient to itself!");
178         }
179         dominantPat.getStudies().addAll(priorPat.getStudies());
180         dominantPat.getMpps().addAll(priorPat.getMpps());
181         dominantPat.getMwlItems().addAll(priorPat.getMwlItems());
182         dominantPat.getGsps().addAll(priorPat.getGsps());
183         dominantPat.getGppps().addAll(priorPat.getGppps());
184         dominantPat.getUPS().addAll(priorPat.getUPS());
185         dominantPat.getMerged().addAll(priorPat.getMerged());
186         if (keepPrior) {
187             priorPat.setMergedWith(dominantPat);
188         } else {
189             priorPat.remove();  
190         }
191     }

202     /**
203      * Update patient data as well as relink study with the patient if the patient
204      * is different than original one.
205      * @throws CreateException 
206      * 
207      * @ejb.interface-method
208      */
209     public void updatePatient(StudyLocal study, Dataset attrs, Dataset modified,
210             PatientMatching matching) throws FinderException, CreateException {
211 		String pid = attrs.getString(Tags.PatientID);
212 
213 		// If the patient id is not included, then we don't have to do any
214 		// patient update. Although patient id is type 2 in DICOM, but for DC,
215 		// we enforce this.
216 		if (pid == null || pid.length() == 0)
217 			return;
218 		
219 		PatientLocal newPatient = updateOrCreate(attrs, modified, matching);
220 		PatientLocal oldPat = study.getPatient();
221 		// Case 1: it's matching the same patient. Do nothing
222 		if(oldPat.getPk() == newPatient.getPk())
223 			return;
224 		if(modified != null) {
225 			modified.putAll( attrs );
226 		}
227 		// Case 2: there's no matching, a new patient is created. The study is updated.
228 		// Case 3: it's matching another existing patient. The study is updated.
229 		study.setPatient(newPatient);
230 		SeriesLocal series;
231 		for ( Iterator it = study.getSeries().iterator() ; it.hasNext() ; ) {
232 		    series = (SeriesLocal) it.next();
233 		    if (series.getMpps() != null) {
234 		        series.getMpps().setPatient(newPatient);
235 		    }
236 		}
237     }

265    private PatientLocal updateOrCreate(Dataset ds, Dataset modified, PatientMatching matching)
266             throws CreateException, FinderException {
267        try {
268             return findAndUpdatePatient(ds, modified, matching);
269        } catch (ObjectNotFoundException e) {
270             PatientLocal pat;
271             try {
272                 pat = patHome.create(ds);
273             } catch (CreateException ce) {
274                 // Check if patient record was inserted by concurrent thread
275                 // with unique index on (pat_id, pat_id_issuer)
276                 try {
277                     return findAndUpdatePatient(ds, modified, matching);
278                 } catch (FinderException fe) {
279                     throw ce;
280                 }
281             }
282             // Check if patient record was also inserted by concurrent thread
283             // with non-unique index on (pat_id, pat_id_issuer)
284             try {
285                 patHome.selectPatient(ds, matching, false);
286                 if(modified!=null) {
287                      modified.putAll(ds);	
288                 }	
289                 return pat;
290             } catch (NonUniquePatientException nupe) {
291                 try {
292                     pat.remove();
293                 } catch (RemoveException e1) {
294                     throw new EJBException(e);
295                 }
296                 return findAndUpdatePatient(ds, modified, matching);
297             }
298         }
299     }
300 
301     private PatientLocal findAndUpdatePatient(Dataset ds, Dataset modified,
302             PatientMatching matching) throws FinderException {
303     	Dataset origModAttrs=null;
304     	if( modified != null ) {
305     	   origModAttrs = DcmObjectFactory.getInstance().newDataset(); 	
306         }
307         PatientLocal pat = patHome.selectPatient(ds, matching, false);
308         if( pat.updateAttributes(ds, origModAttrs) ) {
309             AttrUtils.fetchModifiedAttributes(ds, origModAttrs, modified);
310         }
311         return pat;
312     }

314     /**
315      * @ejb.interface-method
316      */
317     public boolean deletePatient(Dataset ds, PatientMatching matching)
318             throws RemoveException, FinderException {
319         LOG.info("Delete " + ds.getString(Tags.PatientName)
320                 + " with PID " + toPidAndIssuer(ds));
321         try {
322             patHome.selectPatient(ds, matching, false).remove();
323             return true;
324         } catch (ObjectNotFoundException e) {
325             return false;
326         }
327     }

329     /**
330      * @ejb.interface-method
331      */
332     public void patientArrived(Dataset ds, PatientMatching matching)
333             throws FinderException {
334         LOG.info("Change status of SPS for " + ds.getString(Tags.PatientName)
335                 + " with PID " + toPidAndIssuer(ds) + " to ARRIVED");
336         try {
337             PatientLocal pat = patHome.selectPatient(ds, matching, false);
338             Collection c = pat.getMwlItems();
339             for (Iterator iter = c.iterator(); iter.hasNext();) {
340                 MWLItemLocal mwlitem = (MWLItemLocal) iter.next();
341                 if ( mwlitem.getSpsStatusAsInt() == SPSStatus.SCHEDULED )
342                     mwlitem.updateSpsStatus(SPSStatus.ARRIVED);
343             }
344         } catch (ObjectNotFoundException e) {
345         }
346     }

362     private String toPidAndIssuer(Dataset ds) {
363         return ds.getString(Tags.PatientID)+"^^^"+ds.getString(Tags.IssuerOfPatientID, "");
