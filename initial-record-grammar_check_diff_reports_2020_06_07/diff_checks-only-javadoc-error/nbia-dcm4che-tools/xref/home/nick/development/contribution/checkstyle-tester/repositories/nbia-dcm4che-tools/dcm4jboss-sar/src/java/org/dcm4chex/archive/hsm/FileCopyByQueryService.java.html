97      public void setPollInterval(String interval) throws Exception {
98          long l = RetryIntervalls.parseIntervalOrNever(interval);
99          if (l != pollInterval) {
100             this.pollInterval = l;
101             if (getState() == STARTED) {
102                 scheduler.stopScheduler(timerIDFilecopyPolling, schedulerID, this);
103                 schedulerID = scheduler.startScheduler(timerIDFilecopyPolling,
104                         pollInterval, this);
105             }
106         }
107     }

113     public void setQuery(String newSql) throws SQLException {
114         newSql = newSql.trim().replaceAll("\\s\\s+", " ");;
115         if (!newSql.equals(sql)) {
116             sql = newSql;
117             if ( this.getState() == STARTED) 
118                 try {
119                     checkSQL(newSql);
120                     updateCmd();
121                 } catch ( Throwable t) {
122                     log.error("Query String not valid!", t);
123                 }
124         }
125     }

139     private void checkSQL(String sql) throws SQLException {
140         sqlIsValid = false;
141         String sqlUC = sql.toUpperCase();
142         if ( sqlUC.indexOf("DELETE ") != -1 ) {
143             lastCheckResult = "DELETE is not allowed in this SQL statement!";
144             throw new IllegalArgumentException(lastCheckResult);
145         }
146         if ( sqlUC.indexOf("UPDATE ") != -1 ) {
147             lastCheckResult = "UPDATE is not allowed in this SQL statement!";
148             throw new IllegalArgumentException(lastCheckResult);
149         }
150         try {
151             QueryFilecopyCmd cmd = QueryFilecopyCmd.getInstance(sql, this.limit > 0 ? 1 : 0, 1);
152             cmd.setUpdateDatabaseMaxRetries(1);
153             List<Long> chk = cmd.getSeriesPKs(sql.indexOf('?') != -1 ? System.currentTimeMillis() : null, lastSeriesPk);
154             if (log.isDebugEnabled()) 
155                 log.debug("CheckSQL: QueryFilecopyCmd.getSeriesIUIDs done with result:"+chk);
156         } catch (SQLException x) {
157             lastCheckResult = x.getCause() == null ? x.toString() : x.getCause().toString();
158             throw x;
159         }
160         lastCheckResult="OK";
161         sqlIsValid = true;
162     }
163 
164     private void updateCmd() throws SQLException {
165         if (log.isDebugEnabled())
166             log.debug("update QueryFilecopyCmd when state = 3. state:"+getState()+" limit:"+limit+"\nSQL:"+sql);
167         if ( this.getState() == STARTED) {
168             if ( sqlCmd != null ) {
169                 sqlCmd.close();
170             }
171             sqlCmd = QueryFilecopyCmd.getInstance(sql, limit, fetchSize);
172         }
173     }

227     public void handleNotification(Notification notification, Object handback) {
228         synchronized(this) {
229             if (isRunning) {
230                 log.info("FileCopyByQuery is already running!");
231                 return;
232             }
233             isRunning = true;
234         }
235         new Thread(new Runnable(){
236             public void run() {
237                 try {
238                     doCheckFilecopy();
239                 } catch (Exception e) {
240                     log.error("Check for Pending Series Stored failed:", e);
241                 } finally {
242                     isRunning = false;
243                 }
244             }}).start();
245     }
246     
247     public int checkFilecopy() {
248         synchronized(this) {
249             if (isRunning) {
250                 log.info("FilecopyByQuery is already running!");
251                 return -1;
252             }
253             isRunning = true;
254         }
255         return doCheckFilecopy();
256     }
257     
258     public int doCheckFilecopy() {
259         if (lastCheckResult == null) {
260             try {
261                 checkSQL(sql);
262                 if (sqlCmd != null) { 
263                     sqlCmd.close();
264                 }
265                 sqlCmd = QueryFilecopyCmd.getInstance(sql, limit, fetchSize);
266             } catch ( Throwable t) {
267                 log.error("Check of SQL statement failed!",t);
268             }
269         }
270         if (!sqlIsValid) {
271             log.warn("SQL is not marked to be valid! checkFilecopy is disabled!");
272             isRunning = false;
273             return -1;
274         }
275         int nrOfOrders = 0;
276         int notScheduledOrders = 0;
277         try {
278             log.debug("SQL used to find series for copy:"+sqlCmd.getSQL());
279             List<Long> seriesPks = sqlCmd.getSeriesPKs(delay < 0 ? null : new Long(System.currentTimeMillis()-delay), lastSeriesPk);
280             log.info("Found "+seriesPks.size()+" Series for FileCopy! lastSeriesPk:"+lastSeriesPk);
281             Dataset ian;
282             for (int i=0, len=seriesPks.size() ; i < len ; i++) {
283                 ian = lookupContentEdit().getStudyMgtDatasetForSeries(new long[]{seriesPks.get(i)});
284                 if (log.isDebugEnabled()) {
285                     log.debug("IAN for series pk="+seriesPks.get(i)+" IAN:");log.debug(ian);
286                 }
287                 if (ian != null) {
288                     if (scheduleFilecopyOrder(ian))
289                         nrOfOrders++;
290                     else
291                         notScheduledOrders++;
292                 }
293             }
294             if (notScheduledOrders > 0)
295                 log.warn(notScheduledOrders+" Order(s) of FileCopyByQuery service not scheduled! Please check configuration in FileCopy service!");
296             lastSeriesPk = seriesPks.size() > 0 ? seriesPks.get(seriesPks.size()-1) : 0L;
297             return nrOfOrders;
298         } catch (Exception e) {
299             log.error("Error while checking series for Filecopy! Already scheduled filecopy orders:"+nrOfOrders, e);
300             return nrOfOrders;
301         } finally {
302             isRunning = false;
303         }
304     }
305     
306     public String showSQL() {
307         if (lastCheckResult == null) {
308             try {
309                 checkSQL(sql);
310                 if (sqlCmd != null) { 
311                     sqlCmd.close();
312                 }
313                 sqlCmd = QueryFilecopyCmd.getInstance(sql, limit, fetchSize);
314             } catch ( Throwable t) {
315                 return "Check of SQL statement failed!";
316             }
317         }
318         return !sqlIsValid ? "SQL is not marked to be valid!" : sqlCmd.formatSql();
319     }

321     protected boolean scheduleFilecopyOrder(Dataset ian) {
322         try {
323            return (Boolean) server.invoke(filecopyServiceName, "scheduleByIAN", new Object[] {
324                     ian, -1L },
325                     new String[] { Dataset.class.getName(), long.class.getName() });
326         } catch (Exception e) {
327             log.error("Schedule FileCopy Order failed:", e);
328             return false;
329         }
330     }

334                 pollInterval, this);

337     protected void stopService() throws Exception {
338         scheduler.stopScheduler(timerIDFilecopyPolling, schedulerID, this);
339         super.stopService();

342     private ContentEdit lookupContentEdit() throws HomeFactoryException, RemoteException, CreateException {
343         if (contentEdit != null)
344             return contentEdit;
345         ContentEditHome home = (ContentEditHome) EJBHomeFactory.getFactory()
346             .lookup(ContentEditHome.class, ContentEditHome.JNDI_NAME);
347         contentEdit = home.create();
348         return contentEdit;
349     }
