41  import java.rmi.RemoteException;

43  import java.util.Arrays;
44  import java.util.Collection;
45  import java.util.Iterator;
46  import java.util.List;
47  
48  import javax.ejb.EJBException;
49  import javax.ejb.FinderException;
50  import javax.ejb.SessionBean;
51  import javax.ejb.SessionContext;
52  import javax.naming.Context;
53  import javax.naming.InitialContext;
54  import javax.naming.NamingException;
55  
56  import org.dcm4chex.archive.ejb.interfaces.PatientLocal;
57  import org.dcm4chex.archive.ejb.interfaces.PatientLocalHome;
58  
59  /**
60   * @author Gunter Zeilinger <gunterze@gmail.com>
61   * @version $Revision: 3398 $ $Date: 2007-06-21 15:25:32 +0200 (Do, 21 Jun 2007) $
62   * @since Jun 21, 2007
63   * 
64   * @ejb.bean name="PIXQuery" type="Stateless" view-type="remote" jndi-name="ejb/PIXQuery"
65   * @ejb.transaction-type type="Container"
66   * @ejb.transaction type="Required"
67   * @ejb.ejb-ref ejb-name="Patient" view-type="local" ref-name="ejb/Patient" 
68   */
69  public abstract class PIXQueryBean implements SessionBean {
70      
71      private PatientLocalHome patHome;    
72  
73      public void setSessionContext(SessionContext ctx) throws RemoteException {
74          Context jndiCtx = null;
75          try {
76              jndiCtx = new InitialContext();
77              patHome = (PatientLocalHome) jndiCtx
78                      .lookup("java:comp/env/ejb/Patient");
79          } catch (NamingException e) {
80              throw new EJBException(e);
81          } finally {
82              if (jndiCtx != null) {
83                  try {
84                      jndiCtx.close();
85                  } catch (NamingException ignore) {
86                  }
87              }
88          }
89      }
90  
91      public void unsetSessionContext() {
92          patHome = null;
93      }
94      
95      /**
96       * @ejb.interface-method
97       */    
98      public List queryCorrespondingPIDs(String patientID, String issuer,
99              String[] domains) {
100         try {
101             Collection c = isWildCard(patientID) 
102                 ? patHome.findCorrespondingLike(toLIKE(patientID), issuer)
103                 : patHome.findCorresponding(patientID, issuer);
104             return toPIDs(c, domains);
105         } catch (FinderException e) {
106             throw new EJBException(e);
107         }
108     }
109 
110     /**
111      * @ejb.interface-method
112      */    
113     public List queryCorrespondingPIDsByPrimaryPatientID(String patientID,
114             String issuer, String[] domains) {
115         try {
116             Collection c = isWildCard(patientID) 
117                 ? patHome.findCorrespondingByPrimaryPatientIDLike(toLIKE(patientID), issuer)
118                 : patHome.findCorrespondingByPrimaryPatientID(patientID, issuer);
119             return toPIDs(c, domains);
120         } catch (FinderException e) {
121             throw new EJBException(e);
122         }
123     }    
124 
125     /**
126      * @ejb.interface-method
127      */    
128     public List queryCorrespondingPIDsByOtherPatientID(String patientID,
129             String issuer, String[] domains) {
130         try {
131             Collection c = isWildCard(patientID) 
132                 ? patHome.findCorrespondingByOtherPatientIDLike(toLIKE(patientID), issuer)
133                 : patHome.findCorrespondingByOtherPatientID(patientID, issuer);
134             return toPIDs(c, domains);
135         } catch (FinderException e) {
136             throw new EJBException(e);
137         }
138     }    
139 
140     private ArrayList toPIDs(Collection pats, String[] domains) {
141         ArrayList l = new ArrayList(pats.size());
142         List domainList = domains != null ? Arrays.asList(domains) : null;
143         for (Iterator iter = pats.iterator(); iter.hasNext();) {
144             PatientLocal pat = (PatientLocal) iter.next();
145             String iss = pat.getIssuerOfPatientId();
146             if (domainList == null || domainList.contains(iss)) {
147                 l.add(new String[] { pat.getPatientId(), iss });
148             }
149         }
150         return l;
151     }
152     
153     private String toLIKE(String patientID) {
154         StringBuffer sb = new StringBuffer(patientID);
155         for (int i = 0; i < sb.length(); i++) {
156             switch (sb.charAt(i)) {
157                 case '?' :
158                     sb.setCharAt(i, '_');
159                     break;
160                 case '*' :
161                     sb.setCharAt(i, '%');
162                     break;
163                 case '\\' :
164                 case '_' :
165                 case '%' :
166                     sb.insert(i++, '\\');
167                     break;
168             }
169         }
170         return sb.toString();
171     }
172 
173     private boolean isWildCard(String s) {
174         return s.indexOf('*') != -1 || s.indexOf('?') != -1;
175     }
176 
177 }
