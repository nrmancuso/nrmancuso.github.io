110     /**
111      * Get the model for an http request.
112      * <p>
113      * Look in the session for an associated model via <code>MWLMODEL_ATTR_NAME</code><br>
114      * If there is no model stored in session (first request) a new model is created and stored in session.
115      * 
116      * @param request A http request.
117      * 
118      * @return The model for given request.
119      */
120     public static final MWLModel getModel( HttpServletRequest request ) {
121         MWLModel model = (MWLModel) request.getSession().getAttribute(MWLMODEL_ATTR_NAME);
122         if (model == null) {
123             model = new MWLModel(request);
124             request.getSession().setAttribute(MWLMODEL_ATTR_NAME, model);
125             model.filterWorkList( true );
126         }
127         return model;
128     }

209     public void setPatMergeAttributes(Map map) {
210         patients.clear();
211         stickyPatients.clear();
212         if ( map != null ) {
213             dominantPat = new PatientModel( (Dataset) map.get("dominant") );
214             stickyPatients.add(new Long(dominantPat.getPk()));
215             patients.add(dominantPat);
216             Dataset[] dsa = (Dataset[]) map.get("priorPats");
217             PatientModel pat;
218             for ( int i = 0 ; i < dsa.length ; i++ ) {
219                 pat = new PatientModel( dsa[i]);
220                 stickyPatients.add(new Long(pat.getPk()));
221                 patients.add(pat);
222             }
223             mergeReason="Linking MWL to MPPS requires patient merge!";
224         } else {
225             dominantPat = null;
226             mergeReason = null;
227         }
228 
229     }

251     /**
252      * Update the list of MWLEntries for the view.
253      * <p>
254      * The query use the search criteria values from the filter and use offset and limit for paging.
255      * <p>
256      * if <code>newSearch is true</code> will reset paging (set <code>offset</code> to 0!)
257      * @param newSearch
258      */
259     public void filterWorkList(boolean newSearch) {
260 
261         if ( newSearch ) setOffset(0);
262         Dataset searchDS = mwlFilter.toDataset();
263         log.debug("Use searchDS:"); log.debug(searchDS);
264         isLocal = MWLConsoleCtrl.getMwlScuDelegate().isLocal();
265         List l = MWLConsoleCtrl.getMwlScuDelegate().findMWLEntries( searchDS );
266         Collections.sort( l, comparator );
267         int total = l.size();
268         int offset = getOffset();
269         int limit = getLimit();
270         int end;
271         if ( offset >= total ) {
272             offset = 0;
273             end = limit < total ? limit : total;
274         } else {
275             end = offset + limit;
276             if ( end > total ) end = total;
277         }
278         Dataset ds;
279         mwlEntries.clear();
280         int countNull = 0;
281         MWLEntry entry;
282         for ( int i = offset ; i < end ; i++ ){
283             ds = (Dataset) l.get( i );
284             if ( ds != null ) {
285                 entry = new MWLEntry( ds );
286                 mwlEntries.put( String.valueOf(entry.getInternalId()), entry );
287             } else {
288                 countNull++;
289             }
290         }
291         setTotal(total - countNull); // the real total (without null entries!)
292     }

301     /**
302      * Inner class that compares two datasets for sorting Scheduled Procedure Steps 
303      * according scheduled Procedure step start date.
304      * 
305      * @author franz.willer
306      *
307      * TODO To change the template for this generated type comment go to
308      * Window - Preferences - Java - Code Style - Code Templates
309      */
310     public class SpsDSComparator implements Comparator {
311 
312         private final Date DATE_0 = new Date(0l);
313         public SpsDSComparator() {
314 
315         }
316 
317         /**
318          * Compares the scheduled procedure step start date and time of two Dataset objects.
319          * <p>
320          * USe either SPSStartDateAndTime or SPSStartDate and SPSStartTime to get the date.
321          * <p>
322          * Use the '0' Date (new Date(0l)) if the date is not in the Dataset.
323          * <p>
324          * Compares its two arguments for order. Returns a negative integer, zero, or a positive integer 
325          * as the first argument is less than, equal to, or greater than the second.
326          * <p>
327          * Throws an Exception if one of the arguments is null or not a Dataset object.
328          * 
329          * @param arg0 	First argument
330          * @param arg1	Second argument
331          * 
332          * @return <0 if arg0<arg1, 0 if equal and >0 if arg0>arg1
333          */
334         public int compare(Object arg0, Object arg1) {
335             Dataset ds1 = (Dataset) arg0;
336             Dataset ds2 = (Dataset) arg1;
337             Date d1 = _getStartDateAsLong( ds1 );
338             return d1.compareTo( _getStartDateAsLong( ds2 ) );
339         }
340 
341         /**
342          * @param ds1 The dataset
343          * 
344          * @return the date of this SPS Dataset.
345          */
346         private Date _getStartDateAsLong(Dataset ds) {
347             if ( ds == null ) return DATE_0;
348             DcmElement e = ds.get( Tags.SPSSeq );
349             if ( e == null ) return DATE_0;
350             Dataset spsItem = e.getItem();//scheduled procedure step sequence item.
351             Date d = spsItem.getDate( Tags.SPSStartDateAndTime );
352             if ( d == null ) {
353                 d = spsItem.getDateTime( Tags.SPSStartDate, Tags.SPSStartTime );
354             }
355             if ( d == null ) d = DATE_0;
356             return d;
357         }
358     }
