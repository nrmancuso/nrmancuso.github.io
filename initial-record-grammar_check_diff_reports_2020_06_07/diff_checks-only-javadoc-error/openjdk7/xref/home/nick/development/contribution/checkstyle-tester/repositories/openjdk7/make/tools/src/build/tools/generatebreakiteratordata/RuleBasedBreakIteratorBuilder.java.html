36  /**
37   * This class has the job of constructing a RuleBasedBreakIterator from a
38   * textual description. A Builder is constructed by GenerateBreakIteratorData,
39   * which uses it to construct the iterator itself and then throws it away.
40   * <p>The construction logic is separated out into its own class for two primary
41   * reasons:
42   * <ul>
43   * <li>The construction logic is quite sophisticated and large. Separating
44   * it out into its own class means the code must only be loaded into memory
45   * while a RuleBasedBreakIterator is being constructed, and can be purged after
46   * that.
47   * <li>There is a fair amount of state that must be maintained throughout the
48   * construction process that is not needed by the iterator after construction.
49   * Separating this state out into another class prevents all of the functions
50   * that construct the iterator from having to have really long parameter lists,
51   * (hopefully) contributing to readability and maintainability.
52   * </ul>
53   * <p>
54   * It'd be really nice if this could be an independent class rather than an
55   * inner class, because that would shorten the source file considerably, but
56   * making Builder an inner class of RuleBasedBreakIterator allows it direct
57   * access to RuleBasedBreakIterator's private members, which saves us from
58   * having to provide some kind of "back door" to the Builder class that could
59   * then also be used by other classes.
60   */

207     /**
208      * Thus function has three main purposes:
209      * <ul><li>Perform general syntax checking on the description, so the rest
210      * of the build code can assume that it's parsing a legal description.
211      * <li>Split the description into separate rules
212      * <li>Perform variable-name substitutions (so that no one else sees
213      * variable names)
214      * </ul>
215      */
216     private Vector buildRuleList(String description) {
217         // invariants:
218         // - parentheses must be balanced: ()[]{}<>
219         // - nothing can be nested inside <>
220         // - nothing can be nested inside [] except more []s
221         // - pairs of ()[]{}<> must not be empty
222         // - ; can only occur at the outer level
223         // - | can only appear inside ()
224         // - only one = or / can occur in a single rule
225         // - = and / cannot both occur in the same rule
226         // - <> can only occur on the left side of a = expression
227         //   (because we'll perform substitutions to eliminate them other places)
228         // - the left-hand side of a = expression can only be a single character
229         //   (possibly with \) or text inside <>
230         // - the right-hand side of a = expression must be enclosed in [] or ()
231         // - * may not occur at the beginning of a rule, nor may it follow
232         //   =, /, (, (, |, }, ;, or *
233         // - ? may only follow *
234         // - the rule list must contain at least one / rule
235         // - no rule may be empty
236         // - all printing characters in the ASCII range except letters and digits
237         //   are reserved and must be preceded by \
238         // - ! may only occur at the beginning of a rule
239 
240         // set up a vector to contain the broken-up description (each entry in the
241         // vector is a separate rule) and a stack for keeping track of opening
242         // punctuation
243         Vector tempRuleList = new Vector();
244         Stack parenStack = new Stack();
245 
246         int p = 0;
247         int ruleStart = 0;
248         int c = '\u0000';
249         int lastC = '\u0000';
250         int lastOpen = '\u0000';
251         boolean haveEquals = false;
252         boolean havePipe = false;
253         boolean sawVarName = false;
254         final String charsThatCantPrecedeAsterisk = "=/{(|}*;\u0000";
255 
256         // if the description doesn't end with a semicolon, tack a semicolon onto the end
257         if (description.length() != 0 &&
258             description.codePointAt(description.length() - 1) != ';') {
259             description = description + ";";
260         }
261 
262         // for each character, do...
263         while (p < description.length()) {
264             c = description.codePointAt(p);
265 
266             switch (c) {
267                 // if the character is a backslash, skip the character that follows it
268                 // (it'll get treated as a literal character)
269                 case '\\':
270                     ++p;
271                     break;
272 
273                 // if the character is opening punctuation, verify that no nesting
274                 // rules are broken, and push the character onto the stack
275                 case '{':
276                 case '<':
277                 case '[':
278                 case '(':
279                     if (lastOpen == '<') {
280                         error("Can't nest brackets inside <>", p, description);
281                     }
282                     if (lastOpen == '[' && c != '[') {
283                         error("Can't nest anything in [] but []", p, description);
284                     }
285 
286                     // if we see < anywhere except on the left-hand side of =,
287                     // we must be seeing a variable name that was never defined
288                     if (c == '<' && (haveEquals || havePipe)) {
289                         error("Unknown variable name", p, description);
290                     }
291 
292                     lastOpen = c;
293                     parenStack.push(new Character((char)c));
294                     if (c == '<') {
295                         sawVarName = true;
296                     }
297                     break;
298 
299                 // if the character is closing punctuation, verify that it matches the
300                 // last opening punctuation we saw, and that the brackets contain
301                 // something, then pop the stack
302                 case '}':
303                 case '>':
304                 case ']':
305                 case ')':
306                     char expectedClose = '\u0000';
307                     switch (lastOpen) {
308                         case '{':
309                             expectedClose = '}';
310                             break;
311                         case '[':
312                             expectedClose = ']';
313                             break;
314                         case '(':
315                             expectedClose = ')';
316                             break;
317                         case '<':
318                             expectedClose = '>';
319                             break;
320                     }
321                     if (c != expectedClose) {
322                         error("Unbalanced parentheses", p, description);
323                     }
324                     if (lastC == lastOpen) {
325                         error("Parens don't contain anything", p, description);
326                     }
327                     parenStack.pop();
328                     if (!parenStack.empty()) {
329                         lastOpen = ((Character)(parenStack.peek())).charValue();
330                     }
331                     else {
332                         lastOpen = '\u0000';
333                     }
334 
335                     break;
336 
337                 // if the character is an asterisk, make sure it occurs in a place
338                 // where an asterisk can legally go
339                 case '*':
340                     if (charsThatCantPrecedeAsterisk.indexOf(lastC) != -1) {
341                         error("Misplaced asterisk", p, description);
342                     }
343                     break;
344 
345                 // if the character is a question mark, make sure it follows an asterisk
346                 case '?':
347                     if (lastC != '*') {
348                         error("Misplaced ?", p, description);
349                     }
350                     break;
351 
352                 // if the character is an equals sign, make sure we haven't seen another
353                 // equals sign or a slash yet
354                 case '=':
355                     if (haveEquals || havePipe) {
356                         error("More than one = or / in rule", p, description);
357                     }
358                     haveEquals = true;
359                     break;
360 
361                 // if the character is a slash, make sure we haven't seen another slash
362                 // or an equals sign yet
363                 case '/':
364                     if (haveEquals || havePipe) {
365                         error("More than one = or / in rule", p, description);
366                     }
367                     if (sawVarName) {
368                         error("Unknown variable name", p, description);
369                     }
370                     havePipe = true;
371                     break;
372 
373                 // if the character is an exclamation point, make sure it occurs only
374                 // at the beginning of a rule
375                 case '!':
376                     if (lastC != ';' && lastC != '\u0000') {
377                         error("! can only occur at the beginning of a rule", p, description);
378                     }
379                     break;
380 
381                 // we don't have to do anything special on a period
382                 case '.':
383                     break;
384 
385                 // if the character is a syntax character that can only occur
386                 // inside [], make sure that it does in fact only occur inside [].
387                 case '^':
388                 case '-':
389                 case ':':
390                     if (lastOpen != '[' && lastOpen != '<') {
391                         error("Illegal character", p, description);
392                     }
393                     break;
394 
395                 // if the character is a semicolon, do the following...
396                 case ';':
397                     // make sure the rule contains something and that there are no
398                     // unbalanced parentheses or brackets
399                     if (lastC == ';' || lastC == '\u0000') {
400                         error("Empty rule", p, description);
401                     }
402                     if (!parenStack.empty()) {
403                         error("Unbalanced parenheses", p, description);
404                     }
405 
406                     if (parenStack.empty()) {
407                         // if the rule contained an = sign, call processSubstitution()
408                         // to replace the substitution name with the substitution text
409                         // wherever it appears in the description
410                         if (haveEquals) {
411                             description = processSubstitution(description.substring(ruleStart,
412                                             p), description, p + 1);
413                         }
414                         else {
415                             // otherwise, check to make sure the rule doesn't reference
416                             // any undefined substitutions
417                             if (sawVarName) {
418                                 error("Unknown variable name", p, description);
419                             }
420 
421                             // then add it to tempRuleList
422                             tempRuleList.addElement(description.substring(ruleStart, p));
423                         }
424 
425                         // and reset everything to process the next rule
426                         ruleStart = p + 1;
427                         haveEquals = havePipe = sawVarName = false;
428                     }
429                     break;
430 
431                 // if the character is a vertical bar, check to make sure that it
432                 // occurs inside a () expression and that the character that precedes
433                 // it isn't also a vertical bar
434                 case '|':
435                     if (lastC == '|') {
436                         error("Empty alternative", p, description);
437                     }
438                     if (parenStack.empty() || lastOpen != '(') {
439                         error("Misplaced |", p, description);
440                     }
441                     break;
442 
443                 // if the character is anything else (escaped characters are
444                 // skipped and don't make it here), it's an error
445                 default:
446                     if (c >= ' ' && c < '\u007f' && !Character.isLetter((char)c)
447                         && !Character.isDigit((char)c)) {
448                         error("Illegal character", p, description);
449                     }
450                     if (c >= Character.MIN_SUPPLEMENTARY_CODE_POINT) {
451                         ++p;
452                     }
453                     break;
454             }
455             lastC = c;
456             ++p;
457         }
458         if (tempRuleList.size() == 0) {
459             error("No valid rules in description", p, description);
460         }
461         return tempRuleList;
462     }

464     /**
465      * This function performs variable-name substitutions.  First it does syntax
466      * checking on the variable-name definition.  If it's syntactically valid, it
467      * then goes through the remainder of the description and does a simple
468      * find-and-replace of the variable name with its text.  (The variable text
469      * must be enclosed in either [] or () for this to work.)
470      */
471     protected String processSubstitution(String substitutionRule, String description,
472                     int startPos) {
473         // isolate out the text on either side of the equals sign
474         String replace;
475         String replaceWith;
476         int equalPos = substitutionRule.indexOf('=');
477         replace = substitutionRule.substring(0, equalPos);
478         replaceWith = substitutionRule.substring(equalPos + 1);
479 
480         // check to see whether the substitution name is something we've declared
481         // to be "special".  For RuleBasedBreakIterator itself, this is "<ignore>".
482         // This function takes care of any extra processing that has to be done
483         // with "special" substitution names.
484         handleSpecialSubstitution(replace, replaceWith, startPos, description);
485 
486         // perform various other syntax checks on the rule
487         if (replaceWith.length() == 0) {
488             error("Nothing on right-hand side of =", startPos, description);
489         }
490         if (replace.length() == 0) {
491             error("Nothing on left-hand side of =", startPos, description);
492         }
493         if (replace.length() == 2 && replace.charAt(0) != '\\') {
494             error("Illegal left-hand side for =", startPos, description);
495         }
496         if (replace.length() >= 3 && replace.charAt(0) != '<' &&
497             replace.codePointBefore(equalPos) != '>') {
498             error("Illegal left-hand side for =", startPos, description);
499         }
500         if (!(replaceWith.charAt(0) == '[' &&
501               replaceWith.charAt(replaceWith.length() - 1) == ']') &&
502             !(replaceWith.charAt(0) == '(' &&
503               replaceWith.charAt(replaceWith.length() - 1) == ')')) {
504             error("Illegal right-hand side for =", startPos, description);
505         }
506 
507         // now go through the rest of the description (which hasn't been broken up
508         // into separate rules yet) and replace every occurrence of the
509         // substitution name with the substitution body
510         StringBuffer result = new StringBuffer();
511         result.append(description.substring(0, startPos));
512         int lastPos = startPos;
513         int pos = description.indexOf(replace, startPos);
514         while (pos != -1) {
515             result.append(description.substring(lastPos, pos));
516             result.append(replaceWith);
517             lastPos = pos + replace.length();
518             pos = description.indexOf(replace, lastPos);
519         }
520         result.append(description.substring(lastPos));
521         return result.toString();
522     }

524     /**
525      * This function defines a protocol for handling substitution names that
526      * are "special," i.e., that have some property beyond just being
527      * substitutions.  At the RuleBasedBreakIterator level, we have one
528      * special substitution name, "<ignore>".  Subclasses can override this
529      * function to add more.  Any special processing that has to go on beyond
530      * that which is done by the normal substitution-processing code is done
531      * here.
532      */
533     protected void handleSpecialSubstitution(String replace, String replaceWith,
534                 int startPos, String description) {
535         // if we get a definition for a substitution called "ignore", it defines
536         // the ignore characters for the iterator.  Check to make sure the expression
537         // is a [] expression, and if it is, parse it and store the characters off
538         // to the side.
539         if (replace.equals("<ignore>")) {
540             if (replaceWith.charAt(0) == '(') {
541                 error("Ignore group can't be enclosed in (", startPos, description);
542             }
543             ignoreChars = CharSet.parseString(replaceWith);
544         }
545     }

547     /**
548      * This function builds the character category table.  On entry,
549      * tempRuleList is a vector of break rules that has had variable names substituted.
550      * On exit, the charCategoryTable data member has been initialized to hold the
551      * character category table, and tempRuleList's rules have been munged to contain
552      * character category numbers everywhere a literal character or a [] expression
553      * originally occurred.
554      */
555     protected void buildCharCategories(Vector tempRuleList) {
556         int bracketLevel = 0;
557         int p = 0;
558         int lineNum = 0;
559 
560         // build hash table of every literal character or [] expression in the rule list
561         // and use CharSet.parseString() to derive a CharSet object representing the
562         // characters each refers to
563         expressions = new Hashtable();
564         while (lineNum < tempRuleList.size()) {
565             String line = (String)(tempRuleList.elementAt(lineNum));
566             p = 0;
567             while (p < line.length()) {
568                 int c = line.codePointAt(p);
569                 switch (c) {
570                     // skip over all syntax characters except [
571                     case '{': case '}': case '(': case ')': case '*': case '.':
572                     case '/': case '|': case ';': case '?': case '!':
573                         break;
574 
575                     // for [, find the matching ] (taking nested [] pairs into account)
576                     // and add the whole expression to the expression list
577                     case '[':
578                         int q = p + 1;
579                         ++bracketLevel;
580                         while (q < line.length() && bracketLevel != 0) {
581                             c = line.codePointAt(q);
582                             switch (c) {
583                             case '\\':
584                                 q++;
585                                 break;
586                             case '[':
587                                 ++bracketLevel;
588                                 break;
589                             case ']':
590                                 --bracketLevel;
591                                 break;
592                             }
593                             q = q + Character.charCount(c);
594                         }
595                         if (expressions.get(line.substring(p, q)) == null) {
596                             expressions.put(line.substring(p, q), CharSet.parseString(line.substring(p, q)));
597                         }
598                         p = q - 1;
599                         break;
600 
601                     // for \ sequences, just move to the next character and treat
602                     // it as a single character
603                     case '\\':
604                         ++p;
605                         c = line.codePointAt(p);
606                         // DON'T break; fall through into "default" clause
607 
608                     // for an isolated single character, add it to the expression list
609                     default:
610                         expressions.put(line.substring(p, p + 1), CharSet.parseString(line.substring(p, p + 1)));
611                         break;
612                 }
613                 p += Character.charCount(line.codePointAt(p));
614             }
615             ++lineNum;
616         }
617         // dump CharSet's internal expression cache
618         CharSet.releaseExpressionCache();
619 
620         // create the temporary category table (which is a vector of CharSet objects)
621         categories = new Vector();
622         if (ignoreChars != null) {
623             categories.addElement(ignoreChars);
624         }
625         else {
626             categories.addElement(new CharSet());
627         }
628         ignoreChars = null;
629 
630         // this is a hook to allow subclasses to add categories on their own
631         mungeExpressionList(expressions);
632 
633         // Derive the character categories.  Go through the existing character categories
634         // looking for overlap.  Any time there's overlap, we create a new character
635         // category for the characters that overlapped and remove them from their original
636         // category.  At the end, any characters that are left in the expression haven't
637         // been mentioned in any category, so another new category is created for them.
638         // For example, if the first expression is [abc], then a, b, and c will be placed
639         // into a single character category.  If the next expression is [bcd], we will first
640         // remove b and c from their existing category (leaving a behind), create a new
641         // category for b and c, and then create another new category for d (which hadn't
642         // been mentioned in the previous expression).
643         // At no time should a character ever occur in more than one character category.
644 
645         // for each expression in the expressions list, do...
646         Enumeration iter = expressions.elements();
647         while (iter.hasMoreElements()) {
648             // initialize the working char set to the chars in the current expression
649             CharSet e = (CharSet)iter.nextElement();
650 
651             // for each category in the category list, do...
652             for (int j = categories.size() - 1; !e.empty() && j > 0; j--) {
653 
654                 // if there's overlap between the current working set of chars
655                 // and the current category...
656                 CharSet that = (CharSet)(categories.elementAt(j));
657                 if (!that.intersection(e).empty()) {
658 
659                     // add a new category for the characters that were in the
660                     // current category but not in the working char set
661                     CharSet temp = that.difference(e);
662                     if (!temp.empty()) {
663                         categories.addElement(temp);
664                     }
665 
666                     // remove those characters from the working char set and replace
667                     // the current category with the characters that it did
668                     // have in common with the current working char set
669                     temp = e.intersection(that);
670                     e = e.difference(that);
671                     if (!temp.equals(that)) {
672                         categories.setElementAt(temp, j);
673                     }
674                 }
675             }
676 
677             // if there are still characters left in the working char set,
678             // add a new category containing them
679             if (!e.empty()) {
680                 categories.addElement(e);
681             }
682         }
683 
684         // we have the ignore characters stored in position 0.  Make an extra pass through
685         // the character category list and remove anything from the ignore list that shows
686         // up in some other category
687         CharSet allChars = new CharSet();
688         for (int i = 1; i < categories.size(); i++) {
689             allChars = allChars.union((CharSet)(categories.elementAt(i)));
690         }
691         CharSet ignoreChars = (CharSet)(categories.elementAt(0));
692         ignoreChars = ignoreChars.difference(allChars);
693         categories.setElementAt(ignoreChars, 0);
694 
695         // now that we've derived the character categories, go back through the expression
696         // list and replace each CharSet object with a String that represents the
697         // character categories that expression refers to.  The String is encoded: each
698         // character is a character category number (plus 0x100 to avoid confusing them
699         // with syntax characters in the rule grammar)
700         iter = expressions.keys();
701         while (iter.hasMoreElements()) {
702             String key = (String)iter.nextElement();
703             CharSet cs = (CharSet)expressions.get(key);
704             StringBuffer cats = new StringBuffer();
705 
706             // for each category...
707             for (int j = 0; j < categories.size(); j++) {
708 
709                 // if the current expression contains characters in that category...
710                 CharSet temp = cs.intersection((CharSet)(categories.elementAt(j)));
711                 if (!temp.empty()) {
712 
713                     // then add the encoded category number to the String for this
714                     // expression
715                     cats.append((char)(0x100 + j));
716                     if (temp.equals(cs)) {
717                         break;
718                     }
719                 }
720             }
721 
722             // once we've finished building the encoded String for this expression,
723             // replace the CharSet object with it
724             expressions.put(key, cats.toString());
725         }
726 
727         // and finally, we turn the temporary category table into a permanent category
728         // table, which is a CompactByteArray. (we skip category 0, which by definition
729         // refers to all characters not mentioned specifically in the rules)
730         charCategoryTable = new CompactByteArray((byte)0);
731         supplementaryCharCategoryTable = new SupplementaryCharacterData((byte)0);
732 
733         // for each category...
734         for (int i = 0; i < categories.size(); i++) {
735             CharSet chars = (CharSet)(categories.elementAt(i));
736 
737             // go through the character ranges in the category one by one...
738             Enumeration enum_ = chars.getChars();
739             while (enum_.hasMoreElements()) {
740                 int[] range = (int[])(enum_.nextElement());
741 
742                 // and set the corresponding elements in the CompactArray accordingly
743                 if (i != 0) {
744                     if (range[0] < Character.MIN_SUPPLEMENTARY_CODE_POINT) {
745                         if (range[1] < Character.MIN_SUPPLEMENTARY_CODE_POINT) {
746                             charCategoryTable.setElementAt((char)range[0], (char)range[1], (byte)i);
747                         } else {
748                             charCategoryTable.setElementAt((char)range[0], (char)0xFFFF, (byte)i);
749                             supplementaryCharCategoryTable.appendElement(Character.MIN_SUPPLEMENTARY_CODE_POINT, range[1], (byte)i);
750                         }
751                     } else {
752                         supplementaryCharCategoryTable.appendElement(range[0], range[1], (byte)i);
753                     }
754                 }
755 
756                 // (category 0 is special-- it's the hiding place for the ignore
757                 // characters, whose real category number in the CompactArray is
758                 // -1 [this is because category 0 contains all characters not
759                 // specifically mentioned anywhere in the rules] )
760                 else {
761                     if (range[0] < Character.MIN_SUPPLEMENTARY_CODE_POINT) {
762                         if (range[1] < Character.MIN_SUPPLEMENTARY_CODE_POINT) {
763                             charCategoryTable.setElementAt((char)range[0], (char)range[1], IGNORE);
764                         } else {
765                             charCategoryTable.setElementAt((char)range[0], (char)0xFFFF, IGNORE);
766                             supplementaryCharCategoryTable.appendElement(Character.MIN_SUPPLEMENTARY_CODE_POINT, range[1], IGNORE);
767                         }
768                     } else {
769                         supplementaryCharCategoryTable.appendElement(range[0], range[1], IGNORE);
770                     }
771                 }
772             }
773         }
774 
775         // once we've populated the CompactArray, compact it
776         charCategoryTable.compact();
777 
778         // And, complete the category table for supplementary characters
779         supplementaryCharCategoryTable.complete();
780 
781         // initialize numCategories
782         numCategories = categories.size();
783     }

790     /**
791      * This is the function that builds the forward state table.  Most of the real
792      * work is done in parseRule(), which is called once for each rule in the
793      * description.
794      */
795     private void buildStateTable(Vector tempRuleList) {
796         // initialize our temporary state table, and fill it with two states:
797         // state 0 is a dummy state that allows state 1 to be the starting state
798         // and 0 to represent "stop".  State 1 is added here to seed things
799         // before we start parsing
800         tempStateTable = new Vector();
801         tempStateTable.addElement(new short[numCategories + 1]);
802         tempStateTable.addElement(new short[numCategories + 1]);
803 
804         // call parseRule() for every rule in the rule list (except those which
805         // start with !, which are actually backwards-iteration rules)
806         for (int i = 0; i < tempRuleList.size(); i++) {
807             String rule = (String)tempRuleList.elementAt(i);
808             if (rule.charAt(0) != '!') {
809                 parseRule(rule, true);
810             }
811         }
812 
813         // finally, use finishBuildingStateTable() to minimize the number of
814         // states in the table and perform some other cleanup work
815         finishBuildingStateTable(true);
816     }
817 
818     /**
819      * This is where most of the work really happens.  This routine parses a single
820      * rule in the rule description, adding and modifying states in the state
821      * table according to the new expression.  The state table is kept deterministic
822      * throughout the whole operation, although some ugly postprocessing is needed
823      * to handle the *? token.
824      */
825     private void parseRule(String rule, boolean forward) {
826         // algorithm notes:
827         //   - The basic idea here is to read successive character-category groups
828         //   from the input string.  For each group, you create a state and point
829         //   the appropriate entries in the previous state to it.  This produces a
830         //   straight line from the start state to the end state.  The {}, *, and (|)
831         //   idioms produce branches in this straight line.  These branches (states
832         //   that can transition to more than one other state) are called "decision
833         //   points."  A list of decision points is kept.  This contains a list of
834         //   all states that can transition to the next state to be created.  For a
835         //   straight line progression, the only thing in the decision-point list is
836         //   the current state.  But if there's a branch, the decision-point list
837         //   will contain all of the beginning points of the branch when the next
838         //   state to be created represents the end point of the branch.  A stack is
839         //   used to save decision point lists in the presence of nested parentheses
840         //   and the like.  For example, when a { is encountered, the current decision
841         //   point list is saved on the stack and restored when the corresponding }
842         //   is encountered.  This way, after the } is read, the decision point list
843         //   will contain both the state right before the } _and_ the state before
844         //   the whole {} expression.  Both of these states can transition to the next
845         //   state after the {} expression.
846         //   - one complication arises when we have to stamp a transition value into
847         //   an array cell that already contains one.  The updateStateTable() and
848         //   mergeStates() functions handle this case.  Their basic approach is to
849         //   create a new state that combines the two states that conflict and point
850         //   at it when necessary.  This happens recursively, so if the merged states
851         //   also conflict, they're resolved in the same way, and so on.  There are
852         //   a number of tests aimed at preventing infinite recursion.
853         //   - another complication arises with repeating characters.  It's somewhat
854         //   ambiguous whether the user wants a greedy or non-greedy match in these cases.
855         //   (e.g., whether "[a-z]*abc" means the SHORTEST sequence of letters ending in
856         //   "abc" or the LONGEST sequence of letters ending in "abc".  We've adopted
857         //   the *? to mean "shortest" and * by itself to mean "longest".  (You get the
858         //   same result with both if there's no overlap between the repeating character
859         //   group and the group immediately following it.)  Handling the *? token is
860         //   rather complicated and involves keeping track of whether a state needs to
861         //   be merged (as described above) or merely overwritten when you update one of
862         //   its cells, and copying the contents of a state that loops with a *? token
863         //   into some of the states that follow it after the rest of the table-building
864         //   process is complete ("backfilling").
865         // implementation notes:
866         //   - This function assumes syntax checking has been performed on the input string
867         //   prior to its being passed in here.  It assumes that parentheses are
868         //   balanced, all literal characters are enclosed in [] and turned into category
869         //   numbers, that there are no illegal characters or character sequences, and so
870         //   on.  Violation of these invariants will lead to undefined behavior.
871         //   - It'd probably be better to use linked lists rather than Vector and Stack
872         //   to maintain the decision point list and stack.  I went for simplicity in
873         //   this initial implementation.  If performance is critical enough, we can go
874         //   back and fix this later.
875         //   -There are a number of important limitations on the *? token.  It does not work
876         //   right when followed by a repeating character sequence (e.g., ".*?(abc)*")
877         //   (although it does work right when followed by a single repeating character).
878         //   It will not always work right when nested in parentheses or braces (although
879         //   sometimes it will).  It also will not work right if the group of repeating
880         //   characters and the group of characters that follows overlap partially
881         //   (e.g., "[a-g]*?[e-j]").  None of these capabilites was deemed necessary for
882         //   describing breaking rules we know about, so we left them out for
883         //   expeditiousness.
884         //   - Rules such as "[a-z]*?abc;" will be treated the same as "[a-z]*?aa*bc;"--
885         //   that is, if the string ends in "aaaabc", the break will go before the first
886         //   "a" rather than the last one.  Both of these are limitations in the design
887         //   of RuleBasedBreakIterator and not limitations of the rule parser.
888 
889         int p = 0;
890         int currentState = 1;   // don't use state number 0; 0 means "stop"
891         int lastState = currentState;
892         String pendingChars = "";
893 
894         decisionPointStack = new Stack();
895         decisionPointList = new Vector();
896         loopingStates = new Vector();
897         statesToBackfill = new Vector();
898 
899         short[] state;
900         boolean sawEarlyBreak = false;
901 
902         // if we're adding rules to the backward state table, mark the initial state
903         // as a looping state
904         if (!forward) {
905             loopingStates.addElement(new Integer(1));
906         }
907 
908         // put the current state on the decision point list before we start
909         decisionPointList.addElement(new Integer(currentState)); // we want currentState to
910                                                                  // be 1 here...
911         currentState = tempStateTable.size() - 1;   // but after that, we want it to be
912                                                     // 1 less than the state number of the next state
913         while (p < rule.length()) {
914             int c = rule.codePointAt(p);
915             clearLoopingStates = false;
916 
917             // this section handles literal characters, escaped characters (which are
918             // effectively literal characters too), the . token, and [] expressions
919             if (c == '['
920                 || c == '\\'
921                 || Character.isLetter(c)
922                 || Character.isDigit(c)
923                 || c < ' '
924                 || c == '.'
925                 || c >= '\u007f') {
926 
927                 // if we're not on a period, isolate the expression and look up
928                 // the corresponding category list
929                 if (c != '.') {
930                     int q = p;
931 
932                     // if we're on a backslash, the expression is the character
933                     // after the backslash
934                     if (c == '\\') {
935                         q = p + 2;
936                         ++p;
937                     }
938 
939                     // if we're on an opening bracket, scan to the closing bracket
940                     // to isolate the expression
941                     else if (c == '[') {
942                         int bracketLevel = 1;
943 
944                         q += Character.charCount(rule.codePointAt(q));
945                         while (bracketLevel > 0) {
946                             c = rule.codePointAt(q);
947                             if (c == '[') {
948                                 ++bracketLevel;
949                             }
950                             else if (c == ']') {
951                                 --bracketLevel;
952                             }
953                             else if (c == '\\') {
954                                 c = rule.codePointAt(++q);
955                             }
956                             q += Character.charCount(c);
957                         }
958                     }
959 
960                     // otherwise, the expression is just the character itself
961                     else {
962                         q = p + Character.charCount(c);
963                     }
964 
965                     // look up the category list for the expression and store it
966                     // in pendingChars
967                     pendingChars = (String)expressions.get(rule.substring(p, q));
968 
969                     // advance the current position past the expression
970                     p = q - Character.charCount(rule.codePointBefore(q));
971                 }
972 
973                 // if the character we're on is a period, we end up down here
974                 else {
975                     int rowNum = ((Integer)decisionPointList.lastElement()).intValue();
976                     state = (short[])tempStateTable.elementAt(rowNum);
977 
978                     // if the period is followed by an asterisk, then just set the current
979                     // state to loop back on itself
980                     if (p + 1 < rule.length() && rule.charAt(p + 1) == '*' && state[0] != 0) {
981                         decisionPointList.addElement(new Integer(state[0]));
982                         pendingChars = "";
983                         ++p;
984                     }
985 
986                     // otherwise, fabricate a category list ("pendingChars") with
987                     // every category in it
988                     else {
989                         StringBuffer temp = new StringBuffer();
990                         for (int i = 0; i < numCategories; i++)
991                             temp.append((char)(i + 0x100));
992                         pendingChars = temp.toString();
993                     }
994                 }
995 
996                 // we'll end up in here for all expressions except for .*, which is
997                 // special-cased above
998                 if (pendingChars.length() != 0) {
999 
1000                     // if the expression is followed by an asterisk, then push a copy
1001                     // of the current desicion point list onto the stack (this is
1002                     // the same thing we do on an opening brace)
1003                     if (p + 1 < rule.length() && rule.charAt(p + 1) == '*') {
1004                         decisionPointStack.push(decisionPointList.clone());
1005                     }
1006 
1007                     // create a new state, add it to the list of states to backfill
1008                     // if we have looping states to worry about, set its "don't make
1009                     // me an accepting state" flag if we've seen a slash, and add
1010                     // it to the end of the state table
1011                     int newState = tempStateTable.size();
1012                     if (loopingStates.size() != 0) {
1013                         statesToBackfill.addElement(new Integer(newState));
1014                     }
1015                     state = new short[numCategories + 1];
1016                     if (sawEarlyBreak) {
1017                         state[numCategories] = DONT_LOOP_FLAG;
1018                     }
1019                     tempStateTable.addElement(state);
1020 
1021                     // update everybody in the decision point list to point to
1022                     // the new state (this also performs all the reconciliation
1023                     // needed to make the table deterministic), then clear the
1024                     // decision point list
1025                     updateStateTable(decisionPointList, pendingChars, (short)newState);
1026                     decisionPointList.removeAllElements();
1027 
1028                     // add all states created since the last literal character we've
1029                     // seen to the decision point list
1030                     lastState = currentState;
1031                     do {
1032                         ++currentState;
1033                         decisionPointList.addElement(new Integer(currentState));
1034                     } while (currentState + 1 < tempStateTable.size());
1035                 }
1036             }
1037 
1038             // a { marks the beginning of an optional run of characters.  Push a
1039             // copy of the current decision point list onto the stack.  This saves
1040             // it, preventing it from being affected by whatever's inside the parentheses.
1041             // This decision point list is restored when a } is encountered.
1042             else if (c == '{') {
1043                 decisionPointStack.push(decisionPointList.clone());
1044             }
1045 
1046             // a } marks the end of an optional run of characters.  Pop the last decision
1047             // point list off the stack and merge it with the current decision point list.
1048             // a * denotes a repeating character or group (* after () is handled separately
1049             // below).  In addition to restoring the decision point list, modify the
1050             // current state to point to itself on the appropriate character categories.
1051             else if (c == '}' || c == '*') {
1052                 // when there's a *, update the current state to loop back on itself
1053                 // on the character categories that caused us to enter this state
1054                 if (c == '*') {
1055                     for (int i = lastState + 1; i < tempStateTable.size(); i++) {
1056                         Vector temp = new Vector();
1057                         temp.addElement(new Integer(i));
1058                         updateStateTable(temp, pendingChars, (short)(lastState + 1));
1059                     }
1060                 }
1061 
1062                 // pop the top element off the decision point stack and merge
1063                 // it with the current decision point list (this causes the divergent
1064                 // paths through the state table to come together again on the next
1065                 // new state)
1066                 Vector temp = (Vector)decisionPointStack.pop();
1067                 for (int i = 0; i < decisionPointList.size(); i++)
1068                     temp.addElement(decisionPointList.elementAt(i));
1069                 decisionPointList = temp;
1070             }
1071 
1072             // a ? after a * modifies the behavior of * in cases where there is overlap
1073             // between the set of characters that repeat and the characters which follow.
1074             // Without the ?, all states following the repeating state, up to a state which
1075             // is reached by a character that doesn't overlap, will loop back into the
1076             // repeating state.  With the ?, the mark states following the *? DON'T loop
1077             // back into the repeating state.  Thus, "[a-z]*xyz" will match the longest
1078             // sequence of letters that ends in "xyz," while "[a-z]*? will match the
1079             // _shortest_ sequence of letters that ends in "xyz".
1080             // We use extra bookkeeping to achieve this effect, since everything else works
1081             // according to the "longest possible match" principle.  The basic principle
1082             // is that transitions out of a looping state are written in over the looping
1083             // value instead of being reconciled, and that we copy the contents of the
1084             // looping state into empty cells of all non-terminal states that follow the
1085             // looping state.
1086             else if (c == '?') {
1087                 setLoopingStates(decisionPointList, decisionPointList);
1088             }
1089 
1090             // a ( marks the beginning of a sequence of characters.  Parentheses can either
1091             // contain several alternative character sequences (i.e., "(ab|cd|ef)"), or
1092             // they can contain a sequence of characters that can repeat (i.e., "(abc)*").  Thus,
1093             // A () group can have multiple entry and exit points.  To keep track of this,
1094             // we reserve TWO spots on the decision-point stack.  The top of the stack is
1095             // the list of exit points, which becomes the current decision point list when
1096             // the ) is reached.  The next entry down is the decision point list at the
1097             // beginning of the (), which becomes the current decision point list at every
1098             // entry point.
1099             // In addition to keeping track of the exit points and the active decision
1100             // points before the ( (i.e., the places from which the () can be entered),
1101             // we need to keep track of the entry points in case the expression loops
1102             // (i.e., is followed by *).  We do that by creating a dummy state in the
1103             // state table and adding it to the decision point list (BEFORE it's duplicated
1104             // on the stack).  Nobody points to this state, so it'll get optimized out
1105             // at the end.  It exists only to hold the entry points in case the ()
1106             // expression loops.
1107             else if (c == '(') {
1108 
1109                 // add a new state to the state table to hold the entry points into
1110                 // the () expression
1111                 tempStateTable.addElement(new short[numCategories + 1]);
1112 
1113                 // we have to adjust lastState and currentState to account for the
1114                 // new dummy state
1115                 lastState = currentState;
1116                 ++currentState;
1117 
1118                 // add the current state to the decision point list (add it at the
1119                 // BEGINNING so we can find it later)
1120                 decisionPointList.insertElementAt(new Integer(currentState), 0);
1121 
1122                 // finally, push a copy of the current decision point list onto the
1123                 // stack (this keeps track of the active decision point list before
1124                 // the () expression), followed by an empty decision point list
1125                 // (this will hold the exit points)
1126                 decisionPointStack.push(decisionPointList.clone());
1127                 decisionPointStack.push(new Vector());
1128             }
1129 
1130             // a | separates alternative character sequences in a () expression.  When
1131             // a | is encountered, we add the current decision point list to the exit-point
1132             // list, and restore the decision point list to its state prior to the (.
1133             else if (c == '|') {
1134 
1135                 // pick out the top two decision point lists on the stack
1136                 Vector oneDown = (Vector)decisionPointStack.pop();
1137                 Vector twoDown = (Vector)decisionPointStack.peek();
1138                 decisionPointStack.push(oneDown);
1139 
1140                 // append the current decision point list to the list below it
1141                 // on the stack (the list of exit points), and restore the
1142                 // current decision point list to its state before the () expression
1143                 for (int i = 0; i < decisionPointList.size(); i++)
1144                     oneDown.addElement(decisionPointList.elementAt(i));
1145                 decisionPointList = (Vector)twoDown.clone();
1146             }
1147 
1148             // a ) marks the end of a sequence of characters.  We do one of two things
1149             // depending on whether the sequence repeats (i.e., whether the ) is followed
1150             // by *):  If the sequence doesn't repeat, then the exit-point list is merged
1151             // with the current decision point list and the decision point list from before
1152             // the () is thrown away.  If the sequence does repeat, then we fish out the
1153             // state we were in before the ( and copy all of its forward transitions
1154             // (i.e., every transition added by the () expression) into every state in the
1155             // exit-point list and the current decision point list.  The current decision
1156             // point list is then merged with both the exit-point list AND the saved version
1157             // of the decision point list from before the ().  Then we throw out the *.
1158             else if (c == ')') {
1159 
1160                 // pull the exit point list off the stack, merge it with the current
1161                 // decision point list, and make the merged version the current
1162                 // decision point list
1163                 Vector exitPoints = (Vector)decisionPointStack.pop();
1164                 for (int i = 0; i < decisionPointList.size(); i++)
1165                     exitPoints.addElement(decisionPointList.elementAt(i));
1166                 decisionPointList = exitPoints;
1167 
1168                 // if the ) isn't followed by a *, then all we have to do is throw
1169                 // away the other list on the decision point stack, and we're done
1170                 if (p + 1 >= rule.length() || rule.charAt(p + 1) != '*') {
1171                     decisionPointStack.pop();
1172                 }
1173 
1174                 // but if the sequence repeats, we have a lot more work to do...
1175                 else {
1176 
1177                     // now exitPoints and decisionPointList have to point to equivalent
1178                     // vectors, but not the SAME vector
1179                     exitPoints = (Vector)decisionPointList.clone();
1180 
1181                     // pop the original decision point list off the stack
1182                     Vector temp = (Vector)decisionPointStack.pop();
1183 
1184                     // we squirreled away the row number of our entry point list
1185                     // at the beginning of the original decision point list.  Fish
1186                     // that state number out and retrieve the entry point list
1187                     int tempStateNum = ((Integer)temp.firstElement()).intValue();
1188                     short[] tempState = (short[])tempStateTable.elementAt(tempStateNum);
1189 
1190                     // merge the original decision point list with the current
1191                     // decision point list
1192                     for (int i = 0; i < decisionPointList.size(); i++)
1193                         temp.addElement(decisionPointList.elementAt(i));
1194                     decisionPointList = temp;
1195 
1196                     // finally, copy every forward reference from the entry point
1197                     // list into every state in the new decision point list
1198                     for (int i = 0; i < tempState.length; i++) {
1199                         if (tempState[i] > tempStateNum) {
1200                             updateStateTable(exitPoints,
1201                                              new Character((char)(i + 0x100)).toString(),
1202                                              tempState[i]);
1203                         }
1204                     }
1205 
1206                     // update lastState and currentState, and throw away the *
1207                     lastState = currentState;
1208                     currentState = tempStateTable.size() - 1;
1209                     ++p;
1210                 }
1211             }
1212 
1213             // a / marks the position where the break is to go if the character sequence
1214             // matches this rule.  We update the flag word of every state on the decision
1215             // point list to mark them as ending states, and take note of the fact that
1216             // we've seen the slash
1217             else if (c == '/') {
1218                 sawEarlyBreak = true;
1219                 for (int i = 0; i < decisionPointList.size(); i++) {
1220                     state = (short[])tempStateTable.elementAt(((Integer)decisionPointList.
1221                                     elementAt(i)).intValue());
1222                     state[numCategories] |= LOOKAHEAD_STATE_FLAG;
1223                 }
1224             }
1225 
1226             // if we get here without executing any of the above clauses, we have a
1227             // syntax error.  However, for now we just ignore the offending character
1228             // and move on
1229 
1230             // clearLoopingStates is a signal back from updateStateTable() that we've
1231             // transitioned to a state that won't loop back to the current looping
1232             // state.  (In other words, we've gotten to a point where we can no longer
1233             // go back into a *? we saw earlier.)  Clear out the list of looping states
1234             // and backfill any states that need to be backfilled.
1235             if (clearLoopingStates) {
1236                 setLoopingStates(null, decisionPointList);
1237             }
1238 
1239             // advance to the next character, now that we've processed the current
1240             // character
1241             p += Character.charCount(c);
1242         }
1243 
1244         // this takes care of backfilling any states that still need to be backfilled
1245         setLoopingStates(null, decisionPointList);
1246 
1247         // when we reach the end of the string, we do a postprocessing step to mark the
1248         // end states.  The decision point list contains every state that can transition
1249         // to the end state-- that is, every state that is the last state in a sequence
1250         // that matches the rule.  All of these states are considered "mark states"
1251         // or "accepting states"-- that is, states that cause the position returned from
1252         // next() to be updated.  A mark state represents a possible break position.
1253         // This allows us to look ahead and remember how far the rule matched
1254         // before following the new branch (see next() for more information).
1255         // The temporary state table has an extra "flag column" at the end where this
1256         // information is stored.  We mark the end states by setting a flag in their
1257         // flag column.
1258         // Now if we saw the / in the rule, then everything after it is lookahead
1259         // material and the break really goes where the slash is.  In this case,
1260         // we mark these states as BOTH accepting states and lookahead states.  This
1261         // signals that these states cause the break position to be updated to the
1262         // position of the slash rather than the current break position.
1263         for (int i = 0; i < decisionPointList.size(); i++) {
1264             int rowNum = ((Integer)decisionPointList.elementAt(i)).intValue();
1265             state = (short[])tempStateTable.elementAt(rowNum);
1266             state[numCategories] |= END_STATE_FLAG;
1267             if (sawEarlyBreak) {
1268                 state[numCategories] |= LOOKAHEAD_STATE_FLAG;
1269             }
1270         }
1271     }

1290             newValues[(int)(pendingChars.charAt(i)) - 0x100] = newValue;

1299     /**
1300      * The real work of making the state table deterministic happens here.  This function
1301      * merges a state in the state table (specified by rowNum) with a state that is
1302      * passed in (newValues).  The basic process is to copy the nonzero cells in newStates
1303      * into the state in the state table (we'll call that oldValues).  If there's a
1304      * collision (i.e., if the same cell has a nonzero value in both states, and it's
1305      * not the SAME value), then we have to reconcile the collision.  We do this by
1306      * creating a new state, adding it to the end of the state table, and using this
1307      * function recursively to merge the original two states into a single, combined
1308      * state.  This process may happen recursively (i.e., each successive level may
1309      * involve collisions).  To prevent infinite recursion, we keep a log of merge
1310      * operations.  Any time we're merging two states we've merged before, we can just
1311      * supply the row number for the result of that merge operation rather than creating
1312      * a new state just like it.
1313      * @param rowNum The row number in the state table of the state to be updated
1314      * @param newValues The state to merge it with.
1315      * @param rowsBeingUpdated A copy of the list of rows passed to updateStateTable()
1316      * (itself a copy of the decision point list from parseRule()).  Newly-created
1317      * states get added to the decision point list if their "parents" were on it.
1318      */
1319     private void mergeStates(int rowNum,
1320                              short[] newValues,
1321                              Vector rowsBeingUpdated) {
1322         short[] oldValues = (short[])(tempStateTable.elementAt(rowNum));
1323         boolean isLoopingState = loopingStates.contains(new Integer(rowNum));
1324 
1325         // for each of the cells in the rows we're reconciling, do...
1326         for (int i = 0; i < oldValues.length; i++) {
1327 
1328             // if they contain the same value, we don't have to do anything
1329             if (oldValues[i] == newValues[i]) {
1330                 continue;
1331             }
1332 
1333             // if oldValues is a looping state and the state the current cell points to
1334             // is too, then we can just stomp over the current value of that cell (and
1335             // set the clear-looping-states flag if necessary)
1336             else if (isLoopingState && loopingStates.contains(new Integer(oldValues[i]))) {
1337                 if (newValues[i] != 0) {
1338                     if (oldValues[i] == 0) {
1339                         clearLoopingStates = true;
1340                     }
1341                     oldValues[i] = newValues[i];
1342                 }
1343             }
1344 
1345             // if the current cell in oldValues is 0, copy in the corresponding value
1346             // from newValues
1347             else if (oldValues[i] == 0) {
1348                 oldValues[i] = newValues[i];
1349             }
1350 
1351             // the last column of each row is the flag column.  Take care to merge the
1352             // flag words correctly
1353             else if (i == numCategories) {
1354                 oldValues[i] = (short)((newValues[i] & ALL_FLAGS) | oldValues[i]);
1355             }
1356 
1357             // if both newValues and oldValues have a nonzero value in the current
1358             // cell, and it isn't the same value both places...
1359             else if (oldValues[i] != 0 && newValues[i] != 0) {
1360 
1361                 // look up this pair of cell values in the merge list.  If it's
1362                 // found, update the cell in oldValues to point to the merged state
1363                 int combinedRowNum = searchMergeList(oldValues[i], newValues[i]);
1364                 if (combinedRowNum != 0) {
1365                     oldValues[i] = (short)combinedRowNum;
1366                 }
1367 
1368                 // otherwise, we have to reconcile them...
1369                 else {
1370                     // copy our row numbers into variables to make things easier
1371                     int oldRowNum = oldValues[i];
1372                     int newRowNum = newValues[i];
1373                     combinedRowNum = tempStateTable.size();
1374 
1375                     // add this pair of row numbers to the merge list (create it first
1376                     // if we haven't created the merge list yet)
1377                     if (mergeList == null) {
1378                         mergeList = new Vector();
1379                     }
1380                     mergeList.addElement(new int[] { oldRowNum, newRowNum, combinedRowNum });
1381 
1382                     // create a new row to represent the merged state, and copy the
1383                     // contents of oldRow into it, then add it to the end of the
1384                     // state table and update the original row (oldValues) to point
1385                     // to the new, merged, state
1386                     short[] newRow = new short[numCategories + 1];
1387                     short[] oldRow = (short[])(tempStateTable.elementAt(oldRowNum));
1388                     System.arraycopy(oldRow, 0, newRow, 0, numCategories + 1);
1389                     tempStateTable.addElement(newRow);
1390                     oldValues[i] = (short)combinedRowNum;
1391 
1392                     // if the decision point list contains either of the parent rows,
1393                     // update it to include the new row as well
1394                     if ((decisionPointList.contains(new Integer(oldRowNum))
1395                             || decisionPointList.contains(new Integer(newRowNum)))
1396                         && !decisionPointList.contains(new Integer(combinedRowNum))
1397                     ) {
1398                         decisionPointList.addElement(new Integer(combinedRowNum));
1399                     }
1400 
1401                     // do the same thing with the list of rows being updated
1402                     if ((rowsBeingUpdated.contains(new Integer(oldRowNum))
1403                             || rowsBeingUpdated.contains(new Integer(newRowNum)))
1404                         && !rowsBeingUpdated.contains(new Integer(combinedRowNum))
1405                     ) {
1406                         decisionPointList.addElement(new Integer(combinedRowNum));
1407                     }
1408                     // now (groan) do the same thing for all the entries on the
1409                     // decision point stack
1410                     for (int k = 0; k < decisionPointStack.size(); k++) {
1411                         Vector dpl = (Vector)decisionPointStack.elementAt(k);
1412                         if ((dpl.contains(new Integer(oldRowNum))
1413                                 || dpl.contains(new Integer(newRowNum)))
1414                             && !dpl.contains(new Integer(combinedRowNum))
1415                         ) {
1416                             dpl.addElement(new Integer(combinedRowNum));
1417                         }
1418                     }
1419 
1420                     // FINALLY (puff puff puff), call mergeStates() recursively to copy
1421                     // the row referred to by newValues into the new row and resolve any
1422                     // conflicts that come up at that level
1423                     mergeStates(combinedRowNum, (short[])(tempStateTable.elementAt(
1424                                     newValues[i])), rowsBeingUpdated);
1425                 }
1426             }
1427         }
1428         return;
1429     }

1431     /**
1432      * The merge list is a list of pairs of rows that have been merged somewhere in
1433      * the process of building this state table, along with the row number of the
1434      * row containing the merged state.  This function looks up a pair of row numbers
1435      * and returns the row number of the row they combine into.  (It returns 0 if
1436      * this pair of rows isn't in the merge list.)
1437      */
1438     private int searchMergeList(int a, int b) {
1439         // if there is no merge list, there obviously isn't anything in it
1440         if (mergeList == null) {
1441             return 0;
1442         }
1443 
1444         // otherwise, for each element in the merge list...
1445         else {
1446             int[] entry;
1447             for (int i = 0; i < mergeList.size(); i++) {
1448                 entry = (int[])(mergeList.elementAt(i));
1449 
1450                 // we have a hit if the two row numbers match the two row numbers
1451                 // in the beginning of the entry (the two that combine), in either
1452                 // order
1453                 if ((entry[0] == a && entry[1] == b) || (entry[0] == b && entry[1] == a)) {
1454                     return entry[2];
1455                 }
1456 
1457                 // we also have a hit if one of the two row numbers matches the marged
1458                 // row number and the other one matches one of the original row numbers
1459                 if ((entry[2] == a && (entry[0] == b || entry[1] == b))) {
1460                     return entry[2];
1461                 }
1462                 if ((entry[2] == b && (entry[0] == a || entry[1] == a))) {
1463                     return entry[2];
1464                 }
1465             }
1466             return 0;
1467         }
1468     }

1470     /**
1471      * This function is used to update the list of current loooping states (i.e.,
1472      * states that are controlled by a *? construct).  It backfills values from
1473      * the looping states into unpopulated cells of the states that are currently
1474      * marked for backfilling, and then updates the list of looping states to be
1475      * the new list
1476      * @param newLoopingStates The list of new looping states
1477      * @param endStates The list of states to treat as end states (states that
1478      * can exit the loop).
1479      */
1480     private void setLoopingStates(Vector newLoopingStates, Vector endStates) {
1481 
1482         // if the current list of looping states isn't empty, we have to backfill
1483         // values from the looping states into the states that are waiting to be
1484         // backfilled
1485         if (!loopingStates.isEmpty()) {
1486             int loopingState = ((Integer)loopingStates.lastElement()).intValue();
1487             int rowNum;
1488 
1489             // don't backfill into an end state OR any state reachable from an end state
1490             // (since the search for reachable states is recursive, it's split out into
1491             // a separate function, eliminateBackfillStates(), below)
1492             for (int i = 0; i < endStates.size(); i++) {
1493                 eliminateBackfillStates(((Integer)endStates.elementAt(i)).intValue());
1494             }
1495 
1496             // we DON'T actually backfill the states that need to be backfilled here.
1497             // Instead, we MARK them for backfilling.  The reason for this is that if
1498             // there are multiple rules in the state-table description, the looping
1499             // states may have some of their values changed by a succeeding rule, and
1500             // this wouldn't be reflected in the backfilled states.  We mark a state
1501             // for backfilling by putting the row number of the state to copy from
1502             // into the flag cell at the end of the row
1503             for (int i = 0; i < statesToBackfill.size(); i++) {
1504                 rowNum = ((Integer)statesToBackfill.elementAt(i)).intValue();
1505                 short[] state = (short[])tempStateTable.elementAt(rowNum);
1506                 state[numCategories] =
1507                     (short)((state[numCategories] & ALL_FLAGS) | loopingState);
1508             }
1509             statesToBackfill.removeAllElements();
1510             loopingStates.removeAllElements();
1511         }
1512 
1513         if (newLoopingStates != null) {
1514             loopingStates = (Vector)newLoopingStates.clone();
1515         }
1516     }

1519      * This removes "ending states" and states reachable from them from the
1520      * list of states to backfill.
1521      * @param The row number of the state to remove from the backfill list
1522      */
1523     private void eliminateBackfillStates(int baseState) {
1524 
1525         // don't do anything unless this state is actually in the backfill list...
1526         if (statesToBackfill.contains(new Integer(baseState))) {
1527 
1528             // if it is, take it out
1529             statesToBackfill.removeElement(new Integer(baseState));
1530 
1531             // then go through and recursively call this function for every
1532             // state that the base state points to
1533             short[] state = (short[])tempStateTable.elementAt(baseState);
1534             for (int i = 0; i < numCategories; i++) {
1535                 if (state[i] != 0) {
1536                     eliminateBackfillStates(state[i]);
1537                 }
1538             }
1539         }

1542     /**
1543      * This function completes the backfilling process by actually doing the
1544      * backfilling on the states that are marked for it
1545      */
1546     private void backfillLoopingStates() {
1547         short[] state;
1548         short[] loopingState = null;
1549         int loopingStateRowNum = 0;
1550         int fromState;
1551 
1552         // for each state in the state table...
1553         for (int i = 0; i < tempStateTable.size(); i++) {
1554             state = (short[])tempStateTable.elementAt(i);
1555 
1556             // check the state's flag word to see if it's marked for backfilling
1557             // (it's marked for backfilling if any bits other than the two high-order
1558             // bits are set-- if they are, then the flag word, minus the two high bits,
1559             // is the row number to copy from)
1560             fromState = state[numCategories] & ~ALL_FLAGS;
1561             if (fromState > 0) {
1562 
1563                 // load up the state to copy from (if we haven't already)
1564                 if (fromState != loopingStateRowNum) {
1565                     loopingStateRowNum = fromState;
1566                     loopingState = (short[])tempStateTable.elementAt(loopingStateRowNum);
1567                 }
1568 
1569                 // clear out the backfill part of the flag word
1570                 state[numCategories] &= ALL_FLAGS;
1571 
1572                 // then fill all zero cells in the current state with values
1573                 // from the corresponding cells of the fromState
1574                 for (int j = 0; j < state.length; j++) {
1575                     if (state[j] == 0) {
1576                         state[j] = loopingState[j];
1577                     }
1578                     else if (state[j] == DONT_LOOP_FLAG) {
1579                         state[j] = 0;
1580                     }
1581                 }
1582             }
1583         }
1584     }

1586     /**
1587      * This function completes the state-table-building process by doing several
1588      * postprocessing steps and copying everything into its final resting place
1589      * in the iterator itself
1590      * @param forward True if we're working on the forward state table
1591      */
1592     private void finishBuildingStateTable(boolean forward) {
1593         // start by backfilling the looping states
1594         backfillLoopingStates();
1595 
1596         int[] rowNumMap = new int[tempStateTable.size()];
1597         Stack rowsToFollow = new Stack();
1598         rowsToFollow.push(new Integer(1));
1599         rowNumMap[1] = 1;
1600 
1601         // determine which states are no longer reachable from the start state
1602         // (the reachable states will have their row numbers in the row number
1603         // map, and the nonreachable states will have zero in the row number map)
1604         while (rowsToFollow.size() != 0) {
1605             int rowNum = ((Integer)rowsToFollow.pop()).intValue();
1606             short[] row = (short[])(tempStateTable.elementAt(rowNum));
1607 
1608             for (int i = 0; i < numCategories; i++) {
1609                 if (row[i] != 0) {
1610                     if (rowNumMap[row[i]] == 0) {
1611                         rowNumMap[row[i]] = row[i];
1612                         rowsToFollow.push(new Integer(row[i]));
1613                     }
1614                 }
1615             }
1616         }
1617 
1618         boolean madeChange;
1619         int newRowNum;
1620 
1621         // algorithm for minimizing the number of states in the table adapted from
1622         // Aho & Ullman, "Principles of Compiler Design"
1623         // The basic idea here is to organize the states into classes.  When we're done,
1624         // all states in the same class can be considered identical and all but one eliminated.
1625 
1626         // initially assign states to classes based on the number of populated cells they
1627         // contain (the class number is the number of populated cells)
1628         int[] stateClasses = new int[tempStateTable.size()];
1629         int nextClass = numCategories + 1;
1630         short[] state1, state2;
1631         for (int i = 1; i < stateClasses.length; i++) {
1632             if (rowNumMap[i] == 0) {
1633                 continue;
1634             }
1635             state1 = (short[])tempStateTable.elementAt(i);
1636             for (int j = 0; j < numCategories; j++) {
1637                 if (state1[j] != 0) {
1638                     ++stateClasses[i];
1639                 }
1640             }
1641             if (stateClasses[i] == 0) {
1642                 stateClasses[i] = nextClass;
1643             }
1644         }
1645         ++nextClass;
1646 
1647         // then, for each class, elect the first member of that class as that class's
1648         // "representative".  For each member of the class, compare it to the "representative."
1649         // If there's a column position where the state being tested transitions to a
1650         // state in a DIFFERENT class from the class where the "representative" transitions,
1651         // then move the state into a new class.  Repeat this process until no new classes
1652         // are created.
1653         int currentClass;
1654         int lastClass;
1655         boolean split;
1656 
1657         do {
1658             currentClass = 1;
1659             lastClass = nextClass;
1660             while (currentClass < nextClass) {
1661                 split = false;
1662                 state1 = state2 = null;
1663                 for (int i = 0; i < stateClasses.length; i++) {
1664                     if (stateClasses[i] == currentClass) {
1665                         if (state1 == null) {
1666                             state1 = (short[])tempStateTable.elementAt(i);
1667                         }
1668                         else {
1669                             state2 = (short[])tempStateTable.elementAt(i);
1670                             for (int j = 0; j < state2.length; j++) {
1671                                 if ((j == numCategories && state1[j] != state2[j] && forward)
1672                                         || (j != numCategories && stateClasses[state1[j]]
1673                                         != stateClasses[state2[j]])) {
1674                                     stateClasses[i] = nextClass;
1675                                     split = true;
1676                                     break;
1677                                 }
1678                             }
1679                         }
1680                     }
1681                 }
1682                 if (split) {
1683                     ++nextClass;
1684                 }
1685                 ++currentClass;
1686             }
1687         } while (lastClass != nextClass);
1688 
1689         // at this point, all of the states in a class except the first one (the
1690         //"representative") can be eliminated, so update the row-number map accordingly
1691         int[] representatives = new int[nextClass];
1692         for (int i = 1; i < stateClasses.length; i++)
1693             if (representatives[stateClasses[i]] == 0) {
1694                 representatives[stateClasses[i]] = i;
1695             }
1696             else {
1697                 rowNumMap[i] = representatives[stateClasses[i]];
1698             }
1699 
1700         // renumber all remaining rows...
1701         // first drop all that are either unreferenced or not a class representative
1702         for (int i = 1; i < rowNumMap.length; i++) {
1703             if (rowNumMap[i] != i) {
1704                 tempStateTable.setElementAt(null, i);
1705             }
1706         }
1707 
1708         // then calculate everybody's new row number and update the row
1709         // number map appropriately (the first pass updates the row numbers
1710         // of all the class representatives [the rows we're keeping], and the
1711         // second pass updates the cross references for all the rows that
1712         // are being deleted)
1713         newRowNum = 1;
1714         for (int i = 1; i < rowNumMap.length; i++) {
1715             if (tempStateTable.elementAt(i) != null) {
1716                 rowNumMap[i] = newRowNum++;
1717             }
1718         }
1719         for (int i = 1; i < rowNumMap.length; i++) {
1720             if (tempStateTable.elementAt(i) == null) {
1721                 rowNumMap[i] = rowNumMap[rowNumMap[i]];
1722             }
1723         }
1724 
1725         // allocate the permanent state table, and copy the remaining rows into it
1726         // (adjusting all the cell values, of course)
1727 
1728         // this section does that for the forward state table
1729         if (forward) {
1730             endStates = new boolean[newRowNum];
1731             lookaheadStates = new boolean[newRowNum];
1732             stateTable = new short[newRowNum * numCategories];
1733             int p = 0;
1734             int p2 = 0;
1735             for (int i = 0; i < tempStateTable.size(); i++) {
1736                 short[] row = (short[])(tempStateTable.elementAt(i));
1737                 if (row == null) {
1738                     continue;
1739                 }
1740                 for (int j = 0; j < numCategories; j++) {
1741                     stateTable[p] = (short)(rowNumMap[row[j]]);
1742                     ++p;
1743                 }
1744                 endStates[p2] = ((row[numCategories] & END_STATE_FLAG) != 0);
1745                 lookaheadStates[p2] = ((row[numCategories] & LOOKAHEAD_STATE_FLAG) != 0);
1746                 ++p2;
1747             }
1748         }
1749 
1750         // and this section does it for the backward state table
1751         else {
1752             backwardsStateTable = new short[newRowNum * numCategories];
1753             int p = 0;
1754             for (int i = 0; i < tempStateTable.size(); i++) {
1755                 short[] row = (short[])(tempStateTable.elementAt(i));
1756                 if (row == null) {
1757                     continue;
1758                 }
1759                 for (int j = 0; j < numCategories; j++) {
1760                     backwardsStateTable[p] = (short)(rowNumMap[row[j]]);
1761                     ++p;
1762                 }
1763             }
1764         }
1765     }

1767     /**
1768      * This function builds the backward state table from the forward state
1769      * table and any additional rules (identified by the ! on the front)
1770      * supplied in the description
1771      */
1772     private void buildBackwardsStateTable(Vector tempRuleList) {
1773 
1774         // create the temporary state table and seed it with two rows (row 0
1775         // isn't used for anything, and we have to create row 1 (the initial
1776         // state) before we can do anything else
1777         tempStateTable = new Vector();
1778         tempStateTable.addElement(new short[numCategories + 1]);
1779         tempStateTable.addElement(new short[numCategories + 1]);
1780 
1781         // although the backwards state table is built automatically from the forward
1782         // state table, there are some situations (the default sentence-break rules,
1783         // for example) where this doesn't yield enough stop states, causing a dramatic
1784         // drop in performance.  To help with these cases, the user may supply
1785         // supplemental rules that are added to the backward state table.  These have
1786         // the same syntax as the normal break rules, but begin with '!' to distinguish
1787         // them from normal break rules
1788         for (int i = 0; i < tempRuleList.size(); i++) {
1789             String rule = (String)tempRuleList.elementAt(i);
1790             if (rule.charAt(0) == '!') {
1791                 parseRule(rule.substring(1), false);
1792             }
1793         }
1794         backfillLoopingStates();
1795 
1796         // Backwards iteration is qualitatively different from forwards iteration.
1797         // This is because backwards iteration has to be made to operate from no context
1798         // at all-- the user should be able to ask BreakIterator for the break position
1799         // immediately on either side of some arbitrary offset in the text.  The
1800         // forward iteration table doesn't let us do that-- it assumes complete
1801         // information on the context, which means starting from the beginning of the
1802         // document.
1803         // The way we do backward and random-access iteration is to back up from the
1804         // current (or user-specified) position until we see something we're sure is
1805         // a break position (it may not be the last break position immediately
1806         // preceding our starting point, however).  Then we roll forward from there to
1807         // locate the actual break position we're after.
1808         // This means that the backwards state table doesn't have to identify every
1809         // break position, allowing the building algorithm to be much simpler.  Here,
1810         // we use a "pairs" approach, scanning the forward-iteration state table for
1811         // pairs of character categories we ALWAYS break between, and building a state
1812         // table from that information.  No context is required-- all this state table
1813         // looks at is a pair of adjacent characters.
1814 
1815         // It's possible that the user has supplied supplementary rules (see above).
1816         // This has to be done first to keep parseRule() and friends from becoming
1817         // EVEN MORE complicated.  The automatically-generated states are appended
1818         // onto the end of the state table, and then the two sets of rules are
1819         // stitched together at the end.  Take note of the row number of the
1820         // first row of the auromatically-generated part.
1821         int backTableOffset = tempStateTable.size();
1822         if (backTableOffset > 2) {
1823             ++backTableOffset;
1824         }
1825 
1826         // the automatically-generated part of the table models a two-dimensional
1827         // array where the two dimensions represent the two characters we're currently
1828         // looking at.  To model this as a state table, we actually need one additional
1829         // row to represent the initial state.  It gets populated with the row numbers
1830         // of the other rows (in order).
1831         for (int i = 0; i < numCategories + 1; i++)
1832             tempStateTable.addElement(new short[numCategories + 1]);
1833 
1834         short[] state = (short[])tempStateTable.elementAt(backTableOffset - 1);
1835         for (int i = 0; i < numCategories; i++)
1836             state[i] = (short)(i + backTableOffset);
1837 
1838         // scavenge the forward state table for pairs of character categories
1839         // that always have a break between them.  The algorithm is as follows:
1840         // Look down each column in the state table.  For each nonzero cell in
1841         // that column, look up the row it points to.  For each nonzero cell in
1842         // that row, populate a cell in the backwards state table: the row number
1843         // of that cell is the number of the column we were scanning (plus the
1844         // offset that locates this sub-table), and the column number of that cell
1845         // is the column number of the nonzero cell we just found.  This cell is
1846         // populated with its own column number (adjusted according to the actual
1847         // location of the sub-table).  This process will produce a state table
1848         // whose behavior is the same as looking up successive pairs of characters
1849         // in an array of Booleans to determine whether there is a break.
1850         int numRows = stateTable.length / numCategories;
1851         for (int column = 0; column < numCategories; column++) {
1852             for (int row = 0; row < numRows; row++) {
1853                 int nextRow = lookupState(row, column);
1854                 if (nextRow != 0) {
1855                     for (int nextColumn = 0; nextColumn < numCategories; nextColumn++) {
1856                         int cellValue = lookupState(nextRow, nextColumn);
1857                         if (cellValue != 0) {
1858                             state = (short[])tempStateTable.elementAt(nextColumn +
1859                                             backTableOffset);
1860                             state[column] = (short)(column + backTableOffset);
1861                         }
1862                     }
1863                 }
1864             }
1865         }
1866 
1867         // if the user specified some backward-iteration rules with the ! token,
1868         // we have to merge the resulting state table with the auto-generated one
1869         // above.  First copy the populated cells from row 1 over the populated
1870         // cells in the auto-generated table.  Then copy values from row 1 of the
1871         // auto-generated table into all of the the unpopulated cells of the
1872         // rule-based table.
1873         if (backTableOffset > 1) {
1874 
1875             // for every row in the auto-generated sub-table, if a cell is
1876             // populated that is also populated in row 1 of the rule-based
1877             // sub-table, copy the value from row 1 over the value in the
1878             // auto-generated sub-table
1879             state = (short[])tempStateTable.elementAt(1);
1880             for (int i = backTableOffset - 1; i < tempStateTable.size(); i++) {
1881                 short[] state2 = (short[])tempStateTable.elementAt(i);
1882                 for (int j = 0; j < numCategories; j++) {
1883                     if (state[j] != 0 && state2[j] != 0) {
1884                         state2[j] = state[j];
1885                     }
1886                 }
1887             }
1888 
1889             // now, for every row in the rule-based sub-table that is not
1890             // an end state, fill in all unpopulated cells with the values
1891             // of the corresponding cells in the first row of the auto-
1892             // generated sub-table.
1893             state = (short[])tempStateTable.elementAt(backTableOffset - 1);
1894             for (int i = 1; i < backTableOffset - 1; i++) {
1895                 short[] state2 = (short[])tempStateTable.elementAt(i);
1896                 if ((state2[numCategories] & END_STATE_FLAG) == 0) {
1897                     for (int j = 0; j < numCategories; j++) {
1898                         if (state2[j] == 0) {
1899                             state2[j] = state[j];
1900                         }
1901                     }
1902                 }
1903             }
1904         }
1905 
1906         // finally, clean everything up and copy it into the actual BreakIterator
1907         // by calling finishBuildingStateTable()
1908         finishBuildingStateTable(false);
1909     }

1959     /**
1960      * Read datafile. The datafile's format is as follows:
1961      * <pre>
1962      *   BreakIteratorData {
1963      *       u1           magic[7];
1964      *       u1           version;
1965      *       u4           totalDataSize;
1966      *       header_info  header;
1967      *       body         value;
1968      *   }
1969      * </pre>
1970      * <code>totalDataSize</code> is the summation of the size of
1971      * <code>header_info</code> and <code>body</code> in byte count.
1972      * <p>
1973      * In <code>header</code>, each field except for checksum implies the
1974      * length of each field. Since <code>BMPdataLength</code> is a fixed-length
1975      *  data(512 entries), its length isn't included in <code>header</code>.
1976      * <code>checksum</code> is a CRC32 value of all in <code>body</code>.
1977      * <pre>
1978      *   header_info {
1979      *       u4           stateTableLength;
1980      *       u4           backwardsStateTableLength;
1981      *       u4           endStatesLength;
1982      *       u4           lookaheadStatesLength;
1983      *       u4           BMPdataLength;
1984      *       u4           nonBMPdataLength;
1985      *       u4           additionalDataLength;
1986      *       u8           checksum;
1987      *   }
1988      * </pre>
1989      * <p>
1990      *
1991      * Finally, <code>BMPindices</code> and <code>BMPdata</code> are set to
1992      * <code>charCategoryTable</code>. <code>nonBMPdata</code> is set to
1993      * <code>supplementaryCharCategoryTable</code>.
1994      * <pre>
1995      *   body {
1996      *       u2           stateTable[stateTableLength];
1997      *       u2           backwardsStateTable[backwardsStateTableLength];
1998      *       u1           endStates[endStatesLength];
1999      *       u1           lookaheadStates[lookaheadStatesLength];
2000      *       u2           BMPindices[512];
2001      *       u1           BMPdata[BMPdataLength];
2002      *       u4           nonBMPdata[numNonBMPdataLength];
2003      *       u1           additionalData[additionalDataLength];
2004      *   }
2005      * </pre>
2006      */
2007     protected void writeTables(String datafile) {
2008         final String filename;
2009         final String outputDir;
2010         String tmpbuf = GenerateBreakIteratorData.getOutputDirectory();
2011 
2012         if (tmpbuf.equals("")) {
2013             filename = datafile;
2014             outputDir = "";
2015         } else {
2016             char sep = File.separatorChar;
2017             if (sep == '/') {
2018                 outputDir = tmpbuf;
2019             } else if (sep == '\\') {
2020                 outputDir = tmpbuf.replaceAll("/", "\\\\");
2021             } else {
2022                 outputDir = tmpbuf.replaceAll("/", String.valueOf(sep));
2023             }
2024 
2025             filename = outputDir + sep + datafile;
2026         }
2027 
2028         try {
2029             if (!outputDir.equals("")) {
2030                 new File(outputDir).mkdirs();
2031             }
2032             BufferedOutputStream out = new BufferedOutputStream(new FileOutputStream(filename));
2033 
2034             byte[] BMPdata = charCategoryTable.getStringArray();
2035             short[] BMPindices = charCategoryTable.getIndexArray();
2036             int[] nonBMPdata = supplementaryCharCategoryTable.getArray();
2037 
2038             if (BMPdata.length <= 0) {
2039                 throw new InternalError("Wrong BMP data length(" + BMPdata.length + ")");
2040             }
2041             if (BMPindices.length != BMP_INDICES_LENGTH) {
2042                 throw new InternalError("Wrong BMP indices length(" + BMPindices.length + ")");
2043             }
2044             if (nonBMPdata.length <= 0) {
2045                 throw new InternalError("Wrong non-BMP data length(" + nonBMPdata.length + ")");
2046             }
2047 
2048             int len;
2049 
2050             /* Compute checksum */
2051             CRC32 crc32 = new CRC32();
2052             len = stateTable.length;
2053             for (int i = 0; i < len; i++) {
2054                 crc32.update(stateTable[i]);
2055             }
2056             len = backwardsStateTable.length;
2057             for (int i = 0; i < len; i++) {
2058                 crc32.update(backwardsStateTable[i]);
2059             }
2060             crc32.update(toByteArray(endStates));
2061             crc32.update(toByteArray(lookaheadStates));
2062             for (int i = 0; i < BMP_INDICES_LENGTH; i++) {
2063                 crc32.update(BMPindices[i]);
2064             }
2065             crc32.update(BMPdata);
2066             len = nonBMPdata.length;
2067             for (int i = 0; i < len; i++) {
2068                 crc32.update(nonBMPdata[i]);
2069             }
2070             if (additionalData != null) {
2071                 len = additionalData.length;
2072                 for (int i = 0; i < len; i++) {
2073                     crc32.update(additionalData[i]);
2074                 }
2075             }
2076 
2077             /* First, write magic, version, and totalDataSize. */
2078             len = HEADER_LENGTH +
2079                   (stateTable.length + backwardsStateTable.length) * 2 +
2080                   endStates.length + lookaheadStates.length + 1024 +
2081                   BMPdata.length + nonBMPdata.length * 4 +
2082                   ((additionalData == null) ? 0 : additionalData.length);
2083             out.write(LABEL);
2084             out.write(supportedVersion);
2085             out.write(toByteArray(len));
2086 
2087             /* Write header_info. */
2088             out.write(toByteArray(stateTable.length));
2089             out.write(toByteArray(backwardsStateTable.length));
2090             out.write(toByteArray(endStates.length));
2091             out.write(toByteArray(lookaheadStates.length));
2092             out.write(toByteArray(BMPdata.length));
2093             out.write(toByteArray(nonBMPdata.length));
2094             if (additionalData == null) {
2095                 out.write(toByteArray(0));
2096             } else {
2097                 out.write(toByteArray(additionalData.length));
2098             }
2099             out.write(toByteArray(crc32.getValue()));
2100 
2101             /* Write stateTable[numCategories * numRows] */
2102             len = stateTable.length;
2103             for (int i = 0; i < len; i++) {
2104                 out.write(toByteArray(stateTable[i]));
2105             }
2106 
2107             /* Write backwardsStateTable[numCategories * numRows] */
2108             len = backwardsStateTable.length;
2109             for (int i = 0; i < len; i++) {
2110                 out.write(toByteArray(backwardsStateTable[i]));
2111             }
2112 
2113             /* Write endStates[numRows] */
2114             out.write(toByteArray(endStates));
2115 
2116             /* Write lookaheadStates[numRows] */
2117             out.write(toByteArray(lookaheadStates));
2118 
2119             for (int i = 0; i < BMP_INDICES_LENGTH; i++) {
2120                 out.write(toByteArray(BMPindices[i]));
2121             }
2122             BMPindices = null;
2123             out.write(BMPdata);
2124             BMPdata = null;
2125 
2126             /* Write a category table for non-BMP characters. */
2127             len = nonBMPdata.length;
2128             for (int i = 0; i < len; i++) {
2129                 out.write(toByteArray(nonBMPdata[i]));
2130             }
2131             nonBMPdata = null;
2132 
2133             /* Write additional data */
2134             if (additionalData != null) {
2135                 out.write(additionalData);
2136             }
2137 
2138             out.close();
2139         }
2140         catch (Exception e) {
2141             throw new InternalError(e.toString());
2142         }
2143     }
