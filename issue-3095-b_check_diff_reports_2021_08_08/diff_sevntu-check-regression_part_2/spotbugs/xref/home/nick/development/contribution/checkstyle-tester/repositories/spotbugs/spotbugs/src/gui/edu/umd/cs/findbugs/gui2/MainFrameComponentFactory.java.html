217     Component bugSummaryComponent(BugAnnotation value, BugInstance bug) {
218         JLabel label = new JLabel();
219         label.setFont(label.getFont().deriveFont(Driver.getFontSize()));
220         label.setFont(label.getFont().deriveFont(Font.PLAIN));
221         label.setForeground(Color.BLACK);
222         ClassAnnotation primaryClass = bug.getPrimaryClass();
223 
224         String sourceCodeLabel = L10N.getLocalString("summary.source_code", "source code.");
225         String summaryLine = L10N.getLocalString("summary.line", "Line");
226         String summaryLines = L10N.getLocalString("summary.lines", "Lines");
227         String clickToGoToText = L10N.getLocalString("tooltip.click_to_go_to", "Click to go to");
228         if (value instanceof SourceLineAnnotation) {
229             final SourceLineAnnotation link = (SourceLineAnnotation) value;
230             if (sourceCodeExists(link)) {
231                 String srcStr = "";
232                 int start = link.getStartLine();
233                 int end = link.getEndLine();
234                 if (start < 0 && end < 0) {
235                     srcStr = sourceCodeLabel;
236                 } else if (start == end) {
237                     srcStr = " [" + summaryLine + " " + start + "]";
238                 } else if (start < end) {
239                     srcStr = " [" + summaryLines + " " + start + " - " + end + "]";
240                 }
241 
242                 label.setToolTipText(clickToGoToText + " " + srcStr);
243 
244                 label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
245             }
246 
247             label.setText(link.toString());
248         } else if (value instanceof BugAnnotationWithSourceLines) {
249             BugAnnotationWithSourceLines note = (BugAnnotationWithSourceLines) value;
250             final SourceLineAnnotation link = note.getSourceLines();
251             String srcStr = "";
252             if (link != null && sourceCodeExists(link)) {
253                 int start = link.getStartLine();
254                 int end = link.getEndLine();
255                 if (start < 0 && end < 0) {
256                     srcStr = sourceCodeLabel;
257                 } else if (start == end) {
258                     srcStr = " [" + summaryLine + " " + start + "]";
259                 } else if (start < end) {
260                     srcStr = " [" + summaryLines + " " + start + " - " + end + "]";
261                 }
262 
263                 if (!"".equals(srcStr)) {
264                     label.setToolTipText(clickToGoToText + " " + srcStr);
265                     label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
266                 }
267             }
268             String noteText;
269             if (note == bug.getPrimaryMethod() || note == bug.getPrimaryField()) {
270                 noteText = note.toString();
271             } else {
272                 noteText = note.toString(primaryClass);
273             }
274             if (!srcStr.equals(sourceCodeLabel)) {
275                 label.setText(noteText + srcStr);
276             } else {
277                 label.setText(noteText);
278             }
279         } else {
280             label.setText(value.toString(primaryClass));
281         }
282 
283         return label;
284     }

286     /**
287      * Creates bug summary component. If obj is a string will create a JLabel
288      * with that string as it's text and return it. If obj is an annotation will
289      * return a JLabel with the annotation's toString(). If that annotation is a
290      * SourceLineAnnotation or has a SourceLineAnnotation connected to it and
291      * the source file is available will attach a listener to the label.
292      */
293     public Component bugSummaryComponent(String str, BugInstance bug) {
294         JLabel label = new JLabel();
295         label.setFont(label.getFont().deriveFont(Driver.getFontSize()));
296         label.setFont(label.getFont().deriveFont(Font.PLAIN));
297         label.setForeground(Color.BLACK);
298 
299         label.setText(str);
300 
301         SourceLineAnnotation link = bug.getPrimarySourceLineAnnotation();
302         if (link != null) {
303             label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
304         }
305 
306         return label;
307     }

309     private boolean sourceCodeExists(@Nonnull SourceLineAnnotation note) {
310         try {
311             mainFrame.getProject().getSourceFinder().findSourceFile(note);
312         } catch (FileNotFoundException e) {
313             return false;
314         } catch (IOException e) {
315             return false;
316         }
317         return true;
318     }

323         public InitializeGUI(final MainFrame mainFrame) {

327         @Override
328         public void run() {
329             mainFrame.setTitle("SpotBugs");
330             // noinspection ConstantConditions
331             if (MainFrame.USE_WINDOWS_LAF && System.getProperty("os.name").toLowerCase().contains("windows")) {
332                 try {
333                     UIManager.setLookAndFeel(UIManager.getSystemLookAndFeelClassName());
334                 } catch (Exception e) {
335                     LOGGER.log(Level.SEVERE, "Could not load Windows Look&Feel", e);
336                 }
337             }
338 
339             try {
340                 mainFrame.getGuiLayout().initialize();
341             } catch (Exception e) {
342                 // If an exception was encountered while initializing, this may
343                 // be because of a bug in the particular look-and-feel selected
344                 // (as in sourceforge bug 1899648). In an attempt to recover
345                 // gracefully, this code reverts to the cross-platform look-
346                 // and-feel and attempts again to initialize the layout.
347                 if (!"Metal".equals(UIManager.getLookAndFeel().getName())) {
348                     System.err.println("Exception caught initializing GUI; reverting to CrossPlatformLookAndFeel");
349                     try {
350                         UIManager.setLookAndFeel(UIManager.getCrossPlatformLookAndFeelClassName());
351                     } catch (Exception e2) {
352                         System.err.println("Exception while setting CrossPlatformLookAndFeel: " + e2);
353                         throw new Error(e2);
354                     }
355                     mainFrame.getGuiLayout().initialize();
356                 } else {
357                     throw new Error(e);
358                 }
359             }
360             mainFrame.mainFrameTree.setBugPopupMenu(mainFrame.mainFrameTree.createBugPopupMenu());
361             mainFrame.mainFrameTree.setBranchPopupMenu(mainFrame.mainFrameTree.createBranchPopUpMenu());
362             mainFrame.updateStatusBar();
363             Rectangle bounds = GUISaveState.getInstance().getFrameBounds();
364             if (bounds != null) {
365                 mainFrame.setBounds(bounds);
366             }
367 
368             mainFrame.setExtendedState(GUISaveState.getInstance().getExtendedWindowState());
369             Toolkit.getDefaultToolkit().setDynamicLayout(true);
370             mainFrame.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
371             mainFrame.setJMenuBar(mainFrame.mainFrameMenu.createMainMenuBar());
372             mainFrame.setVisible(true);
373 
374             mainFrame.getMainFrameLoadSaveHelper().initialize();
375 
376             // Sets the size of the tooltip to match the rest of the GUI. -
377             // Kristin
378             JToolTip tempToolTip = mainFrame.mainFrameTree.getTableheader().createToolTip();
379             UIManager.put("ToolTip.font", new FontUIResource(tempToolTip.getFont().deriveFont(Driver.getFontSize())));
380 
381             setupOSX();
382 
383             String loadFromURL = SystemProperties.getOSDependentProperty("findbugs.loadBugsFromURL");
384 
385             if (loadFromURL != null) {
386                 try {
387                     loadFromURL = SystemProperties.rewriteURLAccordingToProperties(loadFromURL);
388                     URL url = new URL(loadFromURL);
389                     mainFrame.getMainFrameLoadSaveHelper().loadAnalysis(url);
390                 } catch (MalformedURLException e1) {
391                     JOptionPane.showMessageDialog(mainFrame, "Error loading " + loadFromURL);
392                 }
393             }
394 
395             mainFrame.addWindowListener(new WindowAdapter() {
396                 @Override
397                 public void windowClosing(WindowEvent e) {
398                     mainFrame.callOnClose();
399                 }
400             });
401 
402             Driver.removeSplashScreen();
403             mainFrame.waitForMainFrameInitialized();
404         }

416                     .println("This version of Mac OS X does not support the Apple EAWT. Application Menu handling has been disabled ("
417                             + e + ")");
418                 } catch (ClassNotFoundException e) {
419                     // This shouldn't be reached; if there's a problem with the
420                     // OSXAdapter we should get the
421                     // above NoClassDefFoundError first.
422                     System.err
423                     .println("This version of Mac OS X does not support the Apple EAWT. Application Menu handling has been disabled ("
424                             + e + ")");
425                 } catch (Exception e) {
426                     System.err.println("Exception while loading the OSXAdapter: " + e);
427                     e.printStackTrace();
428                     if (MainFrame.GUI2_DEBUG) {
429                         e.printStackTrace();
430                     }
431                 }
432             }
433         }

436     /**
437      * Listens for when cursor is over the label and when it is clicked. When
438      * the cursor is over the label will make the label text blue and the cursor
439      * the hand cursor. When clicked will take the user to the source code tab
440      * and to the lines of code connected to the SourceLineAnnotation.
441      *
442      * @author Kristin Stephens
443      *
444      */
445     private class BugSummaryMouseListener extends MouseAdapter {
446         private final BugInstance bugInstance;
447 
448         private final JLabel label;
449 
450         private final SourceLineAnnotation note;
451 
452         BugSummaryMouseListener(@Nonnull BugInstance bugInstance, @Nonnull JLabel label, @Nonnull SourceLineAnnotation link) {
453             this.bugInstance = bugInstance;
454             this.label = label;
455             this.note = link;
456         }
457 
458         @Override
459         public void mouseClicked(MouseEvent e) {
460             mainFrame.getSourceCodeDisplayer().displaySource(bugInstance, note);
461         }
462 
463         @Override
464         public void mouseEntered(MouseEvent e) {
465             label.setForeground(Color.blue);
466             mainFrame.setCursor(new Cursor(Cursor.HAND_CURSOR));
467         }
468 
469         @Override
470         public void mouseExited(MouseEvent e) {
471             label.setForeground(Color.black);
472             mainFrame.setCursor(new Cursor(Cursor.DEFAULT_CURSOR));
473         }
474     }
