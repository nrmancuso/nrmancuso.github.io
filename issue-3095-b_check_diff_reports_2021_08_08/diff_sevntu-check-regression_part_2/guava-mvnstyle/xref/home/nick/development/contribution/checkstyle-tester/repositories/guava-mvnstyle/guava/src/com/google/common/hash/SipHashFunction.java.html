15  /*
16   * SipHash-c-d was designed by Jean-Philippe Aumasson and Daniel J. Bernstein and is described in
17   * "SipHash: a fast short-input PRF" (available at https://131002.net/siphash/siphash.pdf).
18   */

20  package com.google.common.hash;
21  
22  import static com.google.common.base.Preconditions.checkArgument;
23  
24  import com.google.errorprone.annotations.Immutable;
25  import java.io.Serializable;
26  import java.nio.ByteBuffer;
27  import org.checkerframework.checker.nullness.qual.Nullable;
28  
29  /**
30   * {@link HashFunction} implementation of SipHash-c-d.
31   *
32   * @author Kurt Alfred Kluever
33   * @author Jean-Philippe Aumasson
34   * @author Daniel J. Bernstein
35   */
36  @Immutable
37  final class SipHashFunction extends AbstractHashFunction implements Serializable {
38    static final HashFunction SIP_HASH_24 =
39        new SipHashFunction(2, 4, 0x0706050403020100L, 0x0f0e0d0c0b0a0908L);
40  
41    // The number of compression rounds.
42    private final int c;
43    // The number of finalization rounds.
44    private final int d;
45    // Two 64-bit keys (represent a single 128-bit key).
46    private final long k0;
47    private final long k1;
48  
49    /**
50     * @param c the number of compression rounds (must be positive)
51     * @param d the number of finalization rounds (must be positive)
52     * @param k0 the first half of the key
53     * @param k1 the second half of the key
54     */
55    SipHashFunction(int c, int d, long k0, long k1) {
56      checkArgument(
57          c > 0, "The number of SipRound iterations (c=%s) during Compression must be positive.", c);
58      checkArgument(
59          d > 0, "The number of SipRound iterations (d=%s) during Finalization must be positive.", d);
60      this.c = c;
61      this.d = d;
62      this.k0 = k0;
63      this.k1 = k1;
64    }
65  
66    @Override
67    public int bits() {
68      return 64;
69    }
70  
71    @Override
72    public Hasher newHasher() {
73      return new SipHasher(c, d, k0, k1);
74    }
75  
76    // TODO(kak): Implement and benchmark the hashFoo() shortcuts.
77  
78    @Override
79    public String toString() {
80      return "Hashing.sipHash" + c + "" + d + "(" + k0 + ", " + k1 + ")";
81    }
82  
83    @Override
84    public boolean equals(@Nullable Object object) {
85      if (object instanceof SipHashFunction) {
86        SipHashFunction other = (SipHashFunction) object;
87        return (c == other.c) && (d == other.d) && (k0 == other.k0) && (k1 == other.k1);
88      }
89      return false;
90    }
91  
92    @Override
93    public int hashCode() {
94      return (int) (getClass().hashCode() ^ c ^ d ^ k0 ^ k1);
95    }
96  
97    private static final class SipHasher extends AbstractStreamingHasher {
98      private static final int CHUNK_SIZE = 8;
99  
100     // The number of compression rounds.
101     private final int c;
102     // The number of finalization rounds.
103     private final int d;
104 
105     // Four 64-bit words of internal state.
106     // The initial state corresponds to the ASCII string "somepseudorandomlygeneratedbytes",
107     // big-endian encoded. There is nothing special about this value; the only requirement
108     // was some asymmetry so that the initial v0 and v1 differ from v2 and v3.
109     private long v0 = 0x736f6d6570736575L;
110     private long v1 = 0x646f72616e646f6dL;
111     private long v2 = 0x6c7967656e657261L;
112     private long v3 = 0x7465646279746573L;
113 
114     // The number of bytes in the input.
115     private long b = 0;
116 
117     // The final 64-bit chunk includes the last 0 through 7 bytes of m followed by null bytes
118     // and ending with a byte encoding the positive integer b mod 256.
119     private long finalM = 0;
120 
121     SipHasher(int c, int d, long k0, long k1) {
122       super(CHUNK_SIZE);
123       this.c = c;
124       this.d = d;
125       this.v0 ^= k0;
126       this.v1 ^= k1;
127       this.v2 ^= k0;
128       this.v3 ^= k1;
129     }
130 
131     @Override
132     protected void process(ByteBuffer buffer) {
133       b += CHUNK_SIZE;
134       processM(buffer.getLong());
135     }
136 
137     @Override
138     protected void processRemaining(ByteBuffer buffer) {
139       b += buffer.remaining();
140       for (int i = 0; buffer.hasRemaining(); i += 8) {
141         finalM ^= (buffer.get() & 0xFFL) << i;
142       }
143     }
144 
145     @Override
146     public HashCode makeHash() {
147       // End with a byte encoding the positive integer b mod 256.
148       finalM ^= b << 56;
149       processM(finalM);
150 
151       // Finalization
152       v2 ^= 0xFFL;
153       sipRound(d);
154       return HashCode.fromLong(v0 ^ v1 ^ v2 ^ v3);
155     }
156 
157     private void processM(long m) {
158       v3 ^= m;
159       sipRound(c);
160       v0 ^= m;
161     }
162 
163     private void sipRound(int iterations) {
164       for (int i = 0; i < iterations; i++) {
165         v0 += v1;
166         v2 += v3;
167         v1 = Long.rotateLeft(v1, 13);
168         v3 = Long.rotateLeft(v3, 16);
169         v1 ^= v0;
170         v3 ^= v2;
171         v0 = Long.rotateLeft(v0, 32);
172         v2 += v1;
173         v0 += v3;
174         v1 = Long.rotateLeft(v1, 17);
175         v3 = Long.rotateLeft(v3, 21);
176         v1 ^= v2;
177         v3 ^= v0;
178         v2 = Long.rotateLeft(v2, 32);
179       }
180     }
181   }
182 
183   private static final long serialVersionUID = 0L;
184 }
