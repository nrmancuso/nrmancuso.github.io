15  package com.google.common.hash;
16  
17  import static com.google.common.base.Preconditions.checkArgument;
18  import static com.google.common.base.Preconditions.checkNotNull;
19  import static com.google.common.base.Preconditions.checkState;
20  
21  import com.google.errorprone.annotations.Immutable;
22  import java.io.Serializable;
23  import java.nio.ByteBuffer;
24  import java.security.MessageDigest;
25  import java.security.NoSuchAlgorithmException;
26  import java.util.Arrays;
27  
28  /**
29   * {@link HashFunction} adapter for {@link MessageDigest} instances.
30   *
31   * @author Kevin Bourrillion
32   * @author Dimitris Andreou
33   */
34  @Immutable
35  final class MessageDigestHashFunction extends AbstractHashFunction implements Serializable {
36  
37    @SuppressWarnings("Immutable") // cloned before each use
38    private final MessageDigest prototype;
39  
40    private final int bytes;
41    private final boolean supportsClone;
42    private final String toString;
43  
44    MessageDigestHashFunction(String algorithmName, String toString) {
45      this.prototype = getMessageDigest(algorithmName);
46      this.bytes = prototype.getDigestLength();
47      this.toString = checkNotNull(toString);
48      this.supportsClone = supportsClone(prototype);
49    }
50  
51    MessageDigestHashFunction(String algorithmName, int bytes, String toString) {
52      this.toString = checkNotNull(toString);
53      this.prototype = getMessageDigest(algorithmName);
54      int maxLength = prototype.getDigestLength();
55      checkArgument(
56          bytes >= 4 && bytes <= maxLength, "bytes (%s) must be >= 4 and < %s", bytes, maxLength);
57      this.bytes = bytes;
58      this.supportsClone = supportsClone(prototype);
59    }
60  
61    private static boolean supportsClone(MessageDigest digest) {
62      try {
63        digest.clone();
64        return true;
65      } catch (CloneNotSupportedException e) {
66        return false;
67      }
68    }
69  
70    @Override
71    public int bits() {
72      return bytes * Byte.SIZE;
73    }
74  
75    @Override
76    public String toString() {
77      return toString;
78    }
79  
80    private static MessageDigest getMessageDigest(String algorithmName) {
81      try {
82        return MessageDigest.getInstance(algorithmName);
83      } catch (NoSuchAlgorithmException e) {
84        throw new AssertionError(e);
85      }
86    }
87  
88    @Override
89    public Hasher newHasher() {
90      if (supportsClone) {
91        try {
92          return new MessageDigestHasher((MessageDigest) prototype.clone(), bytes);
93        } catch (CloneNotSupportedException e) {
94          // falls through
95        }
96      }
97      return new MessageDigestHasher(getMessageDigest(prototype.getAlgorithm()), bytes);
98    }
99  
100   private static final class SerializedForm implements Serializable {
101     private final String algorithmName;
102     private final int bytes;
103     private final String toString;
104 
105     private SerializedForm(String algorithmName, int bytes, String toString) {
106       this.algorithmName = algorithmName;
107       this.bytes = bytes;
108       this.toString = toString;
109     }
110 
111     private Object readResolve() {
112       return new MessageDigestHashFunction(algorithmName, bytes, toString);
113     }
114 
115     private static final long serialVersionUID = 0;
116   }
117 
118   Object writeReplace() {
119     return new SerializedForm(prototype.getAlgorithm(), bytes, toString);
120   }
121 
122   /** Hasher that updates a message digest. */
123   private static final class MessageDigestHasher extends AbstractByteHasher {
124     private final MessageDigest digest;
125     private final int bytes;
126     private boolean done;
127 
128     private MessageDigestHasher(MessageDigest digest, int bytes) {
129       this.digest = digest;
130       this.bytes = bytes;
131     }
132 
133     @Override
134     protected void update(byte b) {
135       checkNotDone();
136       digest.update(b);
137     }
138 
139     @Override
140     protected void update(byte[] b, int off, int len) {
141       checkNotDone();
142       digest.update(b, off, len);
143     }
144 
145     @Override
146     protected void update(ByteBuffer bytes) {
147       checkNotDone();
148       digest.update(bytes);
149     }
150 
151     private void checkNotDone() {
152       checkState(!done, "Cannot re-use a Hasher after calling hash() on it");
153     }
154 
155     @Override
156     public HashCode hash() {
157       checkNotDone();
158       done = true;
159       return (bytes == digest.getDigestLength())
160           ? HashCode.fromBytesNoCopy(digest.digest())
161           : HashCode.fromBytesNoCopy(Arrays.copyOf(digest.digest(), bytes));
162     }
163   }
164 }
