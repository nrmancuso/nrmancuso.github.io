19  package org.apache.struts2.dispatcher;
20  
21  import javax.servlet.http.HttpServletRequest;
22  import java.io.Serializable;
23  import java.util.AbstractMap;
24  import java.util.Enumeration;
25  import java.util.HashSet;
26  import java.util.Set;
27  
28  /**
29   * A simple implementation of the {@link java.util.Map} interface to handle a collection of request attributes.
30   */
31  public class RequestMap extends AbstractMap implements Serializable {
32  
33      private static final long serialVersionUID = -7675640869293787926L;
34  
35      private Set<Object> entries;
36      private HttpServletRequest request;
37  
38      /**
39       * Saves the request to use as the backing for getting and setting values
40       *
41       * @param request the http servlet request.
42       */
43      public RequestMap(final HttpServletRequest request) {
44          this.request = request;
45      }
46  
47  
48      /**
49       * Removes all attributes from the request as well as clears entries in this map.
50       */
51      public void clear() {
52          entries = null;
53          Enumeration keys = request.getAttributeNames();
54  
55          while (keys.hasMoreElements()) {
56              String key = (String) keys.nextElement();
57              request.removeAttribute(key);
58          }
59      }
60  
61      /**
62       * Returns a Set of attributes from the http request.
63       *
64       * @return a Set of attributes from the http request.
65       */
66      public Set entrySet() {
67          if (entries == null) {
68              entries = new HashSet<>();
69  
70              Enumeration enumeration = request.getAttributeNames();
71  
72              while (enumeration.hasMoreElements()) {
73                  final String key = enumeration.nextElement().toString();
74                  final Object value = request.getAttribute(key);
75                  entries.add(new Entry() {
76                      public boolean equals(Object obj) {
77                          if (!(obj instanceof Entry)) {
78                              return false;
79                          }
80                          Entry entry = (Entry) obj;
81  
82                          return ((key == null) ? (entry.getKey() == null) : key.equals(entry.getKey())) && ((value == null) ? (entry.getValue() == null) : value.equals(entry.getValue()));
83                      }
84  
85                      public int hashCode() {
86                          return ((key == null) ? 0 : key.hashCode()) ^ ((value == null) ? 0 : value.hashCode());
87                      }
88  
89                      public Object getKey() {
90                          return key;
91                      }
92  
93                      public Object getValue() {
94                          return value;
95                      }
96  
97                      public Object setValue(Object obj) {
98                          request.setAttribute(key, obj);
99  
100                         return value;
101                     }
102                 });
103             }
104         }
105 
106         return entries;
107     }
108 
109     /**
110      * Returns the request attribute associated with the given key or <tt>null</tt> if it doesn't exist.
111      *
112      * @param key the name of the request attribute.
113      * @return the request attribute or <tt>null</tt> if it doesn't exist.
114      */
115     public Object get(Object key) {
116         return request.getAttribute(key.toString());
117     }
118 
119     /**
120      * Saves an attribute in the request.
121      *
122      * @param key   the name of the request attribute.
123      * @param value the value to set.
124      * @return the object that was just set.
125      */
126     public Object put(Object key, Object value) {
127         Object oldValue = get(key);
128         entries = null;
129         request.setAttribute(key.toString(), value);
130         return oldValue;
131     }
132 
133     /**
134      * Removes the specified request attribute.
135      *
136      * @param key the name of the attribute to remove.
137      * @return the value that was removed or <tt>null</tt> if the value was not found (and hence, not removed).
138      */
139     public Object remove(Object key) {
140         entries = null;
141 
142         Object value = get(key);
143         request.removeAttribute(key.toString());
144 
145         return value;
146     }
147 }
