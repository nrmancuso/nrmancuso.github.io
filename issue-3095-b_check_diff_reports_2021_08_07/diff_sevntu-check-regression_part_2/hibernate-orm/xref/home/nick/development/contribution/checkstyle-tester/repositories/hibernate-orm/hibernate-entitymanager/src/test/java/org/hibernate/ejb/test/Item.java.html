2   package org.hibernate.ejb.test;
3   import java.io.Serializable;
4   import java.util.HashSet;
5   import java.util.Set;
6   import javax.persistence.Column;
7   import javax.persistence.Entity;
8   import javax.persistence.EntityResult;
9   import javax.persistence.FieldResult;
10  import javax.persistence.Id;
11  import javax.persistence.NamedNativeQueries;
12  import javax.persistence.NamedNativeQuery;
13  import javax.persistence.NamedQuery;
14  import javax.persistence.OneToMany;
15  import javax.persistence.SqlResultSetMapping;
16  
17  /**
18   * @author Gavin King
19   */
20  @Entity(name = "Item")
21  @SqlResultSetMapping(name = "getItem", entities =
22  	@EntityResult(entityClass = org.hibernate.ejb.test.Item.class, fields = {
23  		@FieldResult(name = "name", column = "itemname"),
24  		@FieldResult(name = "descr", column = "itemdescription")
25  	})
26  )
27  @NamedNativeQueries({
28  	@NamedNativeQuery(
29  		name = "nativeItem1",
30  		query = "select name as itemname, descr as itemdescription from Item",
31  		resultSetMapping = "getItem"
32  	),
33  	@NamedNativeQuery(
34  		name = "nativeItem2",
35  		query = "select * from Item",
36  		resultClass = Item.class
37  	)
38  })
39  @NamedQuery(name = "query-construct", query = "select new Item(i.name,i.descr) from Item i")
40  //@Cache(region="Item", usage=NONSTRICT_READ_WRITE)
41  public class Item implements Serializable {
42  
43  	private String name;
44  	private String descr;
45  	private Set<Distributor> distributors = new HashSet<Distributor>();
46  
47  	public Item() {
48  	}
49  
50  	public Item(String name, String desc) {
51  		this.name = name;
52  		this.descr = desc;
53  	}
54  
55  	@Column(length = 200)
56  	public String getDescr() {
57  		return descr;
58  	}
59  
60  	public void setDescr(String desc) {
61  		this.descr = desc;
62  	}
63  
64  	@Id
65  	@Column(length = 30)
66  	public String getName() {
67  		return name;
68  	}
69  
70  	public void setName(String name) {
71  		this.name = name;
72  	}
73  
74  	@OneToMany
75  	public Set<Distributor> getDistributors() {
76  		return distributors;
77  	}
78  
79  	public void setDistributors(Set<Distributor> distributors) {
80  		this.distributors = distributors;
81  	}
82  
83  	public void addDistributor(Distributor d) {
84  		if ( distributors == null ) distributors = new HashSet();
85  		distributors.add( d );
86  	}
87  }
