22  package org.hibernate.ejb.criteria.components;
23  import java.io.Serializable;
24  import java.util.HashSet;
25  import java.util.Set;
26  import javax.persistence.CascadeType;
27  import javax.persistence.Column;
28  import javax.persistence.Embeddable;
29  import javax.persistence.OneToMany;
30  
31  /**
32   * The name component
33   *
34   * @author alan.oleary
35   */
36  @Embeddable
37  public class Name implements Serializable {
38      private static final long serialVersionUID = 8381969086665589013L;
39  
40      private String firstName;
41      private String lastName;
42  	private Set<Alias> aliases = new HashSet<Alias>(  );
43  
44  	public Name() {
45  	}
46  
47  	public Name(String firstName, String lastName) {
48  		this.firstName = firstName;
49  		this.lastName = lastName;
50  	}
51  
52  	@Column(name = "FIRST_NAME", nullable = false)
53  	public String getFirstName() {
54  		return firstName;
55  	}
56  
57  	public void setFirstName(String firstName) {
58  		this.firstName = firstName;
59  	}
60  
61  	@Column(name = "LAST_NAME", nullable = false)
62  	public String getLastName() {
63  		return lastName;
64  	}
65  
66  	public void setLastName(String lastName) {
67  		this.lastName = lastName;
68  	}
69  
70  	@OneToMany(cascade = CascadeType.ALL)
71  	public Set<Alias> getAliases() {
72  		return aliases;
73  	}
74  
75  	public void setAliases(Set<Alias> aliases) {
76  		this.aliases = aliases;
77  	}
78  }
