3   import org.hibernate.search.annotations.*;
4   import org.hibernate.search.bridge.builtin.IntegerBridge;

6   
7   import java.io.Serializable;
8   import java.util.Date;

10  /**
11   * @author Navin Surtani
12   */
13  @ProvidedId(bridge = @FieldBridge(impl = StringBridge.class))
14  @Indexed(index = "person")
15  @FullTextFilterDefs({
16        @FullTextFilterDef(name = "personFilter", impl = PersonBlurbFilterFactory.class, cache = FilterCacheModeType.INSTANCE_AND_DOCIDSETRESULTS),
17        @FullTextFilterDef(name = "personAgeFilter", impl = PersonAgeFilterFactory.class, cache = FilterCacheModeType.INSTANCE_AND_DOCIDSETRESULTS)
18  })
19  public class Person implements Serializable {
20     @Field(store = Store.YES)
21     private String name;
22     @Field(store = Store.YES)
23     private String blurb;
24     @Field(store = Store.YES, analyze = Analyze.NO, bridge = @FieldBridge(impl = IntegerBridge.class))
25     private int age;
26     @Field(store = Store.YES, analyze = Analyze.NO)
27     @DateBridge(resolution = Resolution.DAY)
28     private Date dateOfGraduation;
29  
30     private String nonSearchableField;
31     private static final long serialVersionUID = 8251606115293644545L;
32  
33     public Person() {
34     }
35  
36     public Person(String name, String blurb, int age) {
37        this.name = name;
38        this.blurb = blurb;
39        this.age = age;
40     }
41  
42     public Person(String name, String blurb, int age, Date dateOfGraduation) {
43        this.name = name;
44        this.blurb = blurb;
45        this.age = age;
46        this.dateOfGraduation = dateOfGraduation;
47     }
48  
49     public String getName() {
50        return name;
51     }
52  
53     public void setName(String name) {
54        this.name = name;
55     }
56  
57     public String getBlurb() {
58        return blurb;
59     }
60  
61     public void setBlurb(String blurb) {
62        this.blurb = blurb;
63     }
64  
65     public int getAge() {
66        return age;
67     }
68  
69     public void setAge(int age) {
70        this.age = age;
71     }
72  
73     public String getNonSearchableField() {
74        return nonSearchableField;
75     }
76  
77     public void setNonSearchableField(String nonSearchableField) {
78        this.nonSearchableField = nonSearchableField;
79     }
80  
81     public Date getDateOfGraduation() {
82        return dateOfGraduation;
83     }
84  
85     public void setDateOfGraduation(Date dateOfGraduation) {
86        this.dateOfGraduation = dateOfGraduation;
87     }
88  
89     @Override
90     public boolean equals(Object o) {
91        if (this == o) return true;
92        if (o == null || getClass() != o.getClass()) return false;
93  
94        Person person = (Person) o;
95  
96        if (age != person.age) return false;
97        if (blurb != null ? !blurb.equals(person.blurb) : person.blurb != null) return false;
98        if (name != null ? !name.equals(person.name) : person.name != null) return false;
99        if (dateOfGraduation != null ? !dateOfGraduation.equals(person.dateOfGraduation) : person.dateOfGraduation != null) return false;
100 
101       return true;
102    }
103 
104    @Override
105    public int hashCode() {
106       int result;
107       result = (name != null ? name.hashCode() : 0);
108       result = 31 * result + (blurb != null ? blurb.hashCode() : 0);
109       result = 31 * result + (dateOfGraduation != null ? dateOfGraduation.hashCode() : 0);
110       result = 31 * result + age;
111       return result;
112    }
113 
114    @Override
115    public String toString() {
116       return "Person{" +
117             "name='" + name + '\'' +
118             ", blurb='" + blurb + '\'' +
119             ", age=" + age +
120             ", dateOfGraduation=" + dateOfGraduation +
121             '}';
122    }
123 }
