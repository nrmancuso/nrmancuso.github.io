24  package org.hibernate.type.descriptor.java;
25  
26  import java.io.Reader;
27  import java.io.StringReader;
28  import java.sql.Clob;
29  import java.util.Arrays;
30  
31  import org.hibernate.engine.jdbc.CharacterStream;
32  import org.hibernate.engine.jdbc.internal.CharacterStreamImpl;
33  import org.hibernate.type.descriptor.WrapperOptions;
34  
35  /**
36   * TODO : javadoc

38   * @author Steve Ebersole
39   */
40  public class CharacterArrayTypeDescriptor extends AbstractTypeDescriptor<Character[]> {
41  	public static final CharacterArrayTypeDescriptor INSTANCE = new CharacterArrayTypeDescriptor();
42  
43  	@SuppressWarnings({ "unchecked" })
44  	public CharacterArrayTypeDescriptor() {
45  		super( Character[].class, ArrayMutabilityPlan.INSTANCE );
46  	}
47  
48  	public String toString(Character[] value) {
49  		return new String( unwrapChars( value ) );
50  	}
51  
52  	public Character[] fromString(String string) {
53  		return wrapChars( string.toCharArray() );
54  	}
55  
56  	@Override
57  	public boolean areEqual(Character[] one, Character[] another) {
58  		return one == another
59  				|| ( one != null && another != null && Arrays.equals( one, another ) );
60  	}
61  
62  	@Override
63  	public int extractHashCode(Character[] chars) {
64  		int hashCode = 1;
65  		for ( Character aChar : chars ) {
66  			hashCode = 31 * hashCode + aChar;
67  		}
68  		return hashCode;
69  	}
70  
71  	@SuppressWarnings({ "unchecked" })
72  	public <X> X unwrap(Character[] value, Class<X> type, WrapperOptions options) {
73  		if ( value == null ) {
74  			return null;
75  		}
76  		if ( Character[].class.isAssignableFrom( type ) ) {
77  			return (X) value;
78  		}
79  		if ( String.class.isAssignableFrom( type ) ) {
80  			return (X) new String( unwrapChars( value ) );
81  		}
82  		if ( Clob.class.isAssignableFrom( type ) ) {
83  			return (X) options.getLobCreator().createClob( new String( unwrapChars( value ) ) );
84  		}
85  		if ( Reader.class.isAssignableFrom( type ) ) {
86  			return (X) new StringReader( new String( unwrapChars( value ) ) );
87  		}
88  		if ( CharacterStream.class.isAssignableFrom( type ) ) {
89  			return (X) new CharacterStreamImpl( new String( unwrapChars( value ) ) );
90  		}
91  		throw unknownUnwrap( type );
92  	}
93  
94  	public <X> Character[] wrap(X value, WrapperOptions options) {
95  		if ( value == null ) {
96  			return null;
97  		}
98  		if ( Character[].class.isInstance( value ) ) {
99  			return (Character[]) value;
100 		}
101 		if ( String.class.isInstance( value ) ) {
102 			return wrapChars( ( (String) value ).toCharArray() );
103 		}
104 		if ( Clob.class.isInstance( value ) ) {
105 			return wrapChars( DataHelper.extractString( ( (Clob) value ) ).toCharArray() );
106 		}
107 		if ( Reader.class.isInstance( value ) ) {
108 			return wrapChars( DataHelper.extractString( (Reader) value ).toCharArray() );
109 		}
110 		throw unknownWrap( value.getClass() );
111 	}
112 
113 	@SuppressWarnings({ "UnnecessaryBoxing" })
114 	private Character[] wrapChars(char[] chars) {
115 		if ( chars == null ) {
116 			return null;
117 		}
118 		final Character[] result = new Character[chars.length];
119 		for ( int i = 0; i < chars.length; i++ ) {
120 			result[i] = Character.valueOf( chars[i] );
121 		}
122 		return result;
123 	}
124 
125 	@SuppressWarnings({ "UnnecessaryUnboxing" })
126 	private char[] unwrapChars(Character[] chars) {
127 		if ( chars == null ) {
128 			return null;
129 		}
130 		final char[] result = new char[chars.length];
131 		for ( int i = 0; i < chars.length; i++ ) {
132 			result[i] = chars[i].charValue();
133 		}
134 		return result;
135 	}
136 }
