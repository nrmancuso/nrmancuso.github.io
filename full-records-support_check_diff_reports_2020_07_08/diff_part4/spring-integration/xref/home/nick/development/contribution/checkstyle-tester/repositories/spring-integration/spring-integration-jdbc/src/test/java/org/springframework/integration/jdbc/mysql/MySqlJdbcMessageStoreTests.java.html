98  	@After
99  	public void afterTest() {
100 		final JdbcTemplate jdbcTemplate = new JdbcTemplate(dataSource);
101 		new TransactionTemplate(this.transactionManager).execute(status -> {
102 			final int deletedGroupToMessageRows = jdbcTemplate.update("delete from INT_GROUP_TO_MESSAGE");
103 			final int deletedMessages = jdbcTemplate.update("delete from INT_MESSAGE");
104 			final int deletedMessageGroups = jdbcTemplate.update("delete from INT_MESSAGE_GROUP");
105 
106 			LOG.info(String.format("Cleaning Database - Deleted Messages: %s, " +
107 							"Deleted GroupToMessage Rows: %s, Deleted Message Groups: %s",
108 					deletedMessages, deletedGroupToMessageRows, deletedMessageGroups));
109 			return null;
110 		});
111 	}

402 		group = messageStore.getMessageGroup(groupX);

404 
405 		Message<?> message2 = messageStore.pollMessageFromGroup(groupX);

407 		assertThat(message2.getPayload()).isEqualTo("bar");
408 
409 		group = messageStore.getMessageGroup(groupX);
410 		assertThat(group.size()).isEqualTo(1);

414 	@Transactional
415 	@Rollback(false)
416 	@Repeat(20)
417 	public void testSameMessageToMultipleGroups() {
418 		final String group1Id = "group1";
419 		final String group2Id = "group2";
420 
421 		final Message<String> message = MessageBuilder.withPayload("foo").build();
422 
423 		final MessageBuilder<String> builder1 = MessageBuilder.fromMessage(message);
424 		final MessageBuilder<String> builder2 = MessageBuilder.fromMessage(message);
425 
426 		builder1.setSequenceNumber(1);
427 		builder2.setSequenceNumber(2);
428 
429 		final Message<?> message1 = builder1.build();
430 		final Message<?> message2 = builder2.build();
431 
432 		messageStore.addMessageToGroup(group1Id, message1);
433 		messageStore.addMessageToGroup(group2Id, message2);
434 
435 		final Message<?> messageFromGroup1 = messageStore.pollMessageFromGroup(group1Id);
436 		final Message<?> messageFromGroup2 = messageStore.pollMessageFromGroup(group2Id);
437 
438 		assertThat(messageFromGroup1).isNotNull();
439 		assertThat(messageFromGroup2).isNotNull();
440 
441 		LOG.info("messageFromGroup1: " + messageFromGroup1.getHeaders().getId()
442 				+ "; Sequence #: " + new IntegrationMessageHeaderAccessor(messageFromGroup1).getSequenceNumber());
443 		LOG.info("messageFromGroup2: " + messageFromGroup2.getHeaders().getId()
444 				+ "; Sequence #: " + new IntegrationMessageHeaderAccessor(messageFromGroup2).getSequenceNumber());
445 
446 		assertThat(messageFromGroup1.getHeaders().get(IntegrationMessageHeaderAccessor.SEQUENCE_NUMBER))
447 				.isEqualTo(1);
448 		assertThat(messageFromGroup2.getHeaders().get(IntegrationMessageHeaderAccessor.SEQUENCE_NUMBER))
449 				.isEqualTo(2);
450 
451 	}
