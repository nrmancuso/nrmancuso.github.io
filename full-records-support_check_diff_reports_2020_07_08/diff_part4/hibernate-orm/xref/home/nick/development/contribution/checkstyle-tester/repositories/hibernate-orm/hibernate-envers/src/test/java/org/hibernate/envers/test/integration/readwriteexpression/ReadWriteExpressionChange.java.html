1   package org.hibernate.envers.test.integration.readwriteexpression;
2   
3   import java.math.BigDecimal;
4   import java.util.List;
5   import javax.persistence.EntityManager;
6   
7   import org.hibernate.dialect.Oracle8iDialect;
8   import org.junit.Assert;
9   import org.junit.Test;
10  
11  import org.hibernate.ejb.Ejb3Configuration;
12  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
13  import org.hibernate.envers.test.Priority;
14  
15  public class ReadWriteExpressionChange extends BaseEnversJPAFunctionalTestCase {
16  
17      private static final Double HEIGHT_INCHES = 73.0d;
18      private static final Double HEIGHT_CENTIMETERS = HEIGHT_INCHES * 2.54d;
19  
20      private Integer id;
21  
22      @Override
23      public void configure(Ejb3Configuration cfg) {
24          cfg.addAnnotatedClass(Staff.class);
25      }
26  
27      @Test
28      @Priority(10)
29      public void initData() {
30          EntityManager em = getEntityManager();
31          em.getTransaction().begin();
32          Staff staff = new Staff(HEIGHT_INCHES, 1);
33          em.persist(staff);
34          em.getTransaction().commit();
35          id = staff.getId();
36      }
37  
38      @Test
39      public void shouldRespectWriteExpression() {
40          EntityManager em = getEntityManager();
41          List resultList = em.createNativeQuery("select size_in_cm from t_staff_AUD where id ="+id).getResultList();
42          Assert.assertEquals(1, resultList.size());
43          Double sizeInCm = null;
44          if (getDialect() instanceof Oracle8iDialect) {
45              sizeInCm = ((BigDecimal) resultList.get(0)).doubleValue();
46          } else {
47              sizeInCm = (Double) resultList.get(0);
48          }
49          Assert.assertEquals(HEIGHT_CENTIMETERS, sizeInCm.doubleValue(), 0.00000001);
50      }
51  
52      @Test
53      public void shouldRespectReadExpression() {
54          List<Number> revisions = getAuditReader().getRevisions(Staff.class, id);
55          Assert.assertEquals(1, revisions.size());
56          Number number = revisions.get(0);
57          Staff staffRev = getAuditReader().find(Staff.class, id, number);
58          Assert.assertEquals(HEIGHT_INCHES, staffRev.getSizeInInches(), 0.00000001);
59      }
60  
61  }
