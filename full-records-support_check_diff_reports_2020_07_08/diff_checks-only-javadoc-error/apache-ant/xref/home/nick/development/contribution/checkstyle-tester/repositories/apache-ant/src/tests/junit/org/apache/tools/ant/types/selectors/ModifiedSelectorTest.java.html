227     public void doDelayUpdateTest(int kind) {
228         // no check for 1<=kind<=3 - only internal use therefore check it
229         // while development
230 
231         // readable form of parameter kind
232         String[] kinds = {"task", "target", "build"};
233 
234         // setup the "Ant project"
235         MockProject project = new MockProject();
236         File base  = new File("base");
237         File file1 = new File("file1");
238         File file2 = new File("file2");
239 
240         // setup the selector
241         ModifiedSelector sel = new ModifiedSelector();
242         sel.setProject(project);
243         sel.setUpdate(true);
244         sel.setDelayUpdate(true);
245         // sorry - otherwise we will get a ClassCastException because the MockCache
246         // is loaded by two different classloader ...
247         sel.setClassLoader(this.getClass().getClassLoader());
248         sel.addClasspath(testclasses);
249 
250         sel.setAlgorithmClass("org.apache.tools.ant.types.selectors.MockAlgorithm");
251         sel.setCacheClass("org.apache.tools.ant.types.selectors.MockCache");
252         sel.configure();
253 
254         // get the cache, so we can check our things
255         MockCache cache = (MockCache)sel.getCache();
256 
257         // the test
258         assertFalse("Cache must not be saved before 1st selection.", cache.saved);
259         sel.isSelected(base, "file1", file1);
260         assertFalse("Cache must not be saved after 1st selection.", cache.saved);
261         sel.isSelected(base, "file2", file2);
262         assertFalse("Cache must not be saved after 2nd selection.", cache.saved);
263         switch (kind) {
264             case 1 : project.fireTaskFinished();   break;
265             case 2 : project.fireTargetFinished(); break;
266             case 3 : project.fireBuildFinished();  break;
267         }
268         assertTrue("Cache must be saved after " + kinds[kind-1] + "Finished-Event.", cache.saved);
269 
270         // MockCache doesnt create a file - therefore no cleanup needed
271     }

274     /**
275      * Extracts the real used algorithm name from the ModifiedSelector using
276      * its toString() method.
277      * @param classname  the classname from the algorithm to use
278      * @return  the algorithm part from the toString() (without brackets)
279      */
280     private String getAlgoName(String classname) {
281         ModifiedSelector sel = new ModifiedSelector();
282         sel.setProject(selectorRule.getProject());
283         // add the test classes to its classpath
284         sel.addClasspath(testclasses);
285         sel.setAlgorithmClass(classname);
286         // let the selector do its checks
287         sel.validate();
288         // extract the algorithm name (and config) from the selectors output
289         String s1 = sel.toString();
290         int posStart = s1.indexOf("algorithm=") + 10;
291         int posEnd   = s1.indexOf(" comparator=");
292         String algo  = s1.substring(posStart, posEnd);
293         // '<' and '>' are only used if the algorithm has properties
294         if (algo.startsWith("<")) algo = algo.substring(1);
295         if (algo.endsWith(">"))   algo = algo.substring(0, algo.length()-1);
296         // return the clean value
297         return algo;
298     }

408     /**
409      * Test the interface semantic of Caches.
410      * This method does some common test for cache implementations.
411      * A cache must return a stored value and a valid iterator.
412      * After calling the delete() the cache must be empty.
413      *
414      * @param cache   configured test object
415      */
416     protected void doTest(Cache cache) {
417         assertTrue("Cache not proper configured.", cache.isValid());
418 
419         String key1   = "key1";
420         String value1 = "value1";
421         String key2   = "key2";
422         String value2 = "value2";
423 
424         // given cache must be empty
425         Iterator it1 = cache.iterator();
426         assertFalse("Cache is not empty", it1.hasNext());
427 
428         // cache must return a stored value
429         cache.put(key1, value1);
430         cache.put(key2, value2);
431         assertEquals("cache returned wrong value", value1, cache.get(key1));
432         assertEquals("cache returned wrong value", value2, cache.get(key2));
433 
434         // test the iterator
435         Iterator it2 = cache.iterator();
436         Object   returned = it2.next();
437         boolean ok = (key1.equals(returned) || key2.equals(returned));
438         String msg = "Iterator returned unexpected value."
439                    + "  key1.equals(returned)="+key1.equals(returned)
440                    + "  key2.equals(returned)="+key2.equals(returned)
441                    + "  returned="+returned
442                    + "  ok="+ok;
443         assertTrue(msg, ok);
444 
445         // clear the cache
446         cache.delete();
447         Iterator it3 = cache.iterator();
448         assertFalse("Cache is not empty", it3.hasNext());
449     }

674     /**
675      * Complex test scenario using default values (DigestAlgorithm with MD5,
676      * PropertiesfileCache with file=cache.properties, EqualComparator
677      * and update=true). <ol>
678      * <li> try fist time --> should select all </li>
679      * <li> try second time --> should select no files (only directories) </li>
680      * <li> modify timestamp of one file and content of a nother one </li>
681      * <li> try third time --> should select only the file with modified
682      *      content </li>
683      */
684     @Test
685     public void testScenario1() {
686         BFT bft = null;
687         ModifiedSelector s = null;
688         try {
689 
690             String results;
691 
692             // Configure the selector - only defaults are used
693             s = new ModifiedSelector();
694 
695             //
696             // *****  First Run  *****
697             // the first call should get all files, because nothing is in
698             // the cache
699             //
700             performTests(s, "TTTTTTTTTTTT");
701 
702             //
703             // *****  Second Run  *****
704             // the second call should get no files, because no content
705             // has changed
706             //
707             performTests(s, "TFFFFFFFFFFT");
708 
709             //
710             // *****  make some files dirty  *****
711             //
712 
713             // these files are made dirty --> 3+4 with different content
714             String f2name = "tar/bz2/asf-logo-huge.tar.bz2";
715             String f3name = "asf-logo.gif.md5";
716             String f4name = "copy.filterset.filtered";
717 
718             // AccessObject to the test-Ant-environment
719             bft = new BFT();
720             // give some values (via property file) to that environment
721             bft.writeProperties("f2name="+f2name);
722             bft.writeProperties("f3name="+f3name);
723             bft.writeProperties("f4name="+f4name);
724             // call the target for making the files dirty
725             bft.doTarget("modifiedselectortest-makeDirty");
726 
727             //
728             // *****  Third Run  *****
729             // third call should get only those files, which CONTENT changed
730             // (no timestamp changes required!)
731             results = selectorRule.selectionString(s);
732 
733             //

1004     /**
1005      * Does the selection test for a given selector and prints the
1006      * filenames of the differing files (selected but shouldn't,
1007      * not selected but should).
1008      * @param selector  The selector to test
1009      * @param expected  The expected result
1010      */
1011     private void performTests(FileSelector selector, String expected) {
1012         String result = selectorRule.selectionString(selector);
1013         String diff = diff(expected, result);
1014         String resolved = resolve(diff);
1015         assertEquals("Differing files: " + resolved, result, expected);
1016     }
1017     /**
1018      *  Checks which files are selected and shouldn't be or which
1019      *  are not selected but should.
1020      *  @param expected    String containing 'F's and 'T's
1021      *  @param result      String containing 'F's and 'T's
1022      *  @return Difference as String containing '-' (equal) and
1023      *          'X' (difference).
1024      */
1025     private String diff(String expected, String result) {
1026         int length1 = expected.length();
1027         int length2 = result.length();
1028         int min = (length1 > length2) ? length2 : length1;
1029         StringBuffer sb = new StringBuffer();
1030         for (int i=0; i<min; i++) {
1031             sb.append(
1032                     (expected.charAt(i) == result.charAt(i))
1033                             ? "-"
1034                             : "X"
1035             );
1036         }
1037         return sb.toString();
1038     }
1039 
1040     /**
1041      * Resolves a diff-String (@see diff()) against the (inherited) filenames-
1042      * and files arrays.
1043      * @param filelist    Diff-String
1044      * @return String containing the filenames for all differing files,
1045      *         separated with semicolons ';'
1046      */
1047     private String resolve(String filelist) {
1048         StringBuffer sb = new StringBuffer();
1049         int min = (selectorRule.getFilenames().length > filelist.length())
1050                 ? filelist.length()
1051                 : selectorRule.getFilenames().length;
1052         for (int i=0; i<min; i++) {
1053             if ('X'==filelist.charAt(i)) {
1054                 sb.append(selectorRule.getFilenames()[i]);
1055                 sb.append(";");
1056             }
1057         }
1058         return sb.toString();
1059     }
