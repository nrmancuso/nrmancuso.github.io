319    /**
320     * @param isRemote true if the command is deserialized and is executed remote.
321     */
322    @Override
323    public void initializeReplicableCommand(ReplicableCommand c, boolean isRemote) {
324       if (c == null) return;
325       switch (c.getCommandId()) {
326          case PutKeyValueCommand.COMMAND_ID:
327             ((PutKeyValueCommand) c).init(notifier, configuration);
328             break;
329          case ReplaceCommand.COMMAND_ID:
330             ((ReplaceCommand) c).init(notifier, configuration);
331             break;
332          case PutMapCommand.COMMAND_ID:
333             ((PutMapCommand) c).init(notifier);
334             break;
335          case RemoveCommand.COMMAND_ID:
336             ((RemoveCommand) c).init(notifier, configuration);
337             break;
338          case MultipleRpcCommand.COMMAND_ID:
339             MultipleRpcCommand rc = (MultipleRpcCommand) c;
340             rc.init(interceptorChain, icf);
341             if (rc.getCommands() != null)
342                for (ReplicableCommand nested : rc.getCommands()) {
343                   initializeReplicableCommand(nested, false);
344                }
345             break;
346          case SingleRpcCommand.COMMAND_ID:
347             SingleRpcCommand src = (SingleRpcCommand) c;
348             src.init(interceptorChain, icf);
349             if (src.getCommand() != null)
350                initializeReplicableCommand(src.getCommand(), false);
351 
352             break;
353          case InvalidateCommand.COMMAND_ID:
354             InvalidateCommand ic = (InvalidateCommand) c;
355             ic.init(notifier, configuration);
356             break;
357          case InvalidateL1Command.COMMAND_ID:
358             InvalidateL1Command ilc = (InvalidateL1Command) c;
359             ilc.init(configuration, distributionManager, notifier, dataContainer);
360             break;
361          case PrepareCommand.COMMAND_ID:
362          case VersionedPrepareCommand.COMMAND_ID:
363          case TotalOrderNonVersionedPrepareCommand.COMMAND_ID:
364          case TotalOrderVersionedPrepareCommand.COMMAND_ID:
365             PrepareCommand pc = (PrepareCommand) c;
366             pc.init(interceptorChain, icf, txTable);
367             pc.initialize(notifier, recoveryManager);
368             if (pc.getModifications() != null)
369                for (ReplicableCommand nested : pc.getModifications())  {
370                   initializeReplicableCommand(nested, false);
371                }
372             pc.markTransactionAsRemote(isRemote);
373             if (configuration.deadlockDetection().enabled() && isRemote) {
374                DldGlobalTransaction transaction = (DldGlobalTransaction) pc.getGlobalTransaction();
375                transaction.setLocksHeldAtOrigin(pc.getAffectedKeys());
376             }
377             break;
378          case CommitCommand.COMMAND_ID:
379          case VersionedCommitCommand.COMMAND_ID:
380          case TotalOrderCommitCommand.COMMAND_ID:
381          case TotalOrderVersionedCommitCommand.COMMAND_ID:
382             CommitCommand commitCommand = (CommitCommand) c;
383             commitCommand.init(interceptorChain, icf, txTable);
384             commitCommand.markTransactionAsRemote(isRemote);
385             break;
386          case RollbackCommand.COMMAND_ID:
387          case TotalOrderRollbackCommand.COMMAND_ID:
388             RollbackCommand rollbackCommand = (RollbackCommand) c;
389             rollbackCommand.init(interceptorChain, icf, txTable);
390             rollbackCommand.markTransactionAsRemote(isRemote);
391             break;
392          case ClearCommand.COMMAND_ID:
393             ClearCommand cc = (ClearCommand) c;
394             cc.init(notifier, dataContainer);
395             break;
396          case ClusteredGetCommand.COMMAND_ID:
397             ClusteredGetCommand clusteredGetCommand = (ClusteredGetCommand) c;
398             clusteredGetCommand.initialize(icf, this, entryFactory,
399                   interceptorChain, distributionManager, txTable,
400                   configuration.dataContainer().keyEquivalence());
401             break;
402          case LockControlCommand.COMMAND_ID:
403             LockControlCommand lcc = (LockControlCommand) c;
404             lcc.init(interceptorChain, icf, txTable);
405             lcc.markTransactionAsRemote(isRemote);
406             if (configuration.deadlockDetection().enabled() && isRemote) {
407                DldGlobalTransaction gtx = (DldGlobalTransaction) lcc.getGlobalTransaction();
408                RemoteTransaction transaction = txTable.getRemoteTransaction(gtx);
409                if (transaction != null) {
410                   if (!configuration.clustering().cacheMode().isDistributed()) {
411                      Set<Object> keys = txTable.getLockedKeysForRemoteTransaction(gtx);
412                      GlobalTransaction gtx2 = transaction.getGlobalTransaction();
413                      ((DldGlobalTransaction) gtx2).setLocksHeldAtOrigin(keys);
414                      gtx.setLocksHeldAtOrigin(keys);
415                   } else {
416                      GlobalTransaction gtx2 = transaction.getGlobalTransaction();
417                      ((DldGlobalTransaction) gtx2).setLocksHeldAtOrigin(gtx.getLocksHeldAtOrigin());
418                   }
419                }
420             }
421             break;
422          case StateRequestCommand.COMMAND_ID:
423             ((StateRequestCommand) c).init(stateProvider);
424             break;
425          case StateResponseCommand.COMMAND_ID:
426             ((StateResponseCommand) c).init(stateConsumer);
427             break;
428          case GetInDoubtTransactionsCommand.COMMAND_ID:
429             GetInDoubtTransactionsCommand gptx = (GetInDoubtTransactionsCommand) c;
430             gptx.init(recoveryManager);
431             break;
432          case TxCompletionNotificationCommand.COMMAND_ID:
433             TxCompletionNotificationCommand ftx = (TxCompletionNotificationCommand) c;
434             ftx.init(txTable, lockManager, recoveryManager, stateTransferManager);
435             break;
436          case MapCombineCommand.COMMAND_ID:
437             MapCombineCommand mrc = (MapCombineCommand)c;
438             mrc.init(mapReduceManager);
439             break;
440          case ReduceCommand.COMMAND_ID:
441             ReduceCommand reduceCommand = (ReduceCommand)c;
442             reduceCommand.init(mapReduceManager);
443             break;
444          case DistributedExecuteCommand.COMMAND_ID:
445             DistributedExecuteCommand dec = (DistributedExecuteCommand)c;
446             dec.init(cache);
447             break;
448          case GetInDoubtTxInfoCommand.COMMAND_ID:
449             GetInDoubtTxInfoCommand gidTxInfoCommand = (GetInDoubtTxInfoCommand)c;
450             gidTxInfoCommand.init(recoveryManager);
451             break;
452          case CompleteTransactionCommand.COMMAND_ID:
453             CompleteTransactionCommand ccc = (CompleteTransactionCommand)c;
454             ccc.init(recoveryManager);
455             break;
456          case ApplyDeltaCommand.COMMAND_ID:
457             break;
458          case CreateCacheCommand.COMMAND_ID:
459             CreateCacheCommand createCacheCommand = (CreateCacheCommand)c;
460             createCacheCommand.init(cache.getCacheManager(), stateTransferManager);
461             break;
462          case XSiteAdminCommand.COMMAND_ID:
463             XSiteAdminCommand xSiteAdminCommand = (XSiteAdminCommand)c;
464             xSiteAdminCommand.init(backupSender);
465             break;
466          case CancelCommand.COMMAND_ID:
467             CancelCommand cancelCommand = (CancelCommand)c;
468             cancelCommand.init(cancellationService);
469             break;
470          case XSiteStateTransferControlCommand.COMMAND_ID:
471             XSiteStateTransferControlCommand xSiteStateTransferControlCommand = (XSiteStateTransferControlCommand) c;
472             xSiteStateTransferControlCommand.initialize(xSiteStateProvider, xSiteStateConsumer, xSiteStateTransferManager);
473             break;
474          case XSiteStatePushCommand.COMMAND_ID:
475             XSiteStatePushCommand xSiteStatePushCommand = (XSiteStatePushCommand) c;
476             xSiteStatePushCommand.initialize(xSiteStateConsumer);
477             break;
478          case EntryRequestCommand.COMMAND_ID:
479             EntryRequestCommand entryRequestCommand = (EntryRequestCommand) c;
480             entryRequestCommand.init(entryRetriever);
481             break;
482          case EntryResponseCommand.COMMAND_ID:
483             EntryResponseCommand entryResponseCommand = (EntryResponseCommand) c;
484             entryResponseCommand.init(entryRetriever);
485             break;
486          case GetKeysInGroupCommand.COMMAND_ID:
487             GetKeysInGroupCommand getKeysInGroupCommand = (GetKeysInGroupCommand) c;
488             getKeysInGroupCommand.setGroupManager(groupManager);
489             break;
490          case ClusteredGetAllCommand.COMMAND_ID:
491             ClusteredGetAllCommand clusteredGetAllCommand = (ClusteredGetAllCommand) c;
492             clusteredGetAllCommand.init(icf, this, entryFactory, interceptorChain, txTable,
493                   configuration.dataContainer().keyEquivalence());
494             break;
495          default:
496             ModuleCommandInitializer mci = moduleCommandInitializers.get(c.getCommandId());
497             if (mci != null) {
498                mci.initializeReplicableCommand(c, isRemote);
499             } else {
500                if (trace) log.tracef("Nothing to initialize for command: %s", c);
501             }
502       }
503    }
