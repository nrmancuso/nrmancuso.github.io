74  /**
75   * Base class for FindBugs tests.
76   * <p>
77   * Subclasses must:
78   * <ul>
79   * <li>implement getTestScenario() to return the required TestScenario.</li>
80   * <li>call setUpTestProject(TestScenario) and tearDownTestProject() during
81   * setUp and tearDown respectively. The argument for the setup must be the same
82   * test scenario as returned by getTestScenario(). The fixture may be shared by
83   * all tests in the same class, if the tests don't modify the project or are
84   * independent from the modifications.</li>
85   * </ul>
86   *
87   * @author Tomás Pollak
88   */
89  public abstract class AbstractPluginTest {
90

97      /**
98       * Hook for subclasses to add extra classpath entries to the test project
99       * during the setup of the test.
100      */
101     protected static void addExtraClassPathEntries(TestScenario scenario) throws CoreException {
102         // Add JUnit if the test scenario requires it
103         if (scenario.usesJUnit()) {
104             addJUnitToProjectClasspath();
105         }
106     }

108     protected static void addJUnitToProjectClasspath() throws JavaModelException {
109         IClasspathEntry cpe = BuildPathSupport.getJUnit4ClasspathEntry();
110         JavaProjectHelper.addToClasspath(getJavaProject(), cpe);
111     }

135     /**
136      * Create a new Java project with a source folder and copy the test files of
137      * the plugin to the source folder. Compile the project.
138      */
139     protected static void setUpTestProject(TestScenario scenario) throws Exception {
140         // Create the test project
141         createJavaProject(TEST_PROJECT, "bin");
142         addRTJar(getJavaProject());
143 
144         // Create default 'src'
145         addSourceContainer(getJavaProject(), SRC);
146 
147         String[] testFilesPaths = scenario.getTestFilesPaths();
148         for (int i = 1; i < testFilesPaths.length; i++) { // Create extra 'srcx'
149             addSourceContainer(getJavaProject(), SRC + (i + 1));
150         }
151         addExtraClassPathEntries(scenario);
152 
153         // Copy test workspace
154         Bundle testBundle = FindbugsTestPlugin.getDefault().getBundle();
155 
156         // Import default 'src'
157         importResources(getProject().getFolder(SRC), testBundle, testFilesPaths[0]);
158 
159         for (int i = 1; i < testFilesPaths.length; i++) { // Import extra 'srcx'
160             importResources(getProject().getFolder(SRC + (i + 1)), testBundle, testFilesPaths[i]);
161         }
162 
163         importResources(getProject(), testBundle, "/testresources");
164 
165         // Compile project
166         getProject().refreshLocal(IResource.DEPTH_INFINITE, null);
167         getProject().build(IncrementalProjectBuilder.FULL_BUILD, null);
168         processUiEvents();
169         waitForJobs();
170         performDummySearch();
171         waitForJobs();
172         processUiEvents(100);
173 //        processUiEvents(10000);
174     }
175 
176     /**
177      * Delete the Java project used for this test.
178      *
179      * @throws CoreException
180      */
181     protected static void tearDownTestProject() throws CoreException {
182         // Delete the test project
183         waitForJobs();
184         delete(getJavaProject().getProject());
185         waitForJobs();
186         processUiEvents();
187     }
188 
189     protected static void waitForJobs() {
190         while (!Job.getJobManager().isIdle()) {
191             processUiEvents();
192         }
193     }
194 
195     protected static void processUiEvents() {
196         while (Display.getDefault().readAndDispatch()) {
197             ;
198         }
199     }
200 
201     protected static void processUiEvents(long delayInMilliseconds) {
202         long start = System.currentTimeMillis();
203         long sleepTime = delayInMilliseconds > 10? 10 : delayInMilliseconds;
204         while (true) {
205             try {
206                 Thread.sleep(sleepTime);
207             } catch (InterruptedException e) {
208                 e.printStackTrace();
209             }
210             processUiEvents();
211             if(System.currentTimeMillis() - start > delayInMilliseconds){
212                 break;
213             }
214         }
215     }

221     @Before
222     public void setUp() throws Exception {
223         // Start with a clean FindBugs state
224         clearBugsState();
225         IWorkbenchPage activePage = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage();
226         activePage.closeAllEditors(false);
227         IViewPart view = activePage.findView("org.eclipse.ui.internal.introview");
228         if (view != null) {
229             activePage.hideView(view);
230         }
231         getPreferenceStore().setValue(FindBugsConstants.ASK_ABOUT_PERSPECTIVE_SWITCH, false);
232         waitForJobs();
233         processUiEvents();
234     }

238         // Clean the FindBugs state
239         clearBugsState();
240         waitForJobs();
241         processUiEvents();
242     }

244     /**
245      * Assert the total number of bugs in the given resource.
246      *
247      * @param expected
248      *            The expected number of bugs.
249      * @param project
250      *            The IProject that contains the bugs.
251      * @throws CoreException
252      */
253     protected void assertBugsCount(int expected, IProject project) throws CoreException {
254         SortedBugCollection bugs = FindbugsPlugin.getBugCollection(project, null);
255         assertEquals(expected, bugs.getCollection().size());
256     }
257 
258     protected void assertExpectedBugs() throws CoreException {
259         assertBugsCount(getExpectedBugsCount(), getProject());
260         for (String bugPattern : getTestScenario().getVisibleBugs()) {
261             int frequency = getTestScenario().getVisibleBugFrequency(bugPattern);
262             assertReportedBugs(bugPattern, frequency, getProject());
263         }
264     }
265 
266     protected void assertExpectedMarkers(IMarker[] markers) throws CoreException {
267         assertEquals(getVisibleBugsCount(), markers.length);
268         for (int i = 0; i < markers.length; i++) {
269             assertTrue(markers[i].isSubtypeOf(FindBugsMarker.NAME));
270         }
271         for (String bugPattern : getTestScenario().getVisibleBugs()) {
272             int frequency = getTestScenario().getVisibleBugFrequency(bugPattern);
273             assertMarkers(bugPattern, frequency, markers);
274         }
275     }
276 
277     protected void assertExpectedMarkers(Set<IMarker> markers) throws CoreException {
278         assertExpectedMarkers(markers.toArray(new IMarker[0]));
279     }
280 
281     /**
282      * Asserts that the number of present markers of the given type match the
283      * given expected count.
284      *
285      * @param expectedBugType
286      *            The expected bug type.
287      * @param expectedBugTypeCount
288      *            The expected bug type count.
289      * @param markers
290      *            The array of markers to assert on.
291      * @throws CoreException
292      */
293     protected void assertMarkers(String expectedBugType, int expectedBugTypeCount, IMarker[] markers) throws CoreException {
294         int seenBugTypeCount = 0;
295         for (int i = 0; i < markers.length; i++) {
296             IMarker marker = markers[i];
297             if (expectedBugType.equals(marker.getAttribute(FindBugsMarker.BUG_TYPE))) {
298                 seenBugTypeCount++;
299             }
300         }
301         assertEquals("Expected " + expectedBugTypeCount + " of markers " + expectedBugType + " but seen " + seenBugTypeCount,
302                 expectedBugTypeCount, seenBugTypeCount);
303     }
304 
305     protected void assertNoBugs() throws CoreException {
306         assertBugsCount(0, getProject());
307     }
308 
309     protected void assertNoMarkers(IMarker[] markers) {
310         assertEquals(0, markers.length);
311     }

313     /**
314      * Asserts that the number of detected bugs of the given type match the
315      * given expected count.
316      *
317      * @param expectedBugType
318      *            The expected bug type.
319      * @param expectedBugCount
320      *            The expected bug type count.
321      * @param project
322      *            The IProject that contains the bugs.
323      * @throws CoreException
324      */
325     protected void assertReportedBugs(String expectedBugType, int expectedBugCount, IProject project) throws CoreException {
326         int seenBugCount = 0;
327         SortedBugCollection bugs = FindbugsPlugin.getBugCollection(project, null);
328         for (BugInstance bug : bugs) {
329             if (expectedBugType.equals(bug.getType())) {
330                 seenBugCount++;
331             }
332         }
333         assertEquals("Expected " + expectedBugCount + " of bugs " + expectedBugType + " but seen " + seenBugCount,
334                 expectedBugCount, seenBugCount);
335     }

386     /**
387      * Suspend the calling thread until all the background jobs belonging to the
388      * given family are done.
389      *
390      * @see org.eclipse.core.runtime.jobs.Job#belongsTo(Object)
391      *
392      * @param family
393      *            The family object that groups the jobs.
394      */
395     protected void joinJobFamily(Object family) {
396         boolean finished = false;
397         while (!finished) {
398             try {
399                 getJobManager().join(family, null);
400                 finished = true;
401             } catch (InterruptedException e) {
402                 // continue waiting
403             }
404         }
405     }

424     /**
425      * Runs the FindBugs worker on the given Java element.
426      */
427     protected void work(FindBugsWorker worker, IJavaElement element) throws CoreException {
428         worker.work(Collections.singletonList(new WorkItem(element)));
429         processUiEvents(ResourceChangeListener.SHORT_DELAY);
430         joinJobFamily(FindbugsPlugin.class); // wait for RefreshJob
431         waitForJobs();
432         processUiEvents();
433     }
