161 	public Bitmap cropCircle(Bitmap bitmap) {
162 	    Bitmap output = Bitmap.createBitmap(bitmap.getWidth(),
163 	            bitmap.getHeight(), Config.ARGB_8888);
164 	    Canvas canvas = new Canvas(output);
165 
166 	    final int color = 0xff424242;
167 	    final Paint paint = new Paint();
168 	    final Rect rect = new Rect(0, 0, bitmap.getWidth(), bitmap.getHeight());
169 
170 	    paint.setAntiAlias(true);
171 	    canvas.drawARGB(0, 0, 0, 0);
172 	    paint.setColor(color);
173 	    canvas.drawCircle(bitmap.getWidth() / 2, bitmap.getHeight() / 2,
174 	            bitmap.getWidth()/2, paint);
175 	    paint.setXfermode(new PorterDuffXfermode(Mode.SRC_IN));
176 	    canvas.drawBitmap(bitmap, rect, rect, paint);
177 	    return output;
178 	}
