280     assertSubtypeTokenBeforeSupertypeToken(types.classes());

286   private static void assertSubtypeTokenBeforeSupertypeToken(
287       Iterable<? extends TypeToken<?>> types) {
288     int i = 0;
289     for (TypeToken<?> left : types) {
290       int j = 0;
291       for (TypeToken<?> right : types) {
292         if (left.isSupertypeOf(right)) {
293           assertTrue(left + " should be after " + right, i >= j);
294         }
295         j++;
296       }
297       i++;
298     }
299   }

301   private static void assertSubtypeBeforeSupertype(Iterable<? extends Class<?>> types) {
302     int i = 0;
303     for (Class<?> left : types) {
304       int j = 0;
305       for (Class<?> right : types) {
306         if (left.isAssignableFrom(right)) {
307           assertTrue(left + " should be after " + right, i >= j);
308         }
309         j++;
310       }
311       i++;
312     }
313   }
