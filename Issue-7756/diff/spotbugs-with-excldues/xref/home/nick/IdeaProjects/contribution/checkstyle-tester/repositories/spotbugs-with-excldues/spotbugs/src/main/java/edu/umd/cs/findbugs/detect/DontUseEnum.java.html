20  package edu.umd.cs.findbugs.detect;
21  
22  import org.apache.bcel.Const;
23  import org.apache.bcel.classfile.Field;
24  import org.apache.bcel.classfile.FieldOrMethod;
25  import org.apache.bcel.classfile.LocalVariable;
26  import org.apache.bcel.classfile.Method;
27  
28  import edu.umd.cs.findbugs.BugInstance;
29  import edu.umd.cs.findbugs.BugReporter;
30  import edu.umd.cs.findbugs.LocalVariableAnnotation;
31  import edu.umd.cs.findbugs.SourceLineAnnotation;
32  import edu.umd.cs.findbugs.bcel.PreorderDetector;
33  
34  public class DontUseEnum extends PreorderDetector {
35  
36      BugReporter bugReporter;
37  
38      public DontUseEnum(BugReporter bugReporter) {
39          this.bugReporter = bugReporter;
40      }
41  
42      @Override
43      public void visit(Method obj) {
44          if (isReservedName(obj.getName())) {
45              BugInstance bug = new BugInstance(this, "NM_FUTURE_KEYWORD_USED_AS_MEMBER_IDENTIFIER", isVisible(obj) ? HIGH_PRIORITY
46                      : NORMAL_PRIORITY).addClassAndMethod(this);
47              bugReporter.reportBug(bug);
48          }
49      }
50  
51      private boolean isVisible(FieldOrMethod obj) {
52          return (obj.getAccessFlags() & Const.ACC_PUBLIC) != 0 || (obj.getAccessFlags() & Const.ACC_PROTECTED) != 0;
53      }
54  
55      private boolean isReservedName(String name) {
56          return "enum".equals(name) || "assert".equals(name);
57      }
58  
59      @Override
60      public void visit(Field obj) {
61          if (isReservedName(obj.getName())) {
62              BugInstance bug = new BugInstance(this, "NM_FUTURE_KEYWORD_USED_AS_MEMBER_IDENTIFIER", isVisible(obj) ? HIGH_PRIORITY
63                      : NORMAL_PRIORITY).addClass(this).addField(this);
64              bugReporter.reportBug(bug);
65          }
66      }
67  
68      @Override
69      public void visit(LocalVariable obj) {
70          if (isReservedName(obj.getName())) {
71              LocalVariableAnnotation var = new LocalVariableAnnotation(obj.getName(), obj.getIndex(), obj.getStartPC());
72              SourceLineAnnotation source = SourceLineAnnotation.fromVisitedInstruction(getClassContext(), this, obj.getStartPC());
73              BugInstance bug = new BugInstance(this, "NM_FUTURE_KEYWORD_USED_AS_IDENTIFIER", NORMAL_PRIORITY)
74              .addClassAndMethod(this).add(var).add(source);
75              bugReporter.reportBug(bug);
76          }
77      }
78  
79  }
