70  /**
71   * <p>A forward dataflow analysis to determine the types of all values in the Java
72   * stack frame at all points in a Java method. The values include local
73   * variables and values on the Java operand stack.
74   * </p>
75   * <p>
76   * As a side effect, the analysis computes the exception set throwable on each
77   * exception edge in the CFG. This information can be used to prune infeasible
78   * exception edges, and mark exception edges which propagate only implicit
79   * exceptions.</p>
80   *
81   * @author David Hovemeyer
82   * @see Dataflow
83   * @see edu.umd.cs.findbugs.ba.DataflowAnalysis
84   * @see TypeFrame
85   */
86  public class TypeAnalysis extends FrameDataflowAnalysis<Type, TypeFrame> implements EdgeTypes {

95      /**
96       * Repository of information about thrown exceptions computed for a basic
97       * block and its outgoing exception edges. It contains a result TypeFrame,
98       * which is used to detect when the exception information needs to be
99       * recomputed for the block.
100      */
101     private class CachedExceptionSet {
102         private final TypeFrame result;
103 
104         private final ExceptionSet exceptionSet;
105 
106         private final Map<Edge, ExceptionSet> edgeExceptionMap;
107 
108         public CachedExceptionSet(TypeFrame result, ExceptionSet exceptionSet) {
109             this.result = result;
110             this.exceptionSet = exceptionSet;
111             this.edgeExceptionMap = new HashMap<>();
112         }
113 
114         public boolean isUpToDate(TypeFrame result) {
115             return this.result.equals(result);
116         }
117 
118         public ExceptionSet getExceptionSet() {
119             return exceptionSet;
120         }
121 
122         public void setEdgeExceptionSet(Edge edge, ExceptionSet exceptionSet) {
123             edgeExceptionMap.put(edge, exceptionSet);
124         }
125 
126         public ExceptionSet getEdgeExceptionSet(Edge edge) {
127             ExceptionSet edgeExceptionSet = edgeExceptionMap.get(edge);
128             if (edgeExceptionSet == null) {
129                 edgeExceptionSet = exceptionSetFactory.createExceptionSet();
130                 edgeExceptionMap.put(edge, edgeExceptionSet);
131             }
132             return edgeExceptionSet;
133         }
134     }

184     /**
185      * Constructor.
186      *
187      * @param method
188      *            TODO
189      * @param methodGen
190      *            the MethodGen whose CFG we'll be analyzing
191      * @param cfg
192      *            the control flow graph
193      * @param dfs
194      *            DepthFirstSearch of the method
195      * @param typeMerger
196      *            object to merge types
197      * @param visitor
198      *            a TypeFrameModelingVisitor to use to model the effect of
199      *            instructions
200      * @param lookupFailureCallback
201      *            lookup failure callback
202      * @param exceptionSetFactory
203      *            factory for creating ExceptionSet objects
204      */
205     public TypeAnalysis(Method method, MethodGen methodGen, CFG cfg, DepthFirstSearch dfs, TypeMerger typeMerger,
206             TypeFrameModelingVisitor visitor, RepositoryLookupFailureCallback lookupFailureCallback,
207             ExceptionSetFactory exceptionSetFactory) {
208         super(dfs);
209         this.method = method;
210         Code code = method.getCode();
211         if (code == null) {
212             throw new IllegalArgumentException(method.getName() + " has no code");
213         }
214         for (Attribute a : code.getAttributes()) {
215             if (a instanceof LocalVariableTypeTable) {
216                 visitor.setLocalTypeTable((LocalVariableTypeTable) a);
217             }
218         }
219         this.methodGen = methodGen;
220         this.cfg = cfg;
221         this.typeMerger = typeMerger;
222         this.visitor = visitor;
223         this.thrownExceptionSetMap = new HashMap<>();
224         this.lookupFailureCallback = lookupFailureCallback;
225         this.exceptionSetFactory = exceptionSetFactory;
226         this.instanceOfCheckMap = new HashMap<>();
227         if (DEBUG) {
228             System.out.println("\n\nAnalyzing " + methodGen);
229         }
230     }

323     @Override
324     public void initEntryFact(TypeFrame result) {
325         // Make the frame valid
326         result.setValid();
327 
328         int slot = 0;
329 
330         // Clear the stack slots in the frame
331         result.clearStack();
332 
333         // Add local for "this" pointer, if present
334         if (!methodGen.isStatic()) {
335             result.setValue(slot++, ObjectTypeFactory.getInstance(methodGen.getClassName()));
336         }
337 
338         // [Added: Support for Generics]
339         // Get a parser that reads the generic signature of the method and
340         // can be used to get the correct GenericObjectType if an argument
341         // has a class type
342         Iterator<String> iter = GenericSignatureParser.getGenericSignatureIterator(method);
343 
344         // Add locals for parameters.
345         // Note that long and double parameters need to be handled
346         // specially because they occupy two locals.
347         Type[] argumentTypes = methodGen.getArgumentTypes();
348         for (Type argType : argumentTypes) {
349             // Add special "extra" type for long or double params.
350             // These occupy the slot before the "plain" type.
351             if (argType.getType() == Const.T_LONG) {
352                 result.setValue(slot++, TypeFrame.getLongExtraType());
353             } else if (argType.getType() == Const.T_DOUBLE) {
354                 result.setValue(slot++, TypeFrame.getDoubleExtraType());
355             }
356 
357             // [Added: Support for Generics]
358             String s = (iter == null || !iter.hasNext()) ? null : iter.next();
359             if (s != null && (argType instanceof ObjectType || argType instanceof ArrayType)
360                     && !(argType instanceof ExceptionObjectType)) {
361                 // replace with a generic version of the type
362                 try {
363                     Type t = GenericUtilities.getType(s);
364                     if (t != null) {
365                         argType = t;
366                     }
367                 } catch (RuntimeException e) {
368                 } // degrade gracefully
369             }
370 
371             // Add the plain parameter type.
372             result.setValue(slot++, argType);
373         }
374 
375         // Set remaining locals to BOTTOM; this will cause any
376         // uses of them to be flagged
377         while (slot < methodGen.getMaxLocals()) {
378             result.setValue(slot++, TypeFrame.getBottomType());
379         }
380     }

409     @Override
410     public void transfer(BasicBlock basicBlock, @CheckForNull InstructionHandle end, TypeFrame start, TypeFrame result)
411             throws DataflowAnalysisException {
412         visitor.startBasicBlock();
413 
414         super.transfer(basicBlock, end, start, result);
415 
416         // Compute thrown exception types
417         computeThrownExceptionTypes(basicBlock, end, result);
418         if (DEBUG) {
419             System.out.println("After " + basicBlock.getFirstInstruction() + " -> " + basicBlock.getLastInstruction());
420             System.out.println("    frame: " + result);
421         }
422 
423         // If this block ends with an instanceof check,
424         // update the cached information about it.
425         instanceOfCheckMap.remove(basicBlock);
426         if (visitor.isInstanceOfFollowedByBranch()) {
427             InstanceOfCheck check = new InstanceOfCheck(visitor.getInstanceOfValueNumber(), visitor.getInstanceOfType());
428             instanceOfCheckMap.put(basicBlock, check);
429         }
430     }

432     private void computeThrownExceptionTypes(BasicBlock basicBlock, @CheckForNull InstructionHandle end, TypeFrame result)
433             throws DataflowAnalysisException {
434 
435         // Do nothing if we're not computing propagated exceptions
436         if (!(FORCE_ACCURATE_EXCEPTIONS || AnalysisContext.currentAnalysisContext().getBoolProperty(
437                 AnalysisFeatures.ACCURATE_EXCEPTIONS))) {
438             return;
439         }
440 
441         // Also, nothing to do if the block is not an exception thrower
442         if (!basicBlock.isExceptionThrower()) {
443             return;
444         }
445 
446         // If cached results are up to date, don't recompute.
447         CachedExceptionSet cachedExceptionSet = getCachedExceptionSet(basicBlock);
448         if (cachedExceptionSet.isUpToDate(result)) {
449             return;
450         }
451 
452         // Figure out what exceptions can be thrown out
453         // of the basic block, and mark each exception edge
454         // with the set of exceptions which can be propagated
455         // along the edge.
456 
457         int exceptionEdgeCount = 0;
458         Edge lastExceptionEdge = null;
459 
460         for (Iterator<Edge> i = cfg.outgoingEdgeIterator(basicBlock); i.hasNext();) {
461             Edge e = i.next();
462             if (e.isExceptionEdge()) {
463                 exceptionEdgeCount++;
464                 lastExceptionEdge = e;
465             }
466         }
467 
468         if (exceptionEdgeCount == 0) {
469             // System.out.println("Shouldn't all blocks have an exception edge");
470             return;
471         }
472         // Compute exceptions that can be thrown by the
473         // basic block.
474         cachedExceptionSet = computeBlockExceptionSet(basicBlock, result);
475 
476         if (exceptionEdgeCount == 1) {
477             cachedExceptionSet.setEdgeExceptionSet(lastExceptionEdge, cachedExceptionSet.getExceptionSet());
478             return;
479         }
480 
481         // For each outgoing exception edge, compute exceptions
482         // that can be thrown. This assumes that the exception
483         // edges are enumerated in decreasing order of priority.
484         // In the process, this will remove exceptions from
485         // the thrown exception set.
486         ExceptionSet thrownExceptionSet = cachedExceptionSet.getExceptionSet();
487         if (!thrownExceptionSet.isEmpty()) {
488             thrownExceptionSet = thrownExceptionSet.duplicate();
489         }
490         for (Iterator<Edge> i = cfg.outgoingEdgeIterator(basicBlock); i.hasNext();) {
491             Edge edge = i.next();
492             if (edge.isExceptionEdge()) {
493                 cachedExceptionSet.setEdgeExceptionSet(edge, computeEdgeExceptionSet(edge, thrownExceptionSet));
494             }
495         }
496     }

498     @Override
499     public void meetInto(TypeFrame fact, Edge edge, TypeFrame result) throws DataflowAnalysisException {
500         BasicBlock basicBlock = edge.getTarget();
501 
502         if (fact.isValid()) {
503             TypeFrame tmpFact = null;
504 
505             // Handling an exception?
506             if (basicBlock.isExceptionHandler()) {
507                 tmpFact = modifyFrame(fact, null);
508 
509                 // Special case: when merging predecessor facts for entry to
510                 // an exception handler, we clear the stack and push a
511                 // single entry for the exception object. That way, the locals
512                 // can still be merged.
513                 CodeExceptionGen exceptionGen = basicBlock.getExceptionGen();
514                 tmpFact.clearStack();
515 
516                 // Determine the type of exception(s) caught.
517                 Type catchType = null;
518 
519                 if (FORCE_ACCURATE_EXCEPTIONS
520                         || AnalysisContext.currentAnalysisContext().getBoolProperty(AnalysisFeatures.ACCURATE_EXCEPTIONS)) {
521                     try {
522                         // Ideally, the exceptions that can be propagated
523                         // on this edge has already been computed.
524                         CachedExceptionSet cachedExceptionSet = getCachedExceptionSet(edge.getSource());
525                         ExceptionSet edgeExceptionSet = cachedExceptionSet.getEdgeExceptionSet(edge);
526                         if (!edgeExceptionSet.isEmpty()) {
527                             // System.out.println("Using computed edge exception set!");
528                             catchType = ExceptionObjectType.fromExceptionSet(edgeExceptionSet);
529                         }
530                     } catch (ClassNotFoundException e) {
531                         lookupFailureCallback.reportMissingClass(e);
532                     }
533                 }
534 
535                 if (catchType == null) {
536                     // No information about propagated exceptions, so
537                     // pick a type conservatively using the handler catch type.
538                     catchType = exceptionGen.getCatchType();
539                     if (catchType == null)
540                     {
541                         catchType = Type.THROWABLE; // handle catches anything
542                         // throwable
543                     }
544                 }
545 
546                 tmpFact.pushValue(catchType);
547             }
548 
549             // See if we can make some types more precise due to
550             // a successful instanceof check in the source block.
551             if (valueNumberDataflow != null) {
552                 tmpFact = handleInstanceOfBranch(fact, tmpFact, edge);
553             }
554 
555             if (tmpFact != null) {
556                 fact = tmpFact;
557             }
558         }
559 
560         mergeInto(fact, result);
561     }

563     private TypeFrame handleInstanceOfBranch(TypeFrame fact, TypeFrame tmpFact, Edge edge) {
564 
565         InstanceOfCheck check = instanceOfCheckMap.get(edge.getSource());
566         if (check == null) {
567             // System.out.println("no instanceof check for block " +
568             // edge.getSource().getId());
569             return tmpFact;
570         }
571 
572         if (check.getValueNumber() == null) {
573             // System.out.println("instanceof check for block " +
574             // edge.getSource().getId() + " has no value number");
575             return tmpFact;
576         }
577 
578         ValueNumber instanceOfValueNumber = check.getValueNumber();
579         ValueNumberFrame vnaFrame = valueNumberDataflow.getStartFact(edge.getTarget());
580         if (!vnaFrame.isValid()) {
581             return tmpFact;
582         }
583 
584         Type instanceOfType = check.getType();
585         if (!(instanceOfType instanceof ReferenceType || instanceOfType instanceof NullType)) {
586             return tmpFact;
587         }
588 
589         short branchOpcode = edge.getSource().getLastInstruction().getInstruction().getOpcode();
590 
591         int edgeType = edge.getType();
592         int numSlots = Math.min(fact.getNumSlots(), vnaFrame.getNumSlots());
593 
594         if ((edgeType == EdgeTypes.IFCMP_EDGE && (branchOpcode == Const.IFNE || branchOpcode == Const.IFGT || branchOpcode == Const.IFNULL))
595 
596                 || (edgeType == EdgeTypes.FALL_THROUGH_EDGE && (branchOpcode == Const.IFEQ || branchOpcode == Const.IFLE || branchOpcode == Const.IFNONNULL))) {
597             // System.out.println("Successful check on edge " + edge);
598 
599             // Successful instanceof check.
600 
601             for (int i = 0; i < numSlots; ++i) {
602                 if (!vnaFrame.getValue(i).equals(instanceOfValueNumber)) {
603                     continue;
604                 }
605 
606                 Type checkedType = fact.getValue(i);
607                 if (!(checkedType instanceof ReferenceType)) {
608                     continue;
609                 }
610 
611                 // Only refine the type if the cast is feasible: i.e., a
612                 // downcast.
613                 // Otherwise, just set it to TOP.
614                 try {
615                     boolean guaranteed = Hierarchy.isSubtype((ReferenceType) checkedType, (ReferenceType) instanceOfType);
616                     if (guaranteed) {
617                         continue;
618                     }
619 
620                     boolean feasibleCheck = instanceOfType.equals(NullType.instance())
621                             || Hierarchy.isSubtype((ReferenceType) instanceOfType, (ReferenceType) checkedType);
622 
623                     if (!feasibleCheck && instanceOfType instanceof ObjectType && checkedType instanceof ObjectType) {
624                         double v = Analyze.deepInstanceOf(((ObjectType) instanceOfType).getClassName(),
625                                 ((ObjectType) checkedType).getClassName());
626                         if (v > 0.0) {
627                             feasibleCheck = true;
628                         }
629                     }
630                     tmpFact = modifyFrame(fact, tmpFact);
631                     if (feasibleCheck) {
632                         tmpFact.setValue(i, instanceOfType);
633                     } else {
634                         tmpFact.setTop();
635                         return tmpFact;
636                     }
637                 } catch (ClassNotFoundException e) {
638                     lookupFailureCallback.reportMissingClass(e);
639                     return tmpFact;
640                 }
641             }
642         } else if (!instanceOfType.equals(NullType.instance())) {
643 
644             for (int i = 0; i < numSlots; ++i) {
645                 if (!vnaFrame.getValue(i).equals(instanceOfValueNumber)) {
646                     continue;
647                 }
648 
649                 Type checkedType = fact.getValue(i);
650                 if (!(checkedType instanceof ReferenceType)) {
651                     continue;
652                 }
653                 try {
654                     boolean guaranteed = Hierarchy.isSubtype((ReferenceType) checkedType, (ReferenceType) instanceOfType);
655                     if (!guaranteed) {
656                         continue;
657                     }
658                     tmpFact = modifyFrame(fact, tmpFact);
659                     tmpFact.setTop();
660                     return tmpFact;
661                 } catch (ClassNotFoundException e) {
662                     lookupFailureCallback.reportMissingClass(e);
663                     return tmpFact;
664                 }
665             }
666         }
667 
668         return tmpFact;
669     }

671     @Override
672     protected void mergeValues(TypeFrame otherFrame, TypeFrame resultFrame, int slot) throws DataflowAnalysisException {
673 
674         Type type2 = resultFrame.getValue(slot);
675         Type type1 = otherFrame.getValue(slot);
676         Type value = typeMerger.mergeTypes(type2, type1);
677         resultFrame.setValue(slot, value);
678 
679         // Result type is exact IFF types are identical and both are exact
680 
681         boolean typesAreIdentical = type1.equals(type2);
682 
683         boolean bothExact = resultFrame.isExact(slot) && otherFrame.isExact(slot);
684 
685         resultFrame.setExact(slot, typesAreIdentical && bothExact);
686     }

688     /**
689      * Get the cached set of exceptions that can be thrown from given basic
690      * block. If this information hasn't been computed yet, then an empty
691      * exception set is returned.
692      *
693      * @param basicBlock
694      *            the block to get the cached exception set for
695      * @return the CachedExceptionSet for the block
696      */
697     private CachedExceptionSet getCachedExceptionSet(BasicBlock basicBlock) {
698         CachedExceptionSet cachedExceptionSet = thrownExceptionSetMap.get(basicBlock);
699         if (cachedExceptionSet == null) {
700             // When creating the cached exception type set for the first time:
701             // - the block result is set to TOP, so it won't match
702             // any block result that has actually been computed
703             // using the analysis transfer function
704             // - the exception set is created as empty (which makes it
705             // return TOP as its common superclass)
706 
707             TypeFrame top = createFact();
708             makeFactTop(top);
709             cachedExceptionSet = new CachedExceptionSet(top, exceptionSetFactory.createExceptionSet());
710 
711             thrownExceptionSetMap.put(basicBlock, cachedExceptionSet);
712         }
713 
714         return cachedExceptionSet;
715     }

717     /**
718      * Compute the set of exceptions that can be thrown from the given basic
719      * block. This should only be called if the existing cached exception set is
720      * out of date.
721      *
722      * @param basicBlock
723      *            the basic block
724      * @param result
725      *            the result fact for the block; this is used to determine
726      *            whether or not the cached exception set is up to date
727      * @return the cached exception set for the block
728      */
729     private CachedExceptionSet computeBlockExceptionSet(BasicBlock basicBlock, TypeFrame result) throws DataflowAnalysisException {
730 
731         ExceptionSet exceptionSet = computeThrownExceptionTypes(basicBlock);
732 
733 
734         TypeFrame copyOfResult = createFact();
735         copy(result, copyOfResult);
736 
737         CachedExceptionSet cachedExceptionSet = new CachedExceptionSet(copyOfResult, exceptionSet);
738         thrownExceptionSetMap.put(basicBlock, cachedExceptionSet);
739 
740         return cachedExceptionSet;
741     }

743     /**
744      * Based on the set of exceptions that can be thrown from the source basic
745      * block, compute the set of exceptions that can propagate along given
746      * exception edge. This method should be called for each outgoing exception
747      * edge in sequence, so the caught exceptions can be removed from the thrown
748      * exception set as needed.
749      *
750      * @param edge
751      *            the exception edge
752      * @param thrownExceptionSet
753      *            current set of exceptions that can be thrown, taking earlier
754      *            (higher priority) exception edges into account
755      * @return the set of exceptions that can propagate along this edge
756      */
757     private ExceptionSet computeEdgeExceptionSet(Edge edge, ExceptionSet thrownExceptionSet) {
758 
759         if (thrownExceptionSet.isEmpty()) {
760             return thrownExceptionSet;
761         }
762         ExceptionSet result = exceptionSetFactory.createExceptionSet();
763 
764         if (edge.getType() == UNHANDLED_EXCEPTION_EDGE) {
765             // The unhandled exception edge always comes
766             // after all of the handled exception edges.
767             result.addAll(thrownExceptionSet);
768             thrownExceptionSet.clear();
769             return result;
770         }
771 
772         BasicBlock handlerBlock = edge.getTarget();
773         CodeExceptionGen handler = handlerBlock.getExceptionGen();
774         ObjectType catchType = handler.getCatchType();
775 
776         if (Hierarchy.isUniversalExceptionHandler(catchType)) {
777             result.addAll(thrownExceptionSet);
778             thrownExceptionSet.clear();
779         } else {
780             // Go through the set of thrown exceptions.
781             // Any that will DEFINITELY be caught be this handler, remove.
782             // Any that MIGHT be caught, but won't definitely be caught,
783             // remain.
784 
785             for (ExceptionSet.ThrownExceptionIterator i = thrownExceptionSet.iterator(); i.hasNext();) {
786                 // ThrownException thrownException = i.next();
787                 ObjectType thrownType = i.next();
788                 boolean explicit = i.isExplicit();
789 
790                 if (DEBUG) {
791                     System.out.println("\texception type " + thrownType + ", catch type " + catchType);
792                 }
793 
794                 try {
795                     if (Hierarchy.isSubtype(thrownType, catchType)) {
796                         // Exception can be thrown along this edge
797                         result.add(thrownType, explicit);
798 
799                         // And it will definitely be caught
800                         i.remove();
801 
802                         if (DEBUG) {
803                             System.out.println("\tException is subtype of catch type: " + "will definitely catch");
804                         }
805                     } else if (Hierarchy.isSubtype(catchType, thrownType)) {
806                         // Exception possibly thrown along this edge
807                         result.add(thrownType, explicit);
808 
809                         if (DEBUG) {
810                             System.out.println("\tException is supertype of catch type: " + "might catch");
811                         }
812                     }
813                 } catch (ClassNotFoundException e) {
814                     // As a special case, if a class hierarchy lookup
815                     // fails, then we will conservatively assume that the
816                     // exception in question CAN, but WON'T NECESSARILY
817                     // be caught by the handler.
818                     AnalysisContext.reportMissingClass(e);
819                     result.add(thrownType, explicit);
820                 }
821             }
822         }
823 
824         return result;
825     }

827     /**
828      * Compute the set of exception types that can be thrown by given basic
829      * block.
830      *
831      * @param basicBlock
832      *            the basic block
833      * @return the set of exceptions that can be thrown by the block
834      */
835     private ExceptionSet computeThrownExceptionTypes(BasicBlock basicBlock) throws
836     DataflowAnalysisException {
837 
838         ExceptionSet exceptionTypeSet = exceptionSetFactory.createExceptionSet();
839         InstructionHandle pei = basicBlock.getExceptionThrower();
840         Instruction ins = pei.getInstruction();
841 
842         // Get the exceptions that BCEL knows about.
843         // Note that all of these are unchecked.
844         ExceptionThrower exceptionThrower = (ExceptionThrower) ins;
845         Class<?>[] exceptionList = exceptionThrower.getExceptions();
846         for (Class<?> aExceptionList : exceptionList) {
847             exceptionTypeSet.addImplicit(ObjectTypeFactory.getInstance(aExceptionList.getName()));
848         }
849 
850         // Assume that an Error may be thrown by any instruction.
851         exceptionTypeSet.addImplicit(Hierarchy.ERROR_TYPE);
852 
853         if (ins instanceof ATHROW) {
854             // For ATHROW instructions, we generate *two* blocks
855             // for which the ATHROW is an exception thrower.
856             //
857             // - The first, empty basic block, does the null check
858             // - The second block, which actually contains the ATHROW,
859             // throws the object on the top of the operand stack
860             //
861             // We make a special case of the block containing the ATHROW,
862             // by removing all of the implicit exceptions,
863             // and using type information to figure out what is thrown.
864 
865             if (basicBlock.containsInstruction(pei)) {
866                 // This is the actual ATHROW, not the null check
867                 // and implicit exceptions.
868                 exceptionTypeSet.clear();
869 
870                 // The frame containing the thrown value is the start fact
871                 // for the block, because ATHROW is guaranteed to be
872                 // the only instruction in the block.
873                 TypeFrame frame = getStartFact(basicBlock);
874 
875                 // Check whether or not the frame is valid.
876                 // Sun's javac sometimes emits unreachable code.
877                 // For example, it will emit code that follows a JSR
878                 // subroutine call that never returns.
879                 // If the frame is invalid, then we can just make
880                 // a conservative assumption that anything could be
881                 // thrown at this ATHROW.
882                 if (!frame.isValid()) {
883                     exceptionTypeSet.addExplicit(Type.THROWABLE);
884                 } else if (frame.getStackDepth() == 0) {
885                     throw new IllegalStateException("empty stack " + " thrown by " + pei + " in "
886                             + SignatureConverter.convertMethodSignature(methodGen));
887                 } else {
888 
889                     Type throwType = frame.getTopValue();
890                     if (throwType instanceof ObjectType) {
891                         exceptionTypeSet.addExplicit((ObjectType) throwType);
892                     } else if (throwType instanceof ExceptionObjectType) {
893                         exceptionTypeSet.addAll(((ExceptionObjectType) throwType).getExceptionSet());
894                     } else {
895                         // Not sure what is being thrown here.
896                         // Be conservative.
897                         if (DEBUG) {
898                             System.out.println("Non object type " + throwType + " thrown by " + pei + " in "
899                                     + SignatureConverter.convertMethodSignature(methodGen));
900                         }
901                         exceptionTypeSet.addExplicit(Type.THROWABLE);
902                     }
903                 }
904             }
905         }
906 
907         // If it's an InvokeInstruction, add declared exceptions and
908         // RuntimeException
909         if (ins instanceof InvokeInstruction) {
910             ConstantPoolGen cpg = methodGen.getConstantPool();
911 
912             InvokeInstruction inv = (InvokeInstruction) ins;
913             ObjectType[] declaredExceptionList = Hierarchy2.findDeclaredExceptions(inv, cpg);
914             if (declaredExceptionList == null) {
915                 // Couldn't find declared exceptions,
916                 // so conservatively assume it could thrown any checked
917                 // exception.
918                 if (DEBUG) {
919                     System.out.println("Couldn't find declared exceptions for "
920                             + SignatureConverter.convertMethodSignature(inv, cpg));
921                 }
922                 exceptionTypeSet.addExplicit(Hierarchy.EXCEPTION_TYPE);
923             } else {
924                 for (ObjectType aDeclaredExceptionList : declaredExceptionList) {
925                     exceptionTypeSet.addExplicit(aDeclaredExceptionList);
926                 }
927             }
928 
929             exceptionTypeSet.addImplicit(Hierarchy.RUNTIME_EXCEPTION_TYPE);
930         }
931 
932         if (DEBUG) {
933             System.out.println(pei + " can throw " + exceptionTypeSet);
934         }
935 
936         return exceptionTypeSet;
937     }
