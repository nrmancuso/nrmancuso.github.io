79    /**
80     * Does {@link MetaTableAccessor#getRegion(Connection, byte[])} and a write
81     * against hbase:meta while its hosted server is restarted to prove our retrying
82     * works.
83     * @throws IOException
84     * @throws InterruptedException
85     */
86    @Test public void testRetrying()
87    throws IOException, InterruptedException {
88      final TableName name =
89          TableName.valueOf("testRetrying");
90      LOG.info("Started " + name);
91      HTable t = UTIL.createMultiRegionTable(name, HConstants.CATALOG_FAMILY);
92      int regionCount = -1;
93      try (RegionLocator r = t.getRegionLocator()) {
94        regionCount = r.getStartKeys().length;
95      }
96      // Test it works getting a region from just made user table.
97      final List<HRegionInfo> regions =
98        testGettingTableRegions(connection, name, regionCount);
99      MetaTask reader = new MetaTask(connection, "reader") {
100       @Override
101       void metaTask() throws Throwable {
102         testGetRegion(connection, regions.get(0));
103         LOG.info("Read " + regions.get(0).getEncodedName());
104       }
105     };
106     MetaTask writer = new MetaTask(connection, "writer") {
107       @Override
108       void metaTask() throws Throwable {
109         MetaTableAccessor.addRegionToMeta(connection, regions.get(0));
110         LOG.info("Wrote " + regions.get(0).getEncodedName());
111       }
112     };
113     reader.start();
114     writer.start();
115 
116     // We're gonna check how it takes. If it takes too long, we will consider
117     //  it as a fail. We can't put that in the @Test tag as we want to close
118     //  the threads nicely
119     final long timeOut = 180000;
120     long startTime = System.currentTimeMillis();
121 
122     try {
123       // Make sure reader and writer are working.
124       assertTrue(reader.isProgressing());
125       assertTrue(writer.isProgressing());
126 
127       // Kill server hosting meta -- twice  . See if our reader/writer ride over the
128       // meta moves.  They'll need to retry.
129       for (int i = 0; i < 2; i++) {
130         LOG.info("Restart=" + i);
131         UTIL.ensureSomeRegionServersAvailable(2);
132         int index = -1;
133         do {
134           index = UTIL.getMiniHBaseCluster().getServerWithMeta();
135         } while (index == -1 &&
136           startTime + timeOut < System.currentTimeMillis());
137 
138         if (index != -1){
139           UTIL.getMiniHBaseCluster().abortRegionServer(index);
140           UTIL.getMiniHBaseCluster().waitOnRegionServer(index);
141         }
142       }
143 
144       assertTrue("reader: " + reader.toString(), reader.isProgressing());
145       assertTrue("writer: " + writer.toString(), writer.isProgressing());
146     } catch (IOException e) {
147       throw e;
148     } finally {
149       reader.stop = true;
150       writer.stop = true;
151       reader.join();
152       writer.join();
153       t.close();
154     }
155     long exeTime = System.currentTimeMillis() - startTime;
156     assertTrue("Timeout: test took " + exeTime / 1000 + " sec", exeTime < timeOut);
157   }

159   /**
160    * Thread that runs a MetaTableAccessor task until asked stop.
161    */
162   abstract static class MetaTask extends Thread {
163     boolean stop = false;
164     int count = 0;
165     Throwable t = null;
166     final Connection connection;
167 
168     MetaTask(final Connection connection, final String name) {
169       super(name);
170       this.connection = connection;
171     }
172 
173     @Override
174     public void run() {
175       try {
176         while(!this.stop) {
177           LOG.info("Before " + this.getName()+ ", count=" + this.count);
178           metaTask();
179           this.count += 1;
180           LOG.info("After " + this.getName() + ", count=" + this.count);
181           Thread.sleep(100);
182         }
183       } catch (Throwable t) {
184         LOG.info(this.getName() + " failed", t);
185         this.t = t;
186       }
187     }
188 
189     boolean isProgressing() throws InterruptedException {
190       int currentCount = this.count;
191       while(currentCount == this.count) {
192         if (!isAlive()) return false;
193         if (this.t != null) return false;
194         Thread.sleep(10);
195       }
196       return true;
197     }
198 
199     @Override
200     public String toString() {
201       return "count=" + this.count + ", t=" +
202         (this.t == null? "null": this.t.toString());
203     }
204 
205     abstract void metaTask() throws Throwable;
206   }

326   @Test
327   public void testMetaLocationsForRegionReplicas() throws IOException {
328     ServerName serverName0 = ServerName.valueOf("foo", 60010, random.nextLong());
329     ServerName serverName1 = ServerName.valueOf("bar", 60010, random.nextLong());
330     ServerName serverName100 = ServerName.valueOf("baz", 60010, random.nextLong());
331 
332     long regionId = System.currentTimeMillis();
333     HRegionInfo primary = new HRegionInfo(TableName.valueOf("table_foo"),
334       HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false, regionId, 0);
335     HRegionInfo replica1 = new HRegionInfo(TableName.valueOf("table_foo"),
336       HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false, regionId, 1);
337     HRegionInfo replica100 = new HRegionInfo(TableName.valueOf("table_foo"),
338       HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW, false, regionId, 100);
339 
340     long seqNum0 = random.nextLong();
341     long seqNum1 = random.nextLong();
342     long seqNum100 = random.nextLong();
343 
344 
345     Table meta = MetaTableAccessor.getMetaHTable(connection);
346     try {
347       MetaTableAccessor.updateRegionLocation(connection, primary, serverName0, seqNum0);
348 
349       // assert that the server, startcode and seqNum columns are there for the primary region
350       assertMetaLocation(meta, primary.getRegionName(), serverName0, seqNum0, 0, true);
351 
352       // add replica = 1
353       MetaTableAccessor.updateRegionLocation(connection, replica1, serverName1, seqNum1);
354       // check whether the primary is still there
355       assertMetaLocation(meta, primary.getRegionName(), serverName0, seqNum0, 0, true);
356       // now check for replica 1
357       assertMetaLocation(meta, primary.getRegionName(), serverName1, seqNum1, 1, true);
358 
359       // add replica = 1
360       MetaTableAccessor.updateRegionLocation(connection, replica100, serverName100, seqNum100);
361       // check whether the primary is still there
362       assertMetaLocation(meta, primary.getRegionName(), serverName0, seqNum0, 0, true);
363       // check whether the replica 1 is still there
364       assertMetaLocation(meta, primary.getRegionName(), serverName1, seqNum1, 1, true);
365       // now check for replica 1
366       assertMetaLocation(meta, primary.getRegionName(), serverName100, seqNum100, 100, true);
367     } finally {
368       meta.close();
369     }
370   }

372   public static void assertMetaLocation(Table meta, byte[] row, ServerName serverName,
373       long seqNum, int replicaId, boolean checkSeqNum) throws IOException {
374     Get get = new Get(row);
375     Result result = meta.get(get);
376     assertTrue(Bytes.equals(
377       result.getValue(HConstants.CATALOG_FAMILY, MetaTableAccessor.getServerColumn(replicaId)),
378       Bytes.toBytes(serverName.getHostAndPort())));
379     assertTrue(Bytes.equals(
380       result.getValue(HConstants.CATALOG_FAMILY, MetaTableAccessor.getStartCodeColumn(replicaId)),
381       Bytes.toBytes(serverName.getStartcode())));
382     if (checkSeqNum) {
383       assertTrue(Bytes.equals(
384         result.getValue(HConstants.CATALOG_FAMILY, MetaTableAccessor.getSeqNumColumn(replicaId)),
385         Bytes.toBytes(seqNum)));
386     }
387   }

389   public static void assertEmptyMetaLocation(Table meta, byte[] row, int replicaId)
390       throws IOException {
391     Get get = new Get(row);
392     Result result = meta.get(get);
393     Cell serverCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
394       MetaTableAccessor.getServerColumn(replicaId));
395     Cell startCodeCell = result.getColumnLatestCell(HConstants.CATALOG_FAMILY,
396       MetaTableAccessor.getStartCodeColumn(replicaId));
397     assertNotNull(serverCell);
398     assertNotNull(startCodeCell);
399     assertEquals(0, serverCell.getValueLength());
400     assertEquals(0, startCodeCell.getValueLength());
401   }
