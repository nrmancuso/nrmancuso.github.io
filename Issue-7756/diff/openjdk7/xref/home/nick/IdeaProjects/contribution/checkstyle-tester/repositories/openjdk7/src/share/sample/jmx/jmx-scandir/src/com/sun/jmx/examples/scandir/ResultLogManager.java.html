54  /**
55   * The <code>ResultLogManager</code> is in charge of managing result logs.
56   * {@link DirectoryScanner DirectoryScanners} can be configured to log a
57   * {@link ResultRecord} whenever they take action upon a file that
58   * matches their set of matching criteria.
59   * The <code>ResultLogManagerMXBean</code> is responsible for storing these
60   * results in its result logs.
61   * <p>The <code>ResultLogManagerMXBean</code> can be configured to log
62   * these records to a flat file, or into a log held in memory, or both.
63   * Both logs (file and memory) can be configured with a maximum capacity.
64   * <br>When the maximum capacity of the memory log is reached - its first
65   * entry (i.e. its eldest entry) is removed to make place for the latest.
66   * <br>When the maximum capacity of the file log is reached, the file is
67   * renamed by appending a tilde '~' to its name and a new result log is created.
68   *
69   *
70   * @author Sun Microsystems, 2006 - All rights reserved.
71   */
72  public class ResultLogManager extends NotificationBroadcasterSupport
73          implements ResultLogManagerMXBean, MBeanRegistration {

77       **/

79              ScanManager.makeSingletonName(ResultLogManagerMXBean.class);

87      // The memory log
88      //
89      private final List<ResultRecord> memoryLog;

91      // Whether the memory log capacity was reached. In that case every
92      // new entry triggers the deletion of the eldest one.
93      //
94      private volatile boolean memCapacityReached = false;
95  
96      // The maximum number of record that the memory log can
97      // contain.
98      //
99      private volatile int memCapacity;
100 
101     // The maximum number of record that the ResultLogManager can
102     // log in the log file before creating a new file.
103     //
104     private volatile long fileCapacity;
105 
106     // The current log file.
107     //
108     private volatile File logFile;
109 
110     // The OutputStream of the current log file.
111     //
112     private volatile OutputStream logStream = null;
113 
114     // number of record that this object has logged in the log file
115     // since the log file was created. Creating a new file or clearing
116     // the log file reset this value to '0'
117     //
118     private volatile long logCount = 0;
119 
120     // The ResultLogManager config - modified whenever
121     // ScanManager.applyConfiguration is called.

125     /**
126      * Create a new ResultLogManagerMXBean. This constructor is package
127      * protected: only the {@link ScanManager} can create a
128      * <code>ResultLogManager</code>.
129      **/
130     ResultLogManager() {
131         // Instantiate the memory log - override the add() method so that
132         // it removes the head of the list when the maximum capacity is
133         // reached. Note that add() is the only method we will be calling,
134         // otherwise we would have to override all the other flavors
135         // of adding methods. Note also that this implies that the memoryLog
136         // will *always* remain encapsulated in this object and is *never*
137         // handed over (otherwise we wouldn't be able to ensure that
138         // add() is the only method ever called to add a record).
139         //
140         memoryLog =
141                 Collections.synchronizedList(new LinkedList<ResultRecord>() {
142             public synchronized boolean add(ResultRecord e) {
143                 final int max = getMemoryLogCapacity();
144                 while (max > 0 && size() >= max) {
145                     memCapacityReached = true;
146                     removeFirst();
147                 }
148                 return super.add(e);
149             }
150         });
151 
152         // default memory capacity
153         memCapacity = 2048;
154 
155         // default file capacity: 0 means infinite ;-)
156         fileCapacity = 0;
157 
158         // by default logging to file is disabled.
159         logFile = null;
160 
161         // Until the ScanManager apply a new configuration, we're going to
162         // work with a default ResultLogConfig object.
163         config = new ResultLogConfig();
164         config.setMemoryMaxRecords(memCapacity);
165         config.setLogFileName(getLogFileName(false));
166         config.setLogFileMaxRecords(fileCapacity);
167     }

223     /**
224      * Allows the MBean to perform any operations needed after having been
225      * unregistered in the MBean server.
226      * <p>Closes the log file stream, if it is still open.</p>
227      */
228     public void postDeregister() {
229         try {
230             if (logStream != null) {
231                 synchronized(this)  {
232                     logStream.flush();
233                     logStream.close();
234                     logFile = null;
235                     logStream = null;
236                 }
237             }
238         } catch (Exception x) {
239             LOG.finest("Failed to close log properly: "+x);
240         }
241     }

243     /**
244      * Create a new empty log file from the given basename, renaming
245      * previously existing file by appending '~' to its name.
246      **/
247     private File createNewLogFile(String basename) throws IOException {
248         return XmlConfigUtils.createNewXmlFile(basename);
249     }

251     /**
252      * Check whether a new log file should be created.
253      * If a new file needs to be created, creates it, renaming
254      * previously existing file by appending '~' to its name.
255      * Also reset the log count and file capacity.
256      * Sends a notification indicating that the log file was changed.
257      * Returns the new log stream;
258      * Creation of a new file can be forced by passing force=true.
259      **/
260     private OutputStream checkLogFile(String basename, long maxRecords,
261                                       boolean force)
262     throws IOException {
263         final OutputStream newStream;
264         synchronized(this) {
265             if ((force==false) && (logCount < maxRecords))
266                 return logStream;
267             final OutputStream oldStream = logStream;
268 
269             // First close the stream. On some platforms you cannot rename
270             // a file that has open streams...
271             //
272             if (oldStream != null) {
273                 oldStream.flush();
274                 oldStream.close();
275             }
276             final File newFile = (basename==null)?null:createNewLogFile(basename);
277 
278             newStream = (newFile==null)?null:new FileOutputStream(newFile,true);
279             logStream = newStream;
280             logFile = newFile;
281             fileCapacity = maxRecords;
282             logCount = 0;
283         }
284         sendNotification(new Notification(LOG_FILE_CHANGED,objectName,
285                 getNextSeqNumber(),
286                 basename));
287         return newStream;
288     }

356     // Clear the memory log, sends a notification indicating that
357     // the memory log was cleared.
358     //
359     private void clearMemoryLog()throws IOException {
360         synchronized(this) {
361             memoryLog.clear();
362             memCapacityReached = false;
363         }
364         sendNotification(new Notification(MEMORY_LOG_CLEARED,
365                 objectName,
366                 getNextSeqNumber(),"memory log cleared"));
367     }
368 
369     // Clears the log file.
370     //
371     private void clearLogFile() throws IOException {
372         // simply force the creation of a new log file.
373         checkLogFile(getLogFileName(),fileCapacity,true);
374     }

376     // Log a record to the memory log. Send a notification if the
377     // maximum capacity of the memory log is reached.
378     //
379     private void logToMemory(ResultRecord record) {
380 
381         final boolean before = memCapacityReached;
382         final boolean after;
383         synchronized(this) {
384             memoryLog.add(record);
385             after = memCapacityReached;
386         }
387         if (before==false && after==true)
388             sendNotification(new Notification(MEMORY_LOG_MAX_CAPACITY,
389                     objectName,
390                     getNextSeqNumber(),"memory log capacity reached"));
391     }
392

394     // Log a record to the memory log. Send a notification if the
395     // maximum capacity of the memory log is reached.
396     //
397     private void logToFile(ResultRecord record) throws IOException {
398         final String basename;
399         final long   maxRecords;
400         synchronized (this) {
401             if (logFile == null) return;
402             basename = getLogFileName(false);
403             maxRecords = fileCapacity;
404         }
405 
406         // Get the stream into which we should log.
407         final OutputStream stream =
408                 checkLogFile(basename,maxRecords,false);
409 
410         // logging to file now disabled - too bad.
411         if (stream == null) return;
412 
413         synchronized (this) {
414             try {
415                 XmlConfigUtils.write(record,stream,true);
416                 stream.flush();
417                 // don't increment logCount if we were not logging in logStream.
418                 if (stream == logStream) logCount++;
419             } catch (JAXBException x) {
420                 final IllegalArgumentException iae =
421                         new IllegalArgumentException("bad record",x);
422                 LOG.finest("Failed to log record: "+x);
423                 throw iae;
424             }
425         }
426     }

470                     Notification.class.getName(),

472                     ,

474                 MEMORY_LOG_CLEARED},
475                     Notification.class.getName(),
476                     "Emitted when the memory log is cleared")
477         };

480     // Return the name of the log file, or null if logging to file is
481     // disabled.
482     private String getLogFileName(boolean absolute) {
483         synchronized (this) {
484             if (logFile == null) return null;
485             if (absolute) return logFile.getAbsolutePath();
486             return logFile.getPath();
487         }
488     }

490     // This method is be called by the ScanManagerMXBean when a configuration
491     // is applied.
492     //
493     void setConfig(ResultLogConfig logConfigBean) throws IOException {
494         if (logConfigBean == null)
495             throw new IllegalArgumentException("logConfigBean is null");
496         synchronized (this) {
497             config = logConfigBean;
498             setMemoryLogCapacity(config.getMemoryMaxRecords());
499         }
500         final String filename = config.getLogFileName();
501         final String logname  = getLogFileName(false);
502         if ((filename != null && !filename.equals(logname))
503         || (filename == null && logname != null)) {
504             newLogFile(config.getLogFileName(),
505                     config.getLogFileMaxRecords());
506         } else {
507             setLogFileCapacity(config.getLogFileMaxRecords());
508         }
509     }
510 
511     // This method is called by the ScanManagerMXBean when
512     // applyCurrentResultLogConfig() is called.
513     //
514     ResultLogConfig getConfig() {
515         return config;
516     }
