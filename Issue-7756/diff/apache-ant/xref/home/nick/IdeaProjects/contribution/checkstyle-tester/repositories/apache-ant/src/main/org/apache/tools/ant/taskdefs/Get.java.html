92      /**
93       * Does the work.
94       *
95       * @exception BuildException Thrown in unrecoverable error.
96       */
97      public void execute() throws BuildException {
98          checkAttributes();
99  
100         for (Resource r : sources) {
101             URLProvider up = r.as(URLProvider.class);
102             URL source = up.getURL();
103 
104             File dest = destination;
105             if (destination.isDirectory()) {
106                 if (mapperElement == null) {
107                     String path = source.getPath();
108                     if (path.endsWith("/")) {
109                         path = path.substring(0, path.length() - 1);
110                     }
111                     int slash = path.lastIndexOf("/");
112                     if (slash > -1) {
113                         path = path.substring(slash + 1);
114                     }
115                     dest = new File(destination, path);
116                 } else {
117                     FileNameMapper mapper = mapperElement.getImplementation();
118                     String[] d = mapper.mapFileName(source.toString());
119                     if (d == null) {
120                         log("skipping " + r + " - mapper can't handle it",
121                             Project.MSG_WARN);
122                         continue;
123                     } else if (d.length == 0) {
124                         log("skipping " + r + " - mapper returns no file name",
125                             Project.MSG_WARN);
126                         continue;
127                     } else if (d.length > 1) {
128                         log("skipping " + r + " - mapper returns multiple file"
129                             + " names", Project.MSG_WARN);
130                         continue;
131                     }
132                     dest = new File(destination, d[0]);
133                 }
134             }
135 
136         //set up logging
137         int logLevel = Project.MSG_INFO;
138         DownloadProgress progress = null;
139         if (verbose) {
140             progress = new VerboseProgress(System.out);
141         }
142 
143         //execute the get
144         try {
145             doGet(source, dest, logLevel, progress);
146         } catch (IOException ioe) {
147             log("Error getting " + source + " to " + dest);
148             if (!ignoreErrors) {
149                 throw new BuildException(ioe, getLocation());
150             }
151         }
152         }
153     }

156      * make a get request, with the supplied progress and logging info.
157      * All the other config parameters are set at the task level,
158      * source, dest, ignoreErrors, etc.
159      * @param logLevel level to log at, see {@link Project#log(String, int)}
160      * @param progress progress callback; null for no-callbacks
161      * @return true for a successful download, false otherwise.
162      * The return value is only relevant when {@link #ignoreErrors} is true, as
163      * when false all failures raise BuildExceptions.
164      * @throws IOException for network trouble
165      * @throws BuildException for argument errors, or other trouble when ignoreErrors
166      * is false.
167      * @deprecated only gets the first configured resource
168      */
169     public boolean doGet(int logLevel, DownloadProgress progress)
170             throws IOException {
171         checkAttributes();
172         for (Resource r : sources) {
173             URLProvider up = r.as(URLProvider.class);
174             URL source = up.getURL();
175             return doGet(source, destination, logLevel, progress);
176         }
177         /*NOTREACHED*/
178         return false;
179     }

181     /**
182      * make a get request, with the supplied progress and logging info.
183      *
184      * All the other config parameters like ignoreErrors are set at
185      * the task level.
186      * @param source the URL to get
187      * @param dest the target file
188      * @param logLevel level to log at, see {@link Project#log(String, int)}
189      * @param progress progress callback; null for no-callbacks
190      * @return true for a successful download, false otherwise.
191      * The return value is only relevant when {@link #ignoreErrors} is true, as
192      * when false all failures raise BuildExceptions.
193      * @throws IOException for network trouble
194      * @throws BuildException for argument errors, or other trouble when ignoreErrors
195      * is false.
196      * @since Ant 1.8.0
197      */
198     public boolean doGet(URL source, File dest, int logLevel,
199                          DownloadProgress progress)
200         throws IOException {
201 
202         if (dest.exists() && skipExisting) {
203             log("Destination already exists (skipping): "
204                 + dest.getAbsolutePath(), logLevel);
205             return true;
206         }
207 
208         //dont do any progress, unless asked
209         if (progress == null) {
210             progress = new NullProgress();
211         }
212         log("Getting: " + source, logLevel);
213         log("To: " + dest.getAbsolutePath(), logLevel);
214 
215         //set the timestamp to the file date.
216         long timestamp = 0;
217 
218         boolean hasTimestamp = false;
219         if (useTimestamp && dest.exists()) {
220             timestamp = dest.lastModified();
221             if (verbose) {
222                 Date t = new Date(timestamp);
223                 log("local file date : " + t.toString(), logLevel);
224             }
225             hasTimestamp = true;
226         }
227 
228         GetThread getThread = new GetThread(source, dest,
229                                             hasTimestamp, timestamp, progress,
230                                             logLevel, userAgent);
231         getThread.setDaemon(true);
232         getProject().registerThreadTask(getThread, this);
233         getThread.start();
234         try {
235             getThread.join(maxTime * 1000);
236         } catch (InterruptedException ie) {
237             log("interrupted waiting for GET to finish",
238                 Project.MSG_VERBOSE);
239         }
240 
241         if (getThread.isAlive()) {
242             String msg = "The GET operation took longer than " + maxTime
243                 + " seconds, stopping it.";
244             if (ignoreErrors) {
245                 log(msg);
246             }
247             getThread.closeStreams();
248             if (!ignoreErrors) {
249                 throw new BuildException(msg);
250             }
251             return false;
252         }
253 
254         return getThread.wasSuccessful();
255     }

264     /**
265      * Check the attributes.
266      */
267     private void checkAttributes() {
268 
269         if (userAgent == null || userAgent.trim().length() == 0) {
270             throw new BuildException("userAgent may not be null or empty");
271         }
272 
273         if (sources.size() == 0) {
274             throw new BuildException("at least one source is required",
275                                      getLocation());
276         }
277         for (Resource r : sources) {
278             URLProvider up = r.as(URLProvider.class);
279             if (up == null) {
280                 throw new BuildException("Only URLProvider resources are"
281                                          + " supported", getLocation());
282             }
283         }
284 
285         if (destination == null) {
286             throw new BuildException("dest attribute is required", getLocation());
287         }
288 
289         if (destination.exists() && sources.size() > 1
290             && !destination.isDirectory()) {
291             throw new BuildException("The specified destination is not a"
292                                      + " directory",
293                                      getLocation());
294         }
295 
296         if (destination.exists() && !destination.canWrite()) {
297             throw new BuildException("Can't write to "
298                                      + destination.getAbsolutePath(),
299                                      getLocation());
300         }
301 
302         if (sources.size() > 1 && !destination.exists()) {
303             destination.mkdirs();
304         }
305     }

361     /**
362      * If true, conditionally download a file based on the timestamp
363      * of the local copy.
364      *
365      * <p>In this situation, the if-modified-since header is set so
366      * that the file is only fetched if it is newer than the local
367      * file (or there is no local file) This flag is only valid on
368      * HTTP connections, it is ignored in other cases.  When the flag
369      * is set, the local copy of the downloaded file will also have
370      * its timestamp set to the remote file time.</p>
371      *
372      * <p>Note that remote files of date 1/1/1970 (GMT) are treated as
373      * 'no timestamp', and web servers often serve files with a
374      * timestamp in the future by replacing their timestamp with that
375      * of the current time. Also, inter-computer clock differences can
376      * cause no end of grief.</p>
377      * @param v "true" to enable file time fetching
378      */
379     public void setUseTimestamp(boolean v) {
380         useTimestamp = v;

578          */

580             out.println();
581             out.flush();

585     private class GetThread extends Thread {
586 
587         private final URL source;
588         private final File dest;
589         private final boolean hasTimestamp;
590         private final long timestamp;
591         private final DownloadProgress progress;
592         private final int logLevel;
593 
594         private boolean success = false;
595         private IOException ioexception = null;
596         private BuildException exception = null;
597         private InputStream is = null;
598         private OutputStream os = null;
599         private URLConnection connection;
600         private int redirections = 0;
601         private String userAgent = null;

603         GetThread(URL source, File dest,
604                   boolean h, long t, DownloadProgress p, int l, String userAgent) {
605             this.source = source;
606             this.dest = dest;

608             timestamp = t;

614         public void run() {
615             try {
616                 success = get();
617             } catch (IOException ioex) {

619             } catch (BuildException bex) {

624         private boolean get() throws IOException, BuildException {
625 
626             connection = openConnection(source);
627 
628             if (connection == null)
629             {
630                 return false;
631             }
632 
633             boolean downloadSucceeded = downloadFile();
634 
635             //if (and only if) the use file time option is set, then
636             //the saved file now has its timestamp set to that of the
637             //downloaded file
638             if (downloadSucceeded && useTimestamp)  {
639                 updateTimeStamp();
640             }
641 
642             return downloadSucceeded;
643         }

646         private boolean redirectionAllowed(URL aSource, URL aDest) {
647             if (!(aSource.getProtocol().equals(aDest.getProtocol()) || (HTTP
648                     .equals(aSource.getProtocol()) && HTTPS.equals(aDest
649                     .getProtocol())))) {
650                 String message = "Redirection detected from "
651                         + aSource.getProtocol() + " to " + aDest.getProtocol()
652                         + ". Protocol switch unsafe, not allowed.";
653                 if (ignoreErrors) {
654                     log(message, logLevel);
655                     return false;
656                 } else {
657                     throw new BuildException(message);
658                 }
659             }
660 
661             redirections++;
662             if (redirections > REDIRECT_LIMIT) {
663                 String message = "More than " + REDIRECT_LIMIT
664                         + " times redirected, giving up";
665                 if (ignoreErrors) {
666                     log(message, logLevel);
667                     return false;
668                 } else {
669                     throw new BuildException(message);
670                 }
671             }
672 
673 
674             return true;
675         }

677         private URLConnection openConnection(URL aSource) throws IOException {
678 
679             // set up the URL connection
680             URLConnection connection = aSource.openConnection();
681             // modify the headers
682             // NB: things like user authentication could go in here too.
683             if (hasTimestamp) {
684                 connection.setIfModifiedSince(timestamp);
685             }
686             // Set the user agent
687             connection.addRequestProperty("User-Agent", this.userAgent);
688             
689             // prepare Java 1.1 style credentials
690             if (uname != null || pword != null) {
691                 String up = uname + ":" + pword;
692                 String encoding;
693                 // we do not use the sun impl for portability,
694                 // and always use our own implementation for consistent
695                 // testing
696                 Base64Converter encoder = new Base64Converter();
697                 encoding = encoder.encode(up.getBytes());
698                 connection.setRequestProperty("Authorization", "Basic "
699                         + encoding);
700             }
701 
702             connection.setRequestProperty("Accept-Encoding", GZIP_CONTENT_ENCODING);
703 
704             if (connection instanceof HttpURLConnection) {
705                 ((HttpURLConnection) connection)
706                         .setInstanceFollowRedirects(false);
707                 ((HttpURLConnection) connection)
708                         .setUseCaches(httpUseCaches);
709             }
710             // connect to the remote site (may take some time)
711             try {
712                 connection.connect();
713             } catch (NullPointerException e) {
714                 //bad URLs can trigger NPEs in some JVMs
715                 throw new BuildException("Failed to parse " + source.toString(), e);
716             }
717 
718             // First check on a 301 / 302 (moved) response (HTTP only)
719             if (connection instanceof HttpURLConnection) {
720                 HttpURLConnection httpConnection = (HttpURLConnection) connection;
721                 int responseCode = httpConnection.getResponseCode();
722                 if (responseCode == HttpURLConnection.HTTP_MOVED_PERM || 
723                         responseCode == HttpURLConnection.HTTP_MOVED_TEMP ||
724                         responseCode == HttpURLConnection.HTTP_SEE_OTHER ||
725                         responseCode == HTTP_MOVED_TEMP)
726                 {
727                     String newLocation = httpConnection.getHeaderField("Location");
728                     String message = aSource
729                             + (responseCode == HttpURLConnection.HTTP_MOVED_PERM ? " permanently"
730                                     : "") + " moved to " + newLocation;
731                     log(message, logLevel);
732                     URL newURL = new URL(aSource, newLocation);
733                     if (!redirectionAllowed(aSource, newURL))
734                     {
735                         return null;
736                     }
737                     return openConnection(newURL);
738                 }
739                 // next test for a 304 result (HTTP only)
740                 long lastModified = httpConnection.getLastModified();
741                 if (responseCode == HttpURLConnection.HTTP_NOT_MODIFIED
742                         || (lastModified != 0 && hasTimestamp && timestamp >= lastModified)) {
743                     // not modified so no file download. just return
744                     // instead and trace out something so the user
745                     // doesn't think that the download happened when it
746                     // didn't
747                     log("Not modified - so not downloaded", logLevel);
748                     return null;
749                 }
750                 // test for 401 result (HTTP only)
751                 if (responseCode == HttpURLConnection.HTTP_UNAUTHORIZED) {
752                     String message = "HTTP Authorization failure";
753                     if (ignoreErrors) {
754                         log(message, logLevel);
755                         return null;
756                     } else {
757                         throw new BuildException(message);
758                     }
759                 }
760             }
761 
762             //REVISIT: at this point even non HTTP connections may
763             //support the if-modified-since behaviour -we just check
764             //the date of the content and skip the write if it is not
765             //newer. Some protocols (FTP) don't include dates, of
766             //course.
767             return connection;
768         }

770         private boolean downloadFile()
771                 throws FileNotFoundException, IOException {
772             for (int i = 0; i < numberRetries; i++) {
773                 // this three attempt trick is to get round quirks in different
774                 // Java implementations. Some of them take a few goes to bind
775                 // property; we ignore the first couple of such failures.
776                 try {
777                     is = connection.getInputStream();
778                     break;
779                 } catch (IOException ex) {
780                     log("Error opening connection " + ex, logLevel);
781                 }
782             }
783             if (is == null) {
784                 log("Can't get " + source + " to " + dest, logLevel);
785                 if (ignoreErrors) {
786                     return false;
787                 }
788                 throw new BuildException("Can't get " + source + " to " + dest,
789                         getLocation());
790             }
791 
792             if (GZIP_CONTENT_ENCODING.equals(connection.getContentEncoding())) {
793             	is = new GZIPInputStream(is);
794             }
795 
796             os = new FileOutputStream(dest);
797             progress.beginDownload();
798             boolean finished = false;
799             try {
800                 byte[] buffer = new byte[BIG_BUFFER_SIZE];
801                 int length;
802                 while (!isInterrupted() && (length = is.read(buffer)) >= 0) {
803                     os.write(buffer, 0, length);
804                     progress.onTick();
805                 }
806                 finished = !isInterrupted();
807             } finally {
808                 FileUtils.close(os);
809                 FileUtils.close(is);
810 
811                 // we have started to (over)write dest, but failed.
812                 // Try to delete the garbage we'd otherwise leave
813                 // behind.
814                 if (!finished) {
815                     dest.delete();
816                 }
817             }
818             progress.endDownload();
819             return true;
820         }
821 
822         private void updateTimeStamp() {
823             long remoteTimestamp = connection.getLastModified();
824             if (verbose)  {
825                 Date t = new Date(remoteTimestamp);
826                 log("last modified = " + t.toString()
827                     + ((remoteTimestamp == 0)
828                        ? " - using current time instead"
829                        : ""), logLevel);
830             }
831             if (remoteTimestamp != 0) {
832                 FILE_UTILS.setFileLastModified(dest, remoteTimestamp);
833             }
834         }

836         /**
837          * Has the download completed successfully?
838          *
839          * <p>Re-throws any exception caught during executaion.</p>
840          */
841         boolean wasSuccessful() throws IOException, BuildException {
842             if (ioexception != null) {
843                 throw ioexception;
844             }
845             if (exception != null) {
846                 throw exception;
847             }
848             return success;
849         }

851         /**
852          * Closes streams, interrupts the download, may delete the
853          * output file.
854          */
855         void closeStreams() {
856             interrupt();
857             FileUtils.close(os);
858             FileUtils.close(is);
859             if (!success && dest.exists()) {
860                 dest.delete();
861             }
862         }
