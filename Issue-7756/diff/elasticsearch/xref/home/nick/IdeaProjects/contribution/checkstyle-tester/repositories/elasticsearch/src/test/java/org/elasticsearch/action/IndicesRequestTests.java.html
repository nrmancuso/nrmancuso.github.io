281         for (int i = 0; i < numIndexRequests; i++) {

290             indices.add(indexOrAlias);

292         int numUpdateRequests = iterations(1, 10);
293         for (int i = 0; i < numUpdateRequests; i++) {
294             String indexOrAlias = randomIndexOrAlias();
295             bulkRequest.add(new UpdateRequest(indexOrAlias, "type", "id").doc("field1", "value1"));

297         }
298 
299         internalCluster().clientNodeClient().bulk(bulkRequest).actionGet();

807     private static void assertSameIndices(IndicesRequest originalRequest, boolean optional, String... actions) {
808         for (String action : actions) {
809             List<TransportRequest> requests = consumeTransportRequests(action);
810             if (!optional) {
811                 assertThat("no internal requests intercepted for action [" + action + "]", requests.size(), greaterThan(0));
812             }
813             for (TransportRequest internalRequest : requests) {
814                 assertThat(internalRequest, instanceOf(IndicesRequest.class));
815                 assertThat(internalRequest.getClass().getName(), ((IndicesRequest)internalRequest).indices(), equalTo(originalRequest.indices()));
816                 assertThat(((IndicesRequest)internalRequest).indicesOptions(), equalTo(originalRequest.indicesOptions()));
817             }
818         }
819     }

838     private static void assertIndicesSubset(List<String> indices, String... actions) {
839         //indices returned by each bulk shard request need to be a subset of the original indices
840         for (String action : actions) {
841             List<TransportRequest> requests = consumeTransportRequests(action);
842             assertThat("no internal requests intercepted for action [" + action + "]", requests.size(), greaterThan(0));
843             for (TransportRequest internalRequest : requests) {
844                 assertThat(internalRequest, instanceOf(IndicesRequest.class));
845                 for (String index : ((IndicesRequest) internalRequest).indices()) {
846                     assertThat(indices, hasItem(index));
847                 }
848             }
849         }

870     private String[] randomUniqueIndicesOrAliases() {
871         Set<String> uniqueIndices = new HashSet<>();
872         int count = randomIntBetween(1, this.indices.size());
873         while (uniqueIndices.size() < count) {
874             uniqueIndices.add(randomFrom(this.indices));
875         }

877         int i = 0;
878         for (String index : uniqueIndices) {

880         }
881         return indices;
882     }

905     private static List<TransportRequest> consumeTransportRequests(String action) {
906         List<TransportRequest> requests = new ArrayList<>();
907         Iterable<TransportService> transportServices = internalCluster().getInstances(TransportService.class);
908         for (TransportService transportService : transportServices) {
909             List<TransportRequest> transportRequests = ((InterceptingTransportService) transportService).consumeRequests(action);
910             if (transportRequests != null) {
911                 requests.addAll(transportRequests);
912             }
913         }
914         return requests;
915     }

917     public static class InterceptingTransportService extends TransportService {
918 
919         private final Set<String> actions = new HashSet<>();
920 
921         private final Map<String, List<TransportRequest>> requests = new HashMap<>();
922 
923         @Inject
924         public InterceptingTransportService(Settings settings, Transport transport, ThreadPool threadPool,
925                                             NodeSettingsService nodeSettingsService, @ClusterDynamicSettings DynamicSettings dynamicSettings) {
926             super(settings, transport, threadPool);
927         }
928 
929         synchronized List<TransportRequest> consumeRequests(String action) {
930             return requests.remove(action);
931         }
932 
933         synchronized void interceptTransportActions(String... actions) {
934             Collections.addAll(this.actions, actions);
935         }
936 
937         synchronized void clearInterceptedActions() {
938             actions.clear();
939         }
940 
941         @Override
942         public void registerHandler(String action, TransportRequestHandler handler) {
943             super.registerHandler(action, new InterceptingRequestHandler(action, handler));
944         }
945 
946         private class InterceptingRequestHandler implements TransportRequestHandler {
947 
948             private final TransportRequestHandler requestHandler;
949             private final String action;
950 
951             InterceptingRequestHandler(String action, TransportRequestHandler requestHandler) {
952                 this.requestHandler = requestHandler;
953                 this.action = action;
954             }
955 
956             @Override
957             public TransportRequest newInstance() {
958                 return requestHandler.newInstance();
959             }
960 
961             @Override
962             public void messageReceived(TransportRequest request, TransportChannel channel) throws Exception {
963                 synchronized (InterceptingTransportService.this) {
964                     if (actions.contains(action)) {
965                         List<TransportRequest> requestList = requests.get(action);
966                         if (requestList == null) {
967                             requestList = new ArrayList<>();
968                             requestList.add(request);
969                             requests.put(action, requestList);
970                         } else {
971                             requestList.add(request);
972                         }
973                     }
974                 }
975                 requestHandler.messageReceived(request, channel);
976             }
977 
978             @Override
979             public String executor() {
980                 return requestHandler.executor();
981             }
982 
983             @Override
984             public boolean isForceExecution() {
985                 return requestHandler.isForceExecution();
986             }
987         }
988     }
