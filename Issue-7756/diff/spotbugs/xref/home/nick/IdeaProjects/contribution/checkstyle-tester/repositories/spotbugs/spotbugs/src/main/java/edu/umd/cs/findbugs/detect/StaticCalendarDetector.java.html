54  /**
55   * Detector for static fields of type {@link java.util.Calendar} or
56   * {@link java.text.DateFormat} and their subclasses. Because
57   * {@link java.util.Calendar} is unsafe for multithreaded use, static fields
58   * look suspicous. To work correctly, all access would need to be synchronized
59   * by the client which cannot be guaranteed.
60   *
61   * @author Daniel Schneller
62   */
63  public class StaticCalendarDetector extends OpcodeStackDetector {
64  
65      /** External Debug flag set? */
66      private static final boolean DEBUG = Boolean.getBoolean("debug.staticcal");
67  
68      /**
69       * External flag to determine whether to skip the test for synchronized
70       * blocks (default: if a call on a static Calendar or DateFormat is detected
71       * inside a synchronizationb block, it will not be reported). Setting this
72       * to <code>true</code> will report method calls on static fields if they
73       * are in a synchronized block. As the check currently does not take into
74       * account the lock's mutex it may be useful to switch allow
75       */
76      private static final String PROP_SKIP_SYNCHRONIZED_CHECK = "staticcal.skipsynccheck";
77  
78      /** The reporter to report to */
79      final private BugReporter reporter;
80  
81      final private BugAccumulator bugAccumulator;
82  
83      /** Name of the class being inspected */
84      private String currentClass;
85  
86      /**
87       * {@link org.apache.bcel.generic.ObjectType} for {@link java.util.Calendar}
88       */
89      private final ClassDescriptor calendarType = DescriptorFactory.createClassDescriptor(java.util.Calendar.class);
90  
91      /**
92       * {@link org.apache.bcel.generic.ObjectType} for
93       * {@link java.text.DateFormat}
94       */
95      private final ClassDescriptor dateFormatType = DescriptorFactory.createClassDescriptor(java.text.DateFormat.class);
96  
97      /** Stores the current method */
98      private Method currentMethod = null;
99  
100     /** Stores current Control Flow Graph */
101     private CFG currentCFG;
102 
103     /** Stores current LDF */
104     private LockDataflow currentLockDataFlow;
105 
106     private final Map<XField, BugInstance> pendingBugs = new HashMap<>();
107 
108     /**
109      * Creates a new instance of this Detector.
110      *
111      * @param aReporter
112      *            {@link BugReporter} instance to report found problems to.
113      */
114     public StaticCalendarDetector(BugReporter aReporter) {
115         reporter = aReporter;
116         bugAccumulator = new BugAccumulator(reporter);
117     }
118 
119     Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
120 
121     private boolean sawDateClass;
122 
123     /**
124      * Remembers the class name and resets temporary fields.
125      */
126     @Override
127     public void visit(JavaClass someObj) {
128         currentClass = someObj.getClassName();
129         currentMethod = null;
130         currentCFG = null;
131         currentLockDataFlow = null;
132         sawDateClass = false;
133 
134     }
135 
136     @Override
137     public void visit(ConstantPool pool) {
138         for (Constant constant : pool.getConstantPool()) {
139             if (constant instanceof ConstantClass) {
140                 ConstantClass cc = (ConstantClass) constant;
141                 @SlashedClassName String className = cc.getBytes(pool);
142                 if ("java/util/Calendar".equals(className) || "java/text/DateFormat".equals(className)) {
143                     sawDateClass = true;
144                     break;
145                 }
146                 if (className.charAt(0) != '[') {
147                     try {
148                         ClassDescriptor cDesc = DescriptorFactory.createClassDescriptor(className);
149 
150                         if (subtypes2.isSubtype(cDesc, calendarType) || subtypes2.isSubtype(cDesc, dateFormatType)) {
151                             sawDateClass = true;
152                             break;
153                         }
154                     } catch (ClassNotFoundException e) {
155                         reporter.reportMissingClass(e);
156                     }
157                 }
158             }
159         }
160     }
161 
162     /**
163      * Checks if the visited field is of type {@link java.util.Calendar} or
164      * {@link java.text.DateFormat} or a subclass of either one. If so and the
165      * field is static and non-private it is suspicious and will be reported.
166      */
167     @Override
168     public void visit(Field aField) {
169         if (aField.isPrivate()) {
170             /*
171              * private fields are harmless, as long as they are used correctly
172              * inside their own class. This should be something the rest of this
173              * detector can find out, so do not report them, they might be false
174              * positives.
175              */
176             return;
177         }
178         String superclassName = getSuperclassName();
179         if (!aField.isStatic() && !"java/lang/Enum".equals(superclassName)) {
180             return;
181         }
182         if (!aField.isPublic() && !aField.isProtected()) {
183             return;
184         }
185         ClassDescriptor classOfField = DescriptorFactory.createClassDescriptorFromFieldSignature(aField.getSignature());
186         String tBugType = null;
187         int priority = aField.isPublic() && aField.isFinal() && aField.getName().equals(aField.getName().toUpperCase())
188                 && getThisClass().isPublic() ? HIGH_PRIORITY : NORMAL_PRIORITY;
189         if (classOfField != null) {
190             try {
191                 if (subtypes2.isSubtype(classOfField, calendarType)) {
192                     tBugType = "STCAL_STATIC_CALENDAR_INSTANCE";
193                     priority++;
194                 } else if (subtypes2.isSubtype(classOfField, dateFormatType)) {
195                     tBugType = "STCAL_STATIC_SIMPLE_DATE_FORMAT_INSTANCE";
196                 }
197                 if (getClassContext().getXClass().usesConcurrency()) {
198                     priority--;
199                 }
200                 if (tBugType != null) {
201 
202                     pendingBugs.put(getXField(), new BugInstance(this, tBugType, priority).addClass(currentClass).addField(this));
203                 }
204             } catch (ClassNotFoundException e) {
205                 AnalysisContext.reportMissingClass(e);
206             }
207         }
208 
209     }
210 
211     /*
212      * (non-Javadoc)
213      *
214      * @see
215      * edu.umd.cs.findbugs.visitclass.BetterVisitor#visitMethod(org.apache.bcel
216      * .classfile.Method)
217      */
218     @Override
219     public void visitMethod(Method obj) {
220         if (sawDateClass) {
221             try {
222                 super.visitMethod(obj);
223                 currentMethod = obj;
224                 currentLockDataFlow = getClassContext().getLockDataflow(currentMethod);
225                 currentCFG = getClassContext().getCFG(currentMethod);
226             } catch (CFGBuilderException e) {
227                 reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
228             } catch (DataflowAnalysisException e) {
229                 reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
230             }
231         }
232     }
233 
234     @Override
235     public void visit(Code obj) {
236         if (sawDateClass) {
237             super.visit(obj);
238             bugAccumulator.reportAccumulatedBugs();
239         }
240     }

242     /**
243      * Checks for method invocations (
244      * {@link org.apache.bcel.generic.INVOKEVIRTUAL}) call on a static
245      * {@link java.util.Calendar} or {@link java.text.DateFormat} fields. The
246      * {@link OpcodeStack} is used to determine if an invocation is done on such
247      * a static field.
248      *
249      * @param seen
250      *            An opcode to be analyzed
251      * @see edu.umd.cs.findbugs.visitclass.DismantleBytecode#sawOpcode(int)
252      */
253     @Override
254     public void sawOpcode(int seen) {
255 
256         if (seen == Const.GETSTATIC) {
257             XField f = getXFieldOperand();
258             if (pendingBugs.containsKey(f)) {
259                 if (!isLocked()) {
260                     reporter.reportBug(pendingBugs.remove(f));
261 
262                 }
263 
264             }
265         }
266         // we are only interested in method calls
267         if (seen != Const.INVOKEVIRTUAL) {
268             return;
269         }
270 
271         try {
272             @SlashedClassName String className = getClassConstantOperand();
273 
274             if (className.startsWith("[")) {
275                 // Ignore array classes
276                 return;
277             }
278             ClassDescriptor cDesc = DescriptorFactory.createClassDescriptor(className);
279 
280             // if it is not compatible with Calendar or DateFormat, we are not
281             // interested anymore
282             boolean isCalendar = subtypes2.isSubtype(cDesc, calendarType);
283             boolean isDateFormat = subtypes2.isSubtype(cDesc, dateFormatType);
284             if (!isCalendar && !isDateFormat) {
285                 return;
286             }
287 
288             // determine the number of arguments the method expects
289             int numArguments = getNumberArguments(getSigConstantOperand());
290             // go back on the stack to find what the receiver of the method is
291             OpcodeStack.Item invokedOn = stack.getStackItem(numArguments);
292             XField field = invokedOn.getXField();
293             // find out, if the field is static. if not, we are not interested
294             // anymore
295             if (field == null || !field.isStatic()) {
296                 return;
297             }
298 
299             if (Const.STATIC_INITIALIZER_NAME.equals(getMethodName()) && field.getClassName().equals(getDottedClassName())) {
300                 return;
301             }
302             String invokedName = getNameConstantOperand();
303             if (invokedName.startsWith("get")) {
304                 return;
305             }
306             if ("equals".equals(invokedName) && numArguments == 1) {
307                 OpcodeStack.Item passedAsArgument = stack.getStackItem(0);
308                 field = passedAsArgument.getXField();
309                 if (field == null || !field.isStatic()) {
310                     return;
311                 }
312             }
313 
314             if (!SystemProperties.getBoolean(PROP_SKIP_SYNCHRONIZED_CHECK)) {
315                 // check synchronization
316                 if (isLocked()) {
317                     return;
318                 }
319             }
320 
321             // if we get here, we want to generate a report, depending on the
322             // type
323             String tBugType;
324             if (isCalendar) {
325                 tBugType = "STCAL_INVOKE_ON_STATIC_CALENDAR_INSTANCE";
326             } else if (isDateFormat) {
327                 tBugType = "STCAL_INVOKE_ON_STATIC_DATE_FORMAT_INSTANCE";
328             } else {
329                 throw new IllegalStateException("Not possible");
330             }
331             int priority;
332             if (amVisitingMainMethod()) {
333                 priority = LOW_PRIORITY;
334             } else {
335                 if (getClassContext().getXClass().usesConcurrency()) {
336                     priority = NORMAL_PRIORITY;
337                 } else {
338                     priority = LOW_PRIORITY;
339                 }
340                 if (invokedName.startsWith("set") || "format".equals(invokedName) || "add".equals(invokedName)
341                         || "clear".equals(invokedName) || "parse".equals(invokedName) || "applyPattern".equals(invokedName)) {
342                     priority--;
343                 }
344             }
345             bugAccumulator.accumulateBug(new BugInstance(this, tBugType, priority).addClassAndMethod(this).addCalledMethod(this)
346                     .addOptionalField(field), this);
347 
348         } catch (ClassNotFoundException e) {
349             AnalysisContext.reportMissingClass(e);
350         }
351     }
352 
353     private boolean isLocked() {
354         try {
355             if (currentMethod != null && currentLockDataFlow != null && currentCFG != null) {
356                 Collection<Location> tLocations = currentCFG.getLocationsContainingInstructionWithOffset(getPC());
357                 for (Location tLoc : tLocations) {
358                     LockSet lockSet = currentLockDataFlow.getFactAtLocation(tLoc);
359                     if (lockSet.getNumLockedObjects() > 0) {
360                         // within a synchronized block
361                         return true;
362                     }
363                 }
364             }
365         } catch (DataflowAnalysisException e) {
366             reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
367         }
368         return false;
369     }
370 
371     @Override
372     public void report() {
373         // Report any fields that were not accessed, but are potentially
374         // accessible
375         for (BugInstance bug : pendingBugs.values()) {
376             reporter.reportBug(bug);
377         }
378     }
