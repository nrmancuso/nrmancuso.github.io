115     static public @Nonnull
116     IncompatibleTypes getPriorityForAssumingCompatible(GenericObjectType genericType, Type plainType) {
117         IncompatibleTypes result = IncompatibleTypes.getPriorityForAssumingCompatible(genericType.getObjectType(), plainType);
118         List<? extends ReferenceType> parameters = genericType.getParameters();
119         if (result.getPriority() == Priorities.NORMAL_PRIORITY && parameters != null && parameters.contains(plainType)) {
120             result = UNRELATED_TYPES_BUT_MATCHES_TYPE_PARAMETER;
121         }
122         return result;
123 
124     }

131     static public @Nonnull
132     IncompatibleTypes getPriorityForAssumingCompatible(Type expectedType, Type actualType, boolean pointerEquality) {
133         if (expectedType.equals(actualType)) {
134             return SEEMS_OK;
135         }
136 
137         if (!(expectedType instanceof ReferenceType)) {
138             return SEEMS_OK;
139         }
140         if (!(actualType instanceof ReferenceType)) {
141             return SEEMS_OK;
142         }
143 
144         while (expectedType instanceof ArrayType && actualType instanceof ArrayType) {
145             expectedType = ((ArrayType) expectedType).getElementType();
146             actualType = ((ArrayType) actualType).getElementType();
147         }
148 
149         if (expectedType instanceof BasicType ^ actualType instanceof BasicType) {
150             return PRIMATIVE_ARRAY_AND_OTHER_ARRAY;
151         }
152         if (expectedType instanceof BasicType && actualType instanceof BasicType) {
153             if (!expectedType.equals(actualType)) {
154                 return INCOMPATIBLE_PRIMATIVE_ARRAYS;
155             } else {
156                 return SEEMS_OK;
157             }
158         }
159         if (expectedType instanceof ArrayType) {
160             return getPriorityForAssumingCompatibleWithArray(actualType);
161         }
162         if (actualType instanceof ArrayType) {
163             return getPriorityForAssumingCompatibleWithArray(expectedType);
164         }
165         if (expectedType.equals(actualType)) {
166             return SEEMS_OK;
167         }
168 
169         // For now, ignore the case where either reference is not
170         // of an object type. (It could be either an array or null.)
171         if (!(expectedType instanceof ObjectType) || !(actualType instanceof ObjectType)) {
172             return SEEMS_OK;
173         }
174 
175         return getPriorityForAssumingCompatible((ObjectType) expectedType, (ObjectType) actualType, pointerEquality);
176     }
177 
178     private static IncompatibleTypes getPriorityForAssumingCompatibleWithArray(Type rhsType) {
179         if (rhsType.equals(Type.OBJECT)) {
180             return ARRAY_AND_OBJECT;
181         }
182         String sig = rhsType.getSignature();
183         if ("Ljava/io/Serializable;".equals(sig) || "Ljava/lang/Cloneable;".equals(sig)) {
184             return SEEMS_OK;
185         }
186         return ARRAY_AND_NON_ARRAY;
187     }

189     static @Nonnull
190     XMethod getInvokedMethod(XClass xClass, String name, String sig, boolean isStatic) throws CheckedAnalysisException {
191         IAnalysisCache cache = Global.getAnalysisCache();
192         while (true) {
193             XMethod result = xClass.findMethod(name, sig, isStatic);
194             if (result != null) {
195                 return result;
196             }
197             if (isStatic) {
198                 throw new CheckedAnalysisException();
199             }
200             ClassDescriptor superclassDescriptor = xClass.getSuperclassDescriptor();
201             if (superclassDescriptor == null) {
202                 throw new CheckedAnalysisException();
203             }
204             xClass = cache.getClassAnalysis(XClass.class, superclassDescriptor);
205         }
206 
207     }

209     static public @Nonnull
210     IncompatibleTypes getPriorityForAssumingCompatible(ObjectType expectedType, ObjectType actualType, boolean pointerEquality) {
211         if (expectedType.equals(actualType)) {
212             return SEEMS_OK;
213         }
214 
215         if (actualType.equals(Type.OBJECT)) {
216             return IncompatibleTypes.UNCHECKED;
217         }
218 
219         if (expectedType.equals(Type.OBJECT)) {
220             return IncompatibleTypes.SEEMS_OK;
221         }
222 
223         try {
224 
225             if (!Hierarchy.isSubtype(expectedType, actualType) && !Hierarchy.isSubtype(actualType, expectedType)) {
226                 if (Hierarchy.isSubtype(expectedType, GWT_JAVASCRIPTOBJECT_TYPE)
227                         && Hierarchy.isSubtype(actualType, GWT_JAVASCRIPTOBJECT_TYPE)) {
228                     return SEEMS_OK;
229                 }
230                 // See if the types are related by inheritance.
231                 ClassDescriptor lhsDescriptor = DescriptorFactory.createClassDescriptorFromDottedClassName(expectedType
232                         .getClassName());
233                 ClassDescriptor rhsDescriptor = DescriptorFactory.createClassDescriptorFromDottedClassName(actualType
234                         .getClassName());
235 
236                 return getPriorityForAssumingCompatible(pointerEquality, lhsDescriptor, rhsDescriptor);
237             }
238 
239             if (expectedType instanceof GenericObjectType && actualType instanceof GenericObjectType
240                     && (Hierarchy.isSubtype(expectedType, COLLECTION_TYPE) || Hierarchy.isSubtype(expectedType, MAP_TYPE))) {
241                 List<? extends ReferenceType> lhsParameters = ((GenericObjectType) expectedType).getParameters();
242                 List<? extends ReferenceType> rhsParameters = ((GenericObjectType) actualType).getParameters();
243                 if (lhsParameters != null && rhsParameters != null && lhsParameters.size() == rhsParameters.size()) {
244                     for (int i = 0; i < lhsParameters.size(); i++) {
245                         IncompatibleTypes r = getPriorityForAssumingCompatible(lhsParameters.get(i), rhsParameters.get(i),
246                                 pointerEquality);
247                         if (r.getPriority() <= Priorities.NORMAL_PRIORITY) {
248                             return r;
249                         }
250                     }
251                 }
252 
253             }
254 
255         } catch (ClassNotFoundException e) {
256             AnalysisContext.reportMissingClass(e);
257         } catch (MissingClassException e) {
258             AnalysisContext.reportMissingClass(e.getClassNotFoundException());
259         } catch (CheckedAnalysisException e) {
260             AnalysisContext.logError("Error checking for incompatible types", e);
261         }
262         return SEEMS_OK;
263     }

265     public static IncompatibleTypes getPriorityForAssumingCompatible(boolean pointerEquality, ClassDescriptor lhsDescriptor,
266             ClassDescriptor rhsDescriptor) throws CheckedAnalysisException, ClassNotFoundException {
267         if (lhsDescriptor.equals(rhsDescriptor)) {
268             return SEEMS_OK;
269         }
270 
271         AnalysisContext analysisContext = AnalysisContext.currentAnalysisContext();
272         Subtypes2 subtypes2 = analysisContext.getSubtypes2();
273 
274         IAnalysisCache cache = Global.getAnalysisCache();
275         XClass lhs = cache.getClassAnalysis(XClass.class, lhsDescriptor);
276         XClass rhs = cache.getClassAnalysis(XClass.class, rhsDescriptor);
277         // Look up the classes
278         XMethod lhsEquals = getInvokedMethod(lhs, "equals", "(Ljava/lang/Object;)Z", false);
279         XMethod rhsEquals = getInvokedMethod(rhs, "equals", "(Ljava/lang/Object;)Z", false);
280         String lhsClassName = lhsEquals.getClassName();
281         if (lhsEquals.equals(rhsEquals)) {
282             if ("java.lang.Enum".equals(lhsClassName)) {
283                 return INCOMPATIBLE_CLASSES;
284             }
285             if (!pointerEquality && !"java.lang.Object".equals(lhsClassName)) {
286                 return SEEMS_OK;
287             }
288         }
289 
290         if ((subtypes2.isSubtype(lhsDescriptor, SET_DESCRIPTOR) && subtypes2.isSubtype(rhsDescriptor, SET_DESCRIPTOR)
291                 || subtypes2.isSubtype(lhsDescriptor, MAP_DESCRIPTOR) && subtypes2.isSubtype(rhsDescriptor, MAP_DESCRIPTOR) || subtypes2
292                 .isSubtype(lhsDescriptor, LIST_DESCRIPTOR) && subtypes2.isSubtype(rhsDescriptor, LIST_DESCRIPTOR))) {
293             return SEEMS_OK;
294         }
295 
296         if (!lhs.isInterface() && !rhs.isInterface()) {
297             // Both are class types, and therefore there is no possible
298             // way
299             // the compared objects can have the same runtime type.
300             return INCOMPATIBLE_CLASSES;
301         } else {
302 
303             // Look up the common subtypes of the two types. If the
304             // intersection does not contain at least one
305             // class,
306             // then issue a warning of the appropriate type.
307             Set<ClassDescriptor> commonSubtypes = subtypes2.getTransitiveCommonSubtypes(lhsDescriptor, rhsDescriptor);
308 
309             if (commonSubtypes.isEmpty()) {
310                 if (lhs.isInterface() && rhs.isInterface()) {
311                     if (!subtypes2.hasKnownSubclasses(lhsDescriptor) || !subtypes2.hasKnownSubclasses(rhsDescriptor)) {
312                         return UNRELATED_INTERFACES_WITHOUT_IMPLEMENTATIONS;
313                     }
314                     return UNRELATED_INTERFACES;
315                 }
316                 if (lhs.isFinal() || rhs.isFinal()) {
317                     return UNRELATED_FINAL_CLASS_AND_INTERFACE;
318                 }
319                 if (lhsDescriptor.getClassName().startsWith("java/util/")
320                         || rhsDescriptor.getClassName().startsWith("java/util/")) {
321                     return UNRELATED_UTIL_INTERFACE;
322                 }
323                 return UNRELATED_CLASS_AND_INTERFACE;
324             }
325 
326         }
327         return SEEMS_OK;
328     }
