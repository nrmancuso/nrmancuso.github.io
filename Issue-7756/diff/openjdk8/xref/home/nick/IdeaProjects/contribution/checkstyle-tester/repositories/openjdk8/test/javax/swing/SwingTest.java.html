24  import sun.awt.SunToolkit;
25  import java.awt.Toolkit;
26  import java.lang.reflect.InvocationTargetException;
27  import java.lang.reflect.Method;
28  import java.lang.reflect.Modifier;
29  import java.util.Comparator;
30  import java.util.Iterator;
31  import java.util.Set;
32  import java.util.TreeSet;
33  import javax.swing.JFrame;
34  import javax.swing.SwingUtilities;
35  
36  /**
37   * SwingTest is a utility class for writing regression tests
38   * that require interacting with the UI.
39   * It uses reflection to invoke all public methods without parameters.
40   * All static methods are starting on the current thread.
41   * Other methods including constructor are starting on the EDT.
42   * Between each method invocation all pending events are processed.
43   * The methods are sorted by name and invoked in that order.
44   * Failure of the test is signaled by any method throwing an exception.
45   * If no methods throw an exception the test is assumed to have passed.
46   *
47   * @author Sergey A. Malenkov
48   */
49  final class SwingTest implements Runnable {
50  
51      private static final int WIDTH = 640;
52      private static final int HEIGHT = 480;
53  
54      public static void start(Class<?> type) throws Throwable {
55          new SwingTest(type).start();
56      }
57  
58      private final Class<?> type;
59      private final Iterator<Method> methods;
60  
61      private JFrame frame;
62      private Object object;
63      private Method method;
64      private Throwable error;
65  
66      private SwingTest(Class<?> type) {
67          Set<Method> methods = new TreeSet<Method>(new Comparator<Method>() {
68              public int compare(Method first, Method second) {
69                  return first.getName().compareTo(second.getName());
70              }
71          });
72          for (Method method : type.getMethods()) {
73              if (method.getDeclaringClass().equals(type)) {
74                  if (method.getReturnType().equals(void.class)) {
75                      if (0 == method.getParameterTypes().length) {
76                          methods.add(method);
77                      }
78                  }
79              }
80          }
81          this.type = type;
82          this.methods = methods.iterator();
83      }
84  
85      public void run() {
86          try {
87              if (this.object == null) {
88                  System.out.println(this.type);
89                  this.frame = new JFrame(this.type.getSimpleName());
90                  this.frame.setSize(WIDTH, HEIGHT);
91                  this.frame.setLocationRelativeTo(null);
92                  this.object = this.type.getConstructor(this.frame.getClass()).newInstance(this.frame);
93                  this.frame.setVisible(true);
94              }
95              else if (this.method != null) {
96                  System.out.println(this.method);
97                  this.method.invoke(this.object);
98              }
99              else {
100                 System.out.println((this.error == null) ? "PASSED" : "FAILED"); // NON-NLS: debug
101                 this.frame.dispose();
102                 this.frame = null;
103             }
104         }
105         catch (NoSuchMethodException exception) {
106             this.error = exception;
107         }
108         catch (SecurityException exception) {
109             this.error = exception;
110         }
111         catch (IllegalAccessException exception) {
112             this.error = exception;
113         }
114         catch (IllegalArgumentException exception) {
115             this.error = exception;
116         }
117         catch (InstantiationException exception) {
118             this.error = exception;
119         }
120         catch (InvocationTargetException exception) {
121             this.error = exception.getTargetException();
122         }
123         System.out.flush();
124         this.method = this.methods.hasNext() && (this.error == null)
125                 ? this.methods.next()
126                 : null;
127     }
128 
129     private void start() throws Throwable {
130         do {
131             if ((this.method != null) && Modifier.isStatic(this.method.getModifiers())) {
132                 run(); // invoke static method on the current thread
133             }
134             else {
135                 SwingUtilities.invokeLater(this); // invoke on the event dispatch thread
136             }
137             Toolkit tk = Toolkit.getDefaultToolkit();
138             if (tk instanceof SunToolkit) {
139                 SunToolkit stk = (SunToolkit) tk;
140                 stk.realSync(); // wait until done
141             }
142         }
143         while (this.frame != null);
144         if (this.error != null) {
145             throw this.error;
146         }
147     }
148 }
