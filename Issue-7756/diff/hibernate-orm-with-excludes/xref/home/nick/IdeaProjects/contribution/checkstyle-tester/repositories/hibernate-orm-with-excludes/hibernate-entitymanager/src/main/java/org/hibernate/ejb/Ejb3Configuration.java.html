118  *
119  * A Ejb3Configuration object is only guaranteed to create one EntityManagerFactory.
120  * Multiple usage of {@link #buildEntityManagerFactory()} is not guaranteed.
121  *
122  * After #buildEntityManagerFactory() has been called, you no longer can change the configuration
123  * state (no class adding, no property change etc)
124  *
125  * When serialized / deserialized or retrieved from the JNDI, you no longer can change the

128  * Putting the configuration in the JNDI is an expensive operation that requires a partial
129  * serialization

135  * Manager Factory</i> based on runtime environment.  Additionally this class will be removed in Hibernate release
136  * 5.0 for the same reasoning outlined on {@link Configuration} due to move towards new
137  * {@link org.hibernate.SessionFactory} building methodology.  See
138  * <a href="http://opensource.atlassian.com/projects/hibernate/browse/HHH-6181">HHH-6181</a> and
139  * <a href="http://opensource.atlassian.com/projects/hibernate/browse/HHH-6159">HHH-6159</a> for details

194 	 * create a factory from a parsed persistence.xml
195 	 * Especially the scanning of classes and additional jars is done already at this point.
196 	 * <p/>
197 	 * NOTE: public only for unit testing purposes; not a public API!
198 	 *
199 	 * @param metadata The information parsed from the persistence.xml
200 	 * @param overridesIn Any explicitly passed config settings
201 	 *
202 	 * @return this
203 	 */
204 	@SuppressWarnings({ "unchecked" })
205 	public Ejb3Configuration configure(PersistenceMetadata metadata, Map overridesIn) {
206         LOG.debugf("Creating Factory: %s", metadata.getName());
207 
208 		Map overrides = new HashMap();
209 		if ( overridesIn != null ) {
210 			overrides.putAll( overridesIn );
211 		}
212 
213 		Map workingVars = new HashMap();
214 		workingVars.put( AvailableSettings.PERSISTENCE_UNIT_NAME, metadata.getName() );
215 		this.persistenceUnitName = metadata.getName();
216 
217 		if ( StringHelper.isNotEmpty( metadata.getJtaDatasource() ) ) {
218 			this.setProperty( Environment.DATASOURCE, metadata.getJtaDatasource() );
219 		}
220 		else if ( StringHelper.isNotEmpty( metadata.getNonJtaDatasource() ) ) {
221 			this.setProperty( Environment.DATASOURCE, metadata.getNonJtaDatasource() );
222 		}
223 		else {
224 			final String driver = (String) metadata.getProps().get( AvailableSettings.JDBC_DRIVER );
225 			if ( StringHelper.isNotEmpty( driver ) ) {
226 				this.setProperty( Environment.DRIVER, driver );
227 			}
228 			final String url = (String) metadata.getProps().get( AvailableSettings.JDBC_URL );
229 			if ( StringHelper.isNotEmpty( url ) ) {
230 				this.setProperty( Environment.URL, url );
231 			}
232 			final String user = (String) metadata.getProps().get( AvailableSettings.JDBC_USER );
233 			if ( StringHelper.isNotEmpty( user ) ) {
234 				this.setProperty( Environment.USER, user );
235 			}
236 			final String pass = (String) metadata.getProps().get( AvailableSettings.JDBC_PASSWORD );
237 			if ( StringHelper.isNotEmpty( pass ) ) {
238 				this.setProperty( Environment.PASS, pass );
239 			}
240 		}
241 		defineTransactionType( metadata.getTransactionType(), workingVars );
242 		if ( metadata.getClasses().size() > 0 ) {
243 			workingVars.put( AvailableSettings.CLASS_NAMES, metadata.getClasses() );
244 		}
245 		if ( metadata.getPackages().size() > 0 ) {
246 			workingVars.put( AvailableSettings.PACKAGE_NAMES, metadata.getPackages() );
247 		}
248 		if ( metadata.getMappingFiles().size() > 0 ) {
249 			workingVars.put( AvailableSettings.XML_FILE_NAMES, metadata.getMappingFiles() );
250 		}
251 		if ( metadata.getHbmfiles().size() > 0 ) {
252 			workingVars.put( AvailableSettings.HBXML_FILES, metadata.getHbmfiles() );
253 		}
254 
255 		Properties props = new Properties();
256 		props.putAll( metadata.getProps() );
257 
258 		// validation factory
259 		final Object validationFactory = overrides.get( AvailableSettings.VALIDATION_FACTORY );
260 		if ( validationFactory != null ) {
261 			BeanValidationIntegrator.validateFactory( validationFactory );
262 			props.put( AvailableSettings.VALIDATION_FACTORY, validationFactory );
263 		}
264 		overrides.remove( AvailableSettings.VALIDATION_FACTORY );
265 
266 		// validation-mode (overrides has precedence)
267 		{
268 			final Object integrationValue = overrides.get( AvailableSettings.VALIDATION_MODE );
269 			if ( integrationValue != null ) {
270 				props.put( AvailableSettings.VALIDATION_MODE, integrationValue.toString() );
271 			}
272 			else if ( metadata.getValidationMode() != null ) {
273 				props.put( AvailableSettings.VALIDATION_MODE, metadata.getValidationMode() );
274 			}
275 			overrides.remove( AvailableSettings.VALIDATION_MODE );
276 		}
277 
278 		// shared-cache-mode (overrides has precedence)
279 		{
280 			final Object integrationValue = overrides.get( AvailableSettings.SHARED_CACHE_MODE );
281 			if ( integrationValue != null ) {
282 				props.put( AvailableSettings.SHARED_CACHE_MODE, integrationValue.toString() );
283 			}
284 			else if ( metadata.getSharedCacheMode() != null ) {
285 				props.put( AvailableSettings.SHARED_CACHE_MODE, metadata.getSharedCacheMode() );
286 			}
287 			overrides.remove( AvailableSettings.SHARED_CACHE_MODE );
288 		}
289 
290 		for ( Map.Entry entry : (Set<Map.Entry>) overrides.entrySet() ) {
291 			Object value = entry.getValue();
292 			props.put( entry.getKey(), value == null ? "" :  value ); //alter null, not allowed in properties
293 		}
294 
295 		configure( props, workingVars );
296 		return this;
297 	}

299 	/**
300 	 * Build the configuration from an entity manager name and given the
301 	 * appropriate extra properties. Those properties override the one get through
302 	 * the persistence.xml file.
303 	 * If the persistence unit name is not found or does not match the Persistence Provider, null is returned
304 	 *
305 	 * This method is used in a non managed environment
306 	 *
307 	 * @param persistenceUnitName persistence unit name
308 	 * @param integration properties passed to the persistence provider
309 	 *
310 	 * @return configured Ejb3Configuration or null if no persistence unit match
311 	 *
312 	 * @see HibernatePersistence#createEntityManagerFactory(String, java.util.Map)
313 	 */
314 	@SuppressWarnings({ "unchecked" })
315 	public Ejb3Configuration configure(String persistenceUnitName, Map integration) {
316 		try {
317             LOG.debugf("Look up for persistence unit: %s", persistenceUnitName);
318 			integration = integration == null ?
319 					CollectionHelper.EMPTY_MAP :
320 					Collections.unmodifiableMap( integration );
321 			Enumeration<URL> xmls = ClassLoaderHelper.getContextClassLoader()
322 					.getResources( "META-INF/persistence.xml" );
323             if (!xmls.hasMoreElements()) LOG.unableToFindPersistenceXmlInClasspath();
324 			while ( xmls.hasMoreElements() ) {
325 				URL url = xmls.nextElement();
326                 LOG.trace("Analyzing persistence.xml: " + url);
327 				List<PersistenceMetadata> metadataFiles = PersistenceXmlLoader.deploy(
328 						url,
329 						integration,
330 						cfg.getEntityResolver(),
331 						PersistenceUnitTransactionType.RESOURCE_LOCAL );
332 				for ( PersistenceMetadata metadata : metadataFiles ) {
333                     LOG.trace(metadata);
334 
335 					if ( metadata.getProvider() == null || IMPLEMENTATION_NAME.equalsIgnoreCase(
336 							metadata.getProvider()
337 					) ) {
338 						//correct provider
339 
340 						//lazy load the scanner to avoid unnecessary IOExceptions
341 						Scanner scanner = null;
342 						URL jarURL = null;
343 						if ( metadata.getName() == null ) {
344 							scanner = buildScanner( metadata.getProps(), integration );
345 							jarURL = JarVisitorFactory.getJarURLFromURLEntry( url, "/META-INF/persistence.xml" );
346 							metadata.setName( scanner.getUnqualifiedJarName(jarURL) );
347 						}
348 						if ( persistenceUnitName == null && xmls.hasMoreElements() ) {
349 							throw new PersistenceException( "No name provided and several persistence units found" );
350 						}
351 						else if ( persistenceUnitName == null || metadata.getName().equals( persistenceUnitName ) ) {
352 							if (scanner == null) {
353 								scanner = buildScanner( metadata.getProps(), integration );
354 								jarURL = JarVisitorFactory.getJarURLFromURLEntry( url, "/META-INF/persistence.xml" );
355 							}
356 							//scan main JAR
357 							ScanningContext mainJarScanCtx = new ScanningContext()
358 									.scanner( scanner )
359 									.url( jarURL )
360 									.explicitMappingFiles( metadata.getMappingFiles() )
361 									.searchOrm( true );
362 							setDetectedArtifactsOnScanningContext( mainJarScanCtx, metadata.getProps(), integration,
363 																				metadata.getExcludeUnlistedClasses() );
364 							addMetadataFromScan( mainJarScanCtx, metadata );
365 
366 							ScanningContext otherJarScanCtx = new ScanningContext()
367 									.scanner( scanner )
368 									.explicitMappingFiles( metadata.getMappingFiles() )
369 									.searchOrm( true );
370 							setDetectedArtifactsOnScanningContext( otherJarScanCtx, metadata.getProps(), integration,
371 																				false );
372 							for ( String jarFile : metadata.getJarFiles() ) {
373 								otherJarScanCtx.url( JarVisitorFactory.getURLFromPath( jarFile ) );
374 								addMetadataFromScan( otherJarScanCtx, metadata );
375 							}
376 							return configure( metadata, integration );
377 						}
378 					}
379 				}
380 			}
381 			return null;
382 		}
383 		catch (Exception e) {
384 			if ( e instanceof PersistenceException) {
385 				throw (PersistenceException) e;
386 			}
387 			else {
388 				throw new PersistenceException( getExceptionHeader() + "Unable to configure EntityManagerFactory", e );
389 			}
390 		}
391 	}

393 	private Scanner buildScanner(Properties properties, Map<?,?> integration) {
394 		//read the String or Instance from the integration map first and use the properties as a backup.
395 		Object scanner = integration.get( AvailableSettings.SCANNER );
396 		if (scanner == null) {
397 			scanner = properties.getProperty( AvailableSettings.SCANNER );
398 		}
399 		if (scanner != null) {
400 			Class<?> scannerClass;
401 			if ( scanner instanceof String ) {
402 				try {
403 					scannerClass = ReflectHelper.classForName( (String) scanner, this.getClass() );
404 				}
405 				catch ( ClassNotFoundException e ) {
406 					throw new PersistenceException(  "Cannot find scanner class. " + AvailableSettings.SCANNER + "=" + scanner, e );
407 				}
408 			}
409 			else if (scanner instanceof Class) {
410 				scannerClass = (Class<? extends Scanner>) scanner;
411 			}
412 			else if (scanner instanceof Scanner) {
413 				return (Scanner) scanner;
414 			}
415 			else {
416 				throw new PersistenceException(  "Scanner class configuration error: unknown type on the property. " + AvailableSettings.SCANNER );
417 			}
418 			try {
419 				return (Scanner) scannerClass.newInstance();
420 			}
421 			catch ( InstantiationException e ) {
422 				throw new PersistenceException(  "Unable to load Scanner class: " + scannerClass, e );
423 			}
424 			catch ( IllegalAccessException e ) {
425 				throw new PersistenceException(  "Unable to load Scanner class: " + scannerClass, e );
426 			}
427 		}
428 		else {
429 			return new NativeScanner();
430 		}
431 	}

481 	private static void addScannedEntries(ScanningContext scanningContext, List<String> classes, List<String> packages, List<NamedInputStream> hbmFiles, List<String> mappingFiles) throws IOException {
482 		Scanner scanner = scanningContext.scanner;
483 		if (scanningContext.detectClasses) {
484 			Set<Class<? extends Annotation>> annotationsToExclude = new HashSet<Class<? extends Annotation>>(3);
485 			annotationsToExclude.add( Entity.class );
486 			annotationsToExclude.add( MappedSuperclass.class );
487 			annotationsToExclude.add( Embeddable.class );
488 			Set<Class<?>> matchingClasses = scanner.getClassesInJar( scanningContext.url, annotationsToExclude );
489 			for (Class<?> clazz : matchingClasses) {
490 				classes.add( clazz.getName() );
491 			}
492 
493 			Set<Package> matchingPackages = scanner.getPackagesInJar( scanningContext.url, new HashSet<Class<? extends Annotation>>(0) );
494 			for (Package pkg : matchingPackages) {
495 				packages.add( pkg.getName() );
496 			}
497 		}
498 		Set<String> patterns = new HashSet<String>();
499 		if (scanningContext.searchOrm) {
500 			patterns.add( META_INF_ORM_XML );
501 		}
502 		if (scanningContext.detectHbmFiles) {
503 			patterns.add( "**/*.hbm.xml" );
504 		}
505 		if ( mappingFiles != null) patterns.addAll( mappingFiles );
506 		if (patterns.size() !=0) {
507 			Set<NamedInputStream> files = scanner.getFilesInJar( scanningContext.url, patterns );
508 			for (NamedInputStream file : files) {
509 				hbmFiles.add( file );
510 				if (mappingFiles != null) mappingFiles.remove( file.getName() );
511 			}
512 		}
513 	}

515 	/**
516 	 * Process configuration from a PersistenceUnitInfo object; typically called by the container
517 	 * via {@link javax.persistence.spi.PersistenceProvider#createContainerEntityManagerFactory}.
518 	 * In Hibernate EM, this correlates to {@link HibernatePersistence#createContainerEntityManagerFactory}
519 	 *
520 	 * @param info The persistence unit info passed in by the container (usually from processing a persistence.xml).
521 	 * @param integration The map of integration properties from the container to configure the provider.
522 	 *
523 	 * @return this
524 	 *
525 	 * @see HibernatePersistence#createContainerEntityManagerFactory
526 	 */
527 	@SuppressWarnings({ "unchecked" })
528 	public Ejb3Configuration configure(PersistenceUnitInfo info, Map integration) {
529         if (LOG.isDebugEnabled()) LOG.debugf("Processing %s", LogHelper.logPersistenceUnitInfo(info));
530         else LOG.processingPersistenceUnitInfoName(info.getPersistenceUnitName());
531 
532 		// Spec says the passed map may be null, so handle that to make further processing easier...
533 		integration = integration != null ? Collections.unmodifiableMap( integration ) : CollectionHelper.EMPTY_MAP;
534 
535 		// See if we (Hibernate) are the persistence provider
536 		String provider = (String) integration.get( AvailableSettings.PROVIDER );
537 		if ( provider == null ) {
538 			provider = info.getPersistenceProviderClassName();
539 		}
540 		if ( provider != null && ! provider.trim().startsWith( IMPLEMENTATION_NAME ) ) {
541             LOG.requiredDifferentProvider(provider);
542 			return null;
543 		}
544 
545 		// set the classloader, passed in by the container in info, to set as the TCCL so that
546 		// Hibernate uses it to properly resolve class references.
547 		Thread thread = Thread.currentThread();
548 		ClassLoader contextClassLoader = thread.getContextClassLoader();
549 		boolean sameClassLoader = true;
550 		if (overridenClassLoader != null) {
551 			thread.setContextClassLoader( overridenClassLoader );
552 			sameClassLoader = false;
553 		}
554 		else {
555 			if ( info.getClassLoader() == null ) {
556 				throw new IllegalStateException(
557 						"[PersistenceUnit: " + info.getPersistenceUnitName() == null ? "" : info.getPersistenceUnitName()
558 								+ "] " + "PersistenceUnitInfo.getClassLoader() id null" );
559 			}
560 			sameClassLoader = info.getClassLoader().equals( contextClassLoader );
561 			if ( ! sameClassLoader ) {
562 				overridenClassLoader = info.getClassLoader();
563 				thread.setContextClassLoader( overridenClassLoader );
564 			}
565 			else {
566 				overridenClassLoader = null;
567 			}
568 		}

1202 				);

1204 		}
