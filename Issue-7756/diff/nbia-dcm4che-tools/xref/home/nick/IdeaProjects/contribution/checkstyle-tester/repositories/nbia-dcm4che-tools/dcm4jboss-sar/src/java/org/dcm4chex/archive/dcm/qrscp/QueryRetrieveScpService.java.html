497     boolean isHideWithoutIssuerOfPIDFromAET(String callingAET) {
498         if (hideWithoutIssuerOfPIDFromAETs != null) {
499             for (String aet : hideWithoutIssuerOfPIDFromAETs) {
500                 if (aet.equals(callingAET)) {
501                     return !invertHideWithoutIssuerOfPIDFromAETs;
502                 }
503             }
504         }
505         return invertHideWithoutIssuerOfPIDFromAETs;
506     }
