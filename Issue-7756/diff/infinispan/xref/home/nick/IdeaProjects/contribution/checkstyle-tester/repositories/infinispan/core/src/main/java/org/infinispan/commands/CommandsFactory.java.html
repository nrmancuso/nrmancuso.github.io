66  /**
67   * A factory to build commands, initializing and injecting dependencies accordingly.  Commands built for a specific,
68   * named cache instance cannot be reused on a different cache instance since most commands contain the cache name it
69   * was built for along with references to other named-cache scoped components.
70   *
71   * @author Manik Surtani
72   * @author Mircea.Markus@jboss.com
73   * @author Galder Zamarre√±o
74   * @since 4.0
75   */
76  @Scope(Scopes.NAMED_CACHE)
77  public interface CommandsFactory {
78  
79     /**
80      * Builds a PutKeyValueCommand
81      * @param key key to put
82      * @param value value to put
83      * @param metadata metadata of entry
84      * @param flags Command flags provided by cache
85      * @return a PutKeyValueCommand
86      */
87     PutKeyValueCommand buildPutKeyValueCommand(Object key, Object value, Metadata metadata, Set<Flag> flags);
88  
89     /**
90      * Builds a RemoveCommand
91      * @param key key to remove

93      * @param flags Command flags provided by cache

95      */
96     RemoveCommand buildRemoveCommand(Object key, Object value, Set<Flag> flags);
