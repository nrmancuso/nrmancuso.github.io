20  package edu.umd.cs.findbugs.util;
21  
22  import java.util.Map;
23  
24  /**
25   * @author pugh
26   */
27  public class ProfilingMapCache<K, V> extends MapCache<K, V> {
28  
29      final String name;
30  
31      public ProfilingMapCache(int maxCapacity, String name) {
32          super(maxCapacity);
33          this.name = name;
34          count = new int[maxCapacity];
35          Runtime.getRuntime().addShutdownHook(new Thread() {
36              @Override
37              public void run() {
38                  System.out.println("Profile for map cache " + ProfilingMapCache.this.name);
39                  for (int i = 0; i < count.length; i++) {
40                      System.out.printf("%4d %5d%n", i, count[i]);
41                  }
42              }
43          });
44      }
45  
46      int[] count;
47  
48      @Override
49      public V get(Object k) {
50  
51          int age = count.length - 1;
52          for (Map.Entry<K, V> e : entrySet()) {
53              if (e.getKey().equals(k)) {
54                  count[age]++;
55                  if (age > 20) {
56                      new RuntimeException("Reusing value from " + age + " steps ago in " + name).printStackTrace(System.out);
57                  }
58                  break;
59              }
60              age--;
61          }
62          return super.get(k);
63      }
64  
65      public String getStatistics() {
66          StringBuilder b = new StringBuilder();
67          for (int c : count) {
68              b.append(c).append(" ");
69          }
70          return b.toString();
71      }
72  }
