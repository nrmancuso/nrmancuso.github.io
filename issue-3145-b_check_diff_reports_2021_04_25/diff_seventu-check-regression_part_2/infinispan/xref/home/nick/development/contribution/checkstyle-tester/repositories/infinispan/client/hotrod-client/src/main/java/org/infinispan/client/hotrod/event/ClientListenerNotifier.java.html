39  public class ClientListenerNotifier {
40     private static final Log log = LogFactory.getLog(ClientListenerNotifier.class, Log.class);
41  
42     private static final Map<Class<? extends Annotation>, Class<?>[]> allowedListeners =
43           new HashMap<Class<? extends Annotation>, Class<?>[]>(4);
44  
45     static {
46        allowedListeners.put(ClientCacheEntryCreated.class, new Class[]{ClientCacheEntryCreatedEvent.class, ClientCacheEntryCustomEvent.class});
47        allowedListeners.put(ClientCacheEntryModified.class, new Class[]{ClientCacheEntryModifiedEvent.class, ClientCacheEntryCustomEvent.class});
48        allowedListeners.put(ClientCacheEntryRemoved.class, new Class[]{ClientCacheEntryRemovedEvent.class, ClientCacheEntryCustomEvent.class});
49        allowedListeners.put(ClientCacheFailover.class, new Class[]{ClientCacheFailoverEvent.class});
50     }
51  
52     private final ConcurrentMap<byte[], EventDispatcher> clientListeners = CollectionFactory.makeConcurrentMap(
53           ByteArrayEquivalence.INSTANCE, AnyEquivalence.getInstance());
54  
55     private final ExecutorService executor;
56     private final Codec codec;
57     private final Marshaller marshaller;

59     protected ClientListenerNotifier(ExecutorService executor, Codec codec, Marshaller marshaller) {
60        this.executor = executor;
61        this.codec = codec;
62        this.marshaller = marshaller;
63     }

65     public static ClientListenerNotifier create(Codec codec, Marshaller marshaller) {
66        return new ClientListenerNotifier(Executors.newCachedThreadPool(), codec, marshaller);
67     }

69     public Marshaller getMarshaller() {
70        return marshaller;
71     }

73     public void addClientListener(AddClientListenerOperation op) {
74        Map<Class<? extends Annotation>, List<ClientListenerInvocation>> invocables = findMethods(op.listener);
75        EventDispatcher eventDispatcher = new EventDispatcher(op, invocables);
76        clientListeners.put(op.listenerId, eventDispatcher);
77        if (log.isTraceEnabled())
78           log.tracef("Add client listener with id %s, for listener %s and invocable methods %s",
79                 Util.printArray(op.listenerId), op.listener, invocables);
80     }

82     public void failoverClientListeners(Set<SocketAddress> failedServers) {
83        // Compile all listener ids that need failing over
84        List<byte[]> failoverListenerIds = new ArrayList<>();
85        for (Map.Entry<byte[], EventDispatcher> entry : clientListeners.entrySet()) {
86           EventDispatcher dispatcher = entry.getValue();
87           if (failedServers.contains(dispatcher.transport.getRemoteSocketAddress()))
88              failoverListenerIds.add(entry.getKey());
89        }
90        if (log.isTraceEnabled() && failoverListenerIds.isEmpty())
91           log.tracef("No event listeners registered in faild servers: %s", failedServers);
92  
93        // Remove tracking listeners and read to the fallback transport
94        for (byte[] listenerId : failoverListenerIds) {
95           EventDispatcher dispatcher = clientListeners.get(listenerId);
96           removeClientListener(listenerId);
97           // Invoke failover event callback, if presents
98           invokeFailoverEvent(dispatcher);
99           // Re-execute adding client listener in one of the remaining nodes
100          dispatcher.op.execute();
101          if (log.isTraceEnabled()) {
102             SocketAddress failedServerAddress = dispatcher.transport.getRemoteSocketAddress();
103             log.tracef("Fallback listener id %s from a failed server %s to %s",
104                   Util.printArray(listenerId), failedServerAddress,
105                   dispatcher.op.getDedicatedTransport().getRemoteSocketAddress());
106          }
107       }
108    }
109 
110    private void invokeFailoverEvent(EventDispatcher dispatcher) {
111       List<ClientListenerInvocation> callbacks = dispatcher.invocables.get(ClientCacheFailover.class);
112       if (callbacks != null) {
113          for (ClientListenerInvocation callback : callbacks)
114             callback.invoke(ClientEvents.mkCachefailoverEvent());
115       }
116    }

118    public void startClientListener(byte[] listenerId) {
119       EventDispatcher eventDispatcher = clientListeners.get(listenerId);
120       executor.submit(eventDispatcher);
121    }
122 
123    public void removeClientListener(byte[] listenerId) {
124       EventDispatcher dispatcher = clientListeners.remove(listenerId);
125       dispatcher.transport.release(); // force shutting it
126       if (log.isTraceEnabled())
127          log.tracef("Remove client listener with id %s", Util.printArray(listenerId));
128    }

130    public byte[] findListenerId(Object listener) {
131       for (EventDispatcher dispatcher : clientListeners.values()) {
132          if (dispatcher.op.listener.equals(listener))
133             return dispatcher.op.listenerId;
134       }
135       return null;
136    }

138    public boolean isListenerConnected(byte[] listenerId) {
139       EventDispatcher dispatcher = clientListeners.get(listenerId);
140       // If listener not present, is not active
141       return dispatcher != null && !dispatcher.stopped;
142    }

144    public Transport findTransport(byte[] listenerId) {
145       EventDispatcher dispatcher = clientListeners.get(listenerId);
146       if (dispatcher != null)
147          return dispatcher.transport;
148 
149       return null;
150    }

152    private Map<Class<? extends Annotation>, List<ClientListenerInvocation>> findMethods(Object listener) {
153       Map<Class<? extends Annotation>, List<ClientListenerInvocation>> listenerMethodMap =
154             new HashMap<Class<? extends Annotation>, List<ClientListenerInvocation>>(4, 0.99f);
155 
156       for (Method m : listener.getClass().getMethods()) {
157          // loop through all valid method annotations
158          for (Map.Entry<Class<? extends Annotation>, Class<?>[]> entry : allowedListeners.entrySet()) {
159             Class<? extends Annotation> annotationType = entry.getKey();
160             Class<?>[] eventTypes = entry.getValue();
161             if (m.isAnnotationPresent(annotationType)) {
162                testListenerMethodValidity(m, eventTypes, annotationType.getName());
163                SecurityActions.setAccessible(m);
164                ClientListenerInvocation invocation = new ClientListenerInvocation(listener, m);
165                List<ClientListenerInvocation> invocables = listenerMethodMap.get(annotationType);
166                if (invocables == null) {
167                   invocables = new ArrayList<ClientListenerInvocation>();
168                   listenerMethodMap.put(annotationType, invocables);
169                }
170 
171                invocables.add(invocation);
172             }
173          }
174       }
175 
176       return listenerMethodMap;
177    }
178 
179    private void testListenerMethodValidity(Method m, Class<?>[] allowedParameters, String annotationName) {
180       boolean isAllowed = false;
181       for (Class<?> allowedParameter : allowedParameters) {
182          if (m.getParameterTypes().length == 1 && m.getParameterTypes()[0].isAssignableFrom(allowedParameter)) {
183             isAllowed = true;
184             break;
185          }
186       }
187 
188       if (!isAllowed)
189          throw log.incorrectClientListener(annotationName, Arrays.asList(allowedParameters));
190       if (!m.getReturnType().equals(void.class))
191          throw log.incorrectClientListener(annotationName);
192    }

194    public Set<Object> getListeners(byte[] cacheName) {
195       Set<Object> ret = new HashSet<>(clientListeners.size());
196       for (EventDispatcher dispatcher : clientListeners.values()) {
197          if (Arrays.equals(dispatcher.op.cacheName, cacheName))
198             ret.add(dispatcher.op.listener);
199       }
200 
201       return ret;
202    }

204    public void stop() {
205       for (byte[] listenerId : clientListeners.keySet()) {
206          if (log.isTraceEnabled())
207             log.tracef("Remote cache manager stopping, remove client listener id %s", Util.printArray(listenerId));
208 
209          removeClientListener(listenerId);
210       }
211    }

213    public void invokeEvent(byte[] listenerId, ClientEvent clientEvent) {
214       EventDispatcher eventDispatcher = clientListeners.get(listenerId);
215       eventDispatcher.invokeClientEvent(clientEvent);
216    }

218    private final class EventDispatcher implements Runnable {
219       final Map<Class<? extends Annotation>, List<ClientListenerInvocation>> invocables;
220       final AddClientListenerOperation op;
221       final Transport transport;
222       volatile boolean stopped = false;
223 
224       private EventDispatcher(AddClientListenerOperation op,
225             Map<Class<? extends Annotation>, List<ClientListenerInvocation>> invocables) {
226          this.op = op;
227          this.transport = op.getDedicatedTransport();
228          this.invocables = invocables;
229       }
230 
231       @Override
232       public void run() {
233          Thread.currentThread().setName("Client-Listener-" + Util.toHexString(op.listenerId, 8));
234          while (!Thread.currentThread().isInterrupted()) {
235             ClientEvent clientEvent = null;
236             try {
237                clientEvent = codec.readEvent(transport, op.listenerId, marshaller);
238                invokeClientEvent(clientEvent);
239                // Nullify event, makes it easier to identify network vs invocation error messages
240                clientEvent = null;
241             } catch (TransportException e) {
242                Throwable cause = e.getCause();
243                if (cause instanceof ClosedChannelException) {
244                   // Channel closed, ignore and exit
245                   log.debug("Channel closed, exiting event reader thread");
246                   stopped = true;
247                   return;
248                } else if (cause instanceof SocketTimeoutException) {
249                   log.debug("Timed out reading event, retry");
250                } else if (clientEvent != null) {
251                   log.unexpectedErrorConsumingEvent(clientEvent, e);
252                }  else {
253                   log.unrecoverableErrorReadingEvent(e, transport.getRemoteSocketAddress());
254                   stopped = true;
255                   return; // Server is likely gone!
256                }
257             } catch (CancelledKeyException e) {
258                // Cancelled key exceptions are also thrown when the channel has been closed
259                log.debug("Key cancelled, most likely channel closed, exiting event reader thread");
260                stopped = true;
261                return;
262             } catch (Throwable t) {
263                if (clientEvent != null)
264                   log.unexpectedErrorConsumingEvent(clientEvent, t);
265                else
266                   log.unableToReadEventFromServer(t, transport.getRemoteSocketAddress());
267                if (!transport.isValid()) {
268                   stopped = true;
269                   return;
270                }
271             }
272          }
273       }
274 
275       void invokeClientEvent(ClientEvent clientEvent) {
276          if (log.isTraceEnabled())
277             log.tracef("Event %s received for listener with id=%s", clientEvent, Util.printArray(op.listenerId));
278 
279          switch (clientEvent.getType()) {
280             case CLIENT_CACHE_ENTRY_CREATED:
281                invokeCallbacks(clientEvent, ClientCacheEntryCreated.class);
282                break;
283             case CLIENT_CACHE_ENTRY_MODIFIED:
284                invokeCallbacks(clientEvent, ClientCacheEntryModified.class);
285                break;
286             case CLIENT_CACHE_ENTRY_REMOVED:
287                invokeCallbacks(clientEvent, ClientCacheEntryRemoved.class);
288                break;
289          }
290       }
291 
292       private void invokeCallbacks(ClientEvent event, Class<? extends Annotation> type) {
293          List<ClientListenerInvocation> callbacks = invocables.get(type);
294          if (callbacks != null) {
295             for (ClientListenerInvocation callback : callbacks)
296                callback.invoke(event);
297          }
298       }
299    }
300 
301    private static final class ClientListenerInvocation {
302       private static final Log log = LogFactory.getLog(ClientListenerInvocation.class, Log.class);
303 
304       final Object listener;
305       final Method method;
306 
307       private ClientListenerInvocation(Object listener, Method method) {
308          this.listener = listener;
309          this.method = method;
310       }
311 
312       public void invoke(ClientEvent event) {
313          try {
314             method.invoke(listener, event);
315          } catch (Exception e) {
316             throw log.exceptionInvokingListener(
317                   e.getClass().getName(), method, listener, e);
318          }
319       }
320    }
