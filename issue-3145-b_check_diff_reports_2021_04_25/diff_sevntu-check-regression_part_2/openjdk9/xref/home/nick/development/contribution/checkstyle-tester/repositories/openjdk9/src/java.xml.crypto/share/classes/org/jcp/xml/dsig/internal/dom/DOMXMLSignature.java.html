89      /**
90       * Creates a <code>DOMXMLSignature</code> from the specified components.
91       *
92       * @param si the <code>SignedInfo</code>
93       * @param ki the <code>KeyInfo</code>, or <code>null</code> if not specified
94       * @param objs a list of <code>XMLObject</code>s or <code>null</code>
95       *  if not specified. The list is copied to protect against subsequent
96       *  modification.
97       * @param id an optional id (specify <code>null</code> to omit)
98       * @param signatureValueId an optional id (specify <code>null</code> to
99       *  omit)
100      * @throws NullPointerException if <code>si</code> is <code>null</code>
101      */
102     public DOMXMLSignature(SignedInfo si, KeyInfo ki,
103                            List<? extends XMLObject> objs,
104                            String id, String signatureValueId)
105     {
106         if (si == null) {
107             throw new NullPointerException("signedInfo cannot be null");
108         }
109         this.si = si;
110         this.id = id;
111         this.sv = new DOMSignatureValue(signatureValueId);
112         List<XMLObject> tempList =
113             Collections.checkedList(new ArrayList<XMLObject>(),
114                                     XMLObject.class);
115         if (objs != null) {
116             tempList.addAll(objs);
117         }
118         this.objects = Collections.unmodifiableList(tempList);
119         this.ki = ki;
120     }

122     /**
123      * Creates a <code>DOMXMLSignature</code> from XML.
124      *
125      * @param sigElem Signature element
126      * @throws MarshalException if XMLSignature cannot be unmarshalled
127      */
128     public DOMXMLSignature(Element sigElem, XMLCryptoContext context,
129                            Provider provider)
130         throws MarshalException
131     {
132         localSigElem = sigElem;
133         ownerDoc = localSigElem.getOwnerDocument();
134 
135         // get Id attribute, if specified
136         id = DOMUtils.getAttributeValue(localSigElem, "Id");
137 
138         // unmarshal SignedInfo
139         Element siElem = DOMUtils.getFirstChildElement(localSigElem,
140                                                        "SignedInfo");
141         si = new DOMSignedInfo(siElem, context, provider);
142 
143         // unmarshal SignatureValue
144         Element sigValElem = DOMUtils.getNextSiblingElement(siElem,
145                                                             "SignatureValue");
146         sv = new DOMSignatureValue(sigValElem, context);
147 
148         // unmarshal KeyInfo, if specified
149         Element nextSibling = DOMUtils.getNextSiblingElement(sigValElem);
150         if (nextSibling != null && nextSibling.getLocalName().equals("KeyInfo")) {
151             ki = new DOMKeyInfo(nextSibling, context, provider);
152             nextSibling = DOMUtils.getNextSiblingElement(nextSibling);
153         }
154 
155         // unmarshal Objects, if specified
156         if (nextSibling == null) {
157             objects = Collections.emptyList();
158         } else {
159             List<XMLObject> tempObjects = new ArrayList<XMLObject>();
160             while (nextSibling != null) {
161                 String name = nextSibling.getLocalName();
162                 if (!name.equals("Object")) {
163                     throw new MarshalException("Invalid element name: " + name +
164                                                ", expected KeyInfo or Object");
165                 }
166                 tempObjects.add(new DOMXMLObject(nextSibling,
167                                                  context, provider));
168                 nextSibling = DOMUtils.getNextSiblingElement(nextSibling);
169             }
170             objects = Collections.unmodifiableList(tempObjects);
171         }
172     }

204     public void marshal(Node parent, Node nextSibling, String dsPrefix,
205                         DOMCryptoContext context)
206         throws MarshalException
207     {
208         ownerDoc = DOMUtils.getOwnerDocument(parent);
209         sigElem = DOMUtils.createElement(ownerDoc, "Signature",
210                                          XMLSignature.XMLNS, dsPrefix);
211 
212         // append xmlns attribute
213         if (dsPrefix == null || dsPrefix.length() == 0) {
214             sigElem.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns",
215                                    XMLSignature.XMLNS);
216         } else {
217             sigElem.setAttributeNS("http://www.w3.org/2000/xmlns/", "xmlns:" +
218                                    dsPrefix, XMLSignature.XMLNS);
219         }
220 
221         // create and append SignedInfo element
222         ((DOMSignedInfo)si).marshal(sigElem, dsPrefix, context);
223 
224         // create and append SignatureValue element
225         ((DOMSignatureValue)sv).marshal(sigElem, dsPrefix, context);
226 
227         // create and append KeyInfo element if necessary
228         if (ki != null) {
229             ((DOMKeyInfo)ki).marshal(sigElem, null, dsPrefix, context);
230         }
231 
232         // create and append Object elements if necessary
233         for (int i = 0, size = objects.size(); i < size; i++) {
234             ((DOMXMLObject)objects.get(i)).marshal(sigElem, dsPrefix, context);
235         }
236 
237         // append Id attribute
238         DOMUtils.setAttributeID(sigElem, "Id", id);
239 
240         parent.insertBefore(sigElem, nextSibling);
241     }

243     public boolean validate(XMLValidateContext vc)
244         throws XMLSignatureException
245     {
246         if (vc == null) {
247             throw new NullPointerException("validateContext is null");
248         }
249 
250         if (!(vc instanceof DOMValidateContext)) {
251             throw new ClassCastException
252                 ("validateContext must be of type DOMValidateContext");
253         }
254 
255         if (validated) {
256             return validationStatus;
257         }
258 
259         // validate the signature
260         boolean sigValidity = sv.validate(vc);
261         if (!sigValidity) {
262             validationStatus = false;
263             validated = true;
264             return validationStatus;
265         }
266 
267         // validate all References
268         List<Reference> refs = this.si.getReferences();
269         boolean validateRefs = true;
270         for (int i = 0, size = refs.size(); validateRefs && i < size; i++) {
271             Reference ref = refs.get(i);
272             boolean refValid = ref.validate(vc);
273             if (log.isLoggable(java.util.logging.Level.FINE)) {
274                 log.log(java.util.logging.Level.FINE, "Reference[" + ref.getURI() + "] is valid: " + refValid);
275             }
276             validateRefs &= refValid;
277         }
278         if (!validateRefs) {
279             if (log.isLoggable(java.util.logging.Level.FINE)) {
280                 log.log(java.util.logging.Level.FINE, "Couldn't validate the References");
281             }
282             validationStatus = false;
283             validated = true;
284             return validationStatus;
285         }
286 
287         // validate Manifests, if property set
288         boolean validateMans = true;
289         if (Boolean.TRUE.equals(vc.getProperty
290                                 ("org.jcp.xml.dsig.validateManifests")))
291         {
292             for (int i=0, size=objects.size(); validateMans && i < size; i++) {
293                 XMLObject xo = objects.get(i);
294                 List<XMLStructure> content = xo.getContent();
295                 int csize = content.size();
296                 for (int j = 0; validateMans && j < csize; j++) {
297                     XMLStructure xs = content.get(j);
298                     if (xs instanceof Manifest) {
299                         if (log.isLoggable(java.util.logging.Level.FINE)) {
300                             log.log(java.util.logging.Level.FINE, "validating manifest");
301                         }
302                         Manifest man = (Manifest)xs;
303                         List<Reference> manRefs = man.getReferences();
304                         int rsize = manRefs.size();
305                         for (int k = 0; validateMans && k < rsize; k++) {
306                             Reference ref = manRefs.get(k);
307                             boolean refValid = ref.validate(vc);
308                             if (log.isLoggable(java.util.logging.Level.FINE)) {
309                                 log.log(java.util.logging.Level.FINE,
310                                     "Manifest ref[" + ref.getURI() + "] is valid: " + refValid
311                                 );
312                             }
313                             validateMans &= refValid;
314                         }
315                     }
316                 }
317             }
318         }
319 
320         validationStatus = validateMans;
321         validated = true;
322         return validationStatus;
323     }

325     public void sign(XMLSignContext signContext)
326         throws MarshalException, XMLSignatureException
327     {
328         if (signContext == null) {
329             throw new NullPointerException("signContext cannot be null");
330         }
331         DOMSignContext context = (DOMSignContext)signContext;
332         marshal(context.getParent(), context.getNextSibling(),
333                 DOMUtils.getSignaturePrefix(context), context);
334 
335         // generate references and signature value
336         List<Reference> allReferences = new ArrayList<Reference>();
337 
338         // traverse the Signature and register all objects with IDs that
339         // may contain References
340         signatureIdMap = new HashMap<String, XMLStructure>();
341         signatureIdMap.put(id, this);
342         signatureIdMap.put(si.getId(), si);
343         List<Reference> refs = si.getReferences();
344         for (Reference ref : refs) {
345             signatureIdMap.put(ref.getId(), ref);
346         }
347         for (XMLObject obj : objects) {
348             signatureIdMap.put(obj.getId(), obj);
349             List<XMLStructure> content = obj.getContent();
350             for (XMLStructure xs : content) {
351                 if (xs instanceof Manifest) {
352                     Manifest man = (Manifest)xs;
353                     signatureIdMap.put(man.getId(), man);
354                     List<Reference> manRefs = man.getReferences();
355                     for (Reference ref : manRefs) {
356                         allReferences.add(ref);
357                         signatureIdMap.put(ref.getId(), ref);
358                     }
359                 }
360             }
361         }
362         // always add SignedInfo references after Manifest references so
363         // that Manifest reference are digested first
364         allReferences.addAll(refs);
365 
366         // generate/digest each reference
367         for (Reference ref : allReferences) {
368             digestReference((DOMReference)ref, signContext);
369         }
370 
371         // do final sweep to digest any references that were skipped or missed
372         for (Reference ref : allReferences) {
373             if (((DOMReference)ref).isDigested()) {
374                 continue;
375             }
376             ((DOMReference)ref).digest(signContext);
377         }
378 
379         Key signingKey = null;
380         KeySelectorResult ksr = null;
381         try {
382             ksr = signContext.getKeySelector().select(ki,
383                                                       KeySelector.Purpose.SIGN,
384                                                       si.getSignatureMethod(),
385                                                       signContext);
386             signingKey = ksr.getKey();
387             if (signingKey == null) {
388                 throw new XMLSignatureException("the keySelector did not " +
389                                                 "find a signing key");
390             }
391         } catch (KeySelectorException kse) {
392             throw new XMLSignatureException("cannot find signing key", kse);
393         }
394 
395         // calculate signature value
396         try {
397             byte[] val = ((AbstractDOMSignatureMethod)
398                 si.getSignatureMethod()).sign(signingKey, si, signContext);
399             ((DOMSignatureValue)sv).setValue(val);
400         } catch (InvalidKeyException ike) {
401             throw new XMLSignatureException(ike);
402         }
403 
404         this.localSigElem = sigElem;
405         this.ksr = ksr;
406     }

408     @Override
409     public boolean equals(Object o) {
410         if (this == o) {
411             return true;
412         }
413 
414         if (!(o instanceof XMLSignature)) {
415             return false;
416         }
417         XMLSignature osig = (XMLSignature)o;
418 
419         boolean idEqual =
420             (id == null ? osig.getId() == null : id.equals(osig.getId()));
421         boolean keyInfoEqual =
422             (ki == null ? osig.getKeyInfo() == null
423                         : ki.equals(osig.getKeyInfo()));
424 
425         return (idEqual && keyInfoEqual &&
426                 sv.equals(osig.getSignatureValue()) &&
427                 si.equals(osig.getSignedInfo()) &&
428                 objects.equals(osig.getObjects()));
429     }

431     @Override
432     public int hashCode() {
433         int result = 17;
434         if (id != null) {
435             result = 31 * result + id.hashCode();
436         }
437         if (ki != null) {
438             result = 31 * result + ki.hashCode();
439         }
440         result = 31 * result + sv.hashCode();
441         result = 31 * result + si.hashCode();
442         result = 31 * result + objects.hashCode();
443 
444         return result;
445     }

447     private void digestReference(DOMReference ref, XMLSignContext signContext)
448         throws XMLSignatureException
449     {
450         if (ref.isDigested()) {
451             return;
452         }
453         // check dependencies
454         String uri = ref.getURI();
455         if (Utils.sameDocumentURI(uri)) {
456             String id = Utils.parseIdFromSameDocumentURI(uri);
457             if (id != null && signatureIdMap.containsKey(id)) {
458                 XMLStructure xs = signatureIdMap.get(id);
459                 if (xs instanceof DOMReference) {
460                     digestReference((DOMReference)xs, signContext);
461                 } else if (xs instanceof Manifest) {
462                     Manifest man = (Manifest)xs;
463                     List<Reference> manRefs =
464                         DOMManifest.getManifestReferences(man);
465                     for (int i = 0, size = manRefs.size(); i < size; i++) {
466                         digestReference((DOMReference)manRefs.get(i),
467                                         signContext);
468                     }
469                 }
470             }
471             // if uri="" and there are XPath Transforms, there may be
472             // reference dependencies in the XPath Transform - so be on
473             // the safe side, and skip and do at end in the final sweep
474             if (uri.length() == 0) {
475                 List<Transform> transforms = ref.getTransforms();
476                 for (Transform transform : transforms) {
477                     String transformAlg = transform.getAlgorithm();
478                     if (transformAlg.equals(Transform.XPATH) ||
479                         transformAlg.equals(Transform.XPATH2)) {
480                         return;
481                     }
482                 }
483             }
484         }
485         ref.digest(signContext);
486     }

488     public class DOMSignatureValue extends DOMStructure
489         implements SignatureValue
490     {
491         private String id;
492         private byte[] value;
493         private String valueBase64;
494         private Element sigValueElem;
495         private boolean validated = false;
496         private boolean validationStatus;
497 
498         DOMSignatureValue(String id) {
499             this.id = id;
500         }
501 
502         DOMSignatureValue(Element sigValueElem, XMLCryptoContext context)
503             throws MarshalException
504         {
505             try {
506                 // base64 decode signatureValue
507                 value = Base64.decode(sigValueElem);
508             } catch (Base64DecodingException bde) {
509                 throw new MarshalException(bde);
510             }
511 
512             Attr attr = sigValueElem.getAttributeNodeNS(null, "Id");
513             if (attr != null) {
514                 id = attr.getValue();
515                 sigValueElem.setIdAttributeNode(attr, true);
516             } else {
517                 id = null;
518             }
519             this.sigValueElem = sigValueElem;
520         }
521 
522         public String getId() {
523             return id;
524         }
525 
526         public byte[] getValue() {
527             return (value == null) ? null : value.clone();
528         }
529 
530         public boolean validate(XMLValidateContext validateContext)
531             throws XMLSignatureException
532         {
533             if (validateContext == null) {
534                 throw new NullPointerException("context cannot be null");
535             }
536 
537             if (validated) {
538                 return validationStatus;
539             }
540 
541             // get validating key
542             SignatureMethod sm = si.getSignatureMethod();
543             Key validationKey = null;
544             KeySelectorResult ksResult;
545             try {
546                 ksResult = validateContext.getKeySelector().select
547                     (ki, KeySelector.Purpose.VERIFY, sm, validateContext);
548                 validationKey = ksResult.getKey();
549                 if (validationKey == null) {
550                     throw new XMLSignatureException("the keyselector did not " +
551                                                     "find a validation key");
552                 }
553             } catch (KeySelectorException kse) {
554                 throw new XMLSignatureException("cannot find validation " +
555                                                 "key", kse);
556             }
557 
558             // canonicalize SignedInfo and verify signature
559             try {
560                 validationStatus = ((AbstractDOMSignatureMethod)sm).verify
561                     (validationKey, si, value, validateContext);
562             } catch (Exception e) {
563                 throw new XMLSignatureException(e);
564             }
565 
566             validated = true;
567             ksr = ksResult;
568             return validationStatus;
569         }
570 
571         @Override
572         public boolean equals(Object o) {
573             if (this == o) {
574                 return true;
575             }
576 
577             if (!(o instanceof SignatureValue)) {
578                 return false;
579             }
580             SignatureValue osv = (SignatureValue)o;
581 
582             boolean idEqual =
583                 (id == null ? osv.getId() == null : id.equals(osv.getId()));
584 
585             //XXX compare signature values?
586             return idEqual;
587         }
588 
589         @Override
590         public int hashCode() {
591             int result = 17;
592             if (id != null) {
593                 result = 31 * result + id.hashCode();
594             }
595 
596             return result;
597         }

599         public void marshal(Node parent, String dsPrefix,
600                             DOMCryptoContext context)
601             throws MarshalException
602         {
603             // create SignatureValue element
604             sigValueElem = DOMUtils.createElement(ownerDoc, "SignatureValue",
605                                                   XMLSignature.XMLNS, dsPrefix);
606             if (valueBase64 != null) {
607                 sigValueElem.appendChild(ownerDoc.createTextNode(valueBase64));
608             }
609 
610             // append Id attribute, if specified
611             DOMUtils.setAttributeID(sigValueElem, "Id", id);
612             parent.appendChild(sigValueElem);
613         }
614 
615         void setValue(byte[] value) {
616             this.value = value;
617             valueBase64 = Base64.encode(value);
618             sigValueElem.appendChild(ownerDoc.createTextNode(valueBase64));
619         }
620     }
