40  import java.io.*;
41  import java.security.Permission;
42  import javax.net.ssl.*;
43  import java.security.Principal;
44  import java.security.Security;

54  public class SSL extends SecurityManager {
55  
56      private static String krb5Cipher;
57      private static final int LOOP_LIMIT = 3;
58      private static int loopCount = 0;
59      private static volatile String server;
60      private static volatile int port;
61      private static String sniHostname = null;
62      private static String sniMatcherPattern = null;
63  
64      private static String permChecks = "";
65  
66      // 0-Not started, 1-Start OK, 2-Failure
67      private static volatile int serverState = 0;
68  
69      @Override
70      public void checkPermission(Permission perm, Object context) {
71          checkPermission(perm);
72      }
73  
74      public void checkPermission(Permission perm) {
75          if (!(perm instanceof ServicePermission)) {
76              return;
77          }
78          ServicePermission p = (ServicePermission)perm;
79          // ServicePermissions required to create GSSName are ignored
80          if (!p.getActions().isEmpty()) {
81              permChecks = permChecks
82                  + p.getActions().toUpperCase(Locale.US).charAt(0);
83          }
84      }

86      public static void main(String[] args) throws Exception {
87          // reset the security property to make sure that the algorithms
88          // and keys used in this test are not disabled.
89          Security.setProperty("jdk.tls.disabledAlgorithms", "");
90  
91          krb5Cipher = args[0];
92  
93          boolean unbound = args.length > 1;
94  
95          // Workaround for JDK-8161101, reference the class before
96          // SecurityManager is set.
97          System.out.println("Touching " + ServicePermission.class);
98  
99          System.setSecurityManager(new SSL());
100 
101         KDC kdc = KDC.create(OneKDC.REALM);
102         server = "host." + OneKDC.REALM_LOWER_CASE;
103 
104         if (args.length > 2) {
105             sniHostname = "test." + server;
106             sniMatcherPattern = ".*";
107         }
108 
109         kdc.addPrincipal(OneKDC.USER, OneKDC.PASS);
110         kdc.addPrincipalRandKey("krbtgt/" + OneKDC.REALM);
111         KDC.saveConfig(OneKDC.KRB5_CONF, kdc);
112         System.setProperty("java.security.krb5.conf", OneKDC.KRB5_CONF);
113 
114         // Add 3 versions of keys into keytab
115         KeyTab ktab = KeyTab.create(OneKDC.KTAB);
116         String serviceName = null;
117         if (sniHostname != null) {
118             serviceName = "host/" + sniHostname;
119         } else {
120             serviceName = "host/" + server;
121         }
122         PrincipalName service = new PrincipalName(
123             serviceName, PrincipalName.KRB_NT_SRV_HST);
124         ktab.addEntry(service, "pass1".toCharArray(), 1, true);
125         ktab.addEntry(service, "pass2".toCharArray(), 2, true);
126         ktab.addEntry(service, "pass3".toCharArray(), 3, true);
127         ktab.save();
128 
129         // and use the middle one as the real key
130         kdc.addPrincipal(serviceName, "pass2".toCharArray());
131 
132 
133         // JAAS config entry name ssl
134         System.setProperty("java.security.auth.login.config", OneKDC.JAAS_CONF);
135         File f = new File(OneKDC.JAAS_CONF);
136         FileOutputStream fos = new FileOutputStream(f);
137         fos.write((
138                 "ssl {\n" +
139                 "    com.sun.security.auth.module.Krb5LoginModule required\n" +
140                 (unbound ?
141                     "    principal=*\n" :
142                     "    principal=\"" + serviceName + "\"\n") +
143                 "    useKeyTab=true\n" +
144                 "    keyTab=" + OneKDC.KTAB + "\n" +
145                 "    isInitiator=false\n" +
146                 "    storeKey=true;\n};\n"
147                 ).getBytes());
148         fos.close();
149 
150         Context c;
151         final Context s = Context.fromJAAS("ssl");
152 
153         s.startAsServer(GSSUtil.GSS_KRB5_MECH_OID);
154 
155         Thread server = new Thread(new Runnable() {
156             public void run() {
157                 try {
158                     s.doAs(new JsseServerAction(), null);
159                 } catch (Exception e) {
160                     e.printStackTrace();
161                     serverState = 2;
162                 }
163             }
164         });
165         server.setDaemon(true);
166         server.start();
167 
168         while (serverState == 0) {
169             Thread.sleep(50);
170         }
171 
172         if (serverState == 2) {
173             throw new Exception("Server already failed");
174         }
175 
176         c = Context.fromUserPass(OneKDC.USER, OneKDC.PASS, false);
177         c.startAsClient(serviceName, GSSUtil.GSS_KRB5_MECH_OID);
178         c.doAs(new JsseClientAction(), null);
179 
180         // Add another version of key, make sure it can be loaded
181         Thread.sleep(2000);
182         ktab = KeyTab.getInstance(OneKDC.KTAB);
183         ktab.addEntry(service, "pass4".toCharArray(), 4, true);
184         ktab.save();
185         kdc.addPrincipal(serviceName, "pass4".toCharArray());
186 
187         c = Context.fromUserPass(OneKDC.USER, OneKDC.PASS, false);
188         c.startAsClient(serviceName, GSSUtil.GSS_KRB5_MECH_OID);
189         c.doAs(new JsseClientAction(), null);
190 
191         // Permission checking check. Please note this is highly
192         // implementation related.
193         if (unbound) {
194             // For unbound, server does not know what name to check.
195             // Client checks "initiate", then server gets the name
196             // and checks "accept". Second connection resume.
197             if (!permChecks.equals("IA")) {
198                 throw new Exception(permChecks);
199             }
200         } else {
201             // For bound, JAAS checks "accept" once. Server checks again,
202             // client then checks "initiate". Second connection resume.
203             if (!permChecks.equals("AAI")) {
204                 throw new Exception(permChecks);
205             }
206         }
207     }

209     // Following codes copied from
210     // http://java.sun.com/javase/6/docs/technotes/guides/security/jgss/lab/part2.html#JSSE
211     private static class JsseClientAction implements Action {
212         public byte[] run(Context s, byte[] input) throws Exception {
213             SSLSocketFactory sslsf =
214                 (SSLSocketFactory) SSLSocketFactory.getDefault();
215             System.out.println("Connecting " + server + ":" + port);
216             SSLSocket sslSocket = (SSLSocket) sslsf.createSocket(server, port);
217 
218             // Enable only a KRB5 cipher suite.
219             String enabledSuites[] = {krb5Cipher};
220             sslSocket.setEnabledCipherSuites(enabledSuites);
221             // Should check for exception if enabledSuites is not supported
222 
223             if (sniHostname != null) {
224                 List<SNIServerName> serverNames = new ArrayList<>();
225                 serverNames.add(new SNIHostName(sniHostname));
226                 SSLParameters params = sslSocket.getSSLParameters();
227                 params.setServerNames(serverNames);
228                 sslSocket.setSSLParameters(params);
229             }
230 
231             BufferedReader in = new BufferedReader(new InputStreamReader(
232                 sslSocket.getInputStream()));
233             BufferedWriter out = new BufferedWriter(new OutputStreamWriter(
234                 sslSocket.getOutputStream()));
235 
236             String outStr = "Hello There!\n";
237             out.write(outStr);
238             out.flush();
239             System.out.print("Sending " + outStr);
240 
241             String inStr = in.readLine();
242             System.out.println("Received " + inStr);
243 
244             String cipherSuiteChosen = sslSocket.getSession().getCipherSuite();
245             System.out.println("Cipher suite in use: " + cipherSuiteChosen);
246             Principal self = sslSocket.getSession().getLocalPrincipal();
247             System.out.println("I am: " + self.toString());
248             Principal peer = sslSocket.getSession().getPeerPrincipal();
249             System.out.println("Server is: " + peer.toString());
250 
251             sslSocket.close();
252             // This line should not be needed. It's the server's duty to
253             // forget the old key
254             //sslSocket.getSession().invalidate();
255             return null;
256         }
257     }
258 
259     private static class JsseServerAction implements Action {
260         public byte[] run(Context s, byte[] input) throws Exception {
261             SSLServerSocketFactory sslssf =
262                 (SSLServerSocketFactory) SSLServerSocketFactory.getDefault();
263             SSLServerSocket sslServerSocket =
264                 (SSLServerSocket) sslssf.createServerSocket(0); // any port
265             port = sslServerSocket.getLocalPort();
266             System.out.println("Listening on " + port);
267             serverState = 1;
268 
269             // Enable only a KRB5 cipher suite.
270             String enabledSuites[] = {krb5Cipher};
271             sslServerSocket.setEnabledCipherSuites(enabledSuites);
272             // Should check for exception if enabledSuites is not supported
273 
274             if (sniMatcherPattern != null) {
275                 List<SNIMatcher> matchers = new ArrayList<>();
276                 matchers.add(SNIHostName.createSNIMatcher(sniMatcherPattern));
277                 SSLParameters params = sslServerSocket.getSSLParameters();
278                 params.setSNIMatchers(matchers);
279                 sslServerSocket.setSSLParameters(params);
280             }
281 
282             while (loopCount++ < LOOP_LIMIT) {
283                 System.out.println("Waiting for incoming connection...");
284 
285                 SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
286 
287                 System.out.println("Got connection from client "
288                     + sslSocket.getInetAddress());
289 
290                 BufferedReader in = new BufferedReader(new InputStreamReader(
291                     sslSocket.getInputStream()));
292                 BufferedWriter out = new BufferedWriter(new OutputStreamWriter(
293                     sslSocket.getOutputStream()));
294 
295                 String inStr = in.readLine();
296                 System.out.println("Received " + inStr);
297 
298                 String outStr = inStr + " " + new Date().toString() + "\n";
299                 out.write(outStr);
300                 System.out.println("Sending " + outStr);
301                 out.flush();
302 
303                 String cipherSuiteChosen =
304                     sslSocket.getSession().getCipherSuite();
305                 System.out.println("Cipher suite in use: " + cipherSuiteChosen);
306                 Principal self = sslSocket.getSession().getLocalPrincipal();
307                 System.out.println("I am: " + self.toString());
308                 Principal peer = sslSocket.getSession().getPeerPrincipal();
309                 System.out.println("Client is: " + peer.toString());
310 
311                 sslSocket.close();
312             }
313             return null;
314         }
315     }
