1   package org.infinispan.query.remote.filter;
2   
3   import org.infinispan.commons.marshall.AbstractExternalizer;
4   import org.infinispan.factories.annotations.Inject;
5   import org.infinispan.manager.EmbeddedCacheManager;
6   import org.infinispan.metadata.Metadata;
7   import org.infinispan.notifications.cachelistener.filter.EventType;
8   import org.infinispan.objectfilter.ObjectFilter;
9   import org.infinispan.protostream.ProtobufUtil;
10  import org.infinispan.protostream.SerializationContext;
11  import org.infinispan.query.dsl.embedded.impl.JPACacheEventFilterConverter;
12  import org.infinispan.query.dsl.embedded.impl.JPAFilterAndConverter;
13  import org.infinispan.query.remote.ExternalizerIds;
14  import org.infinispan.query.remote.ProtobufMetadataManager;
15  import org.infinispan.query.remote.client.FilterResult;
16  
17  import java.io.IOException;
18  import java.io.ObjectInput;
19  import java.io.ObjectOutput;
20  import java.util.Collections;
21  import java.util.Set;
22  
23  /**
24   * @author anistor@redhat.com
25   * @since 7.2
26   */
27  public final class JPAProtobufCacheEventFilterConverter extends JPACacheEventFilterConverter<Object, Object, byte[]> {
28  
29     private transient SerializationContext serCtx;
30  
31     public JPAProtobufCacheEventFilterConverter(JPAFilterAndConverter<Object, Object> filterAndConverter) {
32        super(filterAndConverter);
33     }
34  
35     @Inject
36     protected void injectDependencies(EmbeddedCacheManager cacheManager) {
37        serCtx = ProtobufMetadataManager.getSerializationContextInternal(cacheManager);
38     }
39  
40     @Override
41     public byte[] filterAndConvert(Object key, Object oldValue, Metadata oldMetadata, Object newValue, Metadata newMetadata, EventType eventType) {
42        ObjectFilter.FilterResult filterResult = filterAndConverter.filterAndConvert(key, newValue, newMetadata);
43        if (filterResult != null) {
44           try {
45              return ProtobufUtil.toWrappedByteArray(serCtx, new FilterResult(filterResult.getInstance(), filterResult.getProjection(), filterResult.getSortProjection()));
46           } catch (IOException e) {
47              throw new RuntimeException(e);
48           }
49        }
50        return null;
51     }
52  
53     public static final class Externalizer extends AbstractExternalizer<JPAProtobufCacheEventFilterConverter> {
54  
55        @Override
56        public void writeObject(ObjectOutput output, JPAProtobufCacheEventFilterConverter object) throws IOException {
57           output.writeObject(object.filterAndConverter);
58        }
59  
60        @Override
61        public JPAProtobufCacheEventFilterConverter readObject(ObjectInput input) throws IOException, ClassNotFoundException {
62           JPAFilterAndConverter filterAndConverter = (JPAFilterAndConverter) input.readObject();
63           return new JPAProtobufCacheEventFilterConverter(filterAndConverter);
64        }
65  
66        @Override
67        public Integer getId() {
68           return ExternalizerIds.JPA_PROTOBUF_CACHE_EVENT_FILTER_CONVERTER;
69        }
70  
71        @Override
72        public Set<Class<? extends JPAProtobufCacheEventFilterConverter>> getTypeClasses() {
73           return Collections.<Class<? extends JPAProtobufCacheEventFilterConverter>>singleton(JPAProtobufCacheEventFilterConverter.class);
74        }
75     }
76  }
