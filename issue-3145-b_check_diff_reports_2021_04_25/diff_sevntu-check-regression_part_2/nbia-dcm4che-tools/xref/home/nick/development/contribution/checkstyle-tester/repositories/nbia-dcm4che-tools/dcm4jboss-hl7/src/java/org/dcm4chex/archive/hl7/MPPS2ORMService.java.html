290      * (non-Javadoc)

292      * @see javax.management.NotificationListener#handleNotification(javax.management.Notification,
293      *      java.lang.Object)
294      */
295     public void handleNotification(Notification notif, Object handback) {
296         log.debug("handleNotification called! type:"+notif.getType());
297         Dataset mpps = (Dataset) notif.getUserData();
298         boolean linked = MPPSScpService.EVENT_TYPE_MPPS_LINKED.equals(notif.getType());
299         if (!linked) {
300             if (ignoreInProgress
301                     && "IN PROGRESS".equals(mpps.getString(Tags.PPSStatus)))
302                 return;
303     
304             final String iuid = mpps.getString(Tags.SOPInstanceUID);
305             mpps = getMPPS(iuid);
306         }
307         handle(mpps, linked);
308     }
309 
310     private void handle(Dataset mpps, boolean linked) {
311         DcmElement sq = mpps.get(Tags.ScheduledStepAttributesSeq);
312         if (sq == null || sq.isEmpty()) {
313             log
314                     .error("Missing Scheduled Step Attributes Seq in MPPS - "
315                             + mpps.getString(Tags.SOPInstanceUID));
316             return;
317         }
318         if (ignoreUnscheduled
319                 && sq.getItem().getString(Tags.AccessionNumber) == null) {
320             return;
321         }
322         if (oneORMperSPS) {
323             for (int i = 0, n = sq.countItems(); i < n; i++) {
324                 mpps.putSQ(Tags.ScheduledStepAttributesSeq).addItem(
325                         sq.getItem(i));
326                 scheduleORM(makeORM(mpps, linked));
327             }
328         } else {
329             scheduleORM(makeORM(mpps, linked));
330         }
331     }
332 
333     private void scheduleORM(byte[] bs) {
334         if (bs == null)
335             return;
336         try {
337             server.invoke(hl7SendServiceName, "forward", new Object[] { bs },
338                     new String[] { byte[].class.getName() });
339         } catch (Exception e) {
340             log.error("Failed to schedule ORM", e);
341         }
342     }
343 
344     private byte[] makeORM(Dataset mpps, boolean linked) {
345         if (mpps == null)
346             return null;
347         try {
348             if (logXSLT)
349                 try {
350                     logXSLT(mpps, linked);
351                 } catch (Exception e) {
352                     log.warn("Failed to log XSLT:", e);
353                 }
354             ByteArrayOutputStream out = new ByteArrayOutputStream(
355                     INIT_BUFFER_SIZE);
356             TransformerHandler th = getTransformerHandler(linked);
357             XMLWriter xmlWriter = new HL7XMLWriter(
358             		new OutputStreamWriter(out, ISO_8859_1));
359             th.setResult(new SAXResult(xmlWriter.getContentHandler()));
360             mpps.writeDataset2(th, null, null, 10240, null);
361             log.info(new String(out.toByteArray()));
362             return out.toByteArray();
363         } catch (Exception e) {
364             log.error("Failed to convert MPPS to ORM", e);
365             log.error(mpps);
366             return null;
367         }
368     }
369 
370     private void logXSLT(Dataset mpps, boolean linked) throws Exception {
371         SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory
372                 .newInstance();
373         String uid = mpps.getString(Tags.SOPInstanceUID);
374         String fn = (linked ? "mpps-linked-" : "mpps-received-")+uid;
375         logXSLT(mpps, tf.newTransformerHandler(), new File(logDir, fn + ".xml"));
376         logXSLT(mpps, getTransformerHandler(linked), new File(logDir, fn +".orm.xml"));
377     }
378 
379     private void logXSLT(Dataset mpps, TransformerHandler th, File logFile)
380             throws Exception {
381         TagDictionary dict = DictionaryFactory.getInstance()
382                 .getDefaultTagDictionary();
383         FileOutputStream out = new FileOutputStream(logFile);
384         try {
385             th.setResult(new StreamResult(out));
386             mpps.writeDataset2(th, dict, null, 10240, null);
387         } finally {
388             out.close();
389         }
390     }
391 
392     private TransformerHandler getTransformerHandler( boolean linked ) throws Exception {
393         SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory
394                 .newInstance();
395         File xslFile = FileUtils.toExistingFile(xslPath);
396         TransformerHandler th = tf.newTransformerHandler(
397                 templates.getTemplates(xslFile));
398         Transformer t = th.getTransformer();
399         t.setParameter("SendingApplication", linked ? linkedSendingApplication : receivedSendingApplication);
400         t.setParameter("SendingFacility", linked ? linkedSendingFacility : receivedSendingFacility);
401         t.setParameter("ReceivingApplication", linked ? linkedReceivingApplication : receivedReceivingApplication);
402         t.setParameter("ReceivingFacility", linked ? linkedReceivingFacility : receivedReceivingFacility);
403         return th;
404     }
405 
406     private Dataset getMPPS(String iuid) {
407         try {
408             MPPSManager mgr = getMPPSManagerHome().create();
409             try {
410                 return mgr.getMPPS(iuid);
411             } finally {
412                 try {
413                     mgr.remove();
414                 } catch (Exception ignore) {
415                 }
416             }
417         } catch (Exception e) {
418             log.error("Failed to load MPPS - " + iuid, e);
419             return null;
420         }
421     }
