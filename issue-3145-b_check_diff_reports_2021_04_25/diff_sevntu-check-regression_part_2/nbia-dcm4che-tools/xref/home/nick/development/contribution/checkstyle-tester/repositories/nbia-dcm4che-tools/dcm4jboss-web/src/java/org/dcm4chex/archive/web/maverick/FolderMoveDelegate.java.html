86      /**
87       * Move one ore more model instances to another parent.
88       * <p>
89       * The move is restricted with following rules:<br>
90       * 1) the destinations model type must be the same as the source parents model type.<br>
91       * 2) the destination must be different to the source parent.<br>
92       * 3) all source models must have the same parent.<br>
93       * 4) the destination and the source parent must have the same parent.<br>
94       * 
95       * @return the name of the next view.
96       */
97      public boolean move() {
98          folderForm = (FolderForm) ctrl.getForm();
99          int[] srcDest = getSrcAndDest( folderForm.getStickyPatients(),
100                 folderForm.getStickyStudies(),
101                 folderForm.getStickySeries(),
102                 folderForm.getStickyInstances()
103         );
104         int moveType = srcDest[0];
105         boolean ret = false;
106 
107         if ( moveType != MOVE_ERROR) {
108             if (! this.checkStickyPlacement(srcDest[0], srcDest[1]) ) return false;
109 
110             int iDest = srcDest[1];
111             ContentManager cm = null;
112             try {
113                 ContentManagerHome home = (ContentManagerHome) EJBHomeFactory
114                 .getFactory().lookup(ContentManagerHome.class,
115                         ContentManagerHome.JNDI_NAME);
116                 cm = home.create();
117 
118                 if ( moveType == STUDY ) {
119                     _move_studies( cm );
120                 } else if ( moveType == SERIES ) {
121                     _move_series( cm, iDest );
122                 } else if ( moveType == INSTANCE ) {
123                     _move_instances( cm, iDest );
124                 }
125                 ctrl.clearSticky();
126                 ret = true;
127             } catch ( Exception x ) {
128                 folderForm.setPopupMsg("folder.err_move", x.getMessage() );
129             }	finally {
130                 try {
131                     if ( cm != null )
132                         cm.remove();
133                 } catch (Exception e) {
134                 }
135             }
136         }
137         return ret;
138     }

142      * Move selected studies to a patient.

146     private void _move_studies( ContentManager cm ) throws Exception {
147         long iDest = Long.parseLong( (String) folderForm.getStickyPatients().iterator().next().toString() );
148         long[] iaSrc = getLongArrayFromSet( folderForm.getStickyStudies() );
149         delegate.moveStudies( iaSrc, iDest );
150 
151         PatientModel destPat = folderForm.getPatientByPk( iDest );
152         List path = findModelPath( folderForm.getPatients(), iaSrc[0], 1 );
153         PatientModel srcPat = (PatientModel) path.get(0);
154         _updatePatientWithStudies( destPat, cm );
155         _updatePatientWithStudies( srcPat, cm );
156         StudyModel study;
157         for ( int i = 0, len=iaSrc.length ; i < len ; i++ ) {
158             study = new StudyModel( cm.getStudy( iaSrc[i] ) );
159         }
160     }

162     private void _move_series( ContentManager cm, int dest ) throws Exception {
163         long iDestStdy = -1, iDestPat = -1;
164         if ( dest == STUDY ) {
165             iDestStdy = Integer.parseInt( folderForm.getStickyStudies().iterator().next().toString() );
166         } else {
167             iDestPat = Integer.parseInt( folderForm.getStickyPatients().iterator().next().toString() );
168         }
169         long[] iaSrc = getLongArrayFromSet( folderForm.getStickySeries() );
170 
171         List sourcePath = findModelPath( folderForm.getPatients(), iaSrc[0], 2 );
172         StudyModel srcStudy = (StudyModel) sourcePath.get(1);//0..patient,1..study,2..series
173 
174         StudyModel destStudy;
175         PatientModel pat;
176         if ( iDestStdy != -1 ) {
177             List destPath = findModelPath( folderForm.getPatients(), iDestStdy, 1 );
178             destStudy = (StudyModel) destPath.get(1);
179             pat = (PatientModel) destPath.get(0);
180         } else {
181             List destPath = findModelPath( folderForm.getPatients(), iDestPat, 0 );
182             pat = (PatientModel) destPath.get(0);
183             destStudy = new StudyModel( delegate.createStudy(getStudyDS(srcStudy), pat.getPk() ) );
184             pat.getStudies().add(destStudy);
185             iDestStdy = destStudy.getPk();
186         }
187         delegate.moveSeries( iaSrc, iDestStdy );
188         //update model for view
189         _updateStudyWithSeries( destStudy, cm );
190         _updateStudyWithSeries( srcStudy, cm );
191         if ( checkMovedSeriesForPpsIuid(destStudy, iaSrc) ) {
192             folderForm.setPopupMsg("folder.err_move_seriesWithPPS", (String)null);
193         }
194     }

196     private boolean checkMovedSeriesForPpsIuid(StudyModel destStudy, long[] iaSrc) {
197         SeriesModel series;
198         long pk;
199         for ( Iterator it = destStudy.getSeries().iterator() ; it.hasNext() ; ) {
200             series = (SeriesModel) it.next();
201             pk = series.getPk();
202             for ( int i = 0 ; i < iaSrc.length ; i++) {
203                 if ( pk == iaSrc[i] && series.getPPSIUID() != null ){
204                     return true;
205                 }
206             }
207         }
208         return false;
209     }

211     private void _move_instances( ContentManager cm, int dest ) throws Exception {
212         long iDestSeries = -1,iDestStdy = -1, iDestPat = -1;
213         if ( dest == SERIES ) {
214             iDestSeries = Integer.parseInt( (String) folderForm.getStickySeries().iterator().next().toString() );
215         } else if ( dest == STUDY ) {
216             iDestStdy = Integer.parseInt( (String) folderForm.getStickyStudies().iterator().next().toString() );
217         } else {
218             iDestPat = Integer.parseInt( (String) folderForm.getStickyPatients().iterator().next().toString() );
219         }
220         long[] iaSrc = getLongArrayFromSet( folderForm.getStickyInstances() );
221 
222         List sourcePath = findModelPath( folderForm.getPatients(), iaSrc[0], 3 );
223         StudyModel srcStudy = (StudyModel) sourcePath.get(1);//0..patient,1..study,2..series
224         SeriesModel srcSeries = (SeriesModel) sourcePath.get(2);//0..patient,1..study,2..series
225 
226         SeriesModel destSeries;
227         StudyModel destStudy = null;
228         PatientModel pat;
229         if ( iDestSeries != -1 ) {
230             List destPath = findModelPath( folderForm.getPatients(), iDestSeries, 2 );
231             destSeries = (SeriesModel) destPath.get(2);//0..patient,1..study,2..series
232             pat = (PatientModel) destPath.get(0);
233         } else {
234             if ( iDestStdy != -1 ) {
235                 List destPath = findModelPath( folderForm.getPatients(), iDestStdy, 1 );
236                 destStudy = (StudyModel) destPath.get(1);
237                 pat = (PatientModel) destPath.get(0);
238                 destSeries = new SeriesModel(delegate.createSeries(getSeriesDS(srcSeries), iDestStdy));
239                 destStudy.getSeries().add(destSeries);
240             } else {
241                 List destPath = findModelPath( folderForm.getPatients(), iDestPat, 0 );
242                 pat = (PatientModel) destPath.get(0);
243                 destStudy = new StudyModel( delegate.createStudy(getStudyDS(srcStudy), pat.getPk() ) );
244                 pat.getStudies().add(destStudy);
245                 destSeries = new SeriesModel(delegate.createSeries(getSeriesDS(srcSeries), destStudy.getPk()));
246                 destStudy.getSeries().add(destSeries);
247             }
248             iDestSeries = destSeries.getPk();
249         }
250         delegate.moveInstances( iaSrc, iDestSeries );
251 
252         _updateSeriesWithInstances( destSeries, cm );
253         _updateSeriesWithInstances( (SeriesModel) sourcePath.get(2), cm );//0..patient,1..study,2..series,3..instances
254 
255         if ( destStudy != null ) 
256             destStudy.update( cm.getStudy( destStudy.getPk() ) );
257     }

265     private Dataset getSeriesDS(SeriesModel srcSeries) {
266         Dataset ds = dof.newDataset();
267         ds.putCS(Tags.SpecificCharacterSet, "ISO_IR 100");
268         ds.putUI(Tags.SeriesInstanceUID, UIDGenerator.getInstance().createUID() );
269         String desc;
270         if ( srcSeries == null ) {
271             desc = AUTO_GENERATED_BY_MOVE_TEXT;
272         } else {
273             desc = srcSeries.getSeriesDescription();
274             desc = addAutogeneratedInfo(desc);
275             Dataset dsSrc = srcSeries.toDataset();
276             ds.putDA(Tags.SeriesDate, dsSrc.getDate(Tags.SeriesDate) );
277             ds.putTM(Tags.SeriesTime, dsSrc.getDate(Tags.SeriesTime) );
278             ds.putCS(Tags.Modality, dsSrc.getString(Tags.Modality));
279         }
280         ds.putLO(Tags.SeriesDescription,desc);
281         return ds;
282     }

307     private String addAutogeneratedInfo(String desc) {
308         if (desc == null)
309             desc = AUTO_GENERATED_BY_MOVE_TEXT;
310         else if (desc.indexOf(AUTO_GENERATED_BY_MOVE_TEXT) == -1) {
311             desc = desc+" ("+AUTO_GENERATED_BY_MOVE_TEXT+")";
312             if (desc.length() > 64) {
313                 desc = desc.substring(0, 65);
314             }
315         }
316         return desc;
317     }

348     /**
349      * Checks the numbers of selected stickies to get moveMode.
350      * <p>
351      * This method checks if only one destination, one ore more sources and no other stickies are set.
352      * <p>
353      * If one of the following rules are not fulfilled, a corresponding error code is set in FolderForms.
354      * <DL>
355      * <DT>Rule:</DT>
356      * <DD>  iDest must be 1</DD>
357      * <DD>  one of iSrcStudy, iSrcSeries or iSrcInstance must be greater than 0</DD>
358      * <DD>  all other have to be 0</DD>
359      * </DL>
360      * 
361      * @param iDest			Number of selected destinations
362      * @param iSrcStudy		Number of selected sources study level
363      * @param iSrcSeries	Number of selected sources series level
364      * @param iSrcInstance	Number of selected sources instance level
365      * 
366      * @return The move mode or MOVE_ERROR.
367      */
368     private int[] getSrcAndDest( Set stickyPat, Set stickyStudy, Set stickySeries, Set stickyInstance ) {
369         int dest = MOVE_ERROR;
370         int src = MOVE_ERROR;
371         int mask = (stickyPat.isEmpty() ? 0:8) | 
372         (stickyStudy.isEmpty() ? 0:4) | 
373         (stickySeries.isEmpty() ? 0:2) | 
374         (stickyInstance.isEmpty() ? 0:1);
375         switch ( mask ) {
376         case 0x09:
377             dest = PATIENT;
378             src = INSTANCE;
379             break;
380         case 0x0A:
381             dest = PATIENT;
382             src = SERIES;
383             break;
384         case 0x0C:
385             dest = PATIENT;
386             src = STUDY;
387             break;
388 
389         case 0x05:
390             dest = STUDY;
391             src = INSTANCE;
392             break;
393         case 0x06:
394             dest = STUDY;
395             src = SERIES;
396             break;
397         case 0x03:
398             dest = SERIES;
399             src = INSTANCE;
400             break;
401 
402         case 0x00:
403             folderForm.setPopupMsg( "folder.err_move_no_selection", "" );
404             break;
405         default:
406             folderForm.setPopupMsg( "folder.err_move_no_source", "" );
407         }
408         return new int[]{src,dest};
409     }

411     /**
412      * Checks if the sticky informations are at correct positions.
413      * <p>
414      * This method checks following restrictions:<br>
415      * 1) Only one destination must be selected.
416      * 2) the destination must be different to the source parent.<br>
417      * 3) all source models must have the same parent.<br>
418      * <p>
419      * If one of the checks failed a corresponding error code is set in FolderForms.
420      * 
421      * @param stickySrc Sticky information of the childs (the source).
422      * @param stickyDest Sticky information of the parent (the destination).
423      * @return
424      */
425     private boolean checkStickyPlacement( int src, int dest ) {
426         Set stickyDst = getStickies( dest );
427         if ( stickyDst.size() > 1) {
428             folderForm.setPopupMsg( "folder.err_move_toManyDest", "" );
429             return false;
430         } 
431         Set stickySrc = getStickies( src );
432         List listDst = findModelPath( folderForm.getPatients(), (String)stickyDst.iterator().next(), dest );
433         List listSrc = findModelPath( folderForm.getPatients(), (String)stickySrc.iterator().next(), src );
434         int listParentSize = listDst.size()-1;
435         if ( listDst.get(listParentSize).equals( listSrc.get(listSrc.size()-1) ) ) {
436             if ( src == STUDY)
437                 folderForm.setPopupMsg( "folder.err_move_samePatient", "" );
438             else if ( src == SERIES)
439                 folderForm.setPopupMsg( "folder.err_move_sameStudy", "" );
440             else if ( src == INSTANCE)
441                 folderForm.setPopupMsg( "folder.err_move_sameSeries", "" );
442             return false; //same parent;
443         }
444         AbstractModel model = (AbstractModel) listSrc.get(listSrc.size()-2);//parent model of first child
445         Iterator iter = stickySrc.iterator();
446         iter.next(); //skip first element (is used for listChilds)
447         while ( iter.hasNext() ) {
448             if ( !model.containsPK( Long.valueOf( iter.next().toString() ) ) ) {
449                 if ( src == STUDY)
450                     folderForm.setPopupMsg( "folder.err_move_diffStudyParent", "" );
451                 else if ( src == SERIES)
452                     folderForm.setPopupMsg( "folder.err_move_diffSeriesParent", "" );
453                 else if ( src == INSTANCE)
454                     folderForm.setPopupMsg( "folder.err_move_diffInstanceParent", "" );
455                 return false; //this child has not the same parent as first child element
456             }
457         }
458         return true;
459     }

463      * Converts a Set of int values as String to an int array.
464      * 
465      * @param set Set with String objects of int
466      * @return int Array

468     public static long[] getLongArrayFromSet(Set set) {
469         if ( set == null ) return null;
470         long[] ia = new long[ set.size() ];
471         int i = 0;
472         for ( Iterator iter = set.iterator() ; iter.hasNext() ; i++ ) {
473             ia[i] = Long.parseLong( (String) iter.next() );
474         }
475         return ia;
476     }

479     /**
480      * Returns the tree nodes for a given pk.
481      * <p>
482      * Because the pk's are only unique within a model type, it is necessary to use the correct
483      * depth value!
484      * <p>
485      * 
486      * @param parent 	This is the root model of the tree to search.
487      * @param pk		The pk as String to search
488      * @param depth		The tree depth where the pk should be found.
489      * @return A list with all nodes to get the model.
490      */
491     private List findModelPath( List parent, String pk, int depth ) {
492         long lPk = Long.parseLong(pk);
493         return findModelPath( parent, lPk, depth );
494     }
495 
496     /**
497      * Returns the tree nodes for a given pk.
498      * <p>
499      * Because the pk's are only unique within a model type, it is necessary to use the correct
500      * depth value!
501      * <p>
502      * 
503      * @param parent 	This is the root model of the tree to search.
504      * @param pk		The pk as int to search
505      * @param depth		The tree depth where the pk should be found.
506      * @return A list with all nodes to get the model.
507      */
508     private List findModelPath( List parent, long pk, int depth ) {
509         Iterator iter = parent.iterator();
510         AbstractModel model = null;
511         List l = new ArrayList();
512         while ( iter.hasNext() ) {
513             model = (AbstractModel) iter.next();
514             if ( depth == 0 ) {//should pk in this parent?
515                 if ( model.getPk() == pk ) {
516                     l.add( model );
517                     return l;
518                 }
519             } else { //search in next tree segment.
520                 List l1 = findModelPath( model.listOfChilds(), pk, depth-1 );
521                 if ( l1 != null ) {
522                     l.add( model );
523                     l.addAll( l1 );
524                     return l;
525                 }
526             }
527         }
528         return null;
529     }
