71  	/**
72  	 * Register the "hydrated" state of an entity instance, after the first step of 2-phase loading.
73  	 *
74  	 * Add the "hydrated state" (an array) of an uninitialized entity to the session. We don't try
75  	 * to resolve any associations yet, because there might be other entities waiting to be
76  	 * read from the JDBC result set we are currently processing
77  	 */
78  	public static void postHydrate(
79  		final EntityPersister persister,
80  		final Serializable id,
81  		final Object[] values,
82  		final Object rowId,
83  		final Object object,
84  		final LockMode lockMode,
85  		final boolean lazyPropertiesAreUnfetched,
86  		final SessionImplementor session)
87  	throws HibernateException {
88  
89  		Object version = Versioning.getVersion( values, persister );
90  		session.getPersistenceContext().addEntry(
91  				object,
92  				Status.LOADING,
93  				values,
94  				rowId,
95  				id,
96  				version,
97  				lockMode,
98  				true,
99  				persister,
100 				false,
101 				lazyPropertiesAreUnfetched
102 			);
103 
104 		if ( version != null && LOG.isTraceEnabled() ) {
105 			String versionStr = persister.isVersioned()
106 					? persister.getVersionType().toLoggableString( version, session.getFactory() )
107 					: "null";
108 			LOG.tracev( "Version: {0}", versionStr );
109 		}
110 
111 	}

144 	private static void doInitializeEntity(
145 			final Object entity,
146 			final EntityEntry entityEntry,
147 			final boolean readOnly,
148 			final SessionImplementor session,
149 			final PreLoadEvent preLoadEvent,
150 			final PostLoadEvent postLoadEvent) throws HibernateException {
151 		final PersistenceContext persistenceContext = session.getPersistenceContext();
152 		EntityPersister persister = entityEntry.getPersister();
153 		Serializable id = entityEntry.getId();
154 		Object[] hydratedState = entityEntry.getLoadedState();
155 
156 		final boolean debugEnabled = LOG.isDebugEnabled();
157 		if ( debugEnabled ) {
158 			LOG.debugf(
159 					"Resolving associations for %s",
160 					MessageHelper.infoString( persister, id, session.getFactory() )
161 			);
162 		}
163 
164 		Type[] types = persister.getPropertyTypes();
165 		for ( int i = 0; i < hydratedState.length; i++ ) {
166 			final Object value = hydratedState[i];
167 			if ( value!=LazyPropertyInitializer.UNFETCHED_PROPERTY && value!=BackrefPropertyAccessor.UNKNOWN ) {
168 				hydratedState[i] = types[i].resolve( value, session, entity );
169 			}
170 		}
171 
172 		//Must occur after resolving identifiers!
173 		if ( session.isEventSource() ) {
174 			preLoadEvent.setEntity( entity ).setState( hydratedState ).setId( id ).setPersister( persister );
175 
176 			final EventListenerGroup<PreLoadEventListener> listenerGroup = session
177 					.getFactory()
178 					.getServiceRegistry()
179 					.getService( EventListenerRegistry.class )
180 					.getEventListenerGroup( EventType.PRE_LOAD );
181 			for ( PreLoadEventListener listener : listenerGroup.listeners() ) {
182 				listener.onPreLoad( preLoadEvent );
183 			}
184 		}
185 
186 		persister.setPropertyValues( entity, hydratedState );
187 
188 		final SessionFactoryImplementor factory = session.getFactory();
189 		if ( persister.hasCache() && session.getCacheMode().isPutEnabled() ) {
190 
191 			if ( debugEnabled ) {
192 				LOG.debugf(
193 						"Adding entity to second-level cache: %s",
194 						MessageHelper.infoString( persister, id, session.getFactory() )
195 				);
196 			}
197 
198 			Object version = Versioning.getVersion(hydratedState, persister);
199 			CacheEntry entry = persister.buildCacheEntry( entity, hydratedState, version, session );
200 			CacheKey cacheKey = session.generateCacheKey( id, persister.getIdentifierType(), persister.getRootEntityName() );
201 
202 			// explicit handling of caching for rows just inserted and then somehow forced to be read
203 			// from the database *within the same transaction*.  usually this is done by
204 			// 		1) Session#refresh, or
205 			// 		2) Session#clear + some form of load
206 			//
207 			// we need to be careful not to clobber the lock here in the cache so that it can be rolled back if need be
208 			if ( session.getPersistenceContext().wasInsertedDuringTransaction( persister, id ) ) {
209 				persister.getCacheAccessStrategy().update(
210 						cacheKey,
211 						persister.getCacheEntryStructure().structure( entry ),
212 						version,
213 						version
214 				);
215 			}
216 			else {
217 				try {
218 					session.getEventListenerManager().cachePutStart();
219 					final boolean put = persister.getCacheAccessStrategy().putFromLoad(
220 							cacheKey,
221 							persister.getCacheEntryStructure().structure( entry ),
222 							session.getTimestamp(),
223 							version,
224 							useMinimalPuts( session, entityEntry )
225 					);
226 
227 					if ( put && factory.getStatistics().isStatisticsEnabled() ) {
228 						factory.getStatisticsImplementor().secondLevelCachePut( persister.getCacheAccessStrategy().getRegion().getName() );
229 					}
230 				}
231 				finally {
232 					session.getEventListenerManager().cachePutEnd();
233 				}
234 			}
235 		}
236 
237 		if ( persister.hasNaturalIdentifier() ) {
238 			persistenceContext.getNaturalIdHelper().cacheNaturalIdCrossReferenceFromLoad(
239 					persister,
240 					id,
241 					persistenceContext.getNaturalIdHelper().extractNaturalIdValues( hydratedState, persister )
242 			);
243 		}
244 
245 		boolean isReallyReadOnly = readOnly;
246 		if ( !persister.isMutable() ) {
247 			isReallyReadOnly = true;
248 		}
249 		else {
250 			Object proxy = persistenceContext.getProxy( entityEntry.getEntityKey() );
251 			if ( proxy != null ) {
252 				// there is already a proxy for this impl
253 				// only set the status to read-only if the proxy is read-only
254 				isReallyReadOnly = ( ( HibernateProxy ) proxy ).getHibernateLazyInitializer().isReadOnly();
255 			}
256 		}
257 		if ( isReallyReadOnly ) {
258 			//no need to take a snapshot - this is a
259 			//performance optimization, but not really
260 			//important, except for entities with huge
261 			//mutable property values
262 			persistenceContext.setEntryStatus(entityEntry, Status.READ_ONLY);
263 		}
264 		else {
265 			//take a snapshot
266 			TypeHelper.deepCopy(
267 					hydratedState,
268 					persister.getPropertyTypes(),
269 					persister.getPropertyUpdateability(),
270 					hydratedState,  //after setting values to object, entityMode
271 					session
272 			);
273 			persistenceContext.setEntryStatus(entityEntry, Status.MANAGED);
274 		}
275 
276 		persister.afterInitialize(
277 				entity,
278 				entityEntry.isLoadedWithLazyPropertiesUnfetched(),
279 				session
280 		);
281 
282 		if ( debugEnabled ) {
283 			LOG.debugf(
284 					"Done materializing entity %s",
285 					MessageHelper.infoString( persister, id, session.getFactory() )
286 			);
287 		}
288 
289 		if ( factory.getStatistics().isStatisticsEnabled() ) {
290 			factory.getStatisticsImplementor().loadEntity( persister.getEntityName() );
291 		}
292 	}

294 	/**
295 	 * PostLoad cannot occur during initializeEntity, as that call occurs *before*
296 	 * the Set collections are added to the persistence context by Loader.
297 	 * Without the split, LazyInitializationExceptions can occur in the Entity's
298 	 * postLoad if it acts upon the collection.
299 	 * 
300 	 * 
301 	 * HHH-6043
302 	 * 
303 	 * @param entity
304 	 * @param session
305 	 * @param postLoadEvent
306 	 */
307 	public static void postLoad(
308 			final Object entity,
309 			final SessionImplementor session,
310 			final PostLoadEvent postLoadEvent) {
311 		
312 		if ( session.isEventSource() ) {
313 			final PersistenceContext persistenceContext
314 					= session.getPersistenceContext();
315 			final EntityEntry entityEntry = persistenceContext.getEntry(entity);
316 			final Serializable id = entityEntry.getId();
317 			
318 			postLoadEvent.setEntity( entity ).setId( entityEntry.getId() )
319 					.setPersister( entityEntry.getPersister() );
320 
321 			final EventListenerGroup<PostLoadEventListener> listenerGroup
322 					= session
323 							.getFactory()
324 							.getServiceRegistry()
325 							.getService( EventListenerRegistry.class )
326 							.getEventListenerGroup( EventType.POST_LOAD );
327 			for ( PostLoadEventListener listener : listenerGroup.listeners() ) {
328 				listener.onPostLoad( postLoadEvent );
329 			}
330 		}
331 	}

333 	private static boolean useMinimalPuts(SessionImplementor session, EntityEntry entityEntry) {
334 		return ( session.getFactory().getSettings().isMinimalPutsEnabled() &&
335 						session.getCacheMode()!=CacheMode.REFRESH ) ||
336 				( entityEntry.getPersister().hasLazyProperties() &&
337 						entityEntry.isLoadedWithLazyPropertiesUnfetched() &&
338 						entityEntry.getPersister().isLazyPropertiesCacheable() );
339 	}

341 	/**
342 	 * Add an uninitialized instance of an entity class, as a placeholder to ensure object
343 	 * identity. Must be called before <tt>postHydrate()</tt>.
344 	 *
345 	 * Create a "temporary" entry for a newly instantiated entity. The entity is uninitialized,
346 	 * but we need the mapping from id to instance in order to guarantee uniqueness.
347 	 */
348 	public static void addUninitializedEntity(
349 			final EntityKey key,
350 			final Object object,
351 			final EntityPersister persister,
352 			final LockMode lockMode,
353 			final boolean lazyPropertiesAreUnfetched,
354 			final SessionImplementor session
355 	) {
356 		session.getPersistenceContext().addEntity(
357 				object,
358 				Status.LOADING,
359 				null,
360 				key,
361 				null,
362 				lockMode,
363 				true,
364 				persister,
365 				false,
366 				lazyPropertiesAreUnfetched
367 			);
368 	}
