91   /**
92    * This constructor parses the URL parameters and builds the tables.
93    * It parses the requiredfields and newrequiredfields parameters on the URL.
94    * then saves them in 2 tables which are accessed via the methods
95    * getFields() and getValues().
96    * The data stored in the table has had all the decoding applied.
97    *
98    * The data stored in the requiredfields= parameter must be double encoded because
99    * it can contain a "." or a "|" which are the google and/or logical operations
100   * to apply to the query. We only support the "." and operation.
101   *
102   * while the data stored in the newrequiredfields parameter can be singular decoded
103   *
104   * @param req the request object of the servlet
105   */
106  public Bug3559113(final HttpServletRequest req) {
107      String reqfieldParameter=req.getParameter("requiredfields");
108      if (reqfieldParameter != null) {
109 
110          // Decode the data after you split the expression apart
111 
112          //String reqfields=decode(reqfieldParameter);
113          String reqfields=reqfieldParameter;
114 //       try {
115 //           reqfields=new String(reqfieldParameter.getBytes(),"utf8");
116 //       } catch (UnsupportedEncodingException e) {
117 //
118 //           e.printStackTrace();
119 //       }
120 
121          String [] ss_and=null;
122          String [] ss_or=null;
123          if (reqfields != null) {
124              if (reqfields.indexOf('|') > -1 )
125                  ss_or=reqfields.split("\\|");
126              else
127                  ss_and=reqfields.split("\\.");
128 
129              String [] ss_l=ss_or;
130              // This is what we may be parsing
131              // The Meta Tags can be separated by either a "." for and "and" operation
132              // of a "|" for and "or" operation.
133              //
134              //   requiredfields=MetaName1:MetaValue1.MetaName2:MetaValue2.MetaName3:MetaValue3 ...
135              // or
136              //   requiredfields=MetaName1:MetaValue1|MetaName2:MetaValue2|MetaName3:MetaValue3 ...
137              //
138              for (int Logical_operator=1;Logical_operator <3;Logical_operator++) {
139                  // parse both the or case as well as the and case
140                  if (ss_l == null) {
141                      ss_l=ss_and;
142                      continue;
143                  }
144                  final int len=ss_l.length;
145                  for (int l_index=0;l_index<len;l_index++) {
146                      final String [] ss=decode(ss_l[l_index]).split(":");
147                      if (ss.length > 1) {
148                          // First we must separate the multiple keys if they exist.
149                          addField(ss[0], // String fieldname,
150                                   ss[1]); // String fieldvalue,
151                      }
152                  }
153                  ss_l=ss_and;
154              }
155 
156          }
157      }
158      // Because of UTF-8 issues this is double encoded in the browser.
159      // So decode it before it can be used
160      String newRequiredFields=req.getParameter("newrequiredfields");
161      if (newRequiredFields != null) {
162          String  [] newrequiredfields=decode(req.getParameter("newrequiredfields")).split(":");

164          // the ":" may not be there, check to be sure that there is actually 2 items.
