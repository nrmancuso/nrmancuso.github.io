41  /**
42   * Defines a mapping between a Java type and one or more JDBC {@linkplain java.sql.Types types}, as well
43   * as describing the in-memory semantics of the given java type (how do we check it for 'dirtiness', how do
44   * we copy values, etc).
45   * <p/>
46   * Application developers needing custom types can implement this interface (either directly or via subclassing an
47   * existing impl) or by the (slightly more stable, though more limited) {@link org.hibernate.usertype.UserType}
48   * interface.
49   * <p/>
50   * Implementations of this interface must certainly be thread-safe.  It is recommended that they be immutable as
51   * well, though that is difficult to achieve completely given the no-arg constructor requirement for custom types.
52   *
53   * @author Gavin King
54   * @author Steve Ebersole
55   */
56  public interface Type extends Serializable {
57  	/**
58  	 * Return true if the implementation is castable to {@link AssociationType}. This does not necessarily imply that
59  	 * the type actually represents an association.  Essentially a polymorphic version of
60  	 * {@code (type instanceof AssociationType.class)}
61  	 *
62  	 * @return True if this type is also an {@link AssociationType} implementor; false otherwise.
63  	 */
64  	public boolean isAssociationType();
65  
66  	/**
67  	 * Return true if the implementation is castable to {@link CollectionType}. Essentially a polymorphic version of
68  	 * {@code (type instanceof CollectionType.class)}
69  	 * <p/>
70  	 * A {@link CollectionType} is additionally an {@link AssociationType}; so if this method returns true,
71  	 * {@link #isAssociationType()} should also return true.
72  	 *
73  	 * @return True if this type is also an {@link CollectionType} implementor; false otherwise.
74  	 */
75  	public boolean isCollectionType();
76  
77  	/**
78  	 * Return true if the implementation is castable to {@link EntityType}. Essentially a polymorphic
79  	 * version of {@code (type instanceof EntityType.class)}.
80  	 * <p/>
81  	 * An {@link EntityType} is additionally an {@link AssociationType}; so if this method returns true,
82  	 * {@link #isAssociationType()} should also return true.
83  	 *
84  	 * @return True if this type is also an {@link EntityType} implementor; false otherwise.
85  	 */
86  	public boolean isEntityType();
87  
88  	/**
89  	 * Return true if the implementation is castable to {@link AnyType}. Essentially a polymorphic
90  	 * version of {@code (type instanceof AnyType.class)}.
91  	 * <p/>
92  	 * An {@link AnyType} is additionally an {@link AssociationType}; so if this method returns true,
93  	 * {@link #isAssociationType()} should also return true.
94  	 *
95  	 * @return True if this type is also an {@link AnyType} implementor; false otherwise.
96  	 */
97  	public boolean isAnyType();
98  
99  	/**
100 	 * Return true if the implementation is castable to {@link CompositeType}. Essentially a polymorphic
101 	 * version of {@code (type instanceof CompositeType.class)}.  A component type may own collections or
102 	 * associations and hence must provide certain extra functionality.
103 	 *
104 	 * @return True if this type is also an {@link CompositeType} implementor; false otherwise.
105 	 */
106 	public boolean isComponentType();
107

125 	 *

127 	 *
128 	 * @throws MappingException Generally indicates an issue accessing the passed mapping object.
129 	 */

132 	/**
133 	 * Return the column sizes dictated by this type.  For example, the mapping for a {@code char}/{@link Character} would

135 	 * <p/>
136 	 * NOTE: The number of elements in this array matches the return from {@link #getColumnSpan}.
137 	 *
138 	 * @param mapping The mapping object :/

142 	 *
143 	 * @throws MappingException Generally indicates an issue accessing the passed mapping object.
144 	 */
145 	public Size[] dictatedSizes(Mapping mapping) throws MappingException;
146 
147 	/**

149 	 * {@link #dictatedSizes} were given).
150 	 * <p/>
151 	 * NOTE: The number of elements in this array matches the return from {@link #getColumnSpan}.
152 	 *
153 	 * @param mapping The mapping object :/
154 	 * @todo Would be much much better to have this aware of Dialect once the service/metamodel split is done

157 	 *
158 	 * @throws MappingException Generally indicates an issue accessing the passed mapping object.
159 	 */
160 	public Size[] defaultSizes(Mapping mapping) throws MappingException;

178 	/**
179 	 * Compare two instances of the class mapped by this type for persistence "equality" (equality of persistent
180 	 * state) taking a shortcut for entity references.
181 	 * <p/>
182 	 * For most types this should equate to an {@link Object#equals equals} check on the values.  For associations
183 	 * the implication is a bit different.  For most types it is conceivable to simply delegate to {@link #isEqual}
184 	 *
185 	 * @param x The first value
186 	 * @param y The second value
187 	 *
188 	 * @return True if there are considered the same (see discussion above).
189 	 *
190 	 * @throws HibernateException A problem occurred performing the comparison
191 	 */
192 	public boolean isSame(Object x, Object y) throws HibernateException;
193 
194 	/**
195 	 * Compare two instances of the class mapped by this type for persistence "equality" (equality of persistent
196 	 * state).
197 	 * <p/>
198 	 * This should always equate to some form of comparison of the value's internal state.  As an example, for
199 	 * something like a date the comparison should be based on its internal "time" state based on the specific portion
200 	 * it is meant to represent (timestamp, date, time).
201 	 *
202 	 * @param x The first value
203 	 * @param y The second value
204 	 *
205 	 * @return True if there are considered equal (see discussion above).
206 	 *
207 	 * @throws HibernateException A problem occurred performing the comparison
208 	 */
209 	public boolean isEqual(Object x, Object y) throws HibernateException;
210 
211 	/**
212 	 * Compare two instances of the class mapped by this type for persistence "equality" (equality of persistent
213 	 * state).
214 	 * <p/>
215 	 * This should always equate to some form of comparison of the value's internal state.  As an example, for
216 	 * something like a date the comparison should be based on its internal "time" state based on the specific portion
217 	 * it is meant to represent (timestamp, date, time).
218 	 *
219 	 * @param x The first value
220 	 * @param y The second value
221 	 * @param factory The session factory
222 	 *
223 	 * @return True if there are considered equal (see discussion above).
224 	 *
225 	 * @throws HibernateException A problem occurred performing the comparison
226 	 */
227 	public boolean isEqual(Object x, Object y, SessionFactoryImplementor factory) throws HibernateException;

291 	/**
292 	 * Has the value been modified compared to the current database state?  The difference between this
293 	 * and the {@link #isDirty} methods is that here we need to account for "partially" built values.  This is really

298 	 * @param currentState the current state of the object

301 	 *
302 	 * @return true if the field has been modified
303 	 *

305 	 */
