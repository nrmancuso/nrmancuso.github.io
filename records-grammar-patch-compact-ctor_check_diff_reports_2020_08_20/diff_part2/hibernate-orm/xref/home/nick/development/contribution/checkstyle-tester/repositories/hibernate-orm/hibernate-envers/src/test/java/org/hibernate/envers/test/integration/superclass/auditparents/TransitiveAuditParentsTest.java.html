1   package org.hibernate.envers.test.integration.superclass.auditparents;
2   
3   import java.util.Set;
4   import javax.persistence.EntityManager;
5   
6   import org.junit.Assert;
7   import org.junit.Test;
8   
9   import org.hibernate.ejb.Ejb3Configuration;
10  import org.hibernate.envers.Audited;
11  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
12  import org.hibernate.envers.test.Priority;
13  import org.hibernate.envers.test.tools.TestTools;
14  import org.hibernate.mapping.Column;
15  import org.hibernate.mapping.Table;
16  
17  /**
18   * Tests mapping of child entity which parent declares one of its ancestors as audited with {@link Audited#auditParents()}
19   * property. Child entity may mark explicitly its parent as audited or not.
20   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
21   */
22  public class TransitiveAuditParentsTest extends BaseEnversJPAFunctionalTestCase {
23      private long childImpTransId = 1L;
24      private long childExpTransId = 2L;
25      
26      @Override
27      public void configure(Ejb3Configuration cfg) {
28          cfg.addAnnotatedClass(MappedGrandparentEntity.class);
29          cfg.addAnnotatedClass(TransitiveParentEntity.class);
30          cfg.addAnnotatedClass(ImplicitTransitiveChildEntity.class);
31          cfg.addAnnotatedClass(ExplicitTransitiveChildEntity.class);
32      }
33  
34      @Test
35      @Priority(10)
36      public void initData() {
37          EntityManager em = getEntityManager();
38  
39          // Revision 1
40          em.getTransaction().begin();
41          em.persist(new ImplicitTransitiveChildEntity(childImpTransId, "grandparent 1", "notAudited 1", "parent 1", "child 1"));
42          em.getTransaction().commit();
43  
44          // Revision 2
45          em.getTransaction().begin();
46          em.persist(new ExplicitTransitiveChildEntity(childExpTransId, "grandparent 2", "notAudited 2", "parent 2", "child 2"));
47          em.getTransaction().commit();
48      }
49  
50      @Test
51      public void testCreatedAuditTables() {
52          Table explicitTransChildTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.superclass.auditparents.ExplicitTransitiveChildEntity_AUD").getTable();
53          checkTableColumns(TestTools.makeSet("child", "parent", "grandparent", "id"), TestTools.makeSet("notAudited"), explicitTransChildTable);
54  
55          Table implicitTransChildTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.superclass.auditparents.ImplicitTransitiveChildEntity_AUD").getTable();
56          checkTableColumns(TestTools.makeSet("child", "parent", "grandparent", "id"), TestTools.makeSet("notAudited"), implicitTransChildTable);
57      }
58  
59      private void checkTableColumns(Set<String> expectedColumns, Set<String> unexpectedColumns, Table table) {
60          for (String columnName : expectedColumns) {
61              // Check whether expected column exists.
62              Assert.assertNotNull(table.getColumn(new Column(columnName)));
63          }
64          for (String columnName : unexpectedColumns) {
65              // Check whether unexpected column does not exist.
66              Assert.assertNull(table.getColumn(new Column(columnName)));
67          }
68      }
69  
70      @Test
71      public void testImplicitTransitiveAuditParents() {
72          // expectedChild.notAudited shall be null, because it is not audited.
73          ImplicitTransitiveChildEntity expectedChild = new ImplicitTransitiveChildEntity(childImpTransId, "grandparent 1", null, "parent 1", "child 1");
74          ImplicitTransitiveChildEntity child = getAuditReader().find(ImplicitTransitiveChildEntity.class, childImpTransId, 1);
75          Assert.assertEquals(expectedChild, child);
76      }
77  
78      @Test
79      public void testExplicitTransitiveAuditParents() {
80          // expectedChild.notAudited shall be null, because it is not audited.
81          ExplicitTransitiveChildEntity expectedChild = new ExplicitTransitiveChildEntity(childExpTransId, "grandparent 2", null, "parent 2", "child 2");
82          ExplicitTransitiveChildEntity child = getAuditReader().find(ExplicitTransitiveChildEntity.class, childExpTransId, 2);
83          Assert.assertEquals(expectedChild, child);
84      }
85  }
