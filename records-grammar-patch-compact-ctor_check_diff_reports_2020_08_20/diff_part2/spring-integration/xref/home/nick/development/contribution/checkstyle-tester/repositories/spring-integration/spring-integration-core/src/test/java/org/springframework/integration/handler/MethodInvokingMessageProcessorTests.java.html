522 	private void optionalAndRequiredWithAnnotatedMethodGuts(MethodInvokingMessageProcessor processor,
523 			boolean compiled) {
524 
525 		processor.setBeanFactory(mock(BeanFactory.class));
526 		Message<String> message = MessageBuilder.withPayload("foo")
527 				.setHeader("num", 42)
528 				.build();
529 		Object result = processor.processMessage(message);
530 		assertThat(result).isEqualTo("null42");
531 		message = MessageBuilder.withPayload("foo")
532 				.setHeader("prop", "bar")
533 				.setHeader("num", 42)
534 				.build();
535 		result = processor.processMessage(message);
536 		assertThat(result).isEqualTo("bar42");
537 		message = MessageBuilder.withPayload("foo")
538 				.setHeader("prop", "bar")
539 				.build();
540 		try {
541 			result = processor.processMessage(message);
542 			fail("Expected MessageHandlingException");
543 		}
544 		catch (MessageHandlingException e) {
545 			if (compiled) {
546 				assertThat(e.getCause().getMessage()).isEqualTo("required header not available: num");
547 			}
548 			else {
549 				assertThat(e.getCause().getCause().getMessage()).isEqualTo("required header not available: num");
550 			}
551 		}
