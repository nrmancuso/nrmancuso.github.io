94          /**
95           * Builds a map pool-name => pool-index from the SnmpTableHandler
96           * of the JvmMemPoolTable.
97           **/
98          private static Map<String, SnmpOid> buildPoolIndexMap(SnmpTableHandler handler) {
99              // optimization...
100             if (handler instanceof SnmpCachedData)
101                 return buildPoolIndexMap((SnmpCachedData)handler);
102 
103             // not optimizable... too bad.
104             final Map<String, SnmpOid> m = new HashMap<String, SnmpOid>();
105             SnmpOid index=null;
106             while ((index = handler.getNext(index))!=null) {
107                 final MemoryPoolMXBean mpm =
108                     (MemoryPoolMXBean)handler.getData(index);
109                 if (mpm == null) continue;
110                 final String name = mpm.getName();
111                 if (name == null) continue;
112                 m.put(name,index);
113             }
114             return m;
115         }

117         /**
118          * Builds a map pool-name => pool-index from the SnmpTableHandler
119          * of the JvmMemPoolTable.
120          * Optimized algorithm.
121          **/
122         private static Map<String, SnmpOid> buildPoolIndexMap(SnmpCachedData cached) {
123             if (cached == null) return Collections.emptyMap();
124             final SnmpOid[] indexes = cached.indexes;
125             final Object[]  datas   = cached.datas;
126             final int len = indexes.length;
127             final Map<String, SnmpOid> m = new HashMap<String, SnmpOid>(len);
128             for (int i=0; i<len; i++) {
129                 final SnmpOid index = indexes[i];
130                 if (index == null) continue;
131                 final MemoryPoolMXBean mpm =
132                     (MemoryPoolMXBean)datas[i];
133                 if (mpm == null) continue;
134                 final String name = mpm.getName();
135                 if (name == null) continue;
136                 m.put(name,index);
137             }
138             return m;
139         }

141         /**
142          * Return a table handler that holds the jvmMemManagerTable table data.
143          * This method return the cached table data if it is still
144          * valid, recompute it and cache the new value if it's not.
145          * If it needs to recompute the cached data, it first
146          * try to obtain the list of memory managers from the request
147          * contextual cache, and if it is not found, it calls
148          * <code>ManagementFactory.getMemoryMBean().getMemoryManagers()</code>
149          * and caches the value.
150          * This ensures that
151          * <code>ManagementFactory.getMemoryMBean().getMemoryManagers()</code>
152          * is not called more than once per request, thus ensuring a
153          * consistent view of the table.
154          **/
155         protected SnmpCachedData updateCachedDatas(Object userData) {
156             // Get the MemoryManager     table
157             final SnmpTableHandler mmHandler =
158                 meta.getManagerHandler(userData);
159 
160             // Get the MemoryPool        table
161             final SnmpTableHandler mpHandler =
162                 meta.getPoolHandler(userData);
163 
164             // Time stamp for the cache
165             final long time = System.currentTimeMillis();
166 
167             //     Build a Map poolname -> index
168             final Map poolIndexMap = buildPoolIndexMap(mpHandler);
169 
170             // For each memory manager, get the list of memory pools
171             // For each memory pool, find its index in the memory pool table
172             // Create a row in the relation table.
173             final TreeMap<SnmpOid, Object> table =
174                     new TreeMap<SnmpOid, Object>(SnmpCachedData.oidComparator);
175             updateTreeMap(table,userData,mmHandler,mpHandler,poolIndexMap);
176 
177             return new SnmpCachedData(time,table);
178         }

181         /**
182          * Get the list of memory pool associated with the
183          * given MemoryManagerMXBean.
184          **/
185         protected String[] getMemoryPools(Object userData,
186                                       MemoryManagerMXBean mmm, long mmarc) {
187             final String listTag =
188                 "JvmMemManager." + mmarc + ".getMemoryPools";
189 
190             String[] result=null;
191             if (userData instanceof Map) {
192                 result = (String[])((Map)userData).get(listTag);
193                 if (result != null) return result;
194             }
195 
196             if (mmm!=null) {
197                 result = mmm.getMemoryPoolNames();
198             }
199             if ((result!=null)&&(userData instanceof Map)) {
200                 Map<Object, Object> map = Util.cast(userData);
201                 map.put(listTag,result);
202             }
203 
204             return result;
205         }

207         protected void updateTreeMap(TreeMap<SnmpOid, Object> table, Object userData,
208                                      MemoryManagerMXBean mmm,
209                                      SnmpOid mmIndex,
210                                      Map poolIndexMap) {
211 
212             // The MemoryManager index is an int, so it's the first
213             // and only subidentifier.
214             final long mmarc;
215             try {
216                 mmarc = mmIndex.getOidArc(0);
217             } catch (SnmpStatusException x) {
218                 log.debug("updateTreeMap",
219                           "Bad MemoryManager OID index: "+mmIndex);
220                 log.debug("updateTreeMap",x);
221                 return;
222             }
223 
224 
225             // Cache this in userData + get it from cache?
226             final String[] mpList = getMemoryPools(userData,mmm,mmarc);
227             if (mpList == null || mpList.length < 1) return;
228 
229             final String mmmName = mmm.getName();
230             for (int i = 0; i < mpList.length; i++) {
231                 final String mpmName = mpList[i];
232                 if (mpmName == null) continue;
233                 final SnmpOid mpIndex = (SnmpOid)poolIndexMap.get(mpmName);
234                 if (mpIndex == null) continue;
235 
236                 // The MemoryPool index is an int, so it's the first
237                 // and only subidentifier.
238                 final long mparc;
239                 try {
240                     mparc  = mpIndex.getOidArc(0);
241                 } catch (SnmpStatusException x) {
242                     log.debug("updateTreeMap","Bad MemoryPool OID index: " +
243                           mpIndex);
244                     log.debug("updateTreeMap",x);
245                     continue;
246                 }
247                 // The MemoryMgrPoolRel table indexed is composed
248                 // of the MemoryManager index, to which the MemoryPool
249                 // index is appended.
250                 final long[] arcs = { mmarc, mparc };
251 
252                 final SnmpOid index = new SnmpOid(arcs);
253 
254                 table.put(index, new JvmMemMgrPoolRelEntryImpl(mmmName,
255                                                                mpmName,
256                                                                (int)mmarc,
257                                                                (int)mparc));
258             }
259         }

261         protected void updateTreeMap(TreeMap<SnmpOid, Object> table, Object userData,
262                                      SnmpTableHandler mmHandler,
263                                      SnmpTableHandler mpHandler,
264                                      Map poolIndexMap) {
265             if (mmHandler instanceof SnmpCachedData) {
266                 updateTreeMap(table,userData,(SnmpCachedData)mmHandler,
267                               mpHandler,poolIndexMap);
268                 return;
269             }
270 
271             SnmpOid mmIndex=null;
272             while ((mmIndex = mmHandler.getNext(mmIndex))!=null) {
273                 final MemoryManagerMXBean mmm =
274                     (MemoryManagerMXBean)mmHandler.getData(mmIndex);
275                 if (mmm == null) continue;
276                 updateTreeMap(table,userData,mmm,mmIndex,poolIndexMap);
277             }
278         }
279 
280         protected void updateTreeMap(TreeMap<SnmpOid, Object> table, Object userData,
281                                      SnmpCachedData mmHandler,
282                                      SnmpTableHandler mpHandler,
283                                      Map poolIndexMap) {
284 
285             final SnmpOid[] indexes = mmHandler.indexes;
286             final Object[]  datas   = mmHandler.datas;
287             final int size = indexes.length;
288             for (int i=size-1; i>-1; i--) {
289                 final MemoryManagerMXBean mmm =
290                     (MemoryManagerMXBean)datas[i];
291                 if (mmm == null) continue;
292                 updateTreeMap(table,userData,mmm,indexes[i],poolIndexMap);
293             }
294         }

363     // See com.sun.jmx.snmp.agent.SnmpMibTable
364     protected SnmpOid getNextOid(SnmpOid oid, Object userData)
365         throws SnmpStatusException {
366         final boolean dbg = log.isDebugOn();
367         if (dbg) log.debug("getNextOid", "previous=" + oid);
368 
369 
370         // Get the data handler.
371         //
372         SnmpTableHandler handler = getHandler(userData);
373         if (handler == null) {
374             // This should never happen.
375             // If we get here it's a bug.
376             //
377             if (dbg) log.debug("getNextOid", "handler is null!");
378             throw new SnmpStatusException(SnmpStatusException.noSuchInstance);
379         }
380 
381         // Get the next oid
382         //
383         final SnmpOid next = handler.getNext(oid);
384         if (dbg) log.debug("getNextOid", "next=" + next);
385 
386         // if next is null: we reached the end of the table.
387         //
388         if (next == null)
389             throw new SnmpStatusException(SnmpStatusException.noSuchInstance);
390 
391         return next;
392     }

396     protected boolean contains(SnmpOid oid, Object userData) {
397 
398         // Get the handler.
399         //
400         SnmpTableHandler handler = getHandler(userData);
401 
402         // handler should never be null.
403         //
404         if (handler == null)
405             return false;
406 
407         return handler.contains(oid);

410     // See com.sun.jmx.snmp.agent.SnmpMibTable
411     public Object getEntry(SnmpOid oid)
412         throws SnmpStatusException {
413 
414         if (oid == null || oid.getLength() < 2)
415             throw new SnmpStatusException(SnmpStatusException.noSuchInstance);
416 
417         // Get the request contextual cache (userData).
418         //
419         final Map<Object, Object> m = JvmContextFactory.getUserData();
420 
421         // We know in the case of this table that the index is composed
422         // of two integers,
423         //  o The MemoryManager is the first  OID arc of the index OID.
424         //  o The MemoryPool    is the second OID arc of the index OID.
425         //
426         final long   mgrIndex     = oid.getOidArc(0);
427         final long   poolIndex    = oid.getOidArc(1);
428 
429         // We're going to use this name to store/retrieve the entry in
430         // the request contextual cache.
431         //
432         // Revisit: Probably better programming to put all these strings
433         //          in some interface.
434         //
435         final String entryTag = ((m==null)?null:
436                                  ("JvmMemMgrPoolRelTable.entry." +
437                                   mgrIndex + "." + poolIndex));
438 
439         // If the entry is in the cache, simply return it.
440         //
441         if (m != null) {
442             final Object entry = m.get(entryTag);
443             if (entry != null) return entry;
444         }
445 
446         // The entry was not in the cache, make a new one.
447         //
448         // Get the data hanler.
449         //
450         SnmpTableHandler handler = getHandler(m);
451 
452         // handler should never be null.
453         //
454         if (handler == null)
455             throw new SnmpStatusException(SnmpStatusException.noSuchInstance);
456 
457         // Get the data associated with our entry.
458         //
459         final Object data = handler.getData(oid);
460 
461         // data may be null if the OID we were given is not valid.
462         //
463         if (!(data instanceof JvmMemMgrPoolRelEntryImpl))
464             throw new SnmpStatusException(SnmpStatusException.noSuchInstance);
465 
466         // make the new entry (transient object that will be kept only
467         // for the duration of the request.
468         //
469         final Object entry = (JvmMemMgrPoolRelEntryImpl)data;
470         // XXXXX Revisit
471         // new JvmMemMgrPoolRelEntryImpl((MemoryManagerMXBean)data,
472         //                                (int)mgrIndex,(int)poolIndex);
473 
474         // Put the entry in the cache in case we need it later while processing
475         // the request.
476         //
477         if (m != null && entry != null) {
478             m.put(entryTag,entry);
479         }
480 
481         return entry;
482     }

484     /**
485      * Get the SnmpTableHandler that holds the jvmMemManagerTable data.
486      * First look it up in the request contextual cache, and if it is
487      * not found, obtain it from the weak cache.
488      * <br>The request contextual cache will be released at the end of the
489      * current requests, and is used only to process this request.
490      * <br>The weak cache is shared by all requests, and is only
491      * recomputed when it is found to be obsolete.
492      * <br>Note that the data put in the request contextual cache is
493      *     never considered to be obsolete, in order to preserve data
494      *     coherency.
495      **/
496     protected SnmpTableHandler getHandler(Object userData) {
497         final Map<Object, Object> m;
498         if (userData instanceof Map) m=Util.cast(userData);
499         else m=null;
500 
501         // Look in the contextual cache.
502         if (m != null) {
503             final SnmpTableHandler handler =
504                 (SnmpTableHandler)m.get("JvmMemMgrPoolRelTable.handler");
505             if (handler != null) return handler;
506         }
507 
508         // No handler in contextual cache, make a new one.
509         final SnmpTableHandler handler = cache.getTableHandler();
510 
511         if (m != null && handler != null )
512             m.put("JvmMemMgrPoolRelTable.handler",handler);
513 
514         return handler;
515     }
