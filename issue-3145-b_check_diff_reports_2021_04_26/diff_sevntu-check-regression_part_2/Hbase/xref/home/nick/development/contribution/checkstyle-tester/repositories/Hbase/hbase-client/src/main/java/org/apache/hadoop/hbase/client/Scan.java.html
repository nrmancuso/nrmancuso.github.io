46  /**
47   * Used to perform Scan operations.
48   * <p>
49   * All operations are identical to {@link Get} with the exception of
50   * instantiation.  Rather than specifying a single row, an optional startRow
51   * and stopRow may be defined.  If rows are not specified, the Scanner will
52   * iterate over all rows.
53   * <p>
54   * To scan everything for each row, instantiate a Scan object.
55   * <p>
56   * To modify scanner caching for just this scan, use {@link #setCaching(int) setCaching}.
57   * If caching is NOT set, we will use the caching value of the hosting {@link Table}.
58   * In addition to row caching, it is possible to specify a
59   * maximum result size, using {@link #setMaxResultSize(long)}. When both are used,
60   * single server requests are limited by either number of rows or maximum result size, whichever
61   * limit comes first.
62   * <p>
63   * To further define the scope of what to get when scanning, perform additional
64   * methods as outlined below.
65   * <p>
66   * To get all columns from specific families, execute {@link #addFamily(byte[]) addFamily}
67   * for each family to retrieve.
68   * <p>
69   * To get specific columns, execute {@link #addColumn(byte[], byte[]) addColumn}
70   * for each column to retrieve.
71   * <p>
72   * To only retrieve columns within a specific range of version timestamps,
73   * execute {@link #setTimeRange(long, long) setTimeRange}.
74   * <p>
75   * To only retrieve columns with a specific timestamp, execute
76   * {@link #setTimeStamp(long) setTimestamp}.
77   * <p>
78   * To limit the number of versions of each column to be returned, execute
79   * {@link #setMaxVersions(int) setMaxVersions}.
80   * <p>
81   * To limit the maximum number of values returned for each call to next(),
82   * execute {@link #setBatch(int) setBatch}.
83   * <p>
84   * To add a filter, execute {@link #setFilter(org.apache.hadoop.hbase.filter.Filter) setFilter}.
85   * <p>
86   * Expert: To explicitly disable server-side block caching for this scan,
87   * execute {@link #setCacheBlocks(boolean)}.
88   */
89  @InterfaceAudience.Public

102    * Partial {@link Result}s are {@link Result}s must be combined to form a complete {@link Result}.
103    * The {@link Result}s had to be returned in fragments (i.e. as partials) because the size of the
104    * cells in the row exceeded max result size on the server. Typically partial results will be
105    * combined client side into complete results before being delivered to the caller. However, if
106    * this flag is set, the caller is indicating that they do not mind seeing partial results (i.e.
107    * they understand that the results returned from the Scanner may only represent part of a
108    * particular row). In such a case, any attempt to combine the partials into a complete result on
109    * the client side will be skipped, and the caller will be able to see the exact results returned
110    * from the server.

157   /**
158    * Set it true for small scan to get better performance
159    *
160    * Small scan should use pread and big scan can use seek + read
161    *
162    * seek + read is fast but can cause two problem (1) resource contention (2)
163    * cause too much network io
164    *
165    * [89-fb] Using pread for non-compaction read request
166    * https://issues.apache.org/jira/browse/HBASE-7266
167    *
168    * On the other hand, if setting it true, we would do
169    * openScanner,next,closeScanner in one RPC call. It means the better
170    * performance for small scan. [HBASE-9488].
171    *
172    * Generally, if the scan range is within one data block(64KB), it could be
173    * considered as a small scan.

210   /**
211    * Creates a new instance of this class while copying all values.
212    *
213    * @param scan  The scan instance to copy from.
214    * @throws IOException When copying the values fails.
215    */
216   public Scan(Scan scan) throws IOException {
217     startRow = scan.getStartRow();
218     stopRow  = scan.getStopRow();
219     maxVersions = scan.getMaxVersions();
220     batch = scan.getBatch();
221     storeLimit = scan.getMaxResultsPerColumnFamily();
222     storeOffset = scan.getRowOffsetPerColumnFamily();
223     caching = scan.getCaching();
224     maxResultSize = scan.getMaxResultSize();
225     cacheBlocks = scan.getCacheBlocks();
226     getScan = scan.isGetScan();
227     filter = scan.getFilter(); // clone?
228     loadColumnFamiliesOnDemand = scan.getLoadColumnFamiliesOnDemandValue();
229     consistency = scan.getConsistency();
230     reversed = scan.isReversed();
231     small = scan.isSmall();
232     TimeRange ctr = scan.getTimeRange();
233     tr = new TimeRange(ctr.getMin(), ctr.getMax());
234     Map<byte[], NavigableSet<byte[]>> fams = scan.getFamilyMap();
235     for (Map.Entry<byte[],NavigableSet<byte[]>> entry : fams.entrySet()) {
236       byte [] fam = entry.getKey();
237       NavigableSet<byte[]> cols = entry.getValue();
238       if (cols != null && cols.size() > 0) {
239         for (byte[] col : cols) {
240           addColumn(fam, col);
241         }
242       } else {
243         addFamily(fam);
244       }
245     }
246     for (Map.Entry<String, byte[]> attr : scan.getAttributesMap().entrySet()) {
247       setAttribute(attr.getKey(), attr.getValue());
248     }
249   }

379   /**
380    * <p>Set a filter (using stopRow and startRow) so the result set only contains rows where the
381    * rowKey starts with the specified prefix.</p>
382    * <p>This is a utility method that converts the desired rowPrefix into the appropriate values
383    * for the startRow and stopRow to achieve the desired result.</p>
384    * <p>This can safely be used in combination with setFilter.</p>
385    * <p><b>NOTE: Doing a {@link #setStartRow(byte[])} and/or {@link #setStopRow(byte[])}
386    * after this method will yield undefined results.</b></p>
387    * @param rowPrefix the prefix all rows must start with. (Set <i>null</i> to remove the filter.)
388    * @return this
389    */
390   public Scan setRowPrefixFilter(byte[] rowPrefix) {
391     if (rowPrefix == null) {
392       setStartRow(HConstants.EMPTY_START_ROW);
393       setStopRow(HConstants.EMPTY_END_ROW);
394     } else {
395       this.setStartRow(rowPrefix);
396       this.setStopRow(calculateTheClosestNextRowKeyForPrefix(rowPrefix));
397     }
398     return this;
399   }
400 
401   /**
402    * <p>When scanning for a prefix the scan should stop immediately after the the last row that
403    * has the specified prefix. This method calculates the closest next rowKey immediately following
404    * the given rowKeyPrefix.</p>
405    * <p><b>IMPORTANT: This converts a rowKey<u>Prefix</u> into a rowKey</b>.</p>
406    * <p>If the prefix is an 'ASCII' string put into a byte[] then this is easy because you can
407    * simply increment the last byte of the array.
408    * But if your application uses real binary rowids you may run into the scenario that your
409    * prefix is something like:</p>
410    * &nbsp;&nbsp;&nbsp;<b>{ 0x12, 0x23, 0xFF, 0xFF }</b><br/>
411    * Then this stopRow needs to be fed into the actual scan<br/>
412    * &nbsp;&nbsp;&nbsp;<b>{ 0x12, 0x24 }</b> (Notice that it is shorter now)<br/>
413    * This method calculates the correct stop row value for this usecase.
414    *
415    * @param rowKeyPrefix the rowKey<u>Prefix</u>.
416    * @return the closest next rowKey immediately following the given rowKeyPrefix.
417    */
418   private byte[] calculateTheClosestNextRowKeyForPrefix(byte[] rowKeyPrefix) {
419     // Essentially we are treating it like an 'unsigned very very long' and doing +1 manually.
420     // Search for the place where the trailing 0xFFs start
421     int offset = rowKeyPrefix.length;
422     while (offset > 0) {
423       if (rowKeyPrefix[offset - 1] != (byte) 0xFF) {
424         break;
425       }
426       offset--;
427     }
428 
429     if (offset == 0) {
430       // We got an 0xFFFF... (only FFs) stopRow value which is
431       // the last possible prefix before the end of the table.
432       // So set it to stop at the 'end of the table'
433       return HConstants.EMPTY_END_ROW;
434     }
435 
436     // Copy the right length of the original
437     byte[] newStopRow = Arrays.copyOfRange(rowKeyPrefix, 0, offset);
438     // And increment the last one
439     newStopRow[newStopRow.length - 1]++;
440     return newStopRow;
441   }

715   /**
716    * Set the value indicating whether loading CFs on demand should be allowed (cluster
717    * default is false). On-demand CF loading doesn't load column families until necessary, e.g.
718    * if you filter on one column, the other column family data will be loaded only for the rows
719    * that are included in result, not all rows like in normal case.
720    * With column-specific filters, like SingleColumnValueFilter w/filterIfMissing == true,
721    * this can deliver huge perf gains when there's a cf with lots of data; however, it can
722    * also lead to some inconsistent results, as follows:
723    * - if someone does a concurrent update to both column families in question you may get a row
724    *   that never existed, e.g. for { rowKey = 5, { cat_videos => 1 }, { video => "my cat" } }
725    *   someone puts rowKey 5 with { cat_videos => 0 }, { video => "my dog" }, concurrent scan
726    *   filtering on "cat_videos == 1" can get { rowKey = 5, { cat_videos => 1 },
727    *   { video => "my dog" } }.
728    * - if there's a concurrent split and you have more than 2 column families, some rows may be
729    *   missing some column families.
730    */
731   public Scan setLoadColumnFamiliesOnDemand(boolean value) {
732     this.loadColumnFamiliesOnDemand = value;
733     return this;
734   }

751   /**
752    * Compile the table and column family (i.e. schema) information
753    * into a String. Useful for parsing and aggregation by debugging,
754    * logging, and administration tools.
755    * @return Map
756    */
757   @Override
758   public Map<String, Object> getFingerprint() {
759     Map<String, Object> map = new HashMap<String, Object>();
760     List<String> families = new ArrayList<String>();
761     if(this.familyMap.size() == 0) {
762       map.put("families", "ALL");
763       return map;
764     } else {
765       map.put("families", families);
766     }
767     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
768         this.familyMap.entrySet()) {
769       families.add(Bytes.toStringBinary(entry.getKey()));
770     }
771     return map;
772   }
773 
774   /**
775    * Compile the details beyond the scope of getFingerprint (row, columns,
776    * timestamps, etc.) into a Map along with the fingerprinted information.
777    * Useful for debugging, logging, and administration tools.
778    * @param maxCols a limit on the number of columns output prior to truncation
779    * @return Map
780    */
781   @Override
782   public Map<String, Object> toMap(int maxCols) {
783     // start with the fingerpring map and build on top of it
784     Map<String, Object> map = getFingerprint();
785     // map from families to column list replaces fingerprint's list of families
786     Map<String, List<String>> familyColumns =
787       new HashMap<String, List<String>>();
788     map.put("families", familyColumns);
789     // add scalar information first
790     map.put("startRow", Bytes.toStringBinary(this.startRow));
791     map.put("stopRow", Bytes.toStringBinary(this.stopRow));
792     map.put("maxVersions", this.maxVersions);
793     map.put("batch", this.batch);
794     map.put("caching", this.caching);
795     map.put("maxResultSize", this.maxResultSize);
796     map.put("cacheBlocks", this.cacheBlocks);
797     map.put("loadColumnFamiliesOnDemand", this.loadColumnFamiliesOnDemand);
798     List<Long> timeRange = new ArrayList<Long>();
799     timeRange.add(this.tr.getMin());
800     timeRange.add(this.tr.getMax());
801     map.put("timeRange", timeRange);
802     int colCount = 0;
803     // iterate through affected families and list out up to maxCols columns
804     for (Map.Entry<byte [], NavigableSet<byte[]>> entry :
805       this.familyMap.entrySet()) {
806       List<String> columns = new ArrayList<String>();
807       familyColumns.put(Bytes.toStringBinary(entry.getKey()), columns);
808       if(entry.getValue() == null) {
809         colCount++;
810         --maxCols;
811         columns.add("ALL");
812       } else {
813         colCount += entry.getValue().size();
814         if (maxCols <= 0) {
815           continue;
816         }
817         for (byte [] column : entry.getValue()) {
818           if (--maxCols <= 0) {
819             continue;
820           }
821           columns.add(Bytes.toStringBinary(column));
822         }
823       }
824     }
825     map.put("totalColumns", colCount);
826     if (this.filter != null) {
827       map.put("filter", this.filter.toString());
828     }
829     // add the id if set
830     if (getId() != null) {
831       map.put("id", getId());
832     }
833     return map;
834   }

862    * Set whether this scan is a small scan
863    * <p>
864    * Small scan should use pread and big scan can use seek + read
865    *
866    * seek + read is fast but can cause two problem (1) resource contention (2)

873    * openScanner,next,closeScanner in one RPC call. It means the better
874    * performance for small scan. [HBASE-9488].

877    * considered as a small scan.

935    * Utility that creates a Scan that will do a  small scan in reverse from passed row
936    * looking for next closest row.
937    * @param row
938    * @param family
939    * @return An instance of Scan primed with passed <code>row</code> and <code>family</code> to
