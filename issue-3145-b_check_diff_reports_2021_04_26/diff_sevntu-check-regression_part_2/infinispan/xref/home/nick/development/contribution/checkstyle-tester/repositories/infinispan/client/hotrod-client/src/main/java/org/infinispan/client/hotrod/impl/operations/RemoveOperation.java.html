1   package org.infinispan.client.hotrod.impl.operations;
2   
3   import net.jcip.annotations.Immutable;
4   import org.infinispan.client.hotrod.Flag;
5   import org.infinispan.client.hotrod.impl.protocol.Codec;
6   import org.infinispan.client.hotrod.impl.transport.Transport;
7   import org.infinispan.client.hotrod.impl.transport.TransportFactory;
8   
9   import java.util.concurrent.atomic.AtomicInteger;
10  
11  /**
12   * Implement "remove" operation as described in <a href="http://community.jboss.org/wiki/HotRodProtocol">Hot Rod
13   * protocol specification</a>.
14   *
15   * @author Mircea.Markus@jboss.com
16   * @since 4.1
17   */
18  @Immutable
19  public class RemoveOperation extends AbstractKeyOperation<byte[]> {
20  
21     public RemoveOperation(Codec codec, TransportFactory transportFactory,
22              byte[] key, byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
23        super(codec, transportFactory, key, cacheName, topologyId, flags);
24     }
25  
26     @Override
27     public byte[] executeOperation(Transport transport) {
28        short status = sendKeyOperation(key, transport, REMOVE_REQUEST, REMOVE_RESPONSE);
29        byte[] result = returnPossiblePrevValue(transport, status);
30        if (status == KEY_DOES_NOT_EXIST_STATUS)
31           return null;
32  
33        return result; // NO_ERROR_STATUS
34     }
35  }
