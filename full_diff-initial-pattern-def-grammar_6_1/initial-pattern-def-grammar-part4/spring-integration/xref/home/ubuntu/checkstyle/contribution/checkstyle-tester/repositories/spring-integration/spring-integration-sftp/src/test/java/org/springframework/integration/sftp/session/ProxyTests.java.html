36  public class ProxyTests {
37  
38  	@Test
39  	@Disabled // TODO Use SftpTestSupport
40  	/*
41  	 * Needs host and account
42  	 */
43  	public void testSimpleConnect() {
44  		DefaultSftpSessionFactory sf = new DefaultSftpSessionFactory();
45  		sf.setHost("10.0.0.3");
46  		sf.setPort(22);
47  		sf.setUser("ftptest");
48  		sf.setPassword("ftptest");
49  		sf.setAllowUnknownKeys(true);
50  		sf.getSession().close();
51  	}
52  
53  	@Test
54  	@Disabled
55  	/*
56  	 * Needs host and account and...
57  	 * $ ssh -D 1080 -f -N gpr@10.0.0.3
58  	 */
59  	public void testProxyConnect() throws Exception {
60  		DefaultSftpSessionFactory sf = new DefaultSftpSessionFactory();
61  		JschProxyFactoryBean proxyFactoryBean = new JschProxyFactoryBean(JschProxyFactoryBean.Type.SOCKS5, "localhost",
62  				1080, "ftptest", "ftptest");
63  		proxyFactoryBean.afterPropertiesSet();
64  		sf.setHost("10.0.0.3");
65  		sf.setPort(22);
66  		sf.setUser("ftptest");
67  		sf.setPassword("ftptest");
68  		sf.setProxy(proxyFactoryBean.getObject());
69  		sf.setAllowUnknownKeys(true);
70  		sf.getSession().close();
71  	}
72  
73  	@Test
74  	public void testFactoryBean() throws Exception {
75  		JschProxyFactoryBean proxyFactoryBean = new JschProxyFactoryBean(JschProxyFactoryBean.Type.SOCKS5, "localhost",
76  				1080, "ftptest", "pass");
77  		proxyFactoryBean.afterPropertiesSet();
78  		Proxy proxy = proxyFactoryBean.getObject();
79  		assertProxy(proxy, ProxySOCKS5.class);
80  
81  		proxyFactoryBean = new JschProxyFactoryBean(JschProxyFactoryBean.Type.SOCKS4, "localhost",
82  				1080, "ftptest", "pass");
83  		proxyFactoryBean.afterPropertiesSet();
84  		proxy = proxyFactoryBean.getObject();
85  		assertProxy(proxy, ProxySOCKS4.class);
86  
87  		proxyFactoryBean = new JschProxyFactoryBean(JschProxyFactoryBean.Type.HTTP, "localhost",
88  				1080, "ftptest", "pass");
89  		proxyFactoryBean.afterPropertiesSet();
90  		proxy = proxyFactoryBean.getObject();
91  		assertProxy(proxy, ProxyHTTP.class);
92  	}
93  
94  	private void assertProxy(Proxy proxy, Class<? extends Proxy> clazz) {
95  		assertThat(proxy).isInstanceOf(clazz);
96  		assertThat(TestUtils.getPropertyValue(proxy, "user")).isEqualTo("ftptest");
97  		assertThat(TestUtils.getPropertyValue(proxy, "passwd")).isEqualTo("pass");
98  	}
