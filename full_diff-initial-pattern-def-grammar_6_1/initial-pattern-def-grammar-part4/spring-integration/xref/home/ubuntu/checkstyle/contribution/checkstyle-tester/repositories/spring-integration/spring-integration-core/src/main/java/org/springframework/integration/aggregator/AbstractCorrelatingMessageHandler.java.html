65  /**
66   * Abstract Message handler that holds a buffer of correlated messages in a
67   * {@link org.springframework.integration.store.MessageStore}.
68   * This class takes care of correlated groups of messages
69   * that can be completed in batches. It is useful for custom implementation of
70   * MessageHandlers that require correlation and is used as a base class for Aggregator -
71   * {@link AggregatingMessageHandler} and Resequencer - {@link ResequencingMessageHandler},
72   * or custom implementations requiring correlation.
73   * <p>
74   * To customize this handler inject {@link CorrelationStrategy},
75   * {@link ReleaseStrategy}, and {@link MessageGroupProcessor} implementations as
76   * you require.
77   * <p>
78   * By default the {@link CorrelationStrategy} will be a
79   * {@link HeaderAttributeCorrelationStrategy} and the {@link ReleaseStrategy} will be a
80   * {@link SequenceSizeReleaseStrategy}.
81   * <p>
82   * Use proper {@link CorrelationStrategy} for cases when same
83   * {@link org.springframework.integration.store.MessageStore} is used
84   * for multiple handlers to ensure uniqueness of message groups across handlers.
85   *
86   * @author Iwein Fuld
87   * @author Dave Syer
88   * @author Oleg Zhurakousky
89   * @author Gary Russell
90   * @author Artem Bilan
91   * @author David Liu
92   * @author Enrique Rodriguez
93   * @author Meherzad Lahewala
94   * @author Jayadev Sirimamilla
95   *
96   * @since 2.0
97   */
98  public abstract class AbstractCorrelatingMessageHandler extends AbstractMessageProducingHandler

149 	public AbstractCorrelatingMessageHandler(MessageGroupProcessor processor, MessageGroupStore store,
150 			CorrelationStrategy correlationStrategy, ReleaseStrategy releaseStrategy) {
151 
152 		Assert.notNull(processor, "'processor' must not be null");
153 		Assert.notNull(store, "'store' must not be null");
154 
155 		setMessageStore(store);
156 		this.outputProcessor = processor;
157 
158 		this.correlationStrategy =
159 				correlationStrategy == null
160 						? new HeaderAttributeCorrelationStrategy(IntegrationMessageHeaderAccessor.CORRELATION_ID)
161 						: correlationStrategy;
162 
163 		this.releaseStrategy =
164 				releaseStrategy == null
165 						? new SimpleSequenceSizeReleaseStrategy()
166 						: releaseStrategy;
167 
168 		this.releaseStrategySet = releaseStrategy != null;
169 		this.sequenceAware = this.releaseStrategy instanceof SequenceSizeReleaseStrategy;
170 	}

318 	@Override
319 	protected void onInit() {
320 		super.onInit();
321 		Assert.state(!(this.discardChannelName != null && this.discardChannel != null),
322 				"'discardChannelName' and 'discardChannel' are mutually exclusive.");
323 		BeanFactory beanFactory = getBeanFactory();
324 		if (beanFactory != null) {
325 			if (this.outputProcessor instanceof BeanFactoryAware) {
326 				((BeanFactoryAware) this.outputProcessor).setBeanFactory(beanFactory);
327 			}
328 			if (this.correlationStrategy instanceof BeanFactoryAware) {
329 				((BeanFactoryAware) this.correlationStrategy).setBeanFactory(beanFactory);
330 			}
331 			if (this.releaseStrategy instanceof BeanFactoryAware) {
332 				((BeanFactoryAware) this.releaseStrategy).setBeanFactory(beanFactory);
333 			}
334 		}
335 
336 		if (this.releasePartialSequences) {
337 			Assert.isInstanceOf(SequenceSizeReleaseStrategy.class, this.releaseStrategy, () ->
338 					"Release strategy of type [" + this.releaseStrategy.getClass().getSimpleName() +
339 							"] cannot release partial sequences. Use a SequenceSizeReleaseStrategy instead.");
340 			((SequenceSizeReleaseStrategy) this.releaseStrategy)
341 					.setReleasePartialSequences(this.releasePartialSequences);
342 		}
343 
344 		if (this.evaluationContext == null) {
345 			this.evaluationContext = ExpressionUtils.createStandardEvaluationContext(getBeanFactory());
346 		}
347 
348 		if (this.sequenceAware) {
349 			this.logger.warn("Using a SequenceSizeReleaseStrategy with large groups may not perform well, consider "
350 					+ "using a SimpleSequenceSizeReleaseStrategy");
351 		}
352 
353 		/*
354 		 * Disallow any further changes to the lock registry
355 		 * (checked in the setter).
356 		 */
357 		this.lockRegistrySet = true;
358 		this.forceReleaseProcessor = createGroupTimeoutProcessor();
359 	}

361 	private MessageGroupProcessor createGroupTimeoutProcessor() {
362 		MessageGroupProcessor processor = new ForceReleaseMessageGroupProcessor();
363 
364 		if (this.groupTimeoutExpression != null && !CollectionUtils.isEmpty(this.forceReleaseAdviceChain)) {
365 			ProxyFactory proxyFactory = new ProxyFactory(processor);
366 			this.forceReleaseAdviceChain.forEach(proxyFactory::addAdvice);
367 			return (MessageGroupProcessor) proxyFactory.getProxy(getApplicationContext().getClassLoader());
368 		}
369 		return processor;
370 	}

396 		if (channelName == null && this.discardChannel == null) {
397 			channelName = IntegrationContextUtils.NULL_CHANNEL_BEAN_NAME;
398 		}

405 					this.discardChannel = new NullChannel();

407 				else {
408 					throw ex;
409 				}
410 			}
411 			this.discardChannelName = null;
412 		}
413 		return this.discardChannel;
414 	}

452 	@Override
453 	protected void handleMessageInternal(Message<?> message) {
454 		Object correlationKey = this.correlationStrategy.getCorrelationKey(message);
455 		Assert.state(correlationKey != null,
456 				"Null correlation not allowed.  Maybe the CorrelationStrategy is failing?");
457 
458 		if (this.logger.isDebugEnabled()) {
459 			this.logger.debug("Handling message with correlationKey [" + correlationKey + "]: " + message);
460 		}
461 
462 		UUID groupIdUuid = UUIDConverter.getUUID(correlationKey);
463 		Lock lock = this.lockRegistry.obtain(groupIdUuid.toString());
464 
465 		boolean noOutput = true;
466 		try {
467 			lock.lockInterruptibly();
468 		}
469 		catch (InterruptedException e) {
470 			Thread.currentThread().interrupt();
471 			throw new MessageHandlingException(message, "Interrupted getting lock in the [" + this + ']', e);
472 		}
473 		try {
474 			noOutput = processMessageForGroup(message, correlationKey, groupIdUuid, lock);
475 		}
476 		finally {
477 			if (noOutput || !this.releaseLockBeforeSend) {
478 				lock.unlock();
479 			}
480 		}
481 	}
482 
483 	private boolean processMessageForGroup(Message<?> message, Object correlationKey, UUID groupIdUuid, Lock lock) {
484 		boolean noOutput = true;
485 		cancelScheduledFutureIfAny(correlationKey, groupIdUuid, true);
486 		MessageGroup messageGroup = this.messageStore.getMessageGroup(correlationKey);
487 		if (this.sequenceAware) {
488 			messageGroup = new SequenceAwareMessageGroup(messageGroup);
489 		}
490 
491 		if (!messageGroup.isComplete() && messageGroup.canAdd(message)) {
492 			if (this.logger.isTraceEnabled()) {
493 				this.logger.trace("Adding message to group [ " + messageGroup + "]");
494 			}
495 			messageGroup = store(correlationKey, message);
496 
497 			if (this.releaseStrategy.canRelease(messageGroup)) {
498 				Collection<Message<?>> completedMessages = null;
499 				try {
500 					noOutput = false;
501 					completedMessages = completeGroup(message, correlationKey, messageGroup, lock);
502 				}
503 				finally {
504 					// Possible clean (implementation dependency) up
505 					// even if there was an exception processing messages
506 					afterRelease(messageGroup, completedMessages);
507 				}
508 				if (!isExpireGroupsUponCompletion() && this.minimumTimeoutForEmptyGroups > 0) {
509 					removeEmptyGroupAfterTimeout(messageGroup, this.minimumTimeoutForEmptyGroups);
510 				}
511 			}
512 			else {
513 				scheduleGroupToForceComplete(messageGroup);
514 			}
515 		}
516 		else {
517 			noOutput = false;
518 			discardMessage(message, lock);
519 		}
520 		return noOutput;
521 	}

523 	private void cancelScheduledFutureIfAny(Object correlationKey, UUID groupIdUuid, boolean mayInterruptIfRunning) {
524 		ScheduledFuture<?> scheduledFuture = this.expireGroupScheduledFutures.remove(groupIdUuid);
525 		if (scheduledFuture != null) {
526 			boolean canceled = scheduledFuture.cancel(mayInterruptIfRunning);
527 			if (canceled && this.logger.isDebugEnabled()) {
528 				this.logger.debug("Cancel 'ScheduledFuture' for MessageGroup with Correlation Key [ "
529 						+ correlationKey + "].");
530 			}
531 		}
532 	}

538 	private void removeEmptyGroupAfterTimeout(MessageGroup messageGroup, long timeout) {
539 		Object groupId = messageGroup.getGroupId();
540 		UUID groupUuid = UUIDConverter.getUUID(groupId);
541 		ScheduledFuture<?> scheduledFuture =
542 				getTaskScheduler()
543 						.schedule(() -> {
544 							Lock lock = this.lockRegistry.obtain(groupUuid.toString());
545 
546 							try {
547 								lock.lockInterruptibly();
548 								try {
549 									this.expireGroupScheduledFutures.remove(groupUuid);
550 									/*
551 									 * Obtain a fresh state for group from the MessageStore,
552 									 * since it could be changed while we have waited for lock.
553 									 */
554 									MessageGroup groupNow = this.messageStore.getMessageGroup(groupUuid);
555 									boolean removeGroup = groupNow.size() == 0 &&
556 											groupNow.getLastModified()
557 													<= (System.currentTimeMillis() - this.minimumTimeoutForEmptyGroups);
558 									if (removeGroup) {
559 										if (this.logger.isDebugEnabled()) {
560 											this.logger.debug("Removing empty group: " + groupUuid);
561 										}
562 										remove(messageGroup);
563 									}
564 								}
565 								finally {
566 									lock.unlock();
567 								}
568 							}
569 							catch (InterruptedException e) {
570 								Thread.currentThread().interrupt();
571 								if (this.logger.isDebugEnabled()) {
572 									this.logger.debug("Thread was interrupted while trying to obtain lock."
573 											+ "Rescheduling empty MessageGroup [ " + groupId + "] for removal.");
574 								}
575 								removeEmptyGroupAfterTimeout(messageGroup, timeout);
576 							}
577 
578 						}, new Date(System.currentTimeMillis() + timeout));
579 
580 		if (this.logger.isDebugEnabled()) {
581 			this.logger.debug("Schedule empty MessageGroup [ " + groupId + "] for removal.");
582 		}
583 		this.expireGroupScheduledFutures.put(groupUuid, scheduledFuture);
584 	}

586 	private void scheduleGroupToForceComplete(MessageGroup messageGroup) {
587 		final Long groupTimeout = obtainGroupTimeout(messageGroup);
588 		/*
589 		 * When 'groupTimeout' is evaluated to 'null' we do nothing.
590 		 * The 'MessageGroupStoreReaper' can be used to 'forceComplete' message groups.
591 		 */
592 		if (groupTimeout != null) {
593 			if (groupTimeout > 0) {
594 				final Object groupId = messageGroup.getGroupId();
595 				final long timestamp = messageGroup.getTimestamp();
596 				final long lastModified = messageGroup.getLastModified();
597 				ScheduledFuture<?> scheduledFuture =
598 						getTaskScheduler()
599 								.schedule(() -> {
600 									try {
601 										processForceRelease(groupId, timestamp, lastModified);
602 									}
603 									catch (MessageDeliveryException e) {
604 										if (AbstractCorrelatingMessageHandler.this.logger.isWarnEnabled()) {
605 											AbstractCorrelatingMessageHandler.this.logger.warn("The MessageGroup ["
606 													+ groupId + "] is rescheduled by the reason of:", e);
607 										}
608 										scheduleGroupToForceComplete(groupId);
609 									}
610 								}, new Date(System.currentTimeMillis() + groupTimeout));
611 
612 				if (this.logger.isDebugEnabled()) {
613 					this.logger.debug("Schedule MessageGroup [ " + messageGroup + "] to 'forceComplete'.");
614 				}
615 				this.expireGroupScheduledFutures.put(UUIDConverter.getUUID(groupId), scheduledFuture);
616 			}
617 			else {
618 				this.forceReleaseProcessor.processMessageGroup(messageGroup);
619 			}
620 		}
621 	}
622 
623 	private void scheduleGroupToForceComplete(Object groupId) {
624 		MessageGroup messageGroup = this.messageStore.getMessageGroup(groupId);
625 		scheduleGroupToForceComplete(messageGroup);
626 	}
627 
628 	private void processForceRelease(Object groupId, long timestamp, long lastModified) {
629 		MessageGroup messageGroup = this.messageStore.getMessageGroup(groupId);
630 		if (messageGroup.getTimestamp() == timestamp && messageGroup.getLastModified() == lastModified) {
631 			this.forceReleaseProcessor.processMessageGroup(messageGroup);
632 		}
633 	}

635 	private void discardMessage(Message<?> message, Lock lock) {
636 		if (this.releaseLockBeforeSend) {
637 			lock.unlock();
638 		}
639 		discardMessage(message);
640 	}

667 	protected void forceComplete(MessageGroup group) { // NOSONAR Complexity
668 		Object correlationKey = group.getGroupId();
669 		// UUIDConverter is no-op if already converted
670 		UUID groupId = UUIDConverter.getUUID(correlationKey);
671 		Lock lock = this.lockRegistry.obtain(groupId.toString());
672 		boolean removeGroup = true;
673 		boolean noOutput = true;
674 		try {
675 			lock.lockInterruptibly();
676 			try {
677 				cancelScheduledFutureIfAny(correlationKey, groupId, false);
678 				MessageGroup groupNow = group;
679 				/*
680 				 * If the group argument is not already complete,
681 				 * re-fetch it because it might have changed while we were waiting on
682 				 * its lock. If the last modified timestamp changed, defer the completion
683 				 * because the selection condition may have changed such that the group
684 				 * would no longer be eligible. If the timestamp changed, it's a completely new
685 				 * group and should not be reaped on this cycle.
686 				 *
687 				 * If the group argument is already complete, do not re-fetch.
688 				 * Note: not all message stores provide a direct reference to its internal
689 				 * group so the initial 'isComplete()` will only return true for those stores if
690 				 * the group was already complete at the time of its selection as a candidate.
691 				 *
692 				 * If the group is marked complete, only consider it
693 				 * for reaping if it's empty (and both timestamps are unaltered).
694 				 */
695 				if (!group.isComplete()) {
696 					groupNow = this.messageStore.getMessageGroup(correlationKey);
697 				}
698 				long lastModifiedNow = groupNow.getLastModified();
699 				int groupSize = groupNow.size();
700 
701 				if ((!groupNow.isComplete() || groupSize == 0)
702 						&& group.getLastModified() == lastModifiedNow
703 						&& group.getTimestamp() == groupNow.getTimestamp()) {
704 
705 					if (groupSize > 0) {
706 						noOutput = false;
707 						if (this.releaseStrategy.canRelease(groupNow)) {
708 							completeGroup(correlationKey, groupNow, lock);
709 						}
710 						else {
711 							expireGroup(correlationKey, groupNow, lock);
712 						}
713 						if (!this.expireGroupsUponTimeout) {
714 							afterRelease(groupNow, groupNow.getMessages(), true);
715 							removeGroup = false;
716 						}
717 					}
718 					else {
719 						/*
720 						 * By default empty groups are removed on the same schedule as non-empty
721 						 * groups. A longer timeout for empty groups can be enabled by
722 						 * setting minimumTimeoutForEmptyGroups.
723 						 */
724 						removeGroup =
725 								lastModifiedNow <= (System.currentTimeMillis() - this.minimumTimeoutForEmptyGroups);
726 						if (removeGroup && this.logger.isDebugEnabled()) {
727 							this.logger.debug("Removing empty group: " + correlationKey);
728 						}
729 					}
730 				}
731 				else {
732 					removeGroup = false;
733 					if (this.logger.isDebugEnabled()) {
734 						this.logger.debug("Group expiry candidate (" + correlationKey +
735 								") has changed - it may be reconsidered for a future expiration");
736 					}
737 				}
738 			}
739 			catch (MessageDeliveryException e) {
740 				removeGroup = false;
741 				if (this.logger.isDebugEnabled()) {
742 					this.logger.debug("Group expiry candidate (" + correlationKey +
743 							") has been affected by MessageDeliveryException - " +
744 							"it may be reconsidered for a future expiration one more time");
745 				}
746 				throw e;
747 			}
748 			finally {
749 				try {
750 					if (removeGroup) {
751 						remove(group);
752 					}
753 				}
754 				finally {
755 					if (noOutput || !this.releaseLockBeforeSend) {
756 						lock.unlock();
757 					}
758 				}
759 			}
760 		}
761 		catch (@SuppressWarnings("unused") InterruptedException ie) {
762 			Thread.currentThread().interrupt();
763 			this.logger.debug("Thread was interrupted while trying to obtain lock");
764 		}
765 	}

783 	protected void expireGroup(Object correlationKey, MessageGroup group, Lock lock) {
784 		if (this.logger.isInfoEnabled()) {
785 			this.logger.info("Expiring MessageGroup with correlationKey[" + correlationKey + "]");
786 		}
787 		if (this.sendPartialResultOnExpiry) {
788 			if (this.logger.isDebugEnabled()) {
789 				this.logger.debug("Prematurely releasing partially complete group with key ["
790 						+ correlationKey + "] to: " + getOutputChannel());
791 			}
792 			completeGroup(correlationKey, group, lock);
793 		}
794 		else {
795 			if (this.logger.isDebugEnabled()) {
796 				this.logger.debug("Discarding messages of partially complete group with key ["
797 						+ correlationKey + "] to: "
798 						+ (this.discardChannelName != null ? this.discardChannelName : this.discardChannel));
799 			}
800 			if (this.releaseLockBeforeSend) {
801 				lock.unlock();
802 			}
803 			group.getMessages()
804 					.forEach(this::discardMessage);
805 		}
806 		if (this.applicationEventPublisher != null) {
807 			this.applicationEventPublisher.publishEvent(
808 					new MessageGroupExpiredEvent(this, correlationKey, group.size(),
809 							new Date(group.getLastModified()), new Date(), !this.sendPartialResultOnExpiry));
810 		}
811 	}

814 		Message<?> first = null;
815 		if (group != null) {
816 			first = group.getOne();
817 		}
818 		completeGroup(first, correlationKey, group, lock);

821 	@SuppressWarnings("unchecked")
822 	protected Collection<Message<?>> completeGroup(Message<?> message, Object correlationKey, MessageGroup group,
823 			Lock lock) {
824 
825 		Collection<Message<?>> partialSequence = null;
826 		Object result;
827 		try {
828 			if (this.logger.isDebugEnabled()) {
829 				this.logger.debug("Completing group with correlationKey [" + correlationKey + "]");
830 			}
831 
832 			result = this.outputProcessor.processMessageGroup(group);
833 			if (result instanceof Collection<?>) {
834 				verifyResultCollectionConsistsOfMessages((Collection<?>) result);
835 				partialSequence = (Collection<Message<?>>) result;
836 			}
837 
838 			if (this.popSequence && partialSequence == null) {
839 				AbstractIntegrationMessageBuilder<?> messageBuilder = null;
840 				if (result instanceof AbstractIntegrationMessageBuilder<?>) {
841 					messageBuilder = (AbstractIntegrationMessageBuilder<?>) result;
842 				}
843 				else if (!(result instanceof Message<?>)) {
844 					messageBuilder =
845 							getMessageBuilderFactory()
846 							.withPayload(result)
847 							.copyHeaders(message.getHeaders());
848 				}
849 				else if (compareSequences((Message<?>) result, message)) {
850 					messageBuilder =
851 							getMessageBuilderFactory()
852 							.fromMessage((Message<?>) result);
853 				}
854 				result = messageBuilder != null ? messageBuilder.popSequenceDetails() : result;
855 			}
856 		}
857 		finally {
858 			if (this.releaseLockBeforeSend) {
859 				lock.unlock();
860 			}
861 		}
862 		sendOutputs(result, message);
863 		return partialSequence;
864 	}

920 	protected static class SequenceAwareMessageGroup extends SimpleMessageGroup {
921 
922 		private final SimpleMessageGroup sourceGroup;
923 
924 		public SequenceAwareMessageGroup(MessageGroup messageGroup) {
925 			/*
926 			 * Since this group is temporary, and never added to, we simply use the
927 			 * supplied group's message collection for the lookup rather than creating a
928 			 * new group.
929 			 */
930 			super(messageGroup.getMessages(), null, messageGroup.getGroupId(), messageGroup.getTimestamp(),
931 					messageGroup.isComplete(), true);
932 			if (messageGroup instanceof SimpleMessageGroup) {
933 				this.sourceGroup = (SimpleMessageGroup) messageGroup;
934 			}
935 			else {
936 				this.sourceGroup = null;
937 			}
938 		}
939 
940 		/**
941 		 * This method determines whether messages have been added to this group that
942 		 * supersede the given message based on its sequence id. This can be helpful to
943 		 * avoid ending up with sequences larger than their required sequence size or
944 		 * sequences that are missing certain sequence numbers.
945 		 */
946 		@Override
947 		public boolean canAdd(Message<?> message) {
948 			if (this.size() == 0) {
949 				return true;
950 			}
951 			Integer messageSequenceNumber = message.getHeaders()
952 					.get(IntegrationMessageHeaderAccessor.SEQUENCE_NUMBER, Integer.class);
953 			if (messageSequenceNumber != null && messageSequenceNumber > 0) {
954 				Integer messageSequenceSize = message.getHeaders()
955 						.get(IntegrationMessageHeaderAccessor.SEQUENCE_SIZE, Integer.class);
956 				if (messageSequenceSize == null) {
957 					messageSequenceSize = 0;
958 				}
959 				return messageSequenceSize.equals(getSequenceSize())
960 						&& !(this.sourceGroup != null ? this.sourceGroup.containsSequence(messageSequenceNumber)
961 						: containsSequenceNumber(this.getMessages(), messageSequenceNumber));
962 			}
963 			return true;
964 		}
965 
966 		private boolean containsSequenceNumber(Collection<Message<?>> messages, Integer messageSequenceNumber) {
967 			for (Message<?> member : messages) {
968 				if (messageSequenceNumber.equals(member.getHeaders().get(
969 						IntegrationMessageHeaderAccessor.SEQUENCE_NUMBER, Integer.class))) {
970 					return true;
971 				}
972 			}
973 			return false;
974 		}
975 
976 	}
