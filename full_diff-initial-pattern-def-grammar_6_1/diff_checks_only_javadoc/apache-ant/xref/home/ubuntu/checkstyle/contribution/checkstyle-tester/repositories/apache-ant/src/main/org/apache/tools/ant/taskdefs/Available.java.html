223     /**
224      * Entry point when operating as a task.
225      *
226      * @exception BuildException if the task is not configured correctly.
227      */
228     public void execute() throws BuildException {
229         if (property == null) {
230             throw new BuildException("property attribute is required",
231                                      getLocation());
232         }
233 
234         isTask = true;
235         try {
236             if (eval()) {
237                 PropertyHelper ph = PropertyHelper.getPropertyHelper(getProject());
238                 Object oldvalue = ph.getProperty(property);
239                 if (null != oldvalue && !oldvalue.equals(value)) {
240                     log("DEPRECATED - <available> used to override an existing"
241                         + " property."
242                         + StringUtils.LINE_SEP
243                         + "  Build file should not reuse the same property"
244                         + " name for different values.",
245                         Project.MSG_WARN);
246                 }
247                 // NB: this makes use of Project#setProperty rather than Project#setNewProperty
248                 //     due to backwards compatibility reasons
249                 ph.setProperty(property, value, true);
250             }
251         } finally {
252             isTask = false;
253         }
254     }

256     /**
257      * Evaluate the availability of a resource.
258      *
259      * @return boolean is the resource is available.
260      * @exception BuildException if the condition is not configured correctly
261      */
262     public boolean eval() throws BuildException {
263         try {
264             if (classname == null && file == null && resource == null) {
265                 throw new BuildException("At least one of (classname|file|"
266                                          + "resource) is required", getLocation());
267             }
268             if (type != null) {
269                 if (file == null) {
270                     throw new BuildException("The type attribute is only valid "
271                                              + "when specifying the file "
272                                              + "attribute.", getLocation());
273                 }
274             }
275             if (classpath != null) {
276                 classpath.setProject(getProject());
277                 this.loader = getProject().createClassLoader(classpath);
278             }
279             String appendix = "";
280             if (isTask) {
281                 appendix = " to set property " + property;
282             } else {
283                 setTaskName("available");
284             }
285             if ((classname != null) && !checkClass(classname)) {
286                 log("Unable to load class " + classname + appendix,
287                     Project.MSG_VERBOSE);
288                 return false;
289             }
290             if ((file != null) && !checkFile()) {
291                 StringBuffer buf = new StringBuffer("Unable to find ");
292                 if (type != null) {
293                     buf.append(type).append(' ');
294                 }
295                 buf.append(filename).append(appendix);
296                 log(buf.toString(), Project.MSG_VERBOSE);
297                 return false;
298             }
299             if ((resource != null) && !checkResource(resource)) {
300                 log("Unable to load resource " + resource + appendix,
301                     Project.MSG_VERBOSE);
302                 return false;
303             }
304         } finally {
305             if (loader != null) {
306                 loader.cleanup();
307                 loader = null;
308             }
309             if (!isTask) {
310                 setTaskName(null);
311             }
312         }
313         return true;
314     }

316     /**
317      * Search for file/directory either relative to project's
318      * basedir or in the path given as filepath.
319      *
320      * <p>filepath can be a list of directory and/or file names (gen'd
321      * via <fileset>)</p>
322      *
323      * <p>look for:</p><ul>
324      *   <li>full-pathname specified == path in list</li>
325      *   <li>full-pathname specified == parent dir of path in list</li>
326      *   <li>simple name specified   == path in list</li>
327      *   <li>simple name specified   == path in list + name</li>
328      *   <li>simple name specified   == parent dir + name</li>
329      *   <li>simple name specified   == parent of parent dir + name</li>
330      * </ul>
331      */
332     private boolean checkFile() {
333         if (filepath == null) {
334             return checkFile(file, filename);
335         } else {
336             String[] paths = filepath.list();
337             for (int i = 0; i < paths.length; ++i) {
338                 log("Searching " + paths[i], Project.MSG_VERBOSE);
339                 File path = new File(paths[i]);
340 
341                 // **   full-pathname specified == path in list
342                 // **   simple name specified   == path in list
343                 if (path.exists()
344                     && (filename.equals(paths[i])
345                         || filename.equals(path.getName()))) {
346                     if (type == null) {
347                         log("Found: " + path, Project.MSG_VERBOSE);
348                         return true;
349                     } else if (type.isDir()
350                                && path.isDirectory()) {
351                         log("Found directory: " + path, Project.MSG_VERBOSE);
352                         return true;
353                     } else if (type.isFile()
354                                && path.isFile()) {
355                         log("Found file: " + path, Project.MSG_VERBOSE);
356                         return true;
357                     }
358                     // not the requested type
359                     return false;
360                 }
361                 File parent = path.getParentFile();
362                 // **   full-pathname specified == parent dir of path in list
363                 if (parent != null && parent.exists()
364                     && filename.equals(parent.getAbsolutePath())) {
365                     if (type == null) {
366                         log("Found: " + parent, Project.MSG_VERBOSE);
367                         return true;
368                     } else if (type.isDir()) {
369                         log("Found directory: " + parent, Project.MSG_VERBOSE);
370                         return true;
371                     }
372                     // not the requested type
373                     return false;
374                 }
375                 // **   simple name specified   == path in list + name
376                 if (path.exists() && path.isDirectory()) {
377                     if (checkFile(new File(path, filename),
378                                   filename + " in " + path)) {
379                         return true;
380                     }
381                 }
382 
383                 // **   simple name specified   == parent dir + name
384                 while (searchParents && parent != null && parent.exists()) {
385                     if (checkFile(new File(parent, filename),
386                                   filename + " in " + parent)) {
387                         return true;
388                     }
389                     parent = parent.getParentFile();
390                 }
391             }
392         }
393         return false;
394     }

396     /**
397      * Check if a given file exists and matches the required type.
398      */
399     private boolean checkFile(File f, String text) {
400         if (type != null) {
401             if (type.isDir()) {
402                 if (f.isDirectory()) {
403                     log("Found directory: " + text, Project.MSG_VERBOSE);
404                 }
405                 return f.isDirectory();
406             } else if (type.isFile()) {
407                 if (f.isFile()) {
408                     log("Found file: " + text, Project.MSG_VERBOSE);
409                 }
410                 return f.isFile();
411             }
412         }
413         if (f.exists()) {
414             log("Found: " + text, Project.MSG_VERBOSE);
415         }
416         return f.exists();
417     }

419     /**
420      * Check if a given resource can be loaded.
421      */
422     private boolean checkResource(String resource) {
423         if (loader != null) {
424             return (loader.getResourceAsStream(resource) != null);
425         } else {
426             ClassLoader cL = this.getClass().getClassLoader();
427             if (cL != null) {
428                 return (cL.getResourceAsStream(resource) != null);
429             } else {
430                 return
431                     (ClassLoader.getSystemResourceAsStream(resource) != null);
432             }
433         }
434     }
435 
436     /**
437      * Check if a given class can be loaded.
438      */
439     private boolean checkClass(String classname) {
440         try {
441             if (ignoreSystemclasses) {
442                 loader = getProject().createClassLoader(classpath);
443                 loader.setParentFirst(false);
444                 loader.addJavaLibraries();
445                 try {
446                     loader.findClass(classname);
447                 } catch (SecurityException se) {
448                     // class found but restricted name; this is
449                     // actually the case we're looking for in JDK 1.3+,
450                     // so catch the exception and return
451                     return true;
452                 }
453             } else if (loader != null) {
454                 loader.loadClass(classname);
455             } else {
456                 ClassLoader l = this.getClass().getClassLoader();
457                 // Can return null to represent the bootstrap class loader.
458                 // see API docs of Class.getClassLoader.
459                 if (l != null) {
460                     Class.forName(classname, true, l);
461                 } else {
462                     Class.forName(classname);
463                 }
464             }
465             return true;
466         } catch (ClassNotFoundException e) {
467             log("class \"" + classname + "\" was not found",
468                 Project.MSG_DEBUG);
469             return false;
470         } catch (NoClassDefFoundError e) {
471             log("Could not load dependent class \"" + e.getMessage()
472                 + "\" for class \"" + classname + "\"",
473                 Project.MSG_DEBUG);
474             return false;
475         }
476     }

478     /**
479      * EnumeratedAttribute covering the file types to be checked for, either
480      * file or dir.
481      */
482     public static class FileDir extends EnumeratedAttribute {
483 
484         private static final String[] VALUES = {"file", "dir"};
485 
486         /**
487          * @see EnumeratedAttribute#getValues
488          */
489         /** {@inheritDoc}. */
490         public String[] getValues() {
491             return VALUES;
492         }
493 
494         /**
495          * Indicate if the value specifies a directory.
496          *
497          * @return true if the value specifies a directory.
498          */
499         public boolean isDir() {
500             return "dir".equalsIgnoreCase(getValue());
501         }
502 
503         /**
504          * Indicate if the value specifies a file.
505          *
506          * @return true if the value specifies a file.
507          */
508         public boolean isFile() {
509             return "file".equalsIgnoreCase(getValue());
510         }
