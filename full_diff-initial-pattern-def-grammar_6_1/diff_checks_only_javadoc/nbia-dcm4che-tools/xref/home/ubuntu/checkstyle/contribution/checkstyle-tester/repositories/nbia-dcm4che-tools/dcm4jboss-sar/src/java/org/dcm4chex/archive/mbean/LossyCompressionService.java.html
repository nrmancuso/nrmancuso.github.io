139     private final NotificationListener delayedCompressionListener = new NotificationListener() {
140         public void handleNotification(Notification notif, Object handback) {
141             Calendar cal = Calendar.getInstance();
142             int hour = cal.get(Calendar.HOUR_OF_DAY);
143             if (isDisabled(hour)) {
144                 if (log.isDebugEnabled())
145                     log.debug("trigger ignored in time between "
146                             + disabledStartHour + " and " + disabledEndHour
147                             + " !");
148             } else {
149                 new Thread(new Runnable() {
150                     public void run() {
151                         try {
152                             checkForFilesToCompress();
153                         } catch (Exception e) {
154                             log.error("Delayed compression failed!", e);
155                         }
156                     }
157                 }).start();
158             }
159         }
160     };

211     public final String getTaskInterval() {
212         String s = RetryIntervalls.formatIntervalZeroAsNever(taskInterval);
213         return (disabledEndHour == -1) ? s : s + "!" + disabledStartHour + "-"
214                 + disabledEndHour;
215     }
216 
217     public void setTaskInterval(String interval) throws Exception {
218         long oldInterval = taskInterval;
219         int pos = interval.indexOf('!');
220         if (pos == -1) {
221             taskInterval = RetryIntervalls.parseIntervalOrNever(interval);
222             disabledEndHour = -1;
223         } else {
224             taskInterval = RetryIntervalls.parseIntervalOrNever(interval
225                     .substring(0, pos));
226             int pos1 = interval.indexOf('-', pos);
227             disabledStartHour = Integer.parseInt(interval.substring(pos + 1,
228                     pos1));
229             disabledEndHour = Integer.parseInt(interval.substring(pos1 + 1));
230         }
231         if (getState() == STARTED && oldInterval != taskInterval) {
232             scheduler.stopScheduler(timerIDCheckFilesToCompress, listenerID,
233                     delayedCompressionListener);
234             listenerID = scheduler.startScheduler(timerIDCheckFilesToCompress,
235                     taskInterval, delayedCompressionListener);
236         }
237     }

243     public final void setCompressionRules(String rules) {
244         this.compressionRuleList.clear();
245         if (rules == null || rules.trim().length() == 0)
246             return;
247         StringTokenizer st = new StringTokenizer(rules, ",;\n\r\t");
248         while (st.hasMoreTokens()) {
249             String tk = st.nextToken().trim();
250             if (tk.length() == 0)
251                 continue;
252             compressionRuleList.add(new CompressionRule(tk));
253         }
254     }
255 
256     public final String getCompressionRules() {
257         StringBuilder sb = new StringBuilder();
258         for (CompressionRule compressionRule : compressionRuleList)
259             sb.append(compressionRule).append("\r\n");
260         return sb.toString();
261     }

335     public String compressFileJPEGLossy(String inFilePath, String outFilePath,
336             float compressionQuality, String derivationDescription,
337             float estimatedCompressionRatio, boolean newSOPInstanceUID,
338             boolean newSeriesInstanceUID) throws Exception {
339         File inFile = new File(inFilePath.trim());
340         File outFile = new File(outFilePath.trim());
341         float[] actualCompressionRatio = new float[1];
342         CompressCmd.compressFileJPEGLossy(inFile, outFile, null, null,
343                 compressionQuality, derivationDescription,
344                 estimatedCompressionRatio, actualCompressionRatio,
345                 newSOPInstanceUID ? uidGenerator.createUID() : null,
346                 newSeriesInstanceUID ? uidGenerator.createUID() : null,
347                 new byte[bufferSize], null, null);
348         return MessageFormat.format(COMPRESS_FILE_FORMAT,
349                 (float) inFile.length() / outFile.length(),
350                 actualCompressionRatio[0]);
351     }

354         return tsuid.equals(UIDs.ExplicitVRLittleEndian)

356                 || tsuid.equals(UIDs.ImplicitVRLittleEndian);
357     }

359     private boolean isLossyCompressed(String tsuid) {
360         return tsuid.equals(UIDs.JPEGBaseline)
361                 || tsuid.equals(UIDs.JPEGExtended)
362                 || tsuid.equals(UIDs.JPEGLSLossy)
363                 || tsuid.equals(UIDs.JPEG2000Lossy);
364     }

366     public String compressSeriesJPEGLossy(String seriesIUID,
367             float compressionQuality, String derivationDescription,
368             float estimatedCompressionRatio, boolean decompress,
369             boolean archive) throws Exception {
370         byte[] buffer = new byte[bufferSize];
371         int[] planarConfiguration = new int[1];
372         int[] pxdataVR = new int[1];
373         float[] pixelCompressionRatio = new float[1];
374         float fileCompressionRatio;
375         float minFileCompressionRatio = Float.MAX_VALUE;
376         float maxFileCompressionRatio = Float.MIN_VALUE;
377         float sumFileCompressionRatio = 0.f;
378         float minPixelCompressionRatio = Float.MAX_VALUE;;
379         float maxPixelCompressionRatio = Float.MIN_VALUE;
380         float sumPixelCompressionRatio = 0.f;
381         int maxDiffPixelData;
382         int minMaxDiffPixelData = Integer.MAX_VALUE;
383         int maxMaxDiffPixelData = Integer.MIN_VALUE;
384         float sumMaxDiffPixelData = 0.f;
385         int count = 0;
386         String suid = uidGenerator.createUID();
387         Dataset keys = DcmObjectFactory.getInstance().newDataset();
388         keys.putUI(Tags.SeriesInstanceUID, seriesIUID.trim());
389         RetrieveCmd cmd = RetrieveCmd.createSeriesRetrieve(keys);
390         cmd.setFetchSize(fetchSize);
391         FileInfo[][] fileInfoss = cmd.getFileInfos();
392         for (int i = 0; i < fileInfoss.length; i++) {
393             FileInfo[] fileInfos = fileInfoss[i];
394             for (int j = 0; j < fileInfos.length; j++) {
395                 FileInfo fileInfo = fileInfos[j];
396                 if (!fileInfo.fsGroupID.equals(srcFSGroupID)
397                         || isLossyCompressed(fileInfo.tsUID))
398                     continue;
399 
400                 File srcFile = FileUtils.toFile(fileInfo.basedir, fileInfo.fileID);
401                 FileSystemDTO destfs =
402                     fsmgt.selectStorageFileSystem(destFSGroupID);
403                 String destDirPath = destfs.getDirectoryPath();
404                 File destFile = FileUtils.createNewFile(
405                         FileUtils.toFile(destDirPath, fileInfo.fileID)
406                                 .getParentFile(),
407                         (int) Long.parseLong(srcFile.getName(), 16));
408                 File uncFile = null;
409                 File uncFile2 = null;
410                 try {
411                     if (!isUncompressed(fileInfo.tsUID)) {
412                         File absTmpDir = FileUtils.resolve(tmpDir);
413                         if (absTmpDir.mkdirs())
414                             log.info("Create directory for decompressed files");
415                         uncFile = new File(absTmpDir,
416                                 fileInfo.fileID.replace('/', '-') + ".dcm");
417                         DecompressCmd.decompressFile(srcFile, uncFile,
418                                 UIDs.ExplicitVRLittleEndian, -1, VRs.OW, buffer);
419                         srcFile = uncFile;
420                     }
421                     String iuid = uidGenerator.createUID();
422                     Dataset ds = DcmObjectFactory.getInstance().newDataset();
423                     byte[] md5 = CompressCmd.compressFileJPEGLossy(srcFile,
424                             destFile, planarConfiguration, pxdataVR,
425                             compressionQuality, derivationDescription,
426                             estimatedCompressionRatio, pixelCompressionRatio,
427                             iuid, suid, buffer, ds,
428                             fileInfo);
429                     fileCompressionRatio =
430                         (float) srcFile.length() / destFile.length();
431                     if (minFileCompressionRatio > fileCompressionRatio)
432                         minFileCompressionRatio = fileCompressionRatio;
433                     if (maxFileCompressionRatio < fileCompressionRatio)
434                         maxFileCompressionRatio = fileCompressionRatio;
435                     sumFileCompressionRatio += fileCompressionRatio;
436                     if (minPixelCompressionRatio > pixelCompressionRatio[0])
437                         minPixelCompressionRatio = pixelCompressionRatio[0];
438                     if (maxPixelCompressionRatio < pixelCompressionRatio[0])
439                         maxPixelCompressionRatio = pixelCompressionRatio[0];
440                     sumPixelCompressionRatio += pixelCompressionRatio[0];
441                     count++;
442                     if (decompress) {
443                         File absTmpDir = FileUtils.resolve(tmpDir);
444                         if (absTmpDir.mkdirs())
445                             log.info("Create directory for decompressed files");
446                         uncFile2 = new File(absTmpDir,
447                                 fileInfo.fileID.replace('/', '+') + ".dcm");
448                         DecompressCmd.decompressFile(destFile, uncFile2,
449                                 UIDs.ExplicitVRLittleEndian,
450                                 planarConfiguration[0], pxdataVR[0], buffer);
451                         maxDiffPixelData = FileUtils.maxDiffPixelData(srcFile, uncFile2);
452                         if (minMaxDiffPixelData > maxDiffPixelData)
453                             minMaxDiffPixelData = maxDiffPixelData;
454                         if (maxMaxDiffPixelData < maxDiffPixelData)
455                             maxMaxDiffPixelData = maxDiffPixelData;
456                         sumMaxDiffPixelData += maxDiffPixelData;
457                     }
458                     if (archive) {
459                         File baseDir = FileUtils.toFile(destDirPath);
460                         int baseDirPathLength = baseDir.getPath().length();
461                         String destFilePath = destFile.getPath()
462                                 .substring(baseDirPathLength + 1)
463                                 .replace(File.separatorChar, '/');
464                         FileDTO fileDTO = new FileDTO();
465                         fileDTO.setRetrieveAET(destfs.getRetrieveAET());
466                         fileDTO.setFileSystemPk(destfs.getPk());
467                         fileDTO.setFileSystemGroupID(destfs.getGroupID());
468                         fileDTO.setDirectoryPath(destfs.getDirectoryPath());
469                         fileDTO.setAvailability(destfs.getAvailability());
470                         fileDTO.setUserInfo(destfs.getUserInfo());
471                         fileDTO.setFilePath(destFilePath);
472                         fileDTO.setFileTsuid(ds.getFileMetaInfo().getTransferSyntaxUID());
473                         fileDTO.setFileSize((int) destFile.length());
474                         fileDTO.setFileMd5(md5);
475 
476                         updateSeriesDescription(ds);
477                         ds.setPrivateCreatorID(PrivateTags.CreatorID);
478                         ds.putAE(PrivateTags.CallingAET, sourceAET);
479                         ds.setPrivateCreatorID(null);
480                         importFile(fileDTO, ds, suid, i+1 >= fileInfoss.length);
481                         destFile = null;
482                     }
483                 } finally {
484                     if (uncFile != null)
485                         FileUtils.delete(uncFile, false);
486                     if (uncFile2 != null)
487                         FileUtils.delete(uncFile2, false);
488                     if (destFile != null)
489                         FileUtils.delete(destFile, false);
490                 }
491                 break;
492             }
493         }
494         if (count == 0)
495             return "No images compressed";
496 
497         String msg = MessageFormat.format(COMPRESS_SERIES_FORMAT,
498                         count,
499                         minFileCompressionRatio,
500                         sumFileCompressionRatio / count,
501                         maxFileCompressionRatio,
502                         minPixelCompressionRatio,
503                         sumPixelCompressionRatio / count,
504                         maxPixelCompressionRatio);
505         if (decompress)
506             msg += MessageFormat.format(MAX_DERIVATION_FORMAT,
507                     minMaxDiffPixelData,
508                     sumMaxDiffPixelData / count,
509                     maxMaxDiffPixelData);
510         return msg;
511     }

562     private static final class CompressionRule {
563         final String cuid;
564         final String bodyPart;
565         final String srcAET;
566         final long delay;
567         final float quality;
568         final String derivationDescription;
569         final float ratio;
570         final int near;
571         
572         public CompressionRule(String s) {
573             String[] a = StringUtils.split(s, ':');
574             if (a.length != 8)
575                 throw new IllegalArgumentException(s);
576             uidOf(a[0]);
577             cuid = a[0];
578             bodyPart = "*".equals(a[1]) || "".equals(a[1]) ? null : a[1];
579             srcAET = a[2];
580             delay = RetryIntervalls.parseInterval(a[3]);
581             quality = Float.parseFloat(a[4]);
582             derivationDescription = a[5];
583             ratio = Float.parseFloat(a[6]);
584             near = Integer.parseInt(a[7]);
585         }
586 
587         public String toString() {
588             return cuid
589                     + ':' + (bodyPart == null ? "*" : bodyPart)
590                     + ':' + srcAET
591                     + ':' + RetryIntervalls.formatInterval(delay)
592                     + ':' + quality
593                     + ':' + derivationDescription
594                     + ':' + ratio
595                     + ':' + near;
596         }
597 
598     }

600     private static String uidOf(String nameOrUID) {
601         return Character.isDigit(nameOrUID.charAt(0)) ? nameOrUID 
602                 : UIDs.forName(nameOrUID);
603     }

605     public void checkForFilesToCompress() throws FinderException, IOException {
606         synchronized(this) {
607             if (isRunning) {
608                 log.info("checkForFilesToCompress is already running!");
609                 return;
610             }
611             isRunning = true;
612         }
613         try {
614             log.info("Check For Files To Lossy Compress");
615             int limit = limitNumberOfFilesPerTask;
616             byte[] buffer = null;
617             FileSystemMgt2 fsMgtEJB = newFileSystemMgt();
618             for (CompressionRule rule : compressionRuleList) {
619                 Timestamp before = new Timestamp(
620                         System.currentTimeMillis() - rule.delay);
621                 FileDTO[] files = externalRetrieveAET != null
622                     ? fsMgtEJB.findFilesToLossyCompressWithExternalRetrieveAET(
623                             srcFSGroupID, externalRetrieveAET, uidOf(rule.cuid), rule.bodyPart,
624                             rule.srcAET, before, limit)
625                     : copyOnFSGroupID != null
626                     ? fsMgtEJB.findFilesToLossyCompressWithCopyOnOtherFileSystemGroup(
627                             srcFSGroupID, copyOnFSGroupID, uidOf(rule.cuid), rule.bodyPart,
628                             rule.srcAET, before, limit)
629                     : fsMgtEJB.findFilesToLossyCompress(
630                             srcFSGroupID, uidOf(rule.cuid), rule.bodyPart,
631                             rule.srcAET, before, limit);
632                 for (FileDTO fileDTO : files) {
633                     if (buffer == null)
634                         buffer = new byte[bufferSize];
635                     doCompress(fsMgtEJB, fileDTO, rule, buffer);
636                     if (--limit <= 0)
637                         break;
638                 }
639                 if (limit <= 0)
640                     break;
641             }
642         } finally {
643             isRunning = false;
644         }
645     }

647     private void doCompress(FileSystemMgt2 fsMgt, FileDTO fileDTO,
648             CompressionRule rule, byte[] buffer) {
649         String tsuid = fileDTO.getFileTsuid();
650         File srcFile = FileUtils.toFile(fileDTO.getDirectoryPath(),
651                 fileDTO.getFilePath());
652         File destFile = null;
653         File uncFile = null;
654         File uncFile2 = null;
655         int failureStatus = FileStatus.COMPRESS_FAILED;
656         try {
657             FileSystemDTO destfs =
658                     fsmgt.selectStorageFileSystem(destFSGroupID);
659             String destDirPath = destfs.getDirectoryPath();
660             destFile = FileUtils.createNewFile(
661                     FileUtils.toFile(destDirPath,fileDTO.getFilePath())
662                             .getParentFile(),
663                     (int) Long.parseLong(srcFile.getName(), 16));
664             if (!isUncompressed(tsuid)) {
665                 File absTmpDir = FileUtils.resolve(tmpDir);
666                 if (absTmpDir.mkdirs())
667                     log.info("Create directory for decompressed files");
668                 uncFile = new File(absTmpDir,
669                         fileDTO.getFilePath().replace('/', '-') + ".dcm");
670                 DecompressCmd.decompressFile(srcFile, uncFile,
671                         UIDs.ExplicitVRLittleEndian, VRs.OW, -1, buffer);
672                 srcFile = uncFile;
673             }
674             Dataset ds = DcmObjectFactory.getInstance().newDataset();
675             int[] planarConfiguration = new int[1];
676             int[] pxdataVR = new int[1];
677             byte[] md5 = CompressCmd.compressFileJPEGLossy(srcFile, destFile,
678                     planarConfiguration, pxdataVR, rule.quality,
679                     rule.derivationDescription, rule.ratio, null, null, null,
680                     buffer, ds, null);
681             if (rule.near >= 0) {
682                 File absTmpDir = FileUtils.resolve(tmpDir);
683                 if (absTmpDir.mkdirs())
684                     log.info("Create directory for decompressed files");
685                 uncFile2 = new File(absTmpDir,
686                         fileDTO.getFilePath().replace('/', '+') + ".dcm");
687                 DecompressCmd.decompressFile(destFile, uncFile2,
688                         UIDs.ExplicitVRLittleEndian, planarConfiguration[0],
689                         pxdataVR[0], buffer);
690                 int maxDiffPixelData = FileUtils.maxDiffPixelData(srcFile, uncFile2);
691                 if (maxDiffPixelData > rule.near) {
692                     failureStatus = FileStatus.VERIFY_COMPRESS_FAILED;
693                     throw new CompressionFailedException(
694                             "Maximal absolute derivation of pixel sample values: "
695                             + maxDiffPixelData
696                             + " exeeds configured limit in compression rule: "
697                             + rule);
698                 }
699             }
700             File baseDir = FileUtils.toFile(destDirPath);
701             int baseDirPathLength = baseDir.getPath().length();
702             String destFilePath = destFile.getPath()
703                     .substring(baseDirPathLength + 1)
704                     .replace(File.separatorChar, '/');
705             if (log.isDebugEnabled())
706                 log.debug("replace File " + srcFile + " with " + destFile);
707             fsMgt.replaceFileAndCoerceAttributes(destfs.getPk(), fileDTO.getPk(),
708                     destFilePath, ds.getFileMetaInfo().getTransferSyntaxUID(),
709                     destFile.length(), md5, FileStatus.DEFAULT, ds);
710             destFile = null;
711         } catch (Exception e) {
712             log.error("Lossy Compression of " + fileDTO + " failed:", e);
713             try {
714                 fsMgt.setFileStatus(fileDTO.getPk(), failureStatus);
715             } catch (Exception x1) {
716                 log.error("Failed to set status of " + fileDTO + " to "
717                         + failureStatus);
718             }
719         } finally {
720             if (uncFile != null)
721                 FileUtils.delete(uncFile, false);
722             if (uncFile2 != null)
723                 FileUtils.delete(uncFile2, false);
724            if (destFile != null)
725                 FileUtils.delete(destFile, false);
726         }
727     }
