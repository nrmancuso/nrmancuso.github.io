48  import org.dcm4che.net.DcmServiceBase;
49  import org.dcm4che.net.DcmServiceException;
50  import org.dcm4che.net.Dimse;
51  import org.dcm4chex.archive.common.PrivateTags;
52  import org.dcm4chex.archive.ejb.interfaces.AttributesModification;
53  import org.dcm4chex.archive.ejb.interfaces.AttributesModificationHome;
54  import org.dcm4chex.archive.util.EJBHomeFactory;
55  
56  /**
57   * @author Gunter Zeilinger <gunterze@gmail.com>
58   * @version $Revision$ $Date$
59   * @since Aug 6, 2009
60   */
61  public class AttributesModificationScp extends DcmServiceBase {
62  
63      private final AttributesModificationScpService service;
64  
65      public AttributesModificationScp(AttributesModificationScpService service) {
66          this.service = service;
67      }
68  
69      @Override
70      protected void doCStore(ActiveAssociation assoc, Dimse rq, Command rspCmd)
71              throws IOException, DcmServiceException {
72          try {
73              Dataset ds = rq.getDataset();
74              String callingAET = assoc.getAssociation().getCallingAET();
75              ds.setPrivateCreatorID(PrivateTags.CreatorID);
76              ds.putAE(PrivateTags.CallingAET, callingAET);
77              ds.setPrivateCreatorID(null);
78              String modifyingSystem =
79                      ds.getString(Tags.ModifyingSystem, callingAET);
80              String reason = 
81                      ds.getString(Tags.ReasonForTheAttributeModification, "COERCE");
82              if ("PATIENT".equals(ds.getString(Tags.QueryRetrieveLevel))) {
83                  if (getAttributesModification().moveStudyToPatient(ds, 
84                          service.patientMatching(), service.isCreatePatientOnMoveStudy()))
85                      service.sendAttributesModificationNotification(ds);
86              } else if (getAttributesModification().modifyAttributes(
87                      ds, new Date(), modifyingSystem, reason,
88                      service.isUpdateOriginalAttributesSeq(),
89                      service.getEntityNotFoundErrorCode()))
90                  service.sendAttributesModificationNotification(ds);
91          } catch (DcmServiceException e) {
92              throw e;
93          } catch (Exception e) {
94              throw new DcmServiceException(Status.ProcessingFailure, e);
95          }
96      }
97  
98      private AttributesModification getAttributesModification() throws Exception {
99          return ((AttributesModificationHome) EJBHomeFactory.getFactory().lookup(
100                 AttributesModificationHome.class, AttributesModificationHome.JNDI_NAME)).create();
101     }
