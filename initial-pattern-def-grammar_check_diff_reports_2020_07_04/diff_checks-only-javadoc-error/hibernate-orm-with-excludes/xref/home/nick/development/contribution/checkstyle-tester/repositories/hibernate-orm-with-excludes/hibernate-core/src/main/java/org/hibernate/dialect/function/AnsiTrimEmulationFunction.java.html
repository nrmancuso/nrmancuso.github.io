24  package org.hibernate.dialect.function;
25  import org.hibernate.type.StandardBasicTypes;
26  
27  /**
28   * A {@link SQLFunction} implementation that emulates the ANSI SQL trim function
29   * on dialects which do not support the full definition.  However, this function
30   * definition does assume the availability of ltrim, rtrim, and replace functions
31   * which it uses in various combinations to emulate the desired ANSI trim()
32   * functionality.
33   *
34   * @author Steve Ebersole
35   */
36  public class AnsiTrimEmulationFunction extends AbstractAnsiTrimEmulationFunction {
37  	public static final String LTRIM = "ltrim";
38  	public static final String RTRIM = "rtrim";
39  	public static final String REPLACE = "replace";
40  	public static final String SPACE_PLACEHOLDER = "${space}$";
41  
42  	public static final String LEADING_SPACE_TRIM_TEMPLATE = LTRIM + "(?1)";
43  	public static final String TRAILING_SPACE_TRIM_TEMPLATE = RTRIM + "(?1)";
44  	public static final String BOTH_SPACE_TRIM_TEMPLATE = LTRIM + "(" + RTRIM + "(?1))";
45  	public static final String BOTH_SPACE_TRIM_FROM_TEMPLATE = LTRIM + "(" + RTRIM + "(?2))"; //skip the FROM keyword in params
46  
47  	/**
48  	 * A template for the series of calls required to trim non-space chars from the beginning of text.
49  	 * <p/>
50  	 * NOTE : essentially we:</ol>
51  	 * <li>replace all space chars with the text '${space}$'</li>
52  	 * <li>replace all the actual replacement chars with space chars</li>
53  	 * <li>perform left-trimming (that removes any of the space chars we just added which occur at the beginning of the text)</li>
54  	 * <li>replace all space chars with the replacement char</li>
55  	 * <li>replace all the '${space}$' text with space chars</li>
56  	 * </ol>
57  	 */
58  	public static final String LEADING_TRIM_TEMPLATE =
59  			REPLACE + "(" +
60  				REPLACE + "(" +
61  					LTRIM + "(" +
62  						REPLACE + "(" +
63  							REPLACE + "(" +
64  								"?1," +
65  								"' '," +
66  								"'" + SPACE_PLACEHOLDER + "'" +
67  							")," +
68  							"?2," +
69  							"' '" +
70  						")" +
71  					")," +
72  					"' '," +
73  					"?2" +
74  				")," +
75  				"'" + SPACE_PLACEHOLDER + "'," +
76  				"' '" +
77  			")";
78  
79  	/**
80  	 * A template for the series of calls required to trim non-space chars from the end of text.
81  	 * <p/>
82  	 * NOTE: essentially the same series of calls as outlined in {@link #LEADING_TRIM_TEMPLATE} except that here,
83  	 * instead of left-trimming the added spaces, we right-trim them to remove them from the end of the text.
84  	 */
85  	public static final String TRAILING_TRIM_TEMPLATE =
86  			REPLACE + "(" +
87  				REPLACE + "(" +
88  					RTRIM + "(" +
89  						REPLACE + "(" +
90  							REPLACE + "(" +
91  								"?1," +
92  								"' '," +
93  								"'" + SPACE_PLACEHOLDER + "'" +
94  							")," +
95  							"?2," +
96  							"' '" +
97  						")" +
98  					")," +
99  					"' '," +
100 					"?2" +
101 				")," +
102 				"'" + SPACE_PLACEHOLDER + "'," +
103 				"' '" +
104 			")";
105 
106 	/**
107 	 * A template for the series of calls required to trim non-space chars from both the beginning and the end of text.
108 	 * <p/>
109 	 * NOTE: again, we have a series of calls that is essentially the same as outlined in {@link #LEADING_TRIM_TEMPLATE}
110 	 * except that here we perform both left (leading) and right (trailing) trimming.
111 	 */
112 	public static final String BOTH_TRIM_TEMPLATE =
113 			REPLACE + "(" +
114 				REPLACE + "(" +
115 					LTRIM + "(" +
116 						RTRIM + "(" +
117 							REPLACE + "(" +
118 								REPLACE + "(" +
119 									"?1," +
120 									"' '," +
121 									"'" + SPACE_PLACEHOLDER + "'" +
122 								")," +
123 								"?2," +
124 								"' '" +
125 							")" +
126 						")" +
127 					")," +
128 					"' '," +
129 					"?2" +
130 				")," +
131 				"'" + SPACE_PLACEHOLDER + "'," +
132 				"' '" +
133 			")";
134 
135 	private final SQLFunction leadingSpaceTrim;
136 	private final SQLFunction trailingSpaceTrim;
137 	private final SQLFunction bothSpaceTrim;
138 	private final SQLFunction bothSpaceTrimFrom;
139 
140 	private final SQLFunction leadingTrim;
141 	private final SQLFunction trailingTrim;
142 	private final SQLFunction bothTrim;
143 
144 	/**
145 	 * Constructs a new AnsiTrimEmulationFunction using {@link #LTRIM}, {@link #RTRIM}, and {@link #REPLACE}
146 	 * respectively.
147 	 *
148 	 * @see #AnsiTrimEmulationFunction(String,String,String)
149 	 */
150 	public AnsiTrimEmulationFunction() {
151 		this( LTRIM, RTRIM, REPLACE );
152 	}
153 
154 	/**
155 	 * Constructs a <tt>trim()</tt> emulation function definition using the specified function calls.
156 	 *
157 	 * @param ltrimFunctionName The <tt>left trim</tt> function to use.
158 	 * @param rtrimFunctionName The <tt>right trim</tt> function to use.
159 	 * @param replaceFunctionName The <tt>replace</tt> function to use.
160 	 */
161 	public AnsiTrimEmulationFunction(String ltrimFunctionName, String rtrimFunctionName, String replaceFunctionName) {
162 		leadingSpaceTrim = new SQLFunctionTemplate(
163 				StandardBasicTypes.STRING,
164 				LEADING_SPACE_TRIM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
165 		);
166 
167 		trailingSpaceTrim = new SQLFunctionTemplate(
168 				StandardBasicTypes.STRING,
169 				TRAILING_SPACE_TRIM_TEMPLATE.replaceAll( RTRIM, rtrimFunctionName )
170 		);
171 
172 		bothSpaceTrim = new SQLFunctionTemplate(
173 				StandardBasicTypes.STRING,
174 				BOTH_SPACE_TRIM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
175 						.replaceAll( RTRIM, rtrimFunctionName )
176 		);
177 
178 		bothSpaceTrimFrom = new SQLFunctionTemplate(
179 				StandardBasicTypes.STRING,
180 				BOTH_SPACE_TRIM_FROM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
181 						.replaceAll( RTRIM, rtrimFunctionName )
182 		);
183 
184 		leadingTrim = new SQLFunctionTemplate(
185 				StandardBasicTypes.STRING,
186 				LEADING_TRIM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
187 						.replaceAll( RTRIM, rtrimFunctionName )
188 						.replaceAll( REPLACE,replaceFunctionName )
189 		);
190 
191 		trailingTrim = new SQLFunctionTemplate(
192 				StandardBasicTypes.STRING,
193 				TRAILING_TRIM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
194 						.replaceAll( RTRIM, rtrimFunctionName )
195 						.replaceAll( REPLACE,replaceFunctionName )
196 		);
197 
198 		bothTrim = new SQLFunctionTemplate(
199 				StandardBasicTypes.STRING,
200 				BOTH_TRIM_TEMPLATE.replaceAll( LTRIM, ltrimFunctionName )
201 						.replaceAll( RTRIM, rtrimFunctionName )
202 						.replaceAll( REPLACE,replaceFunctionName )
203 		);
204 	}
205 
206 	/**
207 	 * {@inheritDoc}
208 	 */
209 	protected SQLFunction resolveBothSpaceTrimFunction() {
210 		return bothSpaceTrim;
211 	}
212 
213 	/**
214 	 * {@inheritDoc}
215 	 */
216 	protected SQLFunction resolveBothSpaceTrimFromFunction() {
217 		return bothSpaceTrimFrom;
218 	}
219 
220 	/**
221 	 * {@inheritDoc}
222 	 */
223 	protected SQLFunction resolveLeadingSpaceTrimFunction() {
224 		return leadingSpaceTrim;
225 	}
226 
227 	/**
228 	 * {@inheritDoc}
229 	 */
230 	protected SQLFunction resolveTrailingSpaceTrimFunction() {
231 		return trailingSpaceTrim;
232 	}
233 
234 	/**
235 	 * {@inheritDoc}
236 	 */
237 	protected SQLFunction resolveBothTrimFunction() {
238 		return bothTrim;
239 	}
240 
241 	/**
242 	 * {@inheritDoc}
243 	 */
244 	protected SQLFunction resolveLeadingTrimFunction() {
245 		return leadingTrim;
246 	}
247 
248 	/**
249 	 * {@inheritDoc}
250 	 */
251 	protected SQLFunction resolveTrailingTrimFunction() {
252 		return trailingTrim;
253 	}
