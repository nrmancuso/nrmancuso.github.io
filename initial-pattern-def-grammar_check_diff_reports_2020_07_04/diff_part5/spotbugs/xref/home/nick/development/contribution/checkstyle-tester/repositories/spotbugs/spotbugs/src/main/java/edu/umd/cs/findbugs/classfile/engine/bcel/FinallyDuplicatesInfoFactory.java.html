60  public class FinallyDuplicatesInfoFactory implements IMethodAnalysisEngine<FinallyDuplicatesInfoFactory.FinallyDuplicatesInfo> {
61      private static final FinallyDuplicatesInfo NONE_FINALLY_INFO = new FinallyDuplicatesInfo();

63      public static class FinallyDuplicatesInfo {
64          private final List<SortedMap<Integer, Integer>> duplicateBlocks;
65          private final int[] positions;
66  
67          public FinallyDuplicatesInfo(int[] positions, List<SortedMap<Integer, Integer>> duplicateBlocks) {
68              this.positions = positions;
69              this.duplicateBlocks = duplicateBlocks;
70          }
71  
72          public FinallyDuplicatesInfo() {
73              this.duplicateBlocks = null;
74              this.positions = null;
75          }
76  
77          public BitSet getDuplicates(int pos) {
78              if(duplicateBlocks == null) {
79                  return new BitSet();
80              }
81              BitSet current = new BitSet();
82              current.set(pos);
83              boolean changed;
84              do {
85                  changed = false;
86                  for(SortedMap<Integer, Integer> duplicates : duplicateBlocks) {
87                      for (int i = current.nextSetBit(0); i >= 0; i = current.nextSetBit(i+1)) {
88                          int offset = getOffset(duplicates, i);
89                          if(offset >= 0) {
90                              for(Integer key : duplicates.keySet()) {
91                                  int dupPosition = positions[getInstructionNumber(positions, key)+offset];
92                                  if(!current.get(dupPosition)) {
93                                      changed = true;
94                                      current.set(dupPosition);
95                                  }
96                              }
97                          }
98                      }
99                  }
100             } while(changed && duplicateBlocks.size() > 1);
101             current.clear(pos);
102             return current;
103         }
104 
105         public List<Edge> getDuplicates(CFG cfg, Edge edge) {
106             InstructionHandle ih = edge.getSource().getLastInstruction();
107             if(ih == null) {
108                 return Collections.emptyList();
109             }
110             BitSet duplicates = getDuplicates(ih.getPosition());
111             if(duplicates.isEmpty()) {
112                 return Collections.emptyList();
113             }
114             List<Edge> result = new ArrayList<>();
115             for(Iterator<Edge> edgeIterator = cfg.edgeIterator(); edgeIterator.hasNext(); ) {
116                 Edge next = edgeIterator.next();
117                 if(next.getType() != edge.getType()) {
118                     continue;
119                 }
120                 InstructionHandle lastInst = next.getSource().getLastInstruction();
121                 if(lastInst != null && lastInst.getPosition() >= 0 && duplicates.get(lastInst.getPosition())) {
122                     result.add(next);
123                 }
124             }
125             return result;
126         }
127 
128         private int getOffset(SortedMap<Integer, Integer> duplicates, int i) {
129             SortedMap<Integer, Integer> headMap = duplicates.headMap(i+1);
130             if(headMap.isEmpty()) {
131                 return -1;
132             }
133             int end = headMap.get(headMap.lastKey());
134             if(end <= i) {
135                 return -1;
136             }
137             return getInstructionNumber(positions, i)-getInstructionNumber(positions, headMap.lastKey());
138         }
139 
140         @Override
141         public String toString() {
142             return String.valueOf(duplicateBlocks);
143         }
144     }

146     private static class TryBlock {
147         boolean incorrect = false;
148         final int catchAnyAddress;
149         InstructionHandle firstInstruction;
150         SortedMap<Integer, Integer> normalBlocks = new TreeMap<>();
151         SortedMap<Integer, Integer> duplicates = new TreeMap<>();
152 
153         public TryBlock(int catchAnyAddress) {
154             this.catchAnyAddress = catchAnyAddress;
155         }
156 
157         public void update(BitSet exceptionTargets, BitSet branchTargets, InstructionList il, Set<Integer> finallyTargets, BitSet usedTargets) {
158             int lastEnd = -1;
159             InstructionHandle ih = il.findHandle(catchAnyAddress);
160             if(ih == null || !(ih.getInstruction() instanceof ASTORE)) {
161                 incorrect = true;
162                 return;
163             }
164             int varIndex = ((ASTORE)ih.getInstruction()).getIndex();
165             firstInstruction = ih.getNext();
166             if(firstInstruction == null) {
167                 incorrect = true;
168                 return;
169             }
170             int start = firstInstruction.getPosition();
171             int end = start;
172             while(true) {
173                 ih = ih.getNext();
174                 if(ih == null) {
175                     incorrect = true;
176                     return;
177                 }
178                 end = ih.getPosition();
179                 Instruction inst = ih.getInstruction();
180                 if((inst instanceof ALOAD) && ((ALOAD)inst).getIndex() == varIndex) {
181                     ih = ih.getNext();
182                     if(ih == null || !(ih.getInstruction() instanceof ATHROW)) {
183                         incorrect = true;
184                         return;
185                     }
186                     break;
187                 }
188                 if(inst instanceof JSR) {
189                     // We are not interested in JSR finally blocks as they are not duplicated
190                     incorrect = true;
191                     return;
192                 }
193             }
194             duplicates.put(start, end);
195             normalBlocks.put(catchAnyAddress, catchAnyAddress);
196             for(Entry<Integer, Integer> entry : normalBlocks.entrySet()) {
197                 if(lastEnd > -1) {
198                     if(entry.getKey() > lastEnd) {
199                         int candidateStart = lastEnd;
200                         int block2end = equalBlocks(firstInstruction, il.findHandle(candidateStart), end-start, il.getInstructionPositions());
201                         if(block2end > 0 && block2end <= entry.getKey()) {
202                             duplicates.put(candidateStart, block2end);
203                             while(true) {
204                                 int newKey = Math.min(exceptionTargets.nextSetBit(block2end+1), branchTargets.nextSetBit(block2end+1));
205                                 if(newKey < 0 || newKey > entry.getKey()) {
206                                     break;
207                                 }
208                                 InstructionHandle ih2 = il.findHandle(newKey);
209                                 if(exceptionTargets.get(newKey)) {
210                                     ih2 = ih2.getNext(); // Skip astore
211                                 }
212                                 candidateStart = ih2.getPosition();
213                                 block2end = equalBlocks(firstInstruction, ih2, end-start, il.getInstructionPositions());
214                                 if(block2end > 0 && block2end <= entry.getKey()) {
215                                     duplicates.put(candidateStart, block2end);
216                                 } else {
217                                     block2end = newKey;
218                                 }
219                             }
220                         }
221                     }
222                 }
223                 lastEnd = entry.getValue();
224             }
225             ih = ih.getNext();
226             int block2end = equalBlocks(firstInstruction, ih, end-start, il.getInstructionPositions());
227             if(block2end > 0) {
228                 duplicates.put(ih.getPosition(), block2end);
229             }
230         }
231 
232         private int equalBlocks(InstructionHandle ih1, InstructionHandle ih2, int length, int[] positions) {
233             if(length == 0) {
234                 return -1;
235             }
236             if(ih1 == null || ih2 == null) {
237                 return -1;
238             }
239             int start1 = ih1.getPosition();
240             int start2 = ih2.getPosition();
241             int startNum1 = getInstructionNumber(positions, start1);
242             int startNum2 = getInstructionNumber(positions, start2);
243             Map<Integer, Integer> lvMap = new HashMap<>();
244             while(true) {
245                 if(ih1 == null || ih2 == null) {
246                     return -1;
247                 }
248                 Instruction inst1 = ih1.getInstruction();
249                 Instruction inst2 = ih2.getInstruction();
250                 if(!inst1.equals(inst2)) {
251                     if(inst1 instanceof LocalVariableInstruction && inst2 instanceof LocalVariableInstruction) {
252                         if(inst1.getClass() != inst2.getClass()) {
253                             return -1;
254                         }
255                         LocalVariableInstruction lvi1 = (LocalVariableInstruction)inst1;
256                         LocalVariableInstruction lvi2 = (LocalVariableInstruction)inst2;
257                         int lv1 = lvi1.getIndex();
258                         int lv2 = lvi2.getIndex();
259                         Integer targetLV = lvMap.get(lv1);
260                         if(targetLV == null) {
261                             if(!(lvi1 instanceof StoreInstruction)) {
262                                 return -1;
263                             }
264                             lvMap.put(lv1, lv2);
265                         } else if(targetLV != lv2) {
266                             return -1;
267                         }
268                     } else {
269                         if(inst1.getOpcode() != inst2.getOpcode()) {
270                             return -1;
271                         }
272                         if(!(inst1 instanceof BranchInstruction)) {
273                             return -1;
274                         }
275                         int target1 = ((BranchInstruction)inst1).getTarget().getPosition();
276                         int target2 = ((BranchInstruction)inst2).getTarget().getPosition();
277                         if(!(getInstructionNumber(positions, target1)-startNum1 == getInstructionNumber(positions, target2)-startNum2 || (target1 == start1+length))) {
278                             return -1;
279                         }
280                     }
281                 }
282                 if(ih1.getPosition()-start1+inst1.getLength() >= length) {
283                     return ih2.getPosition()+inst2.getLength();
284                 }
285                 ih1 = ih1.getNext();
286                 ih2 = ih2.getNext();
287             }
288         }
289 
290         @Override
291         public String toString() {
292             if(incorrect) {
293                 return "INCORRECT";
294             }
295             return duplicates.toString();
296         }
297     }

299     private static int getInstructionNumber(int[] positions, int position) {
300         return Math.abs(Arrays.binarySearch(positions, position));
301     }

303     @Override
304     public FinallyDuplicatesInfo analyze(IAnalysisCache analysisCache, MethodDescriptor descriptor) throws CheckedAnalysisException {
305         Method method = analysisCache.getMethodAnalysis(Method.class, descriptor);
306         if(method == null) {
307             return NONE_FINALLY_INFO;
308         }
309         BitSet exceptionTargets = new BitSet();
310         Map<Integer, TryBlock> finallyTargets = new LinkedHashMap<>();
311         for(CodeException codeException : method.getCode().getExceptionTable()) {
312             if(codeException.getCatchType() == 0) {
313                 TryBlock block = finallyTargets.get(codeException.getHandlerPC());
314                 if(block == null) {
315                     block = new TryBlock(codeException.getHandlerPC());
316                     finallyTargets.put(codeException.getHandlerPC(), block);
317                 }
318                 if(codeException.getStartPC() != codeException.getHandlerPC()) {
319                     block.normalBlocks.put(codeException.getStartPC(), codeException.getEndPC());
320                 }
321             }
322             exceptionTargets.set(codeException.getHandlerPC());
323         }
324         if(finallyTargets.isEmpty()) {
325             return NONE_FINALLY_INFO;
326         }
327         MethodGen methodGen = analysisCache.getMethodAnalysis(MethodGen.class, descriptor);
328         if(methodGen == null) {
329             return NONE_FINALLY_INFO;
330         }
331         InstructionList il = methodGen.getInstructionList();
332         BitSet branchTargets = new BitSet();
333         for(InstructionHandle ih = il.getStart(); ih != null; ih = ih.getNext()) {
334             Instruction inst = ih.getInstruction();
335             if(inst instanceof BranchInstruction) {
336                 branchTargets.set(((BranchInstruction) inst).getTarget().getPosition());
337             }
338         }
339         BitSet usedTargets = new BitSet();
340         List<SortedMap<Integer, Integer>> duplicates = new ArrayList<>();
341         for(TryBlock block : finallyTargets.values()) {
342             if(usedTargets.get(block.catchAnyAddress)) {
343                 continue;
344             }
345             block.update(exceptionTargets, branchTargets, il, finallyTargets.keySet(), usedTargets);
346             if(!block.incorrect && block.duplicates.size() > 1) {
347                 duplicates.add(block.duplicates);
348             }
349         }
350         if(duplicates.isEmpty()) {
351             return NONE_FINALLY_INFO;
352         }
353         return new FinallyDuplicatesInfo(il.getInstructionPositions(), duplicates);
354     }

356     @Override
357     public void registerWith(IAnalysisCache analysisCache) {
358         analysisCache.registerMethodAnalysisEngine(FinallyDuplicatesInfo.class, this);
