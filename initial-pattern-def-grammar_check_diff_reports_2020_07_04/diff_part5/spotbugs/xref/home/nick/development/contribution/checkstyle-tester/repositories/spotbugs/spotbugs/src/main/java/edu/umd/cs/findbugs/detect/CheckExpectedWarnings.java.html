68   * Check uses of the ExpectWarning and NoWarning annotations. This is for
69   * internal testing of FindBugs (against spotbugsTestCases).

97      public CheckExpectedWarnings(BugReporter bugReporter) {
98          bugCollection = bugReporter.getBugCollection();
99          if (bugCollection != null) {
100             reporter = bugReporter;
101             expectWarning = DescriptorFactory.createClassDescriptor(ExpectWarning.class);
102             noWarning = DescriptorFactory.createClassDescriptor(NoWarning.class);
103             desireWarning = DescriptorFactory.createClassDescriptor(DesireWarning.class);
104             desireNoWarning = DescriptorFactory.createClassDescriptor(DesireNoWarning.class);
105         }
106     }

108     @Override
109     public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
110         if (reporter == null) {
111             if (!warned) {
112                 System.err
113                 .println("*** NOTE ***: CheckExpectedWarnings disabled because bug reporter doesn't use a BugCollection");
114                 warned = true;
115             }
116             return;
117         }
118 
119         if (!initialized) {
120 
121             initialized = true;
122             //
123             // Build index of all warnings reported so far, by method.
124             // Because this detector runs in a later pass than any
125             // reporting detector, all warnings should have been
126             // produced by this point.
127             //
128 
129             warningsByClass = new HashMap<>();
130             warningsByMethod = new HashMap<>();
131             warningsByField = new HashMap<>();
132 
133             for (Iterator<BugInstance> i = bugCollection.iterator(); i.hasNext();) {
134                 BugInstance warning = i.next();
135                 MethodAnnotation method = warning.getPrimaryMethod();
136                 if (method != null) {
137                     MethodDescriptor methodDesc = method.toMethodDescriptor();
138                     Collection<BugInstance> warnings = warningsByMethod.get(methodDesc);
139                     if (warnings == null) {
140                         warnings = new LinkedList<>();
141                         warningsByMethod.put(methodDesc, warnings);
142                     }
143                     warnings.add(warning);
144                 }
145                 FieldAnnotation field = warning.getPrimaryField();
146                 if (field != null) {
147                     if (DEBUG) {
148                         System.out.println("primary field of " + field + " for " + warning);
149                     }
150                     FieldDescriptor fieldDescriptor = field.toFieldDescriptor();
151                     Collection<BugInstance> warnings = warningsByField.get(fieldDescriptor);
152 
153                     if (warnings == null) {
154                         warnings = new LinkedList<>();
155                         warningsByField.put(fieldDescriptor, warnings);
156                     }
157                     warnings.add(warning);
158                 }
159 
160                 ClassAnnotation clazz = warning.getPrimaryClass();
161                 if (clazz != null) {
162                     ClassDescriptor classDesc = clazz.getClassDescriptor();
163                     if(field != null && classDesc.equals(field.getClassDescriptor())) {
164                         continue;
165                     }
166                     if (method != null && classDesc.equals(method.getClassDescriptor())) {
167                         continue;
168                     }
169                     Collection<BugInstance> warnings = warningsByClass.get(classDesc);
170                     if (warnings == null) {
171                         warnings = new LinkedList<>();
172                         warningsByClass.put(classDesc, warnings);
173                     }
174                     warnings.add(warning);
175                 }
176 
177             }
178 
179         }
180 
181         XClass xclass = Global.getAnalysisCache().getClassAnalysis(XClass.class, classDescriptor);
182         List<? extends XMethod> methods = xclass.getXMethods();
183         if (DEBUG) {
184             System.out.println("CEW: checking " + xclass.toString());
185         }
186         if (xclass.isSynthetic()) {
187             if (DEBUG) {
188                 System.out.println("Skipping synthetic classxclass " + xclass.toString());
189             }
190             return;
191         }
192         check(xclass, expectWarning, true, HIGH_PRIORITY);
193         check(xclass, desireWarning, true, NORMAL_PRIORITY);
194         check(xclass, noWarning, false, HIGH_PRIORITY);
195         check(xclass, desireNoWarning, false, NORMAL_PRIORITY);
196 
197         for (XMethod xmethod : methods) {
198             if (DEBUG) {
199                 System.out.println("CEW: checking " + xmethod.toString());
200             }
201             if (xmethod.isSynthetic()) {
202                 if (DEBUG) {
203                     System.out.println("Skipping synthetic method " + xmethod.toString());
204                 }
205                 continue;
206             }
207             check(xmethod, expectWarning, true, HIGH_PRIORITY);
208             check(xmethod, desireWarning, true, NORMAL_PRIORITY);
209             check(xmethod, noWarning, false, HIGH_PRIORITY);
210             check(xmethod, desireNoWarning, false, NORMAL_PRIORITY);
211         }
212         for (XField xfield : xclass.getXFields()) {
213             if (DEBUG) {
214                 System.out.println("CEW: checking " + xfield.toString());
215             }
216             if (xfield.isSynthetic()) {
217                 if (DEBUG) {
218                     System.out.println("Skipping synthetic field " + xfield.toString());
219                 }
220                 continue;
221             }
222             check(xfield, expectWarning, true, HIGH_PRIORITY);
223             check(xfield, desireWarning, true, NORMAL_PRIORITY);
224             check(xfield, noWarning, false, HIGH_PRIORITY);
225             check(xfield, desireNoWarning, false, NORMAL_PRIORITY);
226         }
227 
228     }

230     private void check(XClass xclass, ClassDescriptor annotation, boolean expectWarnings, int priority) {
231         AnnotationValue expect = xclass.getAnnotation(annotation);
232         if (expect == null) {
233             return;
234         }
235         if (DEBUG) {
236             System.out.println("*** Found " + annotation + " annotation on " + xclass);
237         }
238         ClassDescriptor descriptor = xclass.getClassDescriptor();
239         Collection<BugInstance> warnings = warningsByClass.get(descriptor);
240         check(expect, descriptor, warnings, expectWarnings, priority, descriptor);
241     }

243     private void check(XMethod xmethod, ClassDescriptor annotation, boolean expectWarnings, int priority) {
244         AnnotationValue expect = xmethod.getAnnotation(annotation);
245         if (expect == null) {
246             return;
247         }
248         if (DEBUG) {
249             System.out.println("*** Found " + annotation + " annotation on " + xmethod);
250         }
251         FieldOrMethodDescriptor descriptor = xmethod.getMethodDescriptor();
252         Collection<BugInstance> warnings = warningsByMethod.get(descriptor);
253         check(expect, descriptor, warnings, expectWarnings, priority, descriptor.getClassDescriptor());
254     }

256     private void check(XField xfield, ClassDescriptor annotation, boolean expectWarnings, int priority) {
257         AnnotationValue expect = xfield.getAnnotation(annotation);
258         if (expect == null) {
259             return;
260         }
261 
262         if (DEBUG) {
263             System.out.println("*** Found " + annotation + " annotation on " + xfield);
264         }
265         FieldOrMethodDescriptor descriptor = xfield.getFieldDescriptor();
266         Collection<BugInstance> warnings = warningsByField.get(descriptor);
267         check(expect, descriptor, warnings, expectWarnings, priority, descriptor.getClassDescriptor());
268     }

270     private void check(AnnotationValue expect, Object descriptor,
271             Collection<BugInstance> warnings, boolean expectWarnings, int priority, ClassDescriptor cd) {
272 
273         if (expect != null) {
274 
275             String expectedBugCodes = (String) expect.getValue("value");
276             EnumValue wantedConfidence = (EnumValue) expect.getValue("confidence");
277             EnumValue wantedPriority = (EnumValue) expect.getValue("priority");
278             Integer num = (Integer) expect.getValue("num");
279             if (num == null) {
280                 num = (expectWarnings ? 1 : 0);
281             }
282             Integer rank = (Integer) expect.getValue("rank");
283             if (rank == null) {
284                 rank = BugRanker.VISIBLE_RANK_MAX;
285             }
286 
287             int minPriority = Confidence.LOW.getConfidenceValue();
288             if (wantedConfidence != null) {
289                 minPriority = Confidence.valueOf(wantedConfidence.value).getConfidenceValue();
290             } else if (wantedPriority != null) {
291                 minPriority = Priority.valueOf(wantedPriority.value).getPriorityValue();
292             }
293 
294             if (DEBUG)  {
295                 if (warnings == null) {
296                     System.out.println("Checking " + expectedBugCodes + " against no bugs");
297                 } else {
298                     System.out.println("Checking " + expectedBugCodes + " against " + warnings.size() + " bugs");
299                     for (BugInstance b : warnings) {
300                         System.out.println("  " + b.getType());
301                     }
302                 }
303             }
304             if (expectedBugCodes == null || expectedBugCodes.trim().length() == 0) {
305                 checkAnnotation(null, warnings, expectWarnings, priority, rank, num, descriptor, minPriority, cd);
306             } else {
307                 StringTokenizer tok = new StringTokenizer(expectedBugCodes, ",");
308                 while (tok.hasMoreTokens()) {
309                     String bugCode = tok.nextToken().trim();
310                     checkAnnotation(bugCode, warnings, expectWarnings, priority, rank, num, descriptor, minPriority, cd);
311                 }
312             }
313         }
314     }

316     public void checkAnnotation(@CheckForNull String bugCode, Collection<BugInstance> warnings, boolean expectWarnings, int priority,
317             Integer rank, Integer num, Object methodDescriptor, int minPriority, ClassDescriptor cd) {
318 
319         String bugCodeMessage = bugCode != null ? bugCode : "any bug";
320         Collection<SourceLineAnnotation> bugs = countWarnings(warnings, bugCode, minPriority,
321                 rank);
322         if (expectWarnings && bugs.size() < num) {
323             if (DetectorFactoryCollection.instance().isDisabledByDefault(bugCode)) {
324                 return;
325             }
326             BugInstance bug = makeWarning("FB_MISSING_EXPECTED_WARNING", methodDescriptor, priority, cd).addString(bugCodeMessage);
327             if (!bugs.isEmpty()) {
328                 bug.addString(String.format("Expected %d bugs, saw %d", num, bugs.size()));
329             }
330             reporter.reportBug(bug);
331         } else if (bugs.size() > num) {
332             // More bugs than expected
333             BugInstance bug = makeWarning("FB_UNEXPECTED_WARNING", methodDescriptor, priority, cd).addString(bugCodeMessage);
334             if (!expectWarnings) {
335                 // Wanted no more than this many warnings
336                 for (SourceLineAnnotation s : bugs) {
337                     reporter.reportBug(bug.add(s));
338                 }
339             } else if(num > 1){
340                 // For example, we told it that we expected 3 warnings, and saw 4 warnings
341                 // num == 1 is default value. So if we set a non default value, and see more warnings
342                 // as expected, it's a problem
343                 bug.addString(String.format("Expected %d bugs, saw %d", num, bugs.size()));
344                 reporter.reportBug(bug);
345             }
346         }
347     }

349     public BugInstance makeWarning(String bugPattern, Object descriptor, int priority, ClassDescriptor cd) {
350         BugInstance bug = new BugInstance(this, bugPattern, priority).addClass(cd);
351         if (descriptor instanceof FieldDescriptor) {
352             bug.addField((FieldDescriptor)descriptor);
353         } else if (descriptor instanceof MethodDescriptor) {
354             bug.addMethod((MethodDescriptor)descriptor);
355         } else if (descriptor instanceof ClassDescriptor) {
356             bug.addClass((ClassDescriptor)descriptor);
357         }
358         if (DEBUG) {
359             System.out.println("Reporting " + bug);
360         }
361         return bug;
362 
363     }

365     private static Collection<SourceLineAnnotation> countWarnings( Collection<BugInstance> warnings,
366             @CheckForNull String bugCode,
367             int desiredPriority, int rank) {
368 
369         Collection<SourceLineAnnotation> matching = new HashSet<>();
370         DetectorFactoryCollection i18n = DetectorFactoryCollection.instance();
371         boolean matchPattern = false;
372         try {
373             i18n.getBugCode(bugCode);
374         } catch (IllegalArgumentException e) {
375             matchPattern = true;
376         }
377 
378         if (warnings != null) {
379             for (BugInstance warning : warnings) {
380                 if (warning.getPriority() > desiredPriority) {
381                     continue;
382                 }
383                 if (warning.getBugRank() > rank) {
384                     continue;
385                 }
386                 if (bugCode == null) {
387                     matching.add(warning.getPrimarySourceLineAnnotation());
388                     matching.addAll(warning.getAnotherInstanceSourceLineAnnotations());
389                     continue;
390                 }
391                 BugPattern pattern = warning.getBugPattern();
392                 String match;
393                 if (matchPattern) {
394                     match = pattern.getType();
395                 } else {
396                     match = pattern.getAbbrev();
397                 }
398                 if (match.equals(bugCode)) {
399                     matching.add(warning.getPrimarySourceLineAnnotation());
400                     matching.addAll(warning.getAnotherInstanceSourceLineAnnotations());
401 
402                 }
403             }
404         }
405         return matching;
406     }

408     @Override
409     public void finishPass() {
410         HashSet<BugPattern> claimedReported = new HashSet<>();
411         for (DetectorFactory d : DetectorFactoryCollection.instance().getFactories()) {
412             claimedReported.addAll(d.getReportedBugPatterns());
413         }
414         for (BugPattern b : DetectorFactoryCollection.instance().getBugPatterns()) {
415             String category = b.getCategory();
416             if (!b.isDeprecated() && !"EXPERIMENTAL".equals(category) && !claimedReported.contains(b)) {
417                 AnalysisContext.logError("No detector claims " + b.getType());
418             }
419         }
420 
421     }
