24  /**
25   * @test
26   * @bug 4094894
27   * @summary On W95/W98 it's not possible to send a datagram >12k
28   *          via the loopback address.
29   */
30  
31  import java.net.*;
32  import java.io.*;
33  
34  public class Send12k {
35  
36  
37      public static void main(String args[]) throws Exception {
38  
39           int SEND_SIZE=0;
40  
41           if(System.getProperty("os.name").contains("Mac")) {
42               SEND_SIZE = 16 * 576;
43           } else {
44               SEND_SIZE = 16 * 1024;
45           }
46          DatagramSocket s1 = new DatagramSocket();
47          DatagramSocket s2 = new DatagramSocket();
48  
49          byte b1[] = new byte[ SEND_SIZE ];
50          DatagramPacket p1 = new DatagramPacket(b1, 0, b1.length,
51                                                 InetAddress.getLocalHost(),
52                                                 s2.getLocalPort());
53          boolean sendOkay = true;
54          try {
55              s1.send(p1);
56          } catch (SocketException e) {
57              /*
58               * Prior to merlin a send of > 12k to loopback address
59               * would fail silently.
60               */
61              sendOkay = false;
62          }
63  
64          if (sendOkay) {
65              byte b2[] = new byte[ SEND_SIZE * 2];
66              DatagramPacket p2 = new DatagramPacket( b2, SEND_SIZE * 2 );
67              s2.setSoTimeout(2000);
68  
69              try {
70                  s2.receive(p1);
71              } catch (InterruptedIOException ioe) {
72                  throw new Exception("Datagram not received within timeout");
73              }
74  
75              if (p1.getLength() != SEND_SIZE) {
76                  throw new Exception("Received datagram incorrect size");
77              }
78          }
79  
80      }
81  
82  }
