56  /**
57   * <p>
58   * Static Configuration Manager for the FreemarkerResult's configuration
59   * </p>
60   *
61   * <p>
62   * Possible extension points are:
63   *
64   * <ul>
65   *   <li>createConfiguration method</li>
66   *   <li>loadSettings method</li>
67   *   <li>createTemplateLoader method</li>
68   *   <li>populateContext method</li>
69   * </ul>
70   *
71   * <p>
72   * <b> createConfiguration method </b><br>
73   * Create a freemarker Configuration.
74   * </p>
75   *
76   * <p>
77   * <b> loadSettings method </b><br>
78   * Load freemarker settings, default to freemarker.properties (if found in classpath)
79   * </p>
80   *
81   * <p>
82   * <b> createTemplateLoader method</b><br>
83   * create a freemarker TemplateLoader that loads freemarker template in the following order:
84   * </p>
85   *
86   * <ol>
87   *   <li>path defined in ServletContext init parameter named 'templatePath' or 'TemplatePath' (must be an absolute path)</li>
88   *   <li>webapp classpath</li>
89   *   <li>struts's static folder (under [STRUT2_SOURCE]/org/apache/struts2/static/</li>
90   * </ol>
91   *
92   * <p>
93   * <b> populateContext method</b><br>
94   * populate the created model.
95   * </p>
96   */
97  public class FreemarkerManager {

248     public synchronized Configuration getConfiguration(ServletContext servletContext) {
249         if (config == null) {
250             try {
251                 init(servletContext);
252             } catch (TemplateException e) {
253                 LOG.error("Cannot load freemarker configuration: ", e);
254             }
255             // store this configuration in the servlet context
256             servletContext.setAttribute(CONFIG_SERVLET_CONTEXT_KEY, config);
257         }
258         return config;
259     }
260 
261     public void init(ServletContext servletContext) throws TemplateException {
262         config = createConfiguration(servletContext);
263 
264         // Set defaults:
265         config.setTemplateExceptionHandler(TemplateExceptionHandler.HTML_DEBUG_HANDLER);
266         contentType = DEFAULT_CONTENT_TYPE;
267 
268         // Process object_wrapper init-param out of order:
269         wrapper = createObjectWrapper(servletContext);
270         LOG.debug("Using object wrapper of class {}", wrapper.getClass().getName());
271         config.setObjectWrapper(wrapper);
272 
273         // Process TemplatePath init-param out of order:
274         templatePath = servletContext.getInitParameter(INITPARAM_TEMPLATE_PATH);
275         if (templatePath == null) {
276             templatePath = servletContext.getInitParameter("templatePath");
277         }
278 
279         configureTemplateLoader(createTemplateLoader(servletContext, templatePath));
280 
281         loadSettings(servletContext);
282     }

297     /**
298      * Create the instance of the freemarker Configuration object.
299      * <p>
300      * this implementation
301      * </p>
302      *
303      * <ul>
304      * <li>obtains the default configuration from Configuration.getDefaultConfiguration()
305      * <li>sets up template loading from a ClassTemplateLoader and a WebappTemplateLoader
306      * <li>sets up the object wrapper to be the BeansWrapper
307      * <li>loads settings from the classpath file /freemarker.properties
308      * </ul>
309      *
310      * @param servletContext the servlet context
311      * @return a instance of the freemarker configuration object
312      * @throws TemplateException in case of errors during creating the configuration
313      */
314     protected Configuration createConfiguration(ServletContext servletContext) throws TemplateException {
315         Version incompatibleImprovements = getFreemarkerVersion(servletContext);
316 
317         Configuration configuration = new Configuration(incompatibleImprovements);
318 
319         configuration.setTemplateExceptionHandler(TemplateExceptionHandler.HTML_DEBUG_HANDLER);
320 
321         if (mruMaxStrongSize > 0) {
322             LOG.debug("Sets Configuration.CACHE_STORAGE_KEY to strong:{}", mruMaxStrongSize);
323             configuration.setSetting(Configuration.CACHE_STORAGE_KEY, "strong:" + mruMaxStrongSize);
324         }
325         if (templateUpdateDelay != null) {
326             LOG.debug("Sets Configuration.TEMPLATE_UPDATE_DELAY_KEY to {}", templateUpdateDelay);
327             configuration.setSetting(Configuration.TEMPLATE_UPDATE_DELAY_KEY, templateUpdateDelay);
328         }
329         if (encoding != null) {
330             LOG.debug("Sets DefaultEncoding to {}", encoding);
331             configuration.setDefaultEncoding(encoding);
332         }
333         LOG.debug("Disabled localized lookups");
334         configuration.setLocalizedLookup(false);
335         LOG.debug("Enabled whitespace stripping");
336         configuration.setWhitespaceStripping(true);
337         LOG.debug("Sets NewBuiltinClassResolver to TemplateClassResolver.SAFER_RESOLVER");
338         configuration.setNewBuiltinClassResolver(TemplateClassResolver.SAFER_RESOLVER);
339         LOG.debug("Sets HTML as an output format and escaping policy");
340         configuration.setAutoEscapingPolicy(Configuration.ENABLE_IF_DEFAULT_AUTO_ESCAPING_POLICY);
341         configuration.setOutputFormat(HTMLOutputFormat.INSTANCE);
342 
343         return configuration;
344     }

346     protected Version getFreemarkerVersion(ServletContext servletContext) {
347         Version incompatibleImprovements = Configuration.VERSION_2_3_28;
348 
349         String incompatibleImprovementsParam = servletContext.getInitParameter("freemarker." + Configuration.INCOMPATIBLE_IMPROVEMENTS_KEY_SNAKE_CASE);
350         if (incompatibleImprovementsParam != null) {
351             incompatibleImprovements = new Version(incompatibleImprovementsParam);
352         }
353 
354         return incompatibleImprovements;
355     }

358     protected ScopesHashModel buildScopesHashModel(ServletContext servletContext, HttpServletRequest request, HttpServletResponse response, ObjectWrapper wrapper, ValueStack stack) {
359         ScopesHashModel model = new ScopesHashModel(wrapper, servletContext, request, stack);
360 
361         // Create hash model wrapper for servlet context (the application). We need one thread, once per servlet context
362         synchronized (servletContext) {
363             ServletContextHashModel servletContextModel = (ServletContextHashModel) servletContext.getAttribute(ATTR_APPLICATION_MODEL);
364             if (servletContextModel == null) {
365                 // first try a JSP support servlet.  If it fails, default to the servlet.
366                 GenericServlet servlet = JspSupportServlet.jspSupportServlet;
367                 if (servlet != null) {
368                     servletContextModel = new ServletContextHashModel(servlet, wrapper);
369                     servletContext.setAttribute(ATTR_APPLICATION_MODEL, servletContextModel);
370                 } else {
371                     servletContextModel = new ServletContextHashModel(servletContext, wrapper);
372                     servletContext.setAttribute(ATTR_APPLICATION_MODEL, servletContextModel);
373                 }
374                 TaglibFactory taglibs = new TaglibFactory(servletContext);
375                 taglibs.setObjectWrapper(wrapper);
376                 servletContext.setAttribute(ATTR_JSP_TAGLIBS_MODEL, taglibs);
377             }
378             model.put(KEY_APPLICATION, servletContextModel);
379             model.putUnlistedModel(KEY_APPLICATION_PRIVATE, servletContextModel);
380         }
381         model.put(KEY_JSP_TAGLIBS, (TemplateModel) servletContext.getAttribute(ATTR_JSP_TAGLIBS_MODEL));
382 
383         // Create hash model wrapper for session
384         HttpSession session = request.getSession(false);
385         if (session != null) {
386             model.put(KEY_SESSION, new HttpSessionHashModel(session, wrapper));
387         }
388 
389         // Create hash model wrapper for the request attributes
390         HttpRequestHashModel requestModel = (HttpRequestHashModel) request.getAttribute(ATTR_REQUEST_MODEL);
391 
392         if ((requestModel == null) || (requestModel.getRequest() != request)) {
393             requestModel = new HttpRequestHashModel(request, response, wrapper);
394             request.setAttribute(ATTR_REQUEST_MODEL, requestModel);
395         }
396 
397         model.put(KEY_REQUEST, requestModel);
398 
399 
400         // Create hash model wrapper for request parameters
401         HttpRequestParametersHashModel reqParametersModel = (HttpRequestParametersHashModel) request.getAttribute(ATTR_REQUEST_PARAMETERS_MODEL);
402         if (reqParametersModel == null || requestModel.getRequest() != request) {
403             reqParametersModel = new HttpRequestParametersHashModel(request);
404             request.setAttribute(ATTR_REQUEST_PARAMETERS_MODEL, reqParametersModel);
405         }
406         model.put(ATTR_REQUEST_PARAMETERS_MODEL, reqParametersModel);
407         model.put(KEY_REQUEST_PARAMETERS_STRUTS,reqParametersModel);
408 
409         return model;
410     }

412     protected ObjectWrapper createObjectWrapper(ServletContext servletContext) {
413         Version incompatibleImprovements = getFreemarkerVersion(servletContext);
414         StrutsBeanWrapper wrapper = new StrutsBeanWrapper(altMapWrapper, incompatibleImprovements);
415         wrapper.setUseCache(cacheBeanWrapper);
416         return wrapper;
417     }

420      /**
421      * Create the template loader. The default implementation will create a
422      * {@link ClassTemplateLoader} if the template path starts with "class://",
423      * a {@link FileTemplateLoader} if the template path starts with "file://",
424      * and a {@link WebappTemplateLoader} otherwise.
425      *
426      * @param servletContext the servlet path
427      * @param templatePath the template path to create a loader for
428      * @return a newly created template loader
429      */
430     protected TemplateLoader createTemplateLoader(ServletContext servletContext, String templatePath) {
431         TemplateLoader templatePathLoader = null;
432 
433          try {
434              if(templatePath!=null){
435                  if (templatePath.startsWith("class://")) {
436                      // substring(7) is intentional as we "reuse" the last slash
437                      templatePathLoader = new ClassTemplateLoader(getClass(), templatePath.substring(7));
438                  } else if (templatePath.startsWith("file://")) {
439                      templatePathLoader = new FileTemplateLoader(new File(templatePath.substring(7)));
440                  }
441              }
442          } catch (IOException e) {
443              LOG.error("Invalid template path specified: {}", e.getMessage(), e);
444          }
445 
446          // presume that most apps will require the class and webapp template loader
447          // if people wish to
448          return templatePathLoader != null ?
449                  new MultiTemplateLoader(new TemplateLoader[]{
450                          templatePathLoader,
451                          new WebappTemplateLoader(servletContext),
452                          new StrutsClassTemplateLoader()
453                  })
454                  : new MultiTemplateLoader(new TemplateLoader[]{
455                  new WebappTemplateLoader(servletContext),
456                  new StrutsClassTemplateLoader()
457          });
458      }

461     /**
462      * Load the settings from the /freemarker.properties file on the classpath
463      *
464      * @param servletContext the servlet context
465      *
466      * @see freemarker.template.Configuration#setSettings for the definition of valid settings
467      */
468     protected void loadSettings(ServletContext servletContext) {
469         try (InputStream in = fileManager.loadFile(ClassLoaderUtil.getResource("freemarker.properties", getClass()))){
470             if (in != null) {
471                 Properties p = new Properties();
472                 p.load(in);
473 
474                 for (Object o : p.keySet()) {
475                     String name = (String) o;
476                     String value = (String) p.get(name);
477 
478                     if (name == null) {
479                         throw new IOException(
480                                 "init-param without param-name.  Maybe the freemarker.properties is not well-formed?");
481                     }
482                     if (value == null) {
483                         throw new IOException(
484                                 "init-param without param-value.  Maybe the freemarker.properties is not well-formed?");
485                     }
486                     addSetting(name, value);
487                 }
488             }
489         } catch (IOException e) {
490             LOG.error("Error while loading freemarker settings from /freemarker.properties", e);
491         } catch (TemplateException e) {
492             LOG.error("Error while loading freemarker settings from /freemarker.properties", e);
493         }
494     }
495 
496     public void addSetting(String name, String value) throws TemplateException {
497         // Process all other init-params:
498         if (name.equals(INITPARAM_NOCACHE)) {
499             nocache = StringUtil.getYesNo(value);
500         } else if (name.equals(INITPARAM_DEBUG)) {
501             debug = StringUtil.getYesNo(value);
502         } else if (name.equals(INITPARAM_CONTENT_TYPE)) {
503             contentType = value;
504         } else {
505             config.setSetting(name, value);
506         }
507 
508         if (contentType != null && !contentTypeEvaluated) {
509             int i = contentType.toLowerCase().indexOf("charset=");
510             contentTypeEvaluated = true;
511             if (i != -1) {
512                 char c = ' ';
513                 i--;
514                 while (i >= 0) {
515                     c = contentType.charAt(i);
516                     if (!Character.isWhitespace(c)) break;
517                     i--;
518                 }
519                 if (i == -1 || c == ';') {
520                     noCharsetInContentType = false;
521                 }
522             }
523         }
524     }

528     public ScopesHashModel buildTemplateModel(ValueStack stack, Object action, ServletContext servletContext, HttpServletRequest request, HttpServletResponse response, ObjectWrapper wrapper) {
529         ScopesHashModel model = buildScopesHashModel(servletContext, request, response, wrapper, stack);
530         populateContext(model, stack, action, request, response);
531         if (tagLibraries != null) {
532             for (Map.Entry<String, TagLibraryModelProvider> entry : tagLibraries.entrySet()) {
533                 model.put(entry.getKey(), entry.getValue().getModels(stack, request, response));
534             }
535         }
536 
537         //place the model in the request using the special parameter.  This can be retrieved for freemarker and velocity.
538         request.setAttribute(ATTR_TEMPLATE_MODEL, model);
539 
540         return model;
541     }

544     protected void populateContext(ScopesHashModel model, ValueStack stack, Object action, HttpServletRequest request, HttpServletResponse response) {
545         // put the same objects into the context that the velocity result uses
546         Map standard = ContextUtil.getStandardContext(stack, request, response);
547         model.putAll(standard);
548 
549         // support for JSP exception pages, exposing the servlet or JSP exception
550         Throwable exception = (Throwable) request.getAttribute("javax.servlet.error.exception");
551 
552         if (exception == null) {
553             exception = (Throwable) request.getAttribute("javax.servlet.error.JspException");
554         }
555 
556         if (exception != null) {
557             model.put(KEY_EXCEPTION, exception);
558         }
559     }
