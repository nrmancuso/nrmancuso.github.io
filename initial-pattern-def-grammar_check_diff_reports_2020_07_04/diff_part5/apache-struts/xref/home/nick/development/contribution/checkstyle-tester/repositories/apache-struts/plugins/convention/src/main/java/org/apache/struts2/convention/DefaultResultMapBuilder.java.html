47  /**
48   * <p>
49   * This class implements the ResultMapBuilder and traverses the web
50   * application content directory looking for reasonably named JSPs and
51   * other result types as well as annotations. This naming is in this
52   * form:
53   * </p>
54   *
55   * <pre>
56   * /resultPath/namespace/action-&lt;result&gt;.jsp
57   * </pre>
58   *
59   * <p>
60   * If there are any files in these locations than a result is created
61   * for each one and the result names is the last portion of the file
62   * name up to the . (dot).
63   * </p>
64   *
65   * <p>
66   * When results are found, new ResultConfig instances are created. The
67   * result config that is created has a number of thing to be aware of:
68   * </p>
69   *
70   * <ul>
71   * <li>The result config contains the location parameter, which is
72   *  required by most result classes to figure out where to find the result.
73   *  In addition, the config has all the parameters from the default result-type
74   *  configuration.</li>
75   * </ul>
76   *
77   * <p>
78   * After loading the files in the web application, this class will then
79   * use any annotations on the action class to override what was found in
80   * the web application files. These annotations are the {@link Result}
81   * and {@link Results} annotations. These two annotations allow an action
82   * to supply different or non-forward based results for specific return
83   * values of an action method.
84   * </p>
85   *
86   * <p>
87   * The result path used by this class for locating JSPs and other
88   * such result files can be set using the Struts2 constant named
89   * <strong>struts.convention.result.path</strong> or using the
90   * {@link org.apache.struts2.convention.annotation.ResultPath}
91   * annotation.
92   * </p>
93   *
94   * <p>
95   * This class will also locate and configure Results in the classpath,
96   * including velocity and FreeMarker templates inside the classpath.
97   * </p>
98   *
99   * <p>
100  * All results that are configured from resources are given a type corresponding
101  * to the resources extension. The extensions and types are given in the
102  * table below:
103  * </p>
104  *
105  * <table summary="">
106  * <tr><th>Extension</th><th>Type</th></tr>
107  * <tr><td>.jsp</td><td>dispatcher</td></tr>
108  * <tr><td>.jspx</td><td>dispatcher</td></tr>
109  * <tr><td>.html</td><td>dispatcher</td></tr>
110  * <tr><td>.htm</td><td>dispatcher</td></tr>
111  * <tr><td>.vm</td><td>velocity</td></tr>
112  * <tr><td>.ftl</td><td>freemarker</td></tr>
113  * </table>
114  */
115 public class DefaultResultMapBuilder implements ResultMapBuilder {
116     private static final Logger LOG = LogManager.getLogger(DefaultResultMapBuilder.class);
117     private final ServletContext servletContext;
118     private Set<String> relativeResultTypes;
119     private ConventionsService conventionsService;
120     private boolean flatResultLayout = true;
121 
122     /**
123      * Constructs the SimpleResultMapBuilder using the given result location.
124      *
125      * @param   servletContext The ServletContext for finding the resources of the web application.
126      * @param   container The Xwork container
127      * @param   relativeResultTypes The list of result types that can have locations that are relative
128      *          and the result location (which is the resultPath plus the namespace) prepended to them.
129      */
130     @Inject
131     public DefaultResultMapBuilder(ServletContext servletContext, Container container,
132             @Inject(ConventionConstants.CONVENTION_RELATIVE_RESULT_TYPES) String relativeResultTypes) {
133         this.servletContext = servletContext;
134         this.relativeResultTypes = new HashSet<>(Arrays.asList(relativeResultTypes.split("\\s*[,]\\s*")));
135         this.conventionsService = container.getInstance(ConventionsService.class, container.getInstance(String.class, ConventionConstants.CONVENTION_CONVENTIONS_SERVICE));
136     }
137 
138     /**
139      * @param flatResultLayout If 'true' result resources will be expected to be in the form
140      *          ${namespace}/${actionName}-${result}.${extension}, otherwise in the form
141      *          ${namespace}/${actionName}/${result}.${extension}
142      */
143     @Inject(ConventionConstants.CONVENTION_RESULT_FLAT_LAYOUT)
144     public void setFlatResultLayout(String flatResultLayout) {
145         this.flatResultLayout = BooleanUtils.toBoolean(flatResultLayout);
146     }

148     /**
149      * {@inheritDoc}
150      */
151     public Map<String, ResultConfig> build(Class<?> actionClass,
152         org.apache.struts2.convention.annotation.Action annotation, String actionName,
153             PackageConfig packageConfig) {
154 
155         // Get the default result location from the annotation or configuration
156         String defaultResultPath = conventionsService.determineResultPath(actionClass);
157 
158         // Add a slash
159         if (!defaultResultPath.endsWith("/")) {
160             defaultResultPath = defaultResultPath + "/";
161         }
162 
163         // Check for resources with the action name
164         final String namespace = packageConfig.getNamespace();
165         if (namespace != null && namespace.startsWith("/")) {
166              defaultResultPath = defaultResultPath + namespace.substring(1);
167         } else if (namespace != null) {
168             defaultResultPath = defaultResultPath + namespace;
169         }
170 
171         if (LOG.isTraceEnabled()) {
172             LOG.trace("Using final calculated namespace [{}]", namespace);
173         }
174 
175         // Add that ending slash for concatenation
176         if (!defaultResultPath.endsWith("/")) {
177             defaultResultPath += "/";
178         }
179 
180         String resultPrefix = defaultResultPath + actionName;
181 
182         //results from files
183         Map<String, ResultConfig> results = new HashMap<>();
184         Map<String, ResultTypeConfig> resultsByExtension = conventionsService.getResultTypesByExtension(packageConfig);
185         createFromResources(actionClass, results, defaultResultPath, resultPrefix, actionName,
186             packageConfig, resultsByExtension);
187 
188         //get inherited @Results and @Result (class level)
189         for (Class<?> clazz : ReflectionTools.getClassHierarchy(actionClass)) {
190             createResultsFromAnnotations(clazz, packageConfig, defaultResultPath, results, resultsByExtension);
191         }
192 
193 
194         //method level
195         if (annotation != null && annotation.results() != null && annotation.results().length > 0) {
196             createFromAnnotations(results, defaultResultPath, packageConfig, annotation.results(),
197                     actionClass, resultsByExtension);
198         }
199         return results;
200     }
201 
202     /**
203      * Creates results from @Results and @Result annotations
204      * @param actionClass  class to check for annotations
205      * @param packageConfig packageConfig where the action will be located
206      * @param defaultResultPath default result path
207      * @param results map of results
208      * @param resultsByExtension  map of result types keyed by extension
209      */
210     protected void createResultsFromAnnotations(Class<?> actionClass, PackageConfig packageConfig, String defaultResultPath,
211                                                 Map<String, ResultConfig> results, Map<String, ResultTypeConfig> resultsByExtension) {
212         Results resultsAnn = actionClass.getAnnotation(Results.class);
213         if (resultsAnn != null) {
214             createFromAnnotations(results, defaultResultPath, packageConfig, resultsAnn.value(),
215                     actionClass, resultsByExtension);
216         }
217 
218         Result resultAnn = actionClass.getAnnotation(Result.class);
219         if (resultAnn != null) {
220             createFromAnnotations(results, defaultResultPath, packageConfig, new Result[]{resultAnn},
221                     actionClass, resultsByExtension);
222         }
223     }

226     /**
227      * Creates any result types from the resources available in the web application. This scans the
228      * web application resources using the servlet context.
229      *
230      * @param   actionClass The action class the results are being built for.
231      * @param   results The results map to put the result configs created into.
232      * @param   resultPath The calculated path to the resources.
233      * @param   resultPrefix The prefix for the result. This is usually <code>/resultPath/actionName</code>.
234      * @param   actionName The action name which is used only for logging in this implementation.
235      * @param   packageConfig The package configuration which is passed along in order to determine
236      * @param   resultsByExtension The map of extensions to result type configuration instances.
237      */
238     protected void createFromResources(Class<?> actionClass, Map<String, ResultConfig> results,
239             final String resultPath, final String resultPrefix, final String actionName,
240             PackageConfig packageConfig, Map<String, ResultTypeConfig> resultsByExtension) {
241         if (LOG.isTraceEnabled()) {
242             LOG.trace("Searching for results in the Servlet container at [{}]" +
243             		" with result prefix of [#1]", resultPath, resultPrefix);
244         }
245 
246         // Build from web application using the ServletContext
247         @SuppressWarnings("unchecked")
248         Set<String> paths = servletContext.getResourcePaths(flatResultLayout ? resultPath : resultPrefix);
249         if (paths != null) {
250             for (String path : paths) {
251                 LOG.trace("Processing resource path [{}]", path);
252 
253                 String fileName = StringUtils.substringAfterLast(path, "/");
254                 if (StringUtils.isBlank(fileName) || StringUtils.startsWith(fileName, ".")) {
255                     LOG.trace("Ignoring file without name [{}]", path);
256                     continue;
257                 }
258                 else if(fileName.lastIndexOf(".") > 0){
259                     String suffix = fileName.substring(fileName.lastIndexOf(".")+1);
260 
261                     if(conventionsService.getResultTypesByExtension(packageConfig).get(suffix) == null) {
262                         LOG.debug("No result type defined for file suffix : [{}]. Ignoring file {}", suffix, fileName);
263                     	continue;
264                     }
265                 }
266 
267                 makeResults(actionClass, path, resultPrefix, results, packageConfig, resultsByExtension);
268             }
269         }
270 
271         // Building from the classpath
272         String classPathLocation = resultPath.startsWith("/") ?
273             resultPath.substring(1, resultPath.length()) : resultPath;
274         if (LOG.isTraceEnabled()) {
275             LOG.trace("Searching for results in the class path at [{}]"
276                     + " with a result prefix of [{}] and action name [{}]", classPathLocation, resultPrefix,
277                     actionName);
278         }
279 
280         ResourceFinder finder = new ResourceFinder(classPathLocation, getClassLoaderInterface());
281         try {
282             Map<String, URL> matches = finder.getResourcesMap("");
283             if (matches != null) {
284                 Test<URL> resourceTest = getResourceTest(resultPath, actionName);
285                 for (Map.Entry<String, URL> entry : matches.entrySet()) {
286                     if (resourceTest.test(entry.getValue())) {
287                         LOG.trace("Processing URL [{}]", entry.getKey());
288 
289                         String urlStr = entry.getValue().toString();
290                         int index = urlStr.lastIndexOf(resultPrefix);
291                         String path = urlStr.substring(index);
292                         makeResults(actionClass, path, resultPrefix, results, packageConfig, resultsByExtension);
293                     }
294 
295                 }
296             }
297         } catch (IOException ex) {
298             LOG.error("Unable to scan directory [{}] for results", ex, classPathLocation);
299         }
300     }

302     protected ClassLoaderInterface getClassLoaderInterface() {
303 
304         /*
305         if there is a ClassLoaderInterface in the context, use it, otherwise
306         default to the default ClassLoaderInterface (a wrapper around the current
307         thread classloader)
308         using this, other plugins (like OSGi) can plugin their own classloader for a while
309         and it will be used by Convention (it cannot be a bean, as Convention is likely to be
310         called multiple times, and it need to use the default ClassLoaderInterface during normal startup)
311         */
312         ClassLoaderInterface classLoaderInterface = null;
313         ActionContext ctx = ActionContext.getContext();
314         if (ctx != null)
315             classLoaderInterface = (ClassLoaderInterface) ctx.get(ClassLoaderInterface.CLASS_LOADER_INTERFACE);
316 
317         return ObjectUtils.defaultIfNull(classLoaderInterface, new ClassLoaderInterfaceDelegate(Thread.currentThread().getContextClassLoader()));
318 
319     }

321     private Test<URL> getResourceTest(final String resultPath, final String actionName) {
322         return new Test<URL>() {
323             public boolean test(URL url) {
324                 String urlStr = url.toString();
325                 int index = urlStr.lastIndexOf(resultPath);
326                 String path = urlStr.substring(index + resultPath.length());
327                 return path.startsWith(actionName);
328             }
329         };
330     }

332     /**
333      * Makes all the results for the given path.
334      *
335      * @param   actionClass The action class the results are being built for.
336      * @param   path The path to build the result for.
337      * @param   resultPrefix The is the result prefix which is the result location plus the action name.
338      *          This is used to determine if the path contains a result code or not.
339      * @param   results The Map to place the result(s)
340      * @param   packageConfig The package config the results belong to.
341      * @param   resultsByExtension The map of extensions to result type configuration instances.
342      */
343     protected void makeResults(Class<?> actionClass, String path, String resultPrefix,
344             Map<String, ResultConfig> results, PackageConfig packageConfig,
345             Map<String, ResultTypeConfig> resultsByExtension) {
346 
347         if (path.startsWith(resultPrefix)) {
348             int indexOfDot = path.indexOf('.', resultPrefix.length());
349 
350             // This case is when the path doesn't contain a result code
351             if (indexOfDot == resultPrefix.length()) {
352                 if (LOG.isTraceEnabled()) {
353                     LOG.trace("The result file [{}] has no result code and therefore" +
354                         " will be associated with success, input and error by default. This might" +
355                         " be overridden by another result file or an annotation.", path);
356                 }
357 
358                 addResult(actionClass, path, results, packageConfig, resultsByExtension, Action.SUCCESS);
359                 addResult(actionClass, path, results, packageConfig, resultsByExtension, Action.INPUT);
360                 addResult(actionClass, path, results, packageConfig, resultsByExtension, Action.ERROR);
361 
362             // This case is when the path contains a result code
363             } else if (indexOfDot > resultPrefix.length()) {
364                 if (LOG.isTraceEnabled()) {
365                     LOG.trace("The result file [{}] has a result code and therefore" +
366                         " will be associated with only that result code.", path);
367                 }
368 
369                 String resultCode = path.substring(resultPrefix.length() + 1, indexOfDot);
370                 ResultConfig result = createResultConfig(actionClass,
371                     new ResultInfo(resultCode, path, packageConfig, resultsByExtension),
372                     packageConfig, null);
373                 results.put(resultCode, result);
374             }
375         }
376     }

378     /**
379      * Checks if result was already assigned, if not checks global results first and if exists, adds reference to it.
380      * If not, creates package specific result.
381      *
382      * @param   actionClass The action class the results are being built for.
383      * @param   path The path to build the result for.
384      * @param   results The Map to place the result(s)
385      * @param   packageConfig The package config the results belong to.
386      * @param   resultsByExtension The map of extensions to result type configuration instances.
387      * @param   resultKey The result name to use
388      */
389     protected void addResult(Class<?> actionClass, String path, Map<String, ResultConfig> results,
390                            PackageConfig packageConfig, Map<String, ResultTypeConfig> resultsByExtension,
391                            String resultKey) {
392 
393         if (!results.containsKey(resultKey)) {
394             Map<String, ResultConfig> globalResults = packageConfig.getAllGlobalResults();
395             if (globalResults.containsKey(resultKey)) {
396                 results.put(resultKey, globalResults.get(resultKey));
397             } else {
398                 ResultConfig resultConfig = createResultConfig(actionClass,
399                         new ResultInfo(resultKey, path, packageConfig, resultsByExtension),
400                         packageConfig, null);
401                 results.put(resultKey, resultConfig);
402             }
403         }
404     }

406     protected void createFromAnnotations(Map<String, ResultConfig> resultConfigs,
407             String resultPath, PackageConfig packageConfig, Result[] results,
408             Class<?> actionClass, Map<String, ResultTypeConfig> resultsByExtension) {
409         // Check for multiple results on the class
410         for (Result result : results) {
411             for (String name : result.name()) {
412                 ResultConfig config = createResultConfig(actionClass, new ResultInfo(
413                         name, result, packageConfig, resultPath, actionClass,
414                         resultsByExtension), packageConfig, result);
415                 if (config != null) {
416                     resultConfigs.put(config.getName(), config);
417                 }
418             }
419         }
420     }

422     /**
423      * Creates the result configuration for the single result annotation. This will use all the
424      * information from the annotation and anything that isn't specified will be fetched from the
425      * PackageConfig defaults (if they exist).
426      *
427      * @param   actionClass The action class the results are being built for.
428      * @param   info The result info that is used to create the ResultConfig instance.
429      * @param   packageConfig The PackageConfig to use to fetch defaults for result and parameters.
430      * @param   result (Optional) The result annotation to pull additional information from.
431      * @return  The ResultConfig or null if the Result annotation is given and the annotation is
432      *          targeted to some other action than this one.
433      */
434     @SuppressWarnings(value = {"unchecked"})
435     protected ResultConfig createResultConfig(Class<?> actionClass, ResultInfo info,
436             PackageConfig packageConfig, Result result) {
437         // Look up by the type that was determined from the annotation or by the extension in the
438         // ResultInfo class
439         ResultTypeConfig resultTypeConfig = packageConfig.getAllResultTypeConfigs().get(info.type);
440         if (resultTypeConfig == null) {
441             throw new ConfigurationException("The Result type [" + info.type + "] which is" +
442                 " defined in the Result annotation on the class [" + actionClass + "] or determined" +
443                 " by the file extension or is the default result type for the PackageConfig of the" +
444                 " action, could not be found as a result-type defined for the Struts/XWork package [" +
445                 packageConfig.getName() + "]");
446         }
447 
448         // Add the default parameters for the result type config (if any)
449         HashMap<String, String> params = new HashMap<>();
450         if (resultTypeConfig.getParams() != null) {
451             params.putAll(resultTypeConfig.getParams());
452         }
453 
454         // Handle the annotation
455         if (result != null) {
456             params.putAll(StringTools.createParameterMap(result.params()));
457         }
458 
459         // Map the location to the default param for the result or a param named location
460         if (info.location != null) {
461             String defaultParamName = resultTypeConfig.getDefaultResultParam();
462             if (!params.containsKey(defaultParamName)) {
463                 params.put(defaultParamName, info.location);
464             }
465         }
466 
467         return new ResultConfig.Builder(info.name, resultTypeConfig.getClassName()).addParams(params).build();
468     }

470     protected class ResultInfo {
471         public final String name;
472         public final String location;
473         public final String type;
474 
475         public ResultInfo(String name, String location, PackageConfig packageConfig,
476                 Map<String, ResultTypeConfig> resultsByExtension) {
477             this.name = name;
478             this.location = location;
479             this.type = determineType(location, packageConfig, resultsByExtension);
480         }
481 
482         public ResultInfo(String name, Result result, PackageConfig packageConfig,
483                 String resultPath, Class<?> actionClass,
484                 Map<String, ResultTypeConfig> resultsByExtension) {
485             this.name = name;
486             if (StringUtils.isNotBlank(result.type())) {
487                 this.type = result.type();
488             } else if (StringUtils.isNotBlank(result.location())) {
489                 this.type = determineType(result.location(), packageConfig, resultsByExtension);
490             } else {
491                 throw new ConfigurationException("The action class [" + actionClass + "] contains a " +
492                     "result annotation that has no type parameter and no location parameter. One of " +
493                     "these must be defined.");
494             }
495 
496             // See if we can handle relative locations or not
497             if (StringUtils.isNotBlank(result.location())) {
498                 if (relativeResultTypes.contains(this.type) && !result.location().startsWith("/")) {
499                     location = resultPath + result.location();
500                 } else {
501                     location = result.location();
502                 }
503             } else {
504                 this.location = null;
505             }
506         }
507 
508         String determineType(String location, PackageConfig packageConfig,
509                 Map<String, ResultTypeConfig> resultsByExtension) {
510             int indexOfDot = location.lastIndexOf(".");
511             if (indexOfDot > 0) {
512                 String extension = location.substring(indexOfDot + 1);
513                 ResultTypeConfig resultTypeConfig = resultsByExtension.get(extension);
514                 if (resultTypeConfig != null) {
515                     return resultTypeConfig.getName();
516                 } else
517                     throw new ConfigurationException("Unable to find a result type for extension [" + extension + "] " +
518                     		"in location attribute [" + location + "].");
519             } else {
520                 return packageConfig.getFullDefaultResultType();
521             }
522         }
523     }
