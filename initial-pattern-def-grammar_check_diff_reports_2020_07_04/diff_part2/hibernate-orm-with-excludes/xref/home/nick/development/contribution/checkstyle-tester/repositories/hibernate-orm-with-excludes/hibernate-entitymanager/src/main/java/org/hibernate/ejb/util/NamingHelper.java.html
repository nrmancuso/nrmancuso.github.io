24  package org.hibernate.ejb.util;
25  
26  import javax.naming.event.NamespaceChangeListener;
27  import javax.naming.event.NamingEvent;
28  import javax.naming.event.NamingExceptionEvent;
29  
30  import org.jboss.logging.Logger;
31  
32  import org.hibernate.ejb.AvailableSettings;
33  import org.hibernate.ejb.Ejb3Configuration;
34  import org.hibernate.ejb.internal.EntityManagerMessageLogger;
35  import org.hibernate.service.jndi.JndiException;
36  import org.hibernate.service.jndi.JndiNameException;
37  import org.hibernate.service.jndi.internal.JndiServiceImpl;
38  
39  /**
40   * @author Emmanuel Bernard
41   */
42  public class NamingHelper {
43  	private NamingHelper() {}
44  
45      private static final EntityManagerMessageLogger LOG = Logger.getMessageLogger(EntityManagerMessageLogger.class, NamingHelper.class.getName());
46  
47  	public static void bind(Ejb3Configuration cfg) {
48  		String name = cfg.getHibernateConfiguration().getProperty( AvailableSettings.CONFIGURATION_JNDI_NAME );
49          if ( name == null ) {
50  			LOG.debug( "No JNDI name configured for binding Ejb3Configuration" );
51  		}
52  		else {
53              LOG.ejb3ConfigurationName( name );
54  
55  			// todo : instantiating the JndiService here is temporary until HHH-6159 is resolved.
56  			JndiServiceImpl jndiService = new JndiServiceImpl( cfg.getProperties() );
57  			try {
58  				jndiService.bind( name, cfg );
59  				LOG.boundEjb3ConfigurationToJndiName( name );
60  				try {
61  					jndiService.addListener( name, LISTENER );
62  				}
63  				catch (Exception e) {
64  					LOG.couldNotBindJndiListener();
65  				}
66  			}
67  			catch (JndiNameException e) {
68  				LOG.invalidJndiName( name, e );
69  			}
70  			catch (JndiException e) {
71  				LOG.unableToBindEjb3ConfigurationToJndi( e );
72  			}
73  		}
74  	}
75  
76  	private static final NamespaceChangeListener LISTENER = new NamespaceChangeListener() {
77  		public void objectAdded(NamingEvent evt) {
78              LOG.debugf("An Ejb3Configuration was successfully bound to name: %s", evt.getNewBinding().getName());
79  		}
80  
81  		public void objectRemoved(NamingEvent evt) {
82  			String name = evt.getOldBinding().getName();
83              LOG.ejb3ConfigurationUnboundFromName(name);
84  		}
85  
86  		public void objectRenamed(NamingEvent evt) {
87  			String name = evt.getOldBinding().getName();
88              LOG.ejb3ConfigurationRenamedFromName(name);
89  		}
90  
91  		public void namingExceptionThrown(NamingExceptionEvent evt) {
92              LOG.unableToAccessEjb3Configuration(evt.getException());
93  		}
94  	};
