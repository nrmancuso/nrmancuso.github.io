56  /**
57   * This class implements the LDAPv3 Extended Response for StartTLS as
58   * defined in
59   * <a href="http://www.ietf.org/rfc/rfc2830.txt">Lightweight Directory
60   * Access Protocol (v3): Extension for Transport Layer Security</a>
61   *
62   * The object identifier for StartTLS is 1.3.6.1.4.1.1466.20037
63   * and no extended response value is defined.
64   *
65   *<p>
66   * The Start TLS extended request and response are used to establish
67   * a TLS connection over the existing LDAP connection associated with
68   * the JNDI context on which <tt>extendedOperation()</tt> is invoked.
69   *
70   * @see StartTlsRequest
71   * @author Vincent Ryan
72   */

181     /**
182      * Negotiates a TLS session using an SSL socket factory.
183      * <p>
184      * Creates an SSL socket using the supplied SSL socket factory and
185      * attaches it to the existing connection. Performs the TLS handshake
186      * and returns the negotiated session information.
187      * <p>
188      * If cipher suites have been set via <tt>setEnabledCipherSuites</tt>
189      * then they are enabled before the TLS handshake begins.
190      * <p>
191      * Hostname verification is performed after the TLS handshake completes.
192      * The default check performs a case insensitive match of the server's
193      * hostname against that in the server's certificate. The server's
194      * hostname is extracted from the subjectAltName in the server's
195      * certificate (if present). Otherwise the value of the common name
196      * attribute of the subject name is used. If a callback has
197      * been set via <tt>setHostnameVerifier</tt> then that verifier is used if
198      * the default check fails.
199      * <p>
200      * If an error occurs then the SSL socket is closed and an IOException
201      * is thrown. The underlying connection remains intact.
202      *
203      * @param factory The possibly null SSL socket factory to use.
204      * If null, the default SSL socket factory is used.
205      * @return The negotiated SSL session
206      * @throw IOException If an IO error was encountered while establishing
207      * the TLS session.
208      * @see #setEnabledCipherSuites
209      * @see #setHostnameVerifier
210      */
211     public SSLSession negotiate(SSLSocketFactory factory) throws IOException {
212 
213         if (isClosed && sslSocket != null) {
214             throw new IOException("TLS connection is closed.");
215         }
216 
217         if (factory == null) {
218             factory = getDefaultFactory();
219         }
220 
221         if (debug) {
222             System.out.println("StartTLS: About to start handshake");
223         }
224 
225         SSLSession sslSession = startHandshake(factory).getSession();
226 
227         if (debug) {
228             System.out.println("StartTLS: Completed handshake");
229         }
230 
231         SSLPeerUnverifiedException verifExcep = null;
232         try {
233             if (verify(hostname, sslSession)) {
234                 isClosed = false;
235                 return sslSession;
236             }
237         } catch (SSLPeerUnverifiedException e) {
238             // Save to return the cause
239             verifExcep = e;
240         }
241         if ((verifier != null) &&
242                 verifier.verify(hostname, sslSession)) {
243             isClosed = false;
244             return sslSession;
245         }
246 
247         // Verification failed
248         close();
249         sslSession.invalidate();
250         if (verifExcep == null) {
251             verifExcep = new SSLPeerUnverifiedException(
252                         "hostname of the server '" + hostname +
253                         "' does not match the hostname in the " +
254                         "server's certificate.");
255         }
256         throw verifExcep;
257     }
258 
259     /**
260      * Closes the TLS connection gracefully and reverts back to the underlying
261      * connection.
262      *
263      * @throw IOException If an IO error was encountered while closing the
264      * TLS connection
265      */
266     public void close() throws IOException {
267 
268         if (isClosed) {
269             return;
270         }
271 
272         if (debug) {
273             System.out.println("StartTLS: replacing SSL " +
274                                 "streams with originals");
275         }
276 
277         // Replace SSL streams with the original streams
278         ldapConnection.replaceStreams(
279                         originalInputStream, originalOutputStream);
280 
281         if (debug) {
282             System.out.println("StartTLS: closing SSL Socket");
283         }
284         sslSocket.close();
285 
286         isClosed = true;
287     }

289     /**
290      * Sets the connection for TLS to use. The TLS connection will be attached
291      * to this connection.
292      *
293      * @param ldapConnection The non-null connection to use.
294      * @param hostname The server's hostname. If null, the hostname used to
295      * open the connection will be used instead.
296      */
297     public void setConnection(Connection ldapConnection, String hostname) {
298         this.ldapConnection = ldapConnection;
299         this.hostname = (hostname != null) ? hostname : ldapConnection.host;
300         originalInputStream = ldapConnection.inStream;
301         originalOutputStream = ldapConnection.outStream;
302     }

304     /*
305      * Returns the default SSL socket factory.
306      *
307      * @return The default SSL socket factory.
308      * @throw IOException If TLS is not supported.
309      */
310     private SSLSocketFactory getDefaultFactory() throws IOException {
311 
312         if (defaultFactory != null) {
313             return defaultFactory;
314         }
315 
316         return (defaultFactory =
317             (SSLSocketFactory) SSLSocketFactory.getDefault());
318     }

320     /*
321      * Start the TLS handshake and manipulate the input and output streams.
322      *
323      * @param factory The SSL socket factory to use.
324      * @return The SSL socket.
325      * @throw IOException If an exception occurred while performing the
326      * TLS handshake.
327      */
328     private SSLSocket startHandshake(SSLSocketFactory factory)
329         throws IOException {
330 
331         if (ldapConnection == null) {
332             throw new IllegalStateException("LDAP connection has not been set."
333                 + " TLS requires an existing LDAP connection.");
334         }
335 
336         if (factory != currentFactory) {
337             // Create SSL socket layered over the existing connection
338             sslSocket = (SSLSocket) factory.createSocket(ldapConnection.sock,
339                 ldapConnection.host, ldapConnection.port, false);
340             currentFactory = factory;
341 
342             if (debug) {
343                 System.out.println("StartTLS: Created socket : " + sslSocket);
344             }
345         }
346 
347         if (suites != null) {
348             sslSocket.setEnabledCipherSuites(suites);
349             if (debug) {
350                 System.out.println("StartTLS: Enabled cipher suites");
351             }
352         }
353 
354         // Connection must be quite for handshake to proceed
355 
356         try {
357             if (debug) {
358                 System.out.println(
359                         "StartTLS: Calling sslSocket.startHandshake");
360             }
361             sslSocket.startHandshake();
362             if (debug) {
363                 System.out.println(
364                         "StartTLS: + Finished sslSocket.startHandshake");
365             }
366 
367             // Replace original streams with the new SSL streams
368             ldapConnection.replaceStreams(sslSocket.getInputStream(),
369                 sslSocket.getOutputStream());
370             if (debug) {
371                 System.out.println("StartTLS: Replaced IO Streams");
372             }
373 
374         } catch (IOException e) {
375             if (debug) {
376                 System.out.println("StartTLS: Got IO error during handshake");
377                 e.printStackTrace();
378             }
379 
380             sslSocket.close();
381             isClosed = true;
382             throw e;   // pass up exception
383         }
384 
385         return sslSocket;
386     }

388     /*
389      * Verifies that the hostname in the server's certificate matches the
390      * hostname of the server.
391      * The server's first certificate is examined. If it has a subjectAltName
392      * that contains a dNSName then that is used as the server's hostname.
393      * The server's hostname may contain a wildcard for its left-most name part.
394      * Otherwise, if the certificate has no subjectAltName then the value of
395      * the common name attribute of the subject name is used.
396      *
397      * @param hostname The hostname of the server.
398      * @param session the SSLSession used on the connection to host.
399      * @return true if the hostname is verified, false otherwise.
400      */
401 
402     private boolean verify(String hostname, SSLSession session)
403         throws SSLPeerUnverifiedException {
404 
405         java.security.cert.Certificate[] certs = null;
406 
407         // if IPv6 strip off the "[]"
408         if (hostname != null && hostname.startsWith("[") &&
409                 hostname.endsWith("]")) {
410             hostname = hostname.substring(1, hostname.length() - 1);
411         }
412         try {
413             HostnameChecker checker = HostnameChecker.getInstance(
414                                                 HostnameChecker.TYPE_LDAP);
415             // Use ciphersuite to determine whether Kerberos is active.
416             if (session.getCipherSuite().startsWith("TLS_KRB5")) {
417                 Principal principal = getPeerPrincipal(session);
418                 if (!checker.match(hostname, principal)) {
419                     throw new SSLPeerUnverifiedException(
420                         "hostname of the kerberos principal:" + principal +
421                         " does not match the hostname:" + hostname);
422                 }
423             } else { // X.509
424 
425                 // get the subject's certificate
426                 certs = session.getPeerCertificates();
427                 X509Certificate peerCert;
428                 if (certs[0] instanceof java.security.cert.X509Certificate) {
429                     peerCert = (java.security.cert.X509Certificate) certs[0];
430                 } else {
431                     throw new SSLPeerUnverifiedException(
432                             "Received a non X509Certificate from the server");
433                 }
434                 checker.match(hostname, peerCert);
435             }
436 
437             // no exception means verification passed
438             return true;
439         } catch (SSLPeerUnverifiedException e) {
440 
441             /*
442              * The application may enable an anonymous SSL cipher suite, and
443              * hostname verification is not done for anonymous ciphers
444              */
445             String cipher = session.getCipherSuite();
446             if (cipher != null && (cipher.indexOf("_anon_") != -1)) {
447                 return true;
448             }
449             throw e;
450         } catch (CertificateException e) {
451 
452             /*
453              * Pass up the cause of the failure
454              */
455             throw(SSLPeerUnverifiedException)
456                 new SSLPeerUnverifiedException("hostname of the server '" +
457                                 hostname +
458                                 "' does not match the hostname in the " +
459                                 "server's certificate.").initCause(e);
460         }
461     }
462 
463     /*
464      * Get the peer principal from the session
465      */
466     private static Principal getPeerPrincipal(SSLSession session)
467             throws SSLPeerUnverifiedException {
468         Principal principal;
469         try {
470             principal = session.getPeerPrincipal();
471         } catch (AbstractMethodError e) {
472             // if the JSSE provider does not support it, return null, since
473             // we need it only for Kerberos.
474             principal = null;
475         }
476         return principal;
477     }
