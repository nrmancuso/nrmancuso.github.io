55  	@Test
56  	public void verifySendMulticast() throws Exception {
57  		MulticastSocket socket;
58  		try {
59  			socket = new MulticastSocket();
60  		}
61  		catch (Exception e) {
62  			return;
63  		}
64  		final int testPort = socket.getLocalPort();
65  		final String multicastAddress = this.multicastRule.getGroup();
66  		final String payload = "foo";
67  		final CountDownLatch listening = new CountDownLatch(2);
68  		final CountDownLatch received = new CountDownLatch(2);
69  		Runnable catcher = () -> {
70  			try {
71  				byte[] buffer = new byte[8];
72  				DatagramPacket receivedPacket = new DatagramPacket(buffer, buffer.length);
73  				MulticastSocket socket1 = new MulticastSocket(testPort);
74  				socket1.setNetworkInterface(multicastRule.getNic());
75  				InetAddress group = InetAddress.getByName(multicastAddress);
76  				socket1.joinGroup(new InetSocketAddress(group, 0), null);
77  				listening.countDown();
78  				socket1.receive(receivedPacket);
79  				socket1.close();
80  				byte[] src = receivedPacket.getData();
81  				int length = receivedPacket.getLength();
82  				int offset = receivedPacket.getOffset();
83  				byte[] dest = new byte[length];
84  				System.arraycopy(src, offset, dest, 0, length);
85  				assertThat(new String(dest)).isEqualTo(payload);
86  				received.countDown();
87  			}
88  			catch (Exception e) {
89  				listening.countDown();
90  				e.printStackTrace();
91  			}
92  		};
93  		Executor executor = new SimpleAsyncTaskExecutor();
94  		executor.execute(catcher);
95  		executor.execute(catcher);
96  		assertThat(listening.await(10000, TimeUnit.MILLISECONDS)).isTrue();
97  		MulticastSendingMessageHandler handler = new MulticastSendingMessageHandler(multicastAddress, testPort);
98  		handler.setBeanFactory(mock(BeanFactory.class));
99  		NetworkInterface nic = this.multicastRule.getNic();
100 		if (nic != null) {
101 			String hostName = null;
102 			Enumeration<InetAddress> addressesFromNetworkInterface = nic.getInetAddresses();
103 			while (addressesFromNetworkInterface.hasMoreElements()) {
104 				InetAddress inetAddress = addressesFromNetworkInterface.nextElement();
105 				if (inetAddress.isSiteLocalAddress()
106 						&& !inetAddress.isAnyLocalAddress()
107 						&& !inetAddress.isLinkLocalAddress()
108 						&& !inetAddress.isLoopbackAddress()) {
109 
110 					hostName = inetAddress.getHostName();
111 					break;
112 				}
113 			}
114 
115 			handler.setLocalAddress(hostName);
116 		}
117 		handler.afterPropertiesSet();
118 		handler.handleMessage(MessageBuilder.withPayload(payload).build());
119 		assertThat(received.await(10000, TimeUnit.MILLISECONDS)).isTrue();
120 		handler.stop();
121 		socket.close();
122 	}

124 	@Test
125 	public void verifySendMulticastWithAcks() throws Exception {
126 
127 		MulticastSocket socket;
128 		try {
129 			socket = new MulticastSocket();
130 		}
131 		catch (Exception e) {
132 			return;
133 		}
134 		final int testPort = socket.getLocalPort();
135 		final AtomicInteger ackPort = new AtomicInteger();
136 
137 		final String multicastAddress = this.multicastRule.getGroup();
138 		final String payload = "foobar";
139 		final CountDownLatch listening = new CountDownLatch(2);
140 		final CountDownLatch ackListening = new CountDownLatch(1);
141 		final CountDownLatch ackSent = new CountDownLatch(2);
142 		NetworkInterface nic = this.multicastRule.getNic();
143 		Runnable catcher = () -> {
144 			try {
145 				byte[] buffer = new byte[1000];
146 				DatagramPacket receivedPacket = new DatagramPacket(buffer, buffer.length);
147 				MulticastSocket socket1 = new MulticastSocket(testPort);
148 				socket1.setNetworkInterface(multicastRule.getNic());
149 				socket1.setSoTimeout(8000);
150 				InetAddress group = InetAddress.getByName(multicastAddress);
151 				socket1.joinGroup(new InetSocketAddress(group, 0), null);
152 				listening.countDown();
153 				assertThat(ackListening.await(10, TimeUnit.SECONDS)).isTrue();
154 				socket1.receive(receivedPacket);
155 				socket1.close();
156 				byte[] src = receivedPacket.getData();
157 				int length = receivedPacket.getLength();
158 				int offset = receivedPacket.getOffset();
159 				byte[] dest = new byte[6];
160 				System.arraycopy(src, offset + length - 6, dest, 0, 6);
161 				assertThat(new String(dest)).isEqualTo(payload);
162 				DatagramPacketMessageMapper mapper = new DatagramPacketMessageMapper();
163 				mapper.setAcknowledge(true);
164 				mapper.setLengthCheck(true);
165 				Message<byte[]> message = mapper.toMessage(receivedPacket);
166 				Object id = message.getHeaders().get(IpHeaders.ACK_ID);
167 				byte[] ack = id.toString().getBytes();
168 				InetAddress inetAddress = null;
169 				Enumeration<InetAddress> addressesFromNetworkInterface = nic.getInetAddresses();
170 				while (addressesFromNetworkInterface.hasMoreElements()) {
171 					InetAddress address = addressesFromNetworkInterface.nextElement();
172 					if (address.isSiteLocalAddress()
173 							&& !address.isAnyLocalAddress()
174 							&& !address.isLinkLocalAddress()
175 							&& !address.isLoopbackAddress()) {
176 
177 						inetAddress = address;
178 						break;
179 					}
180 				}
181 				DatagramPacket ackPack = new DatagramPacket(ack, ack.length,
182 						new InetSocketAddress(inetAddress, ackPort.get()));
183 				DatagramSocket out = new DatagramSocket();
184 				out.send(ackPack);
185 				out.close();
186 				ackSent.countDown();
187 				socket1.close();
188 			}
189 			catch (Exception e) {
190 				listening.countDown();
191 				e.printStackTrace();
192 			}
193 		};
194 		Executor executor = new SimpleAsyncTaskExecutor();
195 		executor.execute(catcher);
196 		executor.execute(catcher);
197 		assertThat(listening.await(10000, TimeUnit.MILLISECONDS)).isTrue();
198 		MulticastSendingMessageHandler handler =
199 				new MulticastSendingMessageHandler(multicastAddress, testPort, true, true, "localhost", 0, 10000);
200 
201 		if (nic != null) {
202 			String hostName = null;
203 			Enumeration<InetAddress> addressesFromNetworkInterface = nic.getInetAddresses();
204 			while (addressesFromNetworkInterface.hasMoreElements()) {
205 				InetAddress inetAddress = addressesFromNetworkInterface.nextElement();
206 				if (inetAddress.isSiteLocalAddress()
207 						&& !inetAddress.isAnyLocalAddress()
208 						&& !inetAddress.isLinkLocalAddress()
209 						&& !inetAddress.isLoopbackAddress()) {
210 
211 					hostName = inetAddress.getHostName();
212 					break;
213 				}
214 			}
215 
216 			handler.setLocalAddress(hostName);
217 		}
218 		handler.setMinAcksForSuccess(2);
219 		handler.setBeanFactory(mock(BeanFactory.class));
220 		handler.afterPropertiesSet();
221 		handler.start();
222 		waitAckListening(handler);
223 		ackPort.set(handler.getAckPort());
224 		ackListening.countDown();
225 		handler.handleMessage(MessageBuilder.withPayload(payload).build());
226 		assertThat(ackSent.await(10000, TimeUnit.MILLISECONDS)).isTrue();
227 		handler.stop();
228 		socket.close();
229 	}

231 	public void waitAckListening(UnicastSendingMessageHandler handler) throws InterruptedException {
232 		int n = 0;
233 		while (n++ < 100 && handler.getAckPort() == 0) {
234 			Thread.sleep(100);
235 		}
236 		assertThat(n < 100).isTrue();
237 	}
