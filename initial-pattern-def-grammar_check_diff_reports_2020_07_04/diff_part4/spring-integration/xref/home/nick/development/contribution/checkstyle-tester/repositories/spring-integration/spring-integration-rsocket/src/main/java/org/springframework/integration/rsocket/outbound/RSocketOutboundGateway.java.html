43  /**
44   * An Outbound Messaging Gateway for RSocket requests.
45   * The request logic is fully based on the {@link RSocketRequester}, which can be obtained from the
46   * {@link ClientRSocketConnector} on the client side or from the
47   * {@link RSocketRequesterMethodArgumentResolver#RSOCKET_REQUESTER_HEADER} request message header
48   * on the server side.
49   * <p>
50   * An RSocket operation is determined by the configured {@link RSocketInteractionModel} or respective SpEL
51   * expression to be evaluated at runtime against the request message.
52   * By default the {@link RSocketInteractionModel#requestResponse} operation is used.
53   * <p>
54   * For a {@link Publisher}-based requests, it must be present in the request message {@code payload}.
55   * The flattening via upstream {@link org.springframework.integration.channel.FluxMessageChannel} will work, too,
56   * but this way we will lose a scope of particular request and every {@link Publisher} event
57   * will be send in its own plain request.
58   * <p>
59   * If reply is a {@link reactor.core.publisher.Flux}, it is wrapped to the {@link Mono} to retain a request scope.
60   * The downstream flow is responsible to obtain this {@link reactor.core.publisher.Flux} from a message payload
61   * and subscribe to it by itself. The {@link Mono} reply from this component is subscribed from the downstream
62   * {@link org.springframework.integration.channel.FluxMessageChannel} or it is adapted to the
63   * {@link org.springframework.util.concurrent.ListenableFuture} otherwise.
64   *
65   * @author Artem Bilan
66   *
67   * @since 5.2
68   *
69   * @see RSocketInteractionModel
70   * @see RSocketRequester
71   */
72  public class RSocketOutboundGateway extends AbstractReplyProducingMessageHandler {

94  	/**
95  	 * Instantiate based on the provided RSocket endpoint {@code route}
96  	 * and optional variables to expand route template.
97  	 * @param route the RSocket endpoint route to use.
98  	 * @param routeVariables the variables to expand route template.
99  	 */
100 	public RSocketOutboundGateway(String route, @Nullable Object... routeVariables) {
101 		this(new ValueExpression<>(route));
102 		if (routeVariables != null) {
103 			this.routeVars = Arrays.copyOf(routeVariables, routeVariables.length);
104 		}
105 	}

107 	/**
108 	 * Instantiate based on the provided SpEL expression to evaluate an RSocket endpoint {@code route}
109 	 * at runtime against a request message.
110 	 * If route is a template and variables expansion is required, it is recommended to do that
111 	 * in this expression evaluation, for example using some bean with an appropriate logic.
112 	 * @param routeExpression the SpEL expression to use.
113 	 */
114 	public RSocketOutboundGateway(Expression routeExpression) {
115 		Assert.notNull(routeExpression, "'routeExpression' must not be null");
116 		this.routeExpression = routeExpression;
117 		setAsync(true);
118 		setPrimaryExpression(this.routeExpression);
119 	}

205 	@Override
206 	protected void doInit() {
207 		super.doInit();
208 		this.evaluationContext = ExpressionUtils.createStandardEvaluationContext(getBeanFactory());
209 		if (this.clientRSocketConnector != null) {
210 			this.rsocketRequesterMono = this.clientRSocketConnector.getRSocketRequester();
211 		}
212 	}
213 
214 	@Override
215 	protected Object handleRequestMessage(Message<?> requestMessage) {
216 		RSocketRequester rsocketRequester = requestMessage.getHeaders()
217 				.get(RSocketRequesterMethodArgumentResolver.RSOCKET_REQUESTER_HEADER, RSocketRequester.class);
218 		Mono<RSocketRequester> requesterMono;
219 		if (rsocketRequester != null) {
220 			requesterMono = Mono.just(rsocketRequester);
221 		}
222 		else {
223 			requesterMono = this.rsocketRequesterMono;
224 		}
225 
226 		Assert.notNull(requesterMono,
227 				() -> "The 'RSocketRequester' must be configured via 'ClientRSocketConnector' or provided in the '" +
228 						RSocketRequesterMethodArgumentResolver.RSOCKET_REQUESTER_HEADER + "' request message headers.");
229 
230 		return requesterMono
231 				.map((rSocketRequester) -> createRequestSpec(rSocketRequester, requestMessage))
232 				.map((requestSpec) -> prepareRetrieveSpec(requestSpec, requestMessage))
233 				.flatMap((retrieveSpec) -> performRetrieve(retrieveSpec, requestMessage));
234 	}
235 
236 	@SuppressWarnings("unchecked")
237 	private RSocketRequester.RequestSpec createRequestSpec(RSocketRequester rsocketRequester,
238 			Message<?> requestMessage) {
239 
240 		String route = this.routeExpression.getValue(this.evaluationContext, requestMessage, String.class);
241 		Assert.notNull(route, () -> "The 'routeExpression' [" + this.routeExpression + "] must not evaluate to null");
242 
243 		RSocketRequester.RequestSpec requestSpec = rsocketRequester.route(route, this.routeVars);
244 		if (this.metadataExpression != null) {
245 			Map<Object, MimeType> metadata =
246 					this.metadataExpression.getValue(this.evaluationContext, requestMessage, Map.class);
247 			if (!CollectionUtils.isEmpty(metadata)) {
248 				requestSpec.metadata((spec) -> metadata.forEach(spec::metadata));
249 			}
250 		}
251 
252 		return requestSpec;
253 	}
254 
255 	private RSocketRequester.RetrieveSpec prepareRetrieveSpec(RSocketRequester.RequestSpec requestSpec,
256 			Message<?> requestMessage) {
257 
258 		Object payload = requestMessage.getPayload();
259 		if (payload instanceof Publisher<?> && this.publisherElementTypeExpression != null) {
260 			Object publisherElementType = evaluateExpressionForType(requestMessage, this.publisherElementTypeExpression,
261 					"publisherElementType");
262 			return prepareRequestSpecForPublisher(requestSpec, (Publisher<?>) payload, publisherElementType);
263 		}
264 		else {
265 			return requestSpec.data(payload);
266 		}
267 	}
268 
269 	private RSocketRequester.RetrieveSpec prepareRequestSpecForPublisher(RSocketRequester.RequestSpec requestSpec,
270 			Publisher<?> payload, Object publisherElementType) {
271 
272 		if (publisherElementType instanceof Class<?>) {
273 			return requestSpec.data(payload, (Class<?>) publisherElementType);
274 		}
275 		else {
276 			return requestSpec.data(payload, (ParameterizedTypeReference<?>) publisherElementType);
277 		}
278 	}
279 
280 	private Mono<?> performRetrieve(RSocketRequester.RetrieveSpec retrieveSpec, Message<?> requestMessage) {
281 		RSocketInteractionModel interactionModel = evaluateInteractionModel(requestMessage);
282 		Assert.notNull(interactionModel,
283 				() -> "The 'interactionModelExpression' [" + this.interactionModelExpression +
284 						"] must not evaluate to null");
285 
286 		Object expectedResponseType = null;
287 		if (!RSocketInteractionModel.fireAndForget.equals(interactionModel)) {
288 			expectedResponseType = evaluateExpressionForType(requestMessage, this.expectedResponseTypeExpression,
289 					"expectedResponseType");
290 		}
291 
292 		switch (interactionModel) {
293 			case fireAndForget:
294 				return retrieveSpec.send();
295 			case requestResponse:
296 				if (expectedResponseType instanceof Class<?>) {
297 					return retrieveSpec.retrieveMono((Class<?>) expectedResponseType);
298 				}
299 				else {
300 					return retrieveSpec.retrieveMono((ParameterizedTypeReference<?>) expectedResponseType);
301 				}
302 			case requestStream:
303 			case requestChannel:
304 				if (expectedResponseType instanceof Class<?>) {
305 					return Mono.just(retrieveSpec.retrieveFlux((Class<?>) expectedResponseType));
306 				}
307 				else {
308 					return Mono.just(retrieveSpec.retrieveFlux((ParameterizedTypeReference<?>) expectedResponseType));
309 				}
310 			default:
311 				throw new UnsupportedOperationException("Unsupported interaction model: " + interactionModel);
312 		}
313 	}
314 
315 	private RSocketInteractionModel evaluateInteractionModel(Message<?> requestMessage) {
316 		Object value = this.interactionModelExpression.getValue(this.evaluationContext, requestMessage);
317 		if (value instanceof RSocketInteractionModel) {
318 			return (RSocketInteractionModel) value;
319 		}
320 		else if (value instanceof String) {
321 			return RSocketInteractionModel.valueOf((String) value);
322 		}
323 		else {
324 			throw new IllegalStateException("The 'interactionModelExpression' [" +
325 					this.interactionModelExpression +
326 					"] must evaluate to 'RSocketInteractionModel' or 'String' type, but not into: '" + value + "'");
327 		}
328 	}
329 
330 	private Object evaluateExpressionForType(Message<?> requestMessage, Expression expression, String propertyName) {
331 		Object type = expression.getValue(this.evaluationContext, requestMessage);
332 		Assert.state(type instanceof Class<?>
333 						|| type instanceof String
334 						|| type instanceof ParameterizedTypeReference<?>,
335 				() -> "The '" + propertyName + "' [" + expression +
336 						"] must evaluate to 'String' (class FQN), 'Class<?>' " +
337 						"or 'ParameterizedTypeReference<?>', not to: " + type);
338 
339 		if (type instanceof String) {
340 			try {
341 				return ClassUtils.forName((String) type, getBeanClassLoader());
342 			}
343 			catch (ClassNotFoundException e) {
344 				throw new IllegalStateException(e);
345 			}
346 		}
347 		else {
348 			return type;
349 		}
350 	}
