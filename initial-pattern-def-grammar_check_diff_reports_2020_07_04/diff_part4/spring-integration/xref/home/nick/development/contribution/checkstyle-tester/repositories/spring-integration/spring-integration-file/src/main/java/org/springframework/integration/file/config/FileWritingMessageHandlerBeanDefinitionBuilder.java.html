17  package org.springframework.integration.file.config;
18  
19  import org.w3c.dom.Element;
20  
21  import org.springframework.beans.factory.support.BeanDefinitionBuilder;
22  import org.springframework.beans.factory.xml.ParserContext;
23  import org.springframework.integration.config.ExpressionFactoryBean;
24  import org.springframework.integration.config.xml.IntegrationNamespaceUtils;
25  import org.springframework.integration.file.DefaultFileNameGenerator;
26  import org.springframework.util.StringUtils;
27  
28  /**
29   * A common helper class for the 'outbound-channel-adapter' and 'outbound-gateway'
30   * element parsers. Both of those are responsible for creating an instance of
31   * {@link org.springframework.integration.file.FileWritingMessageHandler}.
32   *
33   * @author Mark Fisher
34   * @author Artem Bilan
35   * @author Gunnar Hillert
36   * @author Tony Falabella
37   * @author Gary Russell
38   *
39   * @since 1.0.3
40   */
41  abstract class FileWritingMessageHandlerBeanDefinitionBuilder {
42  
43  	static BeanDefinitionBuilder configure(Element element, boolean expectReply, ParserContext parserContext) {
44  
45  		BeanDefinitionBuilder builder =
46  				BeanDefinitionBuilder.genericBeanDefinition(FileWritingMessageHandlerFactoryBean.class);
47  
48  		String directory = element.getAttribute("directory");
49  		String directoryExpression = element.getAttribute("directory-expression");
50  
51  		if (!StringUtils.hasText(directory) && !StringUtils.hasText(directoryExpression)) {
52  			parserContext.getReaderContext().error("directory or directory-expression is required", element);
53  		}
54  		else if (StringUtils.hasText(directory) && StringUtils.hasText(directoryExpression)) {
55  			parserContext.getReaderContext()
56  					.error("Either directory or directory-expression must be provided but not both", element);
57  		}
58  
59  		if (StringUtils.hasText(directoryExpression)) {
60  			BeanDefinitionBuilder expressionBuilder =
61  					BeanDefinitionBuilder.genericBeanDefinition(ExpressionFactoryBean.class);
62  			expressionBuilder.addConstructorArgValue(directoryExpression);
63  			builder.addPropertyValue("directoryExpression", expressionBuilder.getBeanDefinition());
64  		}
65  
66  		builder.addPropertyValue("expectReply", expectReply);
67  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "append-new-line");
68  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "directory");
69  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "auto-create-directory");
70  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "delete-source-files");
71  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "temporary-file-suffix");
72  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "mode", "fileExistsMode");
73  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "charset");
74  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "buffer-size");
75  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "flush-interval");
76  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "flush-when-idle");
77  		IntegrationNamespaceUtils.setReferenceIfAttributeDefined(builder, element, "flush-predicate");
78  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "chmod");
79  		IntegrationNamespaceUtils.setValueIfAttributeDefined(builder, element, "preserve-timestamp");
80  		filenameGenerators(element, parserContext, builder);
81  		return builder;
82  	}
83  
84  	private static void filenameGenerators(Element element, ParserContext parserContext,
85  			BeanDefinitionBuilder builder) {
86  
87  		String remoteFileNameGenerator = element.getAttribute("filename-generator");
88  		String remoteFileNameGeneratorExpression = element.getAttribute("filename-generator-expression");
89  		boolean hasRemoteFileNameGenerator = StringUtils.hasText(remoteFileNameGenerator);
90  		boolean hasRemoteFileNameGeneratorExpression = StringUtils.hasText(remoteFileNameGeneratorExpression);
91  		if (hasRemoteFileNameGenerator || hasRemoteFileNameGeneratorExpression) {
92  			if (hasRemoteFileNameGenerator && hasRemoteFileNameGeneratorExpression) {
93  				parserContext.getReaderContext()
94  						.error("at most one of 'filename-generator-expression' or 'filename-generator' " +
95  						"is allowed on file outbound adapter/gateway", element);
96  			}
97  			if (hasRemoteFileNameGenerator) {
98  				builder.addPropertyReference("fileNameGenerator", remoteFileNameGenerator);
99  			}
100 			else {
101 				BeanDefinitionBuilder fileNameGeneratorBuilder = BeanDefinitionBuilder
102 						.genericBeanDefinition(DefaultFileNameGenerator.class);
103 				fileNameGeneratorBuilder.addPropertyValue("expression", remoteFileNameGeneratorExpression);
104 				builder.addPropertyValue("fileNameGenerator", fileNameGeneratorBuilder.getBeanDefinition());
105 			}
106 		}
107 	}
108 
109 }
