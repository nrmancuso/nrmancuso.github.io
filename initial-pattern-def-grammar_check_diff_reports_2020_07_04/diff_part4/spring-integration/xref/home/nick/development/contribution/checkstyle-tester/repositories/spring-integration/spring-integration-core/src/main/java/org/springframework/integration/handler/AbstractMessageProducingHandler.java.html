127 	/**
128 	 * Set header patterns ("xxx*", "*xxx", "*xxx*" or "xxx*yyy")
129 	 * that will NOT be copied from the inbound message if
130 	 * {@link #shouldCopyRequestHeaders() shouldCopyRequestHeaaders} is true.
131 	 * At least one pattern as "*" means do not copy headers at all.
132 	 * @param headers the headers to not propagate from the inbound message.
133 	 * @since 4.3.10
134 	 * @see org.springframework.util.PatternMatchUtils
135 	 */
136 	@Override
137 	public void setNotPropagatedHeaders(String... headers) {
138 		updateNotPropagatedHeaders(headers, false);
139 	}
140 
141 	/**
142 	 * Set or replace not propagated headers. Exposed so that subclasses can set specific
143 	 * headers in a constructor, since {@link #setNotPropagatedHeaders(String...)} is not
144 	 * final.
145 	 * @param headers Header patterns to not propagate.
146 	 * @param merge true to merge with existing patterns; false to replace.
147 	 * @since 5.0.2
148 	 */
149 	protected final void updateNotPropagatedHeaders(String[] headers, boolean merge) {
150 		Set<String> headerPatterns = new HashSet<>();
151 
152 		if (merge && this.notPropagatedHeaders != null) {
153 			headerPatterns.addAll(Arrays.asList(this.notPropagatedHeaders));
154 		}
155 
156 		if (!ObjectUtils.isEmpty(headers)) {
157 			Assert.noNullElements(headers, "null elements are not allowed in 'headers'");
158 
159 			headerPatterns.addAll(Arrays.asList(headers));
160 
161 			this.notPropagatedHeaders = headerPatterns.toArray(new String[0]);
162 		}
163 
164 		if (headerPatterns.contains("*")) {
165 			this.notPropagatedHeaders = new String[] { "*" };
166 			this.noHeadersPropagation = true;
167 		}
168 
169 		this.selectiveHeaderPropagation = !ObjectUtils.isEmpty(this.notPropagatedHeaders);
170 	}

172 	/**
173 	 * Get the header patterns this handler doesn't propagate.
174 	 * @return an immutable {@link java.util.Collection} of headers that will not be
175 	 * copied from the inbound message if {@link #shouldCopyRequestHeaders()} is true.
176 	 * @since 4.3.10
177 	 * @see #setNotPropagatedHeaders(String...)
178 	 * @see org.springframework.util.PatternMatchUtils
179 	 */
180 	@Override
181 	public Collection<String> getNotPropagatedHeaders() {
182 		return this.notPropagatedHeaders != null
183 				? Collections.unmodifiableSet(new HashSet<>(Arrays.asList(this.notPropagatedHeaders)))
184 				: Collections.emptyList();
185 	}

188 	 * Add header patterns ("xxx*", "*xxx", "*xxx*" or "xxx*yyy")
189 	 * that will NOT be copied from the inbound message if
190 	 * {@link #shouldCopyRequestHeaders()} is true, instead of overwriting the existing
191 	 * set.
192 	 * @param headers the headers to not propagate from the inbound message.
193 	 * @since 4.3.10
194 	 * @see #setNotPropagatedHeaders(String...)
195 	 */
196 	@Override
197 	public void addNotPropagatedHeaders(String... headers) {
198 		updateNotPropagatedHeaders(headers, true);

202 	protected void onInit() {
203 		super.onInit();
204 		Assert.state(!(this.outputChannelName != null && this.outputChannel != null), //NOSONAR (inconsistent sync)
205 				"'outputChannelName' and 'outputChannel' are mutually exclusive.");
206 		BeanFactory beanFactory = getBeanFactory();
207 		if (beanFactory != null) {
208 			this.messagingTemplate.setBeanFactory(beanFactory);
209 		}
210 		this.messagingTemplate.setDestinationResolver(getChannelResolver());
211 	}

213 	@Override
214 	@Nullable
215 	public MessageChannel getOutputChannel() {
216 		String channelName = this.outputChannelName;
217 		if (channelName != null) {
218 			this.outputChannel = getChannelResolver().resolveDestination(channelName);
219 			this.outputChannelName = null;
220 		}
221 		return this.outputChannel;
222 	}

224 	protected void sendOutputs(Object result, Message<?> requestMessage) {
225 		if (result instanceof Iterable<?> && shouldSplitOutput((Iterable<?>) result)) {
226 			for (Object o : (Iterable<?>) result) {
227 				produceOutput(o, requestMessage);
228 			}
229 		}
230 		else if (result != null) {
231 			produceOutput(result, requestMessage);
232 		}
233 	}
234 
235 	protected boolean shouldSplitOutput(Iterable<?> reply) {
236 		for (Object next : reply) {
237 			if (next instanceof Message<?> || next instanceof AbstractIntegrationMessageBuilder<?>) {
238 				return true;
239 			}
240 		}
241 		return false;
242 	}
243 
244 	protected void produceOutput(Object replyArg, final Message<?> requestMessage) {
245 		MessageHeaders requestHeaders = requestMessage.getHeaders();
246 		Object reply = replyArg;
247 		Object replyChannel = null;
248 		if (getOutputChannel() == null) {
249 			Map<?, ?> routingSlipHeader = obtainRoutingSlipHeader(requestHeaders, reply);
250 			if (routingSlipHeader != null) {
251 				Assert.isTrue(routingSlipHeader.size() == 1, "The RoutingSlip header value must be a SingletonMap");
252 				Object key = routingSlipHeader.keySet().iterator().next();
253 				Object value = routingSlipHeader.values().iterator().next();
254 				Assert.isInstanceOf(List.class, key, "The RoutingSlip key must be List");
255 				Assert.isInstanceOf(Integer.class, value, "The RoutingSlip value must be Integer");
256 				List<?> routingSlip = (List<?>) key;
257 				AtomicInteger routingSlipIndex = new AtomicInteger((Integer) value);
258 				replyChannel = getOutputChannelFromRoutingSlip(reply, requestMessage, routingSlip, routingSlipIndex);
259 				if (replyChannel != null) {
260 					reply = addRoutingSlipHeader(reply, routingSlip, routingSlipIndex);
261 				}
262 			}
263 			if (replyChannel == null) {
264 				replyChannel = obtainReplyChannel(requestHeaders, reply);
265 			}
266 		}
267 		doProduceOutput(requestMessage, requestHeaders, reply, replyChannel);
268 	}
269 
270 	@Nullable
271 	private Map<?, ?> obtainRoutingSlipHeader(MessageHeaders requestHeaders, Object reply) {
272 		Map<?, ?> routingSlipHeader = requestHeaders.get(IntegrationMessageHeaderAccessor.ROUTING_SLIP, Map.class);
273 		if (routingSlipHeader == null) {
274 			if (reply instanceof Message) {
275 				routingSlipHeader = ((Message<?>) reply).getHeaders()
276 						.get(IntegrationMessageHeaderAccessor.ROUTING_SLIP, Map.class);
277 			}
278 			else if (reply instanceof AbstractIntegrationMessageBuilder<?>) {
279 				routingSlipHeader = ((AbstractIntegrationMessageBuilder<?>) reply)
280 						.getHeader(IntegrationMessageHeaderAccessor.ROUTING_SLIP, Map.class);
281 			}
282 		}
283 		return routingSlipHeader;
284 	}
285 
286 	@Nullable
287 	private Object obtainReplyChannel(MessageHeaders requestHeaders, Object reply) {
288 		Object replyChannel = requestHeaders.getReplyChannel();
289 		if (replyChannel == null) {
290 			if (reply instanceof Message) {
291 				replyChannel = ((Message<?>) reply).getHeaders().getReplyChannel();
292 			}
293 			else if (reply instanceof AbstractIntegrationMessageBuilder<?>) {
294 				replyChannel = ((AbstractIntegrationMessageBuilder<?>) reply)
295 						.getHeader(MessageHeaders.REPLY_CHANNEL, Object.class);
296 			}
297 		}
298 		return replyChannel;
299 	}
300 
301 	private void doProduceOutput(Message<?> requestMessage, MessageHeaders requestHeaders, Object reply,
302 			@Nullable Object replyChannelArg) {
303 
304 		Object replyChannel = replyChannelArg;
305 		if (replyChannel == null) {
306 			replyChannel = getOutputChannel();
307 		}
308 
309 		if (this.async && (reply instanceof ListenableFuture<?> || reply instanceof Publisher<?>)) {
310 			if (reply instanceof Publisher<?> &&
311 					replyChannel instanceof ReactiveStreamsSubscribableChannel) {
312 
313 				((ReactiveStreamsSubscribableChannel) replyChannel)
314 						.subscribeTo(
315 								Flux.from((Publisher<?>) reply)
316 										.doOnError((ex) -> sendErrorMessage(requestMessage, ex))
317 										.map(result -> createOutputMessage(result, requestHeaders)));
318 			}
319 			else {
320 				asyncNonReactiveReply(requestMessage, reply, replyChannel);
321 			}
322 		}
323 		else {
324 			sendOutput(createOutputMessage(reply, requestHeaders), replyChannel, false);
325 		}
326 	}

328 	private AbstractIntegrationMessageBuilder<?> addRoutingSlipHeader(Object reply, List<?> routingSlip,
329 			AtomicInteger routingSlipIndex) {
330 
331 		return messageBuilderForReply(reply)
332 				.setHeader(IntegrationMessageHeaderAccessor.ROUTING_SLIP,
333 						Collections.singletonMap(routingSlip, routingSlipIndex.get()));
334 	}
335 
336 	protected AbstractIntegrationMessageBuilder<?> messageBuilderForReply(Object reply) {
337 		AbstractIntegrationMessageBuilder<?> builder;
338 		if (reply instanceof Message) {
339 			builder = getMessageBuilderFactory().fromMessage((Message<?>) reply);
340 		}
341 		else if (reply instanceof AbstractIntegrationMessageBuilder) {
342 			builder = (AbstractIntegrationMessageBuilder<?>) reply;
343 		}
344 		else {
345 			builder = getMessageBuilderFactory().withPayload(reply);
346 		}
347 		return builder;
348 	}

350 	private void asyncNonReactiveReply(Message<?> requestMessage, Object reply, @Nullable Object replyChannel) {
351 		ListenableFuture<?> future;
352 		if (reply instanceof ListenableFuture<?>) {
353 			future = (ListenableFuture<?>) reply;
354 		}
355 		else {
356 			SettableListenableFuture<Object> settableListenableFuture = new SettableListenableFuture<>();
357 			Mono<?> reactiveReply;
358 			ReactiveAdapter adapter = ReactiveAdapterRegistry.getSharedInstance().getAdapter(null, reply);
359 			if (adapter != null && adapter.isMultiValue()) {
360 				reactiveReply = Mono.just(reply);
361 			}
362 			else {
363 				reactiveReply = Mono.from((Publisher<?>) reply);
364 			}
365 			reactiveReply.subscribe(settableListenableFuture::set, settableListenableFuture::setException);
366 			future = settableListenableFuture;
367 		}
368 		future.addCallback(new ReplyFutureCallback(requestMessage, replyChannel));
369 	}

371 	private Object getOutputChannelFromRoutingSlip(Object reply, Message<?> requestMessage, List<?> routingSlip,
372 			AtomicInteger routingSlipIndex) {
373 
374 		if (routingSlipIndex.get() >= routingSlip.size()) {
375 			return null;
376 		}
377 
378 		Object path = routingSlip.get(routingSlipIndex.get());
379 		Object routingSlipPathValue = null;
380 
381 		if (path instanceof String) {
382 			routingSlipPathValue = getBeanFactory().getBean((String) path);
383 		}
384 		else if (path instanceof RoutingSlipRouteStrategy) {
385 			routingSlipPathValue = path;
386 		}
387 		else {
388 			throw new IllegalArgumentException("The RoutingSlip 'path' can be of " +
389 					"String or RoutingSlipRouteStrategy type, but got: " + path.getClass());
390 		}
391 
392 		if (routingSlipPathValue instanceof MessageChannel) {
393 			routingSlipIndex.incrementAndGet();
394 			return routingSlipPathValue;
395 		}
396 		else {
397 			Object nextPath = ((RoutingSlipRouteStrategy) routingSlipPathValue).getNextPath(requestMessage, reply);
398 			if (nextPath != null && (!(nextPath instanceof String) || StringUtils.hasText((String) nextPath))) {
399 				return nextPath;
400 			}
401 			else {
402 				routingSlipIndex.incrementAndGet();
403 				return getOutputChannelFromRoutingSlip(reply, requestMessage, routingSlip, routingSlipIndex);
404 			}
405 		}
406 	}

408 	protected Message<?> createOutputMessage(Object output, MessageHeaders requestHeaders) {
409 		AbstractIntegrationMessageBuilder<?> builder;
410 		if (output instanceof Message<?>) {
411 			if (this.noHeadersPropagation || !shouldCopyRequestHeaders()) {
412 				return (Message<?>) output;
413 			}
414 			builder = this.getMessageBuilderFactory().fromMessage((Message<?>) output);
415 		}
416 		else if (output instanceof AbstractIntegrationMessageBuilder) {
417 			builder = (AbstractIntegrationMessageBuilder<?>) output;
418 		}
419 		else {
420 			builder = this.getMessageBuilderFactory().withPayload(output);
421 		}
422 		if (!this.noHeadersPropagation && shouldCopyRequestHeaders()) {
423 			builder.filterAndCopyHeadersIfAbsent(requestHeaders,
424 					this.selectiveHeaderPropagation ? this.notPropagatedHeaders : null);
425 		}
426 		return builder.build();
427 	}

429 	/**
430 	 * Send an output Message. The 'replyChannel' will be considered only if this handler's
431 	 * 'outputChannel' is <code>null</code>. In that case, the 'replyChannel' value must not also be
432 	 * <code>null</code>, and it must be an instance of either String or {@link MessageChannel}.
433 	 * @param output the output object to send
434 	 * @param replyChannelArg the 'replyChannel' value from the original request
435 	 * @param useArgChannel - use the replyChannel argument (must not be null), not
436 	 * the configured output channel.
437 	 */
438 	protected void sendOutput(Object output, @Nullable Object replyChannelArg, boolean useArgChannel) {
439 		Object replyChannel = replyChannelArg;
440 		MessageChannel outChannel = getOutputChannel();
441 		if (!useArgChannel && outChannel != null) {
442 			replyChannel = outChannel;
443 		}
444 		if (replyChannel == null) {
445 			throw new DestinationResolutionException("no output-channel or replyChannel header available");
446 		}
447 
448 		if (replyChannel instanceof MessageChannel) {
449 			if (output instanceof Message<?>) {
450 				this.messagingTemplate.send((MessageChannel) replyChannel, (Message<?>) output);
451 			}
452 			else {
453 				this.messagingTemplate.convertAndSend((MessageChannel) replyChannel, output);
454 			}
455 		}
456 		else if (replyChannel instanceof String) {
457 			if (output instanceof Message<?>) {
458 				this.messagingTemplate.send((String) replyChannel, (Message<?>) output);
459 			}
460 			else {
461 				this.messagingTemplate.convertAndSend((String) replyChannel, output);
462 			}
463 		}
464 		else {
465 			throw new MessagingException("replyChannel must be a MessageChannel or String");
466 		}
467 	}

470 	 * Subclasses may override this. True by default.
471 	 * @return true if the request headers should be copied.
472 	 */
473 	protected boolean shouldCopyRequestHeaders() {
474 		return true;
475 	}

477 	protected void sendErrorMessage(Message<?> requestMessage, Throwable ex) {
478 		Object errorChannel = resolveErrorChannel(requestMessage.getHeaders());
479 		Throwable result = ex;
480 		if (!(ex instanceof MessagingException)) {
481 			result = new MessageHandlingException(requestMessage, ex);
482 		}
483 		if (errorChannel == null) {
484 			logger.error("Async exception received and no 'errorChannel' header exists and no default "
485 					+ "'errorChannel' found", result);
486 		}
487 		else {
488 			try {
489 				sendOutput(new ErrorMessage(result), errorChannel, true);
490 			}
491 			catch (Exception e) {
492 				Exception exceptionToLog =
493 						IntegrationUtils.wrapInHandlingExceptionIfNecessary(requestMessage,
494 								() -> "failed to send error message in the [" + this + ']', e);
495 				logger.error("Failed to send async reply", exceptionToLog);
496 			}
497 		}
498 	}
499 
500 	protected Object resolveErrorChannel(final MessageHeaders requestHeaders) {
501 		Object errorChannel = requestHeaders.getErrorChannel();
502 		if (errorChannel == null) {
503 			try {
504 				errorChannel = getChannelResolver().resolveDestination(IntegrationContextUtils.ERROR_CHANNEL_BEAN_NAME);
505 			}
506 			catch (DestinationResolutionException e) {
507 				// ignore
508 			}
509 		}
510 		return errorChannel;
511 	}

513 	private final class ReplyFutureCallback implements ListenableFutureCallback<Object> {
514 
515 		private final Message<?> requestMessage;
516 
517 		@Nullable
518 		private final Object replyChannel;
519 
520 		ReplyFutureCallback(Message<?> requestMessage, @Nullable Object replyChannel) {
521 			this.requestMessage = requestMessage;
522 			this.replyChannel = replyChannel;
523 		}
524 
525 
526 		@Override
527 		public void onSuccess(Object result) {
528 			Message<?> replyMessage = null;
529 			try {
530 				replyMessage = createOutputMessage(result, this.requestMessage.getHeaders());
531 				sendOutput(replyMessage, this.replyChannel, false);
532 			}
533 			catch (Exception ex) {
534 				Exception exceptionToLogAndSend = ex;
535 				if (!(ex instanceof MessagingException)) { // NOSONAR
536 					exceptionToLogAndSend = new MessageHandlingException(this.requestMessage, ex);
537 					if (replyMessage != null) {
538 						exceptionToLogAndSend = new MessagingException(replyMessage, exceptionToLogAndSend);
539 					}
540 				}
541 				logger.error("Failed to send async reply: " + result.toString(), exceptionToLogAndSend);
542 				onFailure(exceptionToLogAndSend);
543 			}
544 		}
545 
546 		@Override
547 		public void onFailure(Throwable ex) {
548 			sendErrorMessage(this.requestMessage, ex);
549 		}
550 
551 	}
