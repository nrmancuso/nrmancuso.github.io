53  	@Test
54  	public void testRollback() throws Exception {
55  		final AtomicBoolean failWhenCopyingBar = new AtomicBoolean(true);
56  		final AtomicInteger count = new AtomicInteger();
57  		SessionFactory<String> sf = new StringSessionFactory();
58  		AbstractInboundFileSynchronizer<String> sync = new AbstractInboundFileSynchronizer<String>(sf) {
59  
60  			@Override
61  			protected boolean isFile(String file) {
62  				return true;
63  			}
64  
65  			@Override
66  			protected String getFilename(String file) {
67  				return file;
68  			}
69  
70  			@Override
71  			protected long getModified(String file) {
72  				return 0;
73  			}
74  
75  			@Override
76  			protected String protocol() {
77  				return "file";
78  			}
79  
80  			@Override
81  			protected boolean copyFileToLocalDirectory(String remoteDirectoryPath, String remoteFile,
82  					File localDirectory, Session<String> session) throws IOException {
83  				if ("bar".equals(remoteFile) && failWhenCopyingBar.getAndSet(false)) {
84  					throw new IOException("fail");
85  				}
86  				count.incrementAndGet();
87  				return true;
88  			}
89  
90  		};
91  		sync.setFilter(new AcceptOnceFileListFilter<>());
92  		sync.setRemoteDirectory("foo");
93  
94  		assertThatExceptionOfType(MessagingException.class)
95  				.isThrownBy(() -> sync.synchronizeToLocalDirectory(mock(File.class)))
96  				.withRootCauseInstanceOf(IOException.class)
97  				.withMessageContaining("fail");
98  
99  		sync.synchronizeToLocalDirectory(mock(File.class));
100 		assertThat(count.get()).isEqualTo(3);
101 		sync.close();
102 	}

104 	@Test
105 	public void testMaxFetchSizeSynchronizer() throws Exception {
106 		final AtomicInteger count = new AtomicInteger();
107 		AbstractInboundFileSynchronizer<String> sync = createLimitingSynchronizer(count);
108 
109 		sync.synchronizeToLocalDirectory(mock(File.class), 1);
110 		assertThat(count.get()).isEqualTo(1);
111 		sync.synchronizeToLocalDirectory(mock(File.class), 1);
112 		assertThat(count.get()).isEqualTo(2);
113 		sync.synchronizeToLocalDirectory(mock(File.class), 1);
114 		assertThat(count.get()).isEqualTo(3);
115 		sync.close();
116 	}

170 	public void testBulkOnlyFilter() {
171 		final AtomicInteger count = new AtomicInteger();
172 		AbstractInboundFileSynchronizer<String> sync = createLimitingSynchronizer(count);
173 		ChainFileListFilter<String> cflf = new ChainFileListFilter<>();
174 		cflf.addFilter(new AcceptOnceFileListFilter<>());
175 		cflf.addFilter(fs -> Stream.of(fs).collect(Collectors.toList()));
176 		sync.setFilter(cflf);
177 		AbstractInboundFileSynchronizingMessageSource<String> source = createSource(sync);
178 		source.afterPropertiesSet();
179 		source.start();
180 		source.receive();
181 		assertThat(count.get()).isEqualTo(1);
182 		source.receive();
183 		assertThat(count.get()).isEqualTo(2);
184 		source.receive();
185 		assertThat(count.get()).isEqualTo(3);
186 		source.receive();
187 		assertThat(count.get()).isEqualTo(3);

221 	private AbstractInboundFileSynchronizingMessageSource<String> createSource(AtomicInteger count) {
222 		return createSource(createLimitingSynchronizer(count));
223 	}

225 	private AbstractInboundFileSynchronizingMessageSource<String> createSource(
226 			AbstractInboundFileSynchronizer<String> sync) {
227 
228 		AbstractInboundFileSynchronizingMessageSource<String> source =
229 				new AbstractInboundFileSynchronizingMessageSource<String>(sync) {
230 
231 					@Override
232 					public String getComponentType() {
233 						return "foo";
234 					}
235 
236 				};
237 		source.setMaxFetchSize(1);
238 		source.setLocalDirectory(new File(System.getProperty("java.io.tmpdir") + File.separator + UUID.randomUUID()));
239 		source.setAutoCreateLocalDirectory(true);
240 		source.setBeanFactory(mock(BeanFactory.class));
241 		source.setBeanName("fooSource");
242 		return source;
243 	}

245 	private AbstractInboundFileSynchronizer<String> createLimitingSynchronizer(final AtomicInteger count) {
246 		SessionFactory<String> sf = new StringSessionFactory();
247 		AbstractInboundFileSynchronizer<String> sync = new AbstractInboundFileSynchronizer<String>(sf) {
248 
249 			@Override
250 			protected boolean isFile(String file) {
251 				return true;
252 			}
253 
254 			@Override
255 			protected String getFilename(String file) {
256 				return file;
257 			}
258 
259 			@Override
260 			protected long getModified(String file) {
261 				return 0;
262 			}
263 
264 			@Override
265 			protected String protocol() {
266 				return "file";
267 			}
268 
269 			@Override
270 			protected boolean copyFileToLocalDirectory(String remoteDirectoryPath, String remoteFile,
271 					File localDirectory, Session<String> session) {
272 				count.incrementAndGet();
273 				return true;
274 			}
275 
276 		};
277 		sync.setFilter(new AcceptOnceFileListFilter<>());
278 		sync.setRemoteDirectory("foo");
279 		sync.setBeanFactory(mock(BeanFactory.class));
280 		return sync;
281 	}
