2   package org.hibernate.test.annotations.duplicatedgenerator;
3   import org.junit.Assert;
4   import org.junit.Test;
5   
6   import org.hibernate.AnnotationException;
7   import org.hibernate.cfg.AnnotationConfiguration;
8   import org.hibernate.cfg.Environment;
9   import org.hibernate.service.ServiceRegistry;
10  import org.hibernate.testing.ServiceRegistryBuilder;
11  
12  /**
13   * @author Emmanuel Bernard
14   */
15  public class DuplicateTest  {
16      @Test
17  	public void testDuplicateEntityName() throws Exception {
18  		AnnotationConfiguration cfg = new AnnotationConfiguration();
19  		cfg.setProperty( Environment.HBM2DDL_AUTO, "create-drop" );
20  		ServiceRegistry serviceRegistry = null;
21  		try {
22  			cfg.addAnnotatedClass( Flight.class );
23  			cfg.addAnnotatedClass( org.hibernate.test.annotations.Flight.class );
24  			cfg.addResource( "org/hibernate/test/annotations/orm.xml" );
25  			cfg.addResource( "org/hibernate/test/annotations/duplicatedgenerator/orm.xml" );
26  			serviceRegistry = ServiceRegistryBuilder.buildServiceRegistry( cfg.getProperties() );
27  			cfg.buildSessionFactory( serviceRegistry );
28              Assert.fail( "Should not be able to map the same entity name twice" );
29  		}
30  		catch (AnnotationException ae) {
31  			//success
32  		}
33  		finally {
34  			if ( serviceRegistry != null ) {
35  				ServiceRegistryBuilder.destroy( serviceRegistry );
36  			}
37  		}
38  	}
39  }
