102 	public static class SessionProxyHandler implements InvocationHandler {
103 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
104 			if ( "getEntityMode".equals( method.getName() ) ) {
105 				return EntityMode.POJO;
106 			}
107 			throw new UnsupportedOperationException( "Unexpected method call : " + method.getName() );
108 		}

377 	protected <T> void runBasicTests(AbstractSingleColumnStandardBasicType<T> type, T original, T copy, T different) {
378 		final boolean nonCopyable = Class.class.isInstance( original ) || Currency.class.isInstance( original );
379 		if ( ! nonCopyable ) {
380 			// these checks exclude classes which cannot really be cloned (singetons/enums)
381 			assertFalse( original == copy );
382 		}
383 
384 		assertTrue( original == type.replace( original, copy, null, null, null ) );
385 
386 		assertTrue( type.isSame( original, copy ) );
387 		assertTrue( type.isEqual( original, copy ) );
388 		assertTrue( type.isEqual( original, copy ) );
389 		assertTrue( type.isEqual( original, copy, null ) );
390 
391 		assertFalse( type.isSame( original, different ) );
392 		assertFalse( type.isEqual( original, different ) );
393 		assertFalse( type.isEqual( original, different ) );
394 		assertFalse( type.isEqual( original, different, null ) );
395 
396 		assertFalse( type.isDirty( original, copy , session ) );
397 		assertFalse( type.isDirty( original, copy , ArrayHelper.FALSE, session ) );
398 		assertFalse( type.isDirty( original, copy , ArrayHelper.TRUE, session ) );
399 
400 		assertTrue( type.isDirty( original, different , session ) );
401 		assertFalse( type.isDirty( original, different , ArrayHelper.FALSE, session ) );
402 		assertTrue( type.isDirty( original, different , ArrayHelper.TRUE, session ) );
403 
404 		assertFalse( type.isModified( original, copy, ArrayHelper.FALSE, session ) );
405 		assertFalse( type.isModified( original, copy, ArrayHelper.TRUE, session ) );
406 
407 		assertTrue( type.isModified( original, different, ArrayHelper.FALSE, session ) );
408 		assertTrue( type.isModified( original, different, ArrayHelper.TRUE, session ) );
409 	}
