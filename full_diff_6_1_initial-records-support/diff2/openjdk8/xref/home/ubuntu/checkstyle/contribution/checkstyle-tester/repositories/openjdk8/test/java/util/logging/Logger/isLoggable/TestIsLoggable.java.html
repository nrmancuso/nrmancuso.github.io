44      // This logger can be configured to override its default level
45      // for a particular set of thread ids
46      public static final class ThreadLogger extends Logger {
47  
48         final Map<Long, Level> threadMap =
49                  Collections.synchronizedMap(new HashMap<Long, Level>());
50  
51          public ThreadLogger(String name) {
52              super(name, null);
53          }
54  
55          @Override
56          public boolean isLoggable(Level level) {
57              final Level threadLevel = threadMap.get(Thread.currentThread().getId());
58              if (threadLevel == null) return super.isLoggable(level);
59              final int levelValue = threadLevel.intValue();
60              final int offValue = Level.OFF.intValue();
61              if (level.intValue() < levelValue || levelValue == offValue) {
62                  return false;
63              }
64              return true;
65          }
66  
67      }
68  
69      public static final class TestHandler extends Handler {
70  
71          final List<String> messages = new CopyOnWriteArrayList<>();
72  
73          @Override
74          public void publish(LogRecord record) {
75              messages.add(record.getMessage());
76          }
77  
78          @Override
79          public void flush() {
80          }
81  
82          @Override
83          public void close() throws SecurityException {
84              messages.clear();
85          }
86  
87      }

89      // Sorted list of standard levels
90      static final List<Level> LEVELS = Collections.unmodifiableList(
91              java.util.Arrays.asList(new Level[] {
92                  Level.SEVERE, Level.WARNING, Level.INFO, Level.CONFIG,
93                  Level.FINE, Level.FINER, Level.FINEST
94              }));

96      // Test cases:
97      //   LEV_   test logger.severe(msg) .. logger.finest(msg)
98      //   LOG_   logger.log(Level.SEVERE, msg) ... logger.log(Level.FINEST, msg)
99      //   LOG1_  logger.log(Level.SEVERE, msg, param1) ...
100     //   LOG2_  logger.log(Level.SEVERE, msg, params[]) ...
101     //   LOG3_  logger.log(Level.SEVERE, msg, throwable) ...
102     //   LOGP_  logger.logp(Level.SEVERE, class, method, msg) ...
103     //   LOGP1_ logger.logp(Level.SEVERE, class, method, msg, param1) ...
104     //   LOGP2_ logger.logp(Level.SEVERE, class, method, msg, params[]) ...
105     //   LOGP3_ logger.logp(Level.SEVERE, class, method, msg, throwable) ...
106     public static enum LogTest {
107         LEV_SEVERE, LEV_WARNING, LEV_INFO, LEV_CONFIG, LEV_FINE, LEV_FINER, LEV_FINEST,
108         LOG_SEVERE, LOG_WARNING, LOG_INFO, LOG_CONFIG, LOG_FINE, LOG_FINER, LOG_FINEST,
109         LOG1_SEVERE, LOG1_WARNING, LOG1_INFO, LOG1_CONFIG, LOG1_FINE, LOG1_FINER, LOG1_FINEST,
110         LOG2_SEVERE, LOG2_WARNING, LOG2_INFO, LOG2_CONFIG, LOG2_FINE, LOG2_FINER, LOG2_FINEST,
111         LOG3_SEVERE, LOG3_WARNING, LOG3_INFO, LOG3_CONFIG, LOG3_FINE, LOG3_FINER, LOG3_FINEST,
112         LOGP_SEVERE, LOGP_WARNING, LOGP_INFO, LOGP_CONFIG, LOGP_FINE, LOGP_FINER, LOGP_FINEST,
113         LOGP1_SEVERE, LOGP1_WARNING, LOGP1_INFO, LOGP1_CONFIG, LOGP1_FINE, LOGP1_FINER, LOGP1_FINEST,
114         LOGP2_SEVERE, LOGP2_WARNING, LOGP2_INFO, LOGP2_CONFIG, LOGP2_FINE, LOGP2_FINER, LOGP2_FINEST,
115         LOGP3_SEVERE, LOGP3_WARNING, LOGP3_INFO, LOGP3_CONFIG, LOGP3_FINE, LOGP3_FINER, LOGP3_FINEST;
116 
117         // call the method Logger.severe() ... Logger.finest() corresponding
118         // to the given level 'l' (severe() for SEVERE etc...)
119         public void loglevel(Level l, Logger logger, String message) {
120             LogTest test = LogTest.valueOf("LEV_"+l.getName());
121             switch(test) {
122                 case LEV_SEVERE:
123                     logger.severe(message);
124                     break;
125                 case LEV_WARNING:
126                     logger.warning(message);
127                     break;
128                 case LEV_INFO:
129                     logger.info(message);
130                     break;
131                 case LEV_CONFIG:
132                     logger.config(message);
133                     break;
134                 case LEV_FINE:
135                     logger.fine(message);
136                     break;
137                 case LEV_FINER:
138                     logger.finer(message);
139                     break;
140                 case LEV_FINEST:
141                     logger.finest(message);
142                     break;
143             }
144         }

146         // The threshold at which the logger is expected to start logging.
147         // trick: we derive the threshold level from the testcase name...
148         public Level threshold() {
149             for (Level l : LEVELS ) {
150                 if (this.toString().endsWith(l.getName())) {
151                     return l;
152                 }
153             }
154             return Level.OFF;
155         }
156 
157         // Levels for which the logger is expected to log something.
158         public List<Level> loggable() {
159             return LEVELS.subList(0, LEVELS.indexOf(threshold())+1);
160         }
161 
162         // Levels which will be blocked because they are weaker than the
163         // threshold()
164         public List<Level> weaker() {
165             return LEVELS.subList(LEVELS.indexOf(threshold())+1, LEVELS.size());
166         }
167 
168         // Log a message at this testcase threshold, using this testcase method.
169         public void log(Logger logger, String message) {
170             log(threshold(), logger, message);
171         }
172 
173         // Log a message at the given level, using this testcase method.
174         public void log(Level level, Logger logger, String message) {
175             if (this.toString().startsWith("LOG_")) {
176                 logger.log(level, message);
177             } else if (this.toString().startsWith("LOG1_")) {
178                 logger.log(level, message, "dummy param");
179             } else if (this.toString().startsWith("LOG2_")) {
180                 logger.log(level, message, new Object[] {"dummy", "param"});
181             } else if (this.toString().startsWith("LOG3_")) {
182                 logger.log(level, message, new Exception("dummy exception"));
183             } else if (this.toString().startsWith("LOGP_")) {
184                 logger.logp(level, "TestCase", "log", message);
185             } else if (this.toString().startsWith("LOGP1_")) {
186                 logger.logp(level, "TestCase", "log", message, "dummy param");
187             } else if (this.toString().startsWith("LOGP2_")) {
188                 logger.logp(level, "TestCase", "log", message,
189                         new Object[] {"dummy", "param"});
190             } else if (this.toString().startsWith("LOGP3_")) {
191                 logger.logp(level, "TestCase", "log", message,
192                         new Exception("dummy exception"));
193             } else if (this.toString().startsWith("LEV_")) {
194                 loglevel(level, logger, message);
195             }
196         }

198         // String description of the logging method called.
199         public String method() {
200             if (this.toString().startsWith("LOG_")) {
201                 return "Logger.log(Level." + threshold().getName() +", msg): ";
202             } else if (this.toString().startsWith("LOG1_")) {
203                 return "Logger.log(Level." + threshold().getName() +", msg, param1): ";
204             } else if (this.toString().startsWith("LOG2_")) {
205                 return "Logger.log(Level." + threshold().getName() +", msg, params[]): ";
206             } else if (this.toString().startsWith("LOG3_")) {
207                 return "Logger.log(Level." + threshold().getName() +", msg, throwable): ";
208             } else if (this.toString().startsWith("LEV_")) {
209                 return "Logger."+threshold().getName().toLowerCase(Locale.ROOT)+"(): ";
210             } else if (this.toString().startsWith("LOGP_")) {
211                 return "Logger.logp(Level." + threshold().getName() +", msg): ";
212             } else if (this.toString().startsWith("LOGP1_")) {
213                 return "Logger.logp(Level." + threshold().getName() +", msg, param1): ";
214             } else if (this.toString().startsWith("LOGP2_")) {
215                 return "Logger.logp(Level." + threshold().getName() +", msg, params[]): ";
216             } else if (this.toString().startsWith("LOGP3_")) {
217                 return "Logger.logp(Level." + threshold().getName() +", msg, throwable): ";
218             }
219             throw new RuntimeException("Unknown test case: "+this);
220         }

223     // The purpose of this test is to verify that the various log methods in
224     // Logger now call Logger.isLoggable().
225     // To do that - we're going to use a subclass of Logger, ThreadLogger, which
226     // only overrides isLoggable() - and compare the level it is given to a level
227     // it finds in a map indexed with the current thread id.
228     // We will register a TestHandler with our ThreadLogger which will store
229     // the messages in a messages map. This will allow us to verify whether the
230     // logging method we're testing has or hasn't logged.
231     //
232     // The TestCase enum above allows us to test a combination of every possible
233     // log method with every possible level inside a loop - with the
234     // exception of exiting/entering/throwing that we will be testing
235     // outside of that loop.
236     //
237     public static void main(String... args) {
238         LogManager manager = LogManager.getLogManager();
239         ThreadLogger logger = new ThreadLogger("foo.bar");
240         //manager.addLogger(logger);
241         TestHandler handler = new TestHandler();
242         logger.addHandler(handler);
243 
244         //By default, logger's level is Level.INFO
245         final List<Level> loggable = LEVELS.subList(0, LEVELS.indexOf(Level.INFO)+1);
246 
247         // Check our test implementation of logger.isLoggable();
248         //
249         // Since we haven't put anything in the threadMap, isLoggable() should
250         // return true for all levels stronger or equals to Level.INFO.
251         // here we're just checking that our implementation of
252         // ThreadLogger.isLoggable() returns what we want - we're just testing
253         // the test code...
254         for (Level level : LEVELS) {
255             if (logger.isLoggable(level) != loggable.contains(level)) {
256                 throw new RuntimeException(level +
257                         ": unexpected result for isLoggable(): expected " +
258                         (loggable.contains(level)));
259             }
260         }
261 
262         // Test that entering/exiting/throwing call isLoggable()
263 
264         // Here we test the default behavior: this call shouldn't log anything
265         //   because by default the logger level is Level.INFO and these
266         //   methods log at Level.FINER.
267         // So by default - these methods don't log anything. We check it here.
268         logger.entering("blah", "blah");
269         logger.entering("blah", "blah", "blah");
270         logger.entering("blah", "blah", new Object[] {"blah"});
271         if (!handler.messages.isEmpty()) {
272             throw new RuntimeException("Expected empty, got "+handler.messages);
273         }
274 
275         logger.exiting("blah", "blah");
276         logger.exiting("blah", "blah", "blah");
277         logger.exiting("blah", "blah", new Object[] {"blah"});
278         if (!handler.messages.isEmpty()) {
279             throw new RuntimeException("Expected empty, got "+handler.messages);
280         }
281 
282         logger.throwing("blah", "blah", new Exception("blah"));
283         if (!handler.messages.isEmpty()) {
284             throw new RuntimeException("Expected empty, got "+handler.messages);
285         }
286 
287         // Now we're going to put each level in turn in the threadMap.
288         // This means that isLoggable(Level.FINER) should now return true if the
289         // level in the map is not one of the level in the 'stronger' list below
290         // (here stronger=stronger than FINER)
291         final List<Level> stronger = LEVELS.subList(0, LEVELS.indexOf(Level.FINER));
292         for (Level l : LEVELS) {
293 
294             logger.threadMap.put(Thread.currentThread().getId(), l);
295 
296             // Check that our implementation of isLoggable(level) now returns true
297             // if 'level' is stronger or equals to 'l' - here we're just checking
298             // that our implementation of ThreadLogger.isLoggable() returns what
299             // we want - we're just testing the test code...
300             final List<Level> loggableLevels = LEVELS.subList(0, LEVELS.indexOf(l)+1);
301             for (Level level : LEVELS) {
302                 if (logger.isLoggable(level) != loggableLevels.contains(level)) {
303                     throw new RuntimeException(level +
304                             ": unexpected result for isLoggable(): expected " +
305                             (loggableLevels.contains(level)));
306                 }
307             }
308 
309             // These methods should now start to log when the level we put in
310             // the map is weaker or equals to Level.FINER.
311             // This validates that these methods now call ThreadLogger.isLoggable()
312             // since the default level for our logger is still Level.INFO.
313             // If the methods didn't call ThreadLogger.isLoggable() they wouldn't
314             // log anything, whatever we put in the threadMap...
315 
316             logger.entering("blah", "blah");
317             logger.entering("blah", "blah", "blah");
318             logger.entering("blah", "blah", new Object[] {"blah"});
319             if (stronger.contains(l)) {
320                 if (!handler.messages.isEmpty()) {
321                     throw new RuntimeException(l +
322                             ": Expected empty, got " + handler.messages);
323                 }
324             } else {
325                 if (handler.messages.size() != 3) {
326                     throw new RuntimeException(l +
327                             ": Expected size 3, got " + handler.messages);
328                 }
329             }
330 
331             logger.exiting("blah", "blah");
332             logger.exiting("blah", "blah", "blah");
333             logger.exiting("blah", "blah", new Object[] {"blah"});
334             if (stronger.contains(l)) {
335                 if (!handler.messages.isEmpty()) {
336                     throw new RuntimeException(l +
337                             ": Expected empty, got " + handler.messages);
338                 }
339             } else {
340                 if (handler.messages.size() != 6) {
341                     throw new RuntimeException(l +
342                             ": Expected size 6, got " + handler.messages);
343                 }
344             }
345 
346             logger.throwing("blah", "blah", new Exception("blah"));
347             if (stronger.contains(l)) {
348                 if (!handler.messages.isEmpty()) {
349                     throw new RuntimeException(l +
350                             ": Expected empty, got " + handler.messages);
351                 }
352             } else {
353                 if (handler.messages.size() != 7) {
354                     throw new RuntimeException(l +
355                             ": Expected size 7, got " + handler.messages);
356                 }
357             }
358             if (!stronger.contains(l)) {
359                 System.out.println(l + ": Logger.entering/exiting/throwing: " +
360                         handler.messages);
361             }
362             handler.messages.clear();
363         }
364 
365         // Cleanup so that we can start the next test with a clean plate...
366         handler.messages.clear();
367         logger.threadMap.clear();
368 
369         // Test that each logging method calls isLoggable()
370         //
371         for (LogTest testCase : LogTest.values()) {
372             // Each test case is a combination of:
373             //    1. A level to put in the threadMap.
374             //    2. A log method to call
375             final String method = testCase.method();
376 
377             // check our implementation of logger.isLoggable();
378             // by default the logger level is Level.INFO, so our implementation
379             // of isLoggable() should return true for all levels stronger or
380             // equal to INFO and false for the others.
381             // We check that here.
382             for (Level level : LEVELS) {
383                 if (logger.isLoggable(level) != loggable.contains(level)) {
384                     throw new RuntimeException(level +
385                             ": unexpected result for isLoggable(): expected " +
386                             (loggable.contains(level)));
387                 }
388             }
389 
390             // Check that by default the log method will not log for level
391             // weaker than Level.INFO.
392             for (Level l : LEVELS.subList(LEVELS.indexOf(Level.INFO) + 1, LEVELS.size())) {
393                 final String test = method + l + ": ";
394                 testCase.log(l, logger, "blah");
395                 if (!handler.messages.isEmpty()) {
396                     throw new RuntimeException(test +
397                             "Expected empty, got " + handler.messages);
398                 }
399             }
400 
401             // Let's put Level.OFF in the threadMap. Nothing should be logged,
402             // whichever level is used...
403             logger.threadMap.put(Thread.currentThread().getId(), Level.OFF);
404 
405             // Check that isLoggable() now always return false.
406             for (Level level : LEVELS) {
407                 if (logger.isLoggable(level)) {
408                     throw new RuntimeException(level +
409                             ": unexpected result for isLoggable(): expected " +
410                             false);
411                 }
412             }
413 
414             // Check that the log method of the test case won't log, whatever
415             // level we pass to it. This validates that level method calls
416             // isLoggable() - because otherwise it would log for levels stronger
417             // or equal to INFO.
418             for (Level l : LEVELS) {
419                 final String test = "[threadMap=OFF] " + method + l + ": ";
420                 testCase.log(l, logger, "blah");
421                 if (!handler.messages.isEmpty()) {
422                     throw new RuntimeException(test +
423                             "Expected empty, got " + handler.messages);
424                 }
425             }
426             System.out.println("[threadMap=OFF] " + method + "logged " + handler.messages);
427 
428             // Now put the testcase's level in the threadMap.
429             logger.threadMap.put(Thread.currentThread().getId(), testCase.threshold());
430 
431             // The levels for which logging should happen are those that are stronger
432             // or equals to the testcase's  thresholds.
433             final List<Level> loggableLevels =
434                     LEVELS.subList(0, LEVELS.indexOf(testCase.threshold())+1);
435 
436             // Check that our implementation of isLoggable() is taking into account
437             // what we put in the map.
438             for (Level level : LEVELS) {
439                 if (logger.isLoggable(level) != loggableLevels.contains(level)) {
440                     throw new RuntimeException(level +
441                             ": unexpected result for isLoggable(): expected " +
442                             (loggableLevels.contains(level)));
443                 }
444             }
445 
446             // Now check that the log method is indeed calling our implementation
447             // of isLoggable(). We do this by first verifying that it won't log
448             // for levels weaker than what we put in the map.
449             //
450             for (Level l : testCase.weaker()) {
451                 final String test = method + l + ": ";
452                 testCase.log(l, logger, "blah");
453                 if (!handler.messages.isEmpty()) {
454                     throw new RuntimeException(test +
455                             "Expected empty, got " + handler.messages);
456                 }
457             }
458 
459             // Then we check that it will log for the testcase threshold.
460             final String test2 = method + testCase.threshold() + ": ";
461             testCase.log(logger, testCase.threshold() + " blah");
462             if (handler.messages.isEmpty()) {
463                 throw new RuntimeException(test2 +
464                         "Expected 1 message, but list is empty");
465             }
466             if (!handler.messages.contains(testCase.threshold() + " blah")) {
467                 throw new RuntimeException(test2 + " blah not found: "
468                         + handler.messages);
469             }
470             handler.messages.clear();
471 
472             // Now we check that it logs for all 'loggable' level (and doesn't
473             // log for the others).
474             for (Level l : LEVELS) {
475                 final String test = method + l + ": ";
476                 testCase.log(l, logger, l + ": blah");
477                 if (testCase.loggable().contains(l)) {
478                     if (!handler.messages.contains(l + ": blah")) {
479                         throw new RuntimeException(test + "blah not found: " +
480                                 handler.messages);
481                     }
482                 } else {
483                     if (handler.messages.contains(l + ": blah")) {
484                         throw new RuntimeException(test + "blah found: " +
485                                 handler.messages);
486                     }
487                 }
488             }
489             if (handler.messages.size() != testCase.loggable().size()) {
490                 throw new RuntimeException(method +
491                         " Sizes don't match: expected " +
492                         testCase.loggable().size() + " got " +
493                         handler.messages);
494             }
495 
496             // Some visual feedback on what happened.
497             System.out.println(method + "logged " + handler.messages);
498 
499             // Cleanup for next step.
500             // Since we're iterating over all possible levels we can be
501             // sure that we haven't missed anything.
502             // For instance - it could be argued that logger.severe() will
503             // always log. But since we have 1 case where we put Level.OFF in
504             // the map and we have verified that severe() didn't log in that
505             // case, but that it logged in any other case, then we know
506             // beyond doubt that it called our implementation of isLoggable().
507             logger.threadMap.clear();
508             handler.messages.clear();
509         }
510 
511     }
