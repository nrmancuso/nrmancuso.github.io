46  /**
47   * @author Artem Bilan
48   *
49   * @since 5.0
50   */
51  @RunWith(SpringRunner.class)
52  @DirtiesContext
53  public class FeedDslTests {
54  
55  	@ClassRule
56  	public static final TemporaryFolder tempFolder = new TemporaryFolder();
57  
58  	@Autowired
59  	private PollableChannel entries;
60  
61  	@Autowired
62  	private PropertiesPersistingMetadataStore metadataStore;
63  
64  	@Test
65  	@SuppressWarnings("unchecked")
66  	public void testFeedEntryMessageSourceFlow() throws Exception {
67  		Message<SyndEntry> message1 = (Message<SyndEntry>) this.entries.receive(10000);
68  		Message<SyndEntry> message2 = (Message<SyndEntry>) this.entries.receive(10000);
69  		Message<SyndEntry> message3 = (Message<SyndEntry>) this.entries.receive(10000);
70  		assertThat(message1).isNotNull();
71  		assertThat(message2).isNotNull();
72  		assertThat(message3).isNotNull();
73  		long time1 = message1.getPayload().getPublishedDate().getTime();
74  		long time2 = message2.getPayload().getPublishedDate().getTime();
75  		long time3 = message3.getPayload().getPublishedDate().getTime();
76  		assertThat(time1 < time2).isTrue();
77  		assertThat(time2 < time3).isTrue();
78  		assertThat(this.entries.receive(10)).isNull();
79  
80  		this.metadataStore.flush();
81  
82  		FileReader metadataStoreFile =
83  				new FileReader(tempFolder.getRoot().getAbsolutePath() + "/metadata-store.properties");
84  		Properties metadataStoreProperties = new Properties();
85  		metadataStoreProperties.load(metadataStoreFile);
86  		assertThat(metadataStoreProperties.isEmpty()).isFalse();
87  		assertThat(metadataStoreProperties.size()).isEqualTo(1);
88  		assertThat(metadataStoreProperties.containsKey("feedTest")).isTrue();
89  	}
90  
91  	@Configuration
92  	@EnableIntegration
93  	public static class ContextConfiguration {
94  
95  		@Value("org/springframework/integration/feed/sample.rss")
96  		private Resource feedResource;
97  
98  		@Bean
99  		public MetadataStore metadataStore() {
100 			PropertiesPersistingMetadataStore metadataStore = new PropertiesPersistingMetadataStore();
101 			metadataStore.setBaseDirectory(tempFolder.getRoot().getAbsolutePath());
102 			return metadataStore;
103 		}
104 
105 		@Bean
106 		public IntegrationFlow feedFlow() {
107 			return IntegrationFlows
108 					.from(Feed.inboundAdapter(this.feedResource, "feedTest")
109 									.metadataStore(metadataStore())
110 									.preserveWireFeed(true),
111 							e -> e.poller(p -> p.fixedDelay(100)))
112 					.channel(c -> c.queue("entries"))
113 					.get();
114 		}
115 
116 	}
