114 	@Test
115 	public void testSmtp() throws Exception {
116 		assertThat(TestUtils.getPropertyValue(this.sendMailHandler, "mailSender.host")).isEqualTo("localhost");
117 
118 		Properties javaMailProperties = TestUtils.getPropertyValue(this.sendMailHandler,
119 				"mailSender.javaMailProperties", Properties.class);
120 		assertThat(javaMailProperties.getProperty("mail.debug")).isEqualTo("false");
121 
122 		this.sendMailChannel.send(MessageBuilder.withPayload("foo").build());
123 
124 		int n = 0;
125 		while (n++ < 100 && smtpServer.getMessages().size() == 0) {
126 			Thread.sleep(100);
127 		}
128 
129 		assertThat(smtpServer.getMessages().size() > 0).isTrue();
130 		String message = smtpServer.getMessages().get(0);
131 		assertThat(message).endsWith("foo\n");
132 		assertThat(message).contains("foo@bar");
133 		assertThat(message).contains("bar@baz");
134 		assertThat(message).contains("user:user");
135 		assertThat(message).contains("password:pw");
136 
137 	}

139 	@Test
140 	public void testPop3() throws IOException {
141 		Message<?> message = this.pop3Channel.receive(10000);
142 		assertThat(message).isNotNull();
143 		MessageHeaders headers = message.getHeaders();
144 		assertThat(headers.get(MailHeaders.TO, String[].class)).containsExactly("Foo <foo@bar>");
145 		assertThat(headers.get(MailHeaders.FROM)).isEqualTo("Bar <bar@baz>");
146 		assertThat(headers.get(MailHeaders.SUBJECT)).isEqualTo("Test Email");
147 		assertThat(message.getPayload()).isEqualTo("foo\r\n\r\n");
148 		assertThat(message.getHeaders().containsKey(IntegrationMessageHeaderAccessor.CLOSEABLE_RESOURCE)).isTrue();
149 		message.getHeaders().get(IntegrationMessageHeaderAccessor.CLOSEABLE_RESOURCE, Closeable.class).close();
150 	}
151 
152 	@Test
153 	public void testImap() throws Exception {
154 		Message<?> message = this.imapChannel.receive(10000);
155 		assertThat(message).isNotNull();
156 		MimeMessage mm = (MimeMessage) message.getPayload();
157 		assertThat(mm.getRecipients(RecipientType.TO)[0].toString()).isEqualTo("Foo <foo@bar>");
158 		assertThat(mm.getFrom()[0].toString()).isEqualTo("Bar <bar@baz>");
159 		assertThat(mm.getSubject()).isEqualTo("Test Email");
160 		assertThat(mm.getContent()).isEqualTo(TestMailServer.MailServer.MailHandler.BODY + "\r\n");
161 		assertThat(message.getHeaders().containsKey(IntegrationMessageHeaderAccessor.CLOSEABLE_RESOURCE)).isTrue();
162 		message.getHeaders().get(IntegrationMessageHeaderAccessor.CLOSEABLE_RESOURCE, Closeable.class).close();
163 	}
164 
165 	@Test
166 	public void testImapIdle() {
167 		Message<?> message = this.imapIdleChannel.receive(10000);
168 		assertThat(message).isNotNull();
169 		MessageHeaders headers = message.getHeaders();
170 		assertThat(headers.get(MailHeaders.TO, String[].class)).containsExactly("Foo <foo@bar>");
171 		assertThat(headers.get(MailHeaders.FROM)).isEqualTo("Bar <bar@baz>");
172 		assertThat(headers.get(MailHeaders.SUBJECT)).isEqualTo("Test Email");
173 		assertThat(message.getPayload()).isEqualTo(TestMailServer.MailServer.MailHandler.MESSAGE + "\r\n");
174 		assertThat(message.getHeaders().containsKey(IntegrationMessageHeaderAccessor.CLOSEABLE_RESOURCE)).isTrue();
175 		this.imapIdleAdapter.stop();
176 		assertThat(TestUtils.getPropertyValue(this.imapIdleAdapter, "shouldReconnectAutomatically", Boolean.class))
177 				.isFalse();
178 	}

213 		@Bean
214 		public IntegrationFlow sendMailFlow() {
215 			return IntegrationFlows.from("sendMailChannel")
216 					.enrichHeaders(Mail.headers()
217 							.subjectFunction(m -> "foo")
218 							.from("foo@bar")
219 							.toFunction(m -> new String[] { "bar@baz" }))
220 					.handle(Mail.outboundAdapter("localhost")
221 									.port(smtpServer.getPort())
222 									.credentials("user", "pw")
223 									.protocol("smtp")
224 									.javaMailProperties(p -> p.put("mail.debug", "false")),
225 							e -> e.id("sendMailEndpoint"))
226 					.get();
227 		}
228 
229 		@Bean
230 		public IntegrationFlow pop3MailFlow() {
231 			return IntegrationFlows
232 					.from(Mail.pop3InboundAdapter("localhost", pop3Server.getPort(), "user", "pw")
233 									.javaMailProperties(p -> p.put("mail.debug", "false"))
234 									.autoCloseFolder(false)
235 									.headerMapper(mailHeaderMapper()),
236 							e -> e.autoStartup(true).poller(p -> p.fixedDelay(1000)))
237 					.enrichHeaders(s -> s.headerExpressions(c -> c.put(MailHeaders.SUBJECT, "payload.subject")
238 							.put(MailHeaders.FROM, "payload.from[0].toString()")))
239 					.channel(MessageChannels.queue("pop3Channel"))
240 					.get();
241 		}
242 
243 		@Bean
244 		public IntegrationFlow imapMailFlow() {
245 			return IntegrationFlows
246 					.from(Mail.imapInboundAdapter("imap://user:pw@localhost:" + imapServer.getPort() + "/INBOX")
247 									.searchTermStrategy(this::fromAndNotSeenTerm)
248 									.userFlag("testSIUserFlag")
249 									.autoCloseFolder(false)
250 									.simpleContent(true)
251 									.javaMailProperties(p -> p.put("mail.debug", "false")),
252 							e -> e.autoStartup(true)
253 									.poller(p -> p.fixedDelay(1000)))
254 					.channel(MessageChannels.queue("imapChannel"))
255 					.get();
256 		}
257 
258 		@Bean
259 		public IntegrationFlow imapIdleFlow() {
260 			return IntegrationFlows
261 					.from(Mail.imapIdleAdapter("imap://user:pw@localhost:" + imapIdleServer.getPort() + "/INBOX")
262 							.autoStartup(true)
263 							.searchTermStrategy(this::fromAndNotSeenTerm)
264 							.userFlag("testSIUserFlag")
265 							.autoCloseFolder(false)
266 							.javaMailProperties(p -> p.put("mail.debug", "false")
267 									.put("mail.imap.connectionpoolsize", "5"))
268 							.shouldReconnectAutomatically(false)
269 							.headerMapper(mailHeaderMapper()))
270 					.channel(MessageChannels.queue("imapIdleChannel"))
271 					.get();
272 		}

274 		@Bean

279 		private SearchTerm fromAndNotSeenTerm(Flags supportedFlags, Folder folder) {
280 			try {
281 				FromTerm fromTerm = new FromTerm(new InternetAddress("bar@baz"));
282 				return new AndTerm(fromTerm, new FlagTerm(new Flags(Flags.Flag.SEEN), false));
283 			}
284 			catch (AddressException e) {
285 				throw new RuntimeException(e);
286 			}
287 
288 		}
