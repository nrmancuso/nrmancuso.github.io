56  public class CallToUnconditionalThrower extends PreorderVisitor implements Detector {
57  
58      static boolean DEBUG = false;
59  
60      BugReporter bugReporter;
61  
62      AnalysisContext analysisContext;
63  
64      private final boolean testingEnabled;
65  
66      public CallToUnconditionalThrower(BugReporter bugReporter) {
67          this.bugReporter = bugReporter;
68          testingEnabled = SystemProperties.getBoolean("report_TESTING_pattern_in_standard_detectors");
69      }
70  
71      @Override
72      public void report() {
73          //
74      }
75  
76      private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
77          if (BCELUtil.isSynthetic(method) || (method.getAccessFlags() & Const.ACC_BRIDGE) == Const.ACC_BRIDGE) {
78              return;
79          }
80          CFG cfg = classContext.getCFG(method);
81  
82          ConstantPoolGen cpg = classContext.getConstantPoolGen();
83          TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
84  
85          for (Iterator<BasicBlock> i = cfg.blockIterator(); i.hasNext();) {
86              BasicBlock basicBlock = i.next();
87  
88              // Check if it's a method invocation.
89              if (!basicBlock.isExceptionThrower()) {
90                  continue;
91              }
92              InstructionHandle thrower = basicBlock.getExceptionThrower();
93              Instruction ins = thrower.getInstruction();
94              if (!(ins instanceof InvokeInstruction)) {
95                  continue;
96              }
97  
98              InvokeInstruction inv = (InvokeInstruction) ins;
99              boolean foundThrower = false;
100             boolean foundNonThrower = false;
101 
102             if (inv instanceof INVOKEINTERFACE || inv instanceof INVOKEDYNAMIC) {
103                 continue;
104             }
105 
106             String className = inv.getClassName(cpg);
107 
108             Location loc = new Location(thrower, basicBlock);
109             TypeFrame typeFrame = typeDataflow.getFactAtLocation(loc);
110             XMethod primaryXMethod = XFactory.createXMethod(inv, cpg);
111             // if (primaryXMethod.isAbstract()) continue;
112             Set<XMethod> targetSet = null;
113             try {
114 
115                 if (className.startsWith("[")) {
116                     continue;
117                 }
118                 String methodSig = inv.getSignature(cpg);
119                 if (!methodSig.endsWith("V")) {
120                     continue;
121                 }
122 
123                 targetSet = Hierarchy2.resolveMethodCallTargets(inv, typeFrame, cpg);
124 
125                 for (XMethod xMethod : targetSet) {
126                     if (DEBUG) {
127                         System.out.println("\tFound " + xMethod);
128                     }
129 
130                     boolean isUnconditionalThrower = xMethod.isUnconditionalThrower() && !xMethod.isUnsupported()
131                             && !xMethod.isSynthetic();
132                     if (isUnconditionalThrower) {
133                         foundThrower = true;
134                         if (DEBUG) {
135                             System.out.println("Found thrower");
136                         }
137                     } else {
138                         foundNonThrower = true;
139                         if (DEBUG) {
140                             System.out.println("Found non thrower");
141                         }
142                     }
143 
144                 }
145             } catch (ClassNotFoundException e) {
146                 analysisContext.getLookupFailureCallback().reportMissingClass(e);
147             }
148             boolean newResult = foundThrower && !foundNonThrower;
149             if (newResult) {
150                 bugReporter.reportBug(new BugInstance(this, "TESTING", Priorities.NORMAL_PRIORITY)
151                 .addClassAndMethod(classContext.getJavaClass(), method)
152                 .addString("Call to method that always throws Exception").addMethod(primaryXMethod)
153                 .describe(MethodAnnotation.METHOD_CALLED).addSourceLine(classContext, method, loc));
154             }
155 
156         }
157 
158     }
159 
160     @Override
161     public void visitClassContext(ClassContext classContext) {
162         if(!testingEnabled){
163             return;
164         }
165         analysisContext = AnalysisContext.currentAnalysisContext();
166         Method[] methodList = classContext.getJavaClass().getMethods();
167         for (Method method : methodList) {
168             if (method.getCode() == null) {
169                 continue;
170             }
171 
172             try {
173 
174                 analyzeMethod(classContext, method);
175             } catch (CFGBuilderException e) {
176                 bugReporter.logError(
177                         "Error checking for infinite recursive loop in "
178                                 + SignatureConverter.convertMethodSignature(classContext.getJavaClass(), method), e);
179             } catch (DataflowAnalysisException e) {
180                 bugReporter.logError(
181                         "Error checking for infinite recursive loop in "
182                                 + SignatureConverter.convertMethodSignature(classContext.getJavaClass(), method), e);
183             }
184         }
185     }
