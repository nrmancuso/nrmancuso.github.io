45  /**
46   * <!-- START SNIPPET: description -->
47   * <p>
48   * Calls the {@link HttpServletResponse#sendRedirect(String) sendRedirect}
49   * method to the location specified. The response is told to redirect the
50   * browser to the specified location (a new request from the client). The
51   * consequence of doing this means that the action (action instance, action
52   * errors, field errors, etc) that was just executed is lost and no longer
53   * available. This is because actions are built on a single-thread model. The
54   * only way to pass data is through the session or with web parameters
55   * (url?name=value) which can be OGNL expressions.
56   * </p>
57   * <!-- END SNIPPET: description -->
58   * <p>
59   * <b>This result type takes the following parameters:</b>
60   * </p>
61   * <!-- START SNIPPET: params -->
62   * 
63   * <ul>
64   * 
65   * <li><b>location (default)</b> - the location to go to after execution.</li>
66   * 
67   * <li><b>parse</b> - true by default. If set to false, the location param will
68   * not be parsed for Ognl expressions.</li>
69   * 
70   * <li><b>anchor</b> - Optional.  Also known as "fragment" or colloquially as 
71   * "hash".  You can specify an anchor for a result.</li>
72   * </ul>
73   * 
74   * <p>
75   * This result follows the same rules from {@link StrutsResultSupport}.
76   * </p>
77   * 
78   * <!-- END SNIPPET: params -->
79   * <p>
80   * <b>Example:</b>
81   * </p>
82   *
83   * <pre>
84   * <!-- START SNIPPET: example -->
85   * &lt;!--
86   *   The redirect URL generated will be:
87   *   /foo.jsp#FRAGMENT
88   * --&gt;
89   * &lt;result name="success" type="redirect"&gt;
90   *   &lt;param name="location"&gt;foo.jsp&lt;/param&gt;
91   *   &lt;param name="parse"&gt;false&lt;/param&gt;
92   *   &lt;param name="anchor"&gt;FRAGMENT&lt;/param&gt;
93   * &lt;/result&gt;
94   * <!-- END SNIPPET: example -->
95   * </pre>
96   * 
97   */
98  public class ServletRedirectResult extends StrutsResultSupport implements ReflectionExceptionHandler, Redirectable {
99

159     public void execute(ActionInvocation invocation) throws Exception {
160         if (anchor != null) {
161             anchor = conditionalParse(anchor, invocation);
162         }
163         super.execute(invocation);
164     }

166     /**
167      * Redirects to the location specified by calling
168      * {@link HttpServletResponse#sendRedirect(String)}.
169      * 
170      * @param finalLocation the location to redirect to.
171      * @param invocation an encapsulation of the action execution state.
172      * @throws Exception if an error occurs when redirecting.
173      */
174     protected void doExecute(String finalLocation, ActionInvocation invocation) throws Exception {
175         ActionContext ctx = invocation.getInvocationContext();
176         HttpServletRequest request = (HttpServletRequest) ctx.get(ServletActionContext.HTTP_REQUEST);
177         HttpServletResponse response = (HttpServletResponse) ctx.get(ServletActionContext.HTTP_RESPONSE);
178 
179         if (isPathUrl(finalLocation)) {
180             if (!finalLocation.startsWith("/")) {
181                 ActionMapping mapping = actionMapper.getMapping(request, Dispatcher.getInstance().getConfigurationManager());
182                 String namespace = null;
183                 if (mapping != null) {
184                     namespace = mapping.getNamespace();
185                 }
186 
187                 if ((namespace != null) && (namespace.length() > 0) && (!"/".equals(namespace))) {
188                     finalLocation = namespace + "/" + finalLocation;
189                 } else {
190                     finalLocation = "/" + finalLocation;
191                 }
192             }
193 
194             // if the URL's are relative to the servlet context, append the servlet context path
195             if (prependServletContext && (request.getContextPath() != null) && (request.getContextPath().length() > 0)) {
196                 finalLocation = request.getContextPath() + finalLocation;
197             }
198         }
199         ResultConfig resultConfig = invocation.getProxy().getConfig().getResults().get(invocation.getResultCode());
200         if (resultConfig != null) {
201             Map<String, String> resultConfigParams = resultConfig.getParams();
202 
203             List<String> prohibitedResultParams = getProhibitedResultParams();
204             for (Map.Entry<String, String> e : resultConfigParams.entrySet()) {
205                 if (!prohibitedResultParams.contains(e.getKey())) {
206                     Collection<String> values = conditionalParseCollection(e.getValue(), invocation, suppressEmptyParameters);
207                     if (!suppressEmptyParameters || !values.isEmpty()) {
208                         requestParameters.put(e.getKey(), values);
209                     }
210                 }
211             }
212         }
213 
214         StringBuilder tmpLocation = new StringBuilder(finalLocation);
215         urlHelper.buildParametersString(requestParameters, tmpLocation, "&");
216 
217         // add the anchor
218         if (anchor != null) {
219             tmpLocation.append('#').append(anchor);
220         }
221 
222         finalLocation = response.encodeRedirectURL(tmpLocation.toString());
223 
224         LOG.debug("Redirecting to finalLocation: {}", finalLocation);
225 
226         sendRedirect(response, finalLocation);
227     }
228 
229     protected List<String> getProhibitedResultParams() {
230         return Arrays.asList(
231                 DEFAULT_PARAM,
232                 "namespace",
233                 "method",
234                 "encode",
235                 "parse",
236                 "location",
237                 "prependServletContext",
238                 "suppressEmptyParameters",
239                 "anchor",
240                 "statusCode"
241         );
242     }
243 
244     /**
245      * Sends the redirection. Can be overridden to customize how the redirect is
246      * handled (i.e. to use a different status code)
247      * 
248      * @param response The response
249      * @param finalLocation The location URI
250      * @throws IOException in case of IO errors
251      */
252     protected void sendRedirect(HttpServletResponse response, String finalLocation) throws IOException {
253         try {
254             if (SC_FOUND == statusCode) {
255                 response.sendRedirect(finalLocation);
256             } else {
257                 response.setStatus(statusCode);
258                 response.setHeader("Location", finalLocation);
259                 try {
260                     response.getWriter().write(finalLocation);
261                 } finally {
262                     response.getWriter().close();
263                 }
264             }
265         } catch (IOException ioe) {
266             LOG.warn("Unable to redirect to: {}, code: {}; {}", finalLocation, statusCode, ioe);
267             throw ioe;  // Re-throw required to preserve existing default behaviour (no stacktrace in above warn for this reason)
268         } catch (IllegalStateException ise) {
269             LOG.warn("Unable to redirect to: {}, code: {}; isCommited: {}; {}", finalLocation, statusCode, response.isCommitted(), ise);
270             throw ise;  // Re-throw required to preserve existing default behaviour (no stacktrace in above warn for this reason)
271         }
272 
273     }

275     /**
276      * Checks if url is simple path or either full url
277      *
278      * @param url string
279      * @return true if it's just a path not a full url
280      */
281     protected boolean isPathUrl(String url) {
282         try {
283             String rawUrl = url;
284             if (url.contains("?")) {
285                 rawUrl = url.substring(0, url.indexOf('?'));
286             }
287             URI uri = URI.create(rawUrl.replaceAll(" ", "%20"));
288             if (uri.isAbsolute()) {
289                 URL validUrl = uri.toURL();
290                 LOG.debug("[{}] is full url, not a path", url);
291                 return validUrl.getProtocol() == null;
292             } else {
293                 LOG.debug("[{}] isn't absolute URI, assuming it's a path", url);
294                 return true;
295             }
296         } catch (IllegalArgumentException e) {
297             LOG.debug("[{}] isn't a valid URL, assuming it's a path", url, e);
298             return true;
299         } catch (MalformedURLException e) {
300             LOG.debug("[{}] isn't a valid URL, assuming it's a path", url, e);
301             return true;
302         }
303     }

316      *

318      * @param value The parameter value

323         requestParameters.put(key, String.valueOf(value));
324         return this;
325     }
