188     /**
189      * Initialize the portlet with the init parameters from <tt>portlet.xml</tt>
190      *
191      * @param cfg portlet configuration
192      * @throws PortletException in case of errors
193      */
194     public void init(PortletConfig cfg) throws PortletException {
195         super.init(cfg);
196         LOG.debug("Initializing portlet {}", getPortletName());
197 
198         Map<String, String> params = new HashMap<String, String>();
199         for (Enumeration e = cfg.getInitParameterNames(); e.hasMoreElements(); ) {
200             String name = (String) e.nextElement();
201             String value = cfg.getInitParameter(name);
202             params.put(name, value);
203         }
204 
205         servletContext = new PortletServletContext(cfg.getPortletContext());
206         dispatcherUtils = new Dispatcher(servletContext, params);
207         dispatcherUtils.init();
208 
209         // For testability
210         if (factory == null) {
211             factory = dispatcherUtils.getContainer().getInstance(ActionProxyFactory.class);
212         }
213         portletNamespace = cfg.getInitParameter("portletNamespace");
214         LOG.debug("PortletNamespace: {}", portletNamespace);
215 
216         parseModeConfig(actionMap, cfg, PortletMode.VIEW, "viewNamespace",
217                 "defaultViewAction");
218         parseModeConfig(actionMap, cfg, PortletMode.EDIT, "editNamespace",
219                 "defaultEditAction");
220         parseModeConfig(actionMap, cfg, PortletMode.HELP, "helpNamespace",
221                 "defaultHelpAction");
222         parseModeConfig(actionMap, cfg, new PortletMode("config"), "configNamespace",
223                 "defaultConfigAction");
224         parseModeConfig(actionMap, cfg, new PortletMode("about"), "aboutNamespace",
225                 "defaultAboutAction");
226         parseModeConfig(actionMap, cfg, new PortletMode("print"), "printNamespace",
227                 "defaultPrintAction");
228         parseModeConfig(actionMap, cfg, new PortletMode("preview"), "previewNamespace",
229                 "defaultPreviewAction");
230         parseModeConfig(actionMap, cfg, new PortletMode("edit_defaults"),
231                 "editDefaultsNamespace", "defaultEditDefaultsAction");
232         if (StringUtils.isEmpty(portletNamespace)) {
233             portletNamespace = "";
234         }
235 
236         container = dispatcherUtils.getContainer();
237         actionMapper = container.getInstance(ActionMapper.class);
238     }

240     /**
241      * Parse the mode to namespace mappings configured in portlet.xml
242      * @param actionMap The map with mode <-> default action mapping.
243      * @param portletConfig The PortletConfig.
244      * @param portletMode The PortletMode.
245      * @param nameSpaceParam Name of the init parameter where the namespace for the mode
246      * is configured.
247      * @param defaultActionParam Name of the init parameter where the default action to
248      * execute for the mode is configured.
249      */
250     void parseModeConfig(Map<PortletMode, ActionMapping> actionMap, PortletConfig portletConfig,
251                          PortletMode portletMode, String nameSpaceParam,
252                          String defaultActionParam) {
253         String namespace = portletConfig.getInitParameter(nameSpaceParam);
254         if (StringUtils.isEmpty(namespace)) {
255             namespace = "";
256         }
257         modeMap.put(portletMode, namespace);
258         String defaultAction = portletConfig.getInitParameter(defaultActionParam);
259         String method = null;
260         if (StringUtils.isEmpty(defaultAction)) {
261             defaultAction = DEFAULT_ACTION_NAME;
262         }
263         if (defaultAction.indexOf('!') >= 0) {
264             method = defaultAction.substring(defaultAction.indexOf('!') + 1);
265             defaultAction = defaultAction.substring(0, defaultAction.indexOf('!'));
266         }
267         StringBuilder fullPath = new StringBuilder();
268         if (StringUtils.isNotEmpty(portletNamespace)) {
269             fullPath.append(portletNamespace);
270         }
271         if (StringUtils.isNotEmpty(namespace)) {
272             fullPath.append(namespace).append("/");
273         } else {
274             fullPath.append("/");
275         }
276         fullPath.append(defaultAction);
277         ActionMapping mapping = new ActionMapping();
278         mapping.setName(getActionName(fullPath.toString()));
279         mapping.setNamespace(getNamespace(fullPath.toString()));
280         if (method != null) {
281             mapping.setMethod(method);
282         }
283         actionMap.put(portletMode, mapping);
284     }

286     /**
287      * Service an action from the <tt>event</tt> phase.
288      *
289      * @param request action request
290      * @param response  action response
291      *
292      * @throws PortletException in case of errors
293      * @throws IOException in case of IO errors
294      *
295      * @see javax.portlet.Portlet#processAction(javax.portlet.ActionRequest,
296      *      javax.portlet.ActionResponse)
297      */
298     public void processAction(ActionRequest request, ActionResponse response)
299             throws PortletException, IOException {
300         if (LOG.isDebugEnabled()) {
301             LOG.debug("Entering processAction in mode ", request.getPortletMode().toString());
302         }
303         resetActionContext();
304         try {
305             serviceAction(request, response, getRequestMap(request), getParameterMap(request),
306                     getSessionMap(request), getApplicationMap(),
307                     portletNamespace, PortletPhase.ACTION_PHASE);
308             if (LOG.isDebugEnabled()) LOG.debug("Leaving processAction");
309         } finally {
310             ActionContext.setContext(null);
311         }
312     }

314     /**
315      * Service an action from the <tt>render</tt> phase.
316      *
317      * @param request render request
318      * @param response  render response
319      *
320      * @throws PortletException in case of errors
321      * @throws IOException in case of IO errors
322      *
323      * @see javax.portlet.Portlet#render(javax.portlet.RenderRequest,
324      *      javax.portlet.RenderResponse)
325      */
326     public void render(RenderRequest request, RenderResponse response)
327             throws PortletException, IOException {
328 
329         if (LOG.isDebugEnabled()) {
330             LOG.debug("Entering render in mode ", request.getPortletMode().toString());
331         }
332         resetActionContext();
333         response.setTitle(getTitle(request));
334         if (!request.getWindowState().equals(WindowState.MINIMIZED)) {
335             try {
336                 // Check to see if an event set the render to be included directly
337                 serviceAction(request, response, getRequestMap(request), getParameterMap(request),
338                         getSessionMap(request), getApplicationMap(),
339                         portletNamespace, PortletPhase.RENDER_PHASE);
340                 if (LOG.isDebugEnabled()) LOG.debug("Leaving render");
341             } finally {
342                 resetActionContext();
343             }
344         }
345     }

354     /**
355      * Merges all application and portlet attributes into a single
356      * <tt>HashMap</tt> to represent the entire <tt>Action</tt> context.
357      *
358      * @param requestMap a Map of all request attributes.
359      * @param parameterMap a Map of all request parameters.
360      * @param sessionMap a Map of all session attributes.
361      * @param applicationMap a Map of all servlet context attributes.
362      * @param request the PortletRequest object.
363      * @param response the PortletResponse object.
364      * @param servletRequest the HttpServletRequest object.
365      * @param servletResponse the HttpServletResponse object.
366      * @param servletContext the ServletContext object.
367      * @param portletConfig the PortletConfig object.
368      * @param phase The portlet phase (render or action, see
369      *        {@link PortletConstants})
370      * @return a HashMap representing the <tt>Action</tt> context.
371      *
372      * @throws IOException in case of IO errors
373      */
374     public HashMap<String, Object> createContextMap(Map<String, Object> requestMap, Map<String, String[]> parameterMap,
375                                                     Map<String, Object> sessionMap, Map<String, Object> applicationMap,
376                                                     PortletRequest request, PortletResponse response, HttpServletRequest servletRequest,
377                                                     HttpServletResponse servletResponse, ServletContext servletContext,
378                                                     PortletConfig portletConfig, PortletPhase phase) throws IOException {
379 
380         // TODO Must put http request/response objects into map for use with
381         container.inject(servletRequest);
382 
383         // ServletActionContext
384         HashMap<String, Object> extraContext = new HashMap<String, Object>();
385         // The dummy servlet objects. Eases reuse of existing interceptors that uses the servlet objects.
386         extraContext.put(StrutsStatics.HTTP_REQUEST, servletRequest);
387         extraContext.put(StrutsStatics.HTTP_RESPONSE, servletResponse);
388         extraContext.put(StrutsStatics.SERVLET_CONTEXT, servletContext);
389         // End dummy servlet objects
390         extraContext.put(ActionContext.PARAMETERS, HttpParameters.create(parameterMap).build());
391         extraContext.put(ActionContext.SESSION, sessionMap);
392         extraContext.put(ActionContext.APPLICATION, applicationMap);
393 
394         extraContext.put(ActionContext.LOCALE, getLocale(request));
395 
396         extraContext.put(StrutsStatics.STRUTS_PORTLET_CONTEXT, getPortletContext());
397         extraContext.put(REQUEST, request);
398         extraContext.put(RESPONSE, response);
399         extraContext.put(PORTLET_CONFIG, portletConfig);
400         extraContext.put(PORTLET_NAMESPACE, portletNamespace);
401         extraContext.put(DEFAULT_ACTION_FOR_MODE, actionMap.get(request.getPortletMode()));
402         // helpers to get access to request/session/application scope
403         extraContext.put("request", requestMap);
404         extraContext.put("session", sessionMap);
405         extraContext.put("application", applicationMap);
406         extraContext.put("parameters", parameterMap);
407         extraContext.put(MODE_NAMESPACE_MAP, modeMap);
408         extraContext.put(PortletConstants.DEFAULT_ACTION_MAP, actionMap);
409 
410         extraContext.put(PortletConstants.PHASE, phase);
411 
412         AttributeMap attrMap = new AttributeMap(extraContext);
413         extraContext.put("attr", attrMap);
414 
415         return extraContext;
416     }

418     protected Locale getLocale(PortletRequest request) {
419         String defaultLocale = container.getInstance(String.class, StrutsConstants.STRUTS_LOCALE);
420         Locale locale;
421         if (defaultLocale != null) {
422             try {
423                 locale = LocaleUtils.toLocale(defaultLocale);
424             } catch (IllegalArgumentException e) {
425                 LOG.warn(new ParameterizedMessage("Cannot convert 'struts.locale' = [{}] to proper locale, defaulting to request locale [{}]",
426                         defaultLocale, request.getLocale()), e);
427                 locale = request.getLocale();
428             }
429         } else {
430             locale = request.getLocale();
431         }
432         return locale;
433     }

435     /**
436      * Loads the action and executes it. This method first creates the action
437      * context from the given parameters then loads an <tt>ActionProxy</tt>
438      * from the given action name and namespace. After that, the action is
439      * executed and output channels throught the response object.
440      *
441      * @param request the HttpServletRequest object.
442      * @param response the HttpServletResponse object.
443      * @param requestMap a Map of request attributes.
444      * @param parameterMap a Map of request parameters.
445      * @param sessionMap a Map of all session attributes.
446      * @param applicationMap a Map of all application attributes.
447      * @param portletNamespace the namespace or context of the action.
448      * @param phase The portlet phase (render or action, see {@link PortletConstants})
449      *
450      * @throws PortletException in case of errors
451      */
452     public void serviceAction(PortletRequest request, PortletResponse response, Map<String, Object> requestMap, Map<String, String[]> parameterMap,
453                               Map<String, Object> sessionMap, Map<String, Object> applicationMap, String portletNamespace,
454                               PortletPhase phase) throws PortletException {
455         if (LOG.isDebugEnabled()) LOG.debug("serviceAction");
456         Dispatcher.setInstance(dispatcherUtils);
457         String actionName = null;
458         String namespace;
459         try {
460             HttpServletRequest servletRequest = new PortletServletRequest(request, getPortletContext());
461             HttpServletResponse servletResponse = createPortletServletResponse(response);
462             if (phase.isAction()) {
463                 servletRequest = dispatcherUtils.wrapRequest(servletRequest);
464                 if (servletRequest instanceof MultiPartRequestWrapper) {
465                     // Multipart request. Request parameters are encoded in the multipart data,
466                     // so we need to manually add them to the parameter map.
467                     parameterMap.putAll(servletRequest.getParameterMap());
468                 }
469             }
470             container.inject(servletRequest);
471             ActionMapping mapping = getActionMapping(request, servletRequest);
472             actionName = mapping.getName();
473             if ("renderDirect".equals(actionName)) {
474                 namespace = request.getParameter(PortletConstants.RENDER_DIRECT_NAMESPACE);
475             } else {
476                 namespace = mapping.getNamespace();
477             }
478             HashMap<String, Object> extraContext = createContextMap(requestMap, parameterMap,
479                     sessionMap, applicationMap, request, response, servletRequest, servletResponse,
480                     servletContext, getPortletConfig(), phase);
481             extraContext.put(PortletConstants.ACTION_MAPPING, mapping);
482             if (LOG.isDebugEnabled()) {
483                 LOG.debug("Creating action proxy for name = " + actionName + ", namespace = " + namespace);
484             }
485             ActionProxy proxy = factory.createActionProxy(namespace, actionName, mapping.getMethod(), extraContext);
486             request.setAttribute("struts.valueStack", proxy.getInvocation().getStack());
487             proxy.execute();
488         } catch (ConfigurationException e) {
489             if (LOG.isErrorEnabled()) {
490                 LOG.error("Could not find action", e);
491             }
492             throw new PortletException("Could not find action " + actionName, e);
493         } catch (Exception e) {
494             if (LOG.isErrorEnabled()) {
495                 LOG.error("Could not execute action", e);
496             }
497             throw new PortletException("Error executing action " + actionName, e);
498         } finally {
499             Dispatcher.setInstance(null);
500         }
501     }

513     /**
514      * Gets the namespace of the action from the request. The namespace is the
515      * same as the portlet mode. E.g, view mode is mapped to namespace
516      * <code>view</code>, and edit mode is mapped to the namespace
517      * <code>edit</code>
518      *
519      * @param portletRequest the PortletRequest object.
520      * @param servletRequest the ServletRequest to use
521      *
522      * @return the namespace of the action.
523      */
524     protected ActionMapping getActionMapping(final PortletRequest portletRequest, final HttpServletRequest servletRequest) {
525         ActionMapping mapping;
526         String actionPath = getDefaultActionPath(portletRequest);
527         if (resetAction(portletRequest)) {
528             mapping = actionMap.get(portletRequest.getPortletMode());
529         } else {
530             actionPath = servletRequest.getParameter(ACTION_PARAM);
531             if (StringUtils.isEmpty(actionPath)) {
532                 mapping = actionMap.get(portletRequest.getPortletMode());
533             } else {
534 
535                 // Use the usual action mapper, but it is expecting an action extension
536                 // on the uri, so we add the default one, which should be ok as the
537                 // portlet is a portlet first, a servlet second
538                 mapping = actionMapper.getMapping(servletRequest, dispatcherUtils.getConfigurationManager());
539             }
540         }
541 
542         if (mapping == null) {
543             throw new StrutsException("Unable to locate action mapping for request, probably due to an invalid action path: " + actionPath);
544         }
545         return mapping;
546     }

552     /**
553      * Get the namespace part of the action path.
554      * @param actionPath Full path to action
555      * @return The namespace part.
556      */
557     String getNamespace(String actionPath) {
558         int idx = actionPath.lastIndexOf('/');
559         String namespace = "";
560         if (idx >= 0) {
561             namespace = actionPath.substring(0, idx);
562         }
563         return namespace;
564     }

566     /**
567      * Get the action name part of the action path.
568      * @param actionPath Full path to action
569      * @return The action name.
570      */
571     String getActionName(String actionPath) {
572         int idx = actionPath.lastIndexOf('/');
573         String action = actionPath;
574         if (idx >= 0) {
575             action = actionPath.substring(idx + 1);
576         }
577         return action;
578     }

625     /**
626      * Check to see if the action parameter is valid for the current portlet mode. If the portlet
627      * mode has been changed with the portal widgets, the action name is invalid, since the
628      * action name belongs to the previous executing portlet mode. If this method evaluates to
629      * <code>true</code> the <code>default&lt;Mode&gt;Action</code> is used instead.
630      * @param request The portlet request.
631      * @return <code>true</code> if the action should be reset.
632      */
633     private boolean resetAction(PortletRequest request) {
634         boolean reset = false;
635         Map paramMap = request.getParameterMap();
636         String[] modeParam = (String[]) paramMap.get(MODE_PARAM);
637         if (modeParam != null && modeParam.length == 1) {
638             String originatingMode = modeParam[0];
639             String currentMode = request.getPortletMode().toString();
640             if (!currentMode.equals(originatingMode)) {
641                 reset = true;
642             }
643         }
644         if (reset) {
645             request.setAttribute(ACTION_RESET, Boolean.TRUE);
646         } else {
647             request.setAttribute(ACTION_RESET, Boolean.FALSE);
648         }
649         return reset;
650     }
