137 /**
138  * An RPC server that hosts protobuf described Services.
139  *
140  * An RpcServer instance has a Listener that hosts the socket.  Listener has fixed number
141  * of Readers in an ExecutorPool, 10 by default.  The Listener does an accept and then
142  * round robin a Reader is chosen to do the read.  The reader is registered on Selector.  Read does
143  * total read off the channel and the parse from which it makes a Call.  The call is wrapped in a
144  * CallRunner and passed to the scheduler to be run.  Reader goes back to see if more to be done
145  * and loops till done.
146  *
147  * <p>Scheduler can be variously implemented but default simple scheduler has handlers to which it
148  * has given the queues into which calls (i.e. CallRunner instances) are inserted.  Handlers run
149  * taking from the queue.  They run the CallRunner#run method on each item gotten from queue
150  * and keep taking while the server is up.
151  *
152  * CallRunner#run executes the call.  When done, asks the included Call to put itself on new
153  * queue for Responder to pull from and return result to client.
154  *
155  * @see RpcClientImpl
156  */

377     protected synchronized void setResponse(Object m, final CellScanner cells,
378         Throwable t, String errorMsg) {
379       if (this.isError) return;
380       if (t != null) this.isError = true;
381       BufferChain bc = null;
382       try {
383         ResponseHeader.Builder headerBuilder = ResponseHeader.newBuilder();
384         // Presume it a pb Message.  Could be null.
385         Message result = (Message)m;
386         // Call id.
387         headerBuilder.setCallId(this.id);
388         if (t != null) {
389           ExceptionResponse.Builder exceptionBuilder = ExceptionResponse.newBuilder();
390           exceptionBuilder.setExceptionClassName(t.getClass().getName());
391           exceptionBuilder.setStackTrace(errorMsg);
392           exceptionBuilder.setDoNotRetry(t instanceof DoNotRetryIOException);
393           if (t instanceof RegionMovedException) {
394             // Special casing for this exception.  This is only one carrying a payload.
395             // Do this instead of build a generic system for allowing exceptions carry
396             // any kind of payload.
397             RegionMovedException rme = (RegionMovedException)t;
398             exceptionBuilder.setHostname(rme.getHostname());
399             exceptionBuilder.setPort(rme.getPort());
400           }
401           // Set the exception as the result of the method invocation.
402           headerBuilder.setException(exceptionBuilder.build());
403         }
404         // Pass reservoir to buildCellBlock. Keep reference to returne so can add it back to the
405         // reservoir when finished. This is hacky and the hack is not contained but benefits are
406         // high when we can avoid a big buffer allocation on each rpc.
407         this.cellBlock = ipcUtil.buildCellBlock(this.connection.codec,
408           this.connection.compressionCodec, cells, reservoir);
409         if (this.cellBlock != null) {
410           CellBlockMeta.Builder cellBlockBuilder = CellBlockMeta.newBuilder();
411           // Presumes the cellBlock bytebuffer has been flipped so limit has total size in it.
412           cellBlockBuilder.setLength(this.cellBlock.limit());
413           headerBuilder.setCellBlockMeta(cellBlockBuilder.build());
414         }
415         Message header = headerBuilder.build();
416 
417         // Organize the response as a set of bytebuffers rather than collect it all together inside
418         // one big byte array; save on allocations.
419         ByteBuffer bbHeader = IPCUtil.getDelimitedMessageAsByteBuffer(header);
420         ByteBuffer bbResult = IPCUtil.getDelimitedMessageAsByteBuffer(result);
421         int totalSize = bbHeader.capacity() + (bbResult == null? 0: bbResult.limit()) +
422           (this.cellBlock == null? 0: this.cellBlock.limit());
423         ByteBuffer bbTotalSize = ByteBuffer.wrap(Bytes.toBytes(totalSize));
424         bc = new BufferChain(bbTotalSize, bbHeader, bbResult, this.cellBlock);
425         if (connection.useWrap) {
426           bc = wrapWithSasl(bc);
427         }
428       } catch (IOException e) {
429         LOG.warn("Exception while creating response " + e);
430       }
431       this.response = bc;
432     }

434     private BufferChain wrapWithSasl(BufferChain bc)
435         throws IOException {
436       if (!this.connection.useSasl) return bc;
437       // Looks like no way around this; saslserver wants a byte array.  I have to make it one.
438       // THIS IS A BIG UGLY COPY.
439       byte [] responseBytes = bc.getBytes();
440       byte [] token;
441       // synchronization may be needed since there can be multiple Handler
442       // threads using saslServer to wrap responses.
443       synchronized (connection.saslServer) {
444         token = connection.saslServer.wrap(responseBytes, 0, responseBytes.length);
445       }
446       if (LOG.isTraceEnabled()) {
447         LOG.trace("Adding saslServer wrapped token of size " + token.length
448             + " as call response.");
449       }
450 
451       ByteBuffer bbTokenLength = ByteBuffer.wrap(Bytes.toBytes(token.length));
452       ByteBuffer bbTokenBytes = ByteBuffer.wrap(token);
453       return new BufferChain(bbTokenLength, bbTokenBytes);
454     }

559     private ServerSocketChannel acceptChannel = null; //the accept channel
560     private Selector selector = null; //the selector that we use for the server

562     private int currentReader = 0;
563     private Random rand = new Random();
564     private long lastCleanupRunTime = 0; //the last time when a cleanup connec-
565                                          //-tion (for idle connections) ran

567                                           //two cleanup runs

570     private ExecutorService readPool;

572     public Listener(final String name) throws IOException {
573       super(name);
574       backlogLength = conf.getInt("hbase.ipc.server.listen.queue.size", 128);
575       // Create a new server socket and set to non blocking mode
576       acceptChannel = ServerSocketChannel.open();
577       acceptChannel.configureBlocking(false);
578 
579       // Bind the server socket to the binding addrees (can be different from the default interface)
580       bind(acceptChannel.socket(), bindAddress, backlogLength);
581       port = acceptChannel.socket().getLocalPort(); //Could be an ephemeral port
582       // create a selector;
583       selector= Selector.open();
584 
585       readers = new Reader[readThreads];
586       readPool = Executors.newFixedThreadPool(readThreads,
587         new ThreadFactoryBuilder().setNameFormat(
588           "RpcServer.reader=%d,bindAddress=" + bindAddress.getHostName() +
589           ",port=" + port).setDaemon(true).build());
590       for (int i = 0; i < readThreads; ++i) {
591         Reader reader = new Reader();
592         readers[i] = reader;
593         readPool.execute(reader);
594       }
595       LOG.info(getName() + ": started " + readThreads + " reader(s).");
596 
597       // Register accepts on the server socket with the selector.
598       acceptChannel.register(selector, SelectionKey.OP_ACCEPT);
599       this.setName("RpcServer.listener,port=" + port);
600       this.setDaemon(true);
601     }

603 
604     private class Reader implements Runnable {
605       private volatile boolean adding = false;
606       private final Selector readSelector;
607 
608       Reader() throws IOException {
609         this.readSelector = Selector.open();
610       }
611       @Override
612       public void run() {
613         try {
614           doRunLoop();
615         } finally {
616           try {
617             readSelector.close();
618           } catch (IOException ioe) {
619             LOG.error(getName() + ": error closing read selector in " + getName(), ioe);
620           }
621         }
622       }
623 
624       private synchronized void doRunLoop() {
625         while (running) {
626           try {
627             readSelector.select();
628             while (adding) {
629               this.wait(1000);
630             }
631 
632             Iterator<SelectionKey> iter = readSelector.selectedKeys().iterator();
633             while (iter.hasNext()) {
634               SelectionKey key = iter.next();
635               iter.remove();
636               if (key.isValid()) {
637                 if (key.isReadable()) {
638                   doRead(key);
639                 }
640               }
641             }
642           } catch (InterruptedException e) {

674     /** cleanup connections from connectionList. Choose a random range
675      * to scan and also have a limit on the number of the connections
676      * that will be cleanedup per run. The criteria for cleanup is the time
677      * for which the connection was idle. If 'force' is true then all
678      * connections will be looked at for the cleanup.
679      * @param force all connections will be looked at for cleanup
680      */
681     private void cleanupConnections(boolean force) {
682       if (force || numConnections > thresholdIdleConnections) {
683         long currentTime = System.currentTimeMillis();
684         if (!force && (currentTime - lastCleanupRunTime) < cleanupInterval) {
685           return;
686         }
687         int start = 0;
688         int end = numConnections - 1;
689         if (!force) {
690           start = rand.nextInt() % numConnections;
691           end = rand.nextInt() % numConnections;
692           int temp;
693           if (end < start) {
694             temp = start;
695             start = end;
696             end = temp;
697           }
698         }
699         int i = start;
700         int numNuked = 0;
701         while (i <= end) {
702           Connection c;
703           synchronized (connectionList) {
704             try {
705               c = connectionList.get(i);
706             } catch (Exception e) {return;}
707           }
708           if (c.timedOut(currentTime)) {
709             if (LOG.isDebugEnabled())
710               LOG.debug(getName() + ": disconnecting client " + c.getHostAddress());
711             closeConnection(c);
712             numNuked++;
713             end--;
714             //noinspection UnusedAssignment
715             c = null;
716             if (!force && numNuked == maxConnectionsToNuke) break;
717           }
718           else i++;
719         }
720         lastCleanupRunTime = System.currentTimeMillis();
721       }
722     }

965       long lastPurgeTime = 0;   // last check for old calls.
966       while (running) {

996           } else {

1000             // some thread(s) a chance to finish

1005             } catch (InterruptedException ex) {

1007               return;

1010         } catch (Exception e) {

1018     /**
1019      * If there were some calls that have not been sent out for a
1020      * long time, we close the connection.
1021      * @return the time of the purge.
1022      */
1023     private long purge(long lastPurgeTime) {
1024       long now = System.currentTimeMillis();
1025       if (now < lastPurgeTime + purgeTimeout) {
1026         return lastPurgeTime;
1027       }
1028 
1029       ArrayList<Connection> conWithOldCalls = new ArrayList<Connection>();
1030       // get the list of channels from list of keys.
1031       synchronized (writeSelector.keys()) {
1032         for (SelectionKey key : writeSelector.keys()) {
1033           Connection connection = (Connection) key.attachment();
1034           if (connection == null) {
1035             throw new IllegalStateException("Coding error: SelectionKey key without attachment.");
1036           }
1037           Call call = connection.responseQueue.peekFirst();
1038           if (call != null && now > call.timestamp + purgeTimeout) {
1039             conWithOldCalls.add(call.connection);
1040           }
1041         }
1042       }
1043 
1044       // Seems safer to close the connection outside of the synchronized loop...
1045       for (Connection connection : conWithOldCalls) {
1046         closeConnection(connection);
1047       }
1048 
1049       return now;
1050     }

1052     private void doAsyncWrite(SelectionKey key) throws IOException {
1053       Connection connection = (Connection) key.attachment();
1054       if (connection == null) {
1055         throw new IOException("doAsyncWrite: no connection");
1056       }

1058         throw new IOException("doAsyncWrite: bad channel");
1059       }

1064           //  write anymore.
1065          key.interestOps(0);

1068            * We don't explicitly cancel the key, so not sure if this will
1069            * ever fire.
1070            * This warning could be removed.
1071            */
1072           LOG.warn("Exception while changing ops : " + e);
1073         }

1077     /**
1078      * Process the response for this call. You need to have the lock on
1079      * {@link org.apache.hadoop.hbase.ipc.RpcServer.Connection#responseWriteLock}
1080      *
1081      * @param call the call
1082      * @return true if we proceed the call fully, false otherwise.
1083      * @throws IOException
1084      */
1085     private boolean processResponse(final Call call) throws IOException {
1086       boolean error = true;
1087       try {
1088         // Send as much data as we can in the non-blocking fashion
1089         long numBytes = channelWrite(call.connection.channel, call.response);
1090         if (numBytes < 0) {
1091           throw new HBaseIOException("Error writing on the socket " +
1092             "for the call:" + call.toShortString());
1093         }
1094         error = false;
1095       } finally {
1096         if (error) {
1097           LOG.debug(getName() + call.toShortString() + ": output error -- closing");
1098           closeConnection(call.connection);
1099         }
1100       }
1101 
1102       if (!call.response.hasRemaining()) {
1103         call.done();
1104         return true;
1105       } else {
1106         return false; // Socket can't take more, we will have to come back.
1107       }
1108     }

1110     /**
1111      * Process all the responses for this connection
1112      *
1113      * @return true if all the calls were processed or that someone else is doing it.
1114      * false if there * is still some work to do. In this case, we expect the caller to
1115      * delay us.
1116      * @throws IOException
1117      */
1118     private boolean processAllResponses(final Connection connection) throws IOException {
1119       // We want only one writer on the channel for a connection at a time.
1120       connection.responseWriteLock.lock();
1121       try {
1122         for (int i = 0; i < 20; i++) {
1123           // protection if some handlers manage to need all the responder
1124           Call call = connection.responseQueue.pollFirst();
1125           if (call == null) {
1126             return true;
1127           }
1128           if (!processResponse(call)) {
1129             connection.responseQueue.addFirst(call);
1130             return false;
1131           }
1132         }
1133       } finally {
1134         connection.responseWriteLock.unlock();
1135       }
1136 
1137       return connection.responseQueue.isEmpty();
1138     }

1140     //
1141     // Enqueue a response from the application.
1142     //
1143     void doRespond(Call call) throws IOException {
1144       boolean added = false;
1145 
1146       // If there is already a write in progress, we don't wait. This allows to free the handlers
1147       //  immediately for other tasks.
1148       if (call.connection.responseQueue.isEmpty() && call.connection.responseWriteLock.tryLock()) {
1149         try {
1150           if (call.connection.responseQueue.isEmpty()) {
1151             // If we're alone, we can try to do a direct call to the socket. It's
1152             //  an optimisation to save on context switches and data transfer between cores..
1153             if (processResponse(call)) {
1154               return; // we're done.
1155             }
1156             // Too big to fit, putting ahead.
1157             call.connection.responseQueue.addFirst(call);
1158             added = true; // We will register to the selector later, outside of the lock.
1159           }
1160         } finally {
1161           call.connection.responseWriteLock.unlock();
1162         }
1163       }
1164 
1165       if (!added) {
1166         call.connection.responseQueue.addLast(call);
1167       }
1168       call.responder.registerForWrite(call.connection);
1169 
1170       // set the serve time when the response has to be sent later
1171       call.timestamp = System.currentTimeMillis();
1172     }

1363                         + "hostname part: " + fullName);
1364               }
1365               current.doAs(new PrivilegedExceptionAction<Object>() {
1366                 @Override
1367                 public Object run() throws SaslException {
1368                   saslServer = Sasl.createSaslServer(AuthMethod.KERBEROS
1369                       .getMechanismName(), names[0], names[1],
1370                       SaslUtil.SASL_PROPS, new SaslGssCallbackHandler());
1371                   return null;
1372                 }
1373               });
1374             }
1375             if (saslServer == null)
1376               throw new AccessDeniedException(
1377                   "Unable to find SASL server implementation for "
1378                       + authMethod.getMechanismName());
1379             if (LOG.isDebugEnabled()) {
1380               LOG.debug("Created SASL server with mechanism = " + authMethod.getMechanismName());
1381             }
1382           }
1383           if (LOG.isDebugEnabled()) {
1384             LOG.debug("Have read input token of size " + saslToken.length
1385                 + " for processing by saslServer.evaluateResponse()");
1386           }
1387           replyToken = saslServer.evaluateResponse(saslToken);
1388         } catch (IOException e) {
1389           IOException sendToClient = e;
1390           Throwable cause = e;
1391           while (cause != null) {
1392             if (cause instanceof InvalidToken) {
1393               sendToClient = (InvalidToken) cause;
1394               break;
1395             }
1396             cause = cause.getCause();
1397           }
1398           doRawSaslReply(SaslStatus.ERROR, null, sendToClient.getClass().getName(),
1399             sendToClient.getLocalizedMessage());
1400           metrics.authenticationFailure();
1401           String clientIP = this.toString();
1402           // attempting user could be null
1403           AUDITLOG.warn(AUTH_FAILED_FOR + clientIP + ":" + attemptingUser);
1404           throw e;
1405         }
1406         if (replyToken != null) {
1407           if (LOG.isDebugEnabled()) {
1408             LOG.debug("Will send token of size " + replyToken.length
1409                 + " from saslServer.");
1410           }
1411           doRawSaslReply(SaslStatus.SUCCESS, new BytesWritable(replyToken), null,
1412               null);
1413         }
1414         if (saslServer.isComplete()) {
1415           String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
1416           useWrap = qop != null && !"auth".equalsIgnoreCase(qop);
1417           user = getAuthorizedUgi(saslServer.getAuthorizationID());
1418           if (LOG.isDebugEnabled()) {
1419             LOG.debug("SASL server context established. Authenticated client: "
1420               + user + ". Negotiated QoP is "
1421               + saslServer.getNegotiatedProperty(Sasl.QOP));
1422           }
1423           metrics.authenticationSuccess();
1424           AUDITLOG.info(AUTH_SUCCESSFUL_FOR + user);
1425           saslContextEstablished = true;
1426         }
1427       }
1428     }

1431      * No protobuf encoding of raw sasl messages
1432      */
1433     private void doRawSaslReply(SaslStatus status, Writable rv,
1434         String errorClass, String error) throws IOException {
1435       ByteBufferOutputStream saslResponse = null;
1436       DataOutputStream out = null;
1437       try {
1438         // In my testing, have noticed that sasl messages are usually
1439         // in the ballpark of 100-200. That's why the initial capacity is 256.
1440         saslResponse = new ByteBufferOutputStream(256);
1441         out = new DataOutputStream(saslResponse);
1442         out.writeInt(status.state); // write status
1443         if (status == SaslStatus.SUCCESS) {
1444           rv.write(out);
1445         } else {
1446           WritableUtils.writeString(out, errorClass);
1447           WritableUtils.writeString(out, error);
1448         }
1449         saslCall.setSaslTokenResponse(saslResponse.getByteBuffer());
1450         saslCall.responder = responder;
1451         saslCall.sendResponseIfReady();
1452       } finally {
1453         if (saslResponse != null) {
1454           saslResponse.close();
1455         }
1456         if (out != null) {
1457           out.close();
1458         }
1459       }
1460     }

1473     private int readPreamble() throws IOException {
1474       int count;
1475       // Check for 'HBas' magic.
1476       this.dataLengthBuffer.flip();
1477       if (!Arrays.equals(HConstants.RPC_HEADER, dataLengthBuffer.array())) {
1478         return doBadPreambleHandling("Expected HEADER=" +
1479             Bytes.toStringBinary(HConstants.RPC_HEADER) +
1480             " but received HEADER=" + Bytes.toStringBinary(dataLengthBuffer.array()) +
1481             " from " + toString());
1482       }
1483       // Now read the next two bytes, the version and the auth to use.
1484       ByteBuffer versionAndAuthBytes = ByteBuffer.allocate(2);
1485       count = channelRead(channel, versionAndAuthBytes);
1486       if (count < 0 || versionAndAuthBytes.remaining() > 0) {
1487         return count;
1488       }
1489       int version = versionAndAuthBytes.get(0);
1490       byte authbyte = versionAndAuthBytes.get(1);
1491       this.authMethod = AuthMethod.valueOf(authbyte);
1492       if (version != CURRENT_VERSION) {
1493         String msg = getFatalConnectionString(version, authbyte);
1494         return doBadPreambleHandling(msg, new WrongVersionException(msg));
1495       }
1496       if (authMethod == null) {
1497         String msg = getFatalConnectionString(version, authbyte);
1498         return doBadPreambleHandling(msg, new BadAuthException(msg));
1499       }
1500       if (isSecurityEnabled && authMethod == AuthMethod.SIMPLE) {
1501         AccessDeniedException ae = new AccessDeniedException("Authentication is required");
1502         setupResponse(authFailedResponse, authFailedCall, ae, ae.getMessage());
1503         responder.doRespond(authFailedCall);
1504         throw ae;
1505       }
1506       if (!isSecurityEnabled && authMethod != AuthMethod.SIMPLE) {
1507         doRawSaslReply(SaslStatus.SUCCESS, new IntWritable(
1508             SaslUtil.SWITCH_TO_SIMPLE_AUTH), null, null);
1509         authMethod = AuthMethod.SIMPLE;
1510         // client has already sent the initial Sasl message and we
1511         // should ignore it. Both client and server should fall back
1512         // to simple auth from now on.
1513         skipInitialSaslHandshake = true;
1514       }
1515       if (authMethod != AuthMethod.SIMPLE) {
1516         useSasl = true;
1517       }
1518 
1519       dataLengthBuffer.clear();
1520       connectionPreambleRead = true;
1521       return count;
1522     }

1528         return 0;

1533     /**
1534      * Read off the wire. If there is not enough data to read, update the connection state with
1535      *  what we have and returns.
1536      * @return Returns -1 if failure (and caller will close connection), else zero or more.
1537      * @throws IOException
1538      * @throws InterruptedException
1539      */
1540     public int readAndProcess() throws IOException, InterruptedException {
1541       // Try and read in an int.  If new connection, the int will hold the 'HBas' HEADER.  If it
1542       // does, read in the rest of the connection preamble, the version and the auth method.
1543       // Else it will be length of the data to read (or -1 if a ping).  We catch the integer
1544       // length into the 4-byte this.dataLengthBuffer.
1545       int count = read4Bytes();
1546       if (count < 0 || dataLengthBuffer.remaining() > 0) {
1547         return count;
1548       }
1549 
1550       // If we have not read the connection setup preamble, look to see if that is on the wire.
1551       if (!connectionPreambleRead) {
1552         count = readPreamble();
1553         if (!connectionPreambleRead) {
1554           return count;
1555         }
1556 
1557         count = read4Bytes();
1558         if (count < 0 || dataLengthBuffer.remaining() > 0) {
1559           return count;
1560         }
1561       }
1562 
1563       // We have read a length and we have read the preamble.  It is either the connection header
1564       // or it is a request.
1565       if (data == null) {
1566         dataLengthBuffer.flip();
1567         int dataLength = dataLengthBuffer.getInt();
1568         if (dataLength == RpcClient.PING_CALL_ID) {
1569           if (!useWrap) { //covers the !useSasl too
1570             dataLengthBuffer.clear();
1571             return 0;  //ping message
1572           }
1573         }
1574         if (dataLength < 0) { // A data length of zero is legal.
1575           throw new IllegalArgumentException("Unexpected data length "
1576               + dataLength + "!! from " + getHostAddress());
1577         }
1578         data = ByteBuffer.allocate(dataLength);
1579 
1580         // Increment the rpc count. This counter will be decreased when we write
1581         //  the response.  If we want the connection to be detected as idle properly, we
1582         //  need to keep the inc / dec correct.
1583         incRpcCount();
1584       }
1585 
1586       count = channelRead(channel, data);
1587 
1588       if (count >= 0 && data.remaining() == 0) { // count==0 if dataLength == 0
1589         process();
1590       }
1591 
1592       return count;
1593     }

1595     /**
1596      * Process the data buffer and clean the connection state for the next call.
1597      */
1598     private void process() throws IOException, InterruptedException {
1599       data.flip();
1600       try {
1601         if (skipInitialSaslHandshake) {
1602           skipInitialSaslHandshake = false;
1603           return;
1604         }
1605 
1606         if (useSasl) {
1607           saslReadAndProcess(data.array());
1608         } else {
1609           processOneRpc(data.array());
1610         }
1611 
1612       } finally {
1613         dataLengthBuffer.clear(); // Clean for the next call
1614         data = null; // For the GC
1615       }
1616     }

1637     // Reads the connection header following version
1638     private void processConnectionHeader(byte[] buf) throws IOException {
1639       this.connectionHeader = ConnectionHeader.parseFrom(buf);
1640       String serviceName = connectionHeader.getServiceName();
1641       if (serviceName == null) throw new EmptyServiceNameException();
1642       this.service = getService(services, serviceName);
1643       if (this.service == null) throw new UnknownServiceException(serviceName);
1644       setupCellBlockCodecs(this.connectionHeader);
1645       UserGroupInformation protocolUser = createUser(connectionHeader);
1646       if (!useSasl) {
1647         user = protocolUser;
1648         if (user != null) {
1649           user.setAuthenticationMethod(AuthMethod.SIMPLE.authenticationMethod);
1650         }
1651       } else {
1652         // user is authenticated
1653         user.setAuthenticationMethod(authMethod.authenticationMethod);
1654         //Now we check if this is a proxy user case. If the protocol user is
1655         //different from the 'user', it is a proxy user scenario. However,
1656         //this is not allowed if user authenticated with DIGEST.
1657         if ((protocolUser != null)
1658             && (!protocolUser.getUserName().equals(user.getUserName()))) {
1659           if (authMethod == AuthMethod.DIGEST) {
1660             // Not allowed to doAs if token authentication is used
1661             throw new AccessDeniedException("Authenticated user (" + user
1662                 + ") doesn't match what the client claims to be ("
1663                 + protocolUser + ")");
1664           } else {
1665             // Effective user can be different from authenticated user
1666             // for simple auth or kerberos auth
1667             // The user is the real user. Now we create a proxy user
1668             UserGroupInformation realUser = user;
1669             user = UserGroupInformation.createProxyUser(protocolUser
1670                 .getUserName(), realUser);
1671             // Now the user is a proxy user, set Authentication method Proxy.
1672             user.setAuthenticationMethod(AuthenticationMethod.PROXY);
1673           }
1674         }
1675       }
1676       if (connectionHeader.hasVersionInfo()) {
1677         AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
1678             + " with version info: "
1679             + TextFormat.shortDebugString(connectionHeader.getVersionInfo()));
1680       } else {
1681         AUDITLOG.info("Connection from " + this.hostAddress + " port: " + this.remotePort
1682             + " with unknown version info");
1683       }
1684     }

1710     private void processUnwrappedData(byte[] inBuf) throws IOException,
1711     InterruptedException {
1712       ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(inBuf));
1713       // Read all RPCs contained in the inBuf, even partial ones
1714       while (true) {
1715         int count;
1716         if (unwrappedDataLengthBuffer.remaining() > 0) {
1717           count = channelRead(ch, unwrappedDataLengthBuffer);
1718           if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0)
1719             return;
1720         }
1721 
1722         if (unwrappedData == null) {
1723           unwrappedDataLengthBuffer.flip();
1724           int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
1725 
1726           if (unwrappedDataLength == RpcClient.PING_CALL_ID) {
1727             if (LOG.isDebugEnabled())
1728               LOG.debug("Received ping message");
1729             unwrappedDataLengthBuffer.clear();
1730             continue; // ping message
1731           }
1732           unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
1733         }
1734 
1735         count = channelRead(ch, unwrappedData);
1736         if (count <= 0 || unwrappedData.remaining() > 0)
1737           return;
1738 
1739         if (unwrappedData.remaining() == 0) {
1740           unwrappedDataLengthBuffer.clear();
1741           unwrappedData.flip();
1742           processOneRpc(unwrappedData.array());
1743           unwrappedData = null;
1744         }
1745       }
1746     }

1759         }
1760       }

1763     /**
1764      * @param buf Has the request header and the request param and optionally encoded data buffer
1765      * all in this one array.
1766      * @throws IOException
1767      * @throws InterruptedException
1768      */
1769     protected void processRequest(byte[] buf) throws IOException, InterruptedException {
1770       long totalRequestSize = buf.length;
1771       int offset = 0;
1772       // Here we read in the header.  We avoid having pb
1773       // do its default 4k allocation for CodedInputStream.  We force it to use backing array.
1774       CodedInputStream cis = CodedInputStream.newInstance(buf, offset, buf.length);
1775       int headerSize = cis.readRawVarint32();
1776       offset = cis.getTotalBytesRead();
1777       RequestHeader header = RequestHeader.newBuilder().mergeFrom(buf, offset, headerSize).build();
1778       offset += headerSize;
1779       int id = header.getCallId();
1780       if (LOG.isTraceEnabled()) {
1781         LOG.trace("RequestHeader " + TextFormat.shortDebugString(header) +
1782           " totalRequestSize: " + totalRequestSize + " bytes");
1783       }
1784       // Enforcing the call queue size, this triggers a retry in the client
1785       // This is a bit late to be doing this check - we have already read in the total request.
1786       if ((totalRequestSize + callQueueSize.get()) > maxQueueSize) {
1787         final Call callTooBig =
1788           new Call(id, this.service, null, null, null, null, this,
1789             responder, totalRequestSize, null, null);
1790         ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1791         metrics.exception(CALL_QUEUE_TOO_BIG_EXCEPTION);
1792         setupResponse(responseBuffer, callTooBig, CALL_QUEUE_TOO_BIG_EXCEPTION,
1793             "Call queue is full on " + getListenerAddress() +
1794                 ", is hbase.ipc.server.max.callqueue.size too small?");
1795         responder.doRespond(callTooBig);
1796         return;
1797       }
1798       MethodDescriptor md = null;
1799       Message param = null;
1800       CellScanner cellScanner = null;
1801       try {
1802         if (header.hasRequestParam() && header.getRequestParam()) {
1803           md = this.service.getDescriptorForType().findMethodByName(header.getMethodName());
1804           if (md == null) throw new UnsupportedOperationException(header.getMethodName());
1805           Builder builder = this.service.getRequestPrototype(md).newBuilderForType();
1806           // To read the varint, I need an inputstream; might as well be a CIS.
1807           cis = CodedInputStream.newInstance(buf, offset, buf.length);
1808           int paramSize = cis.readRawVarint32();
1809           offset += cis.getTotalBytesRead();
1810           if (builder != null) {
1811             param = builder.mergeFrom(buf, offset, paramSize).build();
1812           }
1813           offset += paramSize;
1814         }
1815         if (header.hasCellBlockMeta()) {
1816           cellScanner = ipcUtil.createCellScanner(this.codec, this.compressionCodec,
1817             buf, offset, buf.length);
1818         }
1819       } catch (Throwable t) {
1820         String msg = getListenerAddress() + " is unable to read call parameter from client " +
1821             getHostAddress();
1822         LOG.warn(msg, t);
1823 
1824         metrics.exception(t);
1825 
1826         // probably the hbase hadoop version does not match the running hadoop version
1827         if (t instanceof LinkageError) {
1828           t = new DoNotRetryIOException(t);
1829         }
1830         // If the method is not present on the server, do not retry.
1831         if (t instanceof UnsupportedOperationException) {
1832           t = new DoNotRetryIOException(t);
1833         }
1834 
1835         final Call readParamsFailedCall =
1836           new Call(id, this.service, null, null, null, null, this,
1837             responder, totalRequestSize, null, null);
1838         ByteArrayOutputStream responseBuffer = new ByteArrayOutputStream();
1839         setupResponse(responseBuffer, readParamsFailedCall, t,
1840           msg + "; " + t.getMessage());
1841         responder.doRespond(readParamsFailedCall);
1842         return;
1843       }
1844 
1845       TraceInfo traceInfo = header.hasTraceInfo()
1846           ? new TraceInfo(header.getTraceInfo().getTraceId(), header.getTraceInfo().getParentId())
1847           : null;
1848       Call call = new Call(id, this.service, md, header, param, cellScanner, this, responder,
1849               totalRequestSize, traceInfo, RpcServer.getRemoteIp());
1850       scheduler.dispatch(new CallRunner(RpcServer.this, call));
1851     }

1853     private boolean authorizeConnection() throws IOException {
1854       try {
1855         // If auth method is DIGEST, the token was obtained by the
1856         // real user for the effective user, therefore not required to
1857         // authorize real user. doAs is allowed only for simple or kerberos
1858         // authentication
1859         if (user != null && user.getRealUser() != null
1860             && (authMethod != AuthMethod.DIGEST)) {
1861           ProxyUsers.authorize(user, this.getHostAddress(), conf);
1862         }
1863         authorize(user, connectionHeader, getHostInetAddress());
1864         metrics.authorizationSuccess();
1865       } catch (AuthorizationException ae) {
1866         if (LOG.isDebugEnabled()) {
1867           LOG.debug("Connection authorization failed: " + ae.getMessage(), ae);
1868         }
1869         metrics.authorizationFailure();
1870         setupResponse(authFailedResponse, authFailedCall,
1871           new AccessDeniedException(ae), ae.getMessage());
1872         responder.doRespond(authFailedCall);
1873         return false;
1874       }
1875       return true;
1876     }

1895         return null;

2095   /**
2096    * This is a server side method, which is invoked over RPC. On success
2097    * the return response has protobuf response payload. On failure, the
2098    * exception name and the stack trace are returned in the protobuf response.
2099    */
2100   @Override
2101   public Pair<Message, CellScanner> call(BlockingService service, MethodDescriptor md,
2102       Message param, CellScanner cellScanner, long receiveTime, MonitoredRPCHandler status)
2103   throws IOException {
2104     try {
2105       status.setRPC(md.getName(), new Object[]{param}, receiveTime);
2106       // TODO: Review after we add in encoded data blocks.
2107       status.setRPCPacket(param);
2108       status.resume("Servicing call");
2109       //get an instance of the method arg type
2110       long startTime = System.currentTimeMillis();
2111       PayloadCarryingRpcController controller = new PayloadCarryingRpcController(cellScanner);
2112       Message result = service.callBlockingMethod(md, controller, param);
2113       long endTime = System.currentTimeMillis();
2114       int processingTime = (int) (endTime - startTime);
2115       int qTime = (int) (startTime - receiveTime);
2116       int totalTime = (int) (endTime - receiveTime);
2117       if (LOG.isTraceEnabled()) {
2118         LOG.trace(CurCall.get().toString() +
2119             ", response " + TextFormat.shortDebugString(result) +
2120             " queueTime: " + qTime +
2121             " processingTime: " + processingTime +
2122             " totalTime: " + totalTime);
2123       }
2124       metrics.dequeuedCall(qTime);
2125       metrics.processedCall(processingTime);
2126       metrics.totalCall(totalTime);
2127       long responseSize = result.getSerializedSize();
2128       // log any RPC responses that are slower than the configured warn
2129       // response time or larger than configured warning size
2130       boolean tooSlow = (processingTime > warnResponseTime && warnResponseTime > -1);
2131       boolean tooLarge = (responseSize > warnResponseSize && warnResponseSize > -1);
2132       if (tooSlow || tooLarge) {
2133         // when tagging, we let TooLarge trump TooSmall to keep output simple
2134         // note that large responses will often also be slow.
2135         logResponse(new Object[]{param},
2136             md.getName(), md.getName() + "(" + param.getClass().getName() + ")",
2137             (tooLarge ? "TooLarge" : "TooSlow"),
2138             status.getClient(), startTime, processingTime, qTime,
2139             responseSize);
2140       }
2141       return new Pair<Message, CellScanner>(result, controller.cellScanner());
2142     } catch (Throwable e) {
2143       // The above callBlockingMethod will always return a SE.  Strip the SE wrapper before
2144       // putting it on the wire.  Its needed to adhere to the pb Service Interface but we don't
2145       // need to pass it over the wire.
2146       if (e instanceof ServiceException) e = e.getCause();
2147 
2148       // increment the number of requests that were exceptions.
2149       metrics.exception(e);
2150 
2151       if (e instanceof LinkageError) throw new DoNotRetryIOException(e);
2152       if (e instanceof IOException) throw (IOException)e;
2153       LOG.error("Unexpected throwable object ", e);
2154       throw new IOException(e.getMessage(), e);
2155     }
2156   }

2158   /**
2159    * Logs an RPC response to the LOG file, producing valid JSON objects for
2160    * client Operations.
2161    * @param params The parameters received in the call.
2162    * @param methodName The name of the method invoked
2163    * @param call The string representation of the call
2164    * @param tag  The tag that will be used to indicate this event in the log.
2165    * @param clientAddress   The address of the client who made this call.
2166    * @param startTime       The time that the call was initiated, in ms.
2167    * @param processingTime  The duration that the call took to run, in ms.
2168    * @param qTime           The duration that the call spent on the queue
2169    *                        prior to being initiated, in ms.
2170    * @param responseSize    The size in bytes of the response buffer.
2171    */
2172   void logResponse(Object[] params, String methodName, String call, String tag,
2173       String clientAddress, long startTime, int processingTime, int qTime,
2174       long responseSize)
2175           throws IOException {
2176     // base information that is reported regardless of type of call
2177     Map<String, Object> responseInfo = new HashMap<String, Object>();
2178     responseInfo.put("starttimems", startTime);
2179     responseInfo.put("processingtimems", processingTime);
2180     responseInfo.put("queuetimems", qTime);
2181     responseInfo.put("responsesize", responseSize);
2182     responseInfo.put("client", clientAddress);
2183     responseInfo.put("class", server == null? "": server.getClass().getSimpleName());
2184     responseInfo.put("method", methodName);
2185     if (params.length == 2 && server instanceof HRegionServer &&
2186         params[0] instanceof byte[] &&
2187         params[1] instanceof Operation) {
2188       // if the slow process is a query, we want to log its table as well
2189       // as its own fingerprint
2190       TableName tableName = TableName.valueOf(
2191           HRegionInfo.parseRegionName((byte[]) params[0])[0]);
2192       responseInfo.put("table", tableName.getNameAsString());
2193       // annotate the response map with operation details
2194       responseInfo.putAll(((Operation) params[1]).toMap());
2195       // report to the log file
2196       LOG.warn("(operation" + tag + "): " +
2197                MAPPER.writeValueAsString(responseInfo));
2198     } else if (params.length == 1 && server instanceof HRegionServer &&
2199         params[0] instanceof Operation) {
2200       // annotate the response map with operation details
2201       responseInfo.putAll(((Operation) params[0]).toMap());
2202       // report to the log file
2203       LOG.warn("(operation" + tag + "): " +
2204                MAPPER.writeValueAsString(responseInfo));
2205     } else {
2206       // can't get JSON details, so just report call.toString() along with
2207       // a more generic tag.
2208       responseInfo.put("call", call);
2209       LOG.warn("(response" + tag + "): " + MAPPER.writeValueAsString(responseInfo));
2210     }

2290    * When the read or write buffer size is larger than this limit, i/o will be
2291    * done in chunks of this size. Most RPC requests and responses would be
2292    * be smaller.

2294   private static int NIO_BUFFER_LIMIT = 64 * 1024; //should not be more than 64KB.

2296   /**
2297    * This is a wrapper around {@link java.nio.channels.WritableByteChannel#write(java.nio.ByteBuffer)}.
2298    * If the amount of data is large, it writes to channel in smaller chunks.
2299    * This is to avoid jdk from creating many direct buffers as the size of
2300    * buffer increases. This also minimizes extra copies in NIO layer
2301    * as a result of multiple write operations required to write a large
2302    * buffer.
2303    *
2304    * @param channel writable byte channel to write to
2305    * @param bufferChain Chain of buffers to write
2306    * @return number of bytes written
2307    * @throws java.io.IOException e
2308    * @see java.nio.channels.WritableByteChannel#write(java.nio.ByteBuffer)
2309    */
2310   protected long channelWrite(GatheringByteChannel channel, BufferChain bufferChain)
2311   throws IOException {
2312     long count =  bufferChain.write(channel, NIO_BUFFER_LIMIT);
2313     if (count > 0) this.metrics.sentBytes(count);
2314     return count;
2315   }

2317   /**
2318    * This is a wrapper around {@link java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)}.
2319    * If the amount of data is large, it writes to channel in smaller chunks.
2320    * This is to avoid jdk from creating many direct buffers as the size of
2321    * ByteBuffer increases. There should not be any performance degredation.
2322    *
2323    * @param channel writable byte channel to write on
2324    * @param buffer buffer to write
2325    * @return number of bytes written
2326    * @throws java.io.IOException e
2327    * @see java.nio.channels.ReadableByteChannel#read(java.nio.ByteBuffer)
2328    */
2329   protected int channelRead(ReadableByteChannel channel,
2330                                    ByteBuffer buffer) throws IOException {
2331 
2332     int count = (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
2333            channel.read(buffer) : channelIO(channel, null, buffer);
2334     if (count > 0) {
2335       metrics.receivedBytes(count);
2336     }
2337     return count;
2338   }

2340   /**
2341    * Helper for {@link #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)}
2342    * and {@link #channelWrite(GatheringByteChannel, BufferChain)}. Only
2343    * one of readCh or writeCh should be non-null.
2344    *
2345    * @param readCh read channel
2346    * @param writeCh write channel
2347    * @param buf buffer to read or write into/out of
2348    * @return bytes written
2349    * @throws java.io.IOException e
2350    * @see #channelRead(java.nio.channels.ReadableByteChannel, java.nio.ByteBuffer)
2351    * @see #channelWrite(GatheringByteChannel, BufferChain)
2352    */
2353   private static int channelIO(ReadableByteChannel readCh,
2354                                WritableByteChannel writeCh,
2355                                ByteBuffer buf) throws IOException {
2356 
2357     int originalLimit = buf.limit();
2358     int initialRemaining = buf.remaining();
2359     int ret = 0;
2360 
2361     while (buf.remaining() > 0) {
2362       try {
2363         int ioSize = Math.min(buf.remaining(), NIO_BUFFER_LIMIT);
2364         buf.limit(buf.position() + ioSize);
2365 
2366         ret = (readCh == null) ? writeCh.write(buf) : readCh.read(buf);
2367 
2368         if (ret < ioSize) {
2369           break;
2370         }
2371 
2372       } finally {
2373         buf.limit(originalLimit);
2374       }
2375     }
2376 
2377     int nBytes = initialRemaining - buf.remaining();
2378     return (nBytes > 0) ? nBytes : ret;
2379   }
