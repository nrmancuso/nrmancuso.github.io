78  /**
79   * <p>An instance of a bug pattern. A BugInstance consists of several parts:</p>
80   * <ul>
81   * <li>the type, which is a string indicating what kind of bug it is; used as a
82   * key for the FindBugsMessages resource bundle</li>
83   * <li>the priority; how likely this instance is to actually be a bug</li>
84   * <li>a list of <em>annotations</em></li>
85   * </ul>
86   * <p>
87   * The annotations describe classes, methods, fields, source locations, and
88   * other relevant context information about the bug instance. Every BugInstance
89   * must have at least one ClassAnnotation, which describes the class in which
90   * the instance was found. This is the "primary class annotation".
91   * </p>
92   * <p>
93   * BugInstance objects are built up by calling a string of <code>add</code>
94   * methods. (These methods all "return this", so they can be chained). Some of
95   * the add methods are specialized to get information automatically from a
96   * BetterVisitor or DismantleBytecode object.</p>
97   *
98   * @author David Hovemeyer
99   * @see BugAnnotation
100  */

164      * @param type
165      *            the bug type

169     public BugInstance(String type, int priority) {
170         this.type = type.intern();
171         this.priority = priority;
172         lastVersion = -1;
173         annotationList = new ArrayList<>(4);
174         cachedHashCode = INVALID_HASH_CODE;
175 
176         BugPattern p = DetectorFactoryCollection.instance().lookupBugPattern(type);
177         if (p == null) {
178             if ( missingBugTypes.add(type)) {
179                 String msg = "Can't find definition of bug type " + type;
180                 AnalysisContext.logError(msg, new NoSuchBugPattern(type));
181             }
182         } else {
183             this.priority += p.getPriorityAdjustment();
184         }
185         if (adjustExperimental && isExperimental()) {
186             this.priority = Priorities.EXP_PRIORITY;
187         }
188         boundPriority();
189     }

205     @Override
206     public Object clone() {
207         BugInstance dup;
208         try {
209             dup = (BugInstance) super.clone();
210 
211             // Do deep copying of mutable objects
212             for (int i = 0; i < dup.annotationList.size(); ++i) {
213                 dup.annotationList.set(i, (BugAnnotation) dup.annotationList.get(i).clone());
214             }
215             dup.propertyListHead = dup.propertyListTail = null;
216             for (Iterator<BugProperty> i = propertyIterator(); i.hasNext();) {
217                 dup.addProperty((BugProperty) i.next().clone());
218             }
219 
220             return dup;
221         } catch (CloneNotSupportedException e) {
222             throw new AssertionError(e);
223         }
224     }

250     public void adjustForDetector(String detectorName) {
251         DetectorFactory factory = DetectorFactoryCollection.instance().getFactoryByClassName(detectorName);
252         detectorFactory = factory;
253         if (factory != null) {
254             this.priority += factory.getPriorityAdjustment();
255             boundPriority();
256             BugPattern bugPattern = getBugPattern();
257             if (SystemProperties.ASSERTIONS_ENABLED && !"EXPERIMENTAL".equals(bugPattern.getCategory())
258                     && !factory.getReportedBugPatterns().contains(bugPattern)) {
259                 AnalysisContext.logError(factory.getShortName() + " doesn't note that it reports "
260                         + bugPattern + " in category " + bugPattern.getCategory());
261             }
262         }
263     }

461      * Find the first BugAnnotation in the list of annotations that is the same
462      * type or a subtype as the given Class parameter.
463      *
464      * @param cls
465      *            the Class parameter
466      * @return the first matching BugAnnotation of the given type, or null if
467      *         there is no such BugAnnotation
468      */
469     @CheckForNull
470     private <T extends BugAnnotation> T findPrimaryAnnotationOfType(Class<T> cls) {
471         T firstMatch = null;
472         for (Iterator<BugAnnotation> i = annotationIterator(); i.hasNext();) {
473             BugAnnotation annotation = i.next();
474             if (cls.isAssignableFrom(annotation.getClass())) {
475                 if (annotation.getDescription().endsWith("DEFAULT")) {
476                     return cls.cast(annotation);
477                 } else if (firstMatch == null) {
478                     firstMatch = cls.cast(annotation);
479                 }
480             }
481         }
482         return firstMatch;
483     }

494     /**
495      * Get the primary source line annotation. There is guaranteed to be one
496      * (unless some Detector constructed an invalid BugInstance).
497      *
498      * @return the source line annotation
499      */
500     @Nonnull
501     public SourceLineAnnotation getPrimarySourceLineAnnotation() {
502         // Highest priority: return the first top level source line annotation
503         for (BugAnnotation annotation : annotationList) {
504             if (annotation instanceof SourceLineAnnotation
505                     && SourceLineAnnotation.DEFAULT_ROLE.equals(annotation.getDescription())
506                     && !((SourceLineAnnotation) annotation).isUnknown()) {
507                 return (SourceLineAnnotation) annotation;
508             }
509         }
510 
511         for (BugAnnotation annotation : annotationList) {
512             if (annotation instanceof SourceLineAnnotation && !((SourceLineAnnotation) annotation).isUnknown()) {
513                 return (SourceLineAnnotation) annotation;
514             }
515         }
516         // Next: Try primary method, primary field, primary class
517         SourceLineAnnotation srcLine;
518         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryMethod())) != null) {
519             return srcLine;
520         }
521         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryField())) != null) {
522             return srcLine;
523         }
524         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryClass())) != null) {
525             return srcLine;
526         }
527 
528         // Last resort: throw exception
529         throw new IllegalStateException("BugInstance for " + getType()
530                 + " must contain at least one class, method, or field annotation");
531     }

621     private class BugPropertyIterator implements Iterator<BugProperty> {
622         private BugProperty prev, cur;
623 
624         private boolean removed;
625 
626         @Override
627         public boolean hasNext() {
628             return findNext() != null;
629         }
630 
631         @Override
632         public BugProperty next() {
633             BugProperty next = findNext();
634             if (next == null) {
635                 throw new NoSuchElementException();
636             }
637             prev = cur;
638             cur = next;
639             removed = false;
640             return cur;
641         }
642 
643         @Override
644         public void remove() {
645             if (cur == null || removed) {
646                 throw new IllegalStateException();
647             }
648             if (prev == null) {
649                 propertyListHead = cur.getNext();
650             } else {
651                 prev.setNext(cur.getNext());
652             }
653             if (cur == propertyListTail) {
654                 propertyListTail = prev;
655             }
656             removed = true;
657         }
658 
659         private BugProperty findNext() {
660             return cur == null ? propertyListHead : cur.getNext();
661         }
662     }

723     /**
724      * Look up a property by name.

727      *            name of the property to look for
728      * @return the BugProperty with the given name, or null if the property has
729      *         not been set
730      */
731     public BugProperty lookupProperty(String name) {
732         BugProperty prop = propertyListHead;
733 
734         while (prop != null) {
735             if (prop.getName().equals(name)) {
736                 break;
737             }
738             prop = prop.getNext();
739         }
740 
741         return prop;
742     }

744     /**
745      * Delete property with given name.
746      *
747      * @param name
748      *            name of the property to delete
749      * @return true if a property with that name was deleted, or false if there
750      *         is no such property
751      */
752     public boolean deleteProperty(String name) {
753         BugProperty prev = null;
754         BugProperty prop = propertyListHead;
755 
756         while (prop != null) {
757             if (prop.getName().equals(name)) {
758                 break;
759             }
760             prev = prop;
761             prop = prop.getNext();
762         }
763 
764         if (prop != null) {
765             if (prev != null) {
766                 // Deleted node in interior or at tail of list
767                 prev.setNext(prop.getNext());
768             } else {
769                 // Deleted node at head of list
770                 propertyListHead = prop.getNext();
771             }
772 
773             if (prop.getNext() == null) {
774                 // Deleted node at end of list
775                 propertyListTail = prev;
776             }
777 
778             return true;
779         } else {
780             // No such property
781             return false;
782         }
783     }

785     private void addProperty(BugProperty prop) {
786         if (propertyListTail != null) {
787             propertyListTail.setNext(prop);
788             propertyListTail = prop;
789         } else {
790             propertyListHead = propertyListTail = prop;
791         }
792         prop.setNext(null);
793     }
