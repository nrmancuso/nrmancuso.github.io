36  /**
37   * Generated tests for {@link Monitor}.
38   *
39   * <p>This test class generates all of its own test cases in the {@link #suite()} method. Every
40   * {@code enterXxx}, {@code tryEnterXxx}, and {@code waitForXxx} method of the {@code Monitor} class
41   * is analyzed reflectively to determine appropriate test cases based on its signature. Additional
42   * ad hoc test cases can be found in {@link SupplementalMonitorTest}.
43   *
44   * @author Justin T. Sampson
45   */
46  
47  public class GeneratedMonitorTest extends TestCase {
48  
49    public static TestSuite suite() {
50      TestSuite suite = new TestSuite();
51  
52      Method[] methods = Monitor.class.getMethods();
53      sortMethods(methods);
54      for (Method method : methods) {
55        if (isAnyEnter(method) || isWaitFor(method)) {
56          validateMethod(method);
57          addTests(suite, method);
58        }
59      }
60  
61      assertEquals(548, suite.testCount());
62  
63      return suite;
64    }

70    private static final long EXPECTED_HANG_DELAY_MILLIS = 75;

77    /**
78     * Various scenarios to be generated for each method under test. The actual scenario generation
79     * (determining which scenarios are applicable to which methods and what the outcome should be)
80     * takes place in {@link #addTests(TestSuite, Method)}.

82    private enum Scenario {
83      SATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
84      UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,

86      SATISFIED_UNOCCUPIED_AND_INTERRUPTED_BEFORE_ENTERING,

88      SATISFIED_BEFORE_WAITING,
89      SATISFIED_WHILE_WAITING,
90      SATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
91      UNSATISFIED_BEFORE_AND_WHILE_WAITING,
92      UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING;
93  
94      @Override
95      public String toString() {
96        return CaseFormat.UPPER_UNDERSCORE.to(CaseFormat.UPPER_CAMEL, name());
97      }

100   /** Timeout values to combine with each {@link Scenario}. */
101   private enum Timeout {
102     MIN(Long.MIN_VALUE, "-oo"),
103     MINUS_SMALL(-SMALL_TIMEOUT_MILLIS, "-" + SMALL_TIMEOUT_MILLIS + "ms"),
104     ZERO(0L, "0ms"),
105     SMALL(SMALL_TIMEOUT_MILLIS, SMALL_TIMEOUT_MILLIS + "ms"),
106     LARGE(UNEXPECTED_HANG_DELAY_MILLIS * 2, (2 * UNEXPECTED_HANG_DELAY_MILLIS) + "ms"),
107     MAX(Long.MAX_VALUE, "+oo");
108 
109     final long millis;
110     final String label;
111 
112     Timeout(long millis, String label) {
113       this.millis = millis;
114       this.label = label;
115     }
116 
117     @Override
118     public String toString() {
119       return label;
120     }
121   }

123   /** Convenient subsets of the {@link Timeout} enumeration for specifying scenario outcomes. */
124   private enum TimeoutsToUse {
125     ANY(Timeout.values()),
126     PAST(Timeout.MIN, Timeout.MINUS_SMALL, Timeout.ZERO),
127     FUTURE(Timeout.SMALL, Timeout.MAX),
128     SMALL(Timeout.SMALL),
129     FINITE(Timeout.MIN, Timeout.MINUS_SMALL, Timeout.ZERO, Timeout.SMALL),
130     INFINITE(Timeout.LARGE, Timeout.MAX);
131 
132     final ImmutableList<Timeout> timeouts;
133 
134     TimeoutsToUse(Timeout... timeouts) {
135       this.timeouts = ImmutableList.copyOf(timeouts);
136     }
137   }

140   private enum Outcome {
141 
142     /** The method returned normally and is either void or returned true. */
143     SUCCESS,
144 
145     /** The method returned false. */
146     FAILURE,

160   /** Identifies all enterXxx and tryEnterXxx methods. */
161   private static boolean isAnyEnter(Method method) {
162     return method.getName().startsWith("enter") || method.getName().startsWith("tryEnter");
163   }
164 
165   /** Identifies just tryEnterXxx methods (a subset of {@link #isAnyEnter}), which never block. */
166   private static boolean isTryEnter(Method method) {
167     return method.getName().startsWith("tryEnter");
168   }
169 
170   /**
171    * Identifies just enterIfXxx methods (a subset of {@link #isAnyEnter}), which are mostly like the
172    * enterXxx methods but behave like tryEnterXxx in some scenarios.
173    */
174   private static boolean isEnterIf(Method method) {
175     return method.getName().startsWith("enterIf");
176   }
177 
178   /** Identifies all waitForXxx methods, which must be called while occupying the monitor. */
179   private static boolean isWaitFor(Method method) {
180     return method.getName().startsWith("waitFor");
181   }

183   /** Determines whether the given method takes a Guard as its first parameter. */
184   private static boolean isGuarded(Method method) {
185     Class<?>[] parameterTypes = method.getParameterTypes();
186     return parameterTypes.length >= 1 && parameterTypes[0] == Monitor.Guard.class;
187   }
188 
189   /** Determines whether the given method takes a time and unit as its last two parameters. */
190   private static boolean isTimed(Method method) {
191     Class<?>[] parameterTypes = method.getParameterTypes();
192     return parameterTypes.length >= 2
193         && parameterTypes[parameterTypes.length - 2] == long.class
194         && parameterTypes[parameterTypes.length - 1] == TimeUnit.class;
195   }
196 
197   /** Determines whether the given method returns a boolean value. */
198   private static boolean isBoolean(Method method) {
199     return method.getReturnType() == boolean.class;
200   }
201 
202   /** Determines whether the given method can throw InterruptedException. */
203   private static boolean isInterruptible(Method method) {
204     return Arrays.asList(method.getExceptionTypes()).contains(InterruptedException.class);
205   }

207   /** Sorts the given methods primarily by name and secondarily by number of parameters. */
208   private static void sortMethods(Method[] methods) {
209     Arrays.sort(
210         methods,
211         new Comparator<Method>() {
212           @Override
213           public int compare(Method m1, Method m2) {
214             int nameComparison = m1.getName().compareTo(m2.getName());
215             if (nameComparison != 0) {
216               return nameComparison;
217             } else {
218               return Ints.compare(m1.getParameterTypes().length, m2.getParameterTypes().length);
219             }
220           }
221         });
222   }

224   /** Validates that the given method's signature meets all of our assumptions. */
225   private static void validateMethod(Method method) {
226     String desc = method.toString();
227 
228     assertTrue(desc, isAnyEnter(method) || isWaitFor(method));
229 
230     switch (method.getParameterTypes().length) {
231       case 0:
232         assertFalse(desc, isGuarded(method));
233         assertFalse(desc, isTimed(method));
234         break;
235       case 1:
236         assertTrue(desc, isGuarded(method));
237         assertFalse(desc, isTimed(method));
238         break;
239       case 2:
240         assertFalse(desc, isGuarded(method));
241         assertTrue(desc, isTimed(method));
242         break;
243       case 3:
244         assertTrue(desc, isGuarded(method));
245         assertTrue(desc, isTimed(method));
246         break;
247       default:
248         fail(desc);
249     }
250 
251     if (method.getReturnType() == void.class) {
252       assertFalse(desc, isBoolean(method));
253     } else {
254       assertTrue(desc, isBoolean(method));
255     }
256 
257     switch (method.getExceptionTypes().length) {
258       case 0:
259         assertFalse(desc, isInterruptible(method));
260         break;
261       case 1:
262         assertTrue(desc, isInterruptible(method));
263         break;
264       default:
265         fail(desc);
266     }
267 
268     if (isEnterIf(method)) {
269       assertTrue(desc, isGuarded(method));
270       assertTrue(desc, isBoolean(method));
271     } else if (isTryEnter(method)) {
272       assertFalse(desc, isTimed(method));
273       assertTrue(desc, isBoolean(method));
274       assertFalse(desc, isInterruptible(method));
275     } else if (isWaitFor(method)) {
276       assertTrue(desc, isGuarded(method));
277       assertEquals(desc, isTimed(method), isBoolean(method));
278     } else { // any other enterXxx method
279       assertEquals(desc, isTimed(method), isBoolean(method));
280     }
281   }

283   /** Generates all test cases appropriate for the given method. */
284   private static void addTests(TestSuite suite, Method method) {
285     if (isGuarded(method)) {
286       for (boolean fair1 : new boolean[] {true, false}) {
287         for (boolean fair2 : new boolean[] {true, false}) {
288           suite.addTest(generateGuardWithWrongMonitorTestCase(method, fair1, fair2));
289         }
290       }
291     }
292     if (isAnyEnter(method)) {
293       addTests(
294           suite,
295           method,
296           Scenario.SATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
297           TimeoutsToUse.ANY,
298           Outcome.SUCCESS);
299       addTests(
300           suite,
301           method,
302           Scenario.UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
303           TimeoutsToUse.FINITE,
304           isGuarded(method)
305               ? (isBoolean(method) ? Outcome.FAILURE : Outcome.HANG)
306               : Outcome.SUCCESS);
307       addTests(
308           suite,
309           method,
310           Scenario.UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING,
311           TimeoutsToUse.INFINITE,
312           isGuarded(method)
313               ? (isTryEnter(method) || isEnterIf(method) ? Outcome.FAILURE : Outcome.HANG)
314               : Outcome.SUCCESS);
315       addTests(
316           suite,
317           method,
318           Scenario.SATISFIED_AND_OCCUPIED_BEFORE_ENTERING,
319           TimeoutsToUse.FINITE,
320           isBoolean(method) ? Outcome.FAILURE : Outcome.HANG);
321       addTests(
322           suite,
323           method,
324           Scenario.SATISFIED_AND_OCCUPIED_BEFORE_ENTERING,
325           TimeoutsToUse.INFINITE,
326           isGuarded(method) ? Outcome.HANG : (isTryEnter(method) ? Outcome.FAILURE : Outcome.HANG));
327       addTests(
328           suite,
329           method,
330           Scenario.SATISFIED_UNOCCUPIED_AND_INTERRUPTED_BEFORE_ENTERING,
331           TimeoutsToUse.ANY,
332           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.SUCCESS);
333     } else { // any waitForXxx method
334       suite.addTest(generateWaitForWhenNotOccupyingTestCase(method, true));
335       suite.addTest(generateWaitForWhenNotOccupyingTestCase(method, false));
336       addTests(
337           suite, method, Scenario.SATISFIED_BEFORE_WAITING, TimeoutsToUse.ANY, Outcome.SUCCESS);
338       addTests(
339           suite, method, Scenario.SATISFIED_WHILE_WAITING, TimeoutsToUse.INFINITE, Outcome.SUCCESS);
340       addTests(
341           suite, method, Scenario.SATISFIED_WHILE_WAITING, TimeoutsToUse.PAST, Outcome.FAILURE);
342       addTests(
343           suite,
344           method,
345           Scenario.SATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
346           TimeoutsToUse.ANY,
347           Outcome.SUCCESS);
348       addTests(
349           suite,
350           method,
351           Scenario.UNSATISFIED_BEFORE_AND_WHILE_WAITING,
352           TimeoutsToUse.FINITE,
353           Outcome.FAILURE);
354       addTests(
355           suite,
356           method,
357           Scenario.UNSATISFIED_BEFORE_AND_WHILE_WAITING,
358           TimeoutsToUse.INFINITE,
359           Outcome.HANG);
360       addTests(
361           suite,
362           method,
363           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
364           TimeoutsToUse.PAST,
365           // prefer responding to interrupt over timing out
366           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.FAILURE);
367       addTests(
368           suite,
369           method,
370           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
371           TimeoutsToUse.SMALL,
372           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.FAILURE);
373       addTests(
374           suite,
375           method,
376           Scenario.UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING,
377           TimeoutsToUse.INFINITE,
378           isInterruptible(method) ? Outcome.INTERRUPT : Outcome.HANG);
379     }
380   }

382   /**
383    * Generates test cases for the given combination of scenario and timeouts. For methods that take
384    * an explicit timeout value, all of the given timeoutsToUse result in individual test cases. For
385    * methods that do not take an explicit timeout value, a single test case is generated only if the
386    * implicit timeout of that method matches the given timeoutsToUse. For example, enter() is
387    * treated like enter(MAX, MILLIS) and tryEnter() is treated like enter(0, MILLIS).
388    */
389   private static void addTests(
390       TestSuite suite,
391       Method method,
392       Scenario scenario,
393       TimeoutsToUse timeoutsToUse,
394       Outcome expectedOutcome) {
395     for (boolean fair : new boolean[] {true, false}) {
396       if (isTimed(method)) {
397         for (Timeout timeout : timeoutsToUse.timeouts) {
398           suite.addTest(new GeneratedMonitorTest(method, scenario, fair, timeout, expectedOutcome));
399         }
400       } else {
401         Timeout implicitTimeout = (isTryEnter(method) ? Timeout.ZERO : Timeout.MAX);
402         if (timeoutsToUse.timeouts.contains(implicitTimeout)) {
403           suite.addTest(new GeneratedMonitorTest(method, scenario, fair, null, expectedOutcome));
404         }
405       }
406     }
407   }

409   /** A guard that encapsulates a simple, mutable boolean flag. */
410   static class FlagGuard extends Monitor.Guard {
411 
412     private boolean satisfied;
413 
414     protected FlagGuard(Monitor monitor) {
415       super(monitor);
416     }
417 
418     @Override
419     public boolean isSatisfied() {
420       return satisfied;
421     }
422 
423     public void setSatisfied(boolean satisfied) {
424       this.satisfied = satisfied;
425     }
426   }

428   private final Method method;
429   private final Scenario scenario;
430   private final Timeout timeout;
431   private final Outcome expectedOutcome;
432   private final Monitor monitor;
433   private final FlagGuard guard;
434   private final CountDownLatch tearDownLatch;
435   private final CountDownLatch doingCallLatch;
436   private final CountDownLatch callCompletedLatch;
437 
438   private GeneratedMonitorTest(
439       Method method, Scenario scenario, boolean fair, Timeout timeout, Outcome expectedOutcome) {
440     super(nameFor(method, scenario, fair, timeout, expectedOutcome));
441     this.method = method;
442     this.scenario = scenario;
443     this.timeout = timeout;
444     this.expectedOutcome = expectedOutcome;
445     this.monitor = new Monitor(fair);
446     this.guard = new FlagGuard(monitor);
447     this.tearDownLatch = new CountDownLatch(1);
448     this.doingCallLatch = new CountDownLatch(1);
449     this.callCompletedLatch = new CountDownLatch(1);
450   }

452   private static String nameFor(
453       Method method, Scenario scenario, boolean fair, Timeout timeout, Outcome expectedOutcome) {
454     return String.format(
455         Locale.ROOT,
456         "%s%s(%s)/%s->%s",
457         method.getName(),
458         fair ? "(fair)" : "(nonfair)",
459         (timeout == null) ? "untimed" : timeout,
460         scenario,
461         expectedOutcome);
462   }

464   @Override
465   protected void runTest() throws Throwable {
466     final Runnable runChosenTest =
467         new Runnable() {
468           @Override
469           public void run() {
470             runChosenTest();
471           }
472         };
473     final FutureTask<Void> task = new FutureTask<>(runChosenTest, null);
474     startThread(
475         new Runnable() {
476           @Override
477           public void run() {
478             task.run();
479           }
480         });
481     awaitUninterruptibly(doingCallLatch);
482     long hangDelayMillis =
483         (expectedOutcome == Outcome.HANG)
484             ? EXPECTED_HANG_DELAY_MILLIS
485             : UNEXPECTED_HANG_DELAY_MILLIS;
486     boolean hung =
487         !awaitUninterruptibly(callCompletedLatch, hangDelayMillis, TimeUnit.MILLISECONDS);
488     if (hung) {
489       assertEquals(expectedOutcome, Outcome.HANG);
490     } else {
491       assertNull(task.get(UNEXPECTED_HANG_DELAY_MILLIS, TimeUnit.MILLISECONDS));
492     }
493   }

495   @Override
496   protected void tearDown() throws Exception {
497     // We don't want to leave stray threads running after each test. At this point, every thread
498     // launched by this test is either:
499     //
500     // (a) Blocked attempting to enter the monitor.
501     // (b) Waiting for the single guard to become satisfied.
502     // (c) Occupying the monitor and awaiting the tearDownLatch.
503     //
504     // Except for (c), every thread should occupy the monitor very briefly, and every thread leaves
505     // the monitor with the guard satisfied. Therefore as soon as tearDownLatch is triggered, we
506     // should be able to enter the monitor, and then we set the guard to satisfied for the benefit
507     // of any remaining waiting threads.
508 
509     tearDownLatch.countDown();
510     assertTrue(
511         "Monitor still occupied in tearDown()",
512         monitor.enter(UNEXPECTED_HANG_DELAY_MILLIS, TimeUnit.MILLISECONDS));
513     try {
514       guard.setSatisfied(true);
515     } finally {
516       monitor.leave();
517     }
518   }
519 
520   private void runChosenTest() {
521     if (isAnyEnter(method)) {
522       runEnterTest();
523     } else {
524       runWaitTest();
525     }
526   }

528   private void runEnterTest() {
529     assertFalse(Thread.currentThread().isInterrupted());
530     assertFalse(monitor.isOccupiedByCurrentThread());
531 
532     doEnterScenarioSetUp();
533 
534     boolean interruptedBeforeCall = Thread.currentThread().isInterrupted();
535     Outcome actualOutcome = doCall();
536     boolean occupiedAfterCall = monitor.isOccupiedByCurrentThread();
537     boolean interruptedAfterCall = Thread.currentThread().isInterrupted();
538 
539     if (occupiedAfterCall) {
540       guard.setSatisfied(true);
541       monitor.leave();
542       assertFalse(monitor.isOccupiedByCurrentThread());
543     }
544 
545     assertEquals(expectedOutcome, actualOutcome);
546     assertEquals(expectedOutcome == Outcome.SUCCESS, occupiedAfterCall);
547     assertEquals(
548         interruptedBeforeCall && expectedOutcome != Outcome.INTERRUPT, interruptedAfterCall);
549   }
550 
551   private void doEnterScenarioSetUp() {
552     switch (scenario) {
553       case SATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING:
554         enterSatisfyGuardAndLeaveInCurrentThread();
555         break;
556       case UNSATISFIED_AND_UNOCCUPIED_BEFORE_ENTERING:
557         break;
558       case SATISFIED_AND_OCCUPIED_BEFORE_ENTERING:
559         enterSatisfyGuardAndLeaveInCurrentThread();
560         enterAndRemainOccupyingInAnotherThread();
561         break;
562       case SATISFIED_UNOCCUPIED_AND_INTERRUPTED_BEFORE_ENTERING:
563         enterSatisfyGuardAndLeaveInCurrentThread();
564         Thread.currentThread().interrupt();
565         break;
566       default:
567         throw new AssertionError("unsupported scenario: " + scenario);
568     }
569   }
570 
571   private void runWaitTest() {
572     assertFalse(Thread.currentThread().isInterrupted());
573     assertFalse(monitor.isOccupiedByCurrentThread());
574     monitor.enter();
575     try {
576       assertTrue(monitor.isOccupiedByCurrentThread());
577 
578       doWaitScenarioSetUp();
579 
580       boolean interruptedBeforeCall = Thread.currentThread().isInterrupted();
581       Outcome actualOutcome = doCall();
582       boolean occupiedAfterCall = monitor.isOccupiedByCurrentThread();
583       boolean interruptedAfterCall = Thread.currentThread().isInterrupted();
584 
585       assertEquals(expectedOutcome, actualOutcome);
586       assertTrue(occupiedAfterCall);
587       assertEquals(
588           interruptedBeforeCall && expectedOutcome != Outcome.INTERRUPT, interruptedAfterCall);
589     } finally {
590       guard.setSatisfied(true);
591       monitor.leave();
592       assertFalse(monitor.isOccupiedByCurrentThread());
593     }
594   }
595 
596   private void doWaitScenarioSetUp() {
597     switch (scenario) {
598       case SATISFIED_BEFORE_WAITING:
599         guard.setSatisfied(true);
600         break;
601       case SATISFIED_WHILE_WAITING:
602         guard.setSatisfied(false);
603         enterSatisfyGuardAndLeaveInAnotherThread(); // enter blocks until we call waitFor
604         break;
605       case UNSATISFIED_BEFORE_AND_WHILE_WAITING:
606         guard.setSatisfied(false);
607         break;
608       case SATISFIED_AND_INTERRUPTED_BEFORE_WAITING:
609         guard.setSatisfied(true);
610         Thread.currentThread().interrupt();
611         break;
612       case UNSATISFIED_AND_INTERRUPTED_BEFORE_WAITING:
613         guard.setSatisfied(false);
614         Thread.currentThread().interrupt();
615         break;
616       default:
617         throw new AssertionError("unsupported scenario: " + scenario);
618     }
619   }

621   private Outcome doCall() {
622     boolean guarded = isGuarded(method);
623     boolean timed = isTimed(method);
624     Object[] arguments = new Object[(guarded ? 1 : 0) + (timed ? 2 : 0)];
625     if (guarded) {
626       arguments[0] = guard;
627     }
628     if (timed) {
629       arguments[arguments.length - 2] = timeout.millis;
630       arguments[arguments.length - 1] = TimeUnit.MILLISECONDS;
631     }
632     try {
633       Object result;
634       doingCallLatch.countDown();
635       try {
636         result = method.invoke(monitor, arguments);
637       } finally {
638         callCompletedLatch.countDown();
639       }
640       if (result == null) {
641         return Outcome.SUCCESS;
642       } else if ((Boolean) result) {
643         return Outcome.SUCCESS;
644       } else {
645         return Outcome.FAILURE;
646       }
647     } catch (InvocationTargetException targetException) {
648       Throwable actualException = targetException.getTargetException();
649       if (actualException instanceof InterruptedException) {
650         return Outcome.INTERRUPT;
651       } else {
652         throw newAssertionError("unexpected exception", targetException);
653       }
654     } catch (IllegalAccessException e) {
655       throw newAssertionError("unexpected exception", e);
656     }
657   }
658 
659   private void enterSatisfyGuardAndLeaveInCurrentThread() {
660     monitor.enter();
661     try {
662       guard.setSatisfied(true);
663     } finally {
664       monitor.leave();
665     }
666   }
667 
668   private void enterSatisfyGuardAndLeaveInAnotherThread() {
669     final CountDownLatch startedLatch = new CountDownLatch(1);
670     startThread(
671         new Runnable() {
672           @Override
673           public void run() {
674             startedLatch.countDown();
675             enterSatisfyGuardAndLeaveInCurrentThread();
676           }
677         });
678     awaitUninterruptibly(startedLatch);
679   }

681   private void enterAndRemainOccupyingInAnotherThread() {
682     final CountDownLatch enteredLatch = new CountDownLatch(1);
683     startThread(
684         new Runnable() {
685           @Override
686           public void run() {
687             monitor.enter();
688             try {
689               enteredLatch.countDown();
690               awaitUninterruptibly(tearDownLatch);
691               guard.setSatisfied(true);
692             } finally {
693               monitor.leave();
694             }
695           }
696         });
697     awaitUninterruptibly(enteredLatch);
698   }
699 
700   @CanIgnoreReturnValue
701   static Thread startThread(Runnable runnable) {
702     Thread thread = new Thread(runnable);
703     thread.setDaemon(true);
704     thread.start();
705     return thread;
706   }

708   /**
709    * Generates a test case verifying that calling any enterXxx, tryEnterXxx, or waitForXxx method
710    * with a guard that doesn't match the monitor produces an IllegalMonitorStateException.
711    */
712   private static TestCase generateGuardWithWrongMonitorTestCase(
713       final Method method, final boolean fair1, final boolean fair2) {
714     final boolean timed = isTimed(method); // Not going to bother with all timeouts, just 0ms.
715     return new TestCase(method.getName() + (timed ? "(0ms)" : "()") + "/WrongMonitor->IMSE") {
716       @Override
717       protected void runTest() throws Throwable {
718         Monitor monitor1 = new Monitor(fair1);
719         Monitor monitor2 = new Monitor(fair2);
720         FlagGuard guard = new FlagGuard(monitor2);
721         Object[] arguments =
722             (timed ? new Object[] {guard, 0L, TimeUnit.MILLISECONDS} : new Object[] {guard});
723         boolean occupyMonitor = isWaitFor(method);
724         if (occupyMonitor) {
725           // If we don't already occupy the monitor, we'll get an IMSE regardless of the guard (see
726           // generateWaitForWhenNotOccupyingTestCase).
727           monitor1.enter();
728         }
729         try {
730           method.invoke(monitor1, arguments);
731           fail("expected IllegalMonitorStateException");
732         } catch (InvocationTargetException e) {
733           assertEquals(IllegalMonitorStateException.class, e.getTargetException().getClass());
734         } finally {
735           if (occupyMonitor) {
736             monitor1.leave();
737           }
738         }
739       }
740     };
741   }
