19  import com.google.gwt.user.client.rpc.SerializationException;
20  import com.google.gwt.user.client.rpc.SerializationStreamReader;
21  import com.google.gwt.user.client.rpc.SerializationStreamWriter;
22  import java.util.Comparator;
23  
24  /**
25   * This class implements the GWT serialization of {@link TreeBasedTable}.
26   *
27   * @author Hayward Chan
28   */
29  public class TreeBasedTable_CustomFieldSerializer {
30    public static void deserialize(SerializationStreamReader reader, TreeBasedTable<?, ?, ?> table) {}
31  
32    public static TreeBasedTable<Object, Object, Object> instantiate(SerializationStreamReader reader)
33        throws SerializationException {
34      checkGwtRpcEnabled();
35      @SuppressWarnings("unchecked") // The comparator isn't used statically.
36      Comparator<Object> rowComparator = (Comparator<Object>) reader.readObject();
37      @SuppressWarnings("unchecked") // The comparator isn't used statically.
38      Comparator<Object> columnComparator = (Comparator<Object>) reader.readObject();
39  
40      TreeBasedTable<Object, Object, Object> table =
41          TreeBasedTable.create(rowComparator, columnComparator);
42      return Table_CustomFieldSerializerBase.populate(reader, table);
43    }
44  
45    public static void serialize(SerializationStreamWriter writer, TreeBasedTable<?, ?, ?> table)
46        throws SerializationException {
47      checkGwtRpcEnabled();
48      writer.writeObject(table.rowComparator());
49      writer.writeObject(table.columnComparator());
50      Table_CustomFieldSerializerBase.serialize(writer, table);
51    }
52  }
