88          Multimaps.toMultimap(Entry::getKey, Entry::getValue, TreeMultimap::create);

90          Equivalence.equals()
91              .onResultOf((Multimap<?, ?> mm) -> ImmutableList.copyOf(mm.asMap().entrySet()))

319   /**
320    * The supplied multimap will be mutated and an unmodifiable instance used in its stead. If the
321    * multimap does not support null keys or values, alternatives may be specified for tests
322    * involving nulls.
323    */
324   private static void checkUnmodifiableMultimap(
325       Multimap<String, Integer> multimap,
326       boolean permitsDuplicates,
327       @Nullable String nullKey,
328       @Nullable Integer nullValue) {
329     Multimap<String, Integer> unmodifiable =
330         prepareUnmodifiableTests(multimap, permitsDuplicates, nullKey, nullValue);
331 
332     UnmodifiableCollectionTests.assertMultimapIsUnmodifiable(unmodifiable, "test", 123);
333 
334     assertUnmodifiableIterableInTandem(unmodifiable.keys(), multimap.keys());
335 
336     assertUnmodifiableIterableInTandem(unmodifiable.keySet(), multimap.keySet());
337 
338     assertUnmodifiableIterableInTandem(unmodifiable.entries(), multimap.entries());
339 
340     assertUnmodifiableIterableInTandem(
341         unmodifiable.asMap().entrySet(), multimap.asMap().entrySet());
342 
343     assertEquals(multimap.toString(), unmodifiable.toString());
344     assertEquals(multimap.hashCode(), unmodifiable.hashCode());
345     assertEquals(multimap, unmodifiable);
346 
347     assertThat(unmodifiable.asMap().get("bar")).containsExactly(5, -1);
348     assertNull(unmodifiable.asMap().get("missing"));
349 
350     assertFalse(unmodifiable.entries() instanceof Serializable);
351   }

353   /** Prepares the multimap for unmodifiable tests, returning an unmodifiable view of the map. */
354   private static Multimap<String, Integer> prepareUnmodifiableTests(
355       Multimap<String, Integer> multimap,
356       boolean permitsDuplicates,
357       @Nullable String nullKey,
358       @Nullable Integer nullValue) {
359     multimap.clear();
360     multimap.put("foo", 1);
361     multimap.put("foo", 2);
362     multimap.put("foo", 3);
363     multimap.put("bar", 5);
364     multimap.put("bar", -1);
365     multimap.put(nullKey, nullValue);
366     multimap.put("foo", nullValue);
367     multimap.put(nullKey, 5);
368     multimap.put("foo", 2);
369 
370     if (permitsDuplicates) {
371       assertEquals(9, multimap.size());
372     } else {
373       assertEquals(8, multimap.size());
374     }
375 
376     Multimap<String, Integer> unmodifiable;
377     if (multimap instanceof SortedSetMultimap) {
378       unmodifiable =
379           Multimaps.unmodifiableSortedSetMultimap((SortedSetMultimap<String, Integer>) multimap);
380     } else if (multimap instanceof SetMultimap) {
381       unmodifiable = Multimaps.unmodifiableSetMultimap((SetMultimap<String, Integer>) multimap);
382     } else if (multimap instanceof ListMultimap) {
383       unmodifiable = Multimaps.unmodifiableListMultimap((ListMultimap<String, Integer>) multimap);
384     } else {
385       unmodifiable = Multimaps.unmodifiableMultimap(multimap);
386     }
387     return unmodifiable;
388   }

459     assertEquals(multimap.hashCode(), multimapView.hashCode());
460     assertEquals(multimap.size(), multimapView.size());
461     assertTrue(multimapView.containsKey("foo"));

465     assertEquals(Collections.singleton(2), multimapView.get("bar"));
466     try {

470     }

474     } catch (UnsupportedOperationException expected) {
475     }
476     try {
477       multimapView.putAll(multimap);
478       fail("UnsupportedOperationException expected");
479     } catch (UnsupportedOperationException expected) {
480     }
481     try {
482       multimapView.replaceValues("foo", Collections.<Integer>emptySet());
483       fail("UnsupportedOperationException expected");
484     } catch (UnsupportedOperationException expected) {
485     }
486     multimapView.remove("bar", 2);
487     assertFalse(multimapView.containsKey("bar"));
488     assertFalse(map.containsKey("bar"));
489     assertEquals(map.keySet(), multimapView.keySet());
490     assertEquals(map.keySet(), multimapView.keys().elementSet());
491     assertThat(multimapView.keys()).contains("foo");
492     assertThat(multimapView.values()).contains(1);
493     assertThat(multimapView.entries()).contains(Maps.immutableEntry("foo", 1));

496     multimapView.clear();
497     assertFalse(multimapView.containsKey("foo"));
498     assertFalse(map.containsKey("foo"));
499     assertTrue(map.isEmpty());
500     assertTrue(multimapView.isEmpty());
501     multimap.clear();
502     assertEquals(multimap.toString(), multimapView.toString());
503     assertEquals(multimap.hashCode(), multimapView.hashCode());
504     assertEquals(multimap.size(), multimapView.size());
