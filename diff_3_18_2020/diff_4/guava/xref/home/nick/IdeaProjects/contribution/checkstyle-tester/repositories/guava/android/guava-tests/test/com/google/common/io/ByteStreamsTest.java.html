54    public void testCopyFileChannel() throws IOException {
55      final int chunkSize = 14407; // Random prime, unlikely to match any internal chunk size
56      ByteArrayOutputStream out = new ByteArrayOutputStream();
57      WritableByteChannel outChannel = Channels.newChannel(out);
58  
59      File testFile = createTempFile();
60      FileOutputStream fos = new FileOutputStream(testFile);
61      byte[] dummyData = newPreFilledByteArray(chunkSize);
62      try {
63        for (int i = 0; i < 500; i++) {
64          fos.write(dummyData);
65        }
66      } finally {
67        fos.close();
68      }
69      ReadableByteChannel inChannel = new RandomAccessFile(testFile, "r").getChannel();
70      try {
71        ByteStreams.copy(inChannel, outChannel);
72      } finally {
73        inChannel.close();
74      }
75      byte[] actual = out.toByteArray();
76      for (int i = 0; i < 500 * chunkSize; i += chunkSize) {
77        assertEquals(dummyData, Arrays.copyOfRange(actual, i, i + chunkSize));
78      }
79    }

507   /** Stream that will skip a maximum number of bytes at a time. */
508   private static class SlowSkipper extends FilterInputStream {
509     private final long max;
510 
511     public SlowSkipper(InputStream in, long max) {
512       super(in);
513       this.max = max;
514     }
515 
516     @Override
517     public long skip(long n) throws IOException {
518       return super.skip(Math.min(max, n));
519     }
520   }
