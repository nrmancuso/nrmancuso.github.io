106   public void testVectors() {
107     for (int i = 0; i < EXPECTED.length; ++i) {
108       byte[] msg = new byte[i];
109       for (int j = 0; j < i; ++j) {
110         msg[j] = (byte) j;
111       }
112       assertSip(msg, EXPECTED[i]);
113     }
114   }

139   // Test for common pitfall regarding sign extension.
140   // For example: (long) data[i++] | (long) data[i++] << 8 | ...
141   // If data[i] == (byte) 0x80, the first cast will sign-extend it to 0xffffffffffffff80,
142   // masking the remaining seven bytes.
143   // To test this, we give an input where bit 7 is not cleared. For example:
144   // (1) 00 01 02 03 04 05 06 07 80
145   // (2) 00 01 02 03 04 05 06 07 81
146   // (3) 00 01 02 03 04 05 06 07 ff (or anything in between)
147   // A fault implementation will generate collisions for these inputs.
148   public void testCollisionsDueToIncorrectSignExtension() {
149     byte[] col1 = new byte[] {0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, (byte) 0x80};
150     byte[] col2 = new byte[] {0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, (byte) 0x81};
151     byte[] col3 = new byte[] {0x00, 0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, (byte) 0xff};
152 
153     ImmutableSet<HashCode> hashCodes =
154         ImmutableSet.of(
155             SIP_WITH_KEY.hashBytes(col1),
156             SIP_WITH_KEY.hashBytes(col2),
157             SIP_WITH_KEY.hashBytes(col3));
158     assertEquals(3, hashCodes.size());
159   }

161   public void testToString() {
162     assertEquals("Hashing.sipHash24(" + K0 + ", " + K1 + ")", SIP_WITH_KEY.toString());
163     assertEquals("Hashing.sipHash24(" + K0 + ", " + K1 + ")", SIP_WITHOUT_KEY.toString());
164     assertEquals("Hashing.sipHash24(20, 13)", Hashing.sipHash24(20, 13).toString());
165   }

167   private static void assertSip(String input, long expected) {
168     assertEquals(expected, SIP_WITH_KEY.hashString(input, UTF_8).asLong());
169     assertEquals(expected, SIP_WITH_KEY.newHasher().putString(input, UTF_8).hash().asLong());
170     assertEquals(expected, SIP_WITHOUT_KEY.hashString(input, UTF_8).asLong());
171     assertEquals(expected, SIP_WITHOUT_KEY.newHasher().putString(input, UTF_8).hash().asLong());
172   }
173 
174   private static void assertSip(byte[] input, long expected) {
175     assertEquals(expected, SIP_WITH_KEY.hashBytes(input).asLong());
176     assertEquals(expected, SIP_WITH_KEY.newHasher().putBytes(input).hash().asLong());
177     assertEquals(expected, SIP_WITHOUT_KEY.hashBytes(input).asLong());
178     assertEquals(expected, SIP_WITHOUT_KEY.newHasher().putBytes(input).hash().asLong());
179   }
