35  /**
36   * An unbounded {@linkplain BlockingQueue blocking queue} that uses the same ordering rules as class
37   * {@link PriorityQueue} and supplies blocking retrieval operations. While this queue is logically
38   * unbounded, attempted additions may fail due to resource exhaustion (causing
39   * <tt>OutOfMemoryError</tt>). This class does not permit <tt>null</tt> elements. A priority queue
40   * relying on {@linkplain Comparable natural ordering} also does not permit insertion of
41   * non-comparable objects (doing so results in <tt>ClassCastException</tt>).
42   *
43   * <p>This class and its iterator implement all of the <em>optional</em> methods of the {@link
44   * Collection} and {@link Iterator} interfaces. The Iterator provided in method {@link #iterator()}
45   * is <em>not</em> guaranteed to traverse the elements of the MonitorBasedPriorityBlockingQueue in
46   * any particular order. If you need ordered traversal, consider using
47   * <tt>Arrays.sort(pq.toArray())</tt>. Also, method <tt>drainTo</tt> can be used to <em>remove</em>
48   * some or all elements in priority order and place them in another collection.
49   *
50   * <p>Operations on this class make no guarantees about the ordering of elements with equal
51   * priority. If you need to enforce an ordering, you can define custom classes or comparators that
52   * use a secondary key to break ties in primary priority values. For example, here is a class that
53   * applies first-in-first-out tie-breaking to comparable elements. To use it, you would insert a
54   * <tt>new FIFOEntry(anEntry)</tt> instead of a plain entry object.
55   *
56   * <pre>
57   * class FIFOEntry&lt;E extends Comparable&lt;? super E&gt;&gt;
58   *     implements Comparable&lt;FIFOEntry&lt;E&gt;&gt; {
59   *   final static AtomicLong seq = new AtomicLong();
60   *   final long seqNum;
61   *   final E entry;
62   *   public FIFOEntry(E entry) {
63   *     seqNum = seq.getAndIncrement();
64   *     this.entry = entry;
65   *   }
66   *   public E getEntry() { return entry; }
67   *   public int compareTo(FIFOEntry&lt;E&gt; other) {
68   *     int res = entry.compareTo(other.entry);
69   *     if (res == 0 &amp;&amp; other.entry != this.entry)
70   *       res = (seqNum &lt; other.seqNum ? -1 : 1);
71   *     return res;
72   *   }
73   * }</pre>
74   *
75   * @author Doug Lea
76   * @author Justin T. Sampson
77   * @param <E> the type of elements held in this collection
78   */
79  @CanIgnoreReturnValue // TODO(cpovirk): Consider being more strict.
80  public class MonitorBasedPriorityBlockingQueue<E> extends AbstractQueue<E>
81      implements BlockingQueue<E> {

160   /**
161    * Inserts the specified element into this priority queue.
162    *
163    * @param e the element to add
164    * @return <tt>true</tt> (as specified by {@link Queue#offer})
165    * @throws ClassCastException if the specified element cannot be compared with elements currently
166    *     in the priority queue according to the priority queue's ordering
167    * @throws NullPointerException if the specified element is null

169   @Override
170   public boolean offer(E e) {
171     final Monitor monitor = this.monitor;
172     monitor.enter();
173     try {
174       boolean ok = q.offer(e);
175       if (!ok) {
176         throw new AssertionError();
177       }
178       return true;
179     } finally {
180       monitor.leave();
181     }
182   }

227   @Override
228   public E poll(long timeout, TimeUnit unit) throws InterruptedException {
229     final Monitor monitor = this.monitor;
230     if (monitor.enterWhen(notEmpty, timeout, unit)) {
231       try {
232         return q.poll();
233       } finally {
234         monitor.leave();
235       }
236     } else {
237       return null;
238     }

358   /**
359    * Returns an array containing all of the elements in this queue; the runtime type of the returned
360    * array is that of the specified array. The returned array elements are in no particular order.
361    * If the queue fits in the specified array, it is returned therein. Otherwise, a new array is
362    * allocated with the runtime type of the specified array and the size of this queue.
363    *
364    * <p>If this queue fits in the specified array with room to spare (i.e., the array has more
365    * elements than this queue), the element in the array immediately following the end of the queue
366    * is set to <tt>null</tt>.
367    *
368    * <p>Like the {@link #toArray()} method, this method acts as bridge between array-based and
369    * collection-based APIs. Further, this method allows precise control over the runtime type of the
370    * output array, and may, under certain circumstances, be used to save allocation costs.
371    *
372    * <p>Suppose <tt>x</tt> is a queue known to contain only strings. The following code can be used
373    * to dump the queue into a newly allocated array of <tt>String</tt>:
374    *
375    * <pre>
376    *     String[] y = x.toArray(new String[0]);</pre>
377    *
378    * <p>Note that <tt>toArray(new Object[0])</tt> is identical in function to <tt>toArray()</tt>.
379    *
380    * @param a the array into which the elements of the queue are to be stored, if it is big enough;
381    *     otherwise, a new array of the same runtime type is allocated for this purpose
382    * @return an array containing all of the elements in this queue
383    * @throws ArrayStoreException if the runtime type of the specified array is not a supertype of
384    *     the runtime type of every element in this queue
385    * @throws NullPointerException if the specified array is null
386    */
387   @Override
388   public <T> T[] toArray(T[] a) {
389     final Monitor monitor = this.monitor;
390     monitor.enter();
391     try {
392       return q.toArray(a);
393     } finally {
394       monitor.leave();
395     }
396   }

409   /**
410    * @throws UnsupportedOperationException {@inheritDoc}
411    * @throws ClassCastException {@inheritDoc}
412    * @throws NullPointerException {@inheritDoc}
413    * @throws IllegalArgumentException {@inheritDoc}
414    */
415   @Override
416   public int drainTo(Collection<? super E> c) {
417     if (c == null) throw new NullPointerException();
418     if (c == this) throw new IllegalArgumentException();
419     final Monitor monitor = this.monitor;
420     monitor.enter();
421     try {
422       int n = 0;
423       E e;
424       while ((e = q.poll()) != null) {
425         c.add(e);
426         ++n;
427       }
428       return n;
429     } finally {
430       monitor.leave();
431     }
432   }

434   /**
435    * @throws UnsupportedOperationException {@inheritDoc}
436    * @throws ClassCastException {@inheritDoc}
437    * @throws NullPointerException {@inheritDoc}
438    * @throws IllegalArgumentException {@inheritDoc}
439    */
440   @Override
441   public int drainTo(Collection<? super E> c, int maxElements) {
442     if (c == null) throw new NullPointerException();
443     if (c == this) throw new IllegalArgumentException();
444     if (maxElements <= 0) return 0;
445     final Monitor monitor = this.monitor;
446     monitor.enter();
447     try {
448       int n = 0;
449       E e;
450       while (n < maxElements && (e = q.poll()) != null) {
451         c.add(e);
452         ++n;
453       }
454       return n;
455     } finally {
456       monitor.leave();
457     }
458   }

475   /**
476    * Returns an iterator over the elements in this queue. The iterator does not return the elements
477    * in any particular order. The returned <tt>Iterator</tt> is a "weakly consistent" iterator that
478    * will never throw {@link ConcurrentModificationException}, and guarantees to traverse elements
479    * as they existed upon construction of the iterator, and may (but is not guaranteed to) reflect
480    * any modifications subsequent to construction.
481    *
482    * @return an iterator over the elements in this queue
483    */
484   @Override
485   public Iterator<E> iterator() {
486     return new Itr(toArray());
487   }
488 
489   /** Snapshot iterator that works off copy of underlying q array. */
490   private class Itr implements Iterator<E> {
491     final Object[] array; // Array of all elements
492     int cursor; // index of next element to return;
493     int lastRet; // index of last element, or -1 if no such
494 
495     Itr(Object[] array) {
496       lastRet = -1;
497       this.array = array;
498     }
499 
500     @Override
501     public boolean hasNext() {
502       return cursor < array.length;
503     }
504 
505     @Override
506     public E next() {
507       if (cursor >= array.length) throw new NoSuchElementException();
508       lastRet = cursor;
509 
510       // array comes from q.toArray() and so should have only E's in it
511       @SuppressWarnings("unchecked")
512       E e = (E) array[cursor++];
513       return e;
514     }
515 
516     @Override
517     public void remove() {
518       if (lastRet < 0) throw new IllegalStateException();
519       Object x = array[lastRet];
520       lastRet = -1;
521       // Traverse underlying queue to find == element,
522       // not just a .equals element.
523       monitor.enter();
524       try {
525         for (Iterator<E> it = q.iterator(); it.hasNext(); ) {
526           if (it.next() == x) {
527             it.remove();
528             return;
529           }
530         }
531       } finally {
532         monitor.leave();
533       }
534     }
535   }

538    * Saves the state to a stream (that is, serializes it). This merely wraps default serialization
539    * within the monitor. The serialization strategy for items is left to underlying Queue. Note that
