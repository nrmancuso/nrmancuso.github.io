164   public void testConsistentHash_correctness() {
165     long[] interestingValues = {-1, 0, 1, 2, Long.MAX_VALUE, Long.MIN_VALUE};
166     for (long h : interestingValues) {
167       checkConsistentHashCorrectness(h);
168     }
169     Random r = new Random(7);
170     for (int i = 0; i < 20; i++) {
171       checkConsistentHashCorrectness(r.nextLong());
172     }
173   }

175   private void checkConsistentHashCorrectness(long hashCode) {
176     int last = 0;
177     for (int shards = 1; shards <= 100000; shards++) {
178       int b = Hashing.consistentHash(hashCode, shards);
179       if (b != last) {
180         assertEquals(shards - 1, b);
181         last = b;
182       }
183     }
184   }

186   public void testConsistentHash_probabilities() {
187     AtomicLongMap<Integer> map = AtomicLongMap.create();
188     Random r = new Random(9);
189     for (int i = 0; i < ITERS; i++) {
190       countRemaps(r.nextLong(), map);
191     }
192     for (int shard = 2; shard <= MAX_SHARDS; shard++) {
193       // Rough: don't exceed 1.2x the expected number of remaps by more than 20
194       assertTrue(map.get(shard) <= 1.2 * ITERS / shard + 20);
195     }
196   }
197 
198   private void countRemaps(long h, AtomicLongMap<Integer> map) {
199     int last = 0;
200     for (int shards = 2; shards <= MAX_SHARDS; shards++) {
201       int chosen = Hashing.consistentHash(h, shards);
202       if (chosen != last) {
203         map.incrementAndGet(shards);
204         last = chosen;
205       }
206     }
207   }

539    * Tests equality of {@link Hashing#goodFastHash} instances. This test must be separate from
540    * {@link #testSeededHashFunctionEquals} because the parameter to {@code goodFastHash} is a size,
541    * not a seed, and because that size is rounded up. Thus, {@code goodFastHash} instances with
542    * different parameters can be equal. That fact is a problem for {@code
543    * testSeededHashFunctionEquals}.
544    */
545   public void testGoodFastHashEquals() throws Exception {
546     HashFunction hashFunction1a = Hashing.goodFastHash(1);
547     HashFunction hashFunction1b = Hashing.goodFastHash(32);

550     HashFunction hashFunction3a = Hashing.goodFastHash(129);
551     HashFunction hashFunction3b = Hashing.goodFastHash(256);

553     HashFunction hashFunction4b = Hashing.goodFastHash(384);

586   static void assertSeededHashFunctionEquals(Class<?> clazz) throws Exception {
587     Random random = new Random(RANDOM_SEED);
588     for (Method method : clazz.getDeclaredMethods()) {
589       if (method.getReturnType().equals(HashFunction.class) // must return HashFunction
590           && Modifier.isPublic(method.getModifiers()) // only the public methods
591           && method.getParameterTypes().length != 0 // only the seeded hash functions
592           && !method.getName().equals("concatenating") // don't test Hashing.concatenating()
593           && !method.getName().equals("goodFastHash") // tested in testGoodFastHashEquals
594           && !method.getName().startsWith("hmac")) { // skip hmac functions
595         Object[] params1 = new Object[method.getParameterTypes().length];
596         Object[] params2 = new Object[method.getParameterTypes().length];
597         for (int i = 0; i < params1.length; i++) {
598           if (method.getParameterTypes()[i] == int.class) {
599             params1[i] = random.nextInt();
600             params2[i] = random.nextInt();
601           } else if (method.getParameterTypes()[i] == long.class) {
602             params1[i] = random.nextLong();
603             params2[i] = random.nextLong();
604           } else {
605             fail("Unable to create a random parameter for " + method.getParameterTypes()[i]);
606           }
607         }
608         HashFunction hashFunction1a = (HashFunction) method.invoke(clazz, params1);
609         HashFunction hashFunction1b = (HashFunction) method.invoke(clazz, params1);
610         HashFunction hashFunction2 = (HashFunction) method.invoke(clazz, params2);
611 
612         new EqualsTester()
613             .addEqualityGroup(hashFunction1a, hashFunction1b)
614             .addEqualityGroup(hashFunction2)
615             .testEquals();
616 
617         assertEquals(hashFunction1a.toString(), hashFunction1b.toString());
618       }
619     }
