48  /**
49   * Creates, based on your criteria, a JUnit test suite that exhaustively tests a {@code Multiset}
50   * implementation.
51   *
52   * @author Jared Levy
53   * @author Louis Wasserman
54   */
55  @GwtIncompatible
56  public class MultisetTestSuiteBuilder<E>
57      extends AbstractCollectionTestSuiteBuilder<MultisetTestSuiteBuilder<E>, E> {
58    public static <E> MultisetTestSuiteBuilder<E> using(TestMultisetGenerator<E> generator) {
59      return new MultisetTestSuiteBuilder<E>().usingGenerator(generator);
60    }
61  
62    public enum NoRecurse implements Feature<Void> {
63      NO_ENTRY_SET;
64  
65      @Override
66      public Set<Feature<? super Void>> getImpliedFeatures() {
67        return Collections.emptySet();
68      }
69    }
70  
71    @Override
72    protected List<Class<? extends AbstractTester>> getTesters() {
73      List<Class<? extends AbstractTester>> testers = Helpers.copyToList(super.getTesters());
74      testers.add(CollectionSerializationEqualTester.class);
75      testers.add(MultisetAddTester.class);
76      testers.add(MultisetContainsTester.class);
77      testers.add(MultisetCountTester.class);
78      testers.add(MultisetElementSetTester.class);
79      testers.add(MultisetEqualsTester.class);
80      testers.add(MultisetReadsTester.class);
81      testers.add(MultisetSetCountConditionallyTester.class);
82      testers.add(MultisetSetCountUnconditionallyTester.class);
83      testers.add(MultisetRemoveTester.class);
84      testers.add(MultisetEntrySetTester.class);
85      testers.add(MultisetIteratorTester.class);
86      testers.add(MultisetSerializationTester.class);
87      return testers;
88    }

90    private static Set<Feature<?>> computeEntrySetFeatures(Set<Feature<?>> features) {
91      Set<Feature<?>> derivedFeatures = new HashSet<>();
92      derivedFeatures.addAll(features);
93      derivedFeatures.remove(CollectionFeature.GENERAL_PURPOSE);
94      derivedFeatures.remove(CollectionFeature.SUPPORTS_ADD);
95      derivedFeatures.remove(CollectionFeature.ALLOWS_NULL_VALUES);
96      derivedFeatures.add(CollectionFeature.REJECTS_DUPLICATES_AT_CREATION);
97      if (!derivedFeatures.remove(CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS)) {
98        derivedFeatures.remove(CollectionFeature.SERIALIZABLE);
99      }
100     return derivedFeatures;
101   }
102 
103   static Set<Feature<?>> computeElementSetFeatures(Set<Feature<?>> features) {
104     Set<Feature<?>> derivedFeatures = new HashSet<>();
105     derivedFeatures.addAll(features);
106     derivedFeatures.remove(CollectionFeature.GENERAL_PURPOSE);
107     derivedFeatures.remove(CollectionFeature.SUPPORTS_ADD);
108     if (!derivedFeatures.remove(CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS)) {
109       derivedFeatures.remove(CollectionFeature.SERIALIZABLE);
110     }
111     return derivedFeatures;
112   }
113 
114   private static Set<Feature<?>> computeReserializedMultisetFeatures(Set<Feature<?>> features) {
115     Set<Feature<?>> derivedFeatures = new HashSet<>();
116     derivedFeatures.addAll(features);
117     derivedFeatures.remove(CollectionFeature.SERIALIZABLE);
118     derivedFeatures.remove(CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS);
119     return derivedFeatures;
120   }
121 
122   @Override
123   protected List<TestSuite> createDerivedSuites(
124       FeatureSpecificTestSuiteBuilder<?, ? extends OneSizeTestContainerGenerator<Collection<E>, E>>
125           parentBuilder) {
126     List<TestSuite> derivedSuites = new ArrayList<>(super.createDerivedSuites(parentBuilder));
127 
128     derivedSuites.add(createElementSetTestSuite(parentBuilder));
129 
130     if (!parentBuilder.getFeatures().contains(NoRecurse.NO_ENTRY_SET)) {
131       derivedSuites.add(
132           SetTestSuiteBuilder.using(new EntrySetGenerator<E>(parentBuilder.getSubjectGenerator()))
133               .named(getName() + ".entrySet")
134               .withFeatures(computeEntrySetFeatures(parentBuilder.getFeatures()))
135               .suppressing(parentBuilder.getSuppressedTests())
136               .createTestSuite());
137     }
138 
139     if (parentBuilder.getFeatures().contains(CollectionFeature.SERIALIZABLE)) {
140       derivedSuites.add(
141           MultisetTestSuiteBuilder.using(
142                   new ReserializedMultisetGenerator<E>(parentBuilder.getSubjectGenerator()))
143               .named(getName() + " reserialized")
144               .withFeatures(computeReserializedMultisetFeatures(parentBuilder.getFeatures()))
145               .suppressing(parentBuilder.getSuppressedTests())
146               .createTestSuite());
147     }
148     return derivedSuites;
149   }
150 
151   TestSuite createElementSetTestSuite(
152       FeatureSpecificTestSuiteBuilder<?, ? extends OneSizeTestContainerGenerator<Collection<E>, E>>
153           parentBuilder) {
154     return SetTestSuiteBuilder.using(
155             new ElementSetGenerator<E>(parentBuilder.getSubjectGenerator()))
156         .named(getName() + ".elementSet")
157         .withFeatures(computeElementSetFeatures(parentBuilder.getFeatures()))
158         .suppressing(parentBuilder.getSuppressedTests())
159         .createTestSuite();
160   }

162   static class ElementSetGenerator<E> implements TestSetGenerator<E> {
163     final OneSizeTestContainerGenerator<Collection<E>, E> gen;
164 
165     ElementSetGenerator(OneSizeTestContainerGenerator<Collection<E>, E> gen) {
166       this.gen = gen;
167     }
168 
169     @Override
170     public SampleElements<E> samples() {
171       return gen.samples();
172     }
173 
174     @Override
175     public Set<E> create(Object... elements) {
176       Object[] duplicated = new Object[elements.length * 2];
177       for (int i = 0; i < elements.length; i++) {
178         duplicated[i] = elements[i];
179         duplicated[i + elements.length] = elements[i];
180       }
181       return ((Multiset<E>) gen.create(duplicated)).elementSet();
182     }
183 
184     @Override
185     public E[] createArray(int length) {
186       return gen.createArray(length);
187     }
188 
189     @Override
190     public Iterable<E> order(List<E> insertionOrder) {
191       return gen.order(new ArrayList<E>(new LinkedHashSet<E>(insertionOrder)));
192     }
193   }

195   static class EntrySetGenerator<E> implements TestSetGenerator<Multiset.Entry<E>> {
196     final OneSizeTestContainerGenerator<Collection<E>, E> gen;
197 
198     private EntrySetGenerator(OneSizeTestContainerGenerator<Collection<E>, E> gen) {
199       this.gen = gen;
200     }
201 
202     @Override
203     public SampleElements<Multiset.Entry<E>> samples() {
204       SampleElements<E> samples = gen.samples();
205       return new SampleElements<>(
206           Multisets.immutableEntry(samples.e0(), 3),
207           Multisets.immutableEntry(samples.e1(), 4),
208           Multisets.immutableEntry(samples.e2(), 1),
209           Multisets.immutableEntry(samples.e3(), 5),
210           Multisets.immutableEntry(samples.e4(), 2));
211     }
212 
213     @Override
214     public Set<Multiset.Entry<E>> create(Object... entries) {
215       List<Object> contents = new ArrayList<>();
216       Set<E> elements = new HashSet<>();
217       for (Object o : entries) {
218         @SuppressWarnings("unchecked")
219         Multiset.Entry<E> entry = (Entry<E>) o;
220         checkArgument(
221             elements.add(entry.getElement()), "Duplicate keys not allowed in EntrySetGenerator");
222         for (int i = 0; i < entry.getCount(); i++) {
223           contents.add(entry.getElement());
224         }
225       }
226       return ((Multiset<E>) gen.create(contents.toArray())).entrySet();
227     }
228 
229     @SuppressWarnings("unchecked")
230     @Override
231     public Multiset.Entry<E>[] createArray(int length) {
232       return new Multiset.Entry[length];
233     }
234 
235     @Override
236     public Iterable<Entry<E>> order(List<Entry<E>> insertionOrder) {
237       // We mimic the order from gen.
238       Map<E, Entry<E>> map = new LinkedHashMap<>();
239       for (Entry<E> entry : insertionOrder) {
240         map.put(entry.getElement(), entry);
241       }
242 
243       Set<E> seen = new HashSet<>();
244       List<Entry<E>> order = new ArrayList<>();
245       for (E e : gen.order(new ArrayList<E>(map.keySet()))) {
246         if (seen.add(e)) {
247           order.add(map.get(e));
248         }
249       }
250       return order;
251     }
252   }

254   static class ReserializedMultisetGenerator<E> implements TestMultisetGenerator<E> {
255     final OneSizeTestContainerGenerator<Collection<E>, E> gen;
256 
257     private ReserializedMultisetGenerator(OneSizeTestContainerGenerator<Collection<E>, E> gen) {
258       this.gen = gen;
259     }
260 
261     @Override
262     public SampleElements<E> samples() {
263       return gen.samples();
264     }
265 
266     @Override
267     public Multiset<E> create(Object... elements) {
268       return (Multiset<E>) SerializableTester.reserialize(gen.create(elements));
269     }
270 
271     @Override
272     public E[] createArray(int length) {
273       return gen.createArray(length);
274     }
275 
276     @Override
277     public Iterable<E> order(List<E> insertionOrder) {
278       return gen.order(insertionOrder);
279     }
280   }
