53  	private static final int K_PUBLIC       = 0x0001;
54  	private static final int K_PRIVATE      = 0x0002;
55  	private static final int K_PROTECTED    = 0x0004;
56  	private static final int K_STATIC       = 0x0008;
57  	private static final int K_FINAL        = 0x0010;
58  	private static final int K_SYNCHRONIZED = 0x0020;
59  	private static final int K_VOLATILE     = 0x0040;
60  	private static final int K_TRANSIENT    = 0x0080;
61  	private static final int K_NATIVE       = 0x0100;
62  	private static final int K_ABSTRACT     = 0x0400;
63  	private static final int K_STRICTFP     = 0x0800;
64  	private static final Map<String, Integer> TO_FLAG_MAP = ImmutableMap.<String, Integer>builder()
65  		.put("private", K_PRIVATE)
66  		.put("protected", K_PROTECTED)
67  		.put("public", K_PUBLIC)
68  		.put("final", K_FINAL)
69  		.put("native", K_NATIVE)
70  		.put("strictfp", K_STRICTFP)
71  		.put("synchronized", K_SYNCHRONIZED)
72  		.put("abstract", K_ABSTRACT)
73  		.put("static", K_STATIC)
74  		.put("transient", K_TRANSIENT)
75  		.put("volatile", K_VOLATILE)
76  		.build();

78  	private static final int[] METHOD_MODIFIERS_EXCLUSIVITY = {
79  		K_PRIVATE   | K_PROTECTED,
80  		K_PRIVATE   | K_PUBLIC,
81  		K_PROTECTED | K_PUBLIC,
82  		K_NATIVE    | K_STRICTFP,
83  		K_ABSTRACT  | K_FINAL,
84  		K_ABSTRACT  | K_NATIVE,
85  		K_ABSTRACT  | K_STATIC,
86  		K_ABSTRACT  | K_STRICTFP,
87  		K_ABSTRACT  | K_SYNCHRONIZED,
88  	};
89  	
90  	private static final int METHOD_MODIFIERS_LEGAL =
91  		K_PRIVATE | K_PROTECTED | K_PUBLIC | K_ABSTRACT | K_FINAL | K_NATIVE |
92  		K_STATIC | K_STRICTFP | K_SYNCHRONIZED
93  	;

95  	private static final int[] FIELD_MODIFIERS_EXCLUSIVITY = {
96  		K_PRIVATE | K_PROTECTED,
97  		K_PRIVATE | K_PUBLIC,
98  		K_PROTECTED | K_PUBLIC,
99  		K_VOLATILE | K_FINAL,
100 	};

102 	private static final int FIELD_MODIFIERS_LEGAL =
103 		K_PRIVATE | K_PROTECTED | K_PUBLIC | K_STATIC | K_FINAL | K_TRANSIENT | K_VOLATILE;
104 	
105 	private static final int[] TYPE_MODIFIERS_EXCLUSIVITY = {
106 		K_PRIVATE | K_PROTECTED,
107 		K_PRIVATE | K_PUBLIC,
108 		K_PROTECTED | K_PUBLIC,
109 		K_FINAL | K_ABSTRACT,
110 	};
111 	
112 	private static final int TYPE_MODIFIERS_LEGAL =
113 		K_PRIVATE | K_PROTECTED | K_PUBLIC | K_STATIC | K_FINAL | K_ABSTRACT | K_STRICTFP;

115 	public void duplicateKeywordModifierCheck(Modifiers modifiers) {
116 		List<String> keywords = Lists.newArrayList();
117 		
118 		for (KeywordModifier n : modifiers.astKeywords()) {
119 			String k = n.astName();
120 			if (k != null && !k.isEmpty() ) {
121 				if (keywords.contains(k)) {
122 					n.addMessage(error(MODIFIERS_DUPLICATE_KEYWORD, "Duplicate modifier: " + k));
123 				} else {
124 					keywords.add(k);
125 				}
126 			}
127 		}
128 	}

130 	public void methodModifiersCheck(MethodDeclaration md) {
131 		modifiersCheck(md.astModifiers(),
132 				METHOD_MODIFIERS_EXCLUSIVITY, METHOD_MODIFIERS_LEGAL, "method declarations");
133 		checkStaticChain(md.astModifiers());
134 	}

136 	public void annotationMethodModifiersCheck(AnnotationMethodDeclaration md) {
137 		modifiersCheck(md.astModifiers(),
138 				METHOD_MODIFIERS_EXCLUSIVITY, METHOD_MODIFIERS_LEGAL & ~K_STRICTFP,
139 				"annotation method declarations");
140 		checkStaticChain(md.astModifiers());
141 	}

143 	public void fieldModifiersCheck(VariableDeclaration vd) {
144 		TypeDeclaration td = vd.upUpToTypeDeclaration();
145 		if (td == null) return;	//not a field.
146 		VariableDefinition def = vd.astDefinition();
147 		if (def != null) {
148 			Modifiers m = def.astModifiers();
149 			modifiersCheck(m, FIELD_MODIFIERS_EXCLUSIVITY, FIELD_MODIFIERS_LEGAL, "field declarations");
150 			boolean allFieldsHaveInitializers = true;
151 			for (VariableDefinitionEntry entry : def.astVariables()) {
152 				if (entry.rawInitializer() == null) {
153 					allFieldsHaveInitializers = false;
154 					break;
155 				}
156 			}
157 			
158 			if (m.isStatic() && !m.isFinal() && !allFieldsHaveInitializers) {
159 				// Initialized static final fields, assuming the initializer expression is a compile time constant, are 'special' and
160 				// do not need to adhere to the static chain rule. However, we can't determine CTC nature without resolution.
161 				checkStaticChain(m);
162 			}
163 		}
164 	}
165 	
166 	public void localVariableModifiersCheck(VariableDefinition vd) {
167 		boolean applies = vd.upUpIfLocalVariableToBlock() != null;
168 		if (!applies) applies = vd.upToForEach() != null;
169 		if (!applies) applies = vd.upToFor() != null;
170 		if (!applies) return;
171 		
172 		modifiersCheck(vd.astModifiers(), new int[0], K_FINAL, "local variable declarations");
173 	}

175 	public void classDeclarationModifiersCheck(ClassDeclaration cd) {
176 		typeModifiersCheck(cd, "class declarations");
177 	}
178 	
179 	public void interfaceDeclarationModifiersCheck(InterfaceDeclaration id) {
180 		int flags = typeModifiersCheck(id, "interface declarations");
181 		generateNotAllowedKeywordError(id.astModifiers(), flags, K_FINAL, "final", "Interfaces cannot be final");
182 	}

184 	public void annotationDeclarationModifiersCheck(AnnotationDeclaration ad) {
185 		int flags = typeModifiersCheck(ad, "annotation declarations");
186 		generateNotAllowedKeywordError(ad.astModifiers(), flags, K_FINAL, "final", "Annotations cannot be final");
187 	}

189 	public void enumDeclarationModifiersCheck(EnumDeclaration ad) {
190 		int flags = typeModifiersCheck(ad, "enum declarations");
191 		generateNotAllowedKeywordError(ad.astModifiers(), flags, K_FINAL, "final", "Enums cannot be marked final");
192 	}

194 	private int typeModifiersCheck(TypeDeclaration td, String desc) {
195 		int flags = modifiersCheck(td.astModifiers(),
196 				TYPE_MODIFIERS_EXCLUSIVITY, TYPE_MODIFIERS_LEGAL, desc);
197 		boolean staticWarningEmitted = false;
198 		if (td.upIfTopLevelToCompilationUnit() != null) {
199 			generateNotAllowedKeywordError(td.astModifiers(), flags, K_PRIVATE, "private", "Top-level types cannot be private.");
200 			staticWarningEmitted |= generateNotAllowedKeywordError(td.astModifiers(), flags, K_STATIC, "static", "Top-level types cannot be static.");
201 		} else if (td.upToBlock() != null) {
202 			generateNotAllowedKeywordError(td.astModifiers(), flags, K_PRIVATE, "private", "Method-local types cannot be private.");
203 			generateNotAllowedKeywordError(td.astModifiers(), flags, K_PROTECTED, "protected", "Method-local types cannot be protected.");
204 			generateNotAllowedKeywordError(td.astModifiers(), flags, K_PUBLIC, "public", "Method-local types cannot be public.");
205 			staticWarningEmitted |= generateNotAllowedKeywordError(td.astModifiers(), flags, K_STATIC, "static", "Method-local types cannot be static.");
206 		}
207 		
208 		if (!staticWarningEmitted) checkStaticChain(td.astModifiers());
209 		
210 		return flags;
211 	}
212 	
213 	public void checkStaticInitializerInNonStaticType(StaticInitializer node) {
214 		TypeDeclaration parent = node.upUpToTypeDeclaration();
215 		if (parent != null) {
216 			if (!parent.astModifiers().isStatic()) {
217 				node.addMessage(error(INITIALIZER_STATIC_IN_NON_STATIC_TYPE,
218 						"static initializers are only allowed in top-level or static types declarations."));
219 			}
220 		}
221 	}
222 	
223 	private void checkStaticChain(Modifiers modifiers) {
224 		if (!modifiers.isStatic()) return;
225 		
226 		Node p = modifiers.getParent();
227 		while (p != null) {
228 			if (p instanceof CompilationUnit) return;
229 			if (p instanceof TypeDeclaration) {
230 				Modifiers pMods = ((TypeDeclaration)p).astModifiers();
231 				if (!pMods.isStatic()) {
232 					modifiers.getParent().addMessage(error(MODIFIERS_STATIC_CHAIN,
233 							"This declaration is (effectively) static; static declarations or only legal in top-level and static declarations."));
234 				}
235 			}
236 			p = p.getParent();
237 		}
238 	}

240 	private int modifiersCheck(Modifiers modifiers, int[] exclusivity, int legality, String desc) {
241 		int flags = modifiers.getEffectiveModifierFlags();
242 		int implicits = flags & ~modifiers.getExplicitModifierFlags();
243 		
244 		for (Node n : modifiers.rawKeywords()) {
245 			if (n instanceof KeywordModifier) {
246 				String k = ((KeywordModifier)n).astName();
247 				if (k == null || k.isEmpty()) {
248 					n.addMessage(error(MODIFIERS_EMPTY_MODIFIER, "Empty/null modifier."));
249 				}
250 				
251 				if (!TO_FLAG_MAP.containsKey(k)) {
252 					n.addMessage(error(MODIFIERS_UNKNOWN_MODIFIER, "Unknown modifier: " + k));
253 					continue;
254 				}
255 				
256 				int flag = TO_FLAG_MAP.get(k);
257 				if ((legality & flag) == 0) {
258 					n.addMessage(error(MODIFIERS_MODIFIER_NOT_ALLOWED, "Modifier not allowed on " + desc + ": " + k));
259 					continue;
260 				}
261 				
262 				flags |= flag;
263 			}
264 		}
265 		
266 		for (int exclusive : exclusivity) {
267 			if ((flags & exclusive) == exclusive) {
268 				generateExclusivityError(implicits, exclusive, modifiers);
269 			}
270 		}
271 		
272 		return flags;
273 	}
274 	
275 	private boolean generateNotAllowedKeywordError(Modifiers modifiers, int flags, int flag, String keyword, String error) {
276 		if ((flags & flag) == 0) return false;
277 		
278 		for (KeywordModifier n : modifiers.astKeywords()) {
279 			if (keyword.equals(n.astName())) {
280 				n.addMessage(error(MODIFIERS_MODIFIER_NOT_ALLOWED, error));
281 				return true;
282 			}
283 		}
284 		
285 		return false;
286 	}

288 	public void emptyDeclarationMustHaveNoModifiers(EmptyDeclaration node) {
289 		Modifiers modifiers = node.astModifiers();
290 		
291 		if (!modifiers.astKeywords().isEmpty() || !modifiers.astAnnotations().isEmpty()) {
292 			node.addMessage(error(MODIFIERS_MODIFIER_NOT_ALLOWED, "Empty Declarations cannot have modifiers."));
293 		}
294 	}
295 	
296 	private void generateExclusivityError(int implicit, int exclusive, Modifiers modifiers) {
297 		String hit = null;
298 		
299 		int responsibleImplicit = implicit & exclusive;
300 		
301 		if (responsibleImplicit != 0) {
302 			String nameOfResponsibleImplicit = "(unknown)";
303 			for (Map.Entry<String, Integer> x : TO_FLAG_MAP.entrySet()) {
304 				if (x.getValue() == responsibleImplicit) nameOfResponsibleImplicit = x.getKey();
305 			}
306 			
307 			for (Node n : modifiers.rawKeywords()) {
308 				if (n instanceof KeywordModifier) {
309 					String k = ((KeywordModifier)n).astName();
310 					if (!TO_FLAG_MAP.containsKey(k)) continue;
311 					int f = TO_FLAG_MAP.get(k);
312 					if ((f & exclusive) == 0) continue;
313 					
314 					modifiers.addMessage(error(MODIFIERS_MODIFIER_CONFLICT, String.format(
315 							"Modifier %s cannot be used here; it is already implicitly %s.", k, nameOfResponsibleImplicit)));
316 				}
317 			}
318 		} else {
319 			for (Node n : modifiers.rawKeywords()) {
320 				if (n instanceof KeywordModifier) {
321 					String k = ((KeywordModifier)n).astName();
322 					if (!TO_FLAG_MAP.containsKey(k)) continue;
323 					int f = TO_FLAG_MAP.get(k);
324 					if ((f & exclusive) == 0) continue;
325 					
326 					if (hit == null) {
327 						hit = k;
328 						continue;
329 					}
330 					
331 					modifiers.addMessage(error(MODIFIERS_MODIFIER_CONFLICT, String.format(
332 							"Modifier %s cannot be used together with %s here.", k, hit)));
333 				}
334 			}
335 		}
