23  /**
24   * Code generator for structural pattern matching patterns.

26   * @author Daniel Dietrich
27   */
28  public class Generator {

30      private Generator() {
31      }

33      // ENTRY POINT: Expands one @Patterns class
34      public static String generate(String derivedClassName, ClassModel classModel) {
35          final List<MethodModel> methodModels = classModel.getMethods().stream()
36                  .filter(method -> method.isAnnotatedWith(Unapply.class))
37                  .collect(toList());
38          final String _package = classModel.getPackageName();
39          final ImportManager im = ImportManager.forClass(classModel, "io.vavr.API.Match");
40          final String methods = generate(im, classModel, methodModels);
41          return "// @formatter:off\n" +
42                  "// CHECKSTYLE:OFF\n" +
43                  (_package.isEmpty() ? "" : "package " + _package + ";\n\n") +
44                  im.getImports() +
45                  "\n\n// GENERATED BY VAVR <<>> derived from " + classModel.getFullQualifiedName() + "\n\n" +
46                  "@SuppressWarnings(\"deprecation\")\n" +
47                  "public final class " + derivedClassName + " {\n\n" +
48                  "    private " + derivedClassName + "() {\n" +
49                  "    }\n\n" +
50                  methods +
51                  "}\n" +
52                  "// CHECKSTYLE:ON\n" +
53                  "// @formatter:on";
54      }
55  
56      // Expands the @Unapply methods of a @Patterns class
57      private static String generate(ImportManager im, ClassModel classModel, List<MethodModel> methodModels) {
58          final StringBuilder builder = new StringBuilder();
59          for (MethodModel methodModel : methodModels) {
60              generate(im, classModel, methodModel, builder);
61              builder.append("\n");
62          }
63          return builder.toString();
64      }
65  
66      // Expands one @Unapply method
67      private static void generate(ImportManager im, ClassModel classModel, MethodModel methodModel, StringBuilder builder) {
68          final String paramTypeName = im.getType(methodModel.getParameter(0).getType());
69          final String definedName = methodModel.getName();
70          final String generatedName = "$" + definedName;
71          final int arity = Integer.parseInt(methodModel.getReturnType().getClassName().substring("Tuple".length()));
72          final String body;
73          if (arity == 0) {
74              body = pattern(im, 0) + ".of(" + paramTypeName + ".class)";
75          } else {
76              final String args = IntStream.rangeClosed(1, arity).mapToObj(i -> "p" + i).collect(joining(", "));
77              final String unapplyRef = classModel.getFullQualifiedName() + "::" + definedName;
78              body = String.format("%s.of(%s, %s, %s)", pattern(im, arity), paramTypeName + ".class", args, unapplyRef);
79          }
80          final List<String> typeArgs = methodModel.getTypeParameters().stream()
81                  .map(typeParameterModel -> mapToName(im, typeParameterModel))
82                  .collect(toList());
83          final List<String> upperBoundArgs = deriveUpperBounds(typeArgs, methodModel.getReturnType().getTypeParameters().size());
84          final String returnType = genReturnType(im, methodModel, upperBoundArgs, arity);
85          final String method;
86          if (arity == 0 && methodModel.getTypeParameters().size() == 0) {
87              method = String.format("final %s %s = %s;", returnType, generatedName, body);
88          } else {
89              final String generics = genGenerics(im, methodModel, typeArgs, upperBoundArgs);
90              final String params = genParams(im, upperBoundArgs, arity);
91              method = String.format("%s %s %s(%s) {\n        return %s;\n    }", generics, returnType, generatedName, params, body);
92          }
93          builder.append("    public static ").append(method).append("\n");
94      }

96      // Introduces new upper generic type bounds for decomposed object parts
97      private static List<String> deriveUpperBounds(List<String> typeArgs, int count) {
98          final List<String> result = new ArrayList<>();
99          final Set<String> knownTypeArgs = new HashSet<>(typeArgs);
100         for (int i = 0; i < count; i++) {
101             String typeArg = "_" + (i + 1);
102             while (knownTypeArgs.contains(typeArg)) {
103                 typeArg = "_" + typeArg;
104             }
105             result.add(typeArg);
106             knownTypeArgs.add(typeArg);
107         }
108         return result;
109     }
110 
111     // Expands the generics part of a method declaration
112     private static String genGenerics(ImportManager im, MethodModel methodModel, List<String> typeParameters, List<String> upperBoundArgs) {
113         final List<TypeParameterModel> returnTypeArgs = methodModel.getReturnType().getTypeParameters();
114         if (typeParameters.size() + returnTypeArgs.size() == 0) {
115             return "";
116         } else {
117             final List<String> result = new ArrayList<>(typeParameters);
118             for (int i = 0; i < returnTypeArgs.size(); i++) {
119                 final String returnTypeArg = mapToName(im, returnTypeArgs.get(i));
120                 result.add(upperBoundArgs.get(i) + " extends " + returnTypeArg);
121             }
122             return result.stream().collect(joining(", ", "<", ">"));
123         }
124     }
125 
126     // Expands the return type of a method declaration
127     private static String genReturnType(ImportManager im, MethodModel methodModel, List<String> upperBoundArgs, int arity) {
128         final List<String> resultTypes = new ArrayList<>();
129         final String type = mapToName(im, methodModel.getParameter(0).getType());
130         resultTypes.add(type);
131         resultTypes.addAll(upperBoundArgs);
132         return pattern(im, arity) + resultTypes.stream().collect(joining(", ", "<", ">"));
133     }
134 
135     // Expands the parameters of a method declaration
136     private static String genParams(ImportManager im, List<String> upperBoundArgs, int arity) {
137         final String patternType = im.getType("io.vavr", "API.Match.Pattern");
138         return IntStream.range(0, arity)
139                 .mapToObj(i -> patternType + "<" + upperBoundArgs.get(i) + ", ?> p" + (i + 1))
140                 .collect(joining(", "));
141     }
142 
143     // Recursively maps generic type parameters to names according to their kind
144     private static String mapToName(ImportManager im, TypeParameterModel typeParameterModel) {
145         if (typeParameterModel.isType()) {
146             return mapToName(im, typeParameterModel.asType());
147         } else if (typeParameterModel.isTypeVar()) {
148             return typeParameterModel.asTypeVar();
149         } else {
150             throw new IllegalStateException("Unhandled type parameter: " + typeParameterModel.toString());
151         }
152     }
153 
154     // Recursively maps class generics to names
155     private static String mapToName(ImportManager im, ClassModel classModel) {
156         final List<TypeParameterModel> typeParameters = classModel.getTypeParameters();
157         final String simpleName = im.getType(classModel);
158         if (typeParameters.size() == 0) {
159             return simpleName;
160         } else {
161             return simpleName + classModel.getTypeParameters().stream()
162                     .map(typeParam -> mapToName(im, typeParam))
163                     .collect(joining(", ", "<", ">"));
164         }
165     }
166 
167     private static String pattern(ImportManager im, int arity) {
168         return im.getType("io.vavr", "API.Match.Pattern" + arity);
169     }
