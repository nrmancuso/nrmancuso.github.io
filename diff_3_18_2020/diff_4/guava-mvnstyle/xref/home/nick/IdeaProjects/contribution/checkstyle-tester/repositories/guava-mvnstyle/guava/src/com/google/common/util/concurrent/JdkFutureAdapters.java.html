15  package com.google.common.util.concurrent;
16  
17  import static com.google.common.base.Preconditions.checkNotNull;
18  import static com.google.common.util.concurrent.Uninterruptibles.getUninterruptibly;
19  
20  import com.google.common.annotations.Beta;
21  import com.google.common.annotations.GwtIncompatible;
22  import java.util.concurrent.Executor;
23  import java.util.concurrent.Executors;
24  import java.util.concurrent.Future;
25  import java.util.concurrent.ThreadFactory;
26  import java.util.concurrent.atomic.AtomicBoolean;
27  
28  /**
29   * Utilities necessary for working with libraries that supply plain {@link Future} instances. Note
30   * that, whenever possible, it is strongly preferred to modify those libraries to return {@code
31   * ListenableFuture} directly.
32   *
33   * @author Sven Mawson
34   * @since 10.0 (replacing {@code Futures.makeListenable}, which existed in 1.0)
35   */
36  @Beta
37  @GwtIncompatible
38  public final class JdkFutureAdapters {
39    /**
40     * Assigns a thread to the given {@link Future} to provide {@link ListenableFuture} functionality.
41     *
42     * <p><b>Warning:</b> If the input future does not already implement {@code ListenableFuture}, the
43     * returned future will emulate {@link ListenableFuture#addListener} by taking a thread from an
44     * internal, unbounded pool at the first call to {@code addListener} and holding it until the
45     * future is {@linkplain Future#isDone() done}.
46     *
47     * <p>Prefer to create {@code ListenableFuture} instances with {@link SettableFuture}, {@link
48     * MoreExecutors#listeningDecorator( java.util.concurrent.ExecutorService)}, {@link
49     * ListenableFutureTask}, {@link AbstractFuture}, and other utilities over creating plain {@code
50     * Future} instances to be upgraded to {@code ListenableFuture} after the fact.
51     */
52    public static <V> ListenableFuture<V> listenInPoolThread(Future<V> future) {
53      if (future instanceof ListenableFuture) {
54        return (ListenableFuture<V>) future;
55      }
56      return new ListenableFutureAdapter<V>(future);
57    }
58  
59    /**
60     * Submits a blocking task for the given {@link Future} to provide {@link ListenableFuture}
61     * functionality.
62     *
63     * <p><b>Warning:</b> If the input future does not already implement {@code ListenableFuture}, the
64     * returned future will emulate {@link ListenableFuture#addListener} by submitting a task to the
65     * given executor at the first call to {@code addListener}. The task must be started by the
66     * executor promptly, or else the returned {@code ListenableFuture} may fail to work. The task's
67     * execution consists of blocking until the input future is {@linkplain Future#isDone() done}, so
68     * each call to this method may claim and hold a thread for an arbitrary length of time. Use of
69     * bounded executors or other executors that may fail to execute a task promptly may result in
70     * deadlocks.
71     *
72     * <p>Prefer to create {@code ListenableFuture} instances with {@link SettableFuture}, {@link
73     * MoreExecutors#listeningDecorator( java.util.concurrent.ExecutorService)}, {@link
74     * ListenableFutureTask}, {@link AbstractFuture}, and other utilities over creating plain {@code
75     * Future} instances to be upgraded to {@code ListenableFuture} after the fact.
76     *
77     * @since 12.0
78     */
79    public static <V> ListenableFuture<V> listenInPoolThread(Future<V> future, Executor executor) {
80      checkNotNull(executor);
81      if (future instanceof ListenableFuture) {
82        return (ListenableFuture<V>) future;
83      }
84      return new ListenableFutureAdapter<V>(future, executor);
85    }
86  
87    /**
88     * An adapter to turn a {@link Future} into a {@link ListenableFuture}. This will wait on the
89     * future to finish, and when it completes, run the listeners. This implementation will wait on
90     * the source future indefinitely, so if the source future never completes, the adapter will never
91     * complete either.
92     *
93     * <p>If the delegate future is interrupted or throws an unexpected unchecked exception, the
94     * listeners will not be invoked.
95     */
96    private static class ListenableFutureAdapter<V> extends ForwardingFuture<V>
97        implements ListenableFuture<V> {
98  
99      private static final ThreadFactory threadFactory =
100         new ThreadFactoryBuilder()
101             .setDaemon(true)
102             .setNameFormat("ListenableFutureAdapter-thread-%d")
103             .build();
104     private static final Executor defaultAdapterExecutor =
105         Executors.newCachedThreadPool(threadFactory);
106 
107     private final Executor adapterExecutor;
108 
109     // The execution list to hold our listeners.
110     private final ExecutionList executionList = new ExecutionList();
111 
112     // This allows us to only start up a thread waiting on the delegate future when the first
113     // listener is added.
114     private final AtomicBoolean hasListeners = new AtomicBoolean(false);
115 
116     // The delegate future.
117     private final Future<V> delegate;
118 
119     ListenableFutureAdapter(Future<V> delegate) {
120       this(delegate, defaultAdapterExecutor);
121     }
122 
123     ListenableFutureAdapter(Future<V> delegate, Executor adapterExecutor) {
124       this.delegate = checkNotNull(delegate);
125       this.adapterExecutor = checkNotNull(adapterExecutor);
126     }
127 
128     @Override
129     protected Future<V> delegate() {
130       return delegate;
131     }
132 
133     @Override
134     public void addListener(Runnable listener, Executor exec) {
135       executionList.add(listener, exec);
136 
137       // When a listener is first added, we run a task that will wait for the delegate to finish,
138       // and when it is done will run the listeners.
139       if (hasListeners.compareAndSet(false, true)) {
140         if (delegate.isDone()) {
141           // If the delegate is already done, run the execution list immediately on the current
142           // thread.
143           executionList.execute();
144           return;
145         }
146 
147         // TODO(lukes): handle RejectedExecutionException
148         adapterExecutor.execute(
149             new Runnable() {
150               @Override
151               public void run() {
152                 try {
153                   /*
154                    * Threads from our private pool are never interrupted. Threads from a
155                    * user-supplied executor might be, but... what can we do? This is another reason
156                    * to return a proper ListenableFuture instead of using listenInPoolThread.
157                    */
158                   getUninterruptibly(delegate);
159                 } catch (Throwable e) {
160                   // ExecutionException / CancellationException / RuntimeException / Error
161                   // The task is presumably done, run the listeners.
162                 }
163                 executionList.execute();
164               }
165             });
166       }
167     }
168   }
169 
170   private JdkFutureAdapters() {}
171 }
