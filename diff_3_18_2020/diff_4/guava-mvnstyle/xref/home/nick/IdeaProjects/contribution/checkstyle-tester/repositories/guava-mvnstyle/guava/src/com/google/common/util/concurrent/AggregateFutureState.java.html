15  package com.google.common.util.concurrent;
16  
17  import static com.google.common.collect.Sets.newConcurrentHashSet;
18  import static java.util.concurrent.atomic.AtomicIntegerFieldUpdater.newUpdater;
19  import static java.util.concurrent.atomic.AtomicReferenceFieldUpdater.newUpdater;
20  
21  import com.google.common.annotations.GwtCompatible;
22  import com.google.j2objc.annotations.ReflectionSupport;
23  import java.util.Set;
24  import java.util.concurrent.atomic.AtomicIntegerFieldUpdater;
25  import java.util.concurrent.atomic.AtomicReferenceFieldUpdater;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  
29  /**
30   * A helper which does some thread-safe operations for aggregate futures, which must be implemented
31   * differently in GWT. Namely:
32   *
33   * <ul>
34   *   <li>Lazily initializes a set of seen exceptions
35   *   <li>Decrements a counter atomically
36   * </ul>
37   */
38  @GwtCompatible(emulated = true)
39  @ReflectionSupport(value = ReflectionSupport.Level.FULL)
40  abstract class AggregateFutureState<OutputT> extends AbstractFuture.TrustedFuture<OutputT> {
41    // Lazily initialized the first time we see an exception; not released until all the input futures
42    // have completed and we have processed them all.
43    private volatile Set<Throwable> seenExceptions = null;
44  
45    private volatile int remaining;
46  
47    private static final AtomicHelper ATOMIC_HELPER;
48  
49    private static final Logger log = Logger.getLogger(AggregateFutureState.class.getName());
50  
51    static {
52      AtomicHelper helper;
53      Throwable thrownReflectionFailure = null;
54      try {
55        helper =
56            new SafeAtomicHelper(
57                newUpdater(AggregateFutureState.class, (Class) Set.class, "seenExceptions"),
58                newUpdater(AggregateFutureState.class, "remaining"));
59      } catch (Throwable reflectionFailure) {
60        // Some Android 5.0.x Samsung devices have bugs in JDK reflection APIs that cause
61        // getDeclaredField to throw a NoSuchFieldException when the field is definitely there.
62        // For these users fallback to a suboptimal implementation, based on synchronized. This will
63        // be a definite performance hit to those users.
64        thrownReflectionFailure = reflectionFailure;
65        helper = new SynchronizedAtomicHelper();
66      }
67      ATOMIC_HELPER = helper;
68      // Log after all static init is finished; if an installed logger uses any Futures methods, it
69      // shouldn't break in cases where reflection is missing/broken.
70      if (thrownReflectionFailure != null) {
71        log.log(Level.SEVERE, "SafeAtomicHelper is broken!", thrownReflectionFailure);
72      }
73    }
74  
75    AggregateFutureState(int remainingFutures) {
76      this.remaining = remainingFutures;
77    }
78  
79    final Set<Throwable> getOrInitSeenExceptions() {
80      /*
81       * The initialization of seenExceptions has to be more complicated than we'd like. The simple
82       * approach would be for each caller CAS it from null to a Set populated with its exception. But
83       * there's another race: If the first thread fails with an exception and a second thread
84       * immediately fails with the same exception:
85       *
86       * Thread1: calls setException(), which returns true, context switch before it can CAS
87       * seenExceptions to its exception
88       *
89       * Thread2: calls setException(), which returns false, CASes seenExceptions to its exception,
90       * and wrongly believes that its exception is new (leading it to logging it when it shouldn't)
91       *
92       * Our solution is for threads to CAS seenExceptions from null to a Set populated with _the
93       * initial exception_, no matter which thread does the work. This ensures that seenExceptions
94       * always contains not just the current thread's exception but also the initial thread's.
95       */
96      Set<Throwable> seenExceptionsLocal = seenExceptions;
97      if (seenExceptionsLocal == null) {
98        // TODO(cpovirk): Should we use a simpler (presumably cheaper) data structure?
99        /*
100        * Using weak references here could let us release exceptions earlier, but:
101        *
102        * 1. On Android, querying a WeakReference blocks if the GC is doing an otherwise-concurrent
103        * pass.
104        *
105        * 2. We would probably choose to compare exceptions using == instead of equals() (for
106        * consistency with how weak references are cleared). That's a behavior change -- arguably the
107        * removal of a feature.
108        *
109        * Fortunately, exceptions rarely contain references to expensive resources.
110        */
111 
112       //
113       seenExceptionsLocal = newConcurrentHashSet();
114       /*
115        * Other handleException() callers may see this as soon as we publish it. We need to populate
116        * it with the initial failure before we do, or else they may think that the initial failure
117        * has never been seen before.
118        */
119       addInitialException(seenExceptionsLocal);
120 
121       ATOMIC_HELPER.compareAndSetSeenExceptions(this, null, seenExceptionsLocal);
122       /*
123        * If another handleException() caller created the set, we need to use that copy in case yet
124        * other callers have added to it.
125        *
126        * This read is guaranteed to get us the right value because we only set this once (here).
127        */
128       seenExceptionsLocal = seenExceptions;
129     }
130     return seenExceptionsLocal;
131   }
132 
133   /** Populates {@code seen} with the exception that was passed to {@code setException}. */
134   abstract void addInitialException(Set<Throwable> seen);
135 
136   final int decrementRemainingAndGet() {
137     return ATOMIC_HELPER.decrementAndGetRemainingCount(this);
138   }
139 
140   final void clearSeenExceptions() {
141     seenExceptions = null;
142   }
143 
144   private abstract static class AtomicHelper {
145     /** Atomic compare-and-set of the {@link AggregateFutureState#seenExceptions} field. */
146     abstract void compareAndSetSeenExceptions(
147         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update);
148 
149     /** Atomic decrement-and-get of the {@link AggregateFutureState#remaining} field. */
150     abstract int decrementAndGetRemainingCount(AggregateFutureState state);
151   }
152 
153   private static final class SafeAtomicHelper extends AtomicHelper {
154     final AtomicReferenceFieldUpdater<AggregateFutureState, Set<Throwable>> seenExceptionsUpdater;
155 
156     final AtomicIntegerFieldUpdater<AggregateFutureState> remainingCountUpdater;
157 
158     SafeAtomicHelper(
159         AtomicReferenceFieldUpdater seenExceptionsUpdater,
160         AtomicIntegerFieldUpdater remainingCountUpdater) {
161       this.seenExceptionsUpdater = seenExceptionsUpdater;
162       this.remainingCountUpdater = remainingCountUpdater;
163     }
164 
165     @Override
166     void compareAndSetSeenExceptions(
167         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update) {
168       seenExceptionsUpdater.compareAndSet(state, expect, update);
169     }
170 
171     @Override
172     int decrementAndGetRemainingCount(AggregateFutureState state) {
173       return remainingCountUpdater.decrementAndGet(state);
174     }
175   }
176 
177   private static final class SynchronizedAtomicHelper extends AtomicHelper {
178     @Override
179     void compareAndSetSeenExceptions(
180         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update) {
181       synchronized (state) {
182         if (state.seenExceptions == expect) {
183           state.seenExceptions = update;
184         }
185       }
186     }
187 
188     @Override
189     int decrementAndGetRemainingCount(AggregateFutureState state) {
190       synchronized (state) {
191         return --state.remaining;
192       }
193     }
194   }
195 }
