15  package com.google.common.collect;
16  
17  import static com.google.common.collect.Platform.checkGwtRpcEnabled;
18  
19  import com.google.gwt.user.client.rpc.SerializationException;
20  import com.google.gwt.user.client.rpc.SerializationStreamReader;
21  import com.google.gwt.user.client.rpc.SerializationStreamWriter;
22  
23  /**
24   * This class implements the GWT serialization of {@link SingletonImmutableTable}.
25   *
26   * @author Chris Povirk
27   */
28  public class SingletonImmutableTable_CustomFieldSerializer {
29    public static void deserialize(
30        SerializationStreamReader reader, SingletonImmutableTable<?, ?, ?> instance) {}
31  
32    public static SingletonImmutableTable<Object, Object, Object> instantiate(
33        SerializationStreamReader reader) throws SerializationException {
34      checkGwtRpcEnabled();
35      Object rowKey = reader.readObject();
36      Object columnKey = reader.readObject();
37      Object value = reader.readObject();
38      return new SingletonImmutableTable<>(rowKey, columnKey, value);
39    }
40  
41    public static void serialize(
42        SerializationStreamWriter writer, SingletonImmutableTable<?, ?, ?> instance)
43        throws SerializationException {
44      checkGwtRpcEnabled();
45      writer.writeObject(instance.singleRowKey);
46      writer.writeObject(instance.singleColumnKey);
47      writer.writeObject(instance.singleValue);
48    }
49  }
