43  /**
44   * Generates a test suite covering the {@link List} implementations in the {@link java.util}
45   * package. Can be subclassed to specify tests that should be suppressed.
46   *
47   * @author Kevin Bourrillion
48   */
49  @GwtIncompatible
50  public class TestsForListsInJavaUtil {
51    public static Test suite() {
52      return new TestsForListsInJavaUtil().allTests();
53    }
54  
55    public Test allTests() {
56      TestSuite suite = new TestSuite("java.util Lists");
57      suite.addTest(testsForEmptyList());
58      suite.addTest(testsForSingletonList());
59      suite.addTest(testsForArraysAsList());
60      suite.addTest(testsForArrayList());
61      suite.addTest(testsForLinkedList());
62      suite.addTest(testsForCopyOnWriteArrayList());
63      suite.addTest(testsForUnmodifiableList());
64      suite.addTest(testsForCheckedList());
65      suite.addTest(testsForAbstractList());
66      suite.addTest(testsForAbstractSequentialList());
67      suite.addTest(testsForVector());
68      return suite;
69    }
70

72      return Collections.emptySet();
73    }

91    protected Collection<Method> suppressForCopyOnWriteArrayList() {
92      return Arrays.asList(
93          getSubListOriginalListSetAffectsSubListMethod(),
94          getSubListOriginalListSetAffectsSubListLargeListMethod(),
95          getSubListSubListRemoveAffectsOriginalLargeListMethod(),

119   public Test testsForEmptyList() {
120     return ListTestSuiteBuilder.using(
121             new TestStringListGenerator() {
122               @Override
123               public List<String> create(String[] elements) {
124                 return Collections.emptyList();
125               }
126             })
127         .named("emptyList")
128         .withFeatures(CollectionFeature.SERIALIZABLE, CollectionSize.ZERO)
129         .suppressing(suppressForEmptyList())
130         .createTestSuite();
131   }

268   public Test testsForAbstractList() {
269     return ListTestSuiteBuilder.using(
270             new TestStringListGenerator() {
271               @Override
272               protected List<String> create(final String[] elements) {
273                 return new AbstractList<String>() {
274                   @Override
275                   public int size() {
276                     return elements.length;
277                   }
278 
279                   @Override
280                   public String get(int index) {
281                     return elements[index];
282                   }
283                 };
284               }
285             })
286         .named("AbstractList")
287         .withFeatures(
288             CollectionFeature.NONE, CollectionFeature.ALLOWS_NULL_VALUES, CollectionSize.ANY)
289         .suppressing(suppressForAbstractList())
290         .createTestSuite();
291   }

293   public Test testsForAbstractSequentialList() {
294     return ListTestSuiteBuilder.using(
295             new TestStringListGenerator() {
296               @Override
297               protected List<String> create(final String[] elements) {
298                 // For this test we trust ArrayList works
299                 final List<String> list = new ArrayList<>();
300                 Collections.addAll(list, elements);
301                 return new AbstractSequentialList<String>() {
302                   @Override
303                   public int size() {
304                     return list.size();
305                   }
306 
307                   @Override
308                   public ListIterator<String> listIterator(int index) {
309                     return list.listIterator(index);
310                   }
311                 };
312               }
313             })
314         .named("AbstractSequentialList")
315         .withFeatures(
316             ListFeature.GENERAL_PURPOSE, CollectionFeature.ALLOWS_NULL_VALUES, CollectionSize.ANY)
317         .suppressing(suppressForAbstractSequentialList())
318         .createTestSuite();
319   }
