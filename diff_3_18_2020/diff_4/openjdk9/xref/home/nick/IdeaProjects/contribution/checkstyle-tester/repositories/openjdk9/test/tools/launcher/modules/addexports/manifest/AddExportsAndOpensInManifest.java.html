24  /**
25   * @test
26   * @library /lib/testlibrary
27   * @modules jdk.compiler
28   * @build AddExportsAndOpensInManifest Test2 JarUtils jdk.testlibrary.*
29   * @compile --add-exports=java.base/jdk.internal.misc=ALL-UNNAMED Test1.java
30   * @run testng AddExportsAndOpensInManifest
31   * @summary Basic test for Add-Exports and Add-Opens attributes in the
32   *          manifest of a main application JAR
33   */
34  
35  import java.nio.file.Files;
36  import java.nio.file.Path;
37  import java.nio.file.Paths;
38  import java.util.jar.Attributes;
39  import java.util.jar.Manifest;
40  
41  import jdk.testlibrary.OutputAnalyzer;
42  import jdk.testlibrary.ProcessTools;
43  
44  import org.testng.annotations.Test;
45  import static org.testng.Assert.*;
46  
47  
48  @Test
49  public class AddExportsAndOpensInManifest {
50  
51      /**
52       * Package Test1 and Test2 into a JAR file with the given attributes
53       * in the JAR manifest, then execute the JAR file with `java -jar`.
54       */
55      private OutputAnalyzer runTest(String attributes) throws Exception {
56          Manifest man = new Manifest();
57          Attributes attrs = man.getMainAttributes();
58          attrs.put(Attributes.Name.MANIFEST_VERSION, "1.0");
59  
60          for (String nameAndValue : attributes.split(",")) {
61              String[] s = nameAndValue.split("=");
62              if (s.length != 2)
63                  throw new RuntimeException("Malformed: " + nameAndValue);
64              String name = s[0];
65              String value = s[1];
66              attrs.put(new Attributes.Name(name), value);
67          }
68  
69          // create the JAR file with Test1 and Test2
70          Path jarfile = Paths.get("test.jar");
71          Files.deleteIfExists(jarfile);
72  
73          Path classes = Paths.get(System.getProperty("test.classes", ""));
74          JarUtils.createJarFile(jarfile, man, classes,
75                  Paths.get("Test1.class"), Paths.get("Test2.class"));
76  
77          // java -jar test.jar
78          return ProcessTools.executeTestJava("--illegal-access=deny",
79                                              "-jar", jarfile.toString())
80                  .outputTo(System.out)
81                  .errorTo(System.out);
82      }
83  
84      /**
85       * Run test with the given JAR attributes, expecting the test to pass
86       */
87      private void runExpectingPass(String attrs) throws Exception {
88          int exitValue = runTest(attrs).getExitValue();
89          assertTrue(exitValue == 0);
90      }
91  
92      /**
93       * Run test with the given JAR attributes, expecting the test to fail
94       * with at least the given output
95       */
96      private void runExpectingFail(String attrs, String errorString) throws Exception {
97          int exitValue = runTest(attrs).shouldContain(errorString).getExitValue();
98          assertTrue(exitValue != 0);
99      }
100 
101 
102     /**
103      * Run tests to make sure that they fail in the expected way.
104      */
105     public void testSanity() throws Exception {
106         runExpectingFail("Main-Class=Test1", "IllegalAccessError");
107         runExpectingFail("Main-Class=Test2", "InaccessibleObjectException");
108     }
109 
110     /**
111      * Run tests with the Add-Exports attribute in the main manifest.
112      */
113     public void testWithAddExports() throws Exception {
114         runExpectingPass("Main-Class=Test1,Add-Exports=java.base/jdk.internal.misc");
115         runExpectingFail("Main-Class=Test2,Add-Exports=java.base/jdk.internal.misc",
116                          "InaccessibleObjectException");
117 
118         // run with leading and trailing spaces
119         runExpectingPass("Main-Class=Test1,Add-Exports=  java.base/jdk.internal.misc");
120         runExpectingPass("Main-Class=Test1,Add-Exports=java.base/jdk.internal.misc  ");
121 
122         // run with multiple values
123         runExpectingPass("Main-Class=Test1,Add-Exports=java.base/jdk.internal.misc"
124                 + " java.base/jdk.internal.loader");
125         runExpectingPass("Main-Class=Test1,Add-Exports=java.base/jdk.internal.loader"
126                 + " java.base/jdk.internal.misc");
127 
128         // run with duplicate values
129         runExpectingPass("Main-Class=Test1,Add-Exports=java.base/jdk.internal.misc"
130                 + " java.base/jdk.internal.misc");
131     }
132 
133     /**
134      * Run tests with the Add-Opens attribute in the main manifest.
135      */
136     public void testWithAddOpens() throws Exception {
137         runExpectingPass("Main-Class=Test1,Add-Opens=java.base/jdk.internal.misc");
138         runExpectingPass("Main-Class=Test2,Add-Opens=java.base/jdk.internal.misc");
139 
140         // run with leading and trailing spaces
141         runExpectingPass("Main-Class=Test1,Add-Opens=  java.base/jdk.internal.misc");
142         runExpectingPass("Main-Class=Test1,Add-Opens=java.base/jdk.internal.misc  ");
143 
144         // run with multiple values
145         runExpectingPass("Main-Class=Test1,Add-Opens=java.base/jdk.internal.misc"
146                 + " java.base/jdk.internal.loader");
147         runExpectingPass("Main-Class=Test1,Add-Opens=java.base/jdk.internal.loader"
148                 + " java.base/jdk.internal.misc");
149 
150         // run with duplicate values
151         runExpectingPass("Main-Class=Test1,Add-Opens=java.base/jdk.internal.misc"
152                 + " java.base/jdk.internal.misc");
153     }
154 
155     /**
156      * Run tests a bad module or package name
157      */
158     public void testWithBadModuleOrPackage() throws Exception {
159         // Add-Exports with bad module name
160         String attrs = "Main-Class=Test1,Add-Exports=java.DoesNotExist/jdk.internal.misc";
161         runExpectingFail(attrs, "IllegalAccessError");
162 
163         // Add-Exports with bad package name
164         attrs = "Main-Class=Test1,Add-Exports=java.base/jdk.internal.DoesNotExit";
165         runExpectingFail(attrs, "IllegalAccessError");
166 
167         // Add-Opens with bad module name
168         attrs = "Main-Class=Test1,Add-Opens=java.DoesNotExist/jdk.internal.misc";
169         runExpectingFail(attrs, "IllegalAccessError");
170 
171         // Add-Opens with bad package name
172         attrs = "Main-Class=Test1,Add-Opens=java.base/jdk.internal.DoesNotExit";
173         runExpectingFail(attrs, "IllegalAccessError");
174     }
