41  import static org.testng.Assert.assertEquals;
42  
43  public class MRTestBase {
44  
45      public static final int SUCCESS = 0;
46  
47      protected final String src = System.getProperty("test.src", ".");
48      protected final String usr = System.getProperty("user.dir", ".");
49  
50      protected void compile(String test) throws Throwable {
51          Path classes = Paths.get(usr, "classes", "base");
52          Files.createDirectories(classes);
53          Path source = Paths.get(src, "data", test, "base", "version");
54          javac(classes, source.resolve("Main.java"), source.resolve("Version.java"));
55  
56          classes = Paths.get(usr, "classes", "v9");
57          Files.createDirectories(classes);
58          source = Paths.get(src, "data", test, "v9", "version");
59          javac(classes, source.resolve("Version.java"));
60  
61          classes = Paths.get(usr, "classes", "v10");
62          Files.createDirectories(classes);
63          source = Paths.get(src, "data", test, "v10", "version");
64          javac(classes, source.resolve("Version.java"));
65      }
66  
67      protected void checkMultiRelease(String jarFile,
68                                       boolean expected) throws IOException {
69          try (JarFile jf = new JarFile(new File(jarFile), true,
70                  ZipFile.OPEN_READ, JarFile.runtimeVersion())) {
71              assertEquals(jf.isMultiRelease(), expected);
72          }
73      }
74  
75      // compares the bytes found in the jar entries with the bytes found in the
76      // corresponding data files used to create the entries
77      protected void compare(String jarfile,
78                             Map<String, String[]> names) throws IOException {
79          try (JarFile jf = new JarFile(jarfile)) {
80              for (String name : names.keySet()) {
81                  Path path = Paths.get("classes", names.get(name));
82                  byte[] b1 = Files.readAllBytes(path);
83                  byte[] b2;
84                  JarEntry je = jf.getJarEntry(name);
85                  try (InputStream is = jf.getInputStream(je)) {
86                      b2 = is.readAllBytes();
87                  }
88                  assertEquals(b1, b2);
89              }
90          }
91      }
92  
93      void javac(Path dest, Path... sourceFiles) throws Throwable {
94          String javac = JDKToolFinder.getJDKTool("javac");
95  
96          List<String> commands = new ArrayList<>();
97          commands.add(javac);
98          String opts = System.getProperty("test.compiler.opts");
99          if (!opts.isEmpty()) {
100             commands.addAll(Arrays.asList(opts.split(" +")));
101         }
102         commands.addAll(Utils.getForwardVmOptions());
103         commands.add("-d");
104         commands.add(dest.toString());
105         Stream.of(sourceFiles)
106                 .map(Object::toString)
107                 .forEach(x -> commands.add(x));
108 
109         ProcessTools.executeCommand(new ProcessBuilder(commands))
110                 .shouldHaveExitValue(SUCCESS);
111     }
112 
113     OutputAnalyzer jarWithStdin(File stdinSource,
114                                 String... args) throws Throwable {
115 
116         String jar = JDKToolFinder.getJDKTool("jar");
117         List<String> commands = new ArrayList<>();
118         commands.add(jar);
119         commands.addAll(Utils.getForwardVmOptions());
120         Stream.of(args).forEach(x -> commands.add(x));
121         ProcessBuilder p = new ProcessBuilder(commands);
122         if (stdinSource != null)
123             p.redirectInput(stdinSource);
124         return ProcessTools.executeCommand(p);
125     }
126 
127     OutputAnalyzer jar(String... args) throws Throwable {
128         return jarWithStdin(null, args);
129     }
