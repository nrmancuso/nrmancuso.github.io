120     @DataProvider
121     public Object[][] exchanges() throws Exception {
122         List<Object[]> values = new ArrayList<>();
123 
124         for (boolean async : new boolean[] { false, true })
125             for (String uri : new String[] { httpURI, httpsURI })
126                 for (String file : new String[] { smallFilename, midSizedFilename })
127                     for (RequestBody requestBodyType : RequestBody.values())
128                         for (ResponseBody responseBodyType : ResponseBody.values())

135     @Test(dataProvider = "exchanges")
136     void exchange(String target,
137                   RequestBody requestBodyType,
138                   ResponseBody responseBodyType,
139                   String file,
140                   boolean async)
141         throws Exception
142     {
143         Path filePath = Paths.get(fileroot + file);
144         URI uri = new URI(target);
145 
146         HttpRequest request = createRequest(uri, requestBodyType, filePath);
147 
148         checkResponse(client, request, requestBodyType, responseBodyType, filePath, async);
149     }
150 
151     static final int DEFAULT_OFFSET = 10;
152     static final int DEFAULT_LENGTH = 1000;
153 
154     HttpRequest createRequest(URI uri,
155                               RequestBody requestBodyType,
156                               Path file)
157         throws IOException
158     {
159         HttpRequest.Builder rb =  HttpRequest.newBuilder(uri);
160 
161         String filename = file.toFile().getAbsolutePath();
162         byte[] fileAsBytes = getFileBytes(filename);
163         String fileAsString = new String(fileAsBytes, UTF_8);
164 
165         switch (requestBodyType) {
166             case BYTE_ARRAY:
167                 rb.POST(fromByteArray(fileAsBytes));
168                 break;
169             case BYTE_ARRAY_OFFSET:
170                 rb.POST(fromByteArray(fileAsBytes, DEFAULT_OFFSET, DEFAULT_LENGTH));
171                 break;
172             case BYTE_ARRAYS:
173                 Iterable<byte[]> iterable = Arrays.asList(fileAsBytes);
174                 rb.POST(fromByteArrays(iterable));
175                 break;
176             case FILE:
177                 rb.POST(fromFile(file));
178                 break;
179             case INPUTSTREAM:
180                 rb.POST(fromInputStream(fileInputStreamSupplier(file)));
181                 break;
182             case STRING:
183                 rb.POST(fromString(fileAsString));
184                 break;
185             case STRING_WITH_CHARSET:
186                 rb.POST(fromString(new String(fileAsBytes), Charset.defaultCharset()));
187                 break;
188             default:
189                 throw new AssertionError("Unknown request body:" + requestBodyType);
190         }
191         return rb.build();
192     }

194     void checkResponse(HttpClient client,
195                        HttpRequest request,
196                        RequestBody requestBodyType,
197                        ResponseBody responseBodyType,
198                        Path file,
199                        boolean async)
200         throws InterruptedException, IOException
201     {
202         String filename = file.toFile().getAbsolutePath();
203         byte[] fileAsBytes = getFileBytes(filename);
204         if (requestBodyType == RequestBody.BYTE_ARRAY_OFFSET) {
205             // Truncate the expected response body, if only a portion was sent
206             fileAsBytes = Arrays.copyOfRange(fileAsBytes,
207                                              DEFAULT_OFFSET,
208                                              DEFAULT_OFFSET + DEFAULT_LENGTH);
209         }
210         String fileAsString = new String(fileAsBytes, UTF_8);
211         Path tempFile = Paths.get("RequestBodyTest.tmp");
212         FileUtils.deleteFileIfExistsWithRetry(tempFile);
213 
214         switch (responseBodyType) {
215             case BYTE_ARRAY:
216                 HttpResponse<byte[]> bar = getResponse(client, request, asByteArray(), async);
217                 assertEquals(bar.statusCode(), 200);
218                 assertEquals(bar.body(), fileAsBytes);
219                 break;
220             case BYTE_ARRAY_CONSUMER:
221                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
222                 HttpResponse<Void> v = getResponse(client, request,
223                         asByteArrayConsumer(o -> consumerBytes(o, baos) ), async);
224                 byte[] ba = baos.toByteArray();
225                 assertEquals(v.statusCode(), 200);
226                 assertEquals(ba, fileAsBytes);
227                 break;
228             case DISCARD:
229                 Object o = new Object();
230                 HttpResponse<Object> or = getResponse(client, request, discard(o), async);
231                 assertEquals(or.statusCode(), 200);
232                 assertSame(or.body(), o);
233                 break;
234             case FILE:
235                 HttpResponse<Path> fr = getResponse(client, request, asFile(tempFile), async);
236                 assertEquals(fr.statusCode(), 200);
237                 assertEquals(Files.size(tempFile), fileAsString.length());
238                 assertEquals(Files.readAllBytes(tempFile), fileAsBytes);
239                 break;
240             case FILE_WITH_OPTION:
241                 fr = getResponse(client, request, asFile(tempFile, CREATE_NEW, WRITE), async);
242                 assertEquals(fr.statusCode(), 200);
243                 assertEquals(Files.size(tempFile), fileAsString.length());
244                 assertEquals(Files.readAllBytes(tempFile), fileAsBytes);
245                 break;
246             case STRING:
247                 HttpResponse<String> sr = getResponse(client, request, asString(), async);
248                 assertEquals(sr.statusCode(), 200);
249                 assertEquals(sr.body(), fileAsString);
250                 break;
251             case STRING_WITH_CHARSET:
252                 HttpResponse<String> r = getResponse(client, request, asString(StandardCharsets.UTF_8), async);
253                 assertEquals(r.statusCode(), 200);
254                 assertEquals(r.body(), fileAsString);
255                 break;
256             default:
257                 throw new AssertionError("Unknown response body:" + responseBodyType);
258         }
259     }

261     static <T> HttpResponse<T> getResponse(HttpClient client,
262                                            HttpRequest request,
263                                            HttpResponse.BodyHandler<T> handler,
264                                            boolean async)
265         throws InterruptedException, IOException
266     {
267         if (!async)
268             return client.send(request, handler);
269         else
270             return client.sendAsync(request, handler).join();
271     }

273     static byte[] getFileBytes(String path) throws IOException {
274         try (FileInputStream fis = new FileInputStream(path);
275              BufferedInputStream bis = new BufferedInputStream(fis);
276              ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
277             bis.transferTo(baos);
278             return baos.toByteArray();
279         }
280     }
281 
282     static Supplier<FileInputStream> fileInputStreamSupplier(Path f) {
283         return new Supplier<>() {
284             Path file = f;
285             @Override
286             public FileInputStream get() {
287                 try {
288                     return new FileInputStream(file.toFile());
289                 } catch (FileNotFoundException x) {
290                     throw new UncheckedIOException(x);
291                 }
292             }
293         };
294     }
295 
296     static void consumerBytes(Optional<byte[]> bytes, ByteArrayOutputStream baos) {
297         try {
298             if (bytes.isPresent())
299                 baos.write(bytes.get());
300         } catch (IOException x) {
301             throw new UncheckedIOException(x);
302         }
303     }
