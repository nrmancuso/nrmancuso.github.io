76  /**
77   * <p>Parses command line arguments, using a syntax that attempts to take from the best of POSIX {@code getopt()}
78   * and GNU {@code getopt_long()}.</p>
79   *
80   * <p>This parser supports short options and long options.</p>
81   *
82   * <ul>
83   *   <li><dfn>Short options</dfn> begin with a single hyphen ("<kbd>-</kbd>") followed by a single letter or digit,
84   *   or question mark ("<kbd>?</kbd>"), or dot ("<kbd>.</kbd>").</li>
85   *
86   *   <li>Short options can accept single arguments. The argument can be made required or optional. The option's
87   *   argument can occur:
88   *     <ul>
89   *       <li>in the slot after the option, as in <kbd>-d /tmp</kbd></li>
90   *       <li>right up against the option, as in <kbd>-d/tmp</kbd></li>
91   *       <li>right up against the option separated by an equals sign (<kbd>"="</kbd>), as in <kbd>-d=/tmp</kbd></li>
92   *     </ul>
93   *   To specify <em>n</em> arguments for an option, specify the option <em>n</em> times, once for each argument,
94   *   as in <kbd>-d /tmp -d /var -d /opt</kbd>; or, when using the
95   *   {@linkplain ArgumentAcceptingOptionSpec#withValuesSeparatedBy(char) "separated values"} clause of the "fluent
96   *   interface" (see below), give multiple values separated by a given character as a single argument to the
97   *   option.</li>
98   *
99   *   <li>Short options can be clustered, so that <kbd>-abc</kbd> is treated as <kbd>-a -b -c</kbd>. If a short option
100  *   in the cluster can accept an argument, the remaining characters are interpreted as the argument for that
101  *   option.</li>
102  *
103  *   <li>An argument consisting only of two hyphens (<kbd>"--"</kbd>) signals that the remaining arguments are to be
104  *   treated as non-options.</li>
105  *
106  *   <li>An argument consisting only of a single hyphen is considered a non-option argument (though it can be an
107  *   argument of an option). Many Unix programs treat single hyphens as stand-ins for the standard input or standard
108  *   output streams.</li>
109  *
110  *   <li><dfn>Long options</dfn> begin with two hyphens (<kbd>"--"</kbd>), followed by multiple letters, digits,
111  *   hyphens, question marks, or dots. A hyphen cannot be the first character of a long option specification when
112  *   configuring the parser.</li>
113  *
114  *   <li>You can abbreviate long options, so long as the abbreviation is unique.</li>
115  *
116  *   <li>Long options can accept single arguments.  The argument can be made required or optional.  The option's
117  *   argument can occur:
118  *     <ul>
119  *       <li>in the slot after the option, as in <kbd>--directory /tmp</kbd></li>
120  *       <li>right up against the option separated by an equals sign (<kbd>"="</kbd>), as in
121  *       <kbd>--directory=/tmp</kbd>
122  *     </ul>
123  *   Specify multiple arguments for a long option in the same manner as for short options (see above).</li>
124  *
125  *   <li>You can use a single hyphen (<kbd>"-"</kbd>) instead of a double hyphen (<kbd>"--"</kbd>) for a long
126  *   option.</li>
127  *
128  *   <li>The option <kbd>-W</kbd> is reserved.  If you tell the parser to {@linkplain
129  *   #recognizeAlternativeLongOptions(boolean) recognize alternative long options}, then it will treat, for example,
130  *   <kbd>-W foo=bar</kbd> as the long option <kbd>foo</kbd> with argument <kbd>bar</kbd>, as though you had written
131  *   <kbd>--foo=bar</kbd>.</li>
132  *
133  *   <li>You can specify <kbd>-W</kbd> as a valid short option, or use it as an abbreviation for a long option, but
134  *   {@linkplain #recognizeAlternativeLongOptions(boolean) recognizing alternative long options} will always supersede
135  *   this behavior.</li>
136  *
137  *   <li>You can specify a given short or long option multiple times on a single command line. The parser collects
138  *   any arguments specified for those options as a list.</li>
139  *
140  *   <li>If the parser detects an option whose argument is optional, and the next argument "looks like" an option,
141  *   that argument is not treated as the argument to the option, but as a potentially valid option. If, on the other
142  *   hand, the optional argument is typed as a derivative of {@link Number}, then that argument is treated as the
143  *   negative number argument of the option, even if the parser recognizes the corresponding numeric option.
144  *   For example:
145  *   <pre><code>
146  *     OptionParser parser = new OptionParser();
147  *     parser.accepts( "a" ).withOptionalArg().ofType( Integer.class );
148  *     parser.accepts( "2" );
149  *     OptionSet options = parser.parse( "-a", "-2" );
150  *   </code></pre>
151  *   In this case, the option set contains <kbd>"a"</kbd> with argument <kbd>-2</kbd>, not both <kbd>"a"</kbd> and
152  *   <kbd>"2"</kbd>. Swapping the elements in the <em>args</em> array gives the latter.</li>
153  * </ul>
154  *
155  * <p>There are two ways to tell the parser what options to recognize:</p>
156  *
157  * <ol>
158  *   <li>A "fluent interface"-style API for specifying options, available since version 2. Sentences in this fluent
159  *   interface language begin with a call to {@link #accepts(String) accepts} or {@link #acceptsAll(Collection)
160  *   acceptsAll} methods; calls on the ensuing chain of objects describe whether the options can take an argument,
161  *   whether the argument is required or optional, to what type arguments of the options should be converted if any,
162  *   etc. Since version 3, these calls return an instance of {@link OptionSpec}, which can subsequently be used to
163  *   retrieve the arguments of the associated option in a type-safe manner.</li>
164  *
165  *   <li>Since version 1, a more concise way of specifying short options has been to use the special {@linkplain
166  *   #OptionParser(String) constructor}. Arguments of options specified in this manner will be of type {@link String}.
167  *   Here are the rules for the format of the specification strings this constructor accepts:
168  *
169  *     <ul>
170  *       <li>Any letter or digit is treated as an option character.</li>
171  *
172  *       <li>An option character can be immediately followed by an asterisk (*) to indicate that the option is a
173  *       "help" option.</li>
174  *
175  *       <li>If an option character (with possible trailing asterisk) is followed by a single colon (<kbd>":"</kbd>),
176  *       then the option requires an argument.</li>
177  *
178  *       <li>If an option character (with possible trailing asterisk) is followed by two colons (<kbd>"::"</kbd>),
179  *       then the option accepts an optional argument.</li>
180  *
181  *       <li>Otherwise, the option character accepts no argument.</li>
182  *
183  *       <li>If the option specification string begins with a plus sign (<kbd>"+"</kbd>), the parser will behave
184  *       "POSIX-ly correct".</li>
185  *
186  *       <li>If the option specification string contains the sequence <kbd>"W;"</kbd> (capital W followed by a
187  *       semicolon), the parser will recognize the alternative form of long options.</li>
188  *     </ul>
189  *   </li>
190  * </ol>
191  *
192  * <p>Each of the options in a list of options given to {@link #acceptsAll(Collection) acceptsAll} is treated as a
193  * synonym of the others.  For example:
194  *   <pre>
195  *     <code>
196  *     OptionParser parser = new OptionParser();
197  *     parser.acceptsAll( asList( "w", "interactive", "confirmation" ) );
198  *     OptionSet options = parser.parse( "-w" );
199  *     </code>
200  *   </pre>
201  * In this case, <code>options.{@link OptionSet#has(String) has}</code> would answer {@code true} when given arguments
202  * <kbd>"w"</kbd>, <kbd>"interactive"</kbd>, and <kbd>"confirmation"</kbd>. The {@link OptionSet} would give the same
203  * responses to these arguments for its other methods as well.</p>
204  *
205  * <p>By default, as with GNU {@code getopt()}, the parser allows intermixing of options and non-options. If, however,
206  * the parser has been created to be "POSIX-ly correct", then the first argument that does not look lexically like an
207  * option, and is not a required argument of a preceding option, signals the end of options. You can still bind
208  * optional arguments to their options using the abutting (for short options) or <kbd>=</kbd> syntax.</p>
209  *
210  * <p>Unlike GNU {@code getopt()}, this parser does not honor the environment variable {@code POSIXLY_CORRECT}.
211  * "POSIX-ly correct" parsers are configured by either:</p>
212  *
213  * <ol>
214  *   <li>using the method {@link #posixlyCorrect(boolean)}, or</li>
215  *
216  *   <li>using the {@linkplain #OptionParser(String) constructor} with an argument whose first character is a plus sign
217  *   (<kbd>"+"</kbd>)</li>
218  * </ol>
219  *
220  * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
221  * @see <a href="http://www.gnu.org/software/libc/manual">The GNU C Library</a>
222  */

377     /**
378      * Parses the given command line arguments according to the option specifications given to the parser.
379      *
380      * @param arguments arguments to parse
381      * @return an {@link OptionSet} describing the parsed options, their arguments, and any non-option arguments found
382      * @throws OptionException if problems are detected while parsing
383      * @throws NullPointerException if the argument list is {@code null}
384      */
385     public OptionSet parse( String... arguments ) {
386         ArgumentList argumentList = new ArgumentList( arguments );
387         OptionSet detected = new OptionSet( recognizedOptions.toJavaUtilMap() );
388         detected.add( recognizedOptions.get( NonOptionArgumentSpec.NAME ) );
389 
390         while ( argumentList.hasMore() )
391             state.handleArgument( this, argumentList, detected );
392 
393         reset();
394 
395         ensureRequiredOptions( detected );
396 
397         return detected;
398     }
399 
400     private void ensureRequiredOptions( OptionSet options ) {
401         Collection<String> missingRequiredOptions = missingRequiredOptions( options );
402         boolean helpOptionPresent = isHelpOptionPresent( options );
403 
404         if ( !missingRequiredOptions.isEmpty() && !helpOptionPresent )
405             throw new MissingRequiredOptionException( missingRequiredOptions );
406     }
407 
408     private Collection<String> missingRequiredOptions( OptionSet options ) {
409         Collection<String> missingRequiredOptions = new HashSet<String>();
410 
411         for ( AbstractOptionSpec<?> each : recognizedOptions.toJavaUtilMap().values() ) {
412             if ( each.isRequired() && !options.has( each ) )
413                 missingRequiredOptions.addAll( each.options() );
414         }
415 
416         for ( Map.Entry<Collection<String>, Set<OptionSpec<?>>> eachEntry : requiredIf.entrySet() ) {
417             AbstractOptionSpec<?> required = specFor( eachEntry.getKey().iterator().next() );
418 
419             if ( optionsHasAnyOf( options, eachEntry.getValue() ) && !options.has( required ) ) {
420                 missingRequiredOptions.addAll( required.options() );
421             }
422         }
423 
424         for ( Map.Entry<Collection<String>, Set<OptionSpec<?>>> eachEntry : requiredUnless.entrySet() ) {
425             AbstractOptionSpec<?> required = specFor( eachEntry.getKey().iterator().next() );
426 
427             if ( !optionsHasAnyOf( options, eachEntry.getValue() ) && !options.has( required ) ) {
428                 missingRequiredOptions.addAll( required.options() );
429             }
430         }
431 
432         return missingRequiredOptions;
433     }
434 
435     private boolean optionsHasAnyOf( OptionSet options, Collection<OptionSpec<?>> specs ) {
436         for ( OptionSpec<?> each : specs ) {
437             if ( options.has( each ) )
438                 return true;
439         }
440 
441         return false;
442     }
443 
444     private boolean isHelpOptionPresent( OptionSet options ) {
445         boolean helpOptionPresent = false;
446         for ( AbstractOptionSpec<?> each : recognizedOptions.toJavaUtilMap().values() ) {
447             if ( each.isForHelp() && options.has( each ) ) {
448                 helpOptionPresent = true;
449                 break;
450             }
451         }
452         return helpOptionPresent;
453     }

455     void handleLongOptionToken( String candidate, ArgumentList arguments, OptionSet detected ) {
456         KeyValuePair optionAndArgument = parseLongOptionWithArgument( candidate );
457 
458         if ( !isRecognized( optionAndArgument.key ) )
459             throw unrecognizedOption( optionAndArgument.key );
460 
461         AbstractOptionSpec<?> optionSpec = specFor( optionAndArgument.key );
462         optionSpec.handleOption( this, arguments, detected, optionAndArgument.value );
463     }
464 
465     void handleShortOptionToken( String candidate, ArgumentList arguments, OptionSet detected ) {
466         KeyValuePair optionAndArgument = parseShortOptionWithArgument( candidate );
467 
468         if ( isRecognized( optionAndArgument.key ) ) {
469             specFor( optionAndArgument.key ).handleOption( this, arguments, detected, optionAndArgument.value );
470         }
471         else
472             handleShortOptionCluster( candidate, arguments, detected );
473     }
474 
475     private void handleShortOptionCluster( String candidate, ArgumentList arguments, OptionSet detected ) {
476         char[] options = extractShortOptionsFrom( candidate );
477         validateOptionCharacters( options );
478 
479         for ( int i = 0; i < options.length; i++ ) {
480             AbstractOptionSpec<?> optionSpec = specFor( options[ i ] );
481 
482             if ( optionSpec.acceptsArguments() && options.length > i + 1 ) {
483                 String detectedArgument = String.valueOf( options, i + 1, options.length - 1 - i );
484                 optionSpec.handleOption( this, arguments, detected, detectedArgument );
485                 break;
486             }
487 
488             optionSpec.handleOption( this, arguments, detected, null );
489         }
490     }
491 
492     void handleNonOptionArgument( String candidate, ArgumentList arguments, OptionSet detectedOptions ) {
493         specFor( NonOptionArgumentSpec.NAME ).handleOption( this, arguments, detectedOptions, candidate );
494     }

496     void noMoreOptions() {
497         state = OptionParserState.noMoreOptions();
498     }

500     boolean looksLikeAnOption( String argument ) {

502     }

518     }

522     }
523 
524     private void putRequiredOption( Collection<String> precedentSynonyms, OptionSpec<?> required,
525         Map<Collection<String>, Set<OptionSpec<?>>> target ) {
526 
527         for ( String each : precedentSynonyms ) {
528             AbstractOptionSpec<?> spec = specFor( each );
529             if ( spec == null )
530                 throw new UnconfiguredOptionException( precedentSynonyms );
531         }
532 
533         Set<OptionSpec<?>> associated = target.get( precedentSynonyms );
534         if ( associated == null ) {
535             associated = new HashSet<OptionSpec<?>>();
536             target.put( precedentSynonyms, associated );
537         }
538 
539         associated.add( required );
540     }

542     private AbstractOptionSpec<?> specFor( char option ) {
543         return specFor( String.valueOf( option ) );
544     }
545 
546     private AbstractOptionSpec<?> specFor( String option ) {
547         return recognizedOptions.get( option );
548     }
549 
550     private void reset() {
551         state = moreOptions( posixlyCorrect );
552     }
553 
554     private static char[] extractShortOptionsFrom( String argument ) {
555         char[] options = new char[ argument.length() - 1 ];
556         argument.getChars( 1, argument.length(), options, 0 );
557 
558         return options;
559     }
560 
561     private void validateOptionCharacters( char[] options ) {
562         for ( char each : options ) {
563             String option = String.valueOf( each );
564 
565             if ( !isRecognized( option ) )
566                 throw unrecognizedOption( option );
567 
568             if ( specFor( option ).acceptsArguments() )
569                 return;
570         }
571     }
572 
573     private static KeyValuePair parseLongOptionWithArgument( String argument ) {
574         return KeyValuePair.valueOf( argument.substring( 2 ) );
575     }
576 
577     private static KeyValuePair parseShortOptionWithArgument( String argument ) {
578         return KeyValuePair.valueOf( argument.substring( 1 ) );
579     }
