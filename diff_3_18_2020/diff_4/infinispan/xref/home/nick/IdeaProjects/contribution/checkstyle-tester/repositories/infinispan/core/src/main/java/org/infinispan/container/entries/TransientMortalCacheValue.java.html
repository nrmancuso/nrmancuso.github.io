1   package org.infinispan.container.entries;
2   
3   import org.infinispan.commons.io.UnsignedNumeric;
4   import org.infinispan.commons.marshall.AbstractExternalizer;
5   import org.infinispan.commons.util.Util;
6   import org.infinispan.marshall.core.Ids;
7   
8   import java.io.IOException;
9   import java.io.ObjectInput;
10  import java.io.ObjectOutput;
11  import java.util.Set;
12  
13  import static java.lang.Math.min;
14  
15  /**
16   * A transient, mortal cache value to correspond with {@link org.infinispan.container.entries.TransientMortalCacheEntry}
17   *
18   * @author Manik Surtani
19   * @since 4.0
20   */
21  public class TransientMortalCacheValue extends MortalCacheValue {
22     protected long maxIdle = -1;
23     protected long lastUsed;
24  
25     public TransientMortalCacheValue(Object value, long created, long lifespan, long maxIdle, long lastUsed) {
26        this(value, created, lifespan, maxIdle);
27        this.lastUsed = lastUsed;
28     }
29  
30     public TransientMortalCacheValue(Object value, long created, long lifespan, long maxIdle) {
31        super(value, created, lifespan);
32        this.maxIdle = maxIdle;
33     }
34  
35     @Override
36     public long getMaxIdle() {
37        return maxIdle;
38     }
39  
40     public void setMaxIdle(long maxIdle) {
41        this.maxIdle = maxIdle;
42     }
43  
44     @Override
45     public long getLastUsed() {
46        return lastUsed;
47     }
48  
49     public void setLastUsed(long lastUsed) {
50        this.lastUsed = lastUsed;
51     }
52  
53     @Override
54     public boolean isExpired(long now) {
55        return ExpiryHelper.isExpiredTransientMortal(maxIdle, lastUsed, lifespan, created, now);
56     }
57  
58     @Override
59     public boolean isExpired() {
60        return isExpired(System.currentTimeMillis());
61     }
62  
63     @Override
64     public InternalCacheEntry toInternalCacheEntry(Object key) {
65        return new TransientMortalCacheEntry(key, value, maxIdle, lifespan, lastUsed, created);
66     }
67  
68     @Override
69     public long getExpiryTime() {
70        long lset = lifespan > -1 ? created + lifespan : -1;
71        long muet = maxIdle > -1 ? lastUsed + maxIdle : -1;
72        if (lset == -1) return muet;
73        if (muet == -1) return lset;
74        return min(lset, muet);
75     }
76  
77     @Override
78     public boolean equals(Object o) {
79        if (this == o) return true;
80        if (!(o instanceof TransientMortalCacheValue)) return false;
81        if (!super.equals(o)) return false;
82  
83        TransientMortalCacheValue that = (TransientMortalCacheValue) o;
84  
85        if (lastUsed != that.lastUsed) return false;
86        if (maxIdle != that.maxIdle) return false;
87  
88        return true;
89     }
90  
91     @Override
92     public int hashCode() {
93        int result = super.hashCode();
94        result = 31 * result + (int) (maxIdle ^ (maxIdle >>> 32));
95        result = 31 * result + (int) (lastUsed ^ (lastUsed >>> 32));
96        return result;
97     }
98  
99     @Override
100    public String toString() {
101       return "TransientMortalCacheValue{" +
102             "maxIdle=" + maxIdle +
103             ", lastUsed=" + lastUsed +
104             "} " + super.toString();
105    }
106 
107    @Override
108    public TransientMortalCacheValue clone() {
109       return (TransientMortalCacheValue) super.clone();
110    }
111 
112    public static class Externalizer extends AbstractExternalizer<TransientMortalCacheValue> {
113       @Override
114       public void writeObject(ObjectOutput output, TransientMortalCacheValue value) throws IOException {
115          output.writeObject(value.value);
116          UnsignedNumeric.writeUnsignedLong(output, value.created);
117          output.writeLong(value.lifespan); // could be negative so should not use unsigned longs
118          UnsignedNumeric.writeUnsignedLong(output, value.lastUsed);
119          output.writeLong(value.maxIdle); // could be negative so should not use unsigned longs
120       }
121 
122       @Override
123       public TransientMortalCacheValue readObject(ObjectInput input) throws IOException, ClassNotFoundException {
124          Object v = input.readObject();
125          long created = UnsignedNumeric.readUnsignedLong(input);
126          Long lifespan = input.readLong();
127          long lastUsed = UnsignedNumeric.readUnsignedLong(input);
128          Long maxIdle = input.readLong();
129          return new TransientMortalCacheValue(v, created, lifespan, maxIdle, lastUsed);
130       }
131 
132       @Override
133       public Integer getId() {
134          return Ids.TRANSIENT_MORTAL_VALUE;
135       }
136 
137       @Override
138       public Set<Class<? extends TransientMortalCacheValue>> getTypeClasses() {
139          return Util.<Class<? extends TransientMortalCacheValue>>asSet(TransientMortalCacheValue.class);
140       }
141    }
142 }
