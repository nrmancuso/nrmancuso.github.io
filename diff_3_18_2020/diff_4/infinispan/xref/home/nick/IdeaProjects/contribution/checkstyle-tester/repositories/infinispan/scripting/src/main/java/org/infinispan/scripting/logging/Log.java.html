1   package org.infinispan.scripting.logging;
2   
3   import static org.jboss.logging.Logger.Level.ERROR;
4   
5   import org.infinispan.commons.CacheException;
6   import org.jboss.logging.annotations.Cause;
7   import org.jboss.logging.annotations.LogMessage;
8   import org.jboss.logging.annotations.Message;
9   import org.jboss.logging.annotations.MessageLogger;
10  
11  /**
12   * Log abstraction for the Scripting module. For this module, message ids ranging from 21001 to
13   * 22000 inclusively have been reserved.
14   *
15   * @author Tristan Tarrant
16   * @since 7.2
17   */
18  @MessageLogger(projectCode = "ISPN")
19  public interface Log extends org.infinispan.util.logging.Log {
20     @LogMessage(level = ERROR)
21     @Message(value = "Could not register interpreter MBean", id = 21001)
22     void jmxRegistrationFailed();
23  
24     @LogMessage(level = ERROR)
25     @Message(value = "Could not unregister interpreter MBean", id = 21002)
26     void jmxUnregistrationFailed();
27  
28     @Message(value = "Script execution error", id = 21003)
29     CacheException scriptExecutionError(@Cause Throwable t);
30  
31     @Message(value = "Compiler error for script '%s'", id = 21004)
32     CacheException scriptCompilationException(@Cause Throwable t, String name);
33  
34     @Message(value = "No script named '%s'", id = 21005)
35     CacheException noNamedScript(String name);
36  
37     @Message(value = "Unknown script mode: '%s'", id = 21006)
38     CacheException unknownScriptProperty(String value);
39  
40     @Message(value = "Cannot find an appropriate script engine for '%s'", id = 21007)
41     IllegalArgumentException noScriptEngineForScript(String name);
42  
43     @Message(value = "Script '%s' cannot be invoked directly since it specifies mode '%s'", id = 21008)
44     IllegalArgumentException cannotInvokeScriptDirectly(String scriptName, String property);
45  
46     @Message(value = "Distributed script '%s' invoked without a cache binding", id = 21009)
47     IllegalStateException distributedTaskNeedCacheInBinding(String scriptName);
48  
49     @Message(value = "Cannot find an appropriate script engine for script '%s'", id = 21010)
50     IllegalArgumentException noEngineForScript(String name);
51  }
