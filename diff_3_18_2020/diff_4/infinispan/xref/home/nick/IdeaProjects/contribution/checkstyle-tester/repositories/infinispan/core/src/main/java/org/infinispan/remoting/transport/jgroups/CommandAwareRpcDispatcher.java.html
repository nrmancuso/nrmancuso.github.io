75     public CommandAwareRpcDispatcher(Channel channel,
76           JGroupsTransport transport,
77           ExecutorService asyncExecutor,
78           TimeService timeService,
79           InboundInvocationHandler globalHandler) {
80        this.server_obj = transport;
81        this.asyncExecutor = asyncExecutor;
82        this.transport = transport;
83        this.timeService = timeService;
84        this.handler = globalHandler;
85  
86        // MessageDispatcher superclass constructors will call start() so perform all init here
87        this.setMembershipListener(transport);
88        this.setChannel(channel);
89        // If existing up handler is a muxing up handler, setChannel(..) will not have replaced it
90        UpHandler handler = channel.getUpHandler();
91        if (handler instanceof Muxer<?>) {
92           @SuppressWarnings("unchecked")
93           Muxer<UpHandler> mux = (Muxer<UpHandler>) handler;
94           mux.setDefaultHandler(this.prot_adapter);
95        }
96        channel.addChannelListener(this);
97        asyncDispatching(true);

133    /**
134     * @param recipients Guaranteed not to be null.  Must <b>not</b> contain self.
135     */
136    public RspList<Object> invokeRemoteCommands(final List<Address> recipients, final ReplicableCommand command,
137                                                final ResponseMode mode, final long timeout, final RspFilter filter,
138                                                final DeliverOrder deliverOrder, boolean asyncMarshalling,
139                                                final boolean ignoreLeavers) throws InterruptedException {
140       if (asyncMarshalling) {
141          asyncExecutor.submit(new Callable<RspList<Object>>() {
142             @Override
143             public RspList<Object> call() throws Exception {
144                return processCalls(command, recipients == null, timeout, filter, recipients, mode, deliverOrder,
145                                    req_marshaller, CommandAwareRpcDispatcher.this, ignoreLeavers);
146             }
147          });
148          return null; // don't wait for a response!
149       } else {
150          RspList<Object> response;
151          try {
152             response = processCalls(command, recipients == null, timeout, filter, recipients, mode, deliverOrder,
153                                     req_marshaller, this, ignoreLeavers);
154          } catch (InterruptedException e) {
155             throw e;
156          } catch (SuspectedException e) {
157             throw new SuspectException("One of the nodes " + recipients + " was suspected", e);
158          } catch (org.jgroups.TimeoutException e) {
159             throw new TimeoutException("One of the nodes " + recipients + " timed out", e);
160          } catch (Exception e) {
161             throw rewrapAsCacheException(e);
162          }
163          if (mode == ResponseMode.GET_NONE) return null; // "Traditional" async.
164          if (response.isEmpty() || containsOnlyNulls(response))
165             return null;
166          else
167             return response;
168       }
169    }

171    /**
172     * Runs a command on each node for each entry in the provided map.
173     * NOTE: if ignoreLeavers is true and the node is suspected while executing this 
174     * method will return a RspList containing for that node a value of ExceptionResponse
175     * containing the SuspectException
176     * 
177     * @param commands The commands and where to run them
178     * @param mode The response mode to determine how many members must return
179     * @param timeout How long to wait before timing out
180     * @param oob Whether these should be submitted using out of band thread pool
181     * @param asyncMarshalling If marshalling should be done asynchronously
182     * @param ignoreLeavers Whether to ignore leavers.  If this is true and a node leaves
183     *        it will send back a response of SuspectException
184     * @return The responses that came back in the provided time
185     * @throws InterruptedException
186     */
187    public RspList<Object> invokeRemoteCommands(final Map<Address, ReplicableCommand> commands, final ResponseMode mode, final long timeout,
188                                                final boolean oob, boolean asyncMarshalling, final boolean ignoreLeavers) throws InterruptedException {
189       if (asyncMarshalling) {
190          asyncExecutor.submit(new Callable<RspList<Object>>() {
191             @Override
192             public RspList<Object> call() throws Exception {
193                return processCalls(commands, timeout, mode, req_marshaller,
194                      CommandAwareRpcDispatcher.this, oob, ignoreLeavers);
195             }
196          });
197          return null; // don't wait for a response!
198       } else {
199          try {
200             return processCalls(commands, timeout, mode,
201                   req_marshaller, this, oob, ignoreLeavers);
202          } catch (InterruptedException e) {
203             throw e;
204          } catch (SuspectedException e) {
205             throw new SuspectException("One of the nodes " + commands.keySet() + " was suspected", e);
206          } catch (org.jgroups.TimeoutException e) {
207             // TODO consider ignoreTimeout flag
208             throw new TimeoutException("One of the nodes " + commands.keySet() + " timed out", e);
209          } catch (Exception e) {
210             throw rewrapAsCacheException(e);
211          }
212       }
213    }

226    public Response invokeRemoteCommand(final Address recipient, final ReplicableCommand command, final ResponseMode mode,
227                                        final long timeout, final DeliverOrder deliverOrder, boolean asyncMarshalling) throws InterruptedException {
228       if (asyncMarshalling) {
229          asyncExecutor.submit(new Callable<Response>() {
230 
231             @Override
232             public Response call() throws Exception {
233                return processSingleCall(command, timeout, recipient, mode, deliverOrder,
234                                         req_marshaller, CommandAwareRpcDispatcher.this, transport);
235             }
236          });
237          return null; // don't wait for a response!
238       } else {
239          Response response;
240          try {
241             response = processSingleCall(command, timeout, recipient, mode, deliverOrder,
242                                          req_marshaller, this, transport);
243          } catch (InterruptedException e) {
244             throw e;
245          } catch (SuspectedException e) {
246             throw new SuspectException("Node " + recipient + " was suspected", e);
247          } catch (org.jgroups.TimeoutException e) {
248             throw new TimeoutException("Node " + recipient + " timed out", e);
249          } catch (Exception e) {
250             throw rewrapAsCacheException(e);
251          }
252          if (mode == ResponseMode.GET_NONE) return null; // "Traditional" async.
253          return response;
254       }

283    private boolean containsOnlyNulls(RspList<Object> l) {
284       for (Rsp<Object> r : l.values()) {
285          if (r.getValue() != null || !r.wasReceived() || r.wasSuspected()) return false;
286       }
287       return true;
288    }

290    /**
291     * Message contains a Command. Execute it against *this* object and return result.
292     */
293    @Override
294    public void handle(Message req, org.jgroups.blocks.Response response) throws Exception {
295       if (isValid(req)) {
296          ReplicableCommand cmd = null;
297          try {
298             cmd = (ReplicableCommand) req_marshaller.objectFromBuffer(req.getRawBuffer(), req.getOffset(), req.getLength());
299             if (cmd == null)
300                throw new NullPointerException("Unable to execute a null command!  Message was " + req);
301             if (req.getSrc() instanceof SiteAddress) {
302                executeCommandFromRemoteSite(cmd, req, response);
303             } else {
304                executeCommandFromLocalCluster(cmd, req, response);
305             }
306          } catch (InterruptedException e) {
307             log.shutdownHandlingCommand(cmd);
308             reply(response, new ExceptionResponse(new CacheException("Cache is shutting down")), cmd);
309          } catch (IllegalLifecycleStateException e) {
310             if (trace) log.trace("Ignoring command unmarshalling error during shutdown");
311             // If this wasn't a CacheRpcCommand, it means the channel is already stopped, and the response won't matter
312             reply(response, CacheNotFoundResponse.INSTANCE, cmd);
313          } catch (Throwable x) {
314             if (cmd == null)
315                log.errorUnMarshallingCommand(x);
316             else
317                log.exceptionHandlingCommand(cmd, x);
318             reply(response, new ExceptionResponse(new CacheException("Problems invoking command.", x)), cmd);
319          }
320       } else {
321          reply(response, null, null);
322       }
323    }

325    private void executeCommandFromRemoteSite(final ReplicableCommand cmd, final Message req, final org.jgroups.blocks.Response response) throws Throwable {
326       SiteAddress siteAddress = (SiteAddress) req.getSrc();
327       ((XSiteReplicateCommand) cmd).setOriginSite(siteAddress.getSite());
328       handler.handleFromRemoteSite(siteAddress.getSite(), (XSiteReplicateCommand) cmd, new Reply() {
329          @Override
330          public void reply(Object returnValue) {
331             CommandAwareRpcDispatcher.this.reply(response, returnValue, cmd);
332          }
333       }, decodeDeliverMode(req));
334    }

336    private void executeCommandFromLocalCluster(final ReplicableCommand cmd, final Message req, final org.jgroups.blocks.Response response) throws Throwable {
337       handler.handleFromCluster(fromJGroupsAddress(req.getSrc()), cmd, new Reply() {
338          @Override
339          public void reply(Object returnValue) {
340             CommandAwareRpcDispatcher.this.reply(response, returnValue, cmd);
341          }
342       }, decodeDeliverMode(req));

345    private static DeliverOrder decodeDeliverMode(Message request) {
346       boolean noTotalOrder = request.isFlagSet(Message.Flag.NO_TOTAL_ORDER);
347       boolean oob = request.isFlagSet(Message.Flag.OOB);
348       if (!noTotalOrder && oob) {
349          return DeliverOrder.TOTAL;
350       } else if (noTotalOrder && oob) {
351          return DeliverOrder.NONE;
352       } else if (noTotalOrder) {
353          //oob is not set at this point, but the no total order flag should.
354          return DeliverOrder.PER_SENDER;
355       }
356       throw new IllegalArgumentException("Unable to decode message " + request);
357    }
358 
359    private static void encodeDeliverMode(Message request, DeliverOrder deliverOrder) {
360       switch (deliverOrder) {
361          case TOTAL:
362             request.setFlag(Message.Flag.OOB);
363             request.clearFlag(Message.Flag.NO_TOTAL_ORDER);
364             break;
365          case PER_SENDER:
366             request.clearFlag(Message.Flag.OOB);
367             request.setFlag(Message.Flag.NO_TOTAL_ORDER);
368             break;
369          case NONE:
370             request.setFlag(Message.Flag.OOB, Message.Flag.NO_TOTAL_ORDER);
371             break;
372       }
373    }

380    private void reply(org.jgroups.blocks.Response response, Object retVal, ReplicableCommand command) {
381       if (response != null) {
382          if (trace) log.tracef("About to send back response %s for command %s", retVal, command);
383          //exceptionThrown is always false because the exceptions are wrapped in an ExceptionResponse
384          response.send(retVal, false);
385       }
386    }

388    protected static Message constructMessage(Buffer buf, Address recipient, boolean oob, ResponseMode mode, boolean rsvp,
389                                              boolean totalOrder) {
390       DeliverOrder deliverOrder = DeliverOrder.PER_SENDER;
391       if (totalOrder) {
392          deliverOrder = DeliverOrder.TOTAL;
393       } else if (oob) {
394          deliverOrder = DeliverOrder.NONE;
395       }
396       return constructMessage(buf, recipient, mode, rsvp, deliverOrder);
397    }

399    protected static Message constructMessage(Buffer buf, Address recipient,ResponseMode mode, boolean rsvp,
400                                              DeliverOrder deliverOrder) {
401       Message msg = new Message();
402       msg.setBuffer(buf);
403       encodeDeliverMode(msg, deliverOrder);
404       //some issues with the new bundler. put back the DONT_BUNDLE flag.
405       if (deliverOrder == DeliverOrder.NONE || mode != ResponseMode.GET_NONE) msg.setFlag(Message.Flag.DONT_BUNDLE);
406       if (rsvp) msg.setFlag(Message.Flag.RSVP);
407 
408       if (recipient != null) msg.setDest(recipient);
409       return msg;
410    }

413       Buffer buf;

415          buf = marshaller.objectToBuffer(command);

418       }

422    private static Response processSingleCall(ReplicableCommand command, long timeout,
423                                              Address destination, ResponseMode mode, DeliverOrder deliverOrder,
424                                              Marshaller marshaller, CommandAwareRpcDispatcher card,
425                                              JGroupsTransport transport) throws Exception {
426       if (trace) log.tracef("Replication task sending %s to single recipient %s with response mode %s", command, destination, mode);
427       boolean rsvp = isRsvpCommand(command);
428 
429       // Replay capability requires responses from all members!
430       Response retval;
431       Buffer buf;
432       buf = marshallCall(marshaller, command);
433       retval = card.sendMessage(constructMessage(buf, destination, mode, rsvp, deliverOrder),
434                                 new RequestOptions(mode, timeout));
435 
436       // we only bother parsing responses if we are not in ASYNC mode.
437       if (trace) log.tracef("Response: %s", retval);
438       if (mode == ResponseMode.GET_NONE)
439          return null;
440 
441       if (retval != null) {
442          if (!transport.checkResponse(retval, fromJGroupsAddress(destination))) {
443             if (trace) log.tracef("Invalid response from %s", destination);
444             throw new TimeoutException("Received an invalid response " + retval + " from " + destination);
445          }
446       }
447 
448       return retval;
449    }

451    private static RspList<Object> processCalls(ReplicableCommand command, boolean broadcast, long timeout,
452                                                RspFilter filter, List<Address> dests, ResponseMode mode, DeliverOrder deliverOrder,
453                                                Marshaller marshaller, CommandAwareRpcDispatcher card,
454                                                boolean ignoreLeavers) throws Exception {
455       if (trace) log.tracef("Replication task sending %s to addresses %s with response mode %s", command, dests, mode);
456       boolean rsvp = isRsvpCommand(command);
457 
458       RspList<Object> retval = null;
459       Buffer buf;
460       if (deliverOrder == DeliverOrder.TOTAL) {
461          buf = marshallCall(marshaller, command);
462          Message message = constructMessage(buf, new AnycastAddress(dests), mode, rsvp, deliverOrder);
463 
464          retval = card.castMessage(dests, message, new RequestOptions(mode, timeout, false, filter));
465       } else if (broadcast || FORCE_MCAST) {
466          buf = marshallCall(marshaller, command);
467          RequestOptions opts = new RequestOptions(mode, timeout, false, filter);
468 
469          //Only the commands in total order must be received...
470          //For correctness, ispn doesn't need their own message, so add own address to exclusion list
471          opts.setExclusionList(card.getChannel().getAddress());
472 
473          retval = card.castMessage(dests, constructMessage(buf, null, mode, rsvp, deliverOrder),opts);
474       } else {
475          RequestOptions opts = new RequestOptions(mode, timeout);
476 
477          //Only the commands in total order must be received...
478          opts.setExclusionList(card.getChannel().getAddress());
479 
480          if (dests.isEmpty()) return new RspList<>();
481          buf = marshallCall(marshaller, command);
482 
483          // if at all possible, try not to use JGroups' ANYCAST for now.  Multiple (parallel) UNICASTs are much faster.
484          if (filter != null) {
485             // This is possibly a remote GET.
486             // These UNICASTs happen in parallel using sendMessageWithFuture.  Each future has a listener attached
487             // (see FutureCollator) and the first successful response is used.
488             FutureCollator futureCollator = new FutureCollator(filter, dests.size(), timeout, card.timeService);
489             for (Address a : dests) {
490                NotifyingFuture<Object> f = card.sendMessageWithFuture(constructMessage(buf, a, mode, rsvp, deliverOrder), opts);
491                futureCollator.watchFuture(f, a);
492             }
493             retval = futureCollator.getResponseList();
494          } else if (mode == ResponseMode.GET_ALL) {
495             // A SYNC call that needs to go everywhere
496             Map<Address, Future<Object>> futures = new HashMap<>(dests.size());
497 
498             for (Address dest : dests)
499                futures.put(dest, card.sendMessageWithFuture(constructMessage(buf, dest, mode, rsvp, deliverOrder), opts));
500 
501             retval = new RspList<>();
502 
503             // a get() on each future will block till that call completes.
504             for (Map.Entry<Address, Future<Object>> entry : futures.entrySet()) {
505                Address target = entry.getKey();
506                try {
507                   retval.addRsp(target, entry.getValue().get(timeout, MILLISECONDS));
508                } catch (java.util.concurrent.TimeoutException te) {
509                   throw new TimeoutException(formatString("Timed out after %s waiting for a response from %s",
510                                                           prettyPrintTime(timeout), target));
511                } catch (ExecutionException e) {
512                   if (ignoreLeavers && e.getCause() instanceof SuspectedException) {
513                      log.tracef(formatString("Ignoring node %s that left during the remote call", target));
514                      retval.addRsp(target, CacheNotFoundResponse.INSTANCE);
515                   } else {
516                      throw wrapThrowableInException(e.getCause());
517                   }
518                }
519             }
520          } else if (mode == ResponseMode.GET_NONE) {
521             // An ASYNC call.  We don't care about responses.
522             for (Address dest : dests) card.sendMessage(constructMessage(buf, dest, mode, rsvp, deliverOrder), opts);
523          }
524       }
525 
526       // we only bother parsing responses if we are not in ASYNC mode.
527       if (mode != ResponseMode.GET_NONE) {
528 
529          if (trace) log.tracef("Responses: %s", retval);
530 
531          // a null response is 99% likely to be due to a marshalling problem - we throw a NSE, this needs to be changed when
532          // JGroups supports http://jira.jboss.com/jira/browse/JGRP-193
533          // the serialization problem could be on the remote end and this is why we cannot catch this above, when marshalling.
534          if (retval == null)
535             throw new NotSerializableException("RpcDispatcher returned a null.  This is most often caused by args for "
536                                                      + command.getClass().getSimpleName() + " not being serializable.");
537       }
538 
539       return retval;
540    }

542    private static RspList<Object> processCalls(Map<Address, ReplicableCommand> commands, long timeout,
543                                                ResponseMode mode, Marshaller marshaller, CommandAwareRpcDispatcher card,
544                                                boolean oob, boolean ignoreLeavers) throws Exception {
545       if (trace) log.tracef("Replication task sending %s with response mode %s", commands, mode);
546 
547       if (commands.isEmpty()) return new RspList<>();
548 
549       RequestOptions opts = new RequestOptions(mode, timeout);
550       //opts.setExclusionList(card.getChannel().getAddress());
551 
552       Map<Address, Future<Object>> futures = new HashMap<Address, Future<Object>>(commands.size());
553       RspList<Object> retval = new RspList<>();
554 
555       for (Map.Entry<Address, ReplicableCommand> cmd : commands.entrySet()) {
556          Buffer buf = marshallCall(marshaller, cmd.getValue());
557          Address dest = cmd.getKey();
558          boolean rsvp = isRsvpCommand(cmd.getValue());
559          futures.put(dest, card.sendMessageWithFuture(constructMessage(buf, dest, oob, mode, rsvp, false), opts));
560       }
561 
562       // a get() on each future will block till that call completes.
563       TimeService timeService = card.timeService;
564       long waitTime = timeService.expectedEndTime(timeout, MILLISECONDS);
565       for (Map.Entry<Address, Future<Object>> entry : futures.entrySet()) {
566          Address target = entry.getKey();
567          try {
568             retval.addRsp(target, entry.getValue().get(timeService.remainingTime(waitTime, MILLISECONDS), MILLISECONDS));
569          } catch (java.util.concurrent.TimeoutException te) {
570             throw new TimeoutException(formatString("Timed out after %s waiting for a response from %s",
571                   prettyPrintTime(timeout), target));
572          } catch (ExecutionException e) {
573             if (ignoreLeavers && e.getCause() instanceof SuspectedException) {
574                retval.addRsp(target, new ExceptionResponse((SuspectedException) e.getCause()));
575             } else {
576                throw wrapThrowableInException(e.getCause());
577             }
578          }
579       }
580       return retval;
581    }
582 
583    private static Exception wrapThrowableInException(Throwable t) {
584       if (t instanceof Exception) {
585          return (Exception) t;
586       } else {
587          return new CacheException(t);
588       }
589    }

596    static class SenderContainer {
597       final Address address;
598       volatile boolean processed = false;

600       SenderContainer(Address address) {
601          this.address = address;
602       }

609                '}';
610       }

613    final static class FutureCollator implements FutureListener<Object> {
614       final RspFilter filter;
615       final Map<Future<Object>, SenderContainer> futures = new HashMap<>(4);
616       final long timeout;
617       @GuardedBy("this")
618       private RspList<Object> retval;
619       @GuardedBy("this")
620       private Exception exception;
621       @GuardedBy("this")
622       private int expectedResponses;
623       private final TimeService timeService;
624 
625       FutureCollator(RspFilter filter, int expectedResponses, long timeout, TimeService timeService) {
626          this.filter = filter;
627          this.expectedResponses = expectedResponses;
628          this.timeout = timeout;
629          this.timeService = timeService;
630       }
631 
632       public void watchFuture(NotifyingFuture<Object> f, Address address) {
633          futures.put(f, new SenderContainer(address));
634          f.setListener(this);
635       }
636 
637       public synchronized RspList<Object> getResponseList() throws Exception {
638          long expectedEndTime = timeService.expectedEndTime(timeout, MILLISECONDS);
639          long waitingTime;
640          while (expectedResponses > 0 && retval == null &&
641                (waitingTime = timeService.remainingTime(expectedEndTime, MILLISECONDS)) > 0) {
642             try {
643                this.wait(waitingTime);
644             } catch (InterruptedException e) {
645                // reset interruption flag
646                Thread.currentThread().interrupt();
647                expectedResponses = -1;
648             }
649          }
650          // Now we either have the response we need or aren't expecting any more responses - or have run out of time.
651          if (retval != null)
652             return retval;
653          else if (exception != null)
654             throw exception;
655          else if (expectedResponses == 0)
656             throw new RpcException(format("No more valid responses.  Received invalid responses from all of %s", futures.values()));
657          else
658             throw new TimeoutException(format("Timed out waiting for %s for valid responses from any of %s.", Util.prettyPrintTime(timeout), futures.values()));
659       }
660 
661       @Override
662       @SuppressWarnings("unchecked")
663       public synchronized void futureDone(Future<Object> objectFuture) {
664          SenderContainer sc = futures.get(objectFuture);
665          if (sc.processed) {
666             // This can happen - it is a race condition in JGroups' NotifyingFuture.setListener() where a listener
667             // could be notified twice.
668             if (trace) log.tracef("Not processing callback; already processed callback for sender %s", sc.address);
669          } else {
670             sc.processed = true;
671             Address sender = sc.address;
672             boolean done = false;
673             try {
674                if (retval == null) {
675                   Object response = objectFuture.get();
676                   if (trace) log.tracef("Received response: %s from %s", response, sender);
677                   filter.isAcceptable(response, sender);
678                   if (!filter.needMoreResponses()) {
679                      retval = new RspList(Collections.singleton(new Rsp(sender, response)));
680                      done = true;
681                      //TODO cancel other tasks?
682                   }
683                } else {
684                   if (trace) log.tracef("Skipping response from %s since a valid response for this request has already been received", sender);
685                }
686             } catch (InterruptedException e) {
687                Thread.currentThread().interrupt();
688             } catch (ExecutionException e) {
689                Throwable cause = e.getCause();
690                if (cause instanceof org.jgroups.SuspectedException) {
691                   // Do not set the exception field, RpcException should be thrown if there is no other valid response
692                   return;
693                } else if (cause instanceof org.jgroups.TimeoutException) {
694                   exception = new TimeoutException("Timeout!", e);
695                } else if (cause instanceof Exception) {
696                   exception = (Exception) cause;
697                } else {
698                   exception = new CacheException("Caught a throwable", cause);
699                }
700 
701                if (log.isDebugEnabled())
702                   log.debugf("Caught exception from sender %s: %s", sender, exception);
703             } finally {
704                expectedResponses--;
705                if (expectedResponses == 0 || done) {
706                   this.notify(); //make sure to awake waiting thread, but avoid unnecessary wakeups!
707                }
708             }
709          }
710       }
711    }
