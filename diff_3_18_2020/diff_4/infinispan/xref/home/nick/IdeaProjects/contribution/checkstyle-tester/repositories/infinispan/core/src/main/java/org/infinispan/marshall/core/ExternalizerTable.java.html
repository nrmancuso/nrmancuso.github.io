120 /**
121  * The externalizer table maintains information necessary to be able to map a particular type with the corresponding
122  * {@link org.infinispan.commons.marshall.AdvancedExternalizer} implementation that it marshall, and it also keeps information of which {@link org.infinispan.commons.marshall.AdvancedExternalizer}
123  * should be used to read data from a buffer given a particular {@link org.infinispan.commons.marshall.AdvancedExternalizer} identifier.
124  *
125  * These tables govern how either internal Infinispan classes, or user defined classes, are marshalled to a given
126  * output, or how these are unmarshalled from a given input.
127  *
128  * @author Galder Zamarre√±o
129  * @since 5.0
130  */
131 @Scope(Scopes.GLOBAL)
132 public class ExternalizerTable implements ObjectTable {
133    private static final Log log = LogFactory.getLog(ExternalizerTable.class);
134 
135    /**
136     * Contains mapping of classes to their corresponding externalizer classes via ExternalizerAdapter instances.
137     */
138    private final Map<Class<?>, ExternalizerAdapter> writers = new WeakHashMap<Class<?>, ExternalizerAdapter>();
139 
140    /**
141     * Contains mapping of ids to their corresponding AdvancedExternalizer classes via ExternalizerAdapter instances.
142     * This maps contains mappings for both internal and foreign or user defined externalizers.
143     *
144     * Internal ids are only allowed to be unsigned bytes (0 to 254). 255 is an special id that signals the
145     * arrival of a foreign externalizer id. Foreign externalizers are only allowed to use positive ids that between 0
146     * and Integer.MAX_INT. To avoid clashes between foreign and internal ids, foreign ids are transformed into negative
147     * values to be stored in this map. This way, we avoid the need of a second map to hold user defined externalizers.
148     */
149    private final Map<Integer, ExternalizerAdapter> readers = new HashMap<Integer, ExternalizerAdapter>();

165    @Start(priority = 7) // Should start before global marshaller
166    public void start() {
167       loadInternalMarshallables();
168       loadForeignMarshallables(gcr.getGlobalConfiguration());
169       started = true;
170       if (log.isTraceEnabled()) {
171          log.tracef("Constant object table was started and contains these externalizer readers: %s", readers);
172          log.tracef("The externalizer writers collection contains: %s", writers);
173       }
174    }
175 
176    @Stop(priority = 13) // Stop after global marshaller
177    public void stop() {
178       started = false;
179       writers.clear();
180       readers.clear();
181       log.trace("Externalizer reader and writer maps have been cleared and constant object table was stopped");
182    }

184    @Override
185    public Writer getObjectWriter(Object o) throws IOException {
186       Class<?> clazz = o.getClass();
187       if (!started) {
188          throw log.externalizerTableStopped(clazz.getName());
189       }
190       Writer writer = writers.get(clazz);
191       return writer;
192    }
193 
194    @Override
195    public Object readObject(Unmarshaller input) throws IOException, ClassNotFoundException {
196       int readerIndex = input.readUnsignedByte();
197       int foreignId = -1;
198       if (readerIndex == Ids.MAX_ID) {
199          // User defined externalizer
200          foreignId = UnsignedNumeric.readUnsignedInt(input);
201          readerIndex = generateForeignReaderIndex(foreignId);
202       }
203 
204       ExternalizerAdapter adapter = readers.get(readerIndex);
205       if (adapter == null) {
206          if (!started) {
207             log.tracef("Either the marshaller has stopped or hasn't started. Read externalizers are not properly populated: %s", readers);
208 
209             if (Thread.currentThread().isInterrupted()) {
210                throw log.pushReadInterruptionDueToCacheManagerShutdown(readerIndex, new InterruptedException());
211             } else {
212                throw log.cannotResolveExternalizerReader(gcr.getStatus(), readerIndex);
213             }
214          } else {
215             if (log.isTraceEnabled()) {
216                log.tracef("Unknown type. Input stream has %s to read", input.available());
217                log.tracef("Check contents of read externalizers: %s", readers);
218             }
219 
220             if (foreignId > 0)
221                throw log.missingForeignExternalizer(foreignId);
222 
223             throw log.unknownExternalizerReaderIndex(readerIndex);
224          }
225       }
226 
227       return adapter.readObject(input);
228    }

352    void addInternalExternalizer(AdvancedExternalizer<?> ext) {
353       int id = checkInternalIdLimit(ext.getId(), ext);
354       updateExtReadersWritersWithTypes(new ExternalizerAdapter(id, ext));
355    }
356 
357    private void updateExtReadersWritersWithTypes(ExternalizerAdapter adapter) {
358       updateExtReadersWritersWithTypes(adapter, adapter.id);
359    }
360 
361    private void updateExtReadersWritersWithTypes(ExternalizerAdapter adapter, int readerIndex) {
362       Set<Class<?>> typeClasses = adapter.externalizer.getTypeClasses();
363       if (typeClasses.size() > 0) {
364          for (Class<?> typeClass : typeClasses)
365             updateExtReadersWriters(adapter, typeClass, readerIndex);
366       } else {
367          throw log.advanceExternalizerTypeClassesUndefined(adapter.externalizer.getClass().getName());
368       }
369    }
370 
371    private void loadForeignMarshallables(GlobalConfiguration globalCfg) {
372       log.trace("Loading user defined externalizers");
373       for (Map.Entry<Integer, AdvancedExternalizer<?>> config : globalCfg.serialization().advancedExternalizers().entrySet()) {
374          AdvancedExternalizer<?> ext = config.getValue();
375 
376          // If no XML or programmatic config, id in annotation is used
377          // as long as it's not default one (meaning, user did not set it).
378          // If XML or programmatic config in use ignore @Marshalls annotation and use value in config.
379          Integer id = ext.getId();
380          if (config.getKey() == null && id == null)
381             throw new CacheConfigurationException(String.format(
382                   "No advanced externalizer identifier set for externalizer %s",
383                   ext.getClass().getName()));
384          else if (config.getKey() != null)
385             id = config.getKey();
386 
387          id = checkForeignIdLimit(id, ext);
388          updateExtReadersWritersWithTypes(new ForeignExternalizerAdapter(id, ext), generateForeignReaderIndex(id));
389       }
390    }
391 
392    private void updateExtReadersWriters(ExternalizerAdapter adapter, Class<?> typeClass, int readerIndex) {
393       writers.put(typeClass, adapter);
394       ExternalizerAdapter prevReader = readers.put(readerIndex, adapter);
395       // Several externalizers might share same id (i.e. HashMap and TreeMap use MapExternalizer)
396       // but a duplicate is only considered when that particular index has already been entered
397       // in the readers map and the externalizers are different (they're from different classes)
398       if (prevReader != null && !prevReader.equals(adapter))
399          throw log.duplicateExternalizerIdFound(
400                adapter.id, typeClass, prevReader.externalizer.getClass().getName(), readerIndex);
401 
402       if (log.isTraceEnabled())
403          log.tracef("Loaded externalizer %s for %s with id %s and reader index %s",
404                    adapter.externalizer.getClass().getName(), typeClass, adapter.id, readerIndex);
405 
406    }

408    private int checkInternalIdLimit(int id, AdvancedExternalizer<?> ext) {
409       if (id >= Ids.MAX_ID)
410          throw log.internalExternalizerIdLimitExceeded(ext, id, Ids.MAX_ID);
411 
412       return id;
413    }
414 
415    private int checkForeignIdLimit(int id, AdvancedExternalizer<?> ext) {
416       if (id < 0)
417          throw log.foreignExternalizerUsingNegativeId(ext, id);
418 
419       return id;
420    }
421 
422    private int generateForeignReaderIndex(int foreignId) {
423       return 0x80000000 | foreignId;
424    }

426    static class ExternalizerAdapter implements Writer {
427       final int id;
428       final AdvancedExternalizer<Object> externalizer;
429 
430       ExternalizerAdapter(int id, AdvancedExternalizer<?> externalizer) {
431          this.id = id;
432          this.externalizer = (AdvancedExternalizer<Object>) externalizer;
433       }
434 
435       public Object readObject(Unmarshaller input) throws IOException, ClassNotFoundException {
436          return externalizer.readObject(input);
437       }
438 
439       @Override
440       public void writeObject(Marshaller output, Object object) throws IOException {
441          output.write(id);
442          externalizer.writeObject(output, object);
443       }
444 
445       int getExternalizerId() {
446          return id;
447       }
448 
449       @Override
450       public String toString() {
451          // Each adapter is represented by the externalizer it delegates to, so just return the class name
452          return externalizer.getClass().getName();
453       }
454 
455       @Override
456       public boolean equals(Object o) {
457          if (this == o) return true;
458          if (o == null || getClass() != o.getClass()) return false;
459          ExternalizerAdapter that = (ExternalizerAdapter) o;
460          if (id != that.id) return false;
461          if (externalizer != null ? !externalizer.getClass().equals(that.externalizer.getClass()) : that.externalizer != null) return false;
462          return true;
463       }
464 
465       @Override
466       public int hashCode() {
467          int result = id;
468          result = 31 * result + (externalizer.getClass() != null ? externalizer.getClass().hashCode() : 0);
469          return result;
470       }
471    }
472 
473    static class ForeignExternalizerAdapter extends ExternalizerAdapter {
474       final int foreignId;
475 
476       ForeignExternalizerAdapter(int foreignId, AdvancedExternalizer<?> externalizer) {
477          super(Ids.MAX_ID, externalizer);
478          this.foreignId = foreignId;
479       }
480 
481       @Override
482       int getExternalizerId() {
483          return foreignId;
484       }
485 
486       @Override
487       public void writeObject(Marshaller output, Object object) throws IOException {
488          output.write(id);
489          // Write as an unsigned, variable length, integer to safe space
490          UnsignedNumeric.writeUnsignedInt(output, foreignId);
491          externalizer.writeObject(output, object);
492       }
493    }
