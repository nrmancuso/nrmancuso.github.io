11  /**

14   *
15   * @author anistor@redhat.com
16   * @since 6.0
17   */
18  @MessageLogger(projectCode = "ISPN")
19  public interface Log extends org.infinispan.util.logging.Log {
20  
21     @Message(value = "Unknown field %s in type %s", id = 18001)
22     IllegalArgumentException unknownField(String fieldName, String fullyQualifiedTypeName);
23  
24     @Message(value = "Field %s from type %s is not indexed", id = 18002)
25     IllegalArgumentException fieldIsNotIndexed(String fieldName, String fullyQualifiedTypeName);
26  
27     @Message(value = "An exception has occurred during query execution", id = 18003)
28     CacheException errorExecutingQuery(@Cause Throwable cause);
29  }
