1   package org.infinispan.server.endpoint.deployments;
2   
3   import org.infinispan.notifications.cachelistener.filter.CacheEventConverterFactory;
4   import org.jboss.msc.service.ServiceName;
5   import org.jboss.msc.service.StartContext;
6   import org.jboss.msc.service.StopContext;
7   
8   import static org.infinispan.server.endpoint.EndpointLogger.ROOT_LOGGER;
9   
10  public final class ConverterFactoryExtensionProcessor extends AbstractNamedFactoryExtensionProcessor<CacheEventConverterFactory> {
11  
12      public ConverterFactoryExtensionProcessor(ServiceName extensionManagerServiceName) {
13          super(extensionManagerServiceName);
14      }
15  
16      @Override
17      public AbstractExtensionManagerService<CacheEventConverterFactory> createService(String name, CacheEventConverterFactory instance) {
18          return new ConverterFactoryService(name, instance);
19      }
20  
21      @Override
22      public Class<CacheEventConverterFactory> getServiceClass() {
23          return CacheEventConverterFactory.class;
24      }
25  
26      private static class ConverterFactoryService extends AbstractExtensionManagerService<CacheEventConverterFactory> {
27          private ConverterFactoryService(String name, CacheEventConverterFactory converterFactory) {
28              super(name, converterFactory);
29          }
30  
31          @Override
32          public void start(StartContext context) {
33              ROOT_LOGGER.debugf("Started converter service with name = %s", name);
34              extensionManager.getValue().addConverterFactory(name, extension);
35          }
36  
37          @Override
38          public void stop(StopContext context) {
39              ROOT_LOGGER.debugf("Stopped converter service with name = %s", name);
40              extensionManager.getValue().removeConverterFactory(name);
41          }
42  
43          @Override
44          public CacheEventConverterFactory getValue() {
45              return extension;
46          }
47  
48          @Override
49          public String getServiceTypeName() {
50              return "converter-factory";
51          }
52      }
53  
54  }
