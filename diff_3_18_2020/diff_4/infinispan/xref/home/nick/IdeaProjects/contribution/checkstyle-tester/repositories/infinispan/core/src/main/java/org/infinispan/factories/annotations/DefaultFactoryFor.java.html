1   package org.infinispan.factories.annotations;
2   
3   import static java.lang.annotation.ElementType.TYPE;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   /**
9    * An annotation that is used internally, for defining a DEFAULT factory to be used when constructing components.  This
10   * annotation allows you to define which components can be constructed by the annotated factory.
11   *
12   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
13   * @since 4.0
14   */
15  @Target(TYPE)
16  @Retention(RetentionPolicy.RUNTIME)
17  public @interface DefaultFactoryFor {
18     /**
19      * Components that may be constructed by a factory annotated with this annotation.
20      *
21      * @return classes that can be constructed by this factory
22      */
23     public abstract Class<?>[] classes();
24  }
