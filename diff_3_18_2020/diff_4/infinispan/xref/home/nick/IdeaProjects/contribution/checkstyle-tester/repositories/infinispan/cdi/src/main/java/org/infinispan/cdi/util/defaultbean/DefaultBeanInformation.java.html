1   package org.infinispan.cdi.util.defaultbean;
2   
3   import static java.lang.annotation.ElementType.FIELD;
4   import static java.lang.annotation.ElementType.METHOD;
5   import static java.lang.annotation.ElementType.TYPE;
6   import static java.lang.annotation.RetentionPolicy.RUNTIME;
7   
8   import java.lang.annotation.Annotation;
9   import java.lang.annotation.Documented;
10  import java.lang.annotation.Retention;
11  import java.lang.annotation.Target;
12  import java.util.Set;
13  
14  import javax.enterprise.util.AnnotationLiteral;
15  
16  /**
17   * We use this annotation as a carrier of qualifiers so that other extensions have access to the original qualifiers of the bean
18   * (those removed and replaced by synthetic qualifier by the {@link DefaultBeanExtension}).
19   * 
20   * @author Jozef Hartinger
21   * 
22   */
23  @Target({ TYPE, METHOD, FIELD })
24  @Retention(RUNTIME)
25  @Documented
26  public @interface DefaultBeanInformation {
27  
28      @SuppressWarnings("all")
29      public static class Literal extends AnnotationLiteral<DefaultBeanInformation> implements DefaultBeanInformation {
30          private final Set<Annotation> qualifiers;
31  
32          public Literal(Set<Annotation> qualifiers) {
33              this.qualifiers = qualifiers;
34          }
35  
36          public Set<Annotation> getQualifiers() {
37              return qualifiers;
38          }
39      }
40  }
