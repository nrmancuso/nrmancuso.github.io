26  package sun.java2d.d3d;
27  
28  import java.awt.Composite;
29  import sun.java2d.SurfaceData;
30  import sun.java2d.loops.CompositeType;
31  import sun.java2d.loops.GraphicsPrimitive;
32  import sun.java2d.loops.GraphicsPrimitiveMgr;
33  import sun.java2d.loops.SurfaceType;
34  import sun.java2d.pipe.Region;
35  import sun.java2d.pipe.BufferedMaskBlit;
36  import static sun.java2d.loops.CompositeType.*;
37  import static sun.java2d.loops.SurfaceType.*;
38  
39  class D3DMaskBlit extends BufferedMaskBlit {
40  
41      static void register() {
42          GraphicsPrimitive[] primitives = {
43              new D3DMaskBlit(IntArgb,    SrcOver),
44              new D3DMaskBlit(IntArgbPre, SrcOver),
45              new D3DMaskBlit(IntRgb,     SrcOver),
46              new D3DMaskBlit(IntRgb,     SrcNoEa),
47              new D3DMaskBlit(IntBgr,     SrcOver),
48              new D3DMaskBlit(IntBgr,     SrcNoEa),
49          };
50          GraphicsPrimitiveMgr.register(primitives);
51      }
52  
53      private D3DMaskBlit(SurfaceType srcType,
54                          CompositeType compType)
55      {
56          super(D3DRenderQueue.getInstance(),
57                srcType, compType, D3DSurfaceData.D3DSurface);
58      }
59  
60      @Override
61      protected void validateContext(SurfaceData dstData,
62                                     Composite comp, Region clip)
63      {
64          D3DSurfaceData d3dDst = (D3DSurfaceData)dstData;
65          D3DContext.validateContext(d3dDst, d3dDst,
66                                     clip, comp, null, null, null,
67                                     D3DContext.NO_CONTEXT_FLAGS);
68      }
69  }
