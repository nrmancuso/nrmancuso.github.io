26  package sun.nio.cs.ext;
27  
28  import java.nio.charset.Charset;
29  import java.nio.charset.CharsetDecoder;
30  import java.nio.charset.CharsetEncoder;
31  import java.util.Arrays;
32  import sun.nio.cs.HistoricallyNamedCharset;
33  import static sun.nio.cs.CharsetMapping.*;
34  
35  public class IBM942C extends Charset implements HistoricallyNamedCharset
36  {
37      public IBM942C() {
38          super("x-IBM942C", ExtendedCharsets.aliasesFor("x-IBM942C"));
39      }
40  
41      public String historicalName() {
42          return "Cp942C";
43      }
44  
45      public boolean contains(Charset cs) {
46          return ((cs.name().equals("US-ASCII"))
47                  || (cs instanceof IBM942C));
48      }
49  
50      public CharsetDecoder newDecoder() {
51          return new DoubleByte.Decoder(this,
52                                        IBM942.b2c,
53                                        b2cSB,
54                                        0x40,
55                                        0xfc);
56      }
57  
58      public CharsetEncoder newEncoder() {
59          return new DoubleByte.Encoder(this, c2b, c2bIndex);
60      }
61  
62      final static char[] b2cSB;
63      final static char[] c2b;
64      final static char[] c2bIndex;
65  
66      static {
67          IBM942.initb2c();
68  
69          // the mappings need udpate are
70          //    u+001a  <-> 0x1a
71          //    u+001c  <-> 0x1c
72          //    u+005c  <-> 0x5c
73          //    u+007e  <-> 0x7e
74          //    u+007f  <-> 0x7f
75  
76          b2cSB = Arrays.copyOf(IBM942.b2cSB, IBM942.b2cSB.length);
77          b2cSB[0x1a] = 0x1a;
78          b2cSB[0x1c] = 0x1c;
79          b2cSB[0x5c] = 0x5c;
80          b2cSB[0x7e] = 0x7e;
81          b2cSB[0x7f] = 0x7f;
82  
83          IBM942.initc2b();
84          c2b = Arrays.copyOf(IBM942.c2b, IBM942.c2b.length);
85          c2bIndex = Arrays.copyOf(IBM942.c2bIndex, IBM942.c2bIndex.length);
86          c2b[c2bIndex[0] + 0x1a] = 0x1a;
87          c2b[c2bIndex[0] + 0x1c] = 0x1c;
88          c2b[c2bIndex[0] + 0x5c] = 0x5c;
89          c2b[c2bIndex[0] + 0x7e] = 0x7e;
90          c2b[c2bIndex[0] + 0x7f] = 0x7f;
91      }
92  }
