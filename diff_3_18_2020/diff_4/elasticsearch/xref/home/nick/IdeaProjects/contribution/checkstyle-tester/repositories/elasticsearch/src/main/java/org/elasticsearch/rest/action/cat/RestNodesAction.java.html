82      @Override
83      public void doRequest(final RestRequest request, final RestChannel channel, final Client client) {
84          final ClusterStateRequest clusterStateRequest = new ClusterStateRequest();
85          clusterStateRequest.clear().nodes(true);
86          clusterStateRequest.local(request.paramAsBoolean("local", clusterStateRequest.local()));
87          clusterStateRequest.masterNodeTimeout(request.paramAsTime("master_timeout", clusterStateRequest.masterNodeTimeout()));
88  
89          client.admin().cluster().state(clusterStateRequest, new RestActionListener<ClusterStateResponse>(channel) {
90              @Override
91              public void processResponse(final ClusterStateResponse clusterStateResponse) {
92                  NodesInfoRequest nodesInfoRequest = new NodesInfoRequest();
93                  nodesInfoRequest.clear().jvm(true).os(true).process(true);
94                  client.admin().cluster().nodesInfo(nodesInfoRequest, new RestActionListener<NodesInfoResponse>(channel) {
95                      @Override
96                      public void processResponse(final NodesInfoResponse nodesInfoResponse) {
97                          NodesStatsRequest nodesStatsRequest = new NodesStatsRequest();
98                          nodesStatsRequest.clear().jvm(true).os(true).fs(true).indices(true).process(true);
99                          client.admin().cluster().nodesStats(nodesStatsRequest, new RestResponseListener<NodesStatsResponse>(channel) {
100                             @Override
101                             public RestResponse buildResponse(NodesStatsResponse nodesStatsResponse) throws Exception {
102                                 return RestTable.buildResponse(buildTable(request, clusterStateResponse, nodesInfoResponse, nodesStatsResponse), channel);
103                             }
104                         });
105                     }
106                 });
107             }
108         });
109     }

214     private Table buildTable(RestRequest req, ClusterStateResponse state, NodesInfoResponse nodesInfo, NodesStatsResponse nodesStats) {
215         boolean fullId = req.paramAsBoolean("full_id", false);
216 
217         DiscoveryNodes nodes = state.getState().nodes();
218         String masterId = nodes.masterNodeId();
219         Table table = getTableWithHeader(req);
220 
221         for (DiscoveryNode node : nodes) {
222             NodeInfo info = nodesInfo.getNodesMap().get(node.id());
223             NodeStats stats = nodesStats.getNodesMap().get(node.id());
224 
225             JvmInfo jvmInfo = info == null ? null : info.getJvm();
226             OsInfo osInfo = info == null ? null : info.getOs();
227             ProcessInfo processInfo = info == null ? null : info.getProcess();
228 
229             JvmStats jvmStats = stats == null ? null : stats.getJvm();
230             FsStats fsStats = stats == null ? null : stats.getFs();
231             OsStats osStats = stats == null ? null : stats.getOs();
232             ProcessStats processStats = stats == null ? null : stats.getProcess();
233             NodeIndicesStats indicesStats = stats == null ? null : stats.getIndices();
234 
235             table.startRow();
236 
237             table.addCell(fullId ? node.id() : Strings.substring(node.getId(), 0, 4));
238             table.addCell(info == null ? null : info.getProcess().id());
239             table.addCell(node.getHostName());
240             table.addCell(node.getHostAddress());
241             if (node.address() instanceof InetSocketTransportAddress) {
242                 table.addCell(((InetSocketTransportAddress) node.address()).address().getPort());
243             } else {
244                 table.addCell("-");
245             }
246 
247             table.addCell(node.getVersion().number());
248             table.addCell(info == null ? null : info.getBuild().hashShort());
249             table.addCell(jvmInfo == null ? null : jvmInfo.version());
250             table.addCell(fsStats == null ? null : fsStats.getTotal().getAvailable());
251             table.addCell(jvmStats == null ? null : jvmStats.getMem().getHeapUsed());
252             table.addCell(jvmStats == null ? null : jvmStats.getMem().getHeapUsedPrecent());
253             table.addCell(jvmInfo == null ? null : jvmInfo.getMem().getHeapMax());
254             table.addCell(osStats == null ? null : osStats.getMem() == null ? null : osStats.getMem().used());
255             table.addCell(osStats == null ? null : osStats.getMem() == null ? null : osStats.getMem().usedPercent());
256             table.addCell(osInfo == null ? null : osInfo.getMem() == null ? null : osInfo.getMem().total()); // sigar fails to load in IntelliJ
257             table.addCell(processStats == null ? null : processStats.getOpenFileDescriptors());
258             table.addCell(processStats == null || processInfo == null ? null :
259                           calculatePercentage(processStats.getOpenFileDescriptors(), processInfo.getMaxFileDescriptors()));
260             table.addCell(processInfo == null ? null : processInfo.getMaxFileDescriptors());
261 
262             table.addCell(osStats == null ? null : osStats.getLoadAverage().length < 1 ? null : String.format(Locale.ROOT, "%.2f", osStats.getLoadAverage()[0]));
263             table.addCell(jvmStats == null ? null : jvmStats.uptime());
264             table.addCell(node.clientNode() ? "c" : node.dataNode() ? "d" : "-");
265             table.addCell(masterId == null ? "x" : masterId.equals(node.id()) ? "*" : node.masterNode() ? "m" : "-");
266             table.addCell(node.name());
267 
268             CompletionStats completionStats = indicesStats == null ? null : stats.getIndices().getCompletion();
269             table.addCell(completionStats == null ? null : completionStats.getSize());
270 
271             FieldDataStats fdStats = indicesStats == null ? null : stats.getIndices().getFieldData();
272             table.addCell(fdStats == null ? null : fdStats.getMemorySize());
273             table.addCell(fdStats == null ? null : fdStats.getEvictions());
274 
275             FilterCacheStats fcStats = indicesStats == null ? null : indicesStats.getFilterCache();
276             table.addCell(fcStats == null ? null : fcStats.getMemorySize());
277             table.addCell(fcStats == null ? null : fcStats.getEvictions());
278 
279             QueryCacheStats qcStats = indicesStats == null ? null : indicesStats.getQueryCache();
280             table.addCell(qcStats == null ? null : qcStats.getMemorySize());
281             table.addCell(qcStats == null ? null : qcStats.getEvictions());
282             table.addCell(qcStats == null ? null : qcStats.getHitCount());

284 
285             FlushStats flushStats = indicesStats == null ? null : indicesStats.getFlush();

287             table.addCell(flushStats == null ? null : flushStats.getTotalTime());
288 
289             GetStats getStats = indicesStats == null ? null : indicesStats.getGet();
290             table.addCell(getStats == null ? null : getStats.current());
291             table.addCell(getStats == null ? null : getStats.getTime());
292             table.addCell(getStats == null ? null : getStats.getCount());
293             table.addCell(getStats == null ? null : getStats.getExistsTime());
294             table.addCell(getStats == null ? null : getStats.getExistsCount());
295             table.addCell(getStats == null ? null : getStats.getMissingTime());

297 
298             IdCacheStats idCacheStats = indicesStats == null ? null : indicesStats.getIdCache();

300 
301             IndexingStats indexingStats = indicesStats == null ? null : indicesStats.getIndexing();
302             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getDeleteCurrent());
303             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getDeleteTime());
304             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getDeleteCount());
305             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getIndexCurrent());
306             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getIndexTime());
307             table.addCell(indexingStats == null ? null : indexingStats.getTotal().getIndexCount());
308

310             table.addCell(mergeStats == null ? null : mergeStats.getCurrent());
311             table.addCell(mergeStats == null ? null : mergeStats.getCurrentNumDocs());
312             table.addCell(mergeStats == null ? null : mergeStats.getCurrentSize());
313             table.addCell(mergeStats == null ? null : mergeStats.getTotal());
314             table.addCell(mergeStats == null ? null : mergeStats.getTotalNumDocs());
315             table.addCell(mergeStats == null ? null : mergeStats.getTotalSize());

317 
318             PercolateStats percolateStats = indicesStats == null ? null : indicesStats.getPercolate();
319             table.addCell(percolateStats == null ? null : percolateStats.getCurrent());
320             table.addCell(percolateStats == null ? null : percolateStats.getMemorySize());
321             table.addCell(percolateStats == null ? null : percolateStats.getNumQueries());
322             table.addCell(percolateStats == null ? null : percolateStats.getTime());
323             table.addCell(percolateStats == null ? null : percolateStats.getCount());
324 
325             RefreshStats refreshStats = indicesStats == null ? null : indicesStats.getRefresh();
326             table.addCell(refreshStats == null ? null : refreshStats.getTotal());
327             table.addCell(refreshStats == null ? null : refreshStats.getTotalTime());
328 
329             SearchStats searchStats = indicesStats == null ? null : indicesStats.getSearch();
330             table.addCell(searchStats == null ? null : searchStats.getTotal().getFetchCurrent());
331             table.addCell(searchStats == null ? null : searchStats.getTotal().getFetchTime());
332             table.addCell(searchStats == null ? null : searchStats.getTotal().getFetchCount());

341             table.addCell(segmentsStats == null ? null : segmentsStats.getIndexWriterMemory());

349             table.addCell(suggestStats == null ? null : suggestStats.getCount());

357     /**
358      * Calculate the percentage of {@code used} from the {@code max} number.
359      * @param used The currently used number.
360      * @param max The maximum number.
361      * @return 0 if {@code max} is <= 0. Otherwise 100 * {@code used} / {@code max}.
362      */
363     private short calculatePercentage(long used, long max) {
364         return max <= 0 ? 0 : (short)((100d * used) / max);
365     }
