110         // We expect leaves to be segment readers
111         final Map<String, String> diagnostics = segmentReader(reader).getSegmentInfo().info.getDiagnostics();
112         final String source = diagnostics.get(IndexWriter.SOURCE);
113         assert Arrays.asList(IndexWriter.SOURCE_ADDINDEXES_READERS, IndexWriter.SOURCE_FLUSH, IndexWriter.SOURCE_MERGE).contains(source) : "Unknown source " + source;

131         private static final ReleasableLock NOOP_LOCK = new ReleasableLock(new NoOpLock());
132         private final ReleasableLock lockReference = new ReleasableLock(new ReentrantLock());

140         /** Activate throttling, which switches the lock to be a real lock */
141         public void activate() {
142             assert lock == NOOP_LOCK : "throttling activated while already active";

147         public void deactivate() {
148             assert lock != NOOP_LOCK : "throttling deactivated but not active";

150         }

192     final protected GetResult getFromSearcher(Get get) throws EngineException {
193         final Searcher searcher = acquireSearcher("get");
194         final Versions.DocIdAndVersion docIdAndVersion;
195         try {
196             docIdAndVersion = Versions.loadDocIdAndVersion(searcher.reader(), get.uid());
197         } catch (Throwable e) {
198             Releasables.closeWhileHandlingException(searcher);
199             //TODO: A better exception goes here
200             throw new EngineException(shardId, "Couldn't resolve version", e);
201         }
202 
203         if (docIdAndVersion != null) {
204             if (get.versionType().isVersionConflictForReads(docIdAndVersion.version, get.version())) {
205                 Releasables.close(searcher);
206                 Uid uid = Uid.createUid(get.uid().text());
207                 throw new VersionConflictEngineException(shardId, uid.type(), uid.id(), docIdAndVersion.version, get.version());
208             }
209         }
210 
211         if (docIdAndVersion != null) {
212             // don't release the searcher on this path, it is the
213             // responsibility of the caller to call GetResult.release
214             return new GetResult(searcher, docIdAndVersion);
215         } else {
216             Releasables.close(searcher);
217             return GetResult.NOT_EXISTS;
218         }
219     }

223     /**
224      * Returns a new searcher instance. The consumer of this
225      * API is responsible for releasing the returned seacher in a
226      * safe manner, preferably in a try/finally block.
227      *
228      * @see Searcher#close()
229      */
230     public final Searcher acquireSearcher(String source) throws EngineException {
231         boolean success = false;
232          /* Acquire order here is store -> manager since we need
233           * to make sure that the store is not closed before
234           * the searcher is acquired. */
235         store.incRef();
236         try {
237             final SearcherManager manager = getSearcherManager(); // can never be null
238             /* This might throw NPE but that's fine we will run ensureOpen()
239             *  in the catch block and throw the right exception */
240             final IndexSearcher searcher = manager.acquire();
241             try {
242                 final Searcher retVal = newSearcher(source, searcher, manager);
243                 success = true;
244                 return retVal;
245             } finally {
246                 if (!success) {
247                     manager.release(searcher);
248                 }
249             }
250         } catch (EngineClosedException ex) {
251             throw ex;
252         } catch (Throwable ex) {
253             ensureOpen(); // throw EngineCloseException here if we are already closed
254             logger.error("failed to acquire searcher, source {}", ex, source);
255             throw new EngineException(shardId, "failed to acquire searcher, source " + source, ex);
256         } finally {
257             if (!success) {  // release the ref in the case of an error...
258                 store.decRef();
259             }
260         }
261     }
262 
263     protected void ensureOpen() {
264         if (isClosed.get()) {
265             throw new EngineClosedException(shardId, failedEngine);
266         }
267     }

272     public final SegmentsStats segmentsStats() {
273         ensureOpen();
274         try (final Searcher searcher = acquireSearcher("segments_stats")) {
275             SegmentsStats stats = new SegmentsStats();
276             for (AtomicReaderContext reader : searcher.reader().leaves()) {
277                 final SegmentReader segmentReader = segmentReader(reader.reader());
278                 stats.add(1, segmentReader.ramBytesUsed());
279             }
280             writerSegmentStats(stats);

292     protected Segment[] getSegmentInfo(SegmentInfos lastCommittedSegmentInfos) {
293         ensureOpen();
294         Map<String, Segment> segments = new HashMap<>();
295 
296         // first, go over and compute the search ones...
297         Searcher searcher = acquireSearcher("segments");
298         try {
299             for (AtomicReaderContext reader : searcher.reader().leaves()) {
300                 SegmentCommitInfo info = segmentReader(reader.reader()).getSegmentInfo();
301                 assert !segments.containsKey(info.info.name);
302                 Segment segment = new Segment(info.info.name);
303                 segment.search = true;
304                 segment.docCount = reader.reader().numDocs();
305                 segment.delDocCount = reader.reader().numDeletedDocs();
306                 segment.version = info.info.getVersion();
307                 segment.compound = info.info.getUseCompoundFile();
308                 try {
309                     segment.sizeInBytes = info.sizeInBytes();
310                 } catch (IOException e) {
311                     logger.trace("failed to get size for [{}]", e, info.info.name);
312                 }
313                 final SegmentReader segmentReader = segmentReader(reader.reader());
314                 segment.memoryInBytes = segmentReader.ramBytesUsed();
315                 // TODO: add more fine grained mem stats values to per segment info here
316                 segments.put(info.info.name, segment);
317             }
318         } finally {
319             searcher.close();
320         }
321 
322         // now, correlate or add the committed ones...
323         if (lastCommittedSegmentInfos != null) {
324             SegmentInfos infos = lastCommittedSegmentInfos;
325             for (SegmentCommitInfo info : infos) {
326                 Segment segment = segments.get(info.info.name);
327                 if (segment == null) {
328                     segment = new Segment(info.info.name);
329                     segment.search = false;
330                     segment.committed = true;
331                     segment.docCount = info.info.getDocCount();
332                     segment.delDocCount = info.getDelCount();
333                     segment.version = info.info.getVersion();
334                     segment.compound = info.info.getUseCompoundFile();
335                     try {
336                         segment.sizeInBytes = info.sizeInBytes();
337                     } catch (IOException e) {
338                         logger.trace("failed to get size for [{}]", e, info.info.name);
339                     }
340                     segments.put(info.info.name, segment);
341                 } else {
342                     segment.committed = true;
343                 }
344             }
345         }
346 
347         Segment[] segmentsArr = segments.values().toArray(new Segment[segments.values().size()]);
348         Arrays.sort(segmentsArr, new Comparator<Segment>() {
349             @Override
350             public int compare(Segment o1, Segment o2) {
351                 return (int) (o1.getGeneration() - o2.getGeneration());
352             }
353         });
354 
355         return segmentsArr;
356     }

363     public final boolean refreshNeeded() {
364         if (store.tryIncRef()) {
365             /*
366               we need to inc the store here since searcherManager.isSearcherCurrent()
367               acquires a searcher internally and that might keep a file open on the
368               store. this violates the assumption that all files are closed when
369               the store is closed so we need to make sure we increment it here
370              */
371             try {
372                 return !getSearcherManager().isSearcherCurrent();
373             } catch (IOException e) {
374                 logger.error("failed to access searcher manager", e);
375                 failEngine("failed to access searcher manager", e);
376                 throw new EngineException(shardId, "failed to access searcher manager", e);
377             } finally {
378                 store.decRef();
379             }
380         }
381         return false;
382     }

433     /** fail engine due to some error. the engine will also be closed. */
434     public void failEngine(String reason, Throwable failure) {
435         assert failure != null;
436         if (failEngineLock.tryLock()) {
437             store.incRef();
438             try {
439                 try {
440                     // we just go and close this engine - no way to recover
441                     closeNoLock("engine failed on: [" + reason + "]");
442                     // we first mark the store as corrupted before we notify any listeners
443                     // this must happen first otherwise we might try to reallocate so quickly
444                     // on the same node that we don't see the corrupted marker file when
445                     // the shard is initializing
446                     if (Lucene.isCorruptionException(failure)) {
447                         try {
448                             store.markStoreCorrupted(ExceptionsHelper.unwrap(failure, CorruptIndexException.class));
449                         } catch (IOException e) {
450                             logger.warn("Couldn't marks store corrupted", e);
451                         }
452                     }
453                 } finally {
454                     if (failedEngine != null) {
455                         logger.debug("tried to fail engine but engine is already failed. ignoring. [{}]", reason, failure);
456                         return;
457                     }
458                     logger.warn("failed engine [{}]", failure, reason);
459                     // we must set a failure exception, generate one if not supplied
460                     failedEngine = failure;
461                     failedEngineListener.onFailedEngine(shardId, reason, failure);
462                 }
463             } catch (Throwable t) {
464                 // don't bubble up these exceptions up
465                 logger.warn("failEngine threw exception", t);
466             } finally {
467                 store.decRef();
468             }
469         } else {
470             logger.debug("tried to fail engine but could not acquire lock - engine should be failed by now [{}]", reason, failure);
471         }
472     }

474     /** Check whether the engine should be failed */
475     protected boolean maybeFailEngine(String source, Throwable t) {
476         if (Lucene.isCorruptionException(t)) {
477             if (engineConfig.isFailEngineOnCorruption()) {
478                 failEngine("corrupt file detected source: [" + source + "]", t);
479                 return true;
480             } else {
481                 logger.warn("corrupt file detected source: [{}] but [{}] is set to [{}]", t, source,
482                         EngineConfig.INDEX_FAIL_ON_CORRUPTION_SETTING, engineConfig.isFailEngineOnCorruption());
483             }
484         } else if (ExceptionsHelper.isOOM(t)) {
485             failEngine("out of memory", t);
486             return true;
487         }
488         return false;
489     }

491     /** Wrap a Throwable in an {@code EngineClosedException} if the engine is already closed */
492     protected Throwable wrapIfClosed(Throwable t) {
493         if (isClosed.get()) {
494             if (t != failedEngine && failedEngine != null) {
495                 t.addSuppressed(failedEngine);
496             }
497             return new EngineClosedException(shardId, t);
498         }
499         return t;
500     }

506     /**
507      * Recovery allow to start the recovery process. It is built of three phases.
508      * <p/>
509      * <p>The first phase allows to take a snapshot of the master index. Once this
510      * is taken, no commit operations are effectively allowed on the index until the recovery
511      * phases are through.
512      * <p/>
513      * <p>The seconds phase takes a snapshot of the current transaction log.
514      * <p/>
515      * <p>The last phase returns the remaining transaction log. During this phase, no dirty
516      * operations are allowed on the index.
517      */
518     public static interface RecoveryHandler {
519 
520         void phase1(SnapshotIndexCommit snapshot) throws ElasticsearchException;
521 
522         void phase2(Translog.Snapshot snapshot) throws ElasticsearchException;
523 
524         void phase3(Translog.Snapshot snapshot) throws ElasticsearchException;
525     }

1024     /**
1025      * Method to close the engine while the write lock is held.
1026      */
1027     protected abstract void closeNoLock(String reason) throws ElasticsearchException;
1028 
1029     /**
1030      * Flush the engine (committing segments to disk and truncating the
1031      * translog) and close it.
1032      */
1033     public void flushAndClose() throws IOException {
1034         if (isClosed.get() == false) {
1035             logger.trace("flushAndClose now acquire writeLock");
1036             try (ReleasableLock _ = writeLock.acquire()) {
1037                 logger.trace("flushAndClose now acquired writeLock");
1038                 try {
1039                     logger.debug("flushing shard on close - this might take some time to sync files to disk");
1040                     try {
1041                         flush(); // TODO we might force a flush in the future since we have the write lock already even though recoveries are running.
1042                     } catch (FlushNotAllowedEngineException ex) {
1043                         logger.debug("flush not allowed during flushAndClose - skipping");
1044                     } catch (EngineClosedException ex) {
1045                         logger.debug("engine already closed - skipping flushAndClose");
1046                     }
1047                 } finally {
1048                     close(); // double close is not a problem
1049                 }
1050             }
1051         }
1052     }
1053 
1054     @Override
1055     public void close() throws IOException {
1056         // don't acquire the write lock if we are already closed
1057         if (isClosed.get() == false) {
1058             logger.debug("close now acquiring writeLock");
1059             try (ReleasableLock _ = writeLock.acquire()) {
1060                 logger.debug("close acquired writeLock");
1061                 closeNoLock("api");
1062             }
1063         }
1064     }
