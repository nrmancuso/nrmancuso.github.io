1   package org.hibernate.envers;
2   import java.lang.annotation.Retention;
3   import java.lang.annotation.Target;
4   import javax.persistence.MappedSuperclass;
5   
6   import static java.lang.annotation.ElementType.FIELD;
7   import static java.lang.annotation.ElementType.METHOD;
8   import static java.lang.annotation.ElementType.TYPE;
9   import static java.lang.annotation.RetentionPolicy.RUNTIME;
10  
11  /**
12   * The {@code AuditingOverride} annotation is used to override the auditing
13   * behavior of a superclass or single property inherited from {@link MappedSuperclass}
14   * type, or attribute inside an embedded component.
15   *
16   * @author Erik-Berndt Scheper
17   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
18   * @see javax.persistence.Embedded
19   * @see javax.persistence.Embeddable
20   * @see javax.persistence.MappedSuperclass  
21   * @see javax.persistence.AssociationOverride
22   * @see AuditJoinTable
23   */
24  @Target({ TYPE, METHOD, FIELD })
25  @Retention(RUNTIME)
26  public @interface AuditOverride {
27  
28  	/**
29  	 * @return Name of the field (or property) whose mapping is being overridden. Allows empty value if
30  	 * {@link AuditOverride} is used to change auditing behavior of all attributes inherited from
31  	 * {@link MappedSuperclass} type.
32  	 */
33  	String name() default "";
34  
35  	/**
36  	 * @return Indicates if the field (or property) is audited; defaults to {@code true}.
37  	 */
38  	boolean isAudited() default true;
39  
40  	/**
41  	 * @return New {@link AuditJoinTable} used for this field (or property). Its value
42  	 * is ignored if {@link #isAudited()} equals to {@code false}.
43  	 */
44  	AuditJoinTable auditJoinTable() default @AuditJoinTable;
45  
46  	/**
47  	 * @return Specifies class which field (or property) mapping is being overridden. <strong>Required</strong> if
48  	 * {@link AuditOverride} is used to change auditing behavior of attributes inherited from {@link MappedSuperclass}
49  	 * type.
50  	 */
51  	Class forClass() default void.class;
52  }
