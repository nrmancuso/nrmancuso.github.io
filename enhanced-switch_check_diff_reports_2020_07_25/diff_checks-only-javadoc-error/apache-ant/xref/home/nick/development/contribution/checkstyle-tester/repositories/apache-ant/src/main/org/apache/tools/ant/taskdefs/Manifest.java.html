38  /**
39   * Holds the data of a jar manifest.
40   *
41   * Manifests are processed according to the
42   * {@link <a href="http://java.sun.com/j2se/1.5.0/docs/guide/jar/jar.html">Jar
43   * file specification.</a>}.
44   * Specifically, a manifest element consists of
45   * a set of attributes and sections. These sections in turn may contain
46   * attributes. Note in particular that this may result in manifest lines
47   * greater than 72 bytes being wrapped and continued on the next
48   * line. If an application can not handle the continuation mechanism, it
49   * is a defect in the application, not this task.
50   *
51   *
52   * @since Ant 1.4
53   */
54  public class Manifest {
55      /** The standard manifest version header */
56      public static final String ATTRIBUTE_MANIFEST_VERSION
57          = "Manifest-Version";
58  
59      /** The standard Signature Version header */
60      public static final String ATTRIBUTE_SIGNATURE_VERSION
61          = "Signature-Version";
62  
63      /** The Name Attribute is the first in a named section */
64      public static final String ATTRIBUTE_NAME = "Name";
65  
66      /** The From Header is disallowed in a Manifest */
67      public static final String ATTRIBUTE_FROM = "From";
68  
69      /** The Class-Path Header is special - it can be duplicated */
70      public static final String ATTRIBUTE_CLASSPATH = "Class-Path";
71  
72      /** Default Manifest version if one is not specified */
73      public static final  String DEFAULT_MANIFEST_VERSION = "1.0";
74  
75      /** The max length of a line in a Manifest */
76      public static final int MAX_LINE_LENGTH = 72;
77  
78      /**
79       * Max length of a line section which is continued. Need to allow
80       * for the CRLF.
81       */
82      public static final int MAX_SECTION_LENGTH = MAX_LINE_LENGTH - 2;
83  
84      /** The End-Of-Line marker in manifests */
85      public static final String EOL = "\r\n";
86      /** Error for attributes */
87      public static final String ERROR_FROM_FORBIDDEN = "Manifest attributes should not start "
88                          + "with \"" + ATTRIBUTE_FROM + "\" in \"";
89  
90      /** Encoding to be used for JAR files. */
91      public static final String JAR_ENCODING = "UTF-8";
92  
93      private static final String ATTRIBUTE_MANIFEST_VERSION_LC =
94          ATTRIBUTE_MANIFEST_VERSION.toLowerCase(Locale.ENGLISH);
95      private static final String ATTRIBUTE_NAME_LC =
96          ATTRIBUTE_NAME.toLowerCase(Locale.ENGLISH);
97      private static final String ATTRIBUTE_FROM_LC =
98          ATTRIBUTE_FROM.toLowerCase(Locale.ENGLISH);
99      private static final String ATTRIBUTE_CLASSPATH_LC =
100         ATTRIBUTE_CLASSPATH.toLowerCase(Locale.ENGLISH);

102     /**
103      * An attribute for the manifest.
104      * Those attributes that are not nested into a section will be added to the "Main" section.
105      */
106     public static class Attribute {
107 
108         /**
109          * Maximum length of the name to have the value starting on the same
110          * line as the name. This to stay under 72 bytes total line length
111          * (including CRLF).
112          */
113         private static final int MAX_NAME_VALUE_LENGTH = 68;
114 
115         /**
116          * Maximum length of the name according to the jar specification.
117          * In this case the first line will have 74 bytes total line length
118          * (including CRLF). This conflicts with the 72 bytes total line length
119          * max, but is the only possible conclusion from the manifest specification, if
120          * names with 70 bytes length are allowed, have to be on the first line, and
121          * have to be followed by ": ".
122          */
123         private static final int MAX_NAME_LENGTH = 70;
124 
125         /** The attribute's name */
126         private String name = null;
127 
128         /** The attribute's value */
129         private Vector<String> values = new Vector<String>();
130 
131         /**
132          * For multivalued attributes, this is the index of the attribute
133          * currently being defined.
134          */
135         private int currentIndex = 0;
136 
137         /**
138          * Construct an empty attribute */
139         public Attribute() {
140         }
141 
142         /**
143          * Construct an attribute by parsing a line from the Manifest
144          *
145          * @param line the line containing the attribute name and value
146          *
147          * @throws ManifestException if the line is not valid
148          */
149         public Attribute(String line) throws ManifestException {
150             parse(line);
151         }
152 
153         /**
154          * Construct a manifest by specifying its name and value
155          *
156          * @param name the attribute's name
157          * @param value the Attribute's value
158          */
159         public Attribute(String name, String value) {
160             this.name = name;
161             setValue(value);
162         }
163 
164         /**
165          * @see java.lang.Object#hashCode
166          * @return a hashcode based on the key and values.
167          */
168         public int hashCode() {
169             int hashCode = 0;
170 
171             if (name != null) {
172                 hashCode += getKey().hashCode();
173             }
174 
175             hashCode += values.hashCode();
176             return hashCode;
177         }
178 
179         /**
180          * @param rhs the object to check for equality.
181          * @see java.lang.Object#equals
182          * @return true if the key and values are the same.
183          */
184         public boolean equals(Object rhs) {
185             if (rhs == null || rhs.getClass() != getClass()) {
186                 return false;
187             }
188 
189             if (rhs == this) {
190                 return true;
191             }
192 
193             Attribute rhsAttribute = (Attribute) rhs;
194             String lhsKey = getKey();
195             String rhsKey = rhsAttribute.getKey();
196             if ((lhsKey == null && rhsKey != null)
197                  || (lhsKey != null && !lhsKey.equals(rhsKey))) {
198                 return false;
199             }
200 
201             return values.equals(rhsAttribute.values);
202         }
203 
204         /**
205          * Parse a line into name and value pairs
206          *
207          * @param line the line to be parsed
208          *
209          * @throws ManifestException if the line does not contain a colon
210          * separating the name and value
211          */
212         public void parse(String line) throws ManifestException {
213             int index = line.indexOf(": ");
214             if (index == -1) {
215                 throw new ManifestException("Manifest line \"" + line
216                     + "\" is not valid as it does not "
217                     + "contain a name and a value separated by ': ' ");
218             }
219             name = line.substring(0, index);
220             setValue(line.substring(index + 2));
221         }
222 
223         /**
224          * Set the Attribute's name; required
225          *
226          * @param name the attribute's name
227          */
228         public void setName(String name) {
229             this.name = name;
230         }
231 
232         /**
233          * Get the Attribute's name
234          *
235          * @return the attribute's name.
236          */
237         public String getName() {
238             return name;
239         }
240 
241         /**
242          * Get the attribute's Key - its name in lower case.
243          *
244          * @return the attribute's key.
245          */
246         public String getKey() {
247             if (name == null) {
248                 return null;
249             }
250             return name.toLowerCase(Locale.ENGLISH);
251         }
252 
253         /**
254          * Set the Attribute's value; required
255          *
256          * @param value the attribute's value
257          */
258         public void setValue(String value) {
259             if (currentIndex >= values.size()) {
260                 values.addElement(value);
261                 currentIndex = values.size() - 1;
262             } else {
263                 values.setElementAt(value, currentIndex);
264             }
265         }
266 
267         /**
268          * Get the Attribute's value.
269          *
270          * @return the attribute's value.
271          */
272         public String getValue() {
273             if (values.size() == 0) {
274                 return null;
275             }
276 
277             String fullValue = "";
278             for (Enumeration<String> e = getValues(); e.hasMoreElements();) {
279                 String value = e.nextElement();
280                 fullValue += value + " ";
281             }
282             return fullValue.trim();
283         }
284 
285         /**
286          * Add a new value to this attribute - making it multivalued.
287          *
288          * @param value the attribute's additional value
289          */
290         public void addValue(String value) {
291             currentIndex++;
292             setValue(value);
293         }
294 
295         /**
296          * Get all the attribute's values.
297          *
298          * @return an enumeration of the attributes values
299          */
300         public Enumeration<String> getValues() {
301             return values.elements();
302         }
303 
304         /**
305          * Add a continuation line from the Manifest file.
306          *
307          * When lines are too long in a manifest, they are continued on the
308          * next line by starting with a space. This method adds the continuation
309          * data to the attribute value by skipping the first character.
310          *
311          * @param line the continuation line.
312          */
313         public void addContinuation(String line) {
314             String currentValue = (String) values.elementAt(currentIndex);
315             setValue(currentValue + line.substring(1));
316         }
317 
318         /**
319          * Write the attribute out to a print writer without
320          * flattening multi-values attributes (i.e. Class-Path).
321          *
322          * @param writer the Writer to which the attribute is written
323          *
324          * @throws IOException if the attribute value cannot be written
325          */
326         public void write(PrintWriter writer) throws IOException {
327             write(writer, false);
328         }
329 
330         /**
331          * Write the attribute out to a print writer.
332          *
333          * @param writer the Writer to which the attribute is written
334          * @param flatten whether to collapse multi-valued attributes
335          *        (i.e. potentially Class-Path) Class-Path into a
336          *        single attribute.
337          *
338          * @throws IOException if the attribute value cannot be written
339          * @since Ant 1.8.0
340          */
341         public void write(PrintWriter writer, boolean flatten)
342             throws IOException {
343             if (!flatten) {
344             for (Enumeration<String> e = getValues(); e.hasMoreElements();) {
345                 writeValue(writer, e.nextElement());
346             }
347             } else {
348                 writeValue(writer, getValue());
349             }
350         }
351 
352         /**
353          * Write a single attribute value out
354          *
355          * @param writer the Writer to which the attribute is written
356          * @param value the attribute value
357          *
358          * @throws IOException if the attribute value cannot be written
359          */
360         private void writeValue(PrintWriter writer, String value)
361              throws IOException {
362             String line = null;
363             int nameLength = name.getBytes(JAR_ENCODING).length;
364             if (nameLength > MAX_NAME_VALUE_LENGTH) {
365                 if (nameLength > MAX_NAME_LENGTH) {
366                     throw new IOException("Unable to write manifest line "
367                             + name + ": " + value);
368                 }
369                 writer.print(name + ": " + EOL);
370                 line = " " + value;
371             } else {
372                 line = name + ": " + value;
373             }
374             while (line.getBytes(JAR_ENCODING).length > MAX_SECTION_LENGTH) {
375                 // try to find a MAX_LINE_LENGTH byte section
376                 int breakIndex = MAX_SECTION_LENGTH;
377                 if (breakIndex >= line.length()) {
378                     breakIndex = line.length() - 1;
379                 }
380                 String section = line.substring(0, breakIndex);
381                 while (section.getBytes(JAR_ENCODING).length > MAX_SECTION_LENGTH
382                      && breakIndex > 0) {
383                     breakIndex--;
384                     section = line.substring(0, breakIndex);
385                 }
386                 if (breakIndex == 0) {
387                     throw new IOException("Unable to write manifest line "
388                         + name + ": " + value);
389                 }
390                 writer.print(section + EOL);
391                 line = " " + line.substring(breakIndex);
392             }
393             writer.print(line + EOL);
394         }
395     }

397     /**
398      * A manifest section - you can nest attribute elements into sections.
399      * A section consists of a set of attribute values,
400      * separated from other sections by a blank line.
401      */
402     public static class Section {
403         /** Warnings for this section */
404         private Vector<String> warnings = new Vector<String>();
405 
406         /**
407          * The section's name if any. The main section in a
408          * manifest is unnamed.
409          */
410         private String name = null;
411 
412         /** The section's attributes.*/
413         private Map<String, Attribute> attributes = new LinkedHashMap<String, Attribute>();
414 
415         /**
416          * The name of the section; optional -default is the main section.
417          * @param name the section's name
418          */
419         public void setName(String name) {
420             this.name = name;
421         }
422 
423         /**
424          * Get the Section's name.
425          *
426          * @return the section's name.
427          */
428         public String getName() {
429             return name;
430         }
431 
432         /**
433          * Read a section through a reader.
434          *
435          * @param reader the reader from which the section is read
436          *
437          * @return the name of the next section if it has been read as
438          *         part of this section - This only happens if the
439          *         Manifest is malformed.
440          *
441          * @throws ManifestException if the section is not valid according
442          *         to the JAR spec
443          * @throws IOException if the section cannot be read from the reader.
444          */
445         public String read(BufferedReader reader)
446              throws ManifestException, IOException {
447             Attribute attribute = null;
448             while (true) {
449                 String line = reader.readLine();
450                 if (line == null || line.length() == 0) {
451                     return null;
452                 }
453                 if (line.charAt(0) == ' ') {
454                     // continuation line
455                     if (attribute == null) {
456                         if (name != null) {
457                             // a continuation on the first line is a
458                             // continuation of the name - concatenate this
459                             // line and the name
460                             name += line.substring(1);
461                         } else {
462                             throw new ManifestException("Can't start an "
463                                 + "attribute with a continuation line " + line);
464                         }
465                     } else {
466                         attribute.addContinuation(line);
467                     }
468                 } else {
469                     attribute = new Attribute(line);
470                     String nameReadAhead = addAttributeAndCheck(attribute);
471                     // refresh attribute in case of multivalued attributes.
472                     attribute = getAttribute(attribute.getKey());
473                     if (nameReadAhead != null) {
474                         return nameReadAhead;
475                     }
476                 }
477             }
478         }
479 
480         /**
481          * Merge in another section without merging Class-Path attributes.
482          *
483          * @param section the section to be merged with this one.
484          *
485          * @throws ManifestException if the sections cannot be merged.
486          */
487         public void merge(Section section) throws ManifestException {
488             merge(section, false);
489         }
490 
491         /**
492          * Merge in another section
493          *
494          * @param section the section to be merged with this one.
495          * @param mergeClassPaths whether Class-Path attributes should
496          *        be merged.
497          *
498          * @throws ManifestException if the sections cannot be merged.
499          */
500         public void merge(Section section, boolean mergeClassPaths)
501             throws ManifestException {
502             if (name == null && section.getName() != null
503                 || (name != null && section.getName() != null
504                     && !(name.toLowerCase(Locale.ENGLISH)
505                          .equals(section.getName().toLowerCase(Locale.ENGLISH))))
506                 ) {
507                 throw new ManifestException("Unable to merge sections "
508                     + "with different names");
509             }
510 
511             Enumeration<String> e = section.getAttributeKeys();
512             Attribute classpathAttribute = null;
513             while (e.hasMoreElements()) {
514                 String attributeName = e.nextElement();
515                 Attribute attribute = section.getAttribute(attributeName);
516                 if (attributeName.equalsIgnoreCase(ATTRIBUTE_CLASSPATH)) {
517                     if (classpathAttribute == null) {
518                         classpathAttribute = new Attribute();
519                         classpathAttribute.setName(ATTRIBUTE_CLASSPATH);
520                     }
521                     Enumeration<String> cpe = attribute.getValues();
522                     while (cpe.hasMoreElements()) {
523                         String value = cpe.nextElement();
524                         classpathAttribute.addValue(value);
525                     }
526                 } else {
527                     // the merge file always wins
528                     storeAttribute(attribute);
529                 }
530             }
531 
532             if (classpathAttribute != null) {
533                 if (mergeClassPaths) {
534                     Attribute currentCp = getAttribute(ATTRIBUTE_CLASSPATH);
535                     if (currentCp != null) {
536                         for (Enumeration<String> attribEnum = currentCp.getValues();
537                              attribEnum.hasMoreElements(); ) {
538                             String value = attribEnum.nextElement();
539                             classpathAttribute.addValue(value);
540                         }
541                     }
542                 }
543                 storeAttribute(classpathAttribute);
544             }
545 
546             // add in the warnings
547             Enumeration<String> warnEnum = section.warnings.elements();
548             while (warnEnum.hasMoreElements()) {
549                 warnings.addElement(warnEnum.nextElement());
550             }
551         }
552 
553         /**
554          * Write the section out to a print writer without flattening
555          * multi-values attributes (i.e. Class-Path).
556          *
557          * @param writer the Writer to which the section is written
558          *
559          * @throws IOException if the section cannot be written
560          */
561         public void write(PrintWriter writer) throws IOException {
562             write(writer, false);
563         }
564 
565         /**
566          * Write the section out to a print writer.
567          *
568          * @param writer the Writer to which the section is written
569          * @param flatten whether to collapse multi-valued attributes
570          *        (i.e. potentially Class-Path) Class-Path into a
571          *        single attribute.
572          *
573          * @throws IOException if the section cannot be written
574          * @since Ant 1.8.0
575          */
576         public void write(PrintWriter writer, boolean flatten)
577             throws IOException {
578             if (name != null) {
579                 Attribute nameAttr = new Attribute(ATTRIBUTE_NAME, name);
580                 nameAttr.write(writer);
581             }
582             Enumeration<String> e = getAttributeKeys();
583             while (e.hasMoreElements()) {
584                 String key = e.nextElement();
585                 Attribute attribute = getAttribute(key);
586                 attribute.write(writer, flatten);
587             }
588             writer.print(EOL);
589         }

639         /**
640          * Add an attribute to the section.
641          *
642          * @param attribute the attribute to be added to the section
643          *
644          * @exception ManifestException if the attribute is not valid.
645          */
646         public void addConfiguredAttribute(Attribute attribute)
647              throws ManifestException {
648             String check = addAttributeAndCheck(attribute);
649             if (check != null) {
650                 throw new BuildException("Specify the section name using "
651                     + "the \"name\" attribute of the <section> element rather "
652                     + "than using a \"Name\" manifest attribute");
653             }
654         }
655 
656         /**
657          * Add an attribute to the section
658          *
659          * @param attribute the attribute to be added.
660          *
661          * @return the value of the attribute if it is a name
662          *         attribute - null other wise
663          *
664          * @exception ManifestException if the attribute already
665          *            exists in this section.
666          */
667         public String addAttributeAndCheck(Attribute attribute)
668              throws ManifestException {
669             if (attribute.getName() == null || attribute.getValue() == null) {
670                 throw new BuildException("Attributes must have name and value");
671             }
672             String attributeKey = attribute.getKey();
673             if (attributeKey.equals(ATTRIBUTE_NAME_LC)) {
674                 warnings.addElement("\"" + ATTRIBUTE_NAME + "\" attributes "
675                     + "should not occur in the main section and must be the "
676                     + "first element in all other sections: \""
677                     + attribute.getName() + ": " + attribute.getValue() + "\"");
678                 return attribute.getValue();
679             }
680 
681             if (attributeKey.startsWith(ATTRIBUTE_FROM_LC)) {
682                 warnings.addElement(ERROR_FROM_FORBIDDEN
683                     + attribute.getName() + ": " + attribute.getValue() + "\"");
684             } else {
685                 // classpath attributes go into a vector
686                 if (attributeKey.equals(ATTRIBUTE_CLASSPATH_LC)) {
687                     Attribute classpathAttribute =
688                         (Attribute) attributes.get(attributeKey);
689 
690                     if (classpathAttribute == null) {
691                         storeAttribute(attribute);
692                     } else {
693                         warnings.addElement("Multiple Class-Path attributes "
694                             + "are supported but violate the Jar "
695                             + "specification and may not be correctly "
696                             + "processed in all environments");
697                         Enumeration<String> e = attribute.getValues();
698                         while (e.hasMoreElements()) {
699                             String value = e.nextElement();
700                             classpathAttribute.addValue(value);
701                         }
702                     }
703                 } else if (attributes.containsKey(attributeKey)) {
704                     throw new ManifestException("The attribute \""
705                         + attribute.getName() + "\" may not occur more "
706                         + "than once in the same section");
707                 } else {
708                     storeAttribute(attribute);
709                 }
710             }
711             return null;
712         }

714         /**
715          * Clone this section
716          *
717          * @return the cloned Section
718          * @since Ant 1.5.2
719          */
720         public Object clone() {
721             Section cloned = new Section();
722             cloned.setName(name);
723             Enumeration<String> e = getAttributeKeys();
724             while (e.hasMoreElements()) {
725                 String key = e.nextElement();
726                 Attribute attribute = getAttribute(key);
727                 cloned.storeAttribute(new Attribute(attribute.getName(),
728                                                     attribute.getValue()));
729             }
730             return cloned;
731         }

733         /**
734          * Store an attribute and update the index.
735          *
736          * @param attribute the attribute to be stored
737          */
738         private void storeAttribute(Attribute attribute) {
739             if (attribute == null) {
740                 return;
741             }
742             String attributeKey = attribute.getKey();
743             attributes.put(attributeKey, attribute);
744         }

775             }

777             Section rhsSection = (Section) rhs;
778 
779             return attributes.equals(rhsSection.attributes);
780         }

793     /**
794      * Construct a manifest from Ant's default manifest file.
795      *
796      * @return the default manifest.
797      * @exception BuildException if there is a problem loading the
798      *            default manifest
799      */
800     public static Manifest getDefaultManifest() throws BuildException {
801         InputStream in = null;
802         InputStreamReader insr = null;
803         try {
804             String defManifest = "/org/apache/tools/ant/defaultManifest.mf";
805             in = Manifest.class.getResourceAsStream(defManifest);
806             if (in == null) {
807                 throw new BuildException("Could not find default manifest: "
808                     + defManifest);
809             }
810             try {
811                 insr = new InputStreamReader(in, "UTF-8");
812                 Manifest defaultManifest = new Manifest(insr);
813                 String version = System.getProperty("java.runtime.version");
814                 if (version == null) {
815                     version = System.getProperty("java.vm.version");
816                 }
817                 Attribute createdBy = new Attribute("Created-By",
818                     version + " ("
819                     + System.getProperty("java.vm.vendor") + ")");
820                 defaultManifest.getMainSection().storeAttribute(createdBy);
821                 return defaultManifest;
822             } catch (UnsupportedEncodingException e) {
823                 insr = new InputStreamReader(in);
824                 return new Manifest(insr);
825             }
826         } catch (ManifestException e) {
827             throw new BuildException("Default manifest is invalid !!", e);
828         } catch (IOException e) {
829             throw new BuildException("Unable to read default manifest", e);
830         } finally {
831             FileUtils.close(insr);
832             FileUtils.close(in);
833         }
834     }

841     /**
842      * Read a manifest file from the given reader
843      *
844      * @param r is the reader from which the Manifest is read
845      *
846      * @throws ManifestException if the manifest is not valid according
847      *         to the JAR spec
848      * @throws IOException if the manifest cannot be read from the reader.
849      */
850     public Manifest(Reader r) throws ManifestException, IOException {
851         BufferedReader reader = new BufferedReader(r);
852         // This should be the manifest version
853         String nextSectionName = mainSection.read(reader);
854         String readManifestVersion
855             = mainSection.getAttributeValue(ATTRIBUTE_MANIFEST_VERSION);
856         if (readManifestVersion != null) {
857             manifestVersion = readManifestVersion;
858             mainSection.removeAttribute(ATTRIBUTE_MANIFEST_VERSION);
859         }
860 
861         String line = null;
862         while ((line = reader.readLine()) != null) {
863             if (line.length() == 0) {
864                 continue;
865             }
866 
867             Section section = new Section();
868             if (nextSectionName == null) {
869                 Attribute sectionName = new Attribute(line);
870                 if (!sectionName.getName().equalsIgnoreCase(ATTRIBUTE_NAME)) {
871                     throw new ManifestException("Manifest sections should "
872                         + "start with a \"" + ATTRIBUTE_NAME
873                         + "\" attribute and not \""
874                         + sectionName.getName() + "\"");
875                 }
876                 nextSectionName = sectionName.getValue();
877             } else {
878                 // we have already started reading this section
879                 // this line is the first attribute. set it and then
880                 // let the normal read handle the rest
881                 Attribute firstAttribute = new Attribute(line);
882                 section.addAttributeAndCheck(firstAttribute);
883             }
884 
885             section.setName(nextSectionName);
886             nextSectionName = section.read(reader);
887             addConfiguredSection(section);
888         }
889     }

955     /**
956      * Merge the contents of the given manifest into this manifest
957      *
958      * @param other the Manifest to be merged with this one.
959      * @param overwriteMain whether to overwrite the main section
960      *        of the current manifest
961      * @param mergeClassPaths whether Class-Path attributes should be
962      *        merged.
963      *
964      * @throws ManifestException if there is a problem merging the
965      *         manifest according to the Manifest spec.
966      *
967      * @since Ant 1.8.0
968      */
969     public void merge(Manifest other, boolean overwriteMain,
970                       boolean mergeClassPaths)
971          throws ManifestException {
972         if (other != null) {
973              if (overwriteMain) {
974                  mainSection = (Section) other.mainSection.clone();
975              } else {
976                  mainSection.merge(other.mainSection, mergeClassPaths);
977              }
978 
979              if (other.manifestVersion != null) {
980                  manifestVersion = other.manifestVersion;
981              }
982 
983              Enumeration<String> e = other.getSectionNames();
984              while (e.hasMoreElements()) {
985                  String sectionName = e.nextElement();
986                  Section ourSection = sections.get(sectionName);
987                  Section otherSection
988                     = other.sections.get(sectionName);
989                  if (ourSection == null) {
990                      if (otherSection != null) {
991                          addConfiguredSection((Section) otherSection.clone());
992                      }
993                  } else {
994                      ourSection.merge(otherSection, mergeClassPaths);
995                  }
996              }
997          }
998     }

1012     /**
1013     * Write the manifest out to a print writer.
1014     *
1015     * @param writer the Writer to which the manifest is written
1016     * @param flatten whether to collapse multi-valued attributes
1017     *        (i.e. potentially Class-Path) Class-Path into a single
1018     *        attribute.
1019     *
1020     * @throws IOException if the manifest cannot be written
1021     * @since Ant 1.8.0
1022     */
1023     public void write(PrintWriter writer, boolean flatten) throws IOException {
1024         writer.print(ATTRIBUTE_MANIFEST_VERSION + ": " + manifestVersion + EOL);
1025         String signatureVersion
1026             = mainSection.getAttributeValue(ATTRIBUTE_SIGNATURE_VERSION);
1027         if (signatureVersion != null) {
1028             writer.print(ATTRIBUTE_SIGNATURE_VERSION + ": "
1029                 + signatureVersion + EOL);
1030             mainSection.removeAttribute(ATTRIBUTE_SIGNATURE_VERSION);
1031         }
1032         mainSection.write(writer, flatten);
1033 
1034         // add it back
1035         if (signatureVersion != null) {
1036             try {
1037                 Attribute svAttr = new Attribute(ATTRIBUTE_SIGNATURE_VERSION,
1038                     signatureVersion);
1039                 mainSection.addConfiguredAttribute(svAttr);
1040             } catch (ManifestException e) {
1041                 // shouldn't happen - ignore
1042             }
1043         }
1044 
1045         for (String sectionName : sections.keySet()) {
1046             Section section = getSection(sectionName);
1047             section.write(writer, flatten);
1048         }
1049     }

1072     public Enumeration<String> getWarnings() {
1073         Vector<String> warnings = new Vector<String>();
1074

1076         while (warnEnum.hasMoreElements()) {
1077             warnings.addElement(warnEnum.nextElement());
1078         }
1079 
1080         // create a vector and add in the warnings for all the sections
1081         for (Section section : sections.values()) {
1082             Enumeration<String> e2 = section.getWarnings();
1083             while (e2.hasMoreElements()) {
1084                 warnings.addElement(e2.nextElement());
1085             }
1086         }
1087
