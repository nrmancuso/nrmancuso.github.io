41  import java.util.Iterator;
42  import java.util.List;

44  import org.dcm4che.data.Dataset;
45  import org.dcm4che.dict.Tags;
46  import org.dcm4che.util.Base64;
47  import org.dcm4che.util.UIDGenerator;
48  import org.dom4j.Document;
49  import org.dom4j.Element;
50  import org.xml.sax.ContentHandler;
51  
52  /**
53   * @author Gunter Zeilinger <gunterze@gmail.com>
54   * @version $Revision$ $Date$
55   * @since Dec 27, 2007
56   */
57  public class MDMService extends ORU_MDMService {
58  
59      public boolean process(MSH msh, Document msg, ContentHandler hl7out, String[] xslSubdirs)
60              throws HL7Exception {
61          try
62          {
63              byte[] pdf = getPDF(msg);
64              if (pdf == null) {
65                  log.warn("Ignore received " + msh.messageType + "^" +
66                          msh.triggerEvent + " without encapsulated report");
67                  return true;
68              }
69              Dataset doc = xslt(msg, xslPath, xslSubdirs);
70              if (!doc.containsValue(Tags.StudyInstanceUID)) {
71                  log.info("No Study Instance UID in MDM - store report in new Study");
72                  doc.putUI(Tags.StudyInstanceUID,
73                          UIDGenerator.getInstance().createUID());
74              }
75              doc.putOB(Tags.EncapsulatedDocument, pdf);
76              storeSR(doc);
77          }
78          catch (Exception e)
79          {
80              throw new HL7Exception("AE", e.getMessage(), e);
81          }      
82          return true;
83      }
84  
85      static String toString(Object el) {
86          return el != null ? ((Element) el).getText() : "";
87      }
88      
89      private byte[] getPDF(Document msg) {
90          List obxs = msg.getRootElement().elements("OBX");
91          String base64 = "";
92          for (Iterator iter = obxs.iterator(); iter.hasNext();) {
93              Element obx = (Element) iter.next();
94              List fds = obx.elements();
95              if (fds.size() > 4) {
96                  List cmps = ((Element) fds.get(4)).elements();
97                  if (cmps.size() > 3 && "PDF".equals(toString(cmps.get(1)))) {
98                      String obx2 = toString(fds.get(1));
99                      if (!"ED".equals(obx2)) {
100                         log.warn("Detect encapsulated report in OBX with OBX-2: '"
101                                 + obx2 + "' instead 'ED'");
102                     }
103                     base64 += toString(cmps.get(3));
104                 }
105            }
106         }
107         return base64.length() > 0 ? Base64.base64ToByteArray(base64) : null;
108     }
