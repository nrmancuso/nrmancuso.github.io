142     protected void startService() throws Exception {
143         logDir = new File(System.getProperty("jboss.server.log.dir"));

145                 ianListener , IANScuService.NOTIF_FILTER, null);

153     private void onIAN(Dataset mpps) {
154         String aet = mpps.getString(Tags.PerformedStationAET);
155         boolean logMPPS = contains(logAETs, aet);
156         if (logMPPS) {
157             logMPPS(aet, mpps);
158         }
159         Templates stylesheet = templates.getTemplatesForAET(aet, MPPS2GPWL_XSL);
160         if (stylesheet == null) {
161             log.info("No mpps2gpwl.xsl found for " + aet);
162             return;
163         }
164         Dataset wkitems = DcmObjectFactory.getInstance().newDataset();
165         try {
166             XSLTUtils.xslt(mpps, stylesheet, wkitems);
167         } catch (Exception e) {
168             log.error("Failed to create work items triggered by MPPS from " + aet, e);
169             return;
170         }
171         DcmElement wkitemSeq = wkitems.get(PrivateTags.WorkItemSeq);
172         log.info("Creating " + wkitemSeq.countItems() + " work item(s) triggered by MPPS from " + aet);
173         try {
174             createWorkItems(wkitemSeq, contains(pgpAETs, aet), contains(appendAETs, aet));
175         } catch (Exception e) {
176             log.error("Failed to create work item triggered by MPPS from " + aet, e);
177         }
178     }
179 
180     private synchronized void createWorkItems(DcmElement wkitemSeq, 
181             boolean checkPGP, boolean checkAppend) throws Exception {
182         GPWLManager gpwlmgr = getGPWLManager();
183         for (int i = 0, n = wkitemSeq.countItems(); i < n; i++) {
184             gpwlmgr.addWorklistItem(wkitemSeq.getItem(i), checkPGP, checkAppend);
185         }
186     }
187 
188     private static boolean contains(Object[] a, Object e) {
189         for (int i = 0; i < a.length; i++) {
190             if (a[i].equals(e)) {
191                 return true;
192             }
193         }
194         return false;
195     }
196 
197     private GPWLManager getGPWLManager() throws Exception {
198         return ((GPWLManagerHome) EJBHomeFactory.getFactory().lookup(
199                 GPWLManagerHome.class, GPWLManagerHome.JNDI_NAME)).create();
200     }
201 
202     private void logMPPS(String aet, Dataset mpps) {
203         File dir = new File(logDir, aet);
204         SimpleDateFormat df = new SimpleDateFormat(LOGFILE_PATTERN);
205         File f = new File(dir, df.format(new Date()));
206         dir.mkdir();
207         try {
208             log.info("Log MPPS attributes to " + f);
209             XSLTUtils.writeTo(mpps, f);
210         } catch (Exception e) {
211             log.warn("Log MPPS attributes failed:", e);
212         }
213     }
