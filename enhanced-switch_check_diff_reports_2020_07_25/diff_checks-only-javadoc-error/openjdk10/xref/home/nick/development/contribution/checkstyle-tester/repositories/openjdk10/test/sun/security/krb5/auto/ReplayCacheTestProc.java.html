50  /**
51   * This test runs multiple acceptor Procs to mimic AP-REQ replays.
52   * These system properties are supported:
53   *
54   * - test.libs on what types of acceptors to use
55   *   Format: CSV of (J|N|N<suffix>=<libname>|J<suffix>=<launcher>)
56   *   Default: J,N on Solaris and Linux where N is available, or J
57   *   Example: J,N,N14=/krb5-1.14/lib/libgssapi_krb5.so,J8=/java8/bin/java
58   *
59   * - test.runs on manual runs. If empty, a iterate through all pattern
60   *   Format: (req# | client# service#) acceptor# expected, ...
61   *   Default: null
62   *   Example: c0s0Jav,c1s1N14av,r0Jbx means 0th req is new c0->s0 sent to Ja,
63   *            1st req is new c1 to s1 sent to N14a,
64   *            2nd req is old (0th replayed) sent to Jb.
65   *            a/b at the end of acceptor is different acceptors of the same lib
66   *
67   * - test.autoruns on number of automatic runs
68   *   Format: number
69   *   Default: 100
70   */
71  public class ReplayCacheTestProc {

82      static {
83          String tmp = System.getProperty("test.service");
84          SERVICE = (tmp == null) ? "service" : tmp;
85          uid = jdk.internal.misc.VM.geteuid();
86          // Where should the rcache be saved. KRB5RCACHEDIR is not
87          // recognized on Solaris (might be supported on Solaris 12),
88          // and directory name is different when launched by root.
89          // See manpage krb5envvar(5) on KRB5RCNAME.
90          if (System.getProperty("os.name").startsWith("SunOS")) {
91              if (uid == 0) {
92                  cwd = "/var/krb5/rcache/root/";
93              } else {
94                  cwd = "/var/krb5/rcache/";
95              }
96          } else {
97              cwd = System.getProperty("user.dir");
98          }
99      }

113     public static void main0(String[] args) throws Exception {
114         System.setProperty("java.security.krb5.conf", OneKDC.KRB5_CONF);
115         if (args.length == 0) { // The controller
116             int nc = 5;     // number of clients
117             int ns = 5;     // number of services
118             String[] libs;  // available acceptor types:
119                             // J: java
120                             // J<suffix>=<java launcher>: another java
121                             // N: default native lib
122                             // N<suffix>=<libname>: another native lib
123             Ex[] result;
124             int numPerType = 2; // number of acceptors per type
125 
126             KDC kdc = KDC.create(OneKDC.REALM, HOST, 0, true);
127             for (int i=0; i<nc; i++) {
128                 kdc.addPrincipal(client(i), OneKDC.PASS);
129             }
130             kdc.addPrincipalRandKey("krbtgt/" + OneKDC.REALM);
131             for (int i=0; i<ns; i++) {
132                 kdc.addPrincipalRandKey(service(i));
133             }
134 
135             kdc.writeKtab(OneKDC.KTAB);
136             KDC.saveConfig(OneKDC.KRB5_CONF, kdc);
137 
138             // User-provided libs
139             String userLibs = System.getProperty("test.libs");
140 
141             if (userLibs != null) {
142                 libs = userLibs.split(",");
143             } else {
144                 if (Platform.isOSX() || Platform.isWindows()) {
145                     // macOS uses Heimdal and Windows has no native lib
146                     libs = new String[]{"J"};
147                 } else {
148                     if (acceptor("N", "sanity").waitFor() != 0) {
149                         Proc.d("Native mode sanity check failed, only java");
150                         libs = new String[]{"J"};
151                     } else {
152                         libs = new String[]{"J", "N"};
153                     }
154                 }
155             }
156 
157             pi = Proc.create("ReplayCacheTestProc").debug("C")
158                     .args("initiator")
159                     .start();
160 
161             int na = libs.length * numPerType;  // total number of acceptors
162             pa = new Proc[na];
163 
164             // Acceptors, numPerType for 1st, numForType for 2nd, ...
165             for (int i=0; i<na; i++) {
166                 pa[i] = acceptor(libs[i/numPerType],
167                         "" + (char)('a' + i%numPerType));
168             }
169 
170             // Manual runs
171             String userRuns = System.getProperty("test.runs");
172 
173             if (userRuns == null) {
174                 result = new Ex[Integer.parseInt(
175                         System.getProperty("test.autoruns", "100"))];
176                 Random r = new Random();
177                 for (int i = 0; i < result.length; i++) {
178                     boolean expected = reqs.isEmpty() || r.nextBoolean();
179                     result[i] = new Ex(
180                             i,
181                             expected ?
182                                     req(r.nextInt(nc), r.nextInt(ns)) :
183                                     r.nextInt(reqs.size()),
184                             pa[r.nextInt(na)],
185                             expected);
186                 }
187             } else if (userRuns.isEmpty()) {
188                 int count = 0;
189                 result = new Ex[libs.length * libs.length];
190                 for (int i = 0; i < libs.length; i++) {
191                     result[count] = new Ex(
192                             count,
193                             req(0, 0),
194                             pa[i * numPerType],
195                             true);
196                     count++;
197                     for (int j = 0; j < libs.length; j++) {
198                         if (i == j) {
199                             continue;
200                         }
201                         result[count] = new Ex(
202                                 count,
203                                 i,
204                                 pa[j * numPerType],
205                                 false);
206                         count++;
207                     }
208                 }
209             } else {
210                 String[] runs = userRuns.split(",");
211                 result = new Ex[runs.length];
212                 for (int i = 0; i < runs.length; i++) {
213                     UserRun run = new UserRun(runs[i]);
214                     result[i] = new Ex(
215                             i,
216                             run.req() == -1 ?
217                                     req(run.client(), run.service()) :
218                                     result[run.req()].req,
219                             Arrays.stream(pa)
220                                     .filter(p -> p.debug().equals(run.acceptor()))
221                                     .findFirst()
222                                     .orElseThrow(() -> new Exception(
223                                             "no acceptor named " + run.acceptor())),
224                             run.success());
225                 }
226             }
227 
228             for (Ex x : result) {
229                 x.run();
230             }
231 
232             pi.println("END");
233             for (int i=0; i<na; i++) {
234                 pa[i].println("END");
235             }
236             System.out.println("\nAll Test Results\n================");
237             boolean finalOut = true;
238             System.out.println("        req**  client    service  acceptor   Result");
239             System.out.println("----  -------  ------  ---------  --------  -------");
240             for (int i=0; i<result.length; i++) {
241                 boolean out = result[i].expected==result[i].actual;
242                 finalOut &= out;
243                 System.out.printf("%3d:    %3d%s      c%d    s%d %4s  %8s   %s  %s\n",
244                         i,
245                         result[i].req,
246                         result[i].expected ? "**" : "  ",
247                         reqs.get(result[i].req).client,
248                         reqs.get(result[i].req).service,
249                         "(" + result[i].csize + ")",
250                         result[i].acceptor.debug(),
251                         result[i].actual ? "++" : "--",
252                         out ? "   " : "xxx");
253             }
254 
255             System.out.println("\nPath of Reqs\n============");
256             for (int j=0; ; j++) {
257                 boolean found = false;
258                 for (int i=0; i<result.length; i++) {
259                     if (result[i].req == j) {
260                         if (!found) {
261                             System.out.printf("%3d (c%s -> s%s): ", j,
262                                     reqs.get(j).client, reqs.get(j).service);
263                         }
264                         System.out.printf("%s%s(%d)%s",
265                                 found ? " -> " : "",
266                                 result[i].acceptor.debug(),
267                                 i,
268                                 result[i].actual != result[i].expected ?
269                                         "xxx" : "");
270                         found = true;
271                     }
272                 }
273                 System.out.println();
274                 if (!found) {
275                     break;
276                 }
277             }
278             if (!finalOut) throw new Exception();
279         } else if (args[0].equals("Nsanity")) {
280             // Native mode sanity check
281             Proc.d("Detect start");
282             Context s = Context.fromUserKtab("*", OneKDC.KTAB, true);
283             s.startAsServer(GSSUtil.GSS_KRB5_MECH_OID);
284         } else if (args[0].equals("initiator")) {
285             while (true) {
286                 String title = Proc.textIn();
287                 Proc.d("Client see " + title);
288                 if (title.equals("END")) break;
289                 String[] cas = title.split(" ");
290                 Context c = Context.fromUserPass(cas[0], OneKDC.PASS, false);
291                 c.startAsClient(cas[1], GSSUtil.GSS_KRB5_MECH_OID);
292                 c.x().requestCredDeleg(true);
293                 byte[] token = c.take(new byte[0]);
294                 Proc.d("Client AP-REQ generated");
295                 Proc.binOut(token);
296             }
297         } else {
298             Proc.d(System.getProperty("java.vm.version"));
299             Proc.d(System.getProperty("sun.security.jgss.native"));
300             Proc.d(System.getProperty("sun.security.jgss.lib"));
301             Proc.d("---------------------------------\n");
302             Proc.d("Server start");
303             Context s = Context.fromUserKtab("*", OneKDC.KTAB, true);
304             Proc.d("Server login");
305             while (true) {
306                 String title = Proc.textIn();
307                 Proc.d("Server sees " + title);
308                 if (title.equals("END")) break;
309                 s.startAsServer(GSSUtil.GSS_KRB5_MECH_OID);
310                 byte[] token = Proc.binIn();
311                 try {
312                     s.take(token);
313                     Proc.textOut("true");
314                     Proc.d("Good");
315                 } catch (Exception e) {
316                     Proc.textOut("false");
317                     Proc.d("Bad");
318                 }
319             }
320         }
321     }

347     // generates an ap-req and save into reqs, returns the index
348     private static int req(int client, int service) throws Exception {
349         pi.println(client(client) + " " + service(service));
350         Req req = new Req(client, service, pi.readData());
351         reqs.add(req);
352         return reqs.size() - 1;
353     }

355     // create a acceptor
356     private static Proc acceptor(String type, String suffix) throws Exception {
357         Proc p;
358         String label;
359         String lib;
360         int pos = type.indexOf('=');
361         if (pos < 0) {
362             label = type;
363             lib = null;
364         } else {
365             label = type.substring(0, pos);
366             lib = type.substring(pos + 1);
367         }
368         if (type.startsWith("J")) {
369             if (lib == null) {
370                 p = Proc.create("ReplayCacheTestProc");
371             } else {
372                 p = Proc.create("ReplayCacheTestProc", lib);
373             }
374             p.prop("sun.security.krb5.rcache", "dfl")
375                     .prop("java.io.tmpdir", cwd);
376             String useMD5 = System.getProperty("jdk.krb5.rcache.useMD5");
377             if (useMD5 != null) {
378                 p.prop("jdk.krb5.rcache.useMD5", useMD5);
379             }
380         } else {
381             p = Proc.create("ReplayCacheTestProc")
382                     .env("KRB5_CONFIG", OneKDC.KRB5_CONF)
383                     .env("KRB5_KTNAME", OneKDC.KTAB)
384                     .env("KRB5RCACHEDIR", cwd)
385                     .prop("sun.security.jgss.native", "true")
386                     .prop("javax.security.auth.useSubjectCredsOnly", "false")
387                     .prop("sun.security.nativegss.debug", "true");
388             if (lib != null) {
389                 String libDir = lib.substring(0, lib.lastIndexOf('/'));
390                 p.prop("sun.security.jgss.lib", lib)
391                         .env("DYLD_LIBRARY_PATH", libDir)
392                         .env("LD_LIBRARY_PATH", libDir);
393             }
394         }
395         Proc.d(label+suffix+" started");
396         return p.args(label+suffix).debug(label+suffix).start();
397     }

399     // generates hash of authenticator inside ap-req inside initsectoken
400     private static void record(String label, Req req) throws Exception {
401         byte[] data = Base64.getDecoder().decode(req.msg);
402         data = Arrays.copyOfRange(data, 17, data.length);
403 
404         try (PrintStream ps = new PrintStream(
405                 new FileOutputStream("log.txt", true))) {
406             ps.printf("%s:\nmsg: %s\nMD5: %s\nSHA-256: %s\n\n",
407                     label,
408                     req.msg,
409                     hex(md5.digest(data)),
410                     hex(sha256.digest(data)));
411         }
412     }

414     // Returns a compact hexdump for a byte array
415     private static String hex(byte[] hash) {
416         char[] h = new char[hash.length * 2];
417         char[] hexConst = "0123456789ABCDEF".toCharArray();
418         for (int i=0; i<hash.length; i++) {
419             h[2*i] = hexConst[(hash[i]&0xff)>>4];
420             h[2*i+1] = hexConst[hash[i]&0xf];
421         }
422         return new String(h);
423     }

425     // return size of dfl file, excluding the null hash ones
426     private static int csize(int p) throws Exception {
427         try (SeekableByteChannel chan = Files.newByteChannel(
428                 Paths.get(cwd, dfl(p)), StandardOpenOption.READ)) {
429             chan.position(6);
430             int cc = 0;
431             while (true) {
432                 try {
433                     if (AuthTime.readFrom(chan) != null) cc++;
434                 } catch (BufferUnderflowException e) {
435                     break;
436                 }
437             }
438             return cc;
439         } catch (IOException ioe) {
440             return 0;
441         }
442     }

444     // models an experiement
445     private static class Ex {
446         int i;              // #
447         int req;            // which ap-req to send
448         Proc acceptor;      // which acceptor to send to
449         boolean expected;   // expected result
450 
451         boolean actual;     // actual output
452         int csize;          // size of rcache after test
453         String hash;        // the hash of req
454 
455         Ex(int i, int req, Proc acceptor, boolean expected) {
456             this.i = i;
457             this.req = req;
458             this.acceptor = acceptor;
459             this.expected = expected;
460         }
461 
462         void run() throws Exception {
463             Req r = reqs.get(req);
464             acceptor.println("TEST");
465             acceptor.println(r.msg);
466             String reply = acceptor.readData();
467 
468             actual = Boolean.valueOf(reply);
469             csize = csize(r.service);
470 
471             String label = String.format("%03d-client%d-%s%d-%s-%s",
472                     i, r.client, SERVICE, r.service, acceptor.debug(), actual);
473 
474             record(label, r);
475             if (new File(cwd, dfl(r.service)).exists()) {
476                 Files.copy(Paths.get(cwd, dfl(r.service)), Paths.get(label),
477                         StandardCopyOption.COPY_ATTRIBUTES);
478             }
479         }
480     }

482     // models a saved ap-req msg
483     private static class Req {
484         String msg;         // based64-ed req
485         int client;         // which client
486         int service;        // which service
487         Req(int client, int service, String msg) {
488             this.msg = msg;
489             this.client= client;
490             this.service = service;
491         }

494     private static class UserRun {
495         static final Pattern p
496                 = Pattern.compile("(c(\\d)+s(\\d+)|r(\\d+))(.*)(.)");
497         final Matcher m;
498 
499         UserRun(String run) { m = p.matcher(run); m.find(); }
500 
501         int req() { return group(4); }
502         int client() { return group(2); }
503         int service() { return group(3); }
504         String acceptor() { return m.group(5); }
505         boolean success() { return m.group(6).equals("v"); }
506 
507         int group(int i) {
508             String g = m.group(i);
509             return g == null ? -1 : Integer.parseInt(g);
510         }
