161     /* Closes any stream already retrieved for the data.
162      * We want to avoid unnecessarily asking the Doc to create a stream only
163      * to get a reference in order to close it because the job failed.
164      * If the representation class is itself a "stream", this
165      * closes that stream too.
166      */
167     private void closeDataStreams() {
168 
169         if (doc == null) {
170             return;
171         }
172 
173         Object data = null;
174 
175         try {
176             data = doc.getPrintData();
177         } catch (IOException e) {
178             return;
179         }
180 
181         if (instream != null) {
182             try {
183                 instream.close();
184             } catch (IOException e) {
185             } finally {
186                 instream = null;
187             }
188         }
189         else if (reader != null) {
190             try {
191                 reader.close();
192             } catch (IOException e) {
193             } finally {
194                 reader = null;
195             }
196         }
197         else if (data instanceof InputStream) {
198             try {
199                 ((InputStream)data).close();
200             } catch (IOException e) {
201             }
202         }
203         else if (data instanceof Reader) {
204             try {
205                 ((Reader)data).close();
206             } catch (IOException e) {
207             }
208         }
209     }

211     private void notifyEvent(int reason) {
212 
213         /* since this method should always get called, here's where
214          * we will perform the clean up of any data stream supplied.
215          */
216         switch (reason) {
217             case PrintJobEvent.DATA_TRANSFER_COMPLETE:
218             case PrintJobEvent.JOB_CANCELED :
219             case PrintJobEvent.JOB_FAILED :
220             case PrintJobEvent.NO_MORE_EVENTS :
221             case PrintJobEvent.JOB_COMPLETE :
222                 closeDataStreams();
223         }
224 
225         synchronized (this) {
226             if (jobListeners != null) {
227                 PrintJobListener listener;
228                 PrintJobEvent event = new PrintJobEvent(this, reason);
229                 for (int i = 0; i < jobListeners.size(); i++) {
230                     listener = (PrintJobListener)(jobListeners.elementAt(i));
231                     switch (reason) {
232 
233                         case PrintJobEvent.JOB_COMPLETE :
234                             listener.printJobCompleted(event);
235                             break;
236 
237                         case PrintJobEvent.JOB_CANCELED :
238                             listener.printJobCanceled(event);
239                             break;
240 
241                         case PrintJobEvent.JOB_FAILED :
242                             listener.printJobFailed(event);
243                             break;
244 
245                         case PrintJobEvent.DATA_TRANSFER_COMPLETE :
246                             listener.printDataTransferCompleted(event);
247                             break;
248 
249                         case PrintJobEvent.NO_MORE_EVENTS :
250                             listener.printJobNoMoreEvents(event);
251                             break;
252 
253                         default:
254                             break;
255                     }
256                 }
257             }
258        }
259     }

280     public void removePrintJobAttributeListener(
281                                         PrintJobAttributeListener listener) {
282         synchronized (this) {
283             if (listener == null || attrListeners == null ) {
284                 return;
285             }
286             int index = attrListeners.indexOf(listener);
287             if (index == -1) {
288                 return;
289             } else {
290                 attrListeners.remove(index);
291                 listenedAttributeSets.remove(index);
292                 if (attrListeners.isEmpty()) {
293                     attrListeners = null;
294                     listenedAttributeSets = null;
295                 }
296             }
297         }

300     public void print(Doc doc, PrintRequestAttributeSet attributes)
301         throws PrintException {
302 
303         synchronized (this) {
304             if (printing) {
305                 throw new PrintException("already printing");
306             } else {
307                 printing = true;
308             }
309         }
310 
311         PrinterState prnState = (PrinterState)service.getAttribute(
312                                                   PrinterState.class);
313         if (prnState == PrinterState.STOPPED) {
314             PrinterStateReasons prnStateReasons =
315                     (PrinterStateReasons)service.getAttribute(
316                                                  PrinterStateReasons.class);
317                 if ((prnStateReasons != null) &&
318                     (prnStateReasons.containsKey(PrinterStateReason.SHUTDOWN)))
319                 {
320                     throw new PrintException("PrintService is no longer available.");
321                 }
322         }
323 
324         if ((PrinterIsAcceptingJobs)(service.getAttribute(
325                          PrinterIsAcceptingJobs.class)) ==
326                          PrinterIsAcceptingJobs.NOT_ACCEPTING_JOBS) {
327             throw new PrintException("Printer is not accepting job.");
328         }
329 
330 
331         this.doc = doc;
332         /* check if the parameters are valid before doing much processing */
333         DocFlavor flavor = doc.getDocFlavor();
334         Object data;
335 
336         try {
337             data = doc.getPrintData();
338         } catch (IOException e) {
339             notifyEvent(PrintJobEvent.JOB_FAILED);
340             throw new PrintException("can't get print data: " + e.toString());
341         }
342 
343         if (flavor == null || (!service.isDocFlavorSupported(flavor))) {
344             notifyEvent(PrintJobEvent.JOB_FAILED);
345             throw new PrintJobFlavorException("invalid flavor", flavor);
346         }
347 
348         initializeAttributeSets(doc, attributes);
349 
350         getAttributeValues(flavor);
351 
352         String repClassName = flavor.getRepresentationClassName();
353 
354         if (flavor.equals(DocFlavor.INPUT_STREAM.GIF) ||
355             flavor.equals(DocFlavor.INPUT_STREAM.JPEG) ||
356             flavor.equals(DocFlavor.INPUT_STREAM.PNG) ||
357             flavor.equals(DocFlavor.BYTE_ARRAY.GIF) ||
358             flavor.equals(DocFlavor.BYTE_ARRAY.JPEG) ||
359             flavor.equals(DocFlavor.BYTE_ARRAY.PNG)) {
360             try {
361                 instream = doc.getStreamForBytes();
362                 if (instream == null) {
363                     notifyEvent(PrintJobEvent.JOB_FAILED);
364                     throw new PrintException("No stream for data");
365                 }
366                 printableJob(new ImagePrinter(instream));
367                 service.wakeNotifier();
368                 return;
369             } catch (ClassCastException cce) {
370                 notifyEvent(PrintJobEvent.JOB_FAILED);
371                 throw new PrintException(cce);
372             } catch (IOException ioe) {
373                 notifyEvent(PrintJobEvent.JOB_FAILED);
374                 throw new PrintException(ioe);
375             }
376         } else if (flavor.equals(DocFlavor.URL.GIF) ||
377                    flavor.equals(DocFlavor.URL.JPEG) ||
378                    flavor.equals(DocFlavor.URL.PNG)) {
379             try {
380                 printableJob(new ImagePrinter((URL)data));
381                 service.wakeNotifier();
382                 return;
383             } catch (ClassCastException cce) {
384                 notifyEvent(PrintJobEvent.JOB_FAILED);
385                 throw new PrintException(cce);
386             }
387         } else if (repClassName.equals("java.awt.print.Pageable")) {
388             try {
389                 pageableJob((Pageable)doc.getPrintData());
390                 service.wakeNotifier();
391                 return;
392             } catch (ClassCastException cce) {
393                 notifyEvent(PrintJobEvent.JOB_FAILED);
394                 throw new PrintException(cce);
395             } catch (IOException ioe) {
396                 notifyEvent(PrintJobEvent.JOB_FAILED);
397                 throw new PrintException(ioe);
398             }
399         } else if (repClassName.equals("java.awt.print.Printable")) {
400             try {
401                 printableJob((Printable)doc.getPrintData());
402                 service.wakeNotifier();
403                 return;
404             } catch (ClassCastException cce) {
405                 notifyEvent(PrintJobEvent.JOB_FAILED);
406                 throw new PrintException(cce);
407             } catch (IOException ioe) {
408                 notifyEvent(PrintJobEvent.JOB_FAILED);
409                 throw new PrintException(ioe);
410             }
411         } else if (repClassName.equals("[B") ||
412                    repClassName.equals("java.io.InputStream") ||
413                    repClassName.equals("java.net.URL")) {
414 
415             if (repClassName.equals("java.net.URL")) {
416                 URL url = (URL)data;
417                 try {
418                     instream = url.openStream();
419                 } catch (IOException e) {
420                     notifyEvent(PrintJobEvent.JOB_FAILED);
421                     throw new PrintException(e.toString());
422                 }
423             } else {
424                 try {
425                     instream = doc.getStreamForBytes();
426                 } catch (IOException ioe) {
427                     notifyEvent(PrintJobEvent.JOB_FAILED);
428                     throw new PrintException(ioe.toString());
429                 }
430             }
431 
432             if (instream == null) {
433                 notifyEvent(PrintJobEvent.JOB_FAILED);
434                 throw new PrintException("No stream for data");
435             }
436 
437             if (mDestination != null) { // if destination attribute is set
438                 try {
439                     FileOutputStream fos = new FileOutputStream(mDestination);
440                     byte []buffer = new byte[1024];
441                     int cread;
442 
443                     while ((cread = instream.read(buffer, 0, buffer.length)) >=0) {
444                         fos.write(buffer, 0, cread);
445                     }
446                     fos.flush();
447                     fos.close();
448                 } catch (FileNotFoundException fnfe) {
449                     notifyEvent(PrintJobEvent.JOB_FAILED);
450                     throw new PrintException(fnfe.toString());
451                 } catch (IOException ioe) {
452                     notifyEvent(PrintJobEvent.JOB_FAILED);
453                     throw new PrintException(ioe.toString());
454                 }
455                 notifyEvent(PrintJobEvent.DATA_TRANSFER_COMPLETE);
456                 notifyEvent(PrintJobEvent.JOB_COMPLETE);
457                 service.wakeNotifier();
458                 return;
459             }
460 
461             if (!startPrintRawData(service.getName(), jobName)) {
462                 notifyEvent(PrintJobEvent.JOB_FAILED);
463                 throw new PrintException("Print job failed to start.");
464             }
465             BufferedInputStream  bin = new BufferedInputStream(instream);
466             int bread = 0;
467             try {
468                 byte[] buffer = new byte[PRINTBUFFERLEN];
469 
470                 while ((bread = bin.read(buffer, 0, PRINTBUFFERLEN)) >=0) {
471                     if (!printRawData(buffer, bread)) {
472                         bin.close();
473                         notifyEvent(PrintJobEvent.JOB_FAILED);
474                         throw new PrintException ("Problem while spooling data");
475                     }
476                 }
477                 bin.close();
478                 if (!endPrintRawData()) {
479                     notifyEvent(PrintJobEvent.JOB_FAILED);
480                     throw new PrintException("Print job failed to close properly.");
481                 }
482                 notifyEvent(PrintJobEvent.DATA_TRANSFER_COMPLETE);
483             } catch (IOException e) {
484                 notifyEvent(PrintJobEvent.JOB_FAILED);
485                 throw new PrintException (e.toString());
486             } finally {
487                 notifyEvent(PrintJobEvent.NO_MORE_EVENTS);
488             }
489         } else {
490             notifyEvent(PrintJobEvent.JOB_FAILED);
491             throw new PrintException("unrecognized class: "+repClassName);
492         }
493         service.wakeNotifier();
494     }

496     public void printableJob(Printable printable) throws PrintException {
497         try {
498             synchronized(this) {
499                 if (job != null) { // shouldn't happen
500                     throw new PrintException("already printing");
501                 } else {
502                     job = new sun.awt.windows.WPrinterJob();
503                 }
504             }
505             PrintService svc = getPrintService();
506             job.setPrintService(svc);
507             if (copies == 0) {
508                 Copies c = (Copies)svc.getDefaultAttributeValue(Copies.class);
509                 copies = c.getValue();
510             }
511 
512             if (mediaName == null) {
513                 Object media = svc.getDefaultAttributeValue(Media.class);
514                 if (media instanceof MediaSizeName) {
515                     mediaName = (MediaSizeName) media;
516                     mediaSize = MediaSize.getMediaSizeForName(mediaName);
517                 }
518             }
519 
520             if (orient == null) {
521                 orient =
522                     (OrientationRequested)svc.getDefaultAttributeValue(OrientationRequested.class);
523             }
524 
525             job.setCopies(copies);
526             job.setJobName(jobName);
527             PageFormat pf = new PageFormat();
528             if (mediaSize != null) {
529                 Paper p = new Paper();
530                 p.setSize(mediaSize.getX(MediaSize.INCH)*72.0,
531                           mediaSize.getY(MediaSize.INCH)*72.0);
532                 p.setImageableArea(72.0, 72.0, p.getWidth()-144.0,
533                                    p.getHeight()-144.0);
534                 pf.setPaper(p);
535             }
536             if (orient == OrientationRequested.REVERSE_LANDSCAPE) {
537                 pf.setOrientation(PageFormat.REVERSE_LANDSCAPE);
538             } else if (orient == OrientationRequested.LANDSCAPE) {
539                 pf.setOrientation(PageFormat.LANDSCAPE);
540             }
541             job.setPrintable(printable, pf);
542             job.print(reqAttrSet);
543             notifyEvent(PrintJobEvent.DATA_TRANSFER_COMPLETE);
544             return;
545         } catch (PrinterException pe) {
546             notifyEvent(PrintJobEvent.JOB_FAILED);
547             throw new PrintException(pe);
548         } finally {
549             printReturned = true;
550             notifyEvent(PrintJobEvent.NO_MORE_EVENTS);
551         }
552     }

554     public void pageableJob(Pageable pageable) throws PrintException {
555         try {
556             synchronized(this) {
557                 if (job != null) { // shouldn't happen

566                 Copies c = (Copies)svc.getDefaultAttributeValue(Copies.class);

576             notifyEvent(PrintJobEvent.JOB_FAILED);
577             throw new PrintException(pe);
578         } finally {
579             printReturned = true;
580             notifyEvent(PrintJobEvent.NO_MORE_EVENTS);
581         }
582     }

584     /* There's some inefficiency here as the job set is created even though
585      * it may never be requested.
586      */
587     private synchronized void
588         initializeAttributeSets(Doc doc, PrintRequestAttributeSet reqSet) {
589 
590         reqAttrSet = new HashPrintRequestAttributeSet();
591         jobAttrSet = new HashPrintJobAttributeSet();
592 
593         Attribute[] attrs;
594         if (reqSet != null) {
595             reqAttrSet.addAll(reqSet);
596             attrs = reqSet.toArray();
597             for (int i=0; i<attrs.length; i++) {
598                 if (attrs[i] instanceof PrintJobAttribute) {
599                     jobAttrSet.add(attrs[i]);
600                 }
601             }
602         }
603 
604         DocAttributeSet docSet = doc.getAttributes();
605         if (docSet != null) {
606             attrs = docSet.toArray();
607             for (int i=0; i<attrs.length; i++) {
608                 if (attrs[i] instanceof PrintRequestAttribute) {
609                     reqAttrSet.add(attrs[i]);
610                 }
611                 if (attrs[i] instanceof PrintJobAttribute) {
612                     jobAttrSet.add(attrs[i]);
613                 }
614             }
615         }
616 
617         /* add the user name to the job */
618         String userName = "";
619         try {
620           userName = System.getProperty("user.name");
621         } catch (SecurityException se) {
622         }
623 
624         if (userName == null || userName.equals("")) {
625             RequestingUserName ruName =
626                 (RequestingUserName)reqSet.get(RequestingUserName.class);
627             if (ruName != null) {
628                 jobAttrSet.add(
629                     new JobOriginatingUserName(ruName.getValue(),
630                                                ruName.getLocale()));
631             } else {
632                 jobAttrSet.add(new JobOriginatingUserName("", null));
633             }
634         } else {
635             jobAttrSet.add(new JobOriginatingUserName(userName, null));
636         }
637 
638         /* if no job name supplied use doc name (if supplied), if none and
639          * its a URL use that, else finally anything .. */
640         if (jobAttrSet.get(JobName.class) == null) {
641             JobName jobName;
642             if (docSet != null && docSet.get(DocumentName.class) != null) {
643                 DocumentName docName =
644                     (DocumentName)docSet.get(DocumentName.class);
645                 jobName = new JobName(docName.getValue(), docName.getLocale());
646                 jobAttrSet.add(jobName);
647             } else {
648                 String str = "JPS Job:" + doc;
649                 try {
650                     Object printData = doc.getPrintData();
651                     if (printData instanceof URL) {
652                         str = ((URL)(doc.getPrintData())).toString();
653                     }
654                 } catch (IOException e) {
655                 }
656                 jobName = new JobName(str, null);
657                 jobAttrSet.add(jobName);
658             }
659         }
660 
661         jobAttrSet = AttributeSetUtilities.unmodifiableView(jobAttrSet);
662     }

664     private void getAttributeValues(DocFlavor flavor) throws PrintException {
665 
666         if (reqAttrSet.get(Fidelity.class) == Fidelity.FIDELITY_TRUE) {
667             fidelity = true;
668         } else {
669             fidelity = false;
670         }
671 
672         Class category;
673         Attribute [] attrs = reqAttrSet.toArray();
674         for (int i=0; i<attrs.length; i++) {
675             Attribute attr = attrs[i];
676             category = attr.getCategory();
677             if (fidelity == true) {
678                 if (!service.isAttributeCategorySupported(category)) {
679                     notifyEvent(PrintJobEvent.JOB_FAILED);
680                     throw new PrintJobAttributeException(
681                         "unsupported category: " + category, category, null);
682                 } else if
683                     (!service.isAttributeValueSupported(attr, flavor, null)) {
684                     notifyEvent(PrintJobEvent.JOB_FAILED);
685                     throw new PrintJobAttributeException(
686                         "unsupported attribute: " + attr, null, attr);
687                 }
688             }
689             if (category == Destination.class) {
690               URI uri = ((Destination)attr).getURI();
691               if (!"file".equals(uri.getScheme())) {
692                 notifyEvent(PrintJobEvent.JOB_FAILED);
693                 throw new PrintException("Not a file: URI");
694               } else {
695                 try {
696                   mDestination = (new File(uri)).getPath();
697                 } catch (Exception e) {
698                   throw new PrintException(e);
699                 }
700                 // check write access
701                 SecurityManager security = System.getSecurityManager();
702                 if (security != null) {
703                   try {
704                     security.checkWrite(mDestination);
705                   } catch (SecurityException se) {
706                     notifyEvent(PrintJobEvent.JOB_FAILED);
707                     throw new PrintException(se);
708                   }
709                 }
710               }
711             } else if (category == JobName.class) {
712                 jobName = ((JobName)attr).getValue();
713             } else if (category == Copies.class) {
714                 copies = ((Copies)attr).getValue();
715             } else if (category == Media.class) {
716               if (attr instanceof MediaSizeName) {
717                     mediaName = (MediaSizeName)attr;
718                     // If requested MediaSizeName is not supported,
719                     // get the corresponding media size - this will
720                     // be used to create a new PageFormat.
721                     if (!service.isAttributeValueSupported(attr, null, null)) {
722                         mediaSize = MediaSize.getMediaSizeForName(mediaName);
723                     }
724                 }
725             } else if (category == OrientationRequested.class) {
726                 orient = (OrientationRequested)attr;
727             }
728         }

736     /* Cancel PrinterJob jobs that haven't yet completed. */
737    public void cancel() throws PrintException {
738         synchronized (this) {
739             if (!printing) {
740                 throw new PrintException("Job is not yet submitted.");
741             } else if (job != null && !printReturned) {
742                 job.cancel();
743                 notifyEvent(PrintJobEvent.JOB_CANCELED);
744                 return;
745             } else {
746                 throw new PrintException("Job could not be cancelled.");
747             }
748         }
