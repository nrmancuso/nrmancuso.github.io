61  /**
62   * This is Kerberos option in the client key exchange message
63   * (CLIENT -> SERVER). It holds the Kerberos ticket and the encrypted
64   * premaster secret encrypted with the session key sealed in the ticket.
65   * From RFC 2712:
66   *  struct
67   *  {
68   *    opaque Ticket;
69   *    opaque authenticator;            // optional
70   *    opaque EncryptedPreMasterSecret; // encrypted with the session key
71   *                                     // which is sealed in the ticket
72   *  } KerberosWrapper;
73   *
74   *
75   * Ticket and authenticator are encrypted as per RFC 1510 (in ASN.1)
76   * Encrypted pre-master secret has the same structure as it does for RSA
77   * except for Kerberos, the encryption key is the session key instead of
78   * the RSA public key.
79   *
80   * XXX authenticator currently ignored
81   *
82   */
83  public final class KerberosClientKeyExchangeImpl
84      extends sun.security.ssl.KerberosClientKeyExchange {
85  
86      private KerberosPreMasterSecret preMaster;
87      private byte[] encodedTicket;
88      private KerberosPrincipal peerPrincipal;
89      private KerberosPrincipal localPrincipal;
90  
91      public KerberosClientKeyExchangeImpl() {
92      }
93  
94      /**
95       * Creates an instance of KerberosClientKeyExchange consisting of the
96       * Kerberos service ticket, authenticator and encrypted premaster secret.
97       * Called by client handshaker.
98       *
99       * @param serverName name of server with which to do handshake;
100      *             this is used to get the Kerberos service ticket
101      * @param protocolVersion Maximum version supported by client (i.e,
102      *          version it requested in client hello)
103      * @param rand random number generator to use for generating pre-master
104      *          secret
105      */
106     @Override
107     public void init(String serverName, boolean isLoopback,
108         AccessControlContext acc, ProtocolVersion protocolVersion,
109         SecureRandom rand) throws IOException {
110 
111          // Get service ticket
112          KerberosTicket ticket = getServiceTicket(serverName, isLoopback, acc);
113          encodedTicket = ticket.getEncoded();
114 
115          // Record the Kerberos principals
116          peerPrincipal = ticket.getServer();
117          localPrincipal = ticket.getClient();
118 
119          // Optional authenticator, encrypted using session key,
120          // currently ignored
121 
122          // Generate premaster secret and encrypt it using session key
123          EncryptionKey sessionKey = new EncryptionKey(
124                                         ticket.getSessionKeyType(),
125                                         ticket.getSessionKey().getEncoded());
126 
127          preMaster = new KerberosPreMasterSecret(protocolVersion,
128              rand, sessionKey);
129     }
130 
131     /**
132      * Creates an instance of KerberosClientKeyExchange from its ASN.1 encoding.
133      * Used by ServerHandshaker to verify and obtain premaster secret.
134      *
135      * @param protocolVersion current protocol version
136      * @param clientVersion version requested by client in its ClientHello;
137      *          used by premaster secret version check
138      * @param rand random number generator used for generating random
139      *          premaster secret if ticket and/or premaster verification fails
140      * @param input inputstream from which to get ASN.1-encoded KerberosWrapper
141      * @param serverKey server's master secret key
142      */
143     @Override
144     public void init(ProtocolVersion protocolVersion,
145         ProtocolVersion clientVersion,
146         SecureRandom rand, HandshakeInStream input, SecretKey[] secretKeys)
147         throws IOException {
148 
149         KerberosKey[] serverKeys = (KerberosKey[])secretKeys;
150 
151         // Read ticket
152         encodedTicket = input.getBytes16();
153 
154         if (debug != null && Debug.isOn("verbose")) {
155             Debug.println(System.out,
156                 "encoded Kerberos service ticket", encodedTicket);
157         }
158 
159         EncryptionKey sessionKey = null;
160 
161         try {
162             Ticket t = new Ticket(encodedTicket);
163 
164             EncryptedData encPart = t.encPart;
165             PrincipalName ticketSname = t.sname;
166             Realm ticketRealm = t.realm;
167 
168             String serverPrincipal = serverKeys[0].getPrincipal().getName();
169 
170             /*
171              * permission to access and use the secret key of the Kerberized
172              * "host" service is done in ServerHandshaker.getKerberosKeys()
173              * to ensure server has the permission to use the secret key
174              * before promising the client
175              */
176 
177             // Check that ticket Sname matches serverPrincipal
178             String ticketPrinc = ticketSname.toString().concat("@" +
179                                         ticketRealm.toString());
180             if (!ticketPrinc.equals(serverPrincipal)) {
181                 if (debug != null && Debug.isOn("handshake"))
182                    System.out.println("Service principal in Ticket does not"
183                         + " match associated principal in KerberosKey");
184                 throw new IOException("Server principal is " +
185                     serverPrincipal + " but ticket is for " +
186                     ticketPrinc);
187             }
188 
189             // See if we have the right key to decrypt the ticket to get
190             // the session key.
191             int encPartKeyType = encPart.getEType();
192             Integer encPartKeyVersion = encPart.getKeyVersionNumber();
193             KerberosKey dkey = null;
194             try {
195                 dkey = findKey(encPartKeyType, encPartKeyVersion, serverKeys);
196             } catch (KrbException ke) { // a kvno mismatch
197                 throw new IOException(
198                         "Cannot find key matching version number", ke);
199             }
200             if (dkey == null) {
201                 // %%% Should print string repr of etype
202                 throw new IOException(
203         "Cannot find key of appropriate type to decrypt ticket - need etype " +
204                                    encPartKeyType);
205             }
206 
207             EncryptionKey secretKey = new EncryptionKey(
208                 encPartKeyType,
209                 dkey.getEncoded());
210 
211             // Decrypt encPart using server's secret key
212             byte[] bytes = encPart.decrypt(secretKey, KeyUsage.KU_TICKET);
213 
214             // Reset data stream after decryption, remove redundant bytes
215             byte[] temp = encPart.reset(bytes);
216             EncTicketPart encTicketPart = new EncTicketPart(temp);
217 
218             // Record the Kerberos Principals
219             peerPrincipal =
220                 new KerberosPrincipal(encTicketPart.cname.getName());
221             localPrincipal = new KerberosPrincipal(ticketSname.getName());
222 
223             sessionKey = encTicketPart.key;
224 
225             if (debug != null && Debug.isOn("handshake")) {
226                 System.out.println("server principal: " + serverPrincipal);
227                 System.out.println("realm: " + encTicketPart.crealm.toString());
228                 System.out.println("cname: " + encTicketPart.cname.toString());
229             }
230         } catch (IOException e) {
231             throw e;
232         } catch (Exception e) {
233             if (debug != null && Debug.isOn("handshake")) {
234                 System.out.println("KerberosWrapper error getting session key,"
235                         + " generating random secret (" + e.getMessage() + ")");
236             }
237             sessionKey = null;
238         }
239 
240         input.getBytes16();   // XXX Read and ignore authenticator
241 
242         if (sessionKey != null) {
243             preMaster = new KerberosPreMasterSecret(protocolVersion,
244                 clientVersion, rand, input, sessionKey);
245         } else {
246             // Generate bogus premaster secret
247             preMaster = new KerberosPreMasterSecret(clientVersion, rand);
248         }
249     }
250 
251     @Override
252     public int messageLength() {
253         return (6 + encodedTicket.length + preMaster.getEncrypted().length);
254     }
255 
256     @Override
257     public void send(HandshakeOutStream s) throws IOException {
258         s.putBytes16(encodedTicket);
259         s.putBytes16(null); // XXX no authenticator
260         s.putBytes16(preMaster.getEncrypted());
261     }
262 
263     @Override
264     public void print(PrintStream s) throws IOException {
265         s.println("*** ClientKeyExchange, Kerberos");
266 
267         if (debug != null && Debug.isOn("verbose")) {
268             Debug.println(s, "Kerberos service ticket", encodedTicket);
269             Debug.println(s, "Random Secret", preMaster.getUnencrypted());
270             Debug.println(s, "Encrypted random Secret",
271                 preMaster.getEncrypted());
272         }
273     }

275     // Similar to sun.security.jgss.krb5.Krb5InitCredenetial/Krb5Context
276     private static KerberosTicket getServiceTicket(String srvName,
277         boolean isLoopback, final AccessControlContext acc) throws IOException {
278 
279         // get the local hostname if srvName is loopback address
280         String serverName = srvName;
281         if (isLoopback) {
282             String localHost = java.security.AccessController.doPrivileged(
283                 new java.security.PrivilegedAction<String>() {
284                 public String run() {
285                     String hostname;
286                     try {
287                         hostname = InetAddress.getLocalHost().getHostName();
288                     } catch (java.net.UnknownHostException e) {
289                         hostname = "localhost";
290                     }
291                     return hostname;
292                 }
293             });
294           serverName = localHost;
295         }
296 
297         // Resolve serverName (possibly in IP addr form) to Kerberos principal
298         // name for service with hostname
299         String serviceName = "host/" + serverName;
300         PrincipalName principal;
301         try {
302             principal = new PrincipalName(serviceName,
303                                 PrincipalName.KRB_NT_SRV_HST);
304         } catch (SecurityException se) {
305             throw se;
306         } catch (Exception e) {
307             IOException ioe = new IOException("Invalid service principal" +
308                                 " name: " + serviceName);
309             ioe.initCause(e);
310             throw ioe;
311         }
312         String realm = principal.getRealmAsString();
313 
314         final String serverPrincipal = principal.toString();
315         final String tgsPrincipal = "krbtgt/" + realm + "@" + realm;
316         final String clientPrincipal = null;  // use default
317 
318 
319         // check permission to obtain a service ticket to initiate a
320         // context with the "host" service
321         SecurityManager sm = System.getSecurityManager();
322         if (sm != null) {
323            sm.checkPermission(new ServicePermission(serverPrincipal,
324                                 "initiate"), acc);
325         }
326 
327         try {
328             KerberosTicket ticket = AccessController.doPrivileged(
329                 new PrivilegedExceptionAction<KerberosTicket>() {
330                 public KerberosTicket run() throws Exception {
331                     return Krb5Util.getTicketFromSubjectAndTgs(
332                         GSSCaller.CALLER_SSL_CLIENT,
333                         clientPrincipal, serverPrincipal,
334                         tgsPrincipal, acc);
335                         }});
336 
337             if (ticket == null) {
338                 throw new IOException("Failed to find any kerberos service" +
339                         " ticket for " + serverPrincipal);
340             }
341             return ticket;
342         } catch (PrivilegedActionException e) {
343             IOException ioe = new IOException(
344                 "Attempt to obtain kerberos service ticket for " +
345                         serverPrincipal + " failed!");
346             ioe.initCause(e);
347             throw ioe;
348         }
349     }

366     /**
367      * Determines if a kvno matches another kvno. Used in the method
368      * findKey(etype, version, keys). Always returns true if either input
369      * is null or zero, in case any side does not have kvno info available.
370      *
371      * Note: zero is included because N/A is not a legal value for kvno
372      * in javax.security.auth.kerberos.KerberosKey. Therefore, the info
373      * that the kvno is N/A might be lost when converting between
374      * EncryptionKey and KerberosKey.
375      */
376     private static boolean versionMatches(Integer v1, int v2) {
377         if (v1 == null || v1 == 0 || v2 == 0) {
378             return true;
379         }
380         return v1.equals(v2);
381     }
382 
383     private static KerberosKey findKey(int etype, Integer version,
384             KerberosKey[] keys) throws KrbException {
385         int ktype;
386         boolean etypeFound = false;
387         for (int i = 0; i < keys.length; i++) {
388             ktype = keys[i].getKeyType();
389             if (etype == ktype) {
390                 etypeFound = true;
391                 if (versionMatches(version, keys[i].getVersionNumber())) {
392                     return keys[i];
393                 }
394             }
395         }
396         // Key not found.
397         // %%% kludge to allow DES keys to be used for diff etypes
398         if ((etype == EncryptedData.ETYPE_DES_CBC_CRC ||
399             etype == EncryptedData.ETYPE_DES_CBC_MD5)) {
400             for (int i = 0; i < keys.length; i++) {
401                 ktype = keys[i].getKeyType();
402                 if (ktype == EncryptedData.ETYPE_DES_CBC_CRC ||
403                         ktype == EncryptedData.ETYPE_DES_CBC_MD5) {
404                     etypeFound = true;
405                     if (versionMatches(version, keys[i].getVersionNumber())) {
406                         return new KerberosKey(keys[i].getPrincipal(),
407                             keys[i].getEncoded(),
408                             etype,
409                             keys[i].getVersionNumber());
410                     }
411                 }
412             }
413         }
414         if (etypeFound) {
415             throw new KrbException(Krb5.KRB_AP_ERR_BADKEYVER);
416         }
417         return null;
418     }
