184     private void createBrowser(Composite parent) {
185         GridData data = new GridData(GridData.FILL_BOTH);
186         data.grabExcessHorizontalSpace = true;
187         data.grabExcessVerticalSpace = true;
188         try {
189             browser = new Browser(parent, SWT.NO_BACKGROUND);
190             browser.setLayoutData(data);
191             browser.setBackground(parent.getBackground());
192             browser.addOpenWindowListener(new OpenWindowListener() {
193                 @Override
194                 public void open(WindowEvent event) {
195                     event.required = true; // Cancel opening of new windows
196                 }
197             });
198             browser.addLocationListener(new LocationListener() {
199                 @Override
200                 public void changed(LocationEvent event) {
201                     // ignore
202                 }
203 
204                 @Override
205                 public void changing(LocationEvent event) {
206                     // fix for SWT code on Won32 platform: it uses "about:blank"
207                     // before
208                     // set any non-null url. We ignore this url
209                     if (allowUrlChange || "about:blank".equals(event.location)) {
210                         return;
211                     }
212                     // disallow changing of property view content
213                     event.doit = false;
214                     // for any external url clicked by user we should leave
215                     // property view
216                     openBrowserInEditor(event);
217                 }
218             });
219         } catch (SWTError e) {
220             FindbugsPlugin plugin = FindbugsPlugin.getDefault();
221             plugin.logException(new RuntimeException(e.getMessage(), e),
222                     "Could not create org.eclipse.swt.widgets.Composite.Browser");
223 
224         }
225     }

281     private void refreshBrowser() {
282         String html = null;
283         if (browser != null && !browser.isDisposed()) {
284             html = getHtml();
285 
286             // avoid flickering if same input
287             if (!html.equals(oldText)) {
288                 allowUrlChange = true;
289                 browser.setText(html);
290                 allowUrlChange = false;
291             }
292         }
293         oldText = html;
294     }

297     private String getHtml() {
298         if (pattern == null) {
299             return "";
300         }
301         boolean hasBug = bug != null;
302         StringBuilder text = new StringBuilder();
303         if (!hasBug) {
304             text.append("<b>Pattern</b>: ");
305             text.append(pattern.getShortDescription());
306         } else {
307             text.append(pattern.getDetailText());
308         }
309         if (!hasBug) {
310             return text.toString();
311         }
312         if(text.lastIndexOf("</p>") == -1 || text.lastIndexOf("<br>") == -1) {
313             text.append("\n<p>");
314         }
315         text.append(getBugDetails());
316         text.append("<br>");
317         text.append(getPatternDetails());
318         addXmlOutput(text);
319         addDetectorInfo(text);
320         String html = "<b>Bug</b>: " + toSafeHtml(bug.getMessageWithoutPrefix()) + "<br>\n" + text.toString();
321         return html;
322     }

324     private void addXmlOutput(StringBuilder text) {
325         StringWriter stringWriter = new StringWriter();
326         XMLOutput xmlOutput = new OutputStreamXMLOutput(stringWriter);
327         try {
328             bug.writeXML(xmlOutput);
329         } catch (IOException e) {
330             // ignore
331         } finally {
332             try {
333                 xmlOutput.finish();
334             } catch (IOException e) {
335                 // ignore
336             }
337         }
338         text.append("<hr size=\"1\" /><p><b>XML output:</b>");
339         text.append("<pre>");
340         text.append(toSafeHtml(stringWriter.toString()));
341         text.append("</pre></p><hr size=\"1\" />");
342     }

344     private String getBugDetails() {
345         StringBuilder sb = new StringBuilder();
346         int rank = 0;
347         MarkerConfidence confidence = MarkerConfidence.Ignore;
348         if(bug != null) {
349             confidence = MarkerConfidence.getConfidence(bug.getPriority());
350             rank = bug.getBugRank();
351         } else if(marker != null) {
352             confidence = MarkerUtil.findConfidenceForMarker(marker);
353             rank = MarkerUtil.findBugRankForMarker(marker);
354         }
355         sb.append("\n<b>Rank</b>: ");
356         sb.append(BugRankCategory.getRank(rank));
357         sb.append(" (").append(rank).append(")");
358         sb.append(", <b>confidence</b>: ").append(confidence);
359         return sb.toString();
360     }

364             return "";
365         }

385                 text.append("<br>Contributed by plugin: ").append(plugin.getPluginId());
386                 text.append("<br>Provider: ").append(plugin.getProvider());

389                     text.append(" (<a href=\"").append(website).append("\">");

391                 }

397     private static String toSafeHtml(String s) {
398         if (s.indexOf(">") >= 0) {
399             s = s.replace(">", "&gt;");
400         }
401         if (s.indexOf("<") >= 0) {
402             s = s.replace("<", "&lt;");
403         }
404         return s;
405     }

468     private void showAnnotation(IEditorPart activeEditor) {
469         if (showingAnnotation) {
470             FindbugsPlugin.getDefault().logInfo("Recursive showAnnotation");
471         }
472         showingAnnotation = true;
473 
474         try {
475 
476             int index = annotationList.getSelectionIndex() - 1;
477             if (index >= 0) {
478                 BugAnnotation theAnnotation = bug.getAnnotationsForMessage(false).get(index);
479                 findLocation: try {
480 
481                     if (theAnnotation instanceof SourceLineAnnotation) {
482                         SourceLineAnnotation sla = (SourceLineAnnotation) theAnnotation;
483                         int line = sla.getStartLine();
484                         EditorUtil.goToLine(activeEditor, line);
485                         return;
486 
487                     } else if (theAnnotation instanceof MethodAnnotation) {
488                         MethodAnnotation ma = (MethodAnnotation) theAnnotation;
489                         String className = ma.getClassName();
490                         IJavaProject project = getIProject();
491                         IType type = project.findType(className);
492                         if (type == null) {
493                             break findLocation;
494                         }
495                         IMethod m = getIMethod(type, ma);
496                         if (m != null) {
497                             JavaUI.openInEditor(m, true, true);
498 
499                         } else {
500                             activeEditor = JavaUI.openInEditor(type, true, true);
501                             SourceLineAnnotation sla = ma.getSourceLines();
502                             EditorUtil.goToLine(activeEditor, sla.getStartLine());
503                         }
504 
505                         return;
506 
507                     } else if (theAnnotation instanceof FieldAnnotation) {
508                         FieldAnnotation fa = (FieldAnnotation) theAnnotation;
509                         String className = fa.getClassName();
510                         IJavaProject project = getIProject();
511                         IType type = project.findType(className);
512                         if (type == null) {
513                             break findLocation;
514                         }
515 
516                         IField f = type.getField(fa.getFieldName());
517                         if (f != null) {
518                             JavaUI.openInEditor(f, true, true);
519                         } else {
520                             activeEditor = JavaUI.openInEditor(type, true, true);
521                             SourceLineAnnotation sla = fa.getSourceLines();
522                             EditorUtil.goToLine(activeEditor, sla.getStartLine());
523                         }
524                         return;
525                     } else if (theAnnotation instanceof TypeAnnotation) {
526                         TypeAnnotation fa = (TypeAnnotation) theAnnotation;
527                         String className = ClassName.fromFieldSignature(fa.getTypeDescriptor());
528                         if (className == null) {
529                             break findLocation;
530                         }
531                         IJavaProject project = getIProject();
532                         IType type = project.findType(ClassName.toDottedClassName(className));
533                         if (type == null) {
534                             break findLocation;
535                         }
536                         JavaUI.openInEditor(type, true, true);
537                         return;
538 
539                     } else if (theAnnotation instanceof ClassAnnotation) {
540                         ClassAnnotation fa = (ClassAnnotation) theAnnotation;
541                         String className = fa.getClassName();
542                         IJavaProject project = getIProject();
543                         IType type = project.findType(className);
544                         if (type == null) {
545                             break findLocation;
546                         }
547                         JavaUI.openInEditor(type, true, true);
548                         return;
549                     }
550                 } catch (JavaModelException e) {
551                     FindbugsPlugin.getDefault().logException(e, "Could not open editor for " + theAnnotation);
552                 } catch (PartInitException e) {
553                     FindbugsPlugin.getDefault().logException(e, "Could not open editor for " + theAnnotation);
554                 }
555             }
556             if(marker != null) {
557                 int line = marker.getAttribute(IMarker.LINE_NUMBER, EditorUtil.DEFAULT_LINE_IN_EDITOR);
558                 EditorUtil.goToLine(activeEditor, line);
559             }
560         } finally {
561             showingAnnotation = false;
562         }
563     }

565     private static String stripFirstAndLast(String s) {
566         return s.substring(1, s.length()-1);
567     }
568     private static IMethod getIMethod(IType type, MethodAnnotation mma) throws JavaModelException {
569         String name = mma.getMethodName();
570         SignatureParser parser = new SignatureParser(mma.getMethodSignature());
571         String[] arguments = parser.getArguments();
572 
573 
574         nextMethod: for(IMethod m : type.getMethods()) {
575             if (!m.getElementName().equals(name)) {
576                 continue nextMethod;
577             }
578 
579             String [] mArguments = m.getParameterTypes();
580             if (arguments.length != mArguments.length) {
581                 continue nextMethod;
582             }
583 
584             for(int i = 0; i < arguments.length; i++) {
585                 String a = arguments[i];
586                 String ma = mArguments[i];
587                 while (a.startsWith("[") && ma.startsWith("[")) {
588                     a = a.substring(1);
589                     ma = ma.substring(1);
590                 }
591                 if (ma.startsWith("Q")) {
592                     ma = stripFirstAndLast(ma);
593                     ClassDescriptor ad = DescriptorFactory.createClassDescriptorFromFieldSignature(a);
594                     if (ad == null) {
595                         continue nextMethod;
596                     }
597                     a = ad.getSimpleName();
598                 }
599                 if (!ma.equals(a)) {
600                     continue nextMethod;
601                 }
602             }
603             return m;
604         }
605         return null;
606     }

608     private void copyInfoToClipboard() {
609         if(bug == null) {
610             return;
611         }
612         StringBuffer sb = new StringBuffer();
613         sb.append(removeHtmlMarkup(getHtml()));
614         sb.append("\n\n");
615         for(BugAnnotation ba : bug.getAnnotationsForMessage(true)) {
616             sb.append(ba.toString()).append("\n");
617         }
618         sb.append("\n");
619         if (file != null) {
620             sb.append("File: ").append(file.getLocation()).append("\n");
621         }
622         Util.copyToClipboard(sb.toString());
623     }

625     private static String removeHtmlMarkup(String html) {
626         // replace any amount of white space with newline between through one
627         // space
628         html = html.replaceAll("\\s*[\\n]+\\s*", " ");
629         // remove all valid html tags
630         html = html.replaceAll("<[a-zA-Z]+>", "\n");
631         html = html.replaceAll("</[a-zA-Z]+>", "");
632         // convert some of the entities which are used in current FB
633         // messages.xml
634         html = html.replaceAll("&nbsp;", "");
635         html = html.replaceAll("&lt;", "<");
636         html = html.replaceAll("&gt;", ">");
637         html = html.replaceAll("&amp;", "&");
638         return html.trim();
639     }

641     private void selectInEditor(boolean openEditor) {
642         if (bug == null || (file == null && javaElt == null)) {
643             return;
644         }
645         IWorkbenchPage page = contributingPart.getSite().getPage();
646         IEditorPart activeEditor = page.getActiveEditor();
647         IEditorInput input = activeEditor != null ? activeEditor.getEditorInput() : null;
648 
649         if (openEditor && !matchInput(input)) {
650             try {
651                 if (file != null) {
652                     activeEditor = IDE.openEditor(page, file);
653                 } else if (javaElt != null) {
654                     activeEditor = JavaUI.openInEditor(javaElt, true, true);
655                 }
656                 if (activeEditor != null) {
657                     input = activeEditor.getEditorInput();
658                 }
659             } catch (PartInitException e) {
660                 FindbugsPlugin.getDefault().logException(e, "Could not open editor for " + bug.getMessage());
661             } catch (CoreException e) {
662                 FindbugsPlugin.getDefault().logException(e, "Could not open editor for " + bug.getMessage());
663             }
664         }
665         if (matchInput(input)) {
666             showAnnotation(activeEditor);
667 
668         }
669     }

671     private boolean matchInput(IEditorInput input) {
672         if (file != null && (input instanceof IFileEditorInput)) {
673             return file.equals(((IFileEditorInput) input).getFile());
674         }
675         if (javaElt != null && input != null) {
676             IJavaElement javaElement = JavaUI.getEditorInputJavaElement(input);
677             if (javaElt.equals(javaElement)) {
678                 return true;
679             }
680             IJavaElement parent = javaElt.getParent();
681             while (parent != null && !parent.equals(javaElement)) {
682                 parent = parent.getParent();
683             }
684             if (parent != null && parent.equals(javaElement)) {
685                 return true;
686             }
687         }
688         return false;
689     }

691     private void refreshTitle() {
692         if (marker != null) {
693             String bugType = marker.getAttribute(FindBugsMarker.BUG_TYPE, "");
694             pattern = DetectorFactoryCollection.instance().lookupBugPattern(bugType);
695         }
696         if (pattern == null) {
697             return;
698         }
699         if (bug == null) {
700             return;
701         }
702         if(file != null) {
703             setContentDescription(file.getName() +
704                     ": " + marker.getAttribute(IMarker.LINE_NUMBER, 0));
705         } else {
706             setContentDescription("");
707         }
708     }

727     private void showInView(IMarker m) {
728         this.marker = m;
729         if (MarkerUtil.isFindBugsMarker(marker)) {
730             bug = MarkerUtil.findBugInstanceForMarker(marker);
731             file = (IFile) (marker.getResource() instanceof IFile ? marker.getResource() : null);
732             javaElt = MarkerUtil.findJavaElementForMarker(marker);
733 
734             pattern = bug != null ? bug.getBugPattern() : null;
735             refreshTitle();
736             refreshAnnotations();
737             refreshBrowser();
738             rootComposite.layout(true, true);
739         }
740     }
