44  /**
45   * List box with two buttons on the left side: "add" and "remove". First the
46   * viewer must be created, then the buttons.
47   *
48   * @author andrei
49   */
50  public class ManagePathsWidget extends Composite {
51  
52      private TableViewer viewer;
53  
54      public ManagePathsWidget(Composite parent) {
55          super(parent, SWT.NONE);
56      }
57  
58      public CheckboxTableViewer createViewer(String title, String linkText, boolean withCheckBox) {
59          GridLayout layout = new GridLayout(2, false);
60          layout.marginHeight = 0;
61          layout.marginWidth = 0;
62          this.setLayout(layout);
63          this.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true, 2, 1));
64          Label titleLabel = new Label(this, SWT.NULL);
65  
66          titleLabel.setText(title);
67          titleLabel.setLayoutData(new GridData(SWT.LEAD, SWT.CENTER, true, false, 2, 1));
68  
69          if (linkText != null) {
70              Link details = new Link(this, SWT.NULL);
71              details.setText(linkText);
72              details.setLayoutData(new GridData(SWT.LEAD, SWT.CENTER, true, false, 2, 1));
73              details.addSelectionListener(new SelectionListener() {
74                  @Override
75                  public void widgetSelected(SelectionEvent e) {
76                      Program.launch(e.text);
77                  }
78  
79                  @Override
80                  public void widgetDefaultSelected(SelectionEvent e) {
81                      // noop
82                  }
83              });
84          }
85  
86          int style = SWT.MULTI | SWT.BORDER | SWT.H_SCROLL | SWT.V_SCROLL;
87          if(withCheckBox) {
88              style |= SWT.CHECK;
89          }
90          Table table = new Table(this, style);
91          CheckboxTableViewer viewer1 = new CheckboxTableViewer(table);
92          viewer1.getControl().setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true, 1, 2));
93          final PathElementLabelProvider labelProvider = new PathElementLabelProvider();
94          viewer1.setLabelProvider(labelProvider);
95          this.viewer = viewer1;
96          viewer1.getControl().addMouseTrackListener(new MouseTrackAdapter() {
97              @Override
98              public void mouseHover(MouseEvent e) {
99                  String tooltip = "";
100                 ViewerCell cell = viewer.getCell(new Point(e.x, e.y));
101                 if(cell != null) {
102                     tooltip = labelProvider.getToolTip(cell.getElement());
103                 }
104                 viewer.getControl().setToolTipText(tooltip);
105             }
106             @Override
107             public void mouseExit(MouseEvent e) {
108                 viewer.getControl().setToolTipText("");
109             }
110         });
111         return viewer1;
112     }
113 
114     public TableViewer createViewer(String title, String linkText) {
115         return createViewer(title, linkText, false);
116     }
117 
118     public void createButtonsArea(PathsProvider contentProvider) {
119         final Button addButton = new Button(this, SWT.PUSH);
120         String addButtonLabel = getMessage("property.addbutton");
121 
122         addButton.setText(addButtonLabel);
123         addButton.setData("add");
124         addButton.setLayoutData(new GridData(SWT.FILL, SWT.BEGINNING, false, false));
125 
126         addButton.addSelectionListener(contentProvider);
127         final Button removeButton = new Button(this, SWT.PUSH);
128         removeButton.setLayoutData(new GridData(SWT.FILL, SWT.BEGINNING, false, true));
129         String removeButtonLabel = getMessage("property.removebutton");
130 
131         removeButton.setText(removeButtonLabel);
132         removeButton.setData("remove");
133         removeButton.setEnabled(false);
134         removeButton.addSelectionListener(contentProvider);
135 
136         viewer.addSelectionChangedListener(new ISelectionChangedListener() {
137             @Override
138             public void selectionChanged(SelectionChangedEvent event) {
139                 IStructuredSelection selection = (IStructuredSelection) event.getSelection();
140                 if(selection.isEmpty()) {
141                     removeButton.setEnabled(false);
142                 } else {
143                     if(selection.getFirstElement() instanceof IPathElement) {
144                         IPathElement pathElement = (IPathElement) selection.getFirstElement();
145                         removeButton.setEnabled(!pathElement.isSystem());
146                     }
147                 }
148             }
149         });
150     }
151 
152     protected static String getMessage(String key) {
153         return FindbugsPlugin.getDefault().getMessage(key);
154     }
