74          synchronized (CWarningWindow.class) {
75              if (icons == null) {
76                  icons = new IconInfo[4][3];
77                  icons[0][0] = new IconInfo(sun.awt.AWTIcon32_security_icon_bw16_png.security_icon_bw16_png);
78                  icons[0][1] = new IconInfo(sun.awt.AWTIcon32_security_icon_interim16_png.security_icon_interim16_png);
79                  icons[0][2] = new IconInfo(sun.awt.AWTIcon32_security_icon_yellow16_png.security_icon_yellow16_png);
80                  icons[1][0] = new IconInfo(sun.awt.AWTIcon32_security_icon_bw24_png.security_icon_bw24_png);
81                  icons[1][1] = new IconInfo(sun.awt.AWTIcon32_security_icon_interim24_png.security_icon_interim24_png);
82                  icons[1][2] = new IconInfo(sun.awt.AWTIcon32_security_icon_yellow24_png.security_icon_yellow24_png);
83                  icons[2][0] = new IconInfo(sun.awt.AWTIcon32_security_icon_bw32_png.security_icon_bw32_png);
84                  icons[2][1] = new IconInfo(sun.awt.AWTIcon32_security_icon_interim32_png.security_icon_interim32_png);
85                  icons[2][2] = new IconInfo(sun.awt.AWTIcon32_security_icon_yellow32_png.security_icon_yellow32_png);
86                  icons[3][0] = new IconInfo(sun.awt.AWTIcon32_security_icon_bw48_png.security_icon_bw48_png);

95      public CWarningWindow(final Window _ownerWindow, final LWWindowPeer _ownerPeer) {
96          super();
97  
98          this.ownerPeer = new WeakReference<LWWindowPeer>(_ownerPeer);
99          this.ownerWindow = _ownerWindow;
100 
101         initialize(null, null, _ownerPeer.getPlatformWindow());
102 
103         setOpaque(false);
104 
105         String warningString = ownerWindow.getWarningString();
106         if (warningString != null) {
107             contentView.setToolTip(ownerWindow.getWarningString());
108         }
109 
110         updateIconSize();
111     }

122     public void setVisible(boolean visible, boolean doSchedule) {
123         synchronized (scheduler) {
124             if (showingTaskHandle != null) {
125                 showingTaskHandle.cancel(false);
126                 showingTaskHandle = null;
127             }
128 
129             if (hidingTaskHandle != null) {
130                 hidingTaskHandle.cancel(false);
131                 hidingTaskHandle = null;
132             }
133 
134             if (visible) {
135                 if (isVisible()) {
136                     currentIcon = 0;
137                 } else {
138                     currentIcon = 2;
139                 }
140 
141                 showingTaskHandle = scheduler.schedule(showingTask, 50,
142                         TimeUnit.MILLISECONDS);
143 
144             } else {
145                 if (!isVisible()) {
146                     return;
147                 }
148 
149                 if (doSchedule) {
150                     hidingTaskHandle = scheduler.schedule(hidingTask, HIDING_DELAY,
151                             TimeUnit.MILLISECONDS);
152                 } else {
153                     hidingTaskHandle = scheduler.schedule(hidingTask, 50,
154                             TimeUnit.MILLISECONDS);
155                 }
156             }
157         }
158     }

189     @Override
190     public void notifyMouseEvent(int id, long when, int button, int x, int y,
191                                  int screenX, int screenY, int modifiers,
192                                  int clickCount, boolean popupTrigger,
193                                  byte[] bdata) {
194         LWWindowPeer peer = ownerPeer.get();
195         if (id == MouseEvent.MOUSE_EXITED) {
196             if (peer != null) {
197                 peer.updateSecurityWarningVisibility();
198             }
199         } else if(id == MouseEvent.MOUSE_ENTERED) {
200             if (peer != null) {
201                 peer.updateSecurityWarningVisibility();
202             }
203         }

219     @Override
220     public void setVisible(boolean visible) {
221         synchronized (lock) {
222             final long nsWindowPtr = getNSWindowPtr();
223 
224             // Process parent-child relationship when hiding
225             if (!visible) {
226                 // Unparent myself
227                 if (owner != null && owner.isVisible()) {
228                     CWrapper.NSWindow.removeChildWindow(
229                             owner.getNSWindowPtr(), nsWindowPtr);
230                 }
231             }
232 
233             // Actually show or hide the window
234             if (visible) {
235                 CWrapper.NSWindow.orderFront(nsWindowPtr);
236             } else {
237                 CWrapper.NSWindow.orderOut(nsWindowPtr);
238             }
239 
240             this.visible = visible;
241 
242             // Manage parent-child relationship when showing
243             if (visible) {
244                 // Add myself as a child
245                 if (owner != null && owner.isVisible()) {
246                     CWrapper.NSWindow.addChildWindow(owner.getNSWindowPtr(),
247                             nsWindowPtr, CWrapper.NSWindow.NSWindowAbove);
248 
249                     // do not allow security warning to be obscured by other windows
250                     if (ownerWindow.isAlwaysOnTop()) {
251                         CWrapper.NSWindow.setLevel(nsWindowPtr,
252                                 CWrapper.NSWindow.NSFloatingWindowLevel);
253                     }
254                 }
255             }
256         }
257     }

277     protected void deliverMoveResizeEvent(int x, int y, int width, int height,
278                                           boolean byUser) {
279 
280         boolean isResize;
281         synchronized (lock) {
282             isResize = (bounds.width != width || bounds.height != height);
283             bounds = new Rectangle(x, y, width, height);
284         }
285 
286         if (isResize) {
287             replaceSurface();
288         }
289 
290         super.deliverMoveResizeEvent(x, y, width, height, byUser);
291     }

327     private void updateIconSize() {
328         int newSize = -1;
329 
330         if (ownerWindow != null) {
331             Insets insets = ownerWindow.getInsets();
332             int max = Math.max(insets.top, Math.max(insets.bottom,
333                     Math.max(insets.left, insets.right)));
334             if (max < 24) {
335                 newSize = 0;
336             } else if (max < 32) {
337                 newSize = 1;
338             } else if (max < 48) {
339                 newSize = 2;
340             } else {
341                 newSize = 3;
342             }
343         }
344         // Make sure we have a valid size
345         if (newSize == -1) {
346             newSize = 0;
347         }
348 
349         synchronized (lock) {
350             if (newSize != currentSize) {
351                 currentSize = newSize;
352                 IconInfo ico = getSecurityIconInfo(currentSize, 0);
353                 AWTAccessor.getWindowAccessor().setSecurityWarningSize(
354                     ownerWindow, ico.getWidth(), ico.getHeight());
355             }
356         }
357     }

359     private final Graphics getGraphics() {
360         SurfaceData sd = contentView.getSurfaceData();
361         if (ownerWindow == null || sd == null) {
362             return null;
363         }
364 
365         return transformGraphics(new SunGraphics2D(sd, SystemColor.windowText,
366                 SystemColor.window, ownerWindow.getFont()));
367     }

370     private void repaint() {
371         final Graphics g = getGraphics();
372         if (g != null) {
373             try {
374                 ((Graphics2D) g).setComposite(AlphaComposite.Src);
375                 g.drawImage(getSecurityIconInfo().getImage(), 0, 0, null);
376             } finally {
377                 g.dispose();
378             }
379         }
380     }
381 
382     private void replaceSurface() {
383         SurfaceData oldData = contentView.getSurfaceData();
384 
385         replaceSurfaceData();
386 
387         if (oldData != null && oldData != contentView.getSurfaceData()) {
388             oldData.flush();
389         }
390     }

404     private final Runnable hidingTask = new Runnable() {
405         public void run() {
406             synchronized (lock) {
407                 setVisible(false);
408             }
409 
410             synchronized (scheduler) {
411                 hidingTaskHandle = null;
412             }
413         }
414     };
415 
416     private final Runnable showingTask = new Runnable() {
417         public void run() {
418             synchronized (lock) {
419                 if (!isVisible()) {
420                     setVisible(true);
421                 }
422 
423                 repaint();
424             }
425 
426             synchronized (scheduler) {
427                 if (currentIcon > 0) {
428                     currentIcon--;
429                     showingTaskHandle = scheduler.schedule(showingTask, SHOWING_DELAY,
430                             TimeUnit.MILLISECONDS);
431                 } else {
432                     showingTaskHandle = null;
433                 }
434             }
435         }
436     };
437 
438     private final ScheduledExecutorService scheduler =
439             Executors.newSingleThreadScheduledExecutor();
440 
441     private ScheduledFuture hidingTaskHandle;
442     private ScheduledFuture showingTaskHandle;
