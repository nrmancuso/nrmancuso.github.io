443   /**
444    * Nonce manager. Nonces are used to make operations like increment and append idempotent
445    * in the case where client doesn't receive the response from a successful operation and
446    * retries. We track the successful ops for some time via a nonce sent by client and handle
447    * duplicate operations (currently, by failing them; in future we might use MVCC to return
448    * result). Nonces are also recovered from WAL during, recovery; however, the caveats (from
449    * HBASE-3787) are:
450    * - WAL recovery is optimized, and under high load we won't read nearly nonce-timeout worth
451    *   of past records. If we don't read the records, we don't read and recover the nonces.
452    *   Some WALs within nonce-timeout at recovery may not even be present due to rolling/cleanup.
453    * - There's no WAL recovery during normal region move, so nonces will not be transfered.
454    * We can have separate additional "Nonce WAL". It will just contain bunch of numbers and
455    * won't be flushed on main path - because WAL itself also contains nonces, if we only flush
456    * it before memstore flush, for a given nonce we will either see it in the WAL (if it was
457    * never flushed to disk, it will be part of recovery), or we'll see it as part of the nonce
458    * log (or both occasionally, which doesn't matter). Nonce log file can be deleted after the
459    * latest nonce in it expired. It can also be recovered during move.

487   /**
488    * Starts a HRegionServer at the default location
489    * @param conf
490    * @param csm implementation of CoordinatedStateManager to be used
491    * @throws IOException
492    * @throws InterruptedException
493    */
494   public HRegionServer(Configuration conf, CoordinatedStateManager csm)
495       throws IOException, InterruptedException {
496     this.fsOk = true;
497     this.conf = conf;
498     checkCodecs(this.conf);
499     this.userProvider = UserProvider.instantiate(conf);
500     FSUtils.setupShortCircuitRead(this.conf);
501     // Disable usage of meta replicas in the regionserver
502     this.conf.setBoolean(HConstants.USE_META_REPLICAS, false);
503 
504     // Config'ed params
505     this.numRetries = this.conf.getInt(HConstants.HBASE_CLIENT_RETRIES_NUMBER,
506         HConstants.DEFAULT_HBASE_CLIENT_RETRIES_NUMBER);
507     this.threadWakeFrequency = conf.getInt(HConstants.THREAD_WAKE_FREQUENCY, 10 * 1000);
508     this.msgInterval = conf.getInt("hbase.regionserver.msginterval", 3 * 1000);
509 
510     this.sleeper = new Sleeper(this.msgInterval, this);
511 
512     boolean isNoncesEnabled = conf.getBoolean(HConstants.HBASE_RS_NONCES_ENABLED, true);
513     this.nonceManager = isNoncesEnabled ? new ServerNonceManager(this.conf) : null;
514 
515     this.numRegionsToReport = conf.getInt(
516       "hbase.regionserver.numregionstoreport", 10);
517 
518     this.operationTimeout = conf.getInt(
519       HConstants.HBASE_CLIENT_OPERATION_TIMEOUT,
520       HConstants.DEFAULT_HBASE_CLIENT_OPERATION_TIMEOUT);
521 
522     this.shortOperationTimeout = conf.getInt(
523       HConstants.HBASE_RPC_SHORTOPERATION_TIMEOUT_KEY,
524       HConstants.DEFAULT_HBASE_RPC_SHORTOPERATION_TIMEOUT);
525 
526     this.abortRequested = false;
527     this.stopped = false;
528 
529     rpcServices = createRpcServices();
530     this.startcode = System.currentTimeMillis();
531     if (this instanceof HMaster) {
532       useThisHostnameInstead = conf.get(MASTER_HOSTNAME_KEY);
533     } else {
534       useThisHostnameInstead = conf.get(RS_HOSTNAME_KEY);
535     }
536     String hostName = shouldUseThisHostnameInstead() ? useThisHostnameInstead :
537       rpcServices.isa.getHostName();
538     serverName = ServerName.valueOf(hostName, rpcServices.isa.getPort(), startcode);
539 
540     rpcControllerFactory = RpcControllerFactory.instantiate(this.conf);
541     rpcRetryingCallerFactory = RpcRetryingCallerFactory.instantiate(this.conf);
542 
543     // login the zookeeper client principal (if using security)
544     ZKUtil.loginClient(this.conf, "hbase.zookeeper.client.keytab.file",
545       "hbase.zookeeper.client.kerberos.principal", hostName);
546     // login the server principal (if using secure Hadoop)
547     login(userProvider, hostName);
548 
549     regionServerAccounting = new RegionServerAccounting();
550     uncaughtExceptionHandler = new UncaughtExceptionHandler() {
551       @Override
552       public void uncaughtException(Thread t, Throwable e) {
553         abort("Uncaught exception in service thread " + t.getName(), e);
554       }
555     };
556 
557     useZKForAssignment = ConfigUtil.useZKForAssignment(conf);
558 
559     // Set 'fs.defaultFS' to match the filesystem on hbase.rootdir else
560     // underlying hadoop hdfs accessors will be going against wrong filesystem
561     // (unless all is set to defaults).
562     FSUtils.setFsDefault(this.conf, FSUtils.getRootDir(this.conf));
563     // Get fs instance used by this RS.  Do we use checksum verification in the hbase? If hbase
564     // checksum verification enabled, then automatically switch off hdfs checksum verification.
565     boolean useHBaseChecksum = conf.getBoolean(HConstants.HBASE_CHECKSUM_VERIFICATION, true);
566     this.fs = new HFileSystem(this.conf, useHBaseChecksum);
567     this.rootDir = FSUtils.getRootDir(this.conf);
568     this.tableDescriptors = new FSTableDescriptors(
569       this.conf, this.fs, this.rootDir, !canUpdateTableDescriptor(), false);
570 
571     service = new ExecutorService(getServerName().toShortString());
572     spanReceiverHost = SpanReceiverHost.getInstance(getConfiguration());
573 
574     // Some unit tests don't need a cluster, so no zookeeper at all

741   /**
742    * Bring up connection to zk ensemble and then wait until a master for this
743    * cluster and then after that, wait until cluster 'up' flag has been set.
744    * This is the order in which master does things.
745    * Finally open long-living server short-circuit connection.
746    * @throws IOException
747    * @throws InterruptedException
748    */
749   private void initializeZooKeeper() throws IOException, InterruptedException {
750     // Create the master address tracker, register with zk, and start it.  Then
751     // block until a master is available.  No point in starting up if no master
752     // running.
753     blockAndCheckIfStopped(this.masterAddressTracker);
754 
755     // Wait on cluster being up.  Master will set this flag up in zookeeper
756     // when ready.
757     blockAndCheckIfStopped(this.clusterStatusTracker);
758 
759     // Retrieve clusterId
760     // Since cluster status is now up
761     // ID should have already been set by HMaster
762     try {
763       clusterId = ZKClusterId.readClusterIdZNode(this.zooKeeper);
764       if (clusterId == null) {
765         this.abort("Cluster ID has not been set");
766       }
767       LOG.info("ClusterId : "+clusterId);
768     } catch (KeeperException e) {
769       this.abort("Failed to retrieve Cluster ID",e);
770     }
771 
772     // In case colocated master, wait here till it's active.
773     // So backup masters won't start as regionservers.
774     // This is to avoid showing backup masters as regionservers
775     // in master web UI, or assigning any region to them.
776     waitForMasterActive();
777     if (isStopped() || isAborted()) {
778       return; // No need for further initialization
779     }
780 
781     // watch for snapshots and other procedures
782     try {
783       rspmHost = new RegionServerProcedureManagerHost();
784       rspmHost.loadProcedures(conf);
785       rspmHost.initialize(this);
786     } catch (KeeperException e) {
787       this.abort("Failed to reach zk cluster when creating procedure handler.", e);
788     }
789     // register watcher for recovering regions
790     this.recoveringRegionWatcher = new RecoveringRegionWatcher(this.zooKeeper, this);
791   }

866   /**
867    * The HRegionServer sticks in this loop until closed.
868    */
869   @Override
870   public void run() {
871     try {
872       // Do pre-registration initializations; zookeeper, lease threads, etc.
873       preRegistrationInitialization();
874     } catch (Throwable e) {
875       abort("Fatal exception during initialization", e);
876     }
877 
878     try {
879       if (!isStopped() && !isAborted()) {
880         ShutdownHook.install(conf, fs, this, Thread.currentThread());
881         // Set our ephemeral znode up in zookeeper now we have a name.
882         createMyEphemeralNode();
883         // Initialize the RegionServerCoprocessorHost now that our ephemeral
884         // node was created, in case any coprocessors want to use ZooKeeper
885         this.rsHost = new RegionServerCoprocessorHost(this, this.conf);
886       }
887 
888       // Try and register with the Master; tell it we are here.  Break if
889       // server is stopped or the clusterup flag is down or hdfs went wacky.
890       while (keepLooping()) {
891         RegionServerStartupResponse w = reportForDuty();
892         if (w == null) {
893           LOG.warn("reportForDuty failed; sleeping and then retrying.");
894           this.sleeper.sleep();
895         } else {
896           handleReportForDutyResponse(w);
897           break;
898         }
899       }
900 
901       if (!isStopped() && isHealthy()){
902         // start the snapshot handler and other procedure handlers,
903         // since the server is ready to run
904         rspmHost.start();
905       }
906       
907       // Start the Quota Manager
908       rsQuotaManager.start(getRpcServer().getScheduler());
909 
910       // We registered with the Master.  Go into run mode.
911       long lastMsg = System.currentTimeMillis();
912       long oldRequestCount = -1;
913       // The main run loop.
914       while (!isStopped() && isHealthy()) {
915         if (!isClusterUp()) {
916           if (isOnlineRegionsEmpty()) {
917             stop("Exiting; cluster shutdown set and not carrying any regions");
918           } else if (!this.stopping) {
919             this.stopping = true;
920             LOG.info("Closing user regions");
921             closeUserRegions(this.abortRequested);
922           } else if (this.stopping) {
923             boolean allUserRegionsOffline = areAllUserRegionsOffline();
924             if (allUserRegionsOffline) {
925               // Set stopped if no more write requests tp meta tables
926               // since last time we went around the loop.  Any open
927               // meta regions will be closed on our way out.
928               if (oldRequestCount == getWriteRequestCount()) {
929                 stop("Stopped; only catalog regions remaining online");
930                 break;
931               }
932               oldRequestCount = getWriteRequestCount();
933             } else {
934               // Make sure all regions have been closed -- some regions may
935               // have not got it because we were splitting at the time of
936               // the call to closeUserRegions.
937               closeUserRegions(this.abortRequested);
938             }
939             LOG.debug("Waiting on " + getOnlineRegionsAsPrintableString());
940           }
941         }
942         long now = System.currentTimeMillis();
943         if ((now - lastMsg) >= msgInterval) {
944           tryRegionServerReport(lastMsg, now);
945           lastMsg = System.currentTimeMillis();
946           doMetrics();
947         }
948         if (!isStopped() && !isAborted()) {
949           this.sleeper.sleep();
950         }
951       } // for
952     } catch (Throwable t) {
953       if (!rpcServices.checkOOME(t)) {
954         String prefix = t instanceof YouAreDeadException? "": "Unhandled: ";
955         abort(prefix + t.getMessage(), t);
956       }
957     }
958     // Run shutdown.
959     if (mxBean != null) {
960       MBeanUtil.unregisterMBean(mxBean);
961       mxBean = null;
962     }
963     if (this.leases != null) this.leases.closeAfterLeasesExpire();
964     if (this.splitLogWorker != null) {
965       splitLogWorker.stop();
966     }
967     if (this.infoServer != null) {
968       LOG.info("Stopping infoServer");
969       try {
970         this.infoServer.stop();
971       } catch (Exception e) {
972         LOG.error("Failed to stop infoServer", e);
973       }
974     }
975     // Send cache a shutdown.
976     if (cacheConfig != null && cacheConfig.isBlockCacheEnabled()) {
977       cacheConfig.getBlockCache().shutdown();
978     }
979 
980     if (movedRegionsCleaner != null) {
981       movedRegionsCleaner.stop("Region Server stopping");
982     }
983 
984     // Send interrupts to wake up threads if sleeping so they notice shutdown.
985     // TODO: Should we check they are alive? If OOME could have exited already
986     if (this.hMemManager != null) this.hMemManager.stop();
987     if (this.cacheFlusher != null) this.cacheFlusher.interruptIfNecessary();
988     if (this.compactSplitThread != null) this.compactSplitThread.interruptIfNecessary();
989     if (this.compactionChecker != null) this.compactionChecker.cancel(true);
990     if (this.healthCheckChore != null) this.healthCheckChore.cancel(true);
991     if (this.nonceManagerChore != null) this.nonceManagerChore.cancel(true);
992     if (this.storefileRefresher != null) this.storefileRefresher.cancel(true);
993     sendShutdownInterrupt();
994 
995     // Stop the quota manager
996     if (rsQuotaManager != null) {
997       rsQuotaManager.stop();
998     }
999     
1000     // Stop the snapshot and other procedure handlers, forcefully killing all running tasks
1001     if (rspmHost != null) {
1002       rspmHost.stop(this.abortRequested || this.killed);
1003     }
1004 
1005     if (this.killed) {
1006       // Just skip out w/o closing regions.  Used when testing.
1007     } else if (abortRequested) {
1008       if (this.fsOk) {
1009         closeUserRegions(abortRequested); // Don't leave any open file handles
1010       }
1011       LOG.info("aborting server " + this.serverName);
1012     } else {
1013       closeUserRegions(abortRequested);
1014       LOG.info("stopping server " + this.serverName);
1015     }
1016 
1017     // so callers waiting for meta without timeout can stop
1018     if (this.metaTableLocator != null) this.metaTableLocator.stop();
1019     if (this.clusterConnection != null && !clusterConnection.isClosed()) {
1020       try {
1021         this.clusterConnection.close();
1022       } catch (IOException e) {
1023         // Although the {@link Closeable} interface throws an {@link
1024         // IOException}, in reality, the implementation would never do that.
1025         LOG.warn("Attempt to close server's short circuit HConnection failed.", e);
1026       }
1027     }
1028 
1029     // Closing the compactSplit thread before closing meta regions
1030     if (!this.killed && containsMetaTableRegions()) {
1031       if (!abortRequested || this.fsOk) {
1032         if (this.compactSplitThread != null) {
1033           this.compactSplitThread.join();
1034           this.compactSplitThread = null;
1035         }
1036         closeMetaTableRegions(abortRequested);
1037       }
1038     }
1039 
1040     if (!this.killed && this.fsOk) {
1041       waitOnAllRegionsToClose(abortRequested);
1042       LOG.info("stopping server " + this.serverName +
1043         "; all regions closed.");
1044     }
1045 
1046     //fsOk flag may be changed when closing regions throws exception.
1047     if (this.fsOk) {
1048       shutdownWAL(!abortRequested);
1049     }
1050 
1051     // Make sure the proxy is down.
1052     if (this.rssStub != null) {
1053       this.rssStub = null;
1054     }
1055     if (this.rpcClient != null) {
1056       this.rpcClient.close();
1057     }
1058     if (this.leases != null) {
1059       this.leases.close();
1060     }
1061     if (this.pauseMonitor != null) {
1062       this.pauseMonitor.stop();
1063     }
1064 
1065     if (!killed) {
1066       stopServiceThreads();
1067     }
1068 
1069     if (this.rpcServices != null) {
1070       this.rpcServices.stop();
1071     }
1072 
1073     try {
1074       deleteMyEphemeralNode();
1075     } catch (KeeperException.NoNodeException nn) {
1076     } catch (KeeperException e) {
1077       LOG.warn("Failed deleting my ephemeral node", e);
1078     }
1079     // We may have failed to delete the znode at the previous step, but
1080     //  we delete the file anyway: a second attempt to delete the znode is likely to fail again.
1081     ZNodeClearer.deleteMyEphemeralNodeOnDisk();
1082 
1083     if (this.zooKeeper != null) {
1084       this.zooKeeper.close();
1085     }
1086     LOG.info("stopping server " + this.serverName +
1087       "; zookeeper connection closed.");
1088 
1089     LOG.info(Thread.currentThread().getName() + " exiting");
1090   }

1150   ClusterStatusProtos.ServerLoad buildServerLoad(long reportStartTime, long reportEndTime)
1151       throws IOException {
1152     // We're getting the MetricsRegionServerWrapper here because the wrapper computes requests
1153     // per second, and other metrics  As long as metrics are part of ServerLoad it's best to use
1154     // the wrapper to compute those numbers in one place.
1155     // In the long term most of these should be moved off of ServerLoad and the heart beat.
1156     // Instead they should be stored in an HBase table so that external visibility into HBase is
1157     // improved; Additionally the load balancer will be able to take advantage of a more complete
1158     // history.
1159     MetricsRegionServerWrapper regionServerWrapper = metricsRegionServer.getRegionServerWrapper();
1160     Collection<Region> regions = getOnlineRegionsLocalContext();
1161     MemoryUsage memory = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
1162 
1163     ClusterStatusProtos.ServerLoad.Builder serverLoad =
1164       ClusterStatusProtos.ServerLoad.newBuilder();
1165     serverLoad.setNumberOfRequests((int) regionServerWrapper.getRequestsPerSecond());
1166     serverLoad.setTotalNumberOfRequests((int) regionServerWrapper.getTotalRequestCount());
1167     serverLoad.setUsedHeapMB((int)(memory.getUsed() / 1024 / 1024));
1168     serverLoad.setMaxHeapMB((int) (memory.getMax() / 1024 / 1024));
1169     Set<String> coprocessors = getWAL(null).getCoprocessorHost().getCoprocessors();
1170     for (String coprocessor : coprocessors) {
1171       serverLoad.addCoprocessors(
1172         Coprocessor.newBuilder().setName(coprocessor).build());
1173     }
1174     RegionLoad.Builder regionLoadBldr = RegionLoad.newBuilder();
1175     RegionSpecifier.Builder regionSpecifier = RegionSpecifier.newBuilder();
1176     for (Region region : regions) {
1177       serverLoad.addRegionLoads(createRegionLoad(region, regionLoadBldr, regionSpecifier));
1178       for (String coprocessor :
1179           getWAL(region.getRegionInfo()).getCoprocessorHost().getCoprocessors()) {
1180         serverLoad.addCoprocessors(Coprocessor.newBuilder().setName(coprocessor).build());
1181       }
1182     }
1183     serverLoad.setReportStartTime(reportStartTime);
1184     serverLoad.setReportEndTime(reportEndTime);
1185     if (this.infoServer != null) {
1186       serverLoad.setInfoServerPort(this.infoServer.getPort());
1187     } else {
1188       serverLoad.setInfoServerPort(-1);
1189     }
1190 
1191     // for the replicationLoad purpose. Only need to get from one service
1192     // either source or sink will get the same info
1193     ReplicationSourceService rsources = getReplicationSourceService();
1194 
1195     if (rsources != null) {
1196       // always refresh first to get the latest value
1197       ReplicationLoad rLoad = rsources.refreshAndGetReplicationLoad();
1198       if (rLoad != null) {
1199         serverLoad.setReplLoadSink(rLoad.getReplicationLoadSink());
1200         for (ClusterStatusProtos.ReplicationLoadSource rLS : rLoad.getReplicationLoadSourceList()) {
1201           serverLoad.addReplLoadSource(rLS);
1202         }
1203       }
1204     }
1205 
1206     return serverLoad.build();
1207   }

1219    * Wait on regions close.
1220    */
1221   private void waitOnAllRegionsToClose(final boolean abort) {
1222     // Wait till all regions are closed before going out.
1223     int lastCount = -1;
1224     long previousLogTime = 0;
1225     Set<String> closedRegions = new HashSet<String>();
1226     boolean interrupted = false;
1227     try {
1228       while (!isOnlineRegionsEmpty()) {
1229         int count = getNumberOfOnlineRegions();
1230         // Only print a message if the count of regions has changed.
1231         if (count != lastCount) {
1232           // Log every second at most
1233           if (System.currentTimeMillis() > (previousLogTime + 1000)) {
1234             previousLogTime = System.currentTimeMillis();
1235             lastCount = count;
1236             LOG.info("Waiting on " + count + " regions to close");
1237             // Only print out regions still closing if a small number else will
1238             // swamp the log.
1239             if (count < 10 && LOG.isDebugEnabled()) {
1240               LOG.debug(this.onlineRegions);
1241             }
1242           }
1243         }
1244         // Ensure all user regions have been sent a close. Use this to
1245         // protect against the case where an open comes in after we start the
1246         // iterator of onlineRegions to close all user regions.
1247         for (Map.Entry<String, Region> e : this.onlineRegions.entrySet()) {
1248           HRegionInfo hri = e.getValue().getRegionInfo();
1249           if (!this.regionsInTransitionInRS.containsKey(hri.getEncodedNameAsBytes())
1250               && !closedRegions.contains(hri.getEncodedName())) {
1251             closedRegions.add(hri.getEncodedName());
1252             // Don't update zk with this close transition; pass false.
1253             closeRegionIgnoreErrors(hri, abort);
1254               }
1255         }
1256         // No regions in RIT, we could stop waiting now.
1257         if (this.regionsInTransitionInRS.isEmpty()) {
1258           if (!isOnlineRegionsEmpty()) {
1259             LOG.info("We were exiting though online regions are not empty," +
1260                 " because some regions failed closing");
1261           }
1262           break;
1263         }
1264         if (sleep(200)) {
1265           interrupted = true;
1266         }
1267       }
1268     } finally {
1269       if (interrupted) {
1270         Thread.currentThread().interrupt();
1271       }
1272     }
1273   }

1483   /*
1484    * Inner class that runs on a long period checking if regions need compaction.
1485    */
1486   private static class CompactionChecker extends ScheduledChore {
1487     private final HRegionServer instance;
1488     private final int majorCompactPriority;
1489     private final static int DEFAULT_PRIORITY = Integer.MAX_VALUE;
1490     private long iteration = 0;
1491 
1492     CompactionChecker(final HRegionServer h, final int sleepTime,
1493         final Stoppable stopper) {
1494       super("CompactionChecker", stopper, sleepTime);
1495       this.instance = h;
1496       LOG.info(this.getName() + " runs every " + StringUtils.formatTime(sleepTime));
1497 
1498       /* MajorCompactPriority is configurable.
1499        * If not set, the compaction will use default priority.
1500        */
1501       this.majorCompactPriority = this.instance.conf.
1502         getInt("hbase.regionserver.compactionChecker.majorCompactPriority",
1503         DEFAULT_PRIORITY);
1504     }
1505 
1506     @Override
1507     protected void chore() {
1508       for (Region r : this.instance.onlineRegions.values()) {
1509         if (r == null)
1510           continue;
1511         for (Store s : r.getStores()) {
1512           try {
1513             long multiplier = s.getCompactionCheckMultiplier();
1514             assert multiplier > 0;
1515             if (iteration % multiplier != 0) continue;
1516             if (s.needsCompaction()) {
1517               // Queue a compaction. Will recognize if major is needed.
1518               this.instance.compactSplitThread.requestSystemCompaction(r, s, getName()
1519                   + " requests compaction");
1520             } else if (s.isMajorCompaction()) {
1521               if (majorCompactPriority == DEFAULT_PRIORITY
1522                   || majorCompactPriority > ((HRegion)r).getCompactPriority()) {
1523                 this.instance.compactSplitThread.requestCompaction(r, s, getName()
1524                     + " requests major compaction; use default priority", null);
1525               } else {
1526                 this.instance.compactSplitThread.requestCompaction(r, s, getName()
1527                     + " requests major compaction; use configured priority",
1528                   this.majorCompactPriority, null);
1529               }
1530             }
1531           } catch (IOException e) {
1532             LOG.warn("Failed major compaction check on " + r, e);
1533           }
1534         }
1535       }
1536       iteration = (iteration == Long.MAX_VALUE) ? 0 : (iteration + 1);
1537     }
1538   }

1540   static class PeriodicMemstoreFlusher extends ScheduledChore {
1541     final HRegionServer server;
1542     final static int RANGE_OF_DELAY = 20000; //millisec
1543     final static int MIN_DELAY_TIME = 3000; //millisec
1544     public PeriodicMemstoreFlusher(int cacheFlushInterval, final HRegionServer server) {

1551       for (Region r : this.server.onlineRegions.values()) {

1553           continue;
1554         if (((HRegion)r).shouldFlush()) {
1555           FlushRequester requester = server.getFlushRequester();
1556           if (requester != null) {

1561             //is a balanced write-load on the regions in a table, we might end up
1562             //overwhelming the filesystem with too many flushes at once.
1563             requester.requestDelayedFlush(r, randomDelay, false);
1564           }
1565         }
1566       }

2705   /**
2706    * Close asynchronously a region, can be called from the master or internally by the regionserver
2707    * when stopping. If called from the master, the region will update the znode status.
2708    *
2709    * <p>
2710    * If an opening was in progress, this method will cancel it, but will not start a new close. The
2711    * coprocessors are not called in this case. A NotServingRegionException exception is thrown.
2712    * </p>
2713 
2714    * <p>
2715    *   If a close was in progress, this new request will be ignored, and an exception thrown.
2716    * </p>
2717    *
2718    * @param encodedName Region to close
2719    * @param abort True if we are aborting
2720    * @param crd details about closing region coordination-coordinated task
2721    * @return True if closed a region.
2722    * @throws NotServingRegionException if the region is not online
2723    * @throws RegionAlreadyInTransitionException if the region is already closing
2724    */
2725   protected boolean closeRegion(String encodedName, final boolean abort,
2726       CloseRegionCoordination.CloseRegionDetails crd, final ServerName sn)
2727       throws NotServingRegionException, RegionAlreadyInTransitionException {
2728     //Check for permissions to close.
2729     Region actualRegion = this.getFromOnlineRegions(encodedName);
2730     if ((actualRegion != null) && (actualRegion.getCoprocessorHost() != null)) {
2731       try {
2732         actualRegion.getCoprocessorHost().preClose(false);
2733       } catch (IOException exp) {
2734         LOG.warn("Unable to close region: the coprocessor launched an error ", exp);
2735         return false;
2736       }
2737     }
2738 
2739     final Boolean previous = this.regionsInTransitionInRS.putIfAbsent(encodedName.getBytes(),
2740         Boolean.FALSE);
2741 
2742     if (Boolean.TRUE.equals(previous)) {
2743       LOG.info("Received CLOSE for the region:" + encodedName + " , which we are already " +
2744           "trying to OPEN. Cancelling OPENING.");
2745       if (!regionsInTransitionInRS.replace(encodedName.getBytes(), previous, Boolean.FALSE)){
2746         // The replace failed. That should be an exceptional case, but theoretically it can happen.
2747         // We're going to try to do a standard close then.
2748         LOG.warn("The opening for region " + encodedName + " was done before we could cancel it." +
2749             " Doing a standard close now");
2750         return closeRegion(encodedName, abort, crd, sn);
2751       }
2752       // Let's get the region from the online region list again
2753       actualRegion = this.getFromOnlineRegions(encodedName);
2754       if (actualRegion == null) { // If already online, we still need to close it.
2755         LOG.info("The opening previously in progress has been cancelled by a CLOSE request.");
2756         // The master deletes the znode when it receives this exception.
2757         throw new RegionAlreadyInTransitionException("The region " + encodedName +
2758           " was opening but not yet served. Opening is cancelled.");
2759       }
2760     } else if (Boolean.FALSE.equals(previous)) {
2761       LOG.info("Received CLOSE for the region: " + encodedName +
2762         ", which we are already trying to CLOSE, but not completed yet");
2763       // The master will retry till the region is closed. We need to do this since
2764       // the region could fail to close somehow. If we mark the region closed in master
2765       // while it is not, there could be data loss.
2766       // If the region stuck in closing for a while, and master runs out of retries,
2767       // master will move the region to failed_to_close. Later on, if the region
2768       // is indeed closed, master can properly re-assign it.
2769       throw new RegionAlreadyInTransitionException("The region " + encodedName +
2770         " was already closing. New CLOSE request is ignored.");
2771     }
2772 
2773     if (actualRegion == null) {
2774       LOG.error("Received CLOSE for a region which is not online, and we're not opening.");
2775       this.regionsInTransitionInRS.remove(encodedName.getBytes());
2776       // The master deletes the znode when it receives this exception.
2777       throw new NotServingRegionException("The region " + encodedName +
2778           " is not online, and is not opening.");
2779     }
2780 
2781     CloseRegionHandler crh;
2782     final HRegionInfo hri = actualRegion.getRegionInfo();
2783     if (hri.isMetaRegion()) {
2784       crh = new CloseMetaHandler(this, this, hri, abort,
2785         csm.getCloseRegionCoordination(), crd);
2786     } else {
2787       crh = new CloseRegionHandler(this, this, hri, abort,
2788         csm.getCloseRegionCoordination(), crd, sn);
2789     }
2790     this.service.submit(crh);
2791     return true;
2792   }
