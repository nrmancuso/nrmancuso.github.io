1   package org.infinispan.cdi;
2   
3   import org.infinispan.cdi.util.defaultbean.DefaultBean;
4   import org.infinispan.cdi.util.logging.Log;
5   import org.infinispan.commons.logging.LogFactory;
6   import org.infinispan.configuration.cache.Configuration;
7   import org.infinispan.configuration.cache.ConfigurationBuilder;
8   
9   import javax.enterprise.inject.Produces;
10  
11  /**
12   * <p>The default embedded cache {@link Configuration} producer.</p>
13   *
14   * <p>The default embedded cache configuration can be overridden by creating a producer which produces the new default
15   * configuration. The configuration produced must have the scope {@linkplain javax.enterprise.context.Dependent Dependent}
16   * and the {@linkplain javax.enterprise.inject.Default Default} qualifier.</p>
17   *
18   * @author Pete Muir
19   * @author Kevin Pollet <kevin.pollet@serli.com> (C) 2011 SERLI
20   */
21  public class DefaultEmbeddedCacheConfigurationProducer {
22  
23     private static final Log log = LogFactory.getLog(DefaultEmbeddedCacheConfigurationProducer.class, Log.class);
24  
25     /**
26      * Produces the default embedded cache configuration.
27      *
28      * @return the default embedded cache configuration.
29      */
30     @Produces
31     @ConfigureCache
32     @DefaultBean(Configuration.class)
33     public Configuration getDefaultEmbeddedCacheConfiguration() {
34        return new ConfigurationBuilder().build();
35     }
36  }
