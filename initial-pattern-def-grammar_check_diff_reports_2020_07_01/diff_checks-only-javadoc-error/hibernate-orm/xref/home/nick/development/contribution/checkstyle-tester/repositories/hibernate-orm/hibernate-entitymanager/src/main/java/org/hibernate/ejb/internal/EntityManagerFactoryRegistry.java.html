24  package org.hibernate.ejb.internal;
25  
26  import java.util.HashSet;
27  import java.util.Set;
28  import java.util.concurrent.ConcurrentHashMap;
29  import javax.persistence.EntityManagerFactory;
30  
31  import org.jboss.logging.Logger;
32  
33  import org.hibernate.HibernateException;
34  import org.hibernate.ejb.AvailableSettings;
35  import org.hibernate.internal.CoreMessageLogger;
36  
37  /**
38   * An internal registry of all {@link org.hibernate.ejb.EntityManagerFactoryImpl} instances for the same
39   * classloader as this class.
40   *
41   * This registry is used for serialization/deserialization of entity managers.
42   *
43   * @author Scott Marlow
44   */
45  public class EntityManagerFactoryRegistry {
46  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(
47  			CoreMessageLogger.class,
48  			EntityManagerFactoryRegistry.class.getName()
49  	);
50  
51  	public static final EntityManagerFactoryRegistry INSTANCE = new EntityManagerFactoryRegistry();
52  
53  	private final ConcurrentHashMap<String, Set<EntityManagerFactory>> entityManagerFactoryMap = new ConcurrentHashMap<String, Set<EntityManagerFactory>>();
54  
55  	public EntityManagerFactoryRegistry() {
56  		LOG.debugf( "Initializing EntityManagerFactoryRegistry : %s", this );
57  	}
58  
59  	/**
60  	 * register the specified entity manager factory
61  	 *
62  	 * @param name to register the passed entity manager factory
63  	 * @param entityManagerFactory
64  	 */
65  	public void addEntityManagerFactory(String name, EntityManagerFactory entityManagerFactory) {
66  		LOG.debugf( "Registering EntityManagerFactory: %s ", name );
67  		if (name == null) { // allow unit tests that don't specify the pu name to succeed
68  			LOG.tracef( "not registering EntityManagerFactory because name is null");
69  			return;
70  		}
71  		Set<EntityManagerFactory> entityManagerFactorySet = new HashSet<EntityManagerFactory>();
72  		entityManagerFactorySet.add(entityManagerFactory);
73  		Set<EntityManagerFactory> previous = entityManagerFactoryMap.putIfAbsent( name, entityManagerFactorySet);
74  
75  		// if already added under 'name'.  Where 'name' could be session factory name, pu name or uuid (previous
76  		// will be null).  We will give a warning that an EntityManagerFactory is created with the same name
77  		// as is already used for a different EMF.  The best way to avoid the warning is to specify the AvailableSettings.SESSION_FACTORY_NAME
78  		// with a unique name.
79  		if (previous != null) {
80  			LOG.entityManagerFactoryAlreadyRegistered(name, AvailableSettings.ENTITY_MANAGER_FACTORY_NAME);
81  			boolean done = false;
82  			while( !done) {
83  				synchronized (previous) {
84  					if (entityManagerFactoryMap.get(name) == previous) {  // compare and set EMF if same
85  						previous.add(entityManagerFactory);
86  						done = true;
87  					}
88  					else {												// else it was removed or a new set added
89  						previous = entityManagerFactoryMap.get(name);	// get the set added by another thread
90  						if (null == previous) {							// or add it here if not
91  							entityManagerFactoryMap.putIfAbsent( name, new HashSet<EntityManagerFactory>());
92  							previous = entityManagerFactoryMap.get(name);// use the current set
93  						}
94  
95  					}
96  				}
97  			}
98  		}
99  	}
100 
101 	/**
102 	 * remove the specified entity manager factory from the EntityManagerFactoryRegistry
103 	 * @param name
104 	 * @param entityManagerFactory
105 	 * @throws HibernateException if the specified entity manager factory could not be found in the registry
106 	 */
107 	public void removeEntityManagerFactory(String name, EntityManagerFactory entityManagerFactory) throws HibernateException {
108 		LOG.debugf( "Remove: name=%s", name );
109 
110 		if (name == null) { // allow unit tests that don't specify the pu name to succeed
111 			LOG.tracef( "not removing EntityManagerFactory from registry because name is null");
112 			return;
113 		}
114 
115 		Set<EntityManagerFactory> entityManagerFactorySet = entityManagerFactoryMap.get(name);
116 		if (entityManagerFactorySet == null) {
117 			throw new HibernateException( "registry does not contain entity manager factory: " + name);
118 		}
119 		synchronized (entityManagerFactorySet) {
120 			boolean removed = entityManagerFactorySet.remove(entityManagerFactory);
121 
122 			if (entityManagerFactorySet.size() == 0) {
123 				entityManagerFactoryMap.remove( name );
124 			}
125 		}
126 	}
127 
128 	/**
129 	 * Lookup the specified entity manager factory by name
130 	 * @param name
131 	 * @return
132 	 * @throws HibernateException if entity manager factory is not found or if more than one
133 	 * entity manager factory was registered with name.
134 	 */
135 	public EntityManagerFactory getNamedEntityManagerFactory(String name) throws HibernateException {
136 		LOG.debugf( "Lookup: name=%s", name );
137 		Set<EntityManagerFactory> entityManagerFactorySet = entityManagerFactoryMap.get(name);
138 
139 		if (entityManagerFactorySet == null) {
140 			throw new HibernateException( "registry does not contain entity manager factory: " + name);
141 		}
142 
143 		if (entityManagerFactorySet.size() > 1) {
144 			throw new HibernateException( "registry contains more than one (" + entityManagerFactorySet.size()+ ") entity manager factories: " + name);
145 		}
146 		return entityManagerFactorySet.iterator().next();
147 	}
148 }
