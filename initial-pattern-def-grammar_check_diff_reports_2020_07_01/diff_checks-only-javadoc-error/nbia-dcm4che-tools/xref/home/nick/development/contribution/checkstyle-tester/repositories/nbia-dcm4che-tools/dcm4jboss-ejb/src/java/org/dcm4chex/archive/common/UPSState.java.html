39  package org.dcm4chex.archive.common;
40  
41  import java.util.Arrays;
42  
43  /**
44   * @author Gunter Zeilinger <gunterze@gmail.com>
45   * @version $Revision$ $Date:: xxxx-xx-xx $
46   * @since Apr 2, 2010
47   */
48  public class UPSState {
49  
50      private static final String[] ENUM = { "SCHEDULED", "IN PROGRESS",
51          "COMPLETED", "CANCELED" };
52  
53      public static final int SCHEDULED = 0;    
54      public static final int IN_PROGRESS = 1;
55      public static final int COMPLETED = 2;    
56      public static final int CANCELED = 3;
57  
58      public static final String toString(int value) {
59          return ENUM[value];
60      }
61  
62      public static final int toInt(String s) {        
63          final int index = Arrays.asList(ENUM).indexOf(s);
64          if (index == -1)
65              throw new IllegalArgumentException(s);
66          return index;
67      }
68  
69      public static int[] toInts(String[] ss) {
70          if (ss == null) {
71              return null;
72          }
73          int[] ret = new int[ss.length];
74          for (int i = 0; i < ret.length; i++) {
75              ret[i] = toInt(ss[i]);
76          }
77          return ret;
78      }
79  }
