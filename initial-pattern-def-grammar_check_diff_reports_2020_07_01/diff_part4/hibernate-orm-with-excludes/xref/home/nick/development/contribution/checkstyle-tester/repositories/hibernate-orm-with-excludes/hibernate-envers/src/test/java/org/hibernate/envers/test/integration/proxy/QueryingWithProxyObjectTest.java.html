16  /**
17   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
18   */
19  public class QueryingWithProxyObjectTest extends BaseEnversFunctionalTestCase {
20      private Integer id = null;
21  
22      @Override
23      protected Class<?>[] getAnnotatedClasses() {
24          return new Class[]{ StrTestEntity.class};
25      }
26  
27      @Test
28      @Priority(10)
29      public void initData() {
30          // Revision 1
31          getSession().getTransaction().begin();
32          StrTestEntity ste = new StrTestEntity("data");
33          getSession().persist(ste);
34          getSession().getTransaction().commit();
35          id = ste.getId();
36          getSession().close();
37      }
38  
39      @Test
40      @TestForIssue(jiraKey="HHH-4760")
41      @SuppressWarnings("unchecked")
42      public void testQueryingWithProxyObject() {
43          StrTestEntity originalSte = new StrTestEntity("data", id);
44          // Load the proxy instance
45          StrTestEntity proxySte = (StrTestEntity) getSession().load(StrTestEntity.class, id);
46  
47          Assert.assertTrue(getAuditReader().isEntityClassAudited(proxySte.getClass()));
48  
49          StrTestEntity ste = getAuditReader().find(proxySte.getClass(), proxySte.getId(), 1);
50          Assert.assertEquals(originalSte, ste);
51  
52          List<Number> revisions = getAuditReader().getRevisions(proxySte.getClass(), proxySte.getId());
53          Assert.assertEquals(Arrays.asList(1), revisions);
54  
55          List<StrTestEntity> entities = getAuditReader().createQuery().forEntitiesAtRevision(proxySte.getClass(), 1).getResultList();
56          Assert.assertEquals(Arrays.asList(originalSte), entities);
57  
58          ste = (StrTestEntity) getAuditReader().createQuery().forRevisionsOfEntity(proxySte.getClass(), true, false).getSingleResult();
59          Assert.assertEquals(originalSte, ste);
60  
61          ste = (StrTestEntity) getAuditReader().createQuery().forEntitiesModifiedAtRevision(proxySte.getClass(), 1).getSingleResult();
62          Assert.assertEquals(originalSte, ste);
63  
64          getSession().close();
65      }
66  }
