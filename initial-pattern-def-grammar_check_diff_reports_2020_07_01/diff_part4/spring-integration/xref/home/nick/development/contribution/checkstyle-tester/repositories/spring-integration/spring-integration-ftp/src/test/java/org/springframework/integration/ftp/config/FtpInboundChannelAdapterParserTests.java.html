65  @SpringJUnitConfig
66  @DirtiesContext
67  public class FtpInboundChannelAdapterParserTests {
68  
69  	@Autowired
70  	private SourcePollingChannelAdapter ftpInbound;
71  
72  	@Autowired
73  	private SourcePollingChannelAdapter simpleAdapterWithCachedSessions;
74  
75  	@Autowired
76  	private MessageChannel autoChannel;
77  
78  	@Autowired
79  	@Qualifier("autoChannel.adapter")
80  	private SourcePollingChannelAdapter autoChannelAdapter;
81  
82  	@Autowired
83  	private ApplicationContext context;
84  
85  	@Autowired
86  	private DirectoryScanner dirScanner;
87  
88  	@Autowired
89  	private MetadataStore metadataStore;
90  
91  	@Test
92  	public void testFtpInboundChannelAdapterComplete() throws Exception {
93  		assertThat(TestUtils.getPropertyValue(ftpInbound, "autoStartup", Boolean.class)).isFalse();
94  		PriorityBlockingQueue<?> blockingQueue =
95  				TestUtils.getPropertyValue(ftpInbound, "source.fileSource.toBeReceived", PriorityBlockingQueue.class);
96  		Comparator<?> comparator = blockingQueue.comparator();
97  		assertThat(comparator).isNotNull();
98  		assertThat(ftpInbound.getComponentName()).isEqualTo("ftpInbound");
99  		assertThat(ftpInbound.getComponentType()).isEqualTo("ftp:inbound-channel-adapter");
100 		assertThat(TestUtils.getPropertyValue(ftpInbound, "outputChannel")).isEqualTo(context.getBean("ftpChannel"));
101 		FtpInboundFileSynchronizingMessageSource inbound =
102 				(FtpInboundFileSynchronizingMessageSource) TestUtils.getPropertyValue(ftpInbound, "source");
103 
104 		assertThat(TestUtils.getPropertyValue(inbound, "fileSource.scanner")).isSameAs(dirScanner);
105 
106 		FtpInboundFileSynchronizer fisync =
107 				(FtpInboundFileSynchronizer) TestUtils.getPropertyValue(inbound, "synchronizer");
108 		assertThat(TestUtils.getPropertyValue(fisync, "remoteDirectoryExpression", Expression.class)
109 				.getExpressionString()).isEqualTo("'foo/bar'");
110 		assertThat(TestUtils.getPropertyValue(fisync, "localFilenameGeneratorExpression")).isNotNull();
111 		assertThat(TestUtils.getPropertyValue(fisync, "preserveTimestamp", Boolean.class)).isTrue();
112 		assertThat(TestUtils.getPropertyValue(fisync, "temporaryFileSuffix", String.class)).isEqualTo(".foo");
113 		assertThat(TestUtils.getPropertyValue(fisync, "remoteFileMetadataStore", MetadataStore.class))
114 				.isSameAs(this.metadataStore);
115 		assertThat(TestUtils.getPropertyValue(fisync, "metadataStorePrefix", String.class)).isEqualTo("testPrefix");
116 		String remoteFileSeparator = (String) TestUtils.getPropertyValue(fisync, "remoteFileSeparator");
117 		assertThat(remoteFileSeparator).isNotNull();
118 		assertThat(remoteFileSeparator).isEqualTo("");
119 
120 		FileListFilter<?> filter = TestUtils.getPropertyValue(fisync, "filter", FileListFilter.class);
121 		assertThat(filter).isNotNull();
122 		assertThat(filter).isInstanceOf(CompositeFileListFilter.class);
123 		Set<?> fileFilters = TestUtils.getPropertyValue(filter, "fileFilters", Set.class);
124 
125 		Iterator<?> filtersIterator = fileFilters.iterator();
126 		assertThat(filtersIterator.next()).isInstanceOf(FtpSimplePatternFileListFilter.class);
127 		assertThat(filtersIterator.next()).isInstanceOf(FtpPersistentAcceptOnceFileListFilter.class);
128 
129 		Object sessionFactory = TestUtils.getPropertyValue(fisync, "remoteFileTemplate.sessionFactory");
130 		assertThat(sessionFactory).isInstanceOf(DefaultFtpSessionFactory.class);
131 		FileListFilter<?> acceptAllFilter = context.getBean("acceptAllFilter", FileListFilter.class);
132 		assertThat(TestUtils.getPropertyValue(inbound, "fileSource.scanner.filter.fileFilters", Collection.class)
133 				.contains(acceptAllFilter)).isTrue();
134 		final AtomicReference<Method> genMethod = new AtomicReference<>();
135 		ReflectionUtils.doWithMethods(AbstractInboundFileSynchronizer.class, method -> {
136 			method.setAccessible(true);
137 			genMethod.set(method);
138 		}, method -> "generateLocalFileName".equals(method.getName()));
139 		assertThat(genMethod.get().invoke(fisync, "foo")).isEqualTo("FOO.afoo");
140 		assertThat(inbound.getMaxFetchSize()).isEqualTo(42);
141 	}
142 
143 	@Test
144 	public void cachingSessionFactory() {
145 		Object sessionFactory = TestUtils.getPropertyValue(simpleAdapterWithCachedSessions,
146 				"source.synchronizer.remoteFileTemplate.sessionFactory");
147 		assertThat(sessionFactory).isInstanceOf(CachingSessionFactory.class);
148 		FtpInboundFileSynchronizer fisync =
149 				TestUtils.getPropertyValue(simpleAdapterWithCachedSessions, "source.synchronizer",
150 						FtpInboundFileSynchronizer.class);
151 		String remoteFileSeparator = (String) TestUtils.getPropertyValue(fisync, "remoteFileSeparator");
152 		assertThat(remoteFileSeparator).isNotNull();
153 		assertThat(remoteFileSeparator).isEqualTo("/");
154 		assertThat(TestUtils.getPropertyValue(fisync, "remoteDirectoryExpression", Expression.class)
155 				.getExpressionString()).isEqualTo("foo/bar");
156 		assertThat(TestUtils.getPropertyValue(simpleAdapterWithCachedSessions, "source.maxFetchSize"))
157 				.isEqualTo(Integer.MIN_VALUE);
158 	}
159 
160 	@Test
161 	public void testAutoChannel() {
162 		assertThat(TestUtils.getPropertyValue(autoChannelAdapter, "outputChannel")).isSameAs(autoChannel);
163 	}
