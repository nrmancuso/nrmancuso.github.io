17  package org.springframework.integration.handler;
18  
19  import java.lang.reflect.InvocationTargetException;
20  import java.lang.reflect.Method;
21  import java.lang.reflect.Modifier;
22  import java.util.Map;
23  import java.util.Set;
24  
25  import org.apache.commons.logging.Log;
26  import org.apache.commons.logging.LogFactory;
27  
28  import org.springframework.beans.BeansException;
29  import org.springframework.beans.factory.BeanFactory;
30  import org.springframework.beans.factory.BeanFactoryAware;
31  import org.springframework.core.MethodIntrospector;
32  import org.springframework.integration.context.IntegrationContextUtils;
33  import org.springframework.messaging.Message;
34  import org.springframework.messaging.converter.MessageConverter;
35  import org.springframework.util.Assert;
36  import org.springframework.util.ClassUtils;
37  import org.springframework.util.ReflectionUtils;
38  
39  /**
40   * The {@link MessageProcessor} implementation for method invocation on the single method classes
41   * - functional interface implementations.
42   *
43   * @author Artem Bilan
44   * @author Gary Russell
45   *
46   * @since 5.0
47   */
48  public class LambdaMessageProcessor implements MessageProcessor<Object>, BeanFactoryAware {
49  
50  	private static final Log LOGGER = LogFactory.getLog(LambdaMessageProcessor.class);
51  
52  	private final Object target;
53  
54  	private final Method method;
55  
56  	private final Class<?> payloadType;
57  
58  	private final Class<?>[] parameterTypes;
59  
60  	private MessageConverter messageConverter;
61  
62  	public LambdaMessageProcessor(Object target, Class<?> payloadType) {
63  		Assert.notNull(target, "'target' must not be null");
64  		this.target = target;
65  
66  		Set<Method> methods =
67  				MethodIntrospector.selectMethods(target.getClass(),
68  						(ReflectionUtils.MethodFilter) methodCandidate ->
69  								methodCandidate.getDeclaringClass() != Object.class &&
70  										!methodCandidate.getDeclaringClass().getName()
71  												.equals("kotlin.jvm.internal.Lambda") &&
72  										!methodCandidate.isDefault() &&
73  										!Modifier.isStatic(methodCandidate.getModifiers()));
74  
75  		Assert.state(methods.size() == 1,
76  				"LambdaMessageProcessor is applicable for inline or lambda " +
77  						"classes with single method - functional interface implementations.");
78  
79  		this.method = methods.iterator().next();
80  		this.method.setAccessible(true);
81  		this.parameterTypes = this.method.getParameterTypes();
82  		this.payloadType = payloadType;
83  	}
84  
85  	@Override
86  	public void setBeanFactory(BeanFactory beanFactory) throws BeansException {
87  		this.messageConverter =
88  				beanFactory.getBean(IntegrationContextUtils.ARGUMENT_RESOLVER_MESSAGE_CONVERTER_BEAN_NAME,
89  						MessageConverter.class);
90  	}
91  
92  	@Override
93  	public Object processMessage(Message<?> message) {
94  		Object[] args = buildArgs(message);
95  
96  		try {
97  			return this.method.invoke(this.target, args);
98  		}
99  		catch (InvocationTargetException e) {
100 			if (e.getTargetException() instanceof ClassCastException) {
101 				LOGGER.error("Could not invoke the method '" + this.method + "' due to a class cast exception, " +
102 						"if using a lambda in the DSL, consider using an overloaded EIP method " +
103 						"that takes a Class<?> argument to explicitly  specify the type. " +
104 						"An example of when this often occurs is if the lambda is configured to " +
105 						"receive a Message<?> argument.", e.getCause());
106 			}
107 			throw new IllegalStateException(// NOSONAR lost stack trace
108 					"Could not invoke the method '" + this.method + "'", e.getCause());
109 		}
110 		catch (Exception e) {
111 			throw new IllegalStateException(
112 					"error occurred during processing message in 'LambdaMessageProcessor' for method [" +
113 							this.method + "]", e);
114 		}
115 	}
116 
117 	private Object[] buildArgs(Message<?> message) {
118 		Object[] args = new Object[this.parameterTypes.length];
119 		for (int i = 0; i < this.parameterTypes.length; i++) {
120 			Class<?> parameterType = this.parameterTypes[i];
121 			if (Message.class.isAssignableFrom(parameterType)) {
122 				args[i] = message;
123 			}
124 			else if (Map.class.isAssignableFrom(parameterType)) {
125 				if (message.getPayload() instanceof Map && this.parameterTypes.length == 1) {
126 					args[i] = message.getPayload();
127 				}
128 				else {
129 					args[i] = message.getHeaders();
130 				}
131 			}
132 			else {
133 				if (this.payloadType != null &&
134 						!ClassUtils.isAssignable(this.payloadType, message.getPayload().getClass())) {
135 					if (Message.class.isAssignableFrom(this.payloadType)) {
136 						args[i] = message;
137 					}
138 					else {
139 						args[i] = this.messageConverter.fromMessage(message, this.payloadType);
140 					}
141 				}
142 				else {
143 					args[i] = message.getPayload();
144 				}
145 			}
146 		}
147 		return args;
148 	}
149 
150 }
