51  public class CacheWritingMessageHandlerTests {
52  
53  	private static CacheFactoryBean cacheFactoryBean;
54  
55  	private static Region<Object, Object> region;
56  
57  	@BeforeClass
58  	public static void startUp() throws Exception {
59  		cacheFactoryBean = new CacheFactoryBean();
60  		cacheFactoryBean.afterPropertiesSet();
61  		Cache cache = (Cache) cacheFactoryBean.getObject();
62  		region = cache.createRegionFactory().setScope(Scope.LOCAL).create("sig-tests");
63  	}
64  
65  	@AfterClass
66  	public static void cleanUp() throws Exception {
67  		if (region != null) {
68  			region.close();
69  		}
70  		if (cacheFactoryBean != null) {
71  			cacheFactoryBean.destroy();
72  		}
73  	}
74  
75  	@Before
76  	public void prepare() {
77  		if (region != null) {
78  			region.clear();
79  		}
80  	}
81  
82  
83  	@Test
84  	public void mapPayloadWritesToCache() throws Exception {
85  		assertThat(region.size()).isEqualTo(0);
86  
87  		CacheWritingMessageHandler handler = new CacheWritingMessageHandler(region);
88  		handler.setBeanFactory(mock(BeanFactory.class));
89  		handler.afterPropertiesSet();
90  
91  		Map<String, String> map = new HashMap<String, String>();
92  		map.put("foo", "bar");
93  		Message<?> message = MessageBuilder.withPayload(map).build();
94  		handler.handleMessage(message);
95  		assertThat(region.size()).isEqualTo(1);
96  		assertThat(region.get("foo")).isEqualTo("bar");
97  	}
98  
99  	@Test
100 	public void ExpressionsWriteToCache() throws Exception {
101 		assertThat(region.size()).isEqualTo(0);
102 
103 		CacheWritingMessageHandler handler = new CacheWritingMessageHandler(region);
104 
105 		Map<String, String> expressions = new HashMap<String, String>();
106 		expressions.put("'foo'", "'bar'");
107 		expressions.put("payload.toUpperCase()", "headers['bar'].toUpperCase()");
108 		handler.setCacheEntries(expressions);
109 		handler.setBeanFactory(mock(BeanFactory.class));
110 		handler.afterPropertiesSet();
111 
112 		Message<?> message = MessageBuilder.withPayload("foo")
113 				.copyHeaders(Collections.singletonMap("bar", "bar"))
114 				.build();
115 		handler.handleMessage(message);
116 		assertThat(region.size()).isEqualTo(2);
117 		assertThat(region.get("FOO")).isEqualTo("BAR");
118 		assertThat(region.get("foo")).isEqualTo("bar");
119 
120 		handler.setCacheEntryExpressions(Collections.<Expression, Expression>singletonMap(new LiteralExpression("baz"),
121 				new ValueExpression<Long>(10L)));
122 
123 		handler.handleMessage(new GenericMessage<String>("test"));
124 		assertThat(region.size()).isEqualTo(3);
125 		assertThat(region.get("baz")).isEqualTo(10L);
126 	}
