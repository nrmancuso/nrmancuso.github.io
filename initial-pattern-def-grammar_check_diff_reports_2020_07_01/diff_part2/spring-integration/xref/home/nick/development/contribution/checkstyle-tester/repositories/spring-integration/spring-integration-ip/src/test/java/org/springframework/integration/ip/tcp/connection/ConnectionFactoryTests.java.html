100 	public void testObtainConnectionIds(AbstractServerConnectionFactory serverFactory) throws Exception {
101 		final List<IpIntegrationEvent> events =
102 				Collections.synchronizedList(new ArrayList<IpIntegrationEvent>());
103 		int expectedEvents = serverFactory instanceof TcpNetServerConnectionFactory
104 				? 7  // Listening, + OPEN, CLOSE, EXCEPTION for each side
105 				: 5; // Listening, + OPEN, CLOSE (but we *might* get exceptions, depending on timing).
106 		final CountDownLatch serverListeningLatch = new CountDownLatch(1);
107 		final CountDownLatch eventLatch = new CountDownLatch(expectedEvents);
108 		ApplicationEventPublisher publisher = event -> {
109 			LogFactory.getLog(this.getClass()).trace("Received: " + event);
110 			events.add((IpIntegrationEvent) event);
111 			if (event instanceof TcpConnectionServerListeningEvent) {
112 				serverListeningLatch.countDown();
113 			}
114 			eventLatch.countDown();
115 		};
116 		serverFactory.setBeanName("serverFactory");
117 		serverFactory.setApplicationEventPublisher(publisher);
118 		serverFactory = spy(serverFactory);
119 		final CountDownLatch serverConnectionInitLatch = new CountDownLatch(1);
120 		doAnswer(invocation -> {
121 			Object result = invocation.callRealMethod();
122 			serverConnectionInitLatch.countDown();
123 			return result;
124 		}).when(serverFactory).wrapConnection(any(TcpConnectionSupport.class));
125 		ThreadPoolTaskScheduler scheduler = new ThreadPoolTaskScheduler();
126 		scheduler.setPoolSize(10);
127 		scheduler.afterPropertiesSet();
128 		BeanFactory bf = mock(BeanFactory.class);
129 		when(bf.containsBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME)).thenReturn(true);
130 		when(bf.getBean(IntegrationContextUtils.TASK_SCHEDULER_BEAN_NAME, TaskScheduler.class)).thenReturn(scheduler);
131 		serverFactory.setBeanFactory(bf);
132 		TcpReceivingChannelAdapter adapter = new TcpReceivingChannelAdapter();
133 		adapter.setOutputChannel(new NullChannel());
134 		adapter.setConnectionFactory(serverFactory);
135 		adapter.start();
136 		assertThat(serverListeningLatch.await(10, TimeUnit.SECONDS)).as("Listening event not received").isTrue();
137 		assertThat(events.get(0)).isInstanceOf(TcpConnectionServerListeningEvent.class);
138 		assertThat(((TcpConnectionServerListeningEvent) events.get(0)).getPort()).isEqualTo(serverFactory.getPort());
139 		int port = serverFactory.getPort();
140 		TcpNetClientConnectionFactory clientFactory = new TcpNetClientConnectionFactory("localhost", port);
141 		clientFactory.registerListener(message -> false);
142 		clientFactory.setBeanName("clientFactory");
143 		clientFactory.setApplicationEventPublisher(publisher);
144 		clientFactory.start();
145 		TcpConnectionSupport client = clientFactory.getConnection();
146 		List<String> clients = clientFactory.getOpenConnectionIds();
147 		assertThat(clients.size()).isEqualTo(1);
148 		assertThat(clients.contains(client.getConnectionId())).isTrue();
149 		assertThat(serverConnectionInitLatch.await(10, TimeUnit.SECONDS)).as("Server connection failed to register")
150 				.isTrue();
151 		List<String> servers = serverFactory.getOpenConnectionIds();
152 		assertThat(servers.size()).isEqualTo(1);
153 		assertThat(serverFactory.closeConnection(servers.get(0))).isTrue();
154 		servers = serverFactory.getOpenConnectionIds();
155 		assertThat(servers.size()).isEqualTo(0);
156 		await().atMost(Duration.ofSeconds(10)).until(() -> clientFactory.getOpenConnectionIds().size() == 0);
157 		clients = clientFactory.getOpenConnectionIds();
158 		assertThat(clients.size()).isEqualTo(0);
159 		assertThat(eventLatch.await(10, TimeUnit.SECONDS)).isTrue();
160 		assertThat(events.size())
161 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
162 				.isGreaterThanOrEqualTo(expectedEvents);
163 
164 		FooEvent event = new FooEvent(client, "foo");
165 		client.publishEvent(event);
166 		assertThat(events.size())
167 				.as("Expected at least " + expectedEvents + " events; got: " + events.size() + " : " + events)
168 				.isGreaterThanOrEqualTo(expectedEvents + 1);
169 
170 		try {
171 			event = new FooEvent(mock(TcpConnectionSupport.class), "foo");
172 			client.publishEvent(event);
173 			fail("Expected exception");
174 		}
175 		catch (IllegalArgumentException e) {
176 			assertThat("Can only publish events with this as the source".equals(e.getMessage())).isTrue();
177 		}
178 
179 		SocketAddress address = serverFactory.getServerSocketAddress();
180 		if (address instanceof InetSocketAddress) {
181 			InetSocketAddress inetAddress = (InetSocketAddress) address;
182 			assertThat(inetAddress.getPort()).isEqualTo(port);
183 		}
184 		serverFactory.stop();
185 		scheduler.shutdown();
186 	}

200 	private void testEarlyClose(final AbstractServerConnectionFactory factory, String property,
201 			String message) throws Exception {
202 		factory.setApplicationEventPublisher(mock(ApplicationEventPublisher.class));
203 		factory.setBeanName("foo");
204 		factory.registerListener(mock(TcpListener.class));
205 		factory.afterPropertiesSet();
206 		Log logger = spy(TestUtils.getPropertyValue(factory, "logger", Log.class));
207 		new DirectFieldAccessor(factory).setPropertyValue("logger", logger);
208 		final CountDownLatch latch1 = new CountDownLatch(1);
209 		final CountDownLatch latch2 = new CountDownLatch(1);
210 		final CountDownLatch latch3 = new CountDownLatch(1);
211 		when(logger.isInfoEnabled()).thenReturn(true);
212 		when(logger.isDebugEnabled()).thenReturn(true);
213 		doAnswer(invocation -> {
214 			latch1.countDown();
215 			// wait until the stop nulls the channel
216 			latch2.await(10, TimeUnit.SECONDS);
217 			return null;
218 		}).when(logger).info(contains("Listening"));
219 		doAnswer(invocation -> {
220 			latch3.countDown();
221 			return null;
222 		}).when(logger).debug(contains(message));
223 		factory.start();
224 		assertThat(latch1.await(10, TimeUnit.SECONDS)).as("missing info log").isTrue();
225 		// stop on a different thread because it waits for the executor
226 		new SimpleAsyncTaskExecutor()
227 				.execute(factory::stop);
228 		int n = 0;
229 		DirectFieldAccessor accessor = new DirectFieldAccessor(factory);
230 		await("Stop was not invoked in time").atMost(Duration.ofSeconds(20))
231 				.until(() -> accessor.getPropertyValue(property) == null);
232 		latch2.countDown();
233 		assertThat(latch3.await(10, TimeUnit.SECONDS)).as("missing debug log").isTrue();
234 		String expected = "bean 'foo', port=" + factory.getPort() + message;
235 		ArgumentCaptor<String> captor = ArgumentCaptor.forClass(String.class);
236 		verify(logger, atLeast(1)).debug(captor.capture());
237 		assertThat(captor.getAllValues()).contains(expected);
238 		factory.stop();
239 	}

261 	private void healthCheckSuccess(AbstractServerConnectionFactory server, boolean fail) throws InterruptedException {
262 		CountDownLatch serverUp = new CountDownLatch(1);
263 		server.setApplicationEventPublisher(event -> {
264 			if (event instanceof TcpConnectionServerListeningEvent) {
265 				serverUp.countDown();
266 			}
267 		});
268 		server.setBeanFactory(mock(BeanFactory.class));
269 		AtomicReference<TcpConnection> connection = new AtomicReference<>();
270 		server.registerSender(conn -> {
271 			connection.set(conn);
272 		});
273 		AtomicInteger tested = new AtomicInteger();
274 		server.registerListener(msg -> {
275 			if (!(msg instanceof ErrorMessage)) {
276 				String payload = new String((byte[]) msg.getPayload());
277 				if (payload.equals("PING")) {
278 					tested.incrementAndGet();
279 					connection.get().send(new GenericMessage<>(fail ? "PANG" : "PONG"));
280 				}
281 				else {
282 					connection.get().send(msg);
283 				}
284 			}
285 			return false;
286 		});
287 		server.start();
288 		assertThat(serverUp.await(10, TimeUnit.SECONDS)).isTrue();
289 		TcpNetClientConnectionFactory clientFactory = new TcpNetClientConnectionFactory("localhost", server.getPort());
290 		clientFactory.setApplicationEventPublisher(event -> { });
291 		clientFactory.setConnectionTest(conn -> {
292 			CountDownLatch latch = new CountDownLatch(1);
293 			AtomicBoolean result = new AtomicBoolean();
294 			conn.registerTestListener(msg -> {
295 				if (Arrays.equals("PONG".getBytes(), (byte[]) msg.getPayload())) {
296 					result.set(true);
297 				}
298 				latch.countDown();
299 				return false;
300 			});
301 			conn.send(new GenericMessage<>("PING"));
302 			try {
303 				latch.await(10, TimeUnit.SECONDS);
304 			}
305 			catch (InterruptedException e) {
306 				Thread.currentThread().interrupt();
307 			}
308 			return result.get();
309 		});
310 		TcpOutboundGateway gateway = new TcpOutboundGateway();
311 		gateway.setRemoteTimeout(60000);
312 		gateway.setConnectionFactory(clientFactory);
313 		QueueChannel outputChannel = new QueueChannel();
314 		gateway.setOutputChannel(outputChannel);
315 		gateway.start();
316 		if (fail) {
317 			assertThatExceptionOfType(MessagingException.class).isThrownBy(() ->
318 				gateway.handleMessage(new GenericMessage<>("test1")))
319 					.withMessageContaining("Connection test failed for");
320 		}
321 		else {
322 			gateway.handleMessage(new GenericMessage<>("test1"));
323 			Message<?> received = outputChannel.receive(0);
324 			assertThat(received).isNotNull();
325 			assertThat(received.getPayload()).isEqualTo("test1".getBytes());
326 			gateway.handleMessage(new GenericMessage<>("test2"));
327 			received = outputChannel.receive(0);
328 			assertThat(received).isNotNull();
329 			assertThat(received.getPayload()).isEqualTo("test2".getBytes());
330 			assertThat(tested.get()).isEqualTo(1);
331 		}
332 		gateway.stop();
333 		server.stop();
334 	}
