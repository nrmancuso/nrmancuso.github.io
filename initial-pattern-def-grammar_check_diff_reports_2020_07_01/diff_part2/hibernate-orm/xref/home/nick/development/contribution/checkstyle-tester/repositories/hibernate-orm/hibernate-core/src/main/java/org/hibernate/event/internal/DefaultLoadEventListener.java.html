83  	/**
84  	 * Handle the given load event.
85  	 *
86  	 * @param event The load event to be handled.
87  	 * @throws HibernateException
88  	 */
89  	public void onLoad(LoadEvent event, LoadEventListener.LoadType loadType) throws HibernateException {
90  
91  		final SessionImplementor source = event.getSession();
92  
93  		EntityPersister persister;
94  		if ( event.getInstanceToLoad() != null ) {
95  			persister = source.getEntityPersister( null, event.getInstanceToLoad() ); //the load() which takes an entity does not pass an entityName
96  			event.setEntityClassName( event.getInstanceToLoad().getClass().getName() );
97  		}
98  		else {
99  			persister = source.getFactory().getEntityPersister( event.getEntityClassName() );
100 		}
101 
102 		if ( persister == null ) {
103 			throw new HibernateException(
104 					"Unable to locate persister: " +
105 					event.getEntityClassName()
106 				);
107 		}
108 
109 		final Class idClass = persister.getIdentifierType().getReturnedClass();
110 		if ( idClass != null && ! idClass.isInstance( event.getEntityId() ) ) {
111 			// we may have the kooky jpa requirement of allowing find-by-id where
112 			// "id" is the "simple pk value" of a dependent objects parent.  This
113 			// is part of its generally goofy "derived identity" "feature"
114 			if ( persister.getEntityMetamodel().getIdentifierProperty().isEmbedded() ) {
115 				final EmbeddedComponentType dependentIdType =
116 						(EmbeddedComponentType) persister.getEntityMetamodel().getIdentifierProperty().getType();
117 				if ( dependentIdType.getSubtypes().length == 1 ) {
118 					final Type singleSubType = dependentIdType.getSubtypes()[0];
119 					if ( singleSubType.isEntityType() ) {
120 						final EntityType dependentParentType = (EntityType) singleSubType;
121 						final Type dependentParentIdType = dependentParentType.getIdentifierOrUniqueKeyType( source.getFactory() );
122 						if ( dependentParentIdType.getReturnedClass().isInstance( event.getEntityId() ) ) {
123 							// yep that's what we have...
124 							loadByDerivedIdentitySimplePkValue(
125 									event,
126 									loadType,
127 									persister,
128 									dependentIdType,
129 									source.getFactory().getEntityPersister( dependentParentType.getAssociatedEntityName() )
130 							);
131 							return;
132 						}
133 					}
134 				}
135 			}
136 			throw new TypeMismatchException(
137 					"Provided id of the wrong type for class " + persister.getEntityName() + ". Expected: " + idClass + ", got " + event.getEntityId().getClass()
138 			);
139 		}
140 
141 		final  EntityKey keyToLoad = source.generateEntityKey( event.getEntityId(), persister );
142 
143 		try {
144 			if ( loadType.isNakedEntityReturned() ) {
145 				//do not return a proxy!
146 				//(this option indicates we are initializing a proxy)
147 				event.setResult( load(event, persister, keyToLoad, loadType) );
148 			}
149 			else {
150 				//return a proxy if appropriate
151 				if ( event.getLockMode() == LockMode.NONE ) {
152 					event.setResult( proxyOrLoad(event, persister, keyToLoad, loadType) );
153 				}
154 				else {
155 					event.setResult( lockAndLoad(event, persister, keyToLoad, loadType, source) );
156 				}
157 			}
158 		}
159 		catch(HibernateException e) {
160 			LOG.unableToLoadCommand( e );
161 			throw e;
162 		}
163 	}

165 	private void loadByDerivedIdentitySimplePkValue(
166 			LoadEvent event,
167 			LoadEventListener.LoadType options,
168 			EntityPersister dependentPersister,
169 			EmbeddedComponentType dependentIdType,
170 			EntityPersister parentPersister) {
171 		final EntityKey parentEntityKey = event.getSession().generateEntityKey( event.getEntityId(), parentPersister );
172 		final Object parent = doLoad( event, parentPersister, parentEntityKey, options );
173 
174 		final Serializable dependent = (Serializable) dependentIdType.instantiate( parent, event.getSession() );
175 		dependentIdType.setPropertyValues( dependent, new Object[] {parent}, dependentPersister.getEntityMode() );
176 		final EntityKey dependentEntityKey = event.getSession().generateEntityKey( dependent, dependentPersister );
177 		event.setEntityId( dependent );
178 
179 		event.setResult( doLoad( event, dependentPersister, dependentEntityKey, options ) );
180 	}

182 	/**
183 	 * Performs the load of an entity.
184 	 *
185 	 * @param event The initiating load request event
186 	 * @param persister The persister corresponding to the entity to be loaded
187 	 * @param keyToLoad The key of the entity to be loaded
188 	 * @param options The defined load options
189 	 * @return The loaded entity.
190 	 * @throws HibernateException
191 	 */
192 	protected Object load(
193 		final LoadEvent event,
194 		final EntityPersister persister,
195 		final EntityKey keyToLoad,
196 		final LoadEventListener.LoadType options) {
197 
198 		if ( event.getInstanceToLoad() != null ) {
199 			if ( event.getSession().getPersistenceContext().getEntry( event.getInstanceToLoad() ) != null ) {
200 				throw new PersistentObjectException(
201 						"attempted to load into an instance that was already associated with the session: " +
202 						MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() )
203 					);
204 			}
205 			persister.setIdentifier( event.getInstanceToLoad(), event.getEntityId(), event.getSession() );
206 		}
207 
208 		Object entity = doLoad(event, persister, keyToLoad, options);
209 
210 		boolean isOptionalInstance = event.getInstanceToLoad() != null;
211 
212 		if ( !options.isAllowNulls() || isOptionalInstance ) {
213 			if ( entity == null ) {
214 				event.getSession().getFactory().getEntityNotFoundDelegate().handleEntityNotFound( event.getEntityClassName(), event.getEntityId() );
215 			}
216 		}
217 
218 		if ( isOptionalInstance && entity != event.getInstanceToLoad() ) {
219 			throw new NonUniqueObjectException( event.getEntityId(), event.getEntityClassName() );
220 		}
221 
222 		return entity;
223 	}

225 	/**
226 	 * Based on configured options, will either return a pre-existing proxy,
227 	 * generate a new proxy, or perform an actual load.
228 	 *
229 	 * @param event The initiating load request event
230 	 * @param persister The persister corresponding to the entity to be loaded
231 	 * @param keyToLoad The key of the entity to be loaded
232 	 * @param options The defined load options
233 	 * @return The result of the proxy/load operation.
234 	 */
235 	protected Object proxyOrLoad(
236 		final LoadEvent event,
237 		final EntityPersister persister,
238 		final EntityKey keyToLoad,
239 		final LoadEventListener.LoadType options) {
240 
241 		if ( LOG.isTraceEnabled() ) {
242 			LOG.tracev( "Loading entity: {0}",
243 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() ) );
244 		}
245 
246         // this class has no proxies (so do a shortcut)
247         if (!persister.hasProxy()) {
248 			return load(event, persister, keyToLoad, options);
249 		}
250 
251         final PersistenceContext persistenceContext = event.getSession().getPersistenceContext();
252 
253 		// look for a proxy
254         Object proxy = persistenceContext.getProxy(keyToLoad);
255         if (proxy != null) {
256 			return returnNarrowedProxy(event, persister, keyToLoad, options, persistenceContext, proxy);
257 		}
258 
259         if (options.isAllowProxyCreation()) {
260 			return createProxyIfNecessary(event, persister, keyToLoad, options, persistenceContext);
261 		}
262 
263         // return a newly loaded object
264         return load(event, persister, keyToLoad, options);
265 	}

267 	/**
268 	 * Given a proxy, initialize it and/or narrow it provided either
269 	 * is necessary.
270 	 *
271 	 * @param event The initiating load request event
272 	 * @param persister The persister corresponding to the entity to be loaded
273 	 * @param keyToLoad The key of the entity to be loaded
274 	 * @param options The defined load options
275 	 * @param persistenceContext The originating session
276 	 * @param proxy The proxy to narrow
277 	 * @return The created/existing proxy
278 	 */
279 	private Object returnNarrowedProxy(
280 			final LoadEvent event,
281 			final EntityPersister persister,
282 			final EntityKey keyToLoad,
283 			final LoadEventListener.LoadType options,
284 			final PersistenceContext persistenceContext,
285 			final Object proxy) {
286 		LOG.trace( "Entity proxy found in session cache" );
287 		LazyInitializer li = ( (HibernateProxy) proxy ).getHibernateLazyInitializer();
288 		if ( li.isUnwrap() ) {
289 			return li.getImplementation();
290 		}
291 		Object impl = null;
292 		if ( !options.isAllowProxyCreation() ) {
293 			impl = load( event, persister, keyToLoad, options );
294 			if ( impl == null ) {
295 				event.getSession().getFactory().getEntityNotFoundDelegate().handleEntityNotFound( persister.getEntityName(), keyToLoad.getIdentifier());
296 			}
297 		}
298 		return persistenceContext.narrowProxy( proxy, persister, keyToLoad, impl );
299 	}

301 	/**
302 	 * If there is already a corresponding proxy associated with the
303 	 * persistence context, return it; otherwise create a proxy, associate it
304 	 * with the persistence context, and return the just-created proxy.
305 	 *
306 	 * @param event The initiating load request event
307 	 * @param persister The persister corresponding to the entity to be loaded
308 	 * @param keyToLoad The key of the entity to be loaded
309 	 * @param options The defined load options
310 	 * @param persistenceContext The originating session
311 	 * @return The created/existing proxy
312 	 */
313 	private Object createProxyIfNecessary(
314 			final LoadEvent event,
315 			final EntityPersister persister,
316 			final EntityKey keyToLoad,
317 			final LoadEventListener.LoadType options,
318 			final PersistenceContext persistenceContext) {
319 		Object existing = persistenceContext.getEntity( keyToLoad );
320 		if ( existing != null ) {
321 			// return existing object or initialized proxy (unless deleted)
322 			LOG.trace( "Entity found in session cache" );
323 			if ( options.isCheckDeleted() ) {
324 				EntityEntry entry = persistenceContext.getEntry( existing );
325 				Status status = entry.getStatus();
326 				if ( status == Status.DELETED || status == Status.GONE ) {
327 					return null;
328 				}
329 			}
330 			return existing;
331 		}
332 		LOG.trace( "Creating new proxy for entity" );
333 		// return new uninitialized proxy
334 		Object proxy = persister.createProxy( event.getEntityId(), event.getSession() );
335 		persistenceContext.getBatchFetchQueue().addBatchLoadableEntityKey( keyToLoad );
336 		persistenceContext.addProxy( keyToLoad, proxy );
337 		return proxy;
338 	}

340 	/**
341 	 * If the class to be loaded has been configured with a cache, then lock
342 	 * given id in that cache and then perform the load.
343 	 *
344 	 * @param event The initiating load request event
345 	 * @param persister The persister corresponding to the entity to be loaded
346 	 * @param keyToLoad The key of the entity to be loaded
347 	 * @param options The defined load options
348 	 * @param source The originating session
349 	 * @return The loaded entity
350 	 * @throws HibernateException
351 	 */
352 	protected Object lockAndLoad(
353 			final LoadEvent event,
354 			final EntityPersister persister,
355 			final EntityKey keyToLoad,
356 			final LoadEventListener.LoadType options,
357 			final SessionImplementor source) {
358 		SoftLock lock = null;
359 		final CacheKey ck;
360 		if ( persister.hasCache() ) {
361 			ck = source.generateCacheKey(
362 					event.getEntityId(),
363 					persister.getIdentifierType(),
364 					persister.getRootEntityName()
365 			);
366 			lock = persister.getCacheAccessStrategy().lockItem( ck, null );
367 		}
368 		else {
369 			ck = null;
370 		}
371 
372 		Object entity;
373 		try {
374 			entity = load(event, persister, keyToLoad, options);
375 		}
376 		finally {
377 			if ( persister.hasCache() ) {
378 				persister.getCacheAccessStrategy().unlockItem( ck, lock );
379 			}
380 		}
381 
382 		return event.getSession().getPersistenceContext().proxyFor( persister, keyToLoad, entity );
383 	}

386 	/**
387 	 * Coordinates the efforts to load a given entity.  First, an attempt is
388 	 * made to load the entity from the session-level cache.  If not found there,
389 	 * an attempt is made to locate it in second-level cache.  Lastly, an
390 	 * attempt is made to load it directly from the datasource.
391 	 *
392 	 * @param event The load event
393 	 * @param persister The persister for the entity being requested for load
394 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
395 	 * @param options The load options.
396 	 * @return The loaded entity, or null.
397 	 */
398 	protected Object doLoad(
399 			final LoadEvent event,
400 			final EntityPersister persister,
401 			final EntityKey keyToLoad,
402 			final LoadEventListener.LoadType options) {
403 
404 		final boolean traceEnabled = LOG.isTraceEnabled();
405 		if ( traceEnabled ) LOG.tracev( "Attempting to resolve: {0}",
406 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() ) );
407 
408 		Object entity = loadFromSessionCache( event, keyToLoad, options );
409 		if ( entity == REMOVED_ENTITY_MARKER ) {
410 			LOG.debug( "Load request found matching entity in context, but it is scheduled for removal; returning null" );
411 			return null;
412 		}
413 		if ( entity == INCONSISTENT_RTN_CLASS_MARKER ) {
414 			LOG.debug( "Load request found matching entity in context, but the matched entity was of an inconsistent return type; returning null" );
415 			return null;
416 		}
417 		if ( entity != null ) {
418 			if (traceEnabled) LOG.tracev("Resolved object in session cache: {0}",
419 						MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() ) );
420 			return entity;
421 		}
422 
423 		entity = loadFromSecondLevelCache(event, persister, options);
424 		if ( entity != null ) {
425 			if ( traceEnabled ) LOG.tracev( "Resolved object in second-level cache: {0}",
426 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() ) );
427 		}
428 		else {
429 			if ( traceEnabled ) LOG.tracev( "Object not resolved in any cache: {0}",
430 					MessageHelper.infoString( persister, event.getEntityId(), event.getSession().getFactory() ) );
431 			entity = loadFromDatasource(event, persister, keyToLoad, options);
432 		}
433 		
434 		if (entity != null && persister.hasNaturalIdentifier()) {
435 			event.getSession().getPersistenceContext().getNaturalIdHelper().cacheNaturalIdCrossReferenceFromLoad(
436 					persister,
437 					event.getEntityId(),
438 					event.getSession().getPersistenceContext().getNaturalIdHelper().extractNaturalIdValues( entity, persister )
439 			);
440 		}
441 
442 
443 		return entity;
444 	}

446 	/**
447 	 * Performs the process of loading an entity from the configured
448 	 * underlying datasource.
449 	 *
450 	 * @param event The load event
451 	 * @param persister The persister for the entity being requested for load
452 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
453 	 * @param options The load options.
454 	 * @return The object loaded from the datasource, or null if not found.
455 	 */
456 	protected Object loadFromDatasource(
457 			final LoadEvent event,
458 			final EntityPersister persister,
459 			final EntityKey keyToLoad,
460 			final LoadEventListener.LoadType options) {
461 		final SessionImplementor source = event.getSession();
462 		Object entity = persister.load(
463 				event.getEntityId(),
464 				event.getInstanceToLoad(),
465 				event.getLockOptions(),
466 				source
467 		);
468 		
469 		if ( event.isAssociationFetch() && source.getFactory().getStatistics().isStatisticsEnabled() ) {
470 			source.getFactory().getStatisticsImplementor().fetchEntity( event.getEntityClassName() );
471 		}
472 
473 		return entity;
474 	}

476 	/**
477 	 * Attempts to locate the entity in the session-level cache.
478 	 * <p/>
479 	 * If allowed to return nulls, then if the entity happens to be found in
480 	 * the session cache, we check the entity type for proper handling
481 	 * of entity hierarchies.
482 	 * <p/>
483 	 * If checkDeleted was set to true, then if the entity is found in the
484 	 * session-level cache, it's current status within the session cache
485 	 * is checked to see if it has previously been scheduled for deletion.
486 	 *
487 	 * @param event The load event
488 	 * @param keyToLoad The EntityKey representing the entity to be loaded.
489 	 * @param options The load options.
490 	 * @return The entity from the session-level cache, or null.
491 	 * @throws HibernateException Generally indicates problems applying a lock-mode.
492 	 */
493 	protected Object loadFromSessionCache(
494 			final LoadEvent event,
495 			final EntityKey keyToLoad,
496 			final LoadEventListener.LoadType options) throws HibernateException {
497 
498 		SessionImplementor session = event.getSession();
499 		Object old = session.getEntityUsingInterceptor( keyToLoad );
500 
501 		if ( old != null ) {
502 			// this object was already loaded
503 			EntityEntry oldEntry = session.getPersistenceContext().getEntry( old );
504 			if ( options.isCheckDeleted() ) {
505 				Status status = oldEntry.getStatus();
506 				if ( status == Status.DELETED || status == Status.GONE ) {
507 					return REMOVED_ENTITY_MARKER;
508 				}
509 			}
510 			if ( options.isAllowNulls() ) {
511 				final EntityPersister persister = event.getSession().getFactory().getEntityPersister( keyToLoad.getEntityName() );
512 				if ( ! persister.isInstance( old ) ) {
513 					return INCONSISTENT_RTN_CLASS_MARKER;
514 				}
515 			}
516 			upgradeLock( old, oldEntry, event.getLockOptions(), event.getSession() );
517 		}
518 
519 		return old;
520 	}

522 	/**
523 	 * Attempts to load the entity from the second-level cache.
524 	 *
525 	 * @param event The load event
526 	 * @param persister The persister for the entity being requested for load
527 	 * @param options The load options.
528 	 * @return The entity from the second-level cache, or null.
529 	 */
530 	protected Object loadFromSecondLevelCache(
531 			final LoadEvent event,
532 			final EntityPersister persister,
533 			final LoadEventListener.LoadType options) {
534 
535 		final SessionImplementor source = event.getSession();
536 		final boolean useCache = persister.hasCache()
537 				&& source.getCacheMode().isGetEnabled()
538 				&& event.getLockMode().lessThan(LockMode.READ);
539 
540 		if ( ! useCache ) {
541 			// we can't use cache here
542 			return null;
543 		}
544 
545 		final SessionFactoryImplementor factory = source.getFactory();
546 		final CacheKey ck = source.generateCacheKey(
547 				event.getEntityId(),
548 				persister.getIdentifierType(),
549 				persister.getRootEntityName()
550 		);
551 
552 		final Object ce = CacheHelper.fromSharedCache( source, ck, persister.getCacheAccessStrategy() );
553 		if ( factory.getStatistics().isStatisticsEnabled() ) {
554 			if ( ce == null ) {
555 				factory.getStatisticsImplementor().secondLevelCacheMiss(
556 						persister.getCacheAccessStrategy().getRegion().getName()
557 				);
558 			}
559 			else {
560 				factory.getStatisticsImplementor().secondLevelCacheHit(
561 						persister.getCacheAccessStrategy().getRegion().getName()
562 				);
563 			}
564 		}
565 
566 		if ( ce == null ) {
567 			// nothing was found in cache
568 			return null;
569 		}
570 
571 		CacheEntry entry = (CacheEntry) persister.getCacheEntryStructure().destructure( ce, factory );
572 		Object entity = convertCacheEntryToEntity( entry, event.getEntityId(), persister, event );
573 		
574 		if ( !persister.isInstance( entity ) ) {
575 			throw new WrongClassException(
576 					"loaded object was of wrong class " + entity.getClass(),
577 					event.getEntityId(),
578 					persister.getEntityName()
579 				);
580 		}
581 		
582 		return entity;
583 	}

585 	private Object convertCacheEntryToEntity(
586 			CacheEntry entry,
587 			Serializable entityId,
588 			EntityPersister persister,
589 			LoadEvent event) {
590 
591 		final EventSource session = event.getSession();
592 		final SessionFactoryImplementor factory = session.getFactory();
593 		final EntityPersister subclassPersister;
594 
595 		if ( LOG.isTraceEnabled() ) {
596 			LOG.tracef(
597 					"Converting second-level cache entry [%s] into entity : %s",
598 					entry,
599 					MessageHelper.infoString( persister, entityId, factory )
600 			);
601 		}
602 
603 		final Object entity;
604 		if ( entry.isReferenceEntry() ) {
605 			final Object optionalObject = event.getInstanceToLoad();
606 			if ( optionalObject != null ) {
607 				throw new HibernateException(
608 						String.format(
609 								"Attempt to load entity [%s] from cache using provided object instance, but cache " +
610 										"is storing references",
611 								MessageHelper.infoString( persister, entityId, factory )
612 						)
613 				);
614 			}
615 
616 			ReferenceCacheEntryImpl referenceCacheEntry = (ReferenceCacheEntryImpl) entry;
617 			entity = referenceCacheEntry.getReference();
618 			if ( entity == null ) {
619 				throw new IllegalStateException(
620 						"Reference cache entry contained null : " + MessageHelper.infoString( persister, entityId, factory )
621 				);
622 			}
623 			subclassPersister = referenceCacheEntry.getSubclassPersister();
624 		}
625 		else {
626 			subclassPersister = factory.getEntityPersister( entry.getSubclass() );
627 			final Object optionalObject = event.getInstanceToLoad();
628 			entity = optionalObject == null
629 					? session.instantiate( subclassPersister, entityId )
630 					: optionalObject;
631 		}
632 
633 		// make it circular-reference safe
634 		final EntityKey entityKey = session.generateEntityKey( entityId, subclassPersister );
635 		TwoPhaseLoad.addUninitializedCachedEntity(
636 				entityKey,
637 				entity,
638 				subclassPersister,
639 				LockMode.NONE,
640 				entry.areLazyPropertiesUnfetched(),
641 				entry.getVersion(),
642 				session
643 		);
644 
645 		final PersistenceContext persistenceContext = session.getPersistenceContext();
646 		final Object[] values;
647 		final Object version;
648 		final boolean isReadOnly;
649 		if ( entry.isReferenceEntry() ) {
650 			values = null;
651 			version = null;
652 			isReadOnly = true;
653 		}
654 		else {
655 			final Type[] types = subclassPersister.getPropertyTypes();
656 			// initializes the entity by (desired) side-effect
657 			values = ( (StandardCacheEntryImpl) entry).assemble(
658 					entity, entityId, subclassPersister, session.getInterceptor(), session
659 			);
660 			if ( ( (StandardCacheEntryImpl) entry ).isDeepCopyNeeded() ) {
661 				TypeHelper.deepCopy(
662 						values,
663 						types,
664 						subclassPersister.getPropertyUpdateability(),
665 						values,
666 						session
667 				);
668 			}
669 			version = Versioning.getVersion( values, subclassPersister );
670 			LOG.tracef( "Cached Version : %s", version );
671 
672 			final Object proxy = persistenceContext.getProxy( entityKey );
673 			if ( proxy != null ) {
674 				// there is already a proxy for this impl
675 				// only set the status to read-only if the proxy is read-only
676 				isReadOnly = ( ( HibernateProxy ) proxy ).getHibernateLazyInitializer().isReadOnly();
677 			}
678 			else {
679 				isReadOnly = session.isDefaultReadOnly();
680 			}
681 		}
682 
683 		persistenceContext.addEntry(
684 				entity,
685 				( isReadOnly ? Status.READ_ONLY : Status.MANAGED ),
686 				values,
687 				null,
688 				entityId,
689 				version,
690 				LockMode.NONE,
691 				true,
692 				subclassPersister,
693 				false,
694 				entry.areLazyPropertiesUnfetched()
695 		);
696 		subclassPersister.afterInitialize( entity, entry.areLazyPropertiesUnfetched(), session );
697 		persistenceContext.initializeNonLazyCollections();
698 
699 		//PostLoad is needed for EJB3
700 		PostLoadEvent postLoadEvent = new PostLoadEvent( session )
701 				.setEntity( entity )
702 				.setId( entityId )
703 				.setPersister( persister );
704 
705 		for ( PostLoadEventListener listener : postLoadEventListeners( session ) ) {
706 			listener.onPostLoad( postLoadEvent );
707 		}
708 
709 		return entity;
710 	}

712 	private Object assembleCacheEntry(
713 			final StandardCacheEntryImpl entry,
714 			final Serializable id,
715 			final EntityPersister persister,
716 			final LoadEvent event) throws HibernateException {
717 
718 		final Object optionalObject = event.getInstanceToLoad();
719 		final EventSource session = event.getSession();
720 		final SessionFactoryImplementor factory = session.getFactory();
721 
722 		if ( LOG.isTraceEnabled() ) {
723 			LOG.tracev( "Assembling entity from second-level cache: {0}",
724 					MessageHelper.infoString( persister, id, factory ) );
725 		}
726 
727 		EntityPersister subclassPersister = factory.getEntityPersister( entry.getSubclass() );
728 		Object result = optionalObject == null ?
729 				session.instantiate( subclassPersister, id ) : optionalObject;
730 
731 		// make it circular-reference safe
732 		final EntityKey entityKey = session.generateEntityKey( id, subclassPersister );
733 		TwoPhaseLoad.addUninitializedCachedEntity(
734 				entityKey,
735 				result,
736 				subclassPersister,
737 				LockMode.NONE,
738 				entry.areLazyPropertiesUnfetched(),
739 				entry.getVersion(),
740 				session
741 			);
742 
743 		Type[] types = subclassPersister.getPropertyTypes();
744 		Object[] values = entry.assemble( result, id, subclassPersister, session.getInterceptor(), session ); // intializes result by side-effect
745 		TypeHelper.deepCopy(
746 				values,
747 				types,
748 				subclassPersister.getPropertyUpdateability(),
749 				values,
750 				session
751 		);
752 
753 		Object version = Versioning.getVersion( values, subclassPersister );
754 		LOG.tracev( "Cached Version: {0}", version );
755 
756 		final PersistenceContext persistenceContext = session.getPersistenceContext();
757 		boolean isReadOnly = session.isDefaultReadOnly();
758 		if ( persister.isMutable() ) {
759 			Object proxy = persistenceContext.getProxy( entityKey );
760 			if ( proxy != null ) {
761 				// there is already a proxy for this impl
762 				// only set the status to read-only if the proxy is read-only
763 				isReadOnly = ( ( HibernateProxy ) proxy ).getHibernateLazyInitializer().isReadOnly();
764 			}
765 		}
766 		else {
767 			isReadOnly = true;
768 		}
769 		persistenceContext.addEntry(
770 				result,
771 				( isReadOnly ? Status.READ_ONLY : Status.MANAGED ),
772 				values,
773 				null,
774 				id,
775 				version,
776 				LockMode.NONE,
777 				true,
778 				subclassPersister,
779 				false,
780 				entry.areLazyPropertiesUnfetched()
781 		);
782 		subclassPersister.afterInitialize( result, entry.areLazyPropertiesUnfetched(), session );
783 		persistenceContext.initializeNonLazyCollections();
784 		// upgrade the lock if necessary:
785 		//lock(result, lockMode);
786 
787 		//PostLoad is needed for EJB3
788 		//TODO: reuse the PostLoadEvent...
789 		PostLoadEvent postLoadEvent = new PostLoadEvent( session )
790 				.setEntity( result )
791 				.setId( id )
792 				.setPersister( persister );
793 
794 		for ( PostLoadEventListener listener : postLoadEventListeners( session ) ) {
795 			listener.onPostLoad( postLoadEvent );
796 		}
797 
798 		return result;
799 	}
800 
801 	private Iterable<PostLoadEventListener> postLoadEventListeners(EventSource session) {
802 		return session
803 				.getFactory()
804 				.getServiceRegistry()
805 				.getService( EventListenerRegistry.class )
806 				.getEventListenerGroup( EventType.POST_LOAD )
807 				.listeners();
808 	}
