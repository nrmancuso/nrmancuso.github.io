49  	@Override
50  	protected void cleanupTest() {
51  		ParentWithCollection dummyParent = createParent( "dummyParent" );
52  		dummyParent.newChildren( createCollection() );
53  		Child dummyChild = dummyParent.addChild( "dummyChild" );
54  		Session s = openSession();
55  		Transaction tx = s.beginTransaction();
56  		List children = s.createCriteria( dummyChild.getClass() ).list();
57  		List parents = s.createCriteria( dummyParent.getClass() ).list();
58  		for ( Iterator it = parents.iterator(); it.hasNext(); ) {
59  			ParentWithCollection parent = ( ParentWithCollection ) it.next();
60  			parent.clearChildren();
61  			s.delete( parent );
62  		}
63  		for ( Iterator it = children.iterator(); it.hasNext(); ) {
64  			s.delete( it.next() );
65  		}
66  		tx.commit();
67  		s.close();
68  	}
