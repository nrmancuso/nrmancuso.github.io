229 		String dir = "sftpSource/";
230 		long modified = setModifiedOnSource1();
231 		File secondRemote = new File(getSourceRemoteDirectory(), "sftpSource2.txt");
232 		secondRemote.setLastModified(System.currentTimeMillis() - 1_000_000);
233 		this.inboundMGetRecursive.send(new GenericMessage<Object>(dir + "*"));
234 		Message<?> result = this.output.receive(1000);
235 		assertThat(result).isNotNull();
236 		List<File> localFiles = (List<File>) result.getPayload();
237 		assertThat(localFiles).hasSize(3);
238 
239 		for (File file : localFiles) {
240 			assertThat(file.getPath().replaceAll(Matcher.quoteReplacement(File.separator), "/")).contains(dir);
241 			if (file.getPath().contains("localTarget1")) {
242 				assertPreserved(modified, file);
243 			}
244 		}
245 		assertThat(localFiles.get(2).getPath().replaceAll(Matcher.quoteReplacement(File.separator), "/"))
246 				.contains(dir + "subSftpSource");
247 
248 		File secondTarget = new File(getTargetLocalDirectory() + File.separator + "sftpSource", "localTarget2.txt");
249 		ByteArrayOutputStream remoteContents = new ByteArrayOutputStream();
250 		ByteArrayOutputStream localContents = new ByteArrayOutputStream();
251 		FileUtils.copyFile(secondRemote, remoteContents);
252 		FileUtils.copyFile(secondTarget, localContents);
253 		String localAsString = new String(localContents.toByteArray());
254 		assertThat(localAsString).isEqualTo(new String(remoteContents.toByteArray()));
255 		long oldLastModified = secondRemote.lastModified();
256 		FileUtils.copyInputStreamToFile(new ByteArrayInputStream("junk".getBytes()), secondRemote);
257 		long newLastModified = secondRemote.lastModified();
258 		secondRemote.setLastModified(oldLastModified);
259 		this.inboundMGetRecursive.send(new GenericMessage<Object>(dir + "*"));
260 		this.output.receive(0);
261 		localContents = new ByteArrayOutputStream();
262 		FileUtils.copyFile(secondTarget, localContents);
263 		assertThat(new String(localContents.toByteArray())).isEqualTo(localAsString);
264 		secondRemote.setLastModified(newLastModified);
265 		this.inboundMGetRecursive.send(new GenericMessage<Object>(dir + "*"));
266 		this.output.receive(0);
267 		localContents = new ByteArrayOutputStream();
268 		FileUtils.copyFile(secondTarget, localContents);
269 		assertThat(new String(localContents.toByteArray())).isEqualTo("junk");
270 		// restore the remote file contents
271 		FileUtils.copyInputStreamToFile(new ByteArrayInputStream(localAsString.getBytes()), secondRemote);
272 	}

717 			return SftpServerOutboundTests.sessionFactory();

722 			return new SftpRemoteFileTemplate(sf);

727 		}

729 		@Bean
730 		public ApplicationEventListeningMessageProducer events() {
731 			ApplicationEventListeningMessageProducer producer = new ApplicationEventListeningMessageProducer();
732 			producer.setEventTypes(ApacheMinaSftpEvent.class);
733 			producer.setOutputChannel(eventChannel());
734 			return producer;
735 		}
736 
737 		@Bean
738 		public MessageChannel eventChannel() {
739 			return (msg, timeout) -> {
740 				if (this.latch != null) {
741 					if (this.events.size() > 0 || msg.getPayload() instanceof SessionOpenedEvent) {
742 						this.events.add((ApacheMinaSftpEvent) msg.getPayload());
743 						if (msg.getPayload() instanceof SessionClosedEvent) {
744 							this.latch.countDown();
745 						}
746 					}
747 				}
748 				return true;
749 			};
750 		}
