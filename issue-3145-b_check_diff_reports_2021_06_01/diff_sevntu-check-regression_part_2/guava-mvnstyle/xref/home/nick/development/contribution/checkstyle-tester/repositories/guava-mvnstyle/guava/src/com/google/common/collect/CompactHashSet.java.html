47  /**
48   * CompactHashSet is an implementation of a Set. All optional operations (adding and removing) are
49   * supported. The elements can be any objects.
50   *
51   * <p>{@code contains(x)}, {@code add(x)} and {@code remove(x)}, are all (expected and amortized)
52   * constant time operations. Expected in the hashtable sense (depends on the hash function doing a
53   * good job of distributing the elements to the buckets to a distribution not far from uniform), and
54   * amortized since some operations can trigger a hash table resize.
55   *
56   * <p>Unlike {@code java.util.HashSet}, iteration is only proportional to the actual {@code size()},
57   * which is optimal, and <i>not</i> the size of the internal hashtable, which could be much larger
58   * than {@code size()}. Furthermore, this structure only depends on a fixed number of arrays; {@code
59   * add(x)} operations <i>do not</i> create objects for the garbage collector to deal with, and for
60   * every element added, the garbage collector will have to traverse {@code 1.5} references on
61   * average, in the marking phase, not {@code 5.0} as in {@code java.util.HashSet}.
62   *
63   * <p>If there are no removals, then {@link #iterator iteration} order is the same as insertion
64   * order. Any removal invalidates any ordering guarantees.
65   *
66   * <p>This class should not be assumed to be universally superior to {@code java.util.HashSet}.
67   * Generally speaking, this class reduces object allocation and memory consumption at the price of
68   * moderately increased constant factors of CPU. Only use this class when there is a specific reason
69   * to prioritize memory over CPU.
70   *
71   * @author Dimitris Andreou
72   * @author Jon Noack
73   */
74  @GwtIncompatible // not worth using in GWT for now
75  class CompactHashSet<E> extends AbstractSet<E> implements Serializable {
76    // TODO(user): cache all field accesses in local vars

83    /**
84     * Creates a <i>mutable</i> {@code CompactHashSet} instance containing the elements of the given
85     * collection in unspecified order.
86     *
87     * @param collection the elements that the set should contain
88     * @return a new {@code CompactHashSet} containing those elements (minus duplicates)
89     */
90    public static <E> CompactHashSet<E> create(Collection<? extends E> collection) {
91      CompactHashSet<E> set = createWithExpectedSize(collection.size());
92      set.addAll(collection);
93      return set;
94    }

96    /**
97     * Creates a <i>mutable</i> {@code CompactHashSet} instance containing the given elements in
98     * unspecified order.
99     *
100    * @param elements the elements that the set should contain
101    * @return a new {@code CompactHashSet} containing those elements (minus duplicates)
102    */
103   @SafeVarargs
104   public static <E> CompactHashSet<E> create(E... elements) {
105     CompactHashSet<E> set = createWithExpectedSize(elements.length);
106     Collections.addAll(set, elements);
107     return set;
108   }

110   /**
111    * Creates a {@code CompactHashSet} instance, with a high enough "initial capacity" that it
112    * <i>should</i> hold {@code expectedSize} elements without growth.
113    *
114    * @param expectedSize the number of elements you expect to add to the returned set
115    * @return a new, empty {@code CompactHashSet} with enough capacity to hold {@code expectedSize}
116    *     elements without resizing
117    * @throws IllegalArgumentException if {@code expectedSize} is negative
118    */
119   public static <E> CompactHashSet<E> createWithExpectedSize(int expectedSize) {
120     return new CompactHashSet<>(expectedSize);

123   /**
124    * The hashtable. Its values are indexes to the elements and entries arrays.
125    *
126    * <p>Currently, the UNSET value means "null pointer", and any positive value x is the actual
127    * index + 1.
128    *
129    * <p>Its size must be a power of two.
130    */
131   @MonotonicNonNull private transient Object table;
132 
133   /**
134    * Contains the logical entries, in the range of [0, size()). The high bits of each int are the
135    * part of the smeared hash of the element not covered by the hashtable mask, whereas the low bits
136    * are the "next" pointer (pointing to the next entry in the bucket chain), which will always be
137    * less than or equal to the hashtable mask.
138    *
139    * <pre>
140    * hash  = aaaaaaaa
141    * mask  = 0000ffff
142    * next  = 0000bbbb
143    * entry = aaaabbbb
144    * </pre>
145    *
146    * <p>The pointers in [size(), entries.length) are all "null" (UNSET).
147    */
148   private transient int @MonotonicNonNull [] entries;
149 
150   /**
151    * The elements contained in the set, in the range of [0, size()). The elements in [size(),
152    * elements.length) are all {@code null}.
153    */
154   @VisibleForTesting transient Object @MonotonicNonNull [] elements;
155 
156   /**
157    * Keeps track of metadata like the number of hash table bits and modifications of this data
158    * structure (to make it possible to throw ConcurrentModificationException in the iterator). Note
159    * that we choose not to make this volatile, so we do less of a "best effort" to track such
160    * errors, for better performance.
161    */
162   private transient int metadata;
163 
164   /** The number of elements contained in the set. */
165   private transient int size;

181   /** Pseudoconstructor for serialization support. */
182   void init(int expectedSize) {
183     Preconditions.checkArgument(expectedSize >= 0, "Expected size must be >= 0");
184 
185     // Save expectedSize for use in allocArrays()
186     this.metadata = Math.max(1, Math.min(CompactHashing.MAX_SIZE, expectedSize));
187   }

189   /** Returns whether arrays need to be allocated. */
190   @VisibleForTesting
191   boolean needsAllocArrays() {
192     return table == null;
193   }
194 
195   /** Handle lazy allocation of arrays. */
196   @CanIgnoreReturnValue
197   int allocArrays() {
198     Preconditions.checkState(needsAllocArrays(), "Arrays already allocated");
199 
200     int expectedSize = metadata;
201     int buckets = CompactHashing.tableSize(expectedSize);
202     this.table = CompactHashing.createTable(buckets);
203     setHashTableMask(buckets - 1);
204 
205     this.entries = new int[expectedSize];
206     this.elements = new Object[expectedSize];
207 
208     return expectedSize;
209   }
210 
211   /** Stores the hash table mask as the number of bits needed to represent an index. */
212   private void setHashTableMask(int mask) {
213     int hashTableBits = Integer.SIZE - Integer.numberOfLeadingZeros(mask);
214     metadata =
215         CompactHashing.maskCombine(metadata, hashTableBits, CompactHashing.HASH_TABLE_BITS_MASK);
216   }
217 
218   /** Gets the hash table mask using the stored number of hash table bits. */
219   private int hashTableMask() {
220     return (1 << (metadata & CompactHashing.HASH_TABLE_BITS_MASK)) - 1;
221   }
222 
223   void incrementModCount() {
224     metadata += CompactHashing.MODIFICATION_COUNT_INCREMENT;
225   }

227   @CanIgnoreReturnValue
228   @Override
229   public boolean add(@Nullable E object) {
230     if (needsAllocArrays()) {
231       allocArrays();
232     }
233     int[] entries = this.entries;
234     Object[] elements = this.elements;
235 
236     int newEntryIndex = this.size; // current size, and pointer to the entry to be appended
237     int newSize = newEntryIndex + 1;
238     int hash = smearedHash(object);
239     int mask = hashTableMask();
240     int tableIndex = hash & mask;
241     int next = CompactHashing.tableGet(table, tableIndex);
242     if (next == UNSET) { // uninitialized bucket
243       if (newSize > mask) {
244         // Resize and add new entry
245         mask = resizeTable(mask, CompactHashing.newCapacity(mask), hash, newEntryIndex);
246       } else {
247         CompactHashing.tableSet(table, tableIndex, newEntryIndex + 1);
248       }
249     } else {
250       int entryIndex;
251       int entry;
252       int hashPrefix = CompactHashing.getHashPrefix(hash, mask);
253       do {
254         entryIndex = next - 1;
255         entry = entries[entryIndex];
256         if (CompactHashing.getHashPrefix(entry, mask) == hashPrefix
257             && Objects.equal(object, elements[entryIndex])) {
258           return false;
259         }
260         next = CompactHashing.getNext(entry, mask);
261       } while (next != UNSET);
262       if (newSize > mask) {
263         // Resize and add new entry
264         mask = resizeTable(mask, CompactHashing.newCapacity(mask), hash, newEntryIndex);
265       } else {
266         entries[entryIndex] = CompactHashing.maskCombine(entry, newEntryIndex + 1, mask);
267       }
268     }
269     resizeMeMaybe(newSize);
270     insertEntry(newEntryIndex, object, hash, mask);
271     this.size = newSize;
272     incrementModCount();
273     return true;
274   }
275 
276   /**
277    * Creates a fresh entry with the specified object at the specified position in the entry arrays.
278    */
279   void insertEntry(int entryIndex, @Nullable E object, int hash, int mask) {
280     this.entries[entryIndex] = CompactHashing.maskCombine(hash, UNSET, mask);
281     this.elements[entryIndex] = object;
282   }
283 
284   /** Resizes the entries storage if necessary. */
285   private void resizeMeMaybe(int newSize) {
286     int entriesSize = entries.length;
287     if (newSize > entriesSize) {
288       // 1.5x but round up to nearest odd (this is optimal for memory consumption on Android)
289       int newCapacity =
290           Math.min(CompactHashing.MAX_SIZE, (entriesSize + Math.max(1, entriesSize >>> 1)) | 1);
291       if (newCapacity != entriesSize) {
292         resizeEntries(newCapacity);
293       }
294     }
295   }
296 
297   /**
298    * Resizes the internal entries array to the specified capacity, which may be greater or less than
299    * the current capacity.
300    */
301   void resizeEntries(int newCapacity) {
302     this.entries = Arrays.copyOf(entries, newCapacity);
303     this.elements = Arrays.copyOf(elements, newCapacity);
304   }
305 
306   @CanIgnoreReturnValue
307   private int resizeTable(int mask, int newCapacity, int targetHash, int targetEntryIndex) {
308     Object newTable = CompactHashing.createTable(newCapacity);
309     int newMask = newCapacity - 1;
310 
311     if (targetEntryIndex != UNSET) {
312       // Add target first; it must be last in the chain because its entry hasn't yet been created
313       CompactHashing.tableSet(newTable, targetHash & newMask, targetEntryIndex + 1);
314     }
315 
316     Object table = this.table;
317     int[] entries = this.entries;
318 
319     // Loop over current hashtable
320     for (int tableIndex = 0; tableIndex <= mask; tableIndex++) {
321       int next = CompactHashing.tableGet(table, tableIndex);
322       while (next != UNSET) {
323         int entryIndex = next - 1;
324         int entry = entries[entryIndex];
325 
326         // Rebuild hash using entry hashPrefix and tableIndex ("hashSuffix")
327         int hash = CompactHashing.getHashPrefix(entry, mask) | tableIndex;
328 
329         int newTableIndex = hash & newMask;
330         int newNext = CompactHashing.tableGet(newTable, newTableIndex);
331         CompactHashing.tableSet(newTable, newTableIndex, next);
332         entries[entryIndex] = CompactHashing.maskCombine(hash, newNext, newMask);
333 
334         next = CompactHashing.getNext(entry, mask);
335       }
336     }
337 
338     this.table = newTable;
339     setHashTableMask(newMask);
340     return newMask;
341   }

343   @Override
344   public boolean contains(@Nullable Object object) {
345     if (needsAllocArrays()) {
346       return false;
347     }
348     int hash = smearedHash(object);
349     int mask = hashTableMask();
350     int next = CompactHashing.tableGet(table, hash & mask);
351     if (next == UNSET) {
352       return false;
353     }
354     int hashPrefix = CompactHashing.getHashPrefix(hash, mask);
355     do {
356       int entryIndex = next - 1;
357       int entry = entries[entryIndex];
358       if (CompactHashing.getHashPrefix(entry, mask) == hashPrefix
359           && Objects.equal(object, elements[entryIndex])) {
360         return true;
361       }
362       next = CompactHashing.getNext(entry, mask);
363     } while (next != UNSET);
364     return false;
365   }

367   @CanIgnoreReturnValue
368   @Override
369   public boolean remove(@Nullable Object object) {
370     if (needsAllocArrays()) {
371       return false;
372     }
373     int mask = hashTableMask();
374     int index =
375         CompactHashing.remove(
376             object, /* value= */ null, mask, table, entries, elements, /* values= */ null);
377     if (index == -1) {
378       return false;
379     }
380 
381     moveLastEntry(index, mask);
382     size--;
383     incrementModCount();
384 
385     return true;
386   }
387 
388   /**
389    * Moves the last entry in the entry array into {@code dstIndex}, and nulls out its old position.
390    */
391   void moveLastEntry(int dstIndex, int mask) {
392     int srcIndex = size() - 1;
393     if (dstIndex < srcIndex) {
394       // move last entry to deleted spot
395       @Nullable Object object = elements[srcIndex];
396       elements[dstIndex] = object;
397       elements[srcIndex] = null;
398 
399       // move the last entry to the removed spot, just like we moved the element
400       entries[dstIndex] = entries[srcIndex];
401       entries[srcIndex] = 0;
402 
403       // also need to update whoever's "next" pointer was pointing to the last entry place
404       int tableIndex = smearedHash(object) & mask;
405       int next = CompactHashing.tableGet(table, tableIndex);
406       int srcNext = srcIndex + 1;
407       if (next == srcNext) {
408         // we need to update the root pointer
409         CompactHashing.tableSet(table, tableIndex, dstIndex + 1);
410       } else {
411         // we need to update a pointer in an entry
412         int entryIndex;
413         int entry;
414         do {
415           entryIndex = next - 1;
416           entry = entries[entryIndex];
417           next = CompactHashing.getNext(entry, mask);
418         } while (next != srcNext);
419         // here, entries[entryIndex] points to the old entry location; update it
420         entries[entryIndex] = CompactHashing.maskCombine(entry, dstIndex + 1, mask);
421       }
422     } else {
423       elements[dstIndex] = null;
424       entries[dstIndex] = 0;
425     }
426   }

428   int firstEntryIndex() {
429     return isEmpty() ? -1 : 0;
430   }
431 
432   int getSuccessor(int entryIndex) {
433     return (entryIndex + 1 < size) ? entryIndex + 1 : -1;
434   }

436   /**
437    * Updates the index an iterator is pointing to after a call to remove: returns the index of the
438    * entry that should be looked at after a removal on indexRemoved, with indexBeforeRemove as the
439    * index that *was* the next entry that would be looked at.
440    */
441   int adjustAfterRemove(int indexBeforeRemove, @SuppressWarnings("unused") int indexRemoved) {
442     return indexBeforeRemove - 1;
443   }

445   @Override
446   public Iterator<E> iterator() {
447     return new Iterator<E>() {
448       int expectedMetadata = metadata;
449       int currentIndex = firstEntryIndex();
450       int indexToRemove = -1;
451 
452       @Override
453       public boolean hasNext() {
454         return currentIndex >= 0;
455       }
456 
457       @SuppressWarnings("unchecked") // known to be Es
458       @Override
459       public E next() {
460         checkForConcurrentModification();
461         if (!hasNext()) {
462           throw new NoSuchElementException();
463         }
464         indexToRemove = currentIndex;
465         E result = (E) elements[currentIndex];
466         currentIndex = getSuccessor(currentIndex);
467         return result;
468       }
469 
470       @Override
471       public void remove() {
472         checkForConcurrentModification();
473         checkRemove(indexToRemove >= 0);
474         incrementExpectedModCount();
475         CompactHashSet.this.remove(elements[indexToRemove]);
476         currentIndex = adjustAfterRemove(currentIndex, indexToRemove);
477         indexToRemove = -1;
478       }
479 
480       void incrementExpectedModCount() {
481         expectedMetadata += CompactHashing.MODIFICATION_COUNT_INCREMENT;
482       }
483 
484       private void checkForConcurrentModification() {
485         if (metadata != expectedMetadata) {
486           throw new ConcurrentModificationException();
487         }
488       }
489     };
490   }

492   @Override
493   public Spliterator<E> spliterator() {
494     if (needsAllocArrays()) {
495       return Spliterators.spliterator(new Object[0], Spliterator.DISTINCT | Spliterator.ORDERED);
496     }
497     return Spliterators.spliterator(elements, 0, size, Spliterator.DISTINCT | Spliterator.ORDERED);
498   }

500   @SuppressWarnings("unchecked") // known to be Es
501   @Override
502   public void forEach(Consumer<? super E> action) {
503     checkNotNull(action);
504     for (int i = firstEntryIndex(); i >= 0; i = getSuccessor(i)) {
505       action.accept((E) elements[i]);
506     }
507   }

527   @CanIgnoreReturnValue
528   @Override
529   public <T> T[] toArray(T[] a) {
530     if (needsAllocArrays()) {
531       if (a.length > 0) {
532         a[0] = null;
533       }
534       return a;
535     }
536     return ObjectArrays.toArrayImpl(elements, 0, size, a);
537   }

539   /**
540    * Ensures that this {@code CompactHashSet} has the smallest representation in memory, given its
541    * current size.
542    */
543   public void trimToSize() {
544     if (needsAllocArrays()) {
545       return;
546     }
547     int size = this.size;
548     if (size < entries.length) {
549       resizeEntries(size);
550     }
551     int minimumTableSize = CompactHashing.tableSize(size);
552     int mask = hashTableMask();
553     if (minimumTableSize < mask) { // smaller table size will always be less than current mask
554       resizeTable(mask, minimumTableSize, UNSET, UNSET);
555     }
556   }
557 
558   @Override
559   public void clear() {
560     if (needsAllocArrays()) {
561       return;
562     }
563     incrementModCount();
564     Arrays.fill(elements, 0, size, null);
565     CompactHashing.tableClear(table);
566     Arrays.fill(entries, 0, size, 0);
567     this.size = 0;
568   }

570   private void writeObject(ObjectOutputStream stream) throws IOException {
571     stream.defaultWriteObject();
572     stream.writeInt(size);
573     for (int i = firstEntryIndex(); i >= 0; i = getSuccessor(i)) {
574       stream.writeObject(elements[i]);
575     }
576   }
577 
578   @SuppressWarnings("unchecked")
579   private void readObject(ObjectInputStream stream) throws IOException, ClassNotFoundException {
580     stream.defaultReadObject();
581     int elementCount = stream.readInt();
582     if (elementCount < 0) {
583       throw new InvalidObjectException("Invalid size: " + elementCount);
584     }
585     init(elementCount);
586     for (int i = 0; i < elementCount; i++) {
587       E element = (E) stream.readObject();
588       add(element);
589     }
590   }
