38  /**
39   * Websphere deployment tool that augments the ejbjar task.
40   * Searches for the websphere specific deployment descriptors and
41   * adds them to the final ejb jar file. Websphere has two specific descriptors for session
42   * beans:
43   * <ul>
44   *    <li>ibm-ejb-jar-bnd.xmi</li>
45   *    <li>ibm-ejb-jar-ext.xmi</li>
46   * </ul>
47   * and another two for container managed entity beans:
48   * <ul>
49   *    <li>Map.mapxmi</li>
50   *    <li>Schema.dbxmi</li>
51   * </ul>
52   * In terms of WebSphere, the generation of container code and stubs is
53   * called <code>deployment</code>. This step can be performed by the websphere
54   * element as part of the jar generation process. If the switch
55   * <code>ejbdeploy</code> is on, the ejbdeploy tool from the websphere toolset
56   * is called for every ejb-jar. Unfortunately, this step only works, if you
57   * use the ibm jdk. Otherwise, the rmic (called by ejbdeploy) throws a
58   * ClassFormatError. Be sure to switch ejbdeploy off, if run ant with
59   * sun jdk.
60   *
61   */
62  public class WebsphereDeploymentTool extends GenericDeploymentTool {

366     /** {@inheritDoc}. */
367     protected DescriptorHandler getDescriptorHandler(File srcDir) {
368         DescriptorHandler handler = new DescriptorHandler(getTask(), srcDir);
369         // register all the DTDs, both the ones that are known and
370         // any supplied by the user
371         handler.registerDTD(PUBLICID_EJB11, ejb11DTD);
372 
373         for (Iterator i = getConfig().dtdLocations.iterator(); i.hasNext();) {
374             EjbJar.DTDLocation dtdLocation = (EjbJar.DTDLocation) i.next();
375 
376             handler.registerDTD(dtdLocation.getPublicId(), dtdLocation.getLocation());
377         }
378 
379         return handler;
380     }

404     /**
405      * Add any vendor specific files which should be included in the EJB Jar.
406      * @param ejbFiles a hashtable entryname -> file.
407      * @param baseName a prefix to use.
408      */
409     protected void addVendorFiles(Hashtable ejbFiles, String baseName) {
410 
411         String ddPrefix = (usingBaseJarName() ? "" : baseName);
412         String dbPrefix = (dbVendor == null) ? "" : dbVendor + "-";
413 
414         // Get the Extensions document
415         File websphereEXT = new File(getConfig().descriptorDir, ddPrefix + WAS_EXT);
416 
417         if (websphereEXT.exists()) {
418             ejbFiles.put(META_DIR + WAS_EXT,
419                 websphereEXT);
420         } else {
421             log("Unable to locate websphere extensions. "
422                 + "It was expected to be in "
423                 + websphereEXT.getPath(), Project.MSG_VERBOSE);
424         }
425 
426         File websphereBND = new File(getConfig().descriptorDir, ddPrefix + WAS_BND);
427 
428         if (websphereBND.exists()) {
429             ejbFiles.put(META_DIR + WAS_BND,
430                 websphereBND);
431         } else {
432             log("Unable to locate websphere bindings. "
433                 + "It was expected to be in "
434                 + websphereBND.getPath(), Project.MSG_VERBOSE);
435         }
436 
437         if (!newCMP) {
438             log("The old method for locating CMP files has been DEPRECATED.",
439                 Project.MSG_VERBOSE);
440             log("Please adjust your websphere descriptor and set "
441                 + "newCMP=\"true\" to use the new CMP descriptor "
442                 + "inclusion mechanism. ", Project.MSG_VERBOSE);
443         } else {
444             // We attempt to put in the MAP and Schema files of CMP beans
445             try {
446                 // Add the Map file
447                 File websphereMAP = new File(getConfig().descriptorDir,
448                     ddPrefix + dbPrefix + WAS_CMP_MAP);
449 
450                 if (websphereMAP.exists()) {
451                     ejbFiles.put(META_DIR + WAS_CMP_MAP,
452                         websphereMAP);
453                 } else {
454                     log("Unable to locate the websphere Map: "
455                         + websphereMAP.getPath(), Project.MSG_VERBOSE);
456                 }
457 
458                 File websphereSchema = new File(getConfig().descriptorDir,
459                     ddPrefix + dbPrefix + WAS_CMP_SCHEMA);
460 
461                 if (websphereSchema.exists()) {
462                     ejbFiles.put(META_DIR + SCHEMA_DIR + WAS_CMP_SCHEMA,
463                         websphereSchema);
464                 } else {
465                     log("Unable to locate the websphere Schema: "
466                         + websphereSchema.getPath(), Project.MSG_VERBOSE);
467                 }
468                 // Theres nothing else to see here...keep moving sonny
469             } catch (Exception e) {
470                 String msg = "Exception while adding Vendor specific files: "
471                     + e.toString();
472 
473                 throw new BuildException(msg, e);
474             }
475         }
476     }

514             options.append(" -quiet");
515         }
516 
517         if (novalidate) {
518             options.append(" -novalidate");

520 
521         if (nowarn) {
522             options.append(" -nowarn");
523         }
524 
525         if (noinform) {
526             options.append(" -noinform");
527         }
528 
529         if (trace) {
530             options.append(" -trace");
531         }
532

534             options.append(" -35");
535         }
536 
537         if (rmicOptions != null) {
538             options.append(" -rmic \"").append(rmicOptions).append("\"");
539         }
540 
541         return options.toString();

545     /**
546      * Helper method invoked by execute() for each websphere jar to be built.
547      * Encapsulates the logic of constructing a java task for calling
548      * websphere.ejbdeploy and executing it.
549      *
550      * @param sourceJar java.io.File representing the source (EJB1.1) jarfile.
551      * @param destJar java.io.File representing the destination, websphere
552      *      jarfile.
553      */
554     private void buildWebsphereJar(File sourceJar, File destJar) {
555         try {
556             if (ejbdeploy) {
557                 Java javaTask = new Java(getTask());
558                 // Set the JvmArgs
559                 javaTask.createJvmarg().setValue("-Xms64m");
560                 javaTask.createJvmarg().setValue("-Xmx128m");
561 
562                 // Set the Environment variable
563                 Environment.Variable var = new Environment.Variable();
564 
565                 var.setKey("websphere.lib.dir");
566                 File libdir = new File(websphereHome, "lib");
567                 var.setValue(libdir.getAbsolutePath());
568                 javaTask.addSysproperty(var);
569 
570                 // Set the working directory
571                 javaTask.setDir(websphereHome);
572 
573                 // Set the Java class name
574                 javaTask.setTaskName("ejbdeploy");
575                 javaTask.setClassname("com.ibm.etools.ejbdeploy.EJBDeploy");
576 
577                 javaTask.createArg().setValue(sourceJar.getPath());
578                 javaTask.createArg().setValue(tempdir);
579                 javaTask.createArg().setValue(destJar.getPath());
580                 javaTask.createArg().setLine(getOptions());
581                 if (getCombinedClasspath() != null
582                     && getCombinedClasspath().toString().length() > 0) {
583                     javaTask.createArg().setValue("-cp");
584                     javaTask.createArg().setValue(getCombinedClasspath().toString());
585                 }
586 
587                 Path classpath = wasClasspath;
588 
589                 if (classpath == null) {
590                     classpath = getCombinedClasspath();
591                 }
592 
593                 javaTask.setFork(true);
594                 if (classpath != null) {
595                     javaTask.setClasspath(classpath);
596                 }
597 
598                 log("Calling websphere.ejbdeploy for " + sourceJar.toString(),
599                     Project.MSG_VERBOSE);
600 
601                 javaTask.execute();
602             }
603         } catch (Exception e) {
604             // Have to catch this because of the semantics of calling main()
605             String msg = "Exception while calling ejbdeploy. Details: " + e.toString();
606 
607             throw new BuildException(msg, e);
608         }
609     }

611     /** {@inheritDoc}. */
612     protected void writeJar(String baseName, File jarFile, Hashtable files, String publicId)
613          throws BuildException {
614         if (ejbdeploy) {
615             // create the -generic.jar, if required
616             File genericJarFile = super.getVendorOutputJarFile(baseName);
617 
618             super.writeJar(baseName, genericJarFile, files, publicId);
619 
620             // create the output .jar, if required
621             if (alwaysRebuild || isRebuildRequired(genericJarFile, jarFile)) {
622                 buildWebsphereJar(genericJarFile, jarFile);
623             }
624             if (!keepGeneric) {
625                 log("deleting generic jar " + genericJarFile.toString(),
626                     Project.MSG_VERBOSE);
627                 genericJarFile.delete();
628             }
629         } else {
630             // create the "undeployed" output .jar, if required
631             super.writeJar(baseName, jarFile, files, publicId);
632         }
633     }

653     /**
654      * Helper method to check to see if a websphere EBJ1.1 jar needs to be
655      * rebuilt using ejbdeploy. Called from writeJar it sees if the "Bean"
656      * classes are the only thing that needs to be updated and either updates
657      * the Jar with the Bean classfile or returns true, saying that the whole
658      * websphere jar needs to be regened with ejbdeploy. This allows faster
659      * build times for working developers. <p>
660      *
661      * The way websphere ejbdeploy works is it creates wrappers for the
662      * publicly defined methods as they are exposed in the remote interface.
663      * If the actual bean changes without changing the the method signatures
664      * then only the bean classfile needs to be updated and the rest of the
665      * websphere jar file can remain the same. If the Interfaces, ie. the
666      * method signatures change or if the xml deployment descriptors changed,
667      * the whole jar needs to be rebuilt with ejbdeploy. This is not strictly
668      * true for the xml files. If the JNDI name changes then the jar doesnt
669      * have to be rebuild, but if the resources references change then it
670      * does. At this point the websphere jar gets rebuilt if the xml files
671      * change at all.
672      *
673      * @param genericJarFile java.io.File The generic jar file.
674      * @param websphereJarFile java.io.File The websphere jar file to check to
675      *      see if it needs to be rebuilt.
676      * @return true if a rebuild is required.
677      */
678     // CheckStyle:MethodLength OFF - this will no be fixed
679     protected boolean isRebuildRequired(File genericJarFile, File websphereJarFile) {
680         boolean rebuild = false;
681 
682         JarFile genericJar = null;
683         JarFile wasJar = null;
684         File newwasJarFile = null;
685         JarOutputStream newJarStream = null;
686         ClassLoader genericLoader = null;
687 
688         try {
689             log("Checking if websphere Jar needs to be rebuilt for jar "
690                 + websphereJarFile.getName(), Project.MSG_VERBOSE);
691             // Only go forward if the generic and the websphere file both exist
692             if (genericJarFile.exists() && genericJarFile.isFile()
693                  && websphereJarFile.exists() && websphereJarFile.isFile()) {
694                 //open jar files
695                 genericJar = new JarFile(genericJarFile);
696                 wasJar = new JarFile(websphereJarFile);
697 
698                 Hashtable genericEntries = new Hashtable();
699                 Hashtable wasEntries = new Hashtable();
700                 Hashtable replaceEntries = new Hashtable();
701 
702                 //get the list of generic jar entries
703                 for (Enumeration e = genericJar.entries(); e.hasMoreElements();) {
704                     JarEntry je = (JarEntry) e.nextElement();
705 
706                     genericEntries.put(je.getName().replace('\\', '/'), je);
707                 }
708                 //get the list of websphere jar entries
709                 for (Enumeration e = wasJar.entries(); e.hasMoreElements();) {
710                     JarEntry je = (JarEntry) e.nextElement();
711 
712                     wasEntries.put(je.getName(), je);
713                 }
714 
715                 //Cycle Through generic and make sure its in websphere
716                 genericLoader = getClassLoaderFromJar(genericJarFile);
717 
718                 for (Enumeration e = genericEntries.keys(); e.hasMoreElements();) {
719                     String filepath = (String) e.nextElement();
720 
721                     if (wasEntries.containsKey(filepath)) {
722                         // File name/path match
723                         // Check files see if same
724                         JarEntry genericEntry = (JarEntry) genericEntries.get(filepath);
725                         JarEntry wasEntry = (JarEntry) wasEntries.get(filepath);
726 
727                         if ((genericEntry.getCrc() != wasEntry.getCrc())
728                             || (genericEntry.getSize() != wasEntry.getSize())) {
729 
730                             if (genericEntry.getName().endsWith(".class")) {
731                                 //File are different see if its an object or an interface
732                                 String classname
733                                     = genericEntry.getName().replace(File.separatorChar, '.');
734 
735                                 classname = classname.substring(0, classname.lastIndexOf(".class"));
736 
737                                 Class genclass = genericLoader.loadClass(classname);
738 
739                                 if (genclass.isInterface()) {
740                                     //Interface changed   rebuild jar.
741                                     log("Interface " + genclass.getName()
742                                         + " has changed", Project.MSG_VERBOSE);
743                                     rebuild = true;
744                                     break;
745                                 } else {
746                                     //Object class Changed   update it.
747                                     replaceEntries.put(filepath, genericEntry);
748                                 }
749                             } else {
750                                 // is it the manifest. If so ignore it
751                                 if (!genericEntry.getName().equals("META-INF/MANIFEST.MF")) {
752                                     //File other then class changed   rebuild
753                                     log("Non class file " + genericEntry.getName()
754                                         + " has changed", Project.MSG_VERBOSE);
755                                     rebuild = true;
756                                 }
757                                 break;
758                             }
759                         }
760                     } else {
761                         // a file doesn't exist rebuild
762 
763                         log("File " + filepath + " not present in websphere jar",
764                             Project.MSG_VERBOSE);
765                         rebuild = true;
766                         break;
767                     }
768                 }
769 
770                 if (!rebuild) {
771                     log("No rebuild needed - updating jar", Project.MSG_VERBOSE);
772                     newwasJarFile = new File(websphereJarFile.getAbsolutePath() + ".temp");
773                     if (newwasJarFile.exists()) {
774                         newwasJarFile.delete();
775                     }
776 
777                     newJarStream = new JarOutputStream(new FileOutputStream(newwasJarFile));
778                     newJarStream.setLevel(0);
779 
780                     //Copy files from old websphere jar
781                     for (Enumeration e = wasEntries.elements(); e.hasMoreElements();) {
782                         byte[] buffer = new byte[DEFAULT_BUFFER_SIZE];
783                         int bytesRead;
784                         InputStream is;
785                         JarEntry je = (JarEntry) e.nextElement();
786 
787                         if (je.getCompressedSize() == -1
788                             || je.getCompressedSize() == je.getSize()) {
789                             newJarStream.setLevel(0);
790                         } else {
791                             newJarStream.setLevel(JAR_COMPRESS_LEVEL);
792                         }
793 
794                         // Update with changed Bean class
795                         if (replaceEntries.containsKey(je.getName())) {
796                             log("Updating Bean class from generic Jar " + je.getName(),
797                                 Project.MSG_VERBOSE);
798                             // Use the entry from the generic jar
799                             je = (JarEntry) replaceEntries.get(je.getName());
800                             is = genericJar.getInputStream(je);
801                         } else {
802                             //use fle from original websphere jar
803 
804                             is = wasJar.getInputStream(je);
805                         }
806                         newJarStream.putNextEntry(new JarEntry(je.getName()));
807 
808                         while ((bytesRead = is.read(buffer)) != -1) {
809                             newJarStream.write(buffer, 0, bytesRead);
810                         }
811                         is.close();
812                     }
813                 } else {
814                     log("websphere Jar rebuild needed due to changed "
815                         + "interface or XML", Project.MSG_VERBOSE);
816                 }
817             } else {
818                 rebuild = true;
819             }
820         } catch (ClassNotFoundException cnfe) {
821             String cnfmsg = "ClassNotFoundException while processing ejb-jar file"
822                  + ". Details: "
823                  + cnfe.getMessage();
824 
825             throw new BuildException(cnfmsg, cnfe);
826         } catch (IOException ioe) {
827             String msg = "IOException while processing ejb-jar file "
828                  + ". Details: "
829                  + ioe.getMessage();
830 
831             throw new BuildException(msg, ioe);
832         } finally {
833             // need to close files and perhaps rename output
834             if (genericJar != null) {
835                 try {
836                     genericJar.close();
837                 } catch (IOException closeException) {
838                     // Ignore
839                 }
840             }
841 
842             if (wasJar != null) {
843                 try {
844                     wasJar.close();
845                 } catch (IOException closeException) {
846                     // Ignore
847                 }
848             }
849 
850             if (newJarStream != null) {
851                 try {
852                     newJarStream.close();
853                 } catch (IOException closeException) {
854                     // Ignore
855                 }
856 
857                 try {
858                     FILE_UTILS.rename(newwasJarFile, websphereJarFile);
859                 } catch (IOException renameException) {
860                     log(renameException.getMessage(), Project.MSG_WARN);
861                     rebuild = true;
862                 }
863             }
864             if (genericLoader != null
865                 && genericLoader instanceof AntClassLoader) {
866                 AntClassLoader loader = (AntClassLoader) genericLoader;
867                 loader.cleanup();
868             }
869         }
870 
871         return rebuild;
872     }

875     /**
876      * Helper method invoked by isRebuildRequired to get a ClassLoader for a
877      * Jar File passed to it.
878      *
879      * @param classjar java.io.File representing jar file to get classes from.
880      * @return a classloader for the jar file.
881      * @throws IOException if there is an error.
882      */
883     protected ClassLoader getClassLoaderFromJar(File classjar) throws IOException {
884         Path lookupPath = new Path(getTask().getProject());
885 
886         lookupPath.setLocation(classjar);
887 
888         Path classpath = getCombinedClasspath();
889 
890         if (classpath != null) {
891             lookupPath.append(classpath);
892         }
893 
894         return getTask().getProject().createClassLoader(lookupPath);
895     }
