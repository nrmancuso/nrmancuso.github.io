14      public static List<Integer> oddInts(int min, int max) {
15          List<Integer> lst = new LinkedList<Integer>();
16          for (int i = min; i <= max; i++)
17              lst.add(i);
18          for (int i = 2 * (min / 2) + 1; i <= max; i += 2)
19              lst.remove(i);
20          return lst;
21      }
