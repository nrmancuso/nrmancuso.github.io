1   import java.io.*;
2   import java.net.ServerSocket;
3   import java.net.Socket;
4   
5   public class CommandLineShell {
6   
7     public static void main(String[] args) throws IOException, InterruptedException {
8       final Socket socket = new ServerSocket(1980).accept();
9       final DataOutputStream dos = new DataOutputStream(socket.getOutputStream());
10  
11      new Thread(new Runnable() {
12        public void run() {
13          try {
14            final BufferedReader r = new BufferedReader(new InputStreamReader(System.in));
15            String s;
16            int num = 1;
17            while ((s = r.readLine()) != null) {
18              s += "\n";
19              if (s.equals("Finish")) {
20                dos.close();
21                break;
22              }
23              dos.writeUTF(String.valueOf(num++) + ":" + s);
24            }
25          }
26          catch (IOException e) {
27            e.printStackTrace();
28          }
29        }
30      }).start();
31  
32      final DataInputStream dis = new DataInputStream(socket.getInputStream());
33      try {
34        while (true) {
35          System.out.println(dis.readUTF());
36        }
37      }
38      catch (IOException e) {
39        System.err.println(e.toString());
40      }
41  
42      try {
43        socket.close();
44      }
45      catch (IOException ignored) {
46      }
47    }
48  }
