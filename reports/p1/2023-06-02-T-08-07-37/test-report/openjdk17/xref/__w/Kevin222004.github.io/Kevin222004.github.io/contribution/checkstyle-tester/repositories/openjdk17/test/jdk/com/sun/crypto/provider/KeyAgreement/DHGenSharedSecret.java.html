75      public void run() throws Exception {
76          long start, end;
77  
78          BigInteger p = new BigInteger(1, DHPrime);
79          BigInteger g = new BigInteger(1, DHBase);
80          int l = 512;

82          DHParameterSpec spec =

84  
85          // generate keyPairs using parameters
86          KeyPairGenerator keyGen =
87              KeyPairGenerator.getInstance("DH", "SunJCE");
88          keyGen.initialize(spec);
89  
90          // Alice generates her key pairs
91          KeyPair keyA = keyGen.generateKeyPair();
92  
93          // Bob generates his key pairs
94          KeyPair keyB = keyGen.generateKeyPair();
95  
96          // Alice encodes her public key in x509 format
97          // , and sends it over to Bob.
98          byte[] alicePubKeyEnc = keyA.getPublic().getEncoded();
99  
100         // bob encodes his publicKey in x509 format and
101         // sends it over to Alice
102         byte[] bobPubKeyEnc = keyB.getPublic().getEncoded();
103 
104         // bob uses it to generate Secret
105         X509EncodedKeySpec x509Spec =
106             new X509EncodedKeySpec(alicePubKeyEnc);
107         KeyFactory bobKeyFac = KeyFactory.getInstance("DH", "SunJCE");
108         PublicKey alicePubKey = bobKeyFac.generatePublic(x509Spec);
109 
110 
111         KeyAgreement bobAlice = KeyAgreement.getInstance("DH", "SunJCE");
