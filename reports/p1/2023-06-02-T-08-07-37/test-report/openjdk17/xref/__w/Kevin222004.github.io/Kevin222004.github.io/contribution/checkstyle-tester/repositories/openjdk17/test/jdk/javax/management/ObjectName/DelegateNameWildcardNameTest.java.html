24  /*
25   * @test
26   * @bug 4780400
27   * @summary Test that MBeanServerDelegate.DELEGATE_NAME and ObjectName.WILDCARD
28   *          public constants have been initialized properly.
29   * @author Luis-Miguel Alventosa
30   *
31   * @run clean DelegateNameWildcardNameTest
32   * @run build DelegateNameWildcardNameTest
33   * @run main DelegateNameWildcardNameTest
34   */
35  
36  import javax.management.MBeanServerDelegate;
37  import javax.management.ObjectName;
38  
39  public class DelegateNameWildcardNameTest {
40  
41      public static void main(String[] args) throws Exception {
42  
43          System.out.println(
44              "Test that <MBeanServerDelegate.DELEGATE_NAME> equals " +
45              "<new ObjectName(\"JMImplementation:type=MBeanServerDelegate\")>");
46          final ObjectName delegateName =
47                  new ObjectName("JMImplementation:type=MBeanServerDelegate");
48          if (!delegateName.equals(MBeanServerDelegate.DELEGATE_NAME))
49              throw new AssertionError("Unexpected value: " +
50                      "MBeanServerDelegate.DELEGATE_NAME = " +
51                      MBeanServerDelegate.DELEGATE_NAME);
52          System.out.println("MBeanServerDelegate.DELEGATE_NAME = " +
53                  "new ObjectName(\"" + delegateName + "\")");
54  
55          System.out.println("Test that <ObjectName.WILDCARD> " +
56                             "equals <new ObjectName(\"*:*\")>");
57          final ObjectName wildcardName = new ObjectName("*:*");
58          if (!wildcardName.equals(ObjectName.WILDCARD))
59              throw new AssertionError("Unexpected value: " +
60                      "ObjectName.WILDCARD = " +
61                      ObjectName.WILDCARD);
62          System.out.println("ObjectName.WILDCARD = " +
63                  "new ObjectName(\"" + wildcardName + "\")");
64  
65          System.out.println("Test passes: constants were initialized properly");
66      }
67  }
