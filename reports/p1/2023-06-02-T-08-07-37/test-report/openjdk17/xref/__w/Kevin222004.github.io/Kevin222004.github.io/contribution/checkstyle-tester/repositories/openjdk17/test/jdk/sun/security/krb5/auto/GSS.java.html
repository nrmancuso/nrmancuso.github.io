34  import sun.security.jgss.GSSUtil;
35  
36  // Testing JGSS without JAAS
37  public class GSS {
38  
39      public static void main(String[] args) throws Exception {
40  
41          new OneKDC(null).writeJAASConf();
42  
43          Context c, s;
44          c = Context.fromThinAir();
45          s = Context.fromThinAir();
46  
47          // This is the only setting needed for JGSS without JAAS. The default
48          // JAAS config entries are already created by OneKDC.
49          System.setProperty("javax.security.auth.useSubjectCredsOnly", "false");
50  
51          c.startAsClient(OneKDC.SERVER, GSSUtil.GSS_KRB5_MECH_OID);
52          s.startAsServer(GSSUtil.GSS_KRB5_MECH_OID);
53  
54          Context.handshake(c, s);
55  
56          Context.transmit("i say high --", c, s);
57          Context.transmit("   you say low", s, c);
58  
59          s.dispose();
60          c.dispose();
61      }
62  }
