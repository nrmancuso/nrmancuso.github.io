49      public bug6937798() {
50          final JTable table = createCountryTable();
51          table.setShowGrid(true);
52          table.setSize(100, 100);
53  
54          BufferedImage im = new BufferedImage(table.getWidth(), table.getHeight(), BufferedImage.TYPE_INT_ARGB);
55          Graphics g = im.getGraphics();
56          table.print(g);
57          g.dispose();
58  
59          for (int i = 0; i < im.getHeight(); i++) {
60              for (int j = 0; j < im.getWidth(); j++) {
61                  if (im.getRGB(i, j) == table.getGridColor().getRGB()) {
62                      System.out.println("got it!");
63                      return;
64                  }
65              }
66          }
67          throw new RuntimeException("no table's grid detected....");
68      }
