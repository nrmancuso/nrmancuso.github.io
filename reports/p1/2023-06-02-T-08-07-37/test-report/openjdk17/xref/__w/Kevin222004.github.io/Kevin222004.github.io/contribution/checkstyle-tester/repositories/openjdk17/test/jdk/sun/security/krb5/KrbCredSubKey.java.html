66                          null, GSSCredential.INDEFINITE_LIFETIME,
67                          GSSUtil.GSS_KRB5_MECH_OID, GSSCredential.ACCEPT_ONLY));
68                  return ctxt.acceptSecContext(token, 0, token.length);
69              }
70          });
