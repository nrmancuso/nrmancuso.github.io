68      /**
69       * Generates the serializable record classes used by the test. First creates
70       * the initial bytecode for the record class using javac, then adds the
71       * prohibited serialization magic field. Effectively, for example:
72       *
73       *   record R () implements Serializable {
74       *       private static final ObjectStreamField[] serialPersistentFields = {
75       *           new ObjectStreamField("s", String.class),
76       *           new ObjectStreamField("i", int.class),
77       *           new ObjectStreamField("l", long.class),
78       *       };
79       *   }
80       */
81      @BeforeTest
82      public void setup() {
83          {  // R1
84              byte[] byteCode = InMemoryJavaCompiler.compile("R1",
85                      "public record R1 () implements java.io.Serializable { }");
86              ObjectStreamField[] serialPersistentFields = {
87                      new ObjectStreamField("s", String.class),
88                      new ObjectStreamField("i", int.class),
89                      new ObjectStreamField("l", long.class),
90                      new ObjectStreamField("d", double.class)
91              };
92              byteCode = addSerialPersistentFields(byteCode, serialPersistentFields);
93              serializableRecordLoader = new ByteCodeLoader("R1", byteCode, SerialPersistentFieldsTest.class.getClassLoader());
94          }
95          {  // R2
96              byte[] byteCode = InMemoryJavaCompiler.compile("R2",
97                      "public record R2 (int x) implements java.io.Serializable { }");
98              ObjectStreamField[] serialPersistentFields = {
99                      new ObjectStreamField("s", String.class)
100             };
101             byteCode = addSerialPersistentFields(byteCode, serialPersistentFields);
102             serializableRecordLoader = new ByteCodeLoader("R2", byteCode, serializableRecordLoader);
103         }
104         {  // R3
105             byte[] byteCode = InMemoryJavaCompiler.compile("R3",
106                     "public record R3 (int x, int y) implements java.io.Serializable { }");
107             ObjectStreamField[] serialPersistentFields = new ObjectStreamField[0];
108             byteCode = addSerialPersistentFields(byteCode, serialPersistentFields);
109             serializableRecordLoader = new ByteCodeLoader("R3", byteCode, serializableRecordLoader);
110         }
111         {  // R4
112             byte[] byteCode = InMemoryJavaCompiler.compile("R4",
113                     "import java.io.Serializable;" +
114                     "public record R4<U extends Serializable,V extends Serializable>(U u, V v) implements Serializable { }");
115             ObjectStreamField[] serialPersistentFields = {
116                     new ObjectStreamField("v", String.class)
117             };
118             byteCode = addSerialPersistentFields(byteCode, serialPersistentFields);
119             serializableRecordLoader = new ByteCodeLoader("R4", byteCode, serializableRecordLoader);
120         }
121         {  // R5  -- Externalizable
122             byte[] byteCode = InMemoryJavaCompiler.compile("R5",
123                     "import java.io.*;" +
124                     "public record R5 (int x) implements Externalizable {" +
125                     "    @Override public void writeExternal(ObjectOutput out) {\n" +
126                     "        throw new AssertionError(\"should not reach here\");\n" +
127                     "    }\n" +
128                     "    @Override public void readExternal(ObjectInput in) {\n" +
129                     "        throw new AssertionError(\"should not reach here\");\n" +
130                     "    }  }");
131             ObjectStreamField[] serialPersistentFields = {
132                     new ObjectStreamField("v", String.class)
133             };
134             byteCode = addSerialPersistentFields(byteCode, serialPersistentFields);
135             serializableRecordLoader = new ByteCodeLoader("R5", byteCode, serializableRecordLoader);
136         }
137     }

139     /** Constructs a new instance of given named record, with the given args. */
140     Object newRecord(String name, Class<?>[] pTypes, Object[] args) {
141         try {
142             Class<?> c = Class.forName(name, true, serializableRecordLoader);
143             assert c.isRecord();
144             assert c.getRecordComponents() != null;
145             return c.getConstructor(pTypes).newInstance(args);
146         } catch (ReflectiveOperationException e) {
147             throw new AssertionError(e);
148         }
149     }

152         return newRecord("R1", null, null);
153     }

155         return newRecord("R2", new Class[]{int.class}, new Object[]{x});
156     }
157     Object newR3(int x, int y) {

161         return newRecord("R4", new Class[]{Serializable.class, Serializable.class}, new Object[]{u,v});

178     @Test(dataProvider = "recordInstances")
179     public void roundTrip(Object objToSerialize) throws Exception {
180         out.println("\n---");
181         out.println("serializing : " + objToSerialize);
182         var objDeserialized = serializeDeserialize(objToSerialize);
183         out.println("deserialized: " + objDeserialized);
184         assertEquals(objToSerialize, objDeserialized);
185         assertEquals(objDeserialized, objToSerialize);
186     }

188     <T> byte[] serialize(T obj) throws IOException {
189         ByteArrayOutputStream baos = new ByteArrayOutputStream();
190         ObjectOutputStream oos = new ObjectOutputStream(baos);
191         oos.writeObject(obj);
192         oos.close();
193         return baos.toByteArray();
194     }
195 
196     @SuppressWarnings("unchecked")
197     <T> T deserialize(byte[] streamBytes)
198         throws IOException, ClassNotFoundException
199     {
200         ByteArrayInputStream bais = new ByteArrayInputStream(streamBytes);
201         ObjectInputStream ois  = new ObjectInputStream(bais) {
202             @Override
203             protected Class<?> resolveClass(ObjectStreamClass desc)
204                     throws ClassNotFoundException {
205                 return Class.forName(desc.getName(), false, serializableRecordLoader);
206             }
207         };
208         return (T) ois.readObject();
209     }
210 
211     <T> T serializeDeserialize(T obj)
212         throws IOException, ClassNotFoundException
213     {
214         return deserialize(serialize(obj));
215     }

217     // -- machinery for augmenting a record class with prohibited serial field --

219     static byte[] addSerialPersistentFields(byte[] classBytes,
220                                             ObjectStreamField[] spf) {
221         ClassReader reader = new ClassReader(classBytes);
222         ClassWriter writer = new ClassWriter(reader, COMPUTE_MAXS | COMPUTE_FRAMES);
223         reader.accept(new SerialPersistentFieldsVisitor(writer, spf), 0);
224         return writer.toByteArray();
225     }
226 
227     /** A visitor that adds a serialPersistentFields field, and assigns it in clinit. */
228     static final class SerialPersistentFieldsVisitor extends ClassVisitor {
229         static final String FIELD_NAME = "serialPersistentFields";
230         static final String FIELD_DESC = "[Ljava/io/ObjectStreamField;";
231         final ObjectStreamField[] spf;
232         String className;
233         SerialPersistentFieldsVisitor(ClassVisitor cv, ObjectStreamField[] spf) {
234             super(ASM8, cv);
235             this.spf = spf;
236         }
237         @Override
238         public void visit(final int version,
239                           final int access,
240                           final String name,
241                           final String signature,
242                           final String superName,
243                           final String[] interfaces) {
244             this.className = name;
245             cv.visit(version, access, name, signature, superName, interfaces);
246         }
247         @Override
248         public FieldVisitor visitField(final int access,
249                                        final String name,
250                                        final String descriptor,
251                                        final String signature,
252                                        final Object value) {
253             // the field-to-be-added should not already exist
254             assert !name.equals("serialPersistentFields") : "Unexpected " + name + " field";
255             return cv.visitField(access, name, descriptor, signature, value);
256         }
257         @Override
258         public void visitEnd() {
259             {
260                 FieldVisitor fv = cv.visitField(ACC_PRIVATE | ACC_STATIC | ACC_FINAL,
261                                                 FIELD_NAME,
262                                                 FIELD_DESC,
263                                                 null,
264                                                 null);
265                 fv.visitEnd();
266             }
267             {
268                 MethodVisitor mv = cv.visitMethod(ACC_STATIC, "<clinit>", "()V", null, null);
269                 mv.visitCode();
270                 mv.visitIntInsn(BIPUSH, spf.length);
271                 mv.visitTypeInsn(ANEWARRAY, "java/io/ObjectStreamField");
272 
273                 for (int i = 0; i < spf.length; i++) {
274                     ObjectStreamField osf = spf[i];
275                     mv.visitInsn(DUP);
276                     mv.visitIntInsn(BIPUSH, i);
277                     mv.visitTypeInsn(NEW, "java/io/ObjectStreamField");
278                     mv.visitInsn(DUP);
279                     mv.visitLdcInsn(osf.getName());
280                     if (osf.getType().isPrimitive()) {
281                         mv.visitFieldInsn(GETSTATIC,  getPrimitiveBoxClass(osf.getType()), "TYPE", "Ljava/lang/Class;");
282                     } else {
283                         mv.visitLdcInsn(Type.getType(osf.getType()));
284                     }
285                     mv.visitMethodInsn(INVOKESPECIAL, "java/io/ObjectStreamField", "<init>", "(Ljava/lang/String;Ljava/lang/Class;)V", false);
286                     mv.visitInsn(AASTORE);
287                 }
288 
289                 mv.visitFieldInsn(PUTSTATIC, className, "serialPersistentFields", "[Ljava/io/ObjectStreamField;");
290                 mv.visitInsn(RETURN);
291                 mv.visitMaxs(0, 0);
292                 mv.visitEnd();
293             }
294             cv.visitEnd();
295         }
296 
297         static String getPrimitiveBoxClass(final Class<?> clazz) {
298             if (!clazz.isPrimitive())
299                 throw new AssertionError("unexpected non-primitive:" + clazz);
300 
301             if (clazz == Integer.TYPE) {
302                 return "java/lang/Integer";
303             } else if (clazz == Boolean.TYPE) {
304                 return "java/lang/Boolean";
305             } else if (clazz == Byte.TYPE) {
306                 return "java/lang/Byte";
307             } else if (clazz == Character.TYPE) {
308                 return "java/lang/Character";
309             } else if (clazz == Short.TYPE) {
310                 return "java/lang/Short";
311             } else if (clazz == Double.TYPE) {
312                 return "java/lang/Double";
313             } else if (clazz == Float.TYPE) {
314                 return "java/lang/Float";
315             } else if (clazz == Long.TYPE) {
316                 return "java/lang/Long";
317             } else {
318                 throw new AssertionError("unknown:" + clazz);
319             }
320         }
321     }

325     /** Checks to ensure correct operation of the test's generation logic. */
326     @Test(dataProvider = "recordInstances")
327     public void wellFormedGeneratedClasses(Object obj) throws Exception {
328         out.println("\n---");
329         out.println(obj);
330         Field f = obj.getClass().getDeclaredField("serialPersistentFields");
331         assertTrue((f.getModifiers() & Modifier.PRIVATE) != 0);
332         assertTrue((f.getModifiers() & Modifier.STATIC) != 0);
333         assertTrue((f.getModifiers() & Modifier.FINAL) != 0);
334         f.setAccessible(true);
335         ObjectStreamField[] fv = (ObjectStreamField[])f.get(obj);
336         assertTrue(fv != null, "Unexpected null value");
337         assertTrue(fv.length >= 0, "Unexpected negative length:" + fv.length);
338     }
