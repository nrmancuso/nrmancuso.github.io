24  /*
25   * @test
26   * @bug 4756443
27   * @summary REGRESSION: NPE in JarURLConnection.getLastModified after setUseCache(false)
28   */
29  
30  import java.io.File;
31  import java.io.IOException;
32  import java.net.URL;
33  import java.net.URLConnection;
34  
35  public class B4756443 {
36  
37      public static void main(String [] args) throws IOException
38      {
39          String testsrc = System.getProperty ("test.src");
40          URL u;
41          if (testsrc == null || "".equals (testsrc)) {
42              /* not running in jtreg */
43              u = new URL ("jar:file:./foo.jar!/a/b/test.xml");
44          } else {
45              /* running in jtreg */
46              File f = new File(testsrc + File.separator
47                              + "foo.jar");
48              String s = f.toURL().toString();
49              u = new URL ("jar:" + s + "!/a/b/test.xml");
50          }
51          System.out.println ("Testing with: " + u);
52          URLConnection con = u.openConnection ();
53          con.setUseCaches (false);
54          con.connect ();
55          long l = con.getLastModified ();
56  
57          URLConnection con1 = u.openConnection ();
58          con1.setUseCaches (true);
59          con1.connect ();
60          long l1 = con1.getLastModified ();
61  
62          if (l != l1) {
63              throw new RuntimeException ("l != l1 ("+l+"/"+l1+")");
64          }
65      }
66  }
