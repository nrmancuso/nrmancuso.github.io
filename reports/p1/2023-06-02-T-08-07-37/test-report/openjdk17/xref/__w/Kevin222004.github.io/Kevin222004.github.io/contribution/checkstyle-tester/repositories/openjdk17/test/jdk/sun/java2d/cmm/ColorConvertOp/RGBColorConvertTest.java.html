24  /**
25   * @test
26   * @bug 6279846
27   * @summary Verifies that transform between the same ICC color spaces does not
28   * change pixels
29   */
30  
31  import java.awt.image.*;
32  import java.awt.color.ColorSpace;
33  
34  public class RGBColorConvertTest {
35  
36      public static void main(String [] args) {
37          BufferedImage src =
38              new BufferedImage(256,3,BufferedImage.TYPE_INT_RGB);
39          BufferedImage dst =
40              new BufferedImage(256,3,BufferedImage.TYPE_INT_RGB);
41  
42          for (int i = 0; i < 256; i++) {
43              src.setRGB(i,0,i);
44              src.setRGB(i,1,i << 8);
45              src.setRGB(i,2,i << 16);
46          }
47  
48          ColorSpace srcColorSpace = src.getColorModel().getColorSpace();
49  
50          ColorConvertOp op = new ColorConvertOp(srcColorSpace, srcColorSpace,
51                                                 null);
52          op.filter(src, dst);
53  
54          int errCount = 0;
55          for (int i = 0; i < src.getWidth(); i++) {
56              for (int j = 0; j < src.getHeight(); j++) {
57                  int scol = src.getRGB(i,j);
58                  int dcol = dst.getRGB(i,j);
59                  if (scol != dcol) {
60                      System.err.println("(" + i + "," + j + ") : " +
61                                         Integer.toHexString(scol) + "!=" +
62                                         Integer.toHexString(dcol));
63                      errCount++;
64                  }
65              }
66          }
67  
68          if (errCount > 0) {
69              throw new RuntimeException(errCount + " pixels are changed by " +
70                                         "transform between the same ICC color " +
71                                         "spaces");
72          }
73      }
74  }
