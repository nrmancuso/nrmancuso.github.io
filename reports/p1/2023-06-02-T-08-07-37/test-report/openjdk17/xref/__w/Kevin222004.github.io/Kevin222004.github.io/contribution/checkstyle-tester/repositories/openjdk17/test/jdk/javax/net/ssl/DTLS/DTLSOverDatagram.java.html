133     SSLEngine createSSLEngine(boolean isClient) throws Exception {
134         SSLContext context = getDTLSContext();
135         SSLEngine engine = context.createSSLEngine();
136 
137         SSLParameters paras = engine.getSSLParameters();
138         paras.setMaximumPacketSize(MAXIMUM_PACKET_SIZE);
139 
140         engine.setUseClientMode(isClient);
141         engine.setSSLParameters(paras);
142 
143         return engine;
144     }

146     // handshake
147     void handshake(SSLEngine engine, DatagramSocket socket,
148             SocketAddress peerAddr, String side) throws Exception {
149 
150         boolean endLoops = false;
151         int loops = MAX_HANDSHAKE_LOOPS;
152         engine.beginHandshake();
153         while (!endLoops &&
154                 (serverException == null) && (clientException == null)) {
155 
156             if (--loops < 0) {
157                 throw new RuntimeException(
158                         "Too many loops to produce handshake packets");
159             }
160 
161             SSLEngineResult.HandshakeStatus hs = engine.getHandshakeStatus();
162             log(side, "=======handshake(" + loops + ", " + hs + ")=======");
163 
164             switch (hs) {
165                 case NEED_UNWRAP, NEED_UNWRAP_AGAIN -> {
166                     log(side, "Receive DTLS records, handshake status is " + hs);
167 
168                     ByteBuffer iNet;
169                     ByteBuffer iApp;
170 
171                     if (hs == SSLEngineResult.HandshakeStatus.NEED_UNWRAP) {
172                         byte[] buf = new byte[BUFFER_SIZE];
173                         DatagramPacket packet = new DatagramPacket(buf, buf.length);
174                         try {
175                             socket.receive(packet);
176                         } catch (SocketTimeoutException ste) {
177                             log(side, "Warning: " + ste);
178 
179                             List<DatagramPacket> packets = new ArrayList<>();
180                             boolean finished = onReceiveTimeout(
181                                     engine, peerAddr, side, packets);
182 
183                             log(side, "Reproduced " + packets.size() + " packets");
184                             for (DatagramPacket p : packets) {
185                                 printHex("Reproduced packet",
186                                         p.getData(), p.getOffset(), p.getLength());
187                                 socket.send(p);
188                             }
189 
190                             if (finished) {
191                                 log(side, "Handshake status is FINISHED "
192                                         + "after calling onReceiveTimeout(), "
193                                         + "finish the loop");
194                                 endLoops = true;
195                             }
196 
197                             log(side, "New handshake status is "
198                                     + engine.getHandshakeStatus());
199 
200                             continue;
201                         }
202 
203                         iNet = ByteBuffer.wrap(buf, 0, packet.getLength());
204                     } else {
205                         iNet = ByteBuffer.allocate(0);
206                     }
207 
208                     iApp = ByteBuffer.allocate(BUFFER_SIZE);
209 
210                     SSLEngineResult r = engine.unwrap(iNet, iApp);
211                     hs = r.getHandshakeStatus();
212 
213                     verifySSLEngineResultStatus(r, side);
214                     if (hs == SSLEngineResult.HandshakeStatus.FINISHED) {
215                         log(side, "Handshake status is FINISHED, finish the loop");
216                         endLoops = true;
217                     }
218                 }
219                 case NEED_WRAP -> {
220                     List<DatagramPacket> packets = new ArrayList<>();
221                     boolean finished = produceHandshakePackets(
222                             engine, peerAddr, side, packets);
223 
224                     log(side, "Produced " + packets.size() + " packets");
225                     for (DatagramPacket p : packets) {
226                         socket.send(p);
227                     }
228 
229                     if (finished) {
230                         log(side, "Handshake status is FINISHED "
231                                 + "after producing handshake packets, "
232                                 + "finish the loop");
233                         endLoops = true;
234                     }
235                 }
236                 case NEED_TASK -> runDelegatedTasks(engine);
237                 case NOT_HANDSHAKING -> {
238                     log(side,
239                             "Handshake status is NOT_HANDSHAKING, finish the loop");
240                     endLoops = true;
241                 }
242                 case FINISHED -> throw new Exception( "Unexpected status, " +
243                         "SSLEngine.getHandshakeStatus() shouldn't return FINISHED");
244                 default -> throw new Exception("Can't reach here, " +
245                         "handshake status is " + hs);
246             }
247         }
248 
249         SSLEngineResult.HandshakeStatus hs = engine.getHandshakeStatus();
250         log(side, "Handshake finished, status is " + hs);
251 
252         if (engine.getHandshakeSession() != null) {
253             throw new Exception(
254                     "Handshake finished, but handshake session is not null");
255         }
256 
257         SSLSession session = engine.getSession();
258         if (session == null) {
259             throw new Exception("Handshake finished, but session is null");
260         }
261         log(side, "Negotiated protocol is " + session.getProtocol());
262         log(side, "Negotiated cipher suite is " + session.getCipherSuite());
263 
264         // handshake status should be NOT_HANDSHAKING
265         //
266         // According to the spec, SSLEngine.getHandshakeStatus() can't
267         // return FINISHED.
268         if (hs != SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING) {
269             throw new Exception("Unexpected handshake status " + hs);
270         }
271     }

273     void verifySSLEngineResultStatus(SSLEngineResult r, String side) throws Exception {
274         SSLEngineResult.Status rs = r.getStatus();
275         SSLEngineResult.HandshakeStatus hs = r.getHandshakeStatus();
276         switch (rs) {
277             case OK -> log(side, "SSLEngineResult status OK");
278             case BUFFER_OVERFLOW -> {
279                 log(side, "BUFFER_OVERFLOW, handshake status is " + hs);
280                 // the client maximum fragment size config does not work?
281                 throw new Exception("Buffer overflow: " +
282                         "incorrect client maximum fragment size");
283             }
284             case BUFFER_UNDERFLOW -> {
285                 log(side, "BUFFER_UNDERFLOW, handshake status is " + hs);
286                 // bad packet, or the client maximum fragment size
287                 // config does not work?
288                 if (hs != SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING) {
289                     throw new Exception("Buffer underflow: " +
290                             "incorrect client maximum fragment size");
291                 } // otherwise, ignore this packet
292             }
293             case CLOSED -> throw new Exception(
294                     "SSL engine closed, handshake status is " + hs);
295             default -> throw new Exception("Can't reach here, result is " + rs);
296         }
297     }

299     // deliver application data
300     void deliverAppData(SSLEngine engine, DatagramSocket socket,
301             ByteBuffer appData, SocketAddress peerAddr, String side) throws Exception {
302 
303         // Note: have not consider the packet loses
304         List<DatagramPacket> packets =
305                 produceApplicationPackets(engine, appData, peerAddr, side);
306         appData.flip();
307         for (DatagramPacket p : packets) {
308             socket.send(p);
309         }
310     }

312     // receive application data
313     void receiveAppData(SSLEngine engine,
314             DatagramSocket socket, ByteBuffer expectedApp) throws Exception {
315 
316         int loops = MAX_APP_READ_LOOPS;
317         while ((serverException == null) && (clientException == null)) {
318             if (--loops < 0) {
319                 throw new RuntimeException(
320                         "Too much loops to receive application data");
321             }
322 
323             byte[] buf = new byte[BUFFER_SIZE];
324             DatagramPacket packet = new DatagramPacket(buf, buf.length);
325             socket.receive(packet);
326             ByteBuffer netBuffer = ByteBuffer.wrap(buf, 0, packet.getLength());
327             ByteBuffer recBuffer = ByteBuffer.allocate(BUFFER_SIZE);
328             SSLEngineResult rs = engine.unwrap(netBuffer, recBuffer);
329             recBuffer.flip();
330             if (recBuffer.remaining() != 0) {
331                 printHex("Received application data", recBuffer);
332                 if (!recBuffer.equals(expectedApp)) {
333                     System.out.println("Engine status is " + rs);
334                     throw new Exception("Not the right application data");
335                 }
336                 break;
337             }
338         }
339     }

341     // produce handshake packets
342     boolean produceHandshakePackets(SSLEngine engine, SocketAddress socketAddr,
343             String side, List<DatagramPacket> packets) throws Exception {
344 
345         boolean endLoops = false;
346         int loops = MAX_HANDSHAKE_LOOPS / 2;
347         while (!endLoops &&
348                 (serverException == null) && (clientException == null)) {
349 
350             if (--loops < 0) {
351                 throw new RuntimeException(
352                         "Too many loops to produce handshake packets");
353             }
354 
355             ByteBuffer oNet = ByteBuffer.allocate(32768);
356             ByteBuffer oApp = ByteBuffer.allocate(0);
357             SSLEngineResult r = engine.wrap(oApp, oNet);
358             oNet.flip();
359 
360             SSLEngineResult.Status rs = r.getStatus();
361             SSLEngineResult.HandshakeStatus hs = r.getHandshakeStatus();
362             log(side, "----produce handshake packet(" +
363                     loops + ", " + rs + ", " + hs + ")----");
364 
365             verifySSLEngineResultStatus(r, side);
366 
367             if (oNet.hasRemaining()) {
368                 byte[] ba = new byte[oNet.remaining()];
369                 oNet.get(ba);
370                 DatagramPacket packet = createHandshakePacket(ba, socketAddr);
371                 packets.add(packet);
372             }
373 
374             if (hs == SSLEngineResult.HandshakeStatus.FINISHED) {
375                 log(side, "Produce handshake packets: "
376                             + "Handshake status is FINISHED, finish the loop");
377                 return true;
378             }
379 
380             boolean endInnerLoop = false;
381             SSLEngineResult.HandshakeStatus nhs = hs;
382             while (!endInnerLoop) {
383                 switch (nhs) {
384                     case NEED_TASK -> runDelegatedTasks(engine);
385                     case NEED_UNWRAP, NEED_UNWRAP_AGAIN, NOT_HANDSHAKING -> {
386                         endInnerLoop = true;
387                         endLoops = true;
388                     }
389                     case NEED_WRAP -> endInnerLoop = true;
390                     case FINISHED ->  throw new Exception(
391                             "Unexpected status, SSLEngine.getHandshakeStatus() "
392                                     + "should not return FINISHED");
393                     default -> throw new Exception("Can't reach here, handshake status is "
394                             + nhs);
395                 }
396 
397                 nhs = engine.getHandshakeStatus();
398             }
399         }
400 
401         return false;
402     }

404     DatagramPacket createHandshakePacket(byte[] ba, SocketAddress socketAddr) {
405         return new DatagramPacket(ba, ba.length, socketAddr);
406     }

408     // produce application packets
409     List<DatagramPacket> produceApplicationPackets(
410             SSLEngine engine, ByteBuffer source,
411             SocketAddress socketAddr, String side) throws Exception {
412 
413         List<DatagramPacket> packets = new ArrayList<>();
414         ByteBuffer appNet = ByteBuffer.allocate(32768);
415         SSLEngineResult r = engine.wrap(source, appNet);
416         appNet.flip();
417 
418         verifySSLEngineResultStatus(r, side);
419 
420         if (appNet.hasRemaining()) {
421             byte[] ba = new byte[appNet.remaining()];
422             appNet.get(ba);
423             DatagramPacket packet =
424                     new DatagramPacket(ba, ba.length, socketAddr);
425             packets.add(packet);
426         }
427 
428         return packets;
429     }

431     // Get a datagram packet for the specified handshake type.
432     static DatagramPacket getPacket(
433             List<DatagramPacket> packets, byte handshakeType) {
434         boolean matched = false;
435         for (DatagramPacket packet : packets) {
436             byte[] data = packet.getData();
437             int offset = packet.getOffset();
438             int length = packet.getLength();
439 
440             // Normally, this packet should be a handshake message
441             // record.  However, even if the underlying platform
442             // splits the record more, we don't really worry about
443             // the improper packet loss because DTLS implementation
444             // should be able to handle packet loss properly.
445             //
446             // See RFC 6347 for the detailed format of DTLS records.
447             if (handshakeType == -1) {      // ChangeCipherSpec
448                 // Is it a ChangeCipherSpec message?
449                 matched = (length == 14) && (data[offset] == 0x14);
450             } else if ((length >= 25) &&    // 25: handshake mini size
451                 (data[offset] == 0x16)) {   // a handshake message
452 
453                 // check epoch number for initial handshake only
454                 if (data[offset + 3] == 0x00) {     // 3,4: epoch
455                     if (data[offset + 4] == 0x00) { // plaintext
456                         matched =
457                             (data[offset + 13] == handshakeType);
458                     } else {                        // ciphertext
459                         // The 1st ciphertext is a Finished message.
460                         //
461                         // If it is not proposed to loss the Finished
462                         // message, it is not necessary to check the
463                         // following packets any more as a Finished
464                         // message is the last handshake message.
465                         matched = (handshakeType == 20);
466                     }
467                 }
468             }
469 
470             if (matched) {
471                 return packet;
472             }
473         }
474 
475         return null;
476     }

478     // run delegated tasks
479     void runDelegatedTasks(SSLEngine engine) throws Exception {
480         Runnable runnable;
481         while ((runnable = engine.getDelegatedTask()) != null) {
482             runnable.run();
483         }
484 
485         SSLEngineResult.HandshakeStatus hs = engine.getHandshakeStatus();
486         if (hs == SSLEngineResult.HandshakeStatus.NEED_TASK) {
487             throw new Exception("handshake shouldn't need additional tasks");
488         }
489     }

491     // retransmission if timeout
492     boolean onReceiveTimeout(SSLEngine engine, SocketAddress socketAddr,
493             String side, List<DatagramPacket> packets) throws Exception {
494 
495         SSLEngineResult.HandshakeStatus hs = engine.getHandshakeStatus();
496         if (hs == SSLEngineResult.HandshakeStatus.NOT_HANDSHAKING) {
497             return false;
498         } else {
499             // retransmission of handshake messages
500             return produceHandshakePackets(engine, socketAddr, side, packets);
501         }
502     }

504     // get DTSL context
505     SSLContext getDTLSContext() throws Exception {
506         String passphrase = "passphrase";
507         return SSLContextBuilder.builder()
508                 .trustStore(KeyStoreUtils.loadKeyStore(TRUST_FILENAME, passphrase))
509                 .keyStore(KeyStoreUtils.loadKeyStore(KEY_FILENAME, passphrase))
510                 .kmfPassphrase(passphrase)
511                 .protocol("DTLS")
512                 .build();
513     }

526     public final void runTest(DTLSOverDatagram testCase) throws Exception {
527         InetSocketAddress serverSocketAddress = new InetSocketAddress
528                 (InetAddress.getLoopbackAddress(), 0);
529         InetSocketAddress clientSocketAddress = new InetSocketAddress
530                 (InetAddress.getLoopbackAddress(), 0);
531 
532         try (DatagramSocket serverSocket = new DatagramSocket(serverSocketAddress);
533                 DatagramSocket clientSocket = new DatagramSocket(clientSocketAddress)) {
534 
535             serverSocket.setSoTimeout(SOCKET_TIMEOUT);
536             clientSocket.setSoTimeout(SOCKET_TIMEOUT);
537 
538             InetSocketAddress serverSocketAddr = new InetSocketAddress(
539                     InetAddress.getLoopbackAddress(), serverSocket.getLocalPort());
540 
541             InetSocketAddress clientSocketAddr = new InetSocketAddress(
542                     InetAddress.getLoopbackAddress(), clientSocket.getLocalPort());
543 
544             ExecutorService pool = Executors.newFixedThreadPool(2);
545             Future<String> server, client;
546 
547             try {
548                 server = pool.submit(new ServerCallable(
549                         testCase, serverSocket, clientSocketAddr));
550                 client = pool.submit(new ClientCallable(
551                         testCase, clientSocket, serverSocketAddr));
552             } finally {
553                 pool.shutdown();
554             }
555 
556             boolean failed = false;
557 
558             // wait for client to finish
559             try {
560                 System.out.println("Client finished: " + client.get());
561             } catch (CancellationException | InterruptedException
562                         | ExecutionException e) {
563                 System.out.println("Exception on client side: ");
564                 e.printStackTrace(System.out);
565                 failed = true;
566             }
567 
568             // wait for server to finish
569             try {
570                 System.out.println("Client finished: " + server.get());
571             } catch (CancellationException | InterruptedException
572                         | ExecutionException e) {
573                 System.out.println("Exception on server side: ");
574                 e.printStackTrace(System.out);
575                 failed = true;
576             }
577 
578             if (failed) {
579                 throw new RuntimeException("Test failed");
580             }
581         }
582     }

584     record ServerCallable(DTLSOverDatagram testCase, DatagramSocket socket,
585                           InetSocketAddress clientSocketAddr) implements Callable<String> {
586 
587         @Override
588         public String call() throws Exception {
589             try {
590                 testCase.doServerSide(socket, clientSocketAddr);
591             } catch (Exception e) {
592                 System.out.println("Exception in  ServerCallable.call():");
593                 e.printStackTrace(System.out);
594                 serverException = e;
595 
596                 if (testCase.isGoodJob()) {
597                     throw e;
598                 } else {
599                     return "Well done, server!";
600                 }
601             }
602 
603             if (testCase.isGoodJob()) {
604                 return "Well done, server!";
605             } else {
606                 throw new Exception("No expected exception");
607             }
608         }
609     }

639 
640         synchronized (System.out) {
641             System.out.println(prefix);
