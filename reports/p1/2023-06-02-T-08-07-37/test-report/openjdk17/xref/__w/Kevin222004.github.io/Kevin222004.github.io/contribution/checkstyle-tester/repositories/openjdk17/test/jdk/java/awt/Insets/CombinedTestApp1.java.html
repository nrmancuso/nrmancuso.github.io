35  import java.awt.*;
36  import java.awt.event.*;
37  import test.java.awt.regtesthelpers.Util;
38  
39  public class CombinedTestApp1
40  {
41      public static void main(String[] args)
42      {
43          Frame frame = new Frame("A test");
44          frame.setSize (200,200);
45  
46          frame.setLayout(new FlowLayout());
47          MenuBar mbar = new MenuBar();
48          Menu file = new Menu("File");
49          mbar.add(file);
50  
51          MenuItem quit = new MenuItem("Quit", new MenuShortcut(KeyEvent.VK_Q));
52          quit.addActionListener(new ActionListener() {
53                  public void actionPerformed(ActionEvent e) { System.exit(0);}
54              });
55          file.add(quit);
56  
57          frame.setMenuBar(mbar);
58          frame.add(new Button("One"));
59          frame.add(new Button("Two"));
60          frame.add(new Button("Three"));
61  
62          frame.pack();
63          frame.setVisible(true);
64          try {
65              Robot robot = new Robot();
66              Util.waitForIdle(robot);
67          } catch(AWTException e){
68              throw new RuntimeException("Test interrupted.", e);
69          }
70  
71          Insets frameInsets = frame.getInsets();
72          System.out.println(frameInsets);
73          if (frameInsets.bottom < 0 || frameInsets.top < 0 ||
74              frameInsets.right < 0 || frameInsets.left <0){
75              throw new RuntimeException("Failed. Frames' Insets are incorrect.");
76          }
77          System.out.println(" Test Passed. ");
78      }
79  }
