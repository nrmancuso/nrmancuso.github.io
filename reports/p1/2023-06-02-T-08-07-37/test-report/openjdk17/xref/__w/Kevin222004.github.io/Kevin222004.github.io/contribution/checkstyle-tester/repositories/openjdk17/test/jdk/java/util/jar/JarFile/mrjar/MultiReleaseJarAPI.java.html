83      @Test
84      public void isMultiReleaseJar() throws Exception {
85          try (JarFile jf = new JarFile(unversioned)) {
86              Assert.assertFalse(jf.isMultiRelease());
87          }
88  
89          try (JarFile jf = new JarFile(unversioned, true, ZipFile.OPEN_READ, Runtime.version())) {
90              Assert.assertFalse(jf.isMultiRelease());
91          }
92  
93          try (JarFile jf = new JarFile(multirelease)) {
94              Assert.assertTrue(jf.isMultiRelease());
95          }
96  
97          try (JarFile jf = new JarFile(multirelease, true, ZipFile.OPEN_READ, Runtime.version())) {
98              Assert.assertTrue(jf.isMultiRelease());
99          }
100 
101         testCustomMultiReleaseValue("true", true);
102         testCustomMultiReleaseValue("true\r\nOther: value", true);
103         testCustomMultiReleaseValue("true\nOther: value", true);
104         // JDK-8200530: '\r' support in Manifest/Attributes will be addressed separately
105         // testCustomMultiReleaseValue("true\rOther: value", true);
106 
107         testCustomMultiReleaseValue("false", false);
108         testCustomMultiReleaseValue(" true", false);
109         testCustomMultiReleaseValue("true ", false);
110         testCustomMultiReleaseValue("true\n true", false);
111 
112         // JDK-8200530: '\r' support in Manifest/Attributes will be addressed separately
113         testCustomMultiReleaseValue("true\r true", false);
114         testCustomMultiReleaseValue("true\r\n true", false);
115 
116         // "Multi-Release: true/false" not in main attributes
117         testCustomMultiReleaseValue("\r\n\r\nName: test\r\nMulti-Release: true\r\n",
118                                     false);
119         testCustomMultiReleaseValue("\n\nName: entryname\nMulti-Release: true\n",
120                                     false);
121         testCustomMultiReleaseValue("EndOfMainAttr: whatever\r\n" +
122                                     "\r\nName: entryname\r\nMulti-Release: true\r\n",
123                                     false);
124         testCustomMultiReleaseValue("EndOfMainAttr: whatever\r\n" +
125                                     "\nName: entryname\nMulti-Release: true\n",
126                                     false);
127 
128         // generate "random" Strings to use as extra attributes, and
129         // verify that Multi-Release: true is always properly matched
130         for (int i = 0; i < 100; i++) {
131             byte[] keyBytes = new byte[RANDOM.nextInt(70) + 1];
132             Arrays.fill(keyBytes, (byte)('a' + RANDOM.nextInt(24)));
133             byte[] valueBytes = new byte[RANDOM.nextInt(70) + 1];
134             Arrays.fill(valueBytes, (byte)('a' + RANDOM.nextInt(24)));
135 
136             String key = new String(keyBytes, StandardCharsets.UTF_8);
137             String value = new String(valueBytes, StandardCharsets.UTF_8);
138             // test that Multi-Release: true anywhere in the manifest always
139             // return true
140             testCustomMultiReleaseValue("true", Map.of(key, value), true);
141 
142             // test that we don't get any false positives
143             testCustomMultiReleaseValue("false", Map.of(key, value), false);
144         }
145     }

147     private void testCustomMultiReleaseValue(String value, boolean expected)
148             throws Exception {
149         testCustomMultiReleaseValue(value, Map.of(), expected);
150     }
151 
152     private static final AtomicInteger JAR_COUNT = new AtomicInteger(0);
153 
154     private void testCustomMultiReleaseValue(String value,
155             Map<String, String> extraAttributes, boolean expected)
156             throws Exception {
157         String fileName = "custom-mr" + JAR_COUNT.incrementAndGet() + ".jar";
158         creator.buildCustomMultiReleaseJar(fileName, value, extraAttributes);
159         File custom = new File(userdir, fileName);
160         try (JarFile jf = new JarFile(custom, true, ZipFile.OPEN_READ, Runtime.version())) {
161             Assert.assertEquals(jf.isMultiRelease(), expected);
162         }
163         Files.delete(custom.toPath());
164     }

178     @Test(dataProvider="versions")
179     public void testVersioning(Runtime.Version value, int xpected) throws Exception {
180         Runtime.Version expected = Runtime.Version.parse(String.valueOf(xpected));
181         Runtime.Version base = JarFile.baseVersion();
182 
183         // multi-release jar, opened as unversioned
184         try (JarFile jar = new JarFile(multirelease)) {
185             Assert.assertEquals(jar.getVersion(), base);
186         }
187 
188         System.err.println("test versioning for Release " + value);
189         try (JarFile jf = new JarFile(multirelease, true, ZipFile.OPEN_READ, value)) {
190             Assert.assertEquals(jf.getVersion(), expected);
191         }
192 
193         // regular, unversioned, jar
194         try (JarFile jf = new JarFile(unversioned, true, ZipFile.OPEN_READ, value)) {
195             Assert.assertEquals(jf.getVersion(), base);
196         }
197     }

199     @Test(dataProvider="versions")
200     public void testAliasing(Runtime.Version version, int xpected) throws Exception {
201         int n = Math.max(version.major(), JarFile.baseVersion().major());
202         Runtime.Version value = Runtime.Version.parse(String.valueOf(n));
203         System.err.println("test aliasing for Release " + version);
204         String prefix;
205         if (JarFile.baseVersion().equals(value)) {
206             prefix = "";
207         } else {
208             prefix = "META-INF/versions/" + value.major() + "/";
209         }
210         // test both multi-release jars
211         readAndCompare(multirelease, value, "README", prefix + "README");
212         readAndCompare(multirelease, value, "version/Version.class", prefix + "version/Version.class");
213         // and signed multi-release jars
214         readAndCompare(signedmultirelease, value, "README", prefix + "README");
215         readAndCompare(signedmultirelease, value, "version/Version.class", prefix + "version/Version.class");
216     }
217 
218     private void readAndCompare(File jar, Runtime.Version version, String name, String realName) throws Exception {
219         byte[] baseBytes;
220         byte[] versionedBytes;
221         try (JarFile jf = new JarFile(jar, true, ZipFile.OPEN_READ, JarFile.baseVersion())) {
222             ZipEntry ze = jf.getEntry(realName);
223             try (InputStream is = jf.getInputStream(ze)) {
224                 baseBytes = is.readAllBytes();
225             }
226         }
227         assert baseBytes.length > 0;
228 
229         try (JarFile jf = new JarFile(jar, true, ZipFile.OPEN_READ, version)) {
230             ZipEntry ze = jf.getEntry(name);
231             try (InputStream is = jf.getInputStream(ze)) {
232                 versionedBytes = is.readAllBytes();
233             }
234         }
235         assert versionedBytes.length > 0;
236 
237         Assert.assertTrue(Arrays.equals(baseBytes, versionedBytes));
238     }

240     @Test
241     public void testNames() throws Exception {
242         String rname = "version/Version.class";
243         String vname = "META-INF/versions/9/version/Version.class";
244         ZipEntry ze1;
245         ZipEntry ze2;
246         try (JarFile jf = new JarFile(multirelease)) {
247             ze1 = jf.getEntry(vname);
248         }
249         Assert.assertEquals(ze1.getName(), vname);
250         try (JarFile jf = new JarFile(multirelease, true, ZipFile.OPEN_READ, Runtime.Version.parse("9"))) {
251             ze2 = jf.getEntry(rname);
252         }
253         Assert.assertEquals(ze2.getName(), rname);
254         Assert.assertNotEquals(ze1.getName(), ze2.getName());
255     }
