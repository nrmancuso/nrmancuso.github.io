24  import java.awt.*;
25  
26  /*
27   * @test
28   * @bug 8164811
29   * @key headful
30   * @summary Check if a per-pixel translucent window is dragged and resized
31   *          by mouse correctly.
32   * Test Description: Check if PERPIXEL_TRANSLUCENT translucency type is supported
33   *      on the current platform. Proceed if they are supported. Create a window
34   *      with some components in it, make window undecorated, apply translucent
35   *      background of 0.5. Drag and resize the window using AWT Robot and verify
36   *      that translucency is correctly applied with pixels checking. Make the
37   *      window appear on top of a known background. Repeat this for JWindow,
38   *      JDialog, JFrame.
39   * Expected Result: If PERPIXEL_TRANSLUCENT translucency type is supported, the
40   *      window should appear with the translucency. Only window background
41   *      should be translucent, all the controls should be opaque.
42   * @author mrkam
43   * @library /lib/client
44   * @build Common ExtendedRobot
45   * @run main PerPixelTranslucent
46   * @run main/othervm -Dsun.java2d.uiScale=1.5 PerPixelTranslucent
47   */
48  
49  public class PerPixelTranslucent extends Common {
50  
51      public static void main(String[] ignored) throws Exception {
52          if (checkTranslucencyMode(GraphicsDevice.WindowTranslucency.PERPIXEL_TRANSLUCENT))
53              for (Class<Window> windowClass: WINDOWS_TO_TEST)
54                  new PerPixelTranslucent(windowClass).doTest();
55      }
56  
57      public PerPixelTranslucent(Class windowClass) throws Exception {
58          super(windowClass, 1.0f, 0.5f, false);
59      }
60  
61      public void doTest() throws Exception {
62          robot.waitForIdle(delay);
63          checkTranslucent();
64      }
65  }
