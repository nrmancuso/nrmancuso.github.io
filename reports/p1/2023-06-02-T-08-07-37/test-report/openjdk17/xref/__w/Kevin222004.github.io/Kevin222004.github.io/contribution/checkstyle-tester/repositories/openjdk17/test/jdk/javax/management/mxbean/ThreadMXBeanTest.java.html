24  /*
25   * @test
26   * @bug 6305746
27   * @key randomness
28   * @summary Test that the null values returned by the ThreadMXBean work.
29   * @author Eamonn McManus
30   *
31   * @run clean ThreadMXBeanTest
32   * @run build ThreadMXBeanTest
33   * @run main ThreadMXBeanTest
34   */
35  
36  import java.lang.management.*;
37  import java.util.*;
38  import javax.management.*;
39  
40  public class ThreadMXBeanTest {
41      public static void main(String[] args) throws Exception {
42          MBeanServer mbs = MBeanServerFactory.newMBeanServer();
43          ThreadMXBean tmb = ManagementFactory.getThreadMXBean();
44          StandardMBean smb = new StandardMBean(tmb, ThreadMXBean.class, true);
45          ObjectName on = new ObjectName("a:type=ThreadMXBean");
46          mbs.registerMBean(smb, on);
47          ThreadMXBean proxy = JMX.newMXBeanProxy(mbs, on, ThreadMXBean.class);
48          long[] ids1 = proxy.getAllThreadIds();
49  
50          // Add some random ids to the list so we'll get back null ThreadInfo
51          long[] ids2 = new long[ids1.length + 10];
52          System.arraycopy(ids1, 0, ids2, 0, ids1.length);
53          Random r = new Random();
54          for (int i = ids1.length; i < ids2.length; i++)
55              ids2[i] = Math.abs(r.nextLong());
56          // Following line produces an exception if null values not handled
57          ThreadInfo[] info = proxy.getThreadInfo(ids2);
58          boolean sawNull = false;
59          for (ThreadInfo ti : info) {
60              if (ti == null)
61                  sawNull = true;
62          }
63          if (!sawNull)
64              throw new Exception("No null value in returned array");
65          System.out.println("TEST PASSED");
66      }
67  }
