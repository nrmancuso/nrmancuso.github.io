24  /*
25   * @test
26   * @bug 6630275
27   * @summary Tests VetoableChangeSupport specification
28   * @author Sergey Malenkov
29   */
30  
31  import java.beans.PropertyChangeEvent;
32  import java.beans.PropertyVetoException;
33  import java.beans.VetoableChangeListener;
34  import java.beans.VetoableChangeSupport;
35  
36  public class Test6630275 {
37      private static final String PROPERTY = "property"; // NON-NLS: predefined property name
38  
39      public static void main(String[] args) {
40          CheckListener first = new CheckListener(false);
41          CheckListener second = new CheckListener(true);
42          CheckListener third = new CheckListener(false);
43  
44          VetoableChangeSupport vcs = new VetoableChangeSupport(Test6630275.class);
45          vcs.addVetoableChangeListener(first);
46          vcs.addVetoableChangeListener(PROPERTY, first);
47          vcs.addVetoableChangeListener(PROPERTY, second);
48          vcs.addVetoableChangeListener(PROPERTY, third);
49          try {
50              vcs.fireVetoableChange(PROPERTY, true, false);
51          } catch (PropertyVetoException exception) {
52              first.validate();
53              second.validate();
54              third.validate();
55              return; // expected exception
56          }
57          throw new Error("exception should be thrown");
58      }
59  
60      private static class CheckListener implements VetoableChangeListener {
61          private final boolean veto;
62          private boolean odd; // even/odd check for notification
63  
64          private CheckListener(boolean veto) {
65              this.veto = veto;
66          }
67  
68          private void validate() {
69              if (this.veto != this.odd)
70                  throw new Error(this.odd
71                          ? "undo event expected"
72                          : "unexpected undo event");
73          }
74  
75          public void vetoableChange(PropertyChangeEvent event) throws PropertyVetoException {
76              this.odd = !this.odd;
77              if (this.veto)
78                  throw new PropertyVetoException("disable all changes", event);
79          }
80      }
81  }
