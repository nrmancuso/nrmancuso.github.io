77      /*
78       * Create an X509Certificate then attempt to construct a
79       * CRLDistributionPointsExtension object from its extension value bytes.
80       */
81      private static void CRLDistributionPointsExtensionTest(String certStr)
82              throws Exception {
83          CertificateFactory cf = CertificateFactory.getInstance("X.509");
84          ByteArrayInputStream is = new ByteArrayInputStream(certStr.getBytes());
85          X509Certificate cert = (X509Certificate) cf.generateCertificate(is);
86  
87          // oid for CRL Distribution Points = 2.5.29.31
88          byte[] CDPExtBytes = cert.getExtensionValue("2.5.29.31");
89          DerValue val = new DerValue(CDPExtBytes);
90          byte[] data = val.getOctetString();
91          CRLDistributionPointsExtension CDPExt
92                  = new CRLDistributionPointsExtension(false, data);
93      }
94  
95      public static void main(String[] args) throws Exception {
96          /* Try to parse a CRLDistributionPointsExtension URI with a space. */
97          try {
98              CRLDistributionPointsExtensionTest(certWithSpaceInCDPStr);
99              throw new RuntimeException("Illegally parsed a "
100                     + "CRLDistributionPointsExtension uri with a space.");
101         } catch (IOException e) {
102             System.out.println("Caught the correct exception.");
103 
104         }
105 
106         /* Try to parse a CRLDistributionPointsExtension URI with backslashes. */
107         try {
108             CRLDistributionPointsExtensionTest(certWithBackslashesInCDPStr);
109             throw new RuntimeException("Illegally parsed a "
110                     + "CRLDistributionPointsExtension uri with a backslashes.");
111         } catch (IOException e) {
112             System.out.println("Caught the correct exception.");
113         }
114 
115         /* Try to construct a URIName from a uri with a space. */
116         String uriWithSpace = "file://crl file.crl";
117         URIName name;
118         try {
119             name = new URIName(uriWithSpace);
120             throw new RuntimeException("Illegally created a URIName "
121                     + "from a uri with a space.");
122         } catch (IOException e) {
123             System.out.println("Caught the correct exception.");
124         }
125 
126         /* Try to construct a URIName from a uri with backslashes. */
127         String uriWithBackslashes = "file://\\\\CRL\\crl_file.crl";
128         try {
129             name = new URIName(uriWithBackslashes);
130             throw new RuntimeException("Illegally created a URIName "
131                     + "from a uri with backslashes.");
132         } catch (IOException e) {
133             System.out.println("Caught the correct exception.");
134         }
135 
136         System.out.println("Tests passed.");
137     }
