24  import java.awt.*;
25  import java.awt.print.*;
26  import javax.print.*;
27  import javax.print.attribute.*;
28  
29  public class PrintSE implements Printable {
30  
31      public static void main(String[] args) throws Exception {
32          GraphicsEnvironment.getLocalGraphicsEnvironment();
33  
34          PrintService service = PrintServiceLookup.lookupDefaultPrintService();
35          if (service == null) {
36              System.out.println("No print service found.");
37              return;
38          }
39          SimpleDoc doc =
40               new SimpleDoc(new PrintSE(),
41                             DocFlavor.SERVICE_FORMATTED.PRINTABLE,
42                             new HashDocAttributeSet());
43          DocPrintJob job = service.createPrintJob();
44          job.print(doc, new HashPrintRequestAttributeSet());
45      }
46  
47      public int print(Graphics g, PageFormat pf, int pg) {
48         if (pg > 0) return NO_SUCH_PAGE;
49         g.drawString("Test passes.", 100, 100);
50         return PAGE_EXISTS;
51     }
52  }
