58      private static void test() throws Exception {
59          SwingUtilities.invokeAndWait(new Runnable() {
60              public void run() {
61                  final SystemTray tray = SystemTray.getSystemTray();
62                  final TrayIcon trayIcon = new TrayIcon(icon.getImage());
63                  try {
64                      tray.add(trayIcon);
65                  } catch (AWTException e) {
66                      throw new RuntimeException(
67                              "TrayIcon could not be added.");
68                  }
69  
70  
71                  try {
72                      trayIcon.displayMessage(createString(63, 'A'),
73                              createString(255, 'C'), TrayIcon.MessageType.ERROR);
74  
75                      trayIcon.setToolTip(createString(127, 'B'));
76  
77                      trayIcon.displayMessage(createString(64, 'A'),
78                              createString(256, 'C'), TrayIcon.MessageType.ERROR);
79  
80                      trayIcon.setToolTip(createString(128, 'B'));
81  
82                      trayIcon.displayMessage(createString(65, 'A'),
83                              createString(257, 'C'), TrayIcon.MessageType.ERROR);
84  
85                      trayIcon.setToolTip(createString(129, 'B'));
86                  }
87                  finally {
88                      tray.remove(trayIcon);
89                  }
90              }
91          });
92      }
93  
94      private static String createString(int len, char letter) {
95          char[] chars = new char[len];
96          Arrays.fill(chars, letter);
97          chars[len - 2] = '=';
98          chars[len - 1] = '>';
99          return new String(chars);
100     }
