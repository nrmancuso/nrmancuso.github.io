24  /* @test
25     @bug 4140445 4140451
26     @summary Test if constructor will check for illegal
27              arguments.
28  */
29  
30  
31  
32  import java.io.*;
33  
34  public class Constructor {
35      public static void main(String argv[]) throws Exception {
36          int values[] = {Integer.MIN_VALUE, -1, 0, 1, 4, 16, 31,
37                          32, 33, Integer.MAX_VALUE};
38          char b[][] = {null, new char[32]};
39  
40          int i = 0, j = 0, k = 0;
41          boolean nullPtr = false, indexOutBnd = false;
42  
43          for (i = 0; i < b.length; i++) {
44              for ( j = 0; j < values.length; j++) {
45                  for ( k = 0; k < values.length; k++) {
46  
47                      nullPtr = (b[i] == null);
48  
49                      int bufLen = nullPtr ? 0 : b[i].length;
50                      indexOutBnd = (values[j] < 0)
51                          || (values[j] > bufLen)
52                          || (values[k] < 0)
53                          || ((values[j] + values[k]) < 0);
54  
55                      try {
56                          CharArrayReader rdr = new CharArrayReader
57                              (b[i], values[j], values[k]);
58                      } catch (NullPointerException e) {
59                          if (!nullPtr) {
60                              throw new Exception
61                                  ("should not throw NullPointerException");
62                          }
63                          continue;
64                      } catch (IllegalArgumentException e) {
65                          if (!indexOutBnd) {
66                              throw new Exception
67                                  ("should not throw IllegalArgumentException");
68                          }
69                          continue;
70                      }
71  
72                      if (nullPtr || indexOutBnd) {
73                          throw new Exception("Failed to detect illegal argument");
74                      }
75                  }
76              }
77          }
78      }
79  }
