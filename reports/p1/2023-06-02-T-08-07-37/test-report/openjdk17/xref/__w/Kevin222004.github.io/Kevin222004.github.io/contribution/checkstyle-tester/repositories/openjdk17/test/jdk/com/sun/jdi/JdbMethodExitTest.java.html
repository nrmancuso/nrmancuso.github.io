48  class JdbMethodExitTestTarg {
49      // These are the values that will be returned by the methods
50      static URL[] urls = new URL[1];
51      public static byte      byteValue = 89;
52      public static char      charValue = 'x';
53      public static double    doubleValue = 2.2;
54      public static float     floatValue = 3.3f;
55      public static int       intValue = 1;
56      public static short     shortValue = 8;
57      public static boolean   booleanValue = false;
58  
59      public static Class       classValue = Object.class;
60      public static ClassLoader classLoaderValue;
61      {
62          try {
63              urls[0] = new URL("file:/foo");
64          } catch (java.net.MalformedURLException ee) {
65          }
66          classLoaderValue = new URLClassLoader(urls);
67      }
68  
69      public static Thread      threadValue;
70      public static ThreadGroup threadGroupValue;
71      public static String      stringValue = "abc";
72      public static int[]       intArrayValue = new int[] {1, 2, 3};
73  
74      public static JdbMethodExitTestTarg  objectValue =
75          new JdbMethodExitTestTarg();
76      public String ivar = stringValue;
77  
78      // These are the instance methods
79      public byte i_bytef()            { return byteValue; }
80      public char i_charf()            { return charValue; }
81      public double i_doublef()        { return doubleValue; }
82      public float i_floatf()          { return floatValue; }
83      public int i_intf()              { return intValue; }
84      public short i_shortf()          { return shortValue; }
85      public boolean i_booleanf()      { return booleanValue; }
86      public String i_stringf()        { return stringValue; }
87      public Class i_classf()          { return classValue; }
88      public ClassLoader i_classLoaderf()
89                                       { return classLoaderValue; }
90      public Thread i_threadf()        { return threadValue = Thread.currentThread(); }
91      public ThreadGroup i_threadGroupf()
92                                       { return threadGroupValue = threadValue.getThreadGroup(); }
93      public int[] i_intArrayf()       { return intArrayValue; }
94      public Object i_nullObjectf()    { return null; }
95      public Object i_objectf()        { return objectValue; }
96      public void i_voidf()            {}

98      static void doit(JdbMethodExitTestTarg xx) {
99  
100         xx.i_bytef();
101         xx.i_charf();
102         xx.i_doublef();
103         xx.i_floatf();
104         xx.i_intf();
105         xx.i_shortf();
106         xx.i_booleanf();
107         xx.i_stringf();
108         xx.i_intArrayf();
109         xx.i_classf();
110         xx.i_classLoaderf();
111         xx.i_threadf();
112         xx.i_threadGroupf();
113         xx.i_nullObjectf();
114         xx.i_objectf();
115         xx.i_voidf();
116 
117         // Prove it works for native methods too
118         StrictMath.sin(doubleValue);
119         stringValue.intern();
120     }

187 public class JdbMethodExitTest extends JdbTest {
188     public static void main(String argv[]) {
189         new JdbMethodExitTest().run();
190     }
191 
192     private JdbMethodExitTest() {
193         super(DEBUGGEE_CLASS);
194     }
195 
196     private static final String DEBUGGEE_CLASS = JdbMethodExitTestTarg.class.getName();
197 
198     @Override
199     protected void runCases() {
200         setBreakpointsFromTestSource("JdbMethodExitTest.java", 1);
201 
202         // test all possible return types
203         execCommand(JdbCommand.run())
204                 .shouldContain("Breakpoint hit");
205         Integer threadId = Integer.parseInt(
206                 new OutputAnalyzer(getDebuggeeOutput())
207                         .firstMatch("^threadid=(.*)$", 1));
208         jdb.command(JdbCommand.untrace());
209 
210         jdb.command(JdbCommand.traceMethods(false, null));
211         execCommand(JdbCommand.trace())
212                 .shouldContain("trace methods in effect");
213 
214         jdb.command(JdbCommand.traceMethods(true, null));
215         execCommand(JdbCommand.trace())
216                 .shouldContain("trace go methods in effect");
217 
218         jdb.command(JdbCommand.traceMethodExits(false, null));
219         execCommand(JdbCommand.trace())
220                 .shouldContain("trace method exits in effect");
221 
222         jdb.command(JdbCommand.traceMethodExits(true, null));
223         execCommand(JdbCommand.trace())
224                 .shouldContain("trace go method exits in effect");
225 
226         jdb.command(JdbCommand.traceMethodExit(false, null));
227         execCommand(JdbCommand.trace())
228                 .shouldContain("trace method exit in effect for JdbMethodExitTestTarg.bkpt");
229 
230         jdb.command(JdbCommand.traceMethodExit(true, null));
231         execCommand(JdbCommand.trace())
232                 .shouldContain("trace go method exit in effect for JdbMethodExitTestTarg.bkpt");
233 
234 
235         // trace exit of methods with all the return values
236         // (but just check a couple of them)
237         jdb.command(JdbCommand.traceMethodExits(true, threadId));
238         execCommand(JdbCommand.cont())
239                 .shouldContain("instance of JdbMethodExitTestTarg")
240                 .shouldContain("return value = 8");
241 
242         // Get out of bkpt back to the call to traceMethods
243         jdb.command(JdbCommand.stepUp());
244 
245 
246         jdb.command(JdbCommand.traceMethods(false, threadId));
247         execCommand(JdbCommand.cont())
248                 .shouldContain("Method entered:");
249         execCommand(JdbCommand.cont())
250                 .shouldContain("Method exited: return value = \"traceMethods\"");
251         jdb.command(JdbCommand.stepUp());
252 
253 
254         List<String> reply = new LinkedList<>();
255         reply.addAll(jdb.command(JdbCommand.traceMethods(true, threadId)));
256         reply.addAll(jdb.command(JdbCommand.cont()));
257         reply.addAll(jdb.command(JdbCommand.cont()));
258         reply.addAll(jdb.command(JdbCommand.cont()));
259         new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)))
260                 .shouldContain("Method entered: \"thread=main\", JdbMethodExitTestTarg.traceMethods1")
261                 .shouldMatch("Method exited: .* JdbMethodExitTestTarg.traceMethods1");
262         jdb.command(JdbCommand.untrace());
263         jdb.command(JdbCommand.stepUp());
264 
265 
266         reply.clear();
267         reply.addAll(jdb.command(JdbCommand.traceMethodExits(false, threadId)));
268         reply.addAll(jdb.command(JdbCommand.cont()));
269         new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)))
270                 .shouldContain("Method exited: return value = \"traceExits\"");
271         jdb.command(JdbCommand.untrace());
272         jdb.command(JdbCommand.stepUp());
273 
274 
275         reply.clear();
276         reply.addAll(jdb.command(JdbCommand.traceMethodExits(true, threadId)));
277         reply.addAll(jdb.command(JdbCommand.cont()));
278         new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)))
279                 .shouldMatch("Method exited: .* JdbMethodExitTestTarg.traceExits1");
280         jdb.command(JdbCommand.untrace());
281         jdb.command(JdbCommand.stepUp());
282 
283 
284         reply.clear();
285         reply.addAll(jdb.command(JdbCommand.step()));   // step into traceExit()
286         reply.addAll(jdb.command(JdbCommand.traceMethodExit(false, threadId)));
287         reply.addAll(jdb.command(JdbCommand.cont()));
288         new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)))
289                 .shouldContain("Method exited: return value = \"traceExit\"");
290         jdb.command(JdbCommand.untrace());
291         jdb.command(JdbCommand.stepUp());
292 
293 
294         reply.clear();
295         reply.addAll(jdb.command(JdbCommand.step()));
296         reply.addAll(jdb.command(JdbCommand.step()));   // skip over setting return value in caller :-(
297         reply.addAll(jdb.command(JdbCommand.traceMethodExit(true, threadId)));
298         reply.addAll(jdb.command(JdbCommand.cont()));
299         new OutputAnalyzer(reply.stream().collect(Collectors.joining(lineSeparator)))
300                 .shouldMatch("Method exited: .*JdbMethodExitTestTarg.traceExit1");
301 
302         new OutputAnalyzer(getJdbOutput())
303                 .shouldContain("Breakpoint hit");
304     }
