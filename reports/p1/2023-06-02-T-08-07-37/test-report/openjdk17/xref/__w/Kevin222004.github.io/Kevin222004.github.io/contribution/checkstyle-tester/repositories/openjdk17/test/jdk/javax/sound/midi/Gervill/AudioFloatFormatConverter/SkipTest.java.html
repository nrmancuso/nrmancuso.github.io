24  /* @test
25   @summary Test skip method returned from AudioFloatFormatConverter.getAudioInputStream
26   @modules java.desktop/com.sun.media.sound
27  */
28  
29  import java.io.ByteArrayInputStream;
30  
31  import javax.sound.sampled.AudioFormat;
32  import javax.sound.sampled.AudioInputStream;
33  
34  import com.sun.media.sound.AudioFloatFormatConverter;
35  
36  public class SkipTest {
37  
38      public static void main(String[] args) throws Exception {
39          AudioFloatFormatConverter converter = new AudioFloatFormatConverter();
40          byte[] data = { 10, 20, 30, 40, 30, 20, 10 };
41          AudioFormat format = new AudioFormat(8000, 8, 1, true, false);
42          AudioFormat format2 = new AudioFormat(16000, 8, 1, true, false);
43          AudioInputStream ais = new AudioInputStream(new ByteArrayInputStream(
44                  data), format, data.length);
45          AudioInputStream ais2 = converter.getAudioInputStream(format2, ais);
46          byte[] data2 = new byte[30];
47          int ret = ais2.read(data2, 0, data2.length);
48          ais.reset();
49          AudioInputStream ais3 = converter.getAudioInputStream(format2, ais);
50          byte[] data3 = new byte[100];
51          ais3.skip(7);
52          int ret2 = ais3.read(data3, 7, data3.length);
53          if (ret2 != ret - 7)
54              throw new Exception("Skip doesn't work correctly (" + ret2 + " != "
55                      + (ret - 7) + ")");
56          for (int i = 7; i < ret2 + 7; i++) {
57              if (data3[i] != data2[i])
58                  throw new Exception("Skip doesn't work correctly (" + data3[i]
59                          + " != " + data2[i] + ")");
60          }
61      }
62  
63  }
