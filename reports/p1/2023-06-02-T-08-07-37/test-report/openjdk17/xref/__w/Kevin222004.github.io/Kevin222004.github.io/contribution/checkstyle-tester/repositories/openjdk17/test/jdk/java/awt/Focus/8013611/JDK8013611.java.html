35  import java.awt.*;
36  import java.awt.event.*;
37  import javax.swing.*;
38  import test.java.awt.regtesthelpers.Util;
39  
40  import java.awt.*;
41  
42  public class JDK8013611 extends JFrame {
43      static JTextField textField = new JTextField("text");
44      static JButton button1 = new JButton("button1");
45      static JButton button2 = new JButton("button2");
46      static Robot robot;
47  
48      static JDialog dialog;
49      static JButton button3 = new JButton("button3");
50  
51      public static void main(String[] args) {
52          robot = Util.createRobot();
53  
54          JDK8013611 frame = new JDK8013611();
55          frame.setLayout(new FlowLayout());
56          frame.add(textField);
57          frame.add(button1);
58          frame.add(button2);
59          frame.pack();
60  
61          dialog = new JDialog(frame, true);
62          dialog.add(button3);
63          dialog.pack();
64  
65          textField.addFocusListener(new FocusAdapter() {
66              @Override
67              public void focusLost(FocusEvent e) {
68                  dialog.setVisible(true);
69              }
70          });
71  
72          button1.addFocusListener(new FocusAdapter() {
73              @Override
74              public void focusGained(FocusEvent e) {
75                  button2.requestFocusInWindow();
76              }
77          });
78  
79          frame.setVisible(true);
80  
81          frame.test();
82      }
83  
84      public void test() {
85          if (!testFocused(textField)) {
86              Util.clickOnComp(textField, robot);
87              if (!testFocused(textField)) {
88                  throw new RuntimeException("Error: couldn't focus " + textField);
89              }
90          }
91  
92          robot.keyPress(KeyEvent.VK_TAB);
93          robot.delay(50);
94          robot.keyRelease(KeyEvent.VK_TAB);
95  
96          if (!testFocused(button3)) {
97              throw new RuntimeException("Test failed: dialog didn't get focus!");
98          }
99  
100         System.out.println("Test passed.");
101     }
102 
103     boolean testFocused(Component c) {
104         for (int i=0; i<10; i++) {
105             if (KeyboardFocusManager.getCurrentKeyboardFocusManager().getFocusOwner() == c) {
106                 return true;
107             }
108             Util.waitForIdle(robot);
109         }
110         return false;
111     }
112 }
