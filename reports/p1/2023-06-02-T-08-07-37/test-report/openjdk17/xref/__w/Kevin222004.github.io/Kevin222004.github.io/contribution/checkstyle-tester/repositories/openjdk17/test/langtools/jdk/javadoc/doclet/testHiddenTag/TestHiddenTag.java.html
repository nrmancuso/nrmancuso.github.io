43      /**
44       * Perform tests on &#64;hidden tags
45       */

47      public void test1() {
48          javadoc("-d", "out1",
49                  "-sourcepath", testSrc,

53  
54          checkOutput("pkg1/A.html", true,
55                  """

63                      n pkg1">A.VisibleInnerExtendsInvisibleInner</a></code></dd>""");
64  
65          checkOutput("pkg1/A.html", false,
66                  "invisibleField",
67                  "invisibleMethod()");
68  
69          checkOutput("pkg1/A.VisibleInner.html", true,
70                  """
71                      <code><a href="A.html#visibleField">visibleField</a></code>""",
72                  """

74                  """
75                      <h2 id="nested-classes-inherited-from-class-pkg1.A">Nested classes/interfaces in\
76                      herited from class&nbsp;pkg1.<a href="A.html" title="class in pkg1">A</a></h2>
77                      <code><a href="A.VisibleInner.html" title="class in pkg1">A.VisibleInner</a>, <a\
78                       href="A.VisibleInnerExtendsInvisibleInner.html" title="class in pkg1">A.Visible\
79                      InnerExtendsInvisibleInner</a></code></div>
80                      """);

83                  "../pkg1/A.VisibleInner.html#VisibleInner()",

85                  "invisibleMethod()");

88                  """

90                      span class="element-name type-name-label">A.VisibleInnerExtendsInvisibleInner</span>
91                      <span class="extends-implements">extends <a href="A.html" title="class in pkg1">A</a></span></div>""",
92                  """
93                      <code><a href="A.html#visibleField">visibleField</a></code>""",
94                  """

98                  "invisibleField",
99                  "invisibleMethod",
100                 "A.InvisibleInner");
101 
102         checkOutput("pkg1/Intf.html", true,
103                 """
104                     <section class="detail" id="visibleDefaultMethod()">""",
105                 """

110                     </dl>""");
111

113                 "InvisibleParent",
114                 "invisibleDefaultMethod",
115                 "invisibleInterfaceMethod");
116

118                 """

121                 """

147                     </dl>""");
