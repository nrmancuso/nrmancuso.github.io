24  import java.awt.datatransfer.Transferable;
25  import java.awt.datatransfer.DataFlavor;
26  import java.awt.datatransfer.UnsupportedFlavorException;
27  import java.io.File;
28  import java.io.IOException;
29  
30  class FileListTransferable implements Transferable {
31  
32      final private DataFlavor[] supportedFlavors =
33              {DataFlavor.javaFileListFlavor};
34  
35      private java.util.List<File> list;
36  
37      public FileListTransferable(java.util.List<File> list) {
38          this.list = list;
39      }
40  
41      public DataFlavor[] getTransferDataFlavors() {
42          return supportedFlavors;
43      }
44  
45      public boolean isDataFlavorSupported(DataFlavor flavor) {
46          for (DataFlavor df:supportedFlavors) {
47              if (df.equals(flavor)) return true;
48          }
49          return false;
50      }
51  
52      public Object getTransferData(DataFlavor flavor) throws UnsupportedFlavorException, IOException {
53          if (flavor.equals(DataFlavor.javaFileListFlavor)) {
54              return list;
55          }
56          throw new UnsupportedFlavorException(flavor);
57      }
58  }
