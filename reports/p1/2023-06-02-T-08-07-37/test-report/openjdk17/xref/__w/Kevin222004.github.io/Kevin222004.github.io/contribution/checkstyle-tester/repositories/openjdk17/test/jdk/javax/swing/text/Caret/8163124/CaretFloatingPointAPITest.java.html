72              + "If the display does not support HiDPI mode press PASS.\n\n"

74              + " in the text field.\n"

80          countDownLatch = new CountDownLatch(1);

167     static class CustomCaret implements Caret {
168 
169         private JTextComponent component;
170         private boolean visible;
171         private boolean selectionVisible = true;
172         int blinkRate;
173         int dot;
174         int mark;
175         Position.Bias dotBias;
176         Position.Bias markBias;
177         Object selectionTag;
178         Point2D magicCaretPosition;
179 
180         private MouseListener mouseListener = new CaretMouseListener();
181 
182         @Override
183         public void install(JTextComponent c) {
184             this.component = c;
185             c.addMouseListener(mouseListener);
186         }
187 
188         @Override
189         public void deinstall(JTextComponent c) {
190             c.removeMouseListener(mouseListener);
191             this.component = null;
192         }
193 
194         @Override
195         public void paint(Graphics g) {
196 
197             if (component == null) {
198                 return;
199             }
200 
201             int dot = getDot();
202             Rectangle2D r = null;
203             try {
204                 r = component.modelToView2D(dot);
205             } catch (BadLocationException e) {
206                 return;
207             }
208 
209             if (r == null) {
210                 return;
211             }
212 
213             Rectangle2D cr = getCaretRectangle(r);
214             repaint(cr.getBounds());
215 
216             g.setColor(component.getCaretColor());
217             float cx = (float) cr.getX();
218             float cy = (float) cr.getY();
219             float cw = (float) cr.getWidth();
220             float ch = (float) cr.getHeight();
221             float c = cx + cw / 2;
222 
223             Graphics2D g2d = (Graphics2D) g;
224             g2d.draw(new Line2D.Float(c, cy, c, cy + ch));
225             g2d.draw(new Line2D.Float(cx, cy, cx + cw, cy));
226             g2d.draw(new Line2D.Float(cx, cy + ch, cx + cw, cy + ch));
227         }
228 
229         void repaint(Rectangle r) {
230             component.repaint(r);
231         }
232 
233         Rectangle2D getCaretRectangle(Rectangle2D r) {
234             int d = 3;
235             double cx = r.getX() - d;
236             double cy = r.getY();
237             double cw = 2 * d;
238             double ch = r.getHeight();
239             return new Rectangle2D.Double(cx, cy, cw, ch);
240         }

243         public void addChangeListener(ChangeListener l) {
244         }

247         public void removeChangeListener(ChangeListener l) {
248         }

251         public boolean isVisible() {
252             return visible;

265         @Override
266         public void setSelectionVisible(boolean v) {
267             this.selectionVisible = v;
268             updateSelection();
269         }
270 
271         @Override
272         public void setMagicCaretPosition(Point p) {
273             magicCaretPosition = p;
274         }
275 
276         @Override
277         public Point getMagicCaretPosition() {
278             if (magicCaretPosition != null) {
279                 return new Point((int) magicCaretPosition.getX(),
280                                  (int) magicCaretPosition.getY());
281             }
282             return null;
283         }

297             return dot;
298         }
299 
300         @Override
301         public int getMark() {
302             return mark;
303         }
304 
305         @Override
306         public void setDot(int dot) {
307             setDot(dot, Position.Bias.Forward);
308         }
309 
310         private void setDot(int dot, Position.Bias bias) {
311             handleSetDot(dot, bias);
312             updateSelection();
313         }
314 
315         @Override
316         public void moveDot(int dot) {
317             moveDot(dot, Position.Bias.Forward);
318         }
319 
320         private void moveDot(int dot, Position.Bias bias) {
321             changeCaretPosition(dot, bias);
322             updateSelection();
323         }
324 
325         void handleSetDot(int dot, Position.Bias dotBias) {
326 
327             if (component == null) {
328                 return;
329             }
330 
331             Document doc = component.getDocument();
332             if (doc != null) {
333                 dot = Math.min(dot, doc.getLength());
334             }
335 
336             dot = Math.max(dot, 0);
337 
338             if (dot == 0) {
339                 dotBias = Position.Bias.Forward;
340             }
341 
342             mark = dot;
343 
344             if (this.dot != dot || this.dotBias != dotBias) {
345                 changeCaretPosition(dot, dotBias);
346                 updateSelection();
347             }
348 
349             this.markBias = this.dotBias;
350         }
351 
352         void changeCaretPosition(int dot, Position.Bias dotBias) {
353             this.dot = dot;
354             this.dotBias = dotBias;
355             setMagicCaretPosition(null);
356             SwingUtilities.invokeLater(this::repaintNewCaret);
357         }
358 
359         private void updateSelection() {
360             Highlighter h = component.getHighlighter();
361             if (h != null) {
362                 int p0 = Math.min(dot, mark);
363                 int p1 = Math.max(dot, mark);
364 
365                 if (p0 == p1 || !selectionVisible) {
366                     if (selectionTag != null) {
367                         h.removeHighlight(selectionTag);
368                         selectionTag = null;
369                     }
370                 } else {
371                     try {
372                         if (selectionTag != null) {
373                             h.changeHighlight(selectionTag, p0, p1);
374                         } else {
375                             Highlighter.HighlightPainter p = getSelectionPainter();
376                             selectionTag = h.addHighlight(p0, p1, p);
377                         }
378                     } catch (BadLocationException e) {
379                         throw new RuntimeException(e);
380                     }
381                 }
382             }
383         }
384 
385         void repaintNewCaret() {
386             if (component != null) {
387                 TextUI mapper = component.getUI();
388                 Document doc = component.getDocument();
389                 if ((mapper != null) && (doc != null)) {
390                     Rectangle2D newLoc;
391                     try {
392                         newLoc = mapper.modelToView2D(component, this.dot, this.dotBias);
393                     } catch (BadLocationException e) {
394                         newLoc = null;
395                     }
396                     if (newLoc != null) {
397                         adjustVisibility(newLoc.getBounds());
398                         if (getMagicCaretPosition() == null) {
399                             setMagicCaretPosition(new Point((int) newLoc.getX(),
400                                                             (int) newLoc.getY()));
401                         }
402                     }
403                     damage(newLoc.getBounds());
404                 }
405             }
406         }
407 
408         protected Highlighter.HighlightPainter getSelectionPainter() {
409             return DefaultHighlighter.DefaultPainter;
410         }
411 
412         protected void adjustVisibility(Rectangle nloc) {
413             if (component == null) {
414                 return;
415             }
416             if (SwingUtilities.isEventDispatchThread()) {
417                 component.scrollRectToVisible(nloc);
418             } else {
419                 SwingUtilities.invokeLater(() -> {
420                     component.scrollRectToVisible(nloc);
421                 });
422             }
423         }
424 
425         protected synchronized void damage(Rectangle r) {
426             if (r != null && component != null) {
427                 component.repaint(r);
428             }
429         }

431         private class CaretMouseListener extends MouseAdapter {
432 
433             @Override
434             public void mousePressed(MouseEvent e) {
435                 Point pt = new Point(e.getX(), e.getY());
436                 Position.Bias[] biasRet = new Position.Bias[1];
437                 int pos = component.getUI().viewToModel(component, pt, biasRet);
438                 if (biasRet[0] == null) {
439                     biasRet[0] = Position.Bias.Forward;
440                 }
441                 if (pos >= 0) {
442                     setDot(pos);
443                 }
444             }
445         }
446     }
