24  /*
25   * @test
26   * @bug 4157921
27   * @summary Test subclasses of BasicPermission to make sure different
28   * subclasses don't equal or imply each other.
29   */
30  
31  import java.security.Permission;
32  import java.security.BasicPermission;
33  
34  class A extends BasicPermission {
35      public A(String name) { super(name); }
36  }
37  
38  class B extends BasicPermission {
39      public B(String name) { super(name); }
40  }
41  
42  public class EqualsImplies {
43  
44      public static void main(String[]args) throws Exception {
45  
46        Permission p1 = new A("foo");
47        Permission p2 = new B("foo");
48  
49        if (p1.implies(p2) || p2.implies(p1) || p1.equals(p2)) {
50            throw new Exception("Test failed");
51        }
52  
53        // make sure permissions imply and equal themselves
54        if (! (p1.implies(p1) && p1.equals(p1))) {
55            throw new Exception("Test failed");
56        }
57  
58      }
59  }
