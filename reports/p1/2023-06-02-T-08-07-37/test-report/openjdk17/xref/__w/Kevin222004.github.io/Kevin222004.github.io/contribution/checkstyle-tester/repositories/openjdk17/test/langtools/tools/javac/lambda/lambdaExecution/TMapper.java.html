24  /**
25   * Given an input object maps to an appropriate output object. A mapper may
26   * variously provide a mapping between types, object instances or keys and
27   * values or any other form of transformation upon the input.
28   *
29   * <p/>All mapper implementations are expected to:
30   * <ul>
31   *  <li>Provide stable results such that for any {@code t} the result of two
32   * {@code map} operations are always equivalent. ie.<pre>
33   * Foo one = mapper.map(a);
34   * Foo two = mapper.map(a);
35   *
36   * assert one.equals(two) && two.equals(one);
37   * </pre></li>
38   * <li>Equivalent input objects should map to equivalent output objects. ie.<pre>
39   * assert a.equals(b);  // a and b are equivalent
40   *
41   * Foo x = mapper.map(a);
42   * Foo y = mapper.map(b);
43   *
44   * assert x.equals(y); // their mapped results should be as equivalent.
45   * </pre></li>
46   * <li>The mapper should not modify the input object in any way that would
47   * change the mapping.</li>
48   * <li>When used for aggregate operations upon many elements mappers
49   * should not assume that the {@code map} operation will be called upon elements
50   * in any specific order.</li>
51   * </ul>
52   *
53   * @param <R> the type of output objects from {@code map} operation. May be the
54   * @param <T> the type of input objects provided to the {@code map} operation.
55   * same type as {@code <T>}.
56   */
57  public interface TMapper<R, T> {
58  
59      /**
60       * Map the provided input object to an appropriate output object.
61       *
62       * @param t the input object to be mapped.
63       * @return the mapped output object.
64       */
65      R map(T t);
66  }
