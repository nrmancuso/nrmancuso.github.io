239     private void shapedCbActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_shapedCbActionPerformed

241             Shape s = null;
242             if (shapedCb.isSelected()) {

247             testFrame.setShape(s);

314     private void checkEffects() {
315         GraphicsDevice gd = getGraphicsConfiguration().getDevice();
316         if (!gd.isWindowTranslucencySupported(WindowTranslucency.PERPIXEL_TRANSPARENT)) {
317             shapedCb.setEnabled(false);
318         }
319         if (!gd.isWindowTranslucencySupported(WindowTranslucency.TRANSLUCENT)) {
320             transparencySld.setEnabled(false);
321         }
322         if (!gd.isWindowTranslucencySupported(WindowTranslucency.PERPIXEL_TRANSLUCENT)) {
323             nonOpaqueChb.setEnabled(false);
324         }
325     }
