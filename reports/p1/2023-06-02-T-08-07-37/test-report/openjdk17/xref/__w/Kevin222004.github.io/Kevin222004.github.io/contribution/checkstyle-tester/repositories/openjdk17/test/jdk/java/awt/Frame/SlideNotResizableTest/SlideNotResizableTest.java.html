24  import java.awt.*;
25  import java.awt.Dimension;
26  import java.awt.Point;
27  import java.awt.event.InputEvent;
28  
29  /**
30   * @test
31   * @key headful
32   * @bug 8032595
33   * @summary setResizable(false) makes a frame slide down
34   * @author Petr Pchelko
35   */
36  
37  public class SlideNotResizableTest {
38  
39      private static volatile boolean passed = false;
40      private static final Dimension FRAME_SIZE = new Dimension(100, 100);
41      private static final Point FRAME_LOCATION = new Point(200, 200);
42  
43      public static void main(String[] args) throws Throwable {
44          Frame aFrame = null;
45          try {
46              aFrame = new Frame();
47              aFrame.setSize(FRAME_SIZE);
48              aFrame.setLocation(FRAME_LOCATION);
49              aFrame.setResizable(false);
50              aFrame.setVisible(true);
51  
52              sync();
53  
54              if (!aFrame.getLocation().equals(FRAME_LOCATION)) {
55                  throw new RuntimeException("FAILED: Wrong frame position");
56              }
57          } finally {
58              if (aFrame != null) {
59                  aFrame.dispose();
60              }
61          }
62      }
63  
64      private static void sync() throws Exception {
65          Robot robot = new Robot();
66          robot.waitForIdle();
67          Thread.sleep(1000);
68      }
69  }
