222     /**
223      * take removes existing elements until empty, then blocks interruptibly
224      */
225     public void testBlockingTake() throws InterruptedException {
226         final BlockingQueue<Item> q = populatedQueue(SIZE);
227         final CountDownLatch pleaseInterrupt = new CountDownLatch(1);
228         Thread t = newStartedThread(new CheckedRunnable() {
229             public void realRun() throws InterruptedException {
230                 for (int i = 0; i < SIZE; i++) mustEqual(i, q.take());
231 
232                 Thread.currentThread().interrupt();
233                 try {
234                     q.take();
235                     shouldThrow();
236                 } catch (InterruptedException success) {}
237                 assertFalse(Thread.interrupted());
238 
239                 pleaseInterrupt.countDown();
240                 try {
241                     q.take();
242                     shouldThrow();
243                 } catch (InterruptedException success) {}
244                 assertFalse(Thread.interrupted());
245             }});
246 
247         await(pleaseInterrupt);
248         if (randomBoolean()) assertThreadBlocks(t, Thread.State.WAITING);
249         t.interrupt();
250         awaitTermination(t);
251     }

726     /**
