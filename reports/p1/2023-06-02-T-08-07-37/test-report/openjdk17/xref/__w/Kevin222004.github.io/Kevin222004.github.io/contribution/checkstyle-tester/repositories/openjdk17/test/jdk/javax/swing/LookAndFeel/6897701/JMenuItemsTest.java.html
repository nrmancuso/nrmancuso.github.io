24  /*
25   * @test
26   * @key headful
27   * @bug 6897701
28   * @summary Verify JMenu and JMenuItem Disabled state for Nimbus LAF
29   * @run main JMenuItemsTest
30   */
31  
32  import java.awt.Color;
33  import java.awt.Rectangle;
34  import java.awt.image.BufferedImage;
35  import javax.swing.JFrame;
36  import javax.swing.JMenu;
37  import javax.swing.JMenuBar;
38  import javax.swing.JMenuItem;
39  import javax.swing.SwingUtilities;
40  import javax.swing.UIManager;
41  
42  public class JMenuItemsTest {
43  
44      private static JFrame mainFrame;
45      private static JMenu disabledMenu;
46      private static JMenuItem disabledMenuItem;
47  
48      public JMenuItemsTest() {
49          createUI();
50      }
51  
52      private void createUI() {
53  
54          mainFrame = new JFrame("Test");
55  
56          disabledMenu = new JMenu("Disabled Menu");
57          disabledMenu.setForeground(Color.BLUE);
58          disabledMenu.setEnabled(false);
59  
60          disabledMenuItem = new JMenuItem("Disabled MenuItem");
61          disabledMenuItem.setForeground(Color.BLUE);
62          disabledMenuItem.setEnabled(false);
63  
64          JMenuBar menuBar = new JMenuBar();
65          menuBar = new JMenuBar();
66          menuBar.add(disabledMenu);
67          menuBar.add(disabledMenuItem);
68  
69          mainFrame.add(menuBar);
70          mainFrame.pack();
71          mainFrame.setVisible(true);
72      }
73  
74      private void dispose() {
75          mainFrame.dispose();
76      }
77  
78      private void testDisabledStateOfJMenu() {
79  
80          // Test disabled JMenu state
81          Rectangle rect = disabledMenu.getBounds();
82          BufferedImage image = new BufferedImage(rect.width, rect.height,
83                  BufferedImage.TYPE_INT_ARGB);
84          disabledMenu.paint(image.getGraphics());
85          int y = image.getHeight() / 2;
86          for (int x = 0; x < image.getWidth(); x++) {
87              Color c = new Color(image.getRGB(x, y));
88              if (c.equals(Color.BLUE)) {
89                  dispose();
90                  throw new RuntimeException("JMenu Disabled"
91                          + " State not Valid.");
92              }
93          }
94  
95      }
96  
97      private void testDisabledStateOfJMenuItem() {
98  
99          // Test disabled JMenuItem state
100         Rectangle rect = disabledMenuItem.getBounds();
101         BufferedImage image = new BufferedImage(rect.width, rect.height,
102                 BufferedImage.TYPE_INT_ARGB);
103         disabledMenuItem.paint(image.getGraphics());
104         int y = image.getHeight() / 2;
105         for (int x = 0; x < image.getWidth(); x++) {
106             Color c = new Color(image.getRGB(x, y));
107             if (c.equals(Color.BLUE)) {
108                 dispose();
109                 throw new RuntimeException("JMenuItem Disabled"
110                         + " State not Valid.");
111             }
112         }
113 
114     }
115 
116     public static void main(String[] args) throws Exception {
117         UIManager.setLookAndFeel("javax.swing.plaf.nimbus.NimbusLookAndFeel");
118         SwingUtilities.invokeAndWait(() -> {
119 
120             try {
121                 JMenuItemsTest obj = new JMenuItemsTest();
122                 obj.testDisabledStateOfJMenu();
123                 obj.testDisabledStateOfJMenuItem();
124                 obj.dispose();
125 
126             } catch (Exception ex) {
127                 throw ex;
128             }
129 
130         });
131     }
132 }
