24  /*
25   * @test
26   * @bug 6271396 8008577
27   * @modules jdk.localedata
28   * @summary Test case for verifying typo of timezone display name Australia/Lord_Howe
29   * @run main/othervm -Djava.locale.providers=JRE,SPI Bug6271396
30   */
31  
32  import java.util.Locale;
33  import java.util.TimeZone;
34  
35  public class Bug6271396 {
36  
37      public static void main(String[] args) {
38  
39          TimeZone Lord_Howe = TimeZone.getTimeZone("Australia/Lord_Howe");
40          Locale tzLocale = new Locale("fr");
41  
42          if (!Lord_Howe.getDisplayName(false, TimeZone.LONG, tzLocale).equals
43             ("Heure standard de Lord Howe"))
44               throw new RuntimeException("\n" + tzLocale + ": LONG, " +
45                                          "non-daylight saving name for " +
46                                          "Australia/Lord_Howe should be " +
47                                          "\"Heure standard de Lord Howe\"");
48          if (!Lord_Howe.getDisplayName(true, TimeZone.LONG, tzLocale).equals
49             ("Heure d'\u00e9t\u00e9 de Lord Howe"))
50               throw new RuntimeException("\n" + tzLocale + ": LONG, " +
51                                          "daylight saving name for " +
52                                          "Australia/Lord_Howe should be " +
53                                          "\"Heure d'\u00e9t\u00e9 de Lord Howe\"");
54  
55          tzLocale = new Locale("zh", "TW");
56          if (!Lord_Howe.getDisplayName(false, TimeZone.LONG, tzLocale).equals
57             ("\u8c6a\u52f3\u7235\u5cf6\u6a19\u6e96\u6642\u9593"))
58               throw new RuntimeException("\n" + tzLocale + ": LONG, " +
59                                          "non-daylight saving name for " +
60                                          "Australia/Lord_Howe should be " +
61                                          "\"\u8c6a\u52f3\u7235\u5cf6" +
62                                          "\u6a19\u6e96\u6642\u9593\"");
63          if (!Lord_Howe.getDisplayName(true, TimeZone.LONG, tzLocale).equals
64             ("\u8c6a\u52f3\u7235\u5cf6\u590f\u4ee4\u6642\u9593"))
65               throw new RuntimeException("\n" + tzLocale + ": LONG, " +
66                                          "daylight saving name for " +
67                                          "Australia/Lord_Howe should be " +
68                                          "\"\u8c6a\u52f3\u7235\u5cf6" +
69                                          "\u590f\u4ee4\u6642\u9593\"");
70     }
71  }
