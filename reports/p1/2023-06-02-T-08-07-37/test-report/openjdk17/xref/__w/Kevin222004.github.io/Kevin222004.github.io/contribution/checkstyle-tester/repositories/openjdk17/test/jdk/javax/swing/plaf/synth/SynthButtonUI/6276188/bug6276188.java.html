24   * @test 1.4 08/08/05
25   * @key headful
26   * @bug 6276188
27   * @library ../../../../regtesthelpers
28   * @build Util
29   * @author Romain Guy
30   * @summary Tests PRESSED and MOUSE_OVER and FOCUSED state for buttons with Synth.
31   * @run main bug6276188
32   */
33  import java.awt.*;
34  import java.awt.image.*;
35  import java.awt.event.*;
36  
37  import javax.swing.*;
38  import javax.swing.plaf.synth.*;
39  
40  public class bug6276188 extends JFrame {
41  
42      private static JButton button;
43      private static Point p;
44  
45      public static void main(String[] args) throws Throwable {
46          SynthLookAndFeel lookAndFeel = new SynthLookAndFeel();
47          lookAndFeel.load(bug6276188.class.getResourceAsStream("bug6276188.xml"), bug6276188.class);
48  
49          UIManager.setLookAndFeel(lookAndFeel);
50          SwingUtilities.invokeAndWait(new Runnable() {
51              public void run() {
52                  JFrame testFrame = new JFrame();
53                  testFrame.setLayout(new BorderLayout());
54                  testFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
55                  testFrame.add(BorderLayout.CENTER, button = new JButton());
56  
57                  testFrame.setSize(new Dimension(320, 200));
58                  testFrame.setVisible(true);
59              }
60          });
61  
62          Robot robot = new Robot();
63          robot.setAutoDelay(50);
64          robot.waitForIdle();
65          robot.delay(200);
66  
67          p = Util.getCenterPoint(button);
68  
69          robot.mouseMove(p.x , p.y);
70          robot.mousePress(InputEvent.BUTTON1_MASK);
71          robot.waitForIdle();
72          robot.delay(1000);
73  
74          Color color = robot.getPixelColor(p.x, p.y);
75          robot.mouseRelease(InputEvent.BUTTON1_MASK);
76          boolean red = color.getRed() > 0 && color.getGreen() == 0 && color.getBlue() == 0;
77          if (!red) {
78              System.err.println("Red: " + color.getRed() + "; Green: " + color.getGreen() + "; Blue: " + color.getBlue());
79              throw new RuntimeException("Synth ButtonUI does not handle PRESSED & MOUSE_OVER state");
80          }
81      }
82  }
