23  /*
24   * @test
25   * @bug 8154218
26   * @summary  Verifies if owner Frame is associated with print dialog
27   * @run main/manual TestPrintJobFrameAssociation
28   */
29  import java.awt.BorderLayout;
30  import java.awt.Button;
31  import java.awt.FlowLayout;
32  import java.awt.Frame;
33  import java.awt.Label;
34  import java.awt.Panel;
35  import java.awt.JobAttributes;
36  import java.awt.PrintJob;
37  import java.awt.Toolkit;
38  import java.awt.event.WindowAdapter;
39  import java.awt.event.WindowEvent;
40  import javax.swing.JButton;
41  import javax.swing.JDialog;
42  import javax.swing.JPanel;
43  import javax.swing.JTextArea;
44  import javax.swing.SwingUtilities;
45  
46  public class TestPrintJobFrameAssociation {
47      private static Thread mainThread;
48      private static boolean testPassed;
49      private static boolean testGeneratedInterrupt;
50      private static Button print;
51      private static Label dialogName;
52      private static Frame frame;
53      private static boolean start;
54      private static Thread t;
55  
56      public static void main(String args[]) throws Exception {
57          SwingUtilities.invokeAndWait(() -> {
58              doTest(TestPrintJobFrameAssociation::frameTest);
59          });
60          mainThread = Thread.currentThread();
61          try {
62              Thread.sleep(60000);
63          } catch (InterruptedException e) {
64              if (!testPassed && testGeneratedInterrupt) {
65                  throw new RuntimeException("Print dialog not disposed."
66                          + " Print dialog is not associated with owner Frame`");
67              }
68          }
69          if (!testGeneratedInterrupt) {
70              throw new RuntimeException("user has not executed the test");
71          }
72      }
73  
74      private static void frameTest() {
75          Panel panel =new Panel();
76  
77          print = new Button("Print");
78          print.setActionCommand("Print");
79          print.addActionListener((e) -> {
80              JobAttributes ja = new JobAttributes();
81              ja.setDialog(JobAttributes.DialogType.COMMON);
82  
83              t.start();
84              start = true;
85              PrintJob pjob = Toolkit.getDefaultToolkit().getPrintJob(frame,
86                                                           "Printing Test",
87                                                           ja,null);
88          });
89  
90          panel.add(print);
91  
92          frame = new Frame("Test Frame");
93          frame.setLayout (new BorderLayout ());
94          frame.add(panel,"South");
95          frame.pack();
96          frame.setVisible(true);
97  
98          t = new Thread (() -> {
99              if (start) {
100                 try {
101                     Thread.sleep(5000);
102                 } catch (InterruptedException ex) {}
103                 frame.dispose();
104             }
105         });
106     }
107 
108     public static synchronized void pass() {
109         testPassed = true;
110         testGeneratedInterrupt = true;
111         mainThread.interrupt();
112     }
113 
114     public static synchronized void fail() {
115         testPassed = false;
116         testGeneratedInterrupt = true;
117         mainThread.interrupt();
118     }
119 
120     private static void doTest(Runnable action) {
121         String description
122                 = " A Frame with Print Button is shown. Press Print.\n"
123                 + " A cross-platform print dialog will be shown. After 5 secs\n"
124                 + " the frame along with this print dialog will be disposed.\n"
125                 + " If the print dialog is not disposed, press FAIL else press PASS";
126 
127         final JDialog dialog = new JDialog();
128         dialog.setTitle("printSelectionTest");
129         JTextArea textArea = new JTextArea(description);
130         textArea.setEditable(false);
131         final JButton testButton = new JButton("Start Test");
132         final JButton passButton = new JButton("PASS");
133         passButton.setEnabled(false);
134         passButton.addActionListener((e) -> {
135             dialog.dispose();
136             pass();
137         });
138         final JButton failButton = new JButton("FAIL");
139         failButton.setEnabled(false);
140         failButton.addActionListener((e) -> {
141             dialog.dispose();
142             fail();
143         });
144         testButton.addActionListener((e) -> {
145             testButton.setEnabled(false);
146             action.run();
147             passButton.setEnabled(true);
148             failButton.setEnabled(true);
149         });
150         JPanel mainPanel = new JPanel(new BorderLayout());
151         mainPanel.add(textArea, BorderLayout.CENTER);
152         JPanel buttonPanel = new JPanel(new FlowLayout());
153         buttonPanel.add(testButton);
154         buttonPanel.add(passButton);
155         buttonPanel.add(failButton);
156         mainPanel.add(buttonPanel, BorderLayout.SOUTH);
157         dialog.add(mainPanel);
158         dialog.pack();
159         dialog.setVisible(true);
160         dialog.addWindowListener(new WindowAdapter() {
161             @Override
162             public void windowClosing(WindowEvent e) {
163                 System.out.println("main dialog closing");
164                 testGeneratedInterrupt = false;
165                 mainThread.interrupt();
166             }
167         });
168 
169     }
