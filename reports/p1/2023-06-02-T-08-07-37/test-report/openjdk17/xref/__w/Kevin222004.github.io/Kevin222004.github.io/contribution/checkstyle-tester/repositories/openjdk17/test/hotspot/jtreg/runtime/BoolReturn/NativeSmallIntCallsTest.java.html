24  /* @test
25   * @bug 8149170
26   * @summary Test native functions return booleans as 0/1 but differently than java functions
27   * @modules java.base/jdk.internal.misc
28   * @library /test/lib
29   * @compile BoolConstructor.java
30   * @run main/native NativeSmallIntCallsTest
31   */
32  
33  // This test shows that returns from native calls for boolean truncate to JNI_TRUE if value != 0
34  // and JNI_FALSE if value returned is 0.
35  
36  import jdk.test.lib.Asserts;
37  
38  public class NativeSmallIntCallsTest {
39      static native boolean nativeCastToBoolCallTrue();
40      static native boolean nativeCastToBoolCallFalse();
41      static native boolean nativeCallBoolConstructor(int visible, boolean expected);
42      static {
43          System.loadLibrary("NativeSmallIntCalls");
44      }
45  
46      public static void main(java.lang.String[] unused) throws Exception {
47          // Call through jni
48          // JNI makes all results != 0 true for compatibility
49          boolean nativeTrueVal = nativeCastToBoolCallTrue();
50          Asserts.assertTrue(nativeTrueVal, "trueval is false!");
51  
52          boolean nativeFalseVal = nativeCastToBoolCallFalse();
53          Asserts.assertTrue(nativeFalseVal, "falseval is false in native!");
54  
55          // Call a constructor or method that passes jboolean values into Java from native
56          nativeCallBoolConstructor(1, true);
57          nativeCallBoolConstructor(0x10, true);
58          nativeCallBoolConstructor(0x100, false);
59          nativeCallBoolConstructor(0x1000, false);
60      }
61  }
