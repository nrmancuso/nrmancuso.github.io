2   import java.util.Comparator;
3   import java.util.Optional;
4   import java.util.function.BinaryOperator;
5   import java.util.function.Consumer;
6   import java.util.stream.Stream;
7   
8   class Bug {
9   
10    public static void main(String[] args) {
11      Stream<String> tokenStream = Stream.of("hello", "world");
12  
13      Comparator<String> order = Comparator.naturalOrder();
14      Min<String> min = tokenStream.collect(
15        () -> new Min<>(order),
16        Min::accept,
17        Min::combine);
18  
19      System.out.printf("Min = %s%n",
20                        min.asOptional().orElse(null));
21    }
22  
23    static class Min<T> implements Consumer<T> {
24  
25      private final BinaryOperator<T> minOf;
26      private T min;
27  
28      public Min(Comparator<? super T> order) {
29        this.minOf = BinaryOperator.minBy(order);
30      }
31  
32      @Override
33      public void accept(T t) {
34        this.min = min != null ? minOf.apply(min, t) : t;
35      }
36  
37      public void combine(Min<T> other) {
38        if (other.min != null) {
39          accept(other.min);
40        }
41      }
42  
43      public Optional<T> asOptional() {
44        return Optional.ofNullable(min);
45      }
46    }
47  }
