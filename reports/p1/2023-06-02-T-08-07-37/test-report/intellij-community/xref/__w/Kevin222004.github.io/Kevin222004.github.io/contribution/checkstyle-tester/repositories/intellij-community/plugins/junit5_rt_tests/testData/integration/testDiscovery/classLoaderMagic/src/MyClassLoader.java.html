1   import java.io.ByteArrayOutputStream;

8     public MyClassLoader(ClassLoader extClassLoader) {
9       super(extClassLoader);
10    }
11  
12    public static void main(String[] args) throws Exception {
13      doMagic(ClassLoader.getSystemClassLoader().getParent());
14    }

16    public static void doMagic(ClassLoader extClassLoader) throws Exception {
17      byte[] bytes = getMagicClassBytes();
18      MyClassLoader classLoader = new MyClassLoader(extClassLoader);
19      classLoader.defineClass("Magic", bytes, 0, bytes.length);
20      Class<?> magic = classLoader.loadClass("Magic");
21      Method abracadabra = magic.getDeclaredMethod("abracadabra");
22      abracadabra.setAccessible(true);
23      abracadabra.invoke(null);
24    }
25  
26    private static byte[] getMagicClassBytes() throws IOException {
27      InputStream is = getSystemClassLoader().getResourceAsStream("Magic.class");
28      try {
29        ByteArrayOutputStream buffer = new ByteArrayOutputStream();
30  
31        int nRead;
32        byte[] data = new byte[16384];
33  
34        while ((nRead = is.read(data, 0, data.length)) != -1) {
35          buffer.write(data, 0, nRead);
36        }
37  
38        buffer.flush();
39        return buffer.toByteArray();
40      }
41      finally {
42        is.close();
43      }
44    }
45  }
