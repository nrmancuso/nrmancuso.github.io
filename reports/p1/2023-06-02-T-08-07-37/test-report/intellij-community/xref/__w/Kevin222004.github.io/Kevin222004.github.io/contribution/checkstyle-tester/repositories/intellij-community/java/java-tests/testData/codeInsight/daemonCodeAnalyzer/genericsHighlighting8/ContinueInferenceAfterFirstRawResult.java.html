5     List<String> getList(Function<Object, String> function) {
6           /*
7            * When the first argument below is a raw type it turns red because IDEA thinks the return
8            * type is Collection<>.  javac and Eclipse don't care
9            */
10      return  transform(new ArrayList(), new ArrayList<String>(), function);
11    }
12  
13    <R, S, T extends Collection<S>> T transform(Iterable<? extends R> oldCollection, T newCollection, Function<R, S> function) {
14      for (R r : oldCollection) {
15        newCollection.add(function.apply(r));
16      }
17      return newCollection;
18    }
19  
20    interface Function<X, Y> {
21      Y apply(X input);
22    }
