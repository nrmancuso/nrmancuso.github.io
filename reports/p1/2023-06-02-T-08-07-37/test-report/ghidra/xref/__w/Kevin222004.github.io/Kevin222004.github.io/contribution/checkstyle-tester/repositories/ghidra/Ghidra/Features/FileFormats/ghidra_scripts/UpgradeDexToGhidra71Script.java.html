16  //Upgrade DEX program(s) that have function prototypes layed down prior to Ghidra 7.1 
17  //@category    Upgrade
18  
19  import java.util.Map;
20  
21  import ghidra.app.cmd.function.ApplyFunctionSignatureCmd;
22  import ghidra.app.script.GhidraScript;
23  import ghidra.framework.model.*;
24  import ghidra.framework.plugintool.PluginTool;
25  import ghidra.program.model.address.AddressSpace;
26  import ghidra.program.model.data.FunctionDefinitionDataType;
27  import ghidra.program.model.lang.*;
28  import ghidra.program.model.listing.*;
29  import ghidra.program.model.pcode.Varnode;
30  import ghidra.program.model.symbol.SourceType;
31  import ghidra.util.exception.*;
32  
33  /**
34   * There was a major rearrangement of registers in the Dalvik.slaspec from 7.0 -> 7.1 which invalidates function prototypes
35   * laid down by "Android DEX Header Format" analyzer.  This script repairs the prototypes to match the new register layout
36   * If run with a Program already up, the script will make all the changes, letting the user decide if they want to
37   * save (or undo) the changes.  If the script is run from an empty code browser, it will search for all Dalvik programs
38   * in the current project and automatically upgrade and save the function prototypes.
39   *
40   */
41  public class UpgradeDexToGhidra71Script extends GhidraScript {
42  
43      @Override
44      public void run() throws Exception {
45          
46          if ( currentProgram != null ) {
47              processProgram(currentProgram);
48              return;
49          }
50          
51          PluginTool tool = state.getTool();
52          Project project = tool.getProject();
53          ProjectData projectData = project.getProjectData();
54          DomainFolder rootFolder = projectData.getRootFolder();
55          recurseProjectFolder( rootFolder );
56      }
57      
58      private void recurseProjectFolder( DomainFolder domainFolder ) throws Exception {
59          DomainFile[] files = domainFolder.getFiles();
60          for ( DomainFile domainFile : files ) {
61          	monitor.checkCancelled();
62          	try {
63          		processDomainFile( domainFile );
64          	} catch(Exception ex) {
65          		printerr(ex.getMessage());
66          	}
67          }
68          DomainFolder[] folders = domainFolder.getFolders();
69          for ( DomainFolder folder : folders ) {
70          	monitor.checkCancelled();
71              recurseProjectFolder( folder );
72          }
73      }
74      
75  	private void processDomainFile(DomainFile domainFile ) throws Exception {
76  		Map<String, String> metadata = domainFile.getMetadata();
77  		if (metadata == null) {
78  			return;
79  		}
80  		String formatString = metadata.get("Executable Format");
81  		if (formatString == null) {
82  			return;
83  		}
84  		if (!formatString.equals("Dalvik Executable (DEX)")) {
85  			return;
86  		}
87  		DomainObject domainObject = domainFile.getDomainObject(this, true, true, monitor);
88  		try {
89  			Program program = (Program) domainObject;
90  			processProgram(program);
91  			saveProgram(program);
92  		} finally {
93  			domainObject.release(this);
94  		}
95  	}
96  
97  	private void processProgram(Program program) throws CancelledException {
98  		println("Updating program: "+program.getName());
99  		int id = program.startTransaction("Update DEX parameters");
100 		boolean success = false;
101 		try {
102 			for (Function func : program.getFunctionManager().getFunctions(true)) {
103 				monitor.checkCancelled();
104 				processFunction(func);
105 			}
106 			success = true;
107 		} finally {
108 			program.endTransaction(id, success);
109 		}
110 	}
111 
112 	private void processFunction(Function func) {
113 		monitor.setMessage("Updating: "+func.getName());
114 		FunctionDefinitionDataType sig = new FunctionDefinitionDataType(func,false);
115 		try {
116 			sig.setCallingConvention(CompilerSpec.CALLING_CONVENTION_stdcall);
117 		}
118 		catch (InvalidInputException e) {
119 			throw new AssertException(e);
120 		}
121 		func.setCustomVariableStorage(false);
122 		ApplyFunctionSignatureCmd cmd = new ApplyFunctionSignatureCmd(func.getEntryPoint(),sig,SourceType.ANALYSIS);
123 		cmd.applyTo(func.getProgram());
124 		
125 		Program program = func.getProgram();
126 		Language language = program.getLanguage();
127 		AddressSpace registerSpace = program.getAddressFactory().getRegisterSpace();
128 		Variable[] localVariables = func.getLocalVariables();
129 		
130 		for (Variable var : localVariables) {
131 			Varnode varnode = var.getFirstStorageVarnode();
132 			if (!varnode.isRegister()) {
133 				continue;
134 			}
135 			if (varnode.getOffset() >= 0x1000)
136 			 {
137 				continue;		// Already converted
138 			}
139 			long offset = varnode.getOffset() + 0x1000 - 8;
140 			int size = varnode.getSize();
141 			Register localRegister = language.getRegister(registerSpace, offset, size);
142 			try {
143 				LocalVariableImpl newlocal = new LocalVariableImpl( var.getName(), 0, var.getDataType(), localRegister, func.getProgram() );
144 				func.removeVariable(var);
145 				func.addLocalVariable(newlocal, SourceType.ANALYSIS);
146 			} catch (InvalidInputException e) {
147 			} catch (DuplicateNameException e) {
148 			}
149 			
150 		}
151 	}
