16  // Reports the percentage of bytes in any executable section that were disassembled into instructions,
17  // or turned into defined data.
18  // This can be an "indicator" of how well the initial analysis was at finding all code.
19  //
20  // Assumes a program is open.
21  //
22  //@category Examples
23  
24  import ghidra.app.script.GhidraScript;
25  import ghidra.program.model.address.AddressSetView;
26  import ghidra.program.model.listing.*;
27  import ghidra.util.Msg;
28  
29  public class ReportPercentDisassembled extends GhidraScript {
30  
31  	@Override
32  	public void run() throws Exception {
33  
34  		// find all the sections of memory marked as executable
35  		Program prog = currentProgram;
36  		AddressSetView execMemSet = prog.getMemory().getExecuteSet();
37  		/*int myExecSetLen = 0;
38  		MemoryBlock[] blocks = prog.getMemory().getBlocks();
39  		for (int i = 0; i < blocks.length; i++) {
40  			if (blocks[i].isExecute()) {
41  				myExecSetLen += blocks[i].getSize();
42  			}
43  		}*/
44  
45  		// tally up all the bytes that have been marked as instructions
46  		//   (probably faster ways to do this, but it works)
47  		long numPossibleDis = execMemSet.getNumAddresses();
48  		InstructionIterator instIter = prog.getListing().getInstructions(execMemSet, true);
49  		int instCount = 0;
50  		while (instIter.hasNext()) {
51  			Instruction inst = instIter.next();
52  			instCount += inst.getLength();
53  		}
54  		DataIterator dataIter = prog.getListing().getData(execMemSet, true);
55  		int dataCount = 0;
56  		while (dataIter.hasNext()) {
57  			Data data = dataIter.next();
58  			if (data.isDefined()) {
59  				dataCount += data.getLength();
60  			}
61  		}
62  
63  		// dump the info
64  		int total = instCount + dataCount;
65  		if (numPossibleDis != 0) {
66  			float coverage = (float) total / (float) numPossibleDis;
67  //	    	Msg.info(this,"execSetLen = " + numPossibleDis);
68  //	    	Msg.info(this,"MyExecSetLen = " + myExecSetLen);
69  //	    	Msg.info(this,"NumInsts = " + instCount);
70  //	    	Msg.info(this,"numData = " + dataCount);
71  //
72  //	    	Msg.info(this,"totalDis = " + total);
73  			float percentage = coverage * 100;
74  			Msg.info(this, "REPORT DISASSEMBLY EXTENT: " + prog.getName() + ": " + percentage +
75  				"% disassembled.");
76  		}
77  		return;
78  
79  	}
80  
81  }
