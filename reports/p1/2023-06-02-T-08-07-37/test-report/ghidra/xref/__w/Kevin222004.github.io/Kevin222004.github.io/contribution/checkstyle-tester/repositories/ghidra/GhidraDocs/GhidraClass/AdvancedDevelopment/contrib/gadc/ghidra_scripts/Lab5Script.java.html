16  //This script searches through all instructions that are
17  //moving a scalar into a register
18  //and sets an EOL comment in the form "[register] = [value]"
19  //@category GADC
20  
21  import ghidra.app.script.GhidraScript;
22  import ghidra.program.model.lang.Register;
23  import ghidra.program.model.listing.Instruction;
24  import ghidra.program.model.scalar.Scalar;
25  
26  public class Lab5Script extends GhidraScript {
27  
28  	@Override
29  	public void run() throws Exception {
30  
31  		for (Instruction instruction = getFirstInstruction(); instruction != null; instruction =
32  			getInstructionAfter(instruction)) {
33  			if (monitor.isCancelled()) {
34  				break;
35  			}
36  			if (instruction.getNumOperands() != 2) {
37  				continue;
38  			}
39  
40  			Object[] opObjects0 = instruction.getOpObjects(0);
41  			if (opObjects0.length != 1 || !(opObjects0[0] instanceof Register)) {
42  				continue;
43  			}
44  
45  			Object[] opObjects1 = instruction.getOpObjects(1);
46  			if (opObjects1.length != 1 || !(opObjects1[0] instanceof Scalar)) {
47  				continue;
48  			}
49  
50  			Register register = (Register) opObjects0[0];
51  			Scalar scalar = (Scalar) opObjects1[0];
52  			String comment =
53  				"[" + register.getName() + "]=[" + scalar.toString(16, false, false, "", "") + "]";
54  			setEOLComment(instruction.getMinAddress(), comment);
55  		}
56  	}
57  }
