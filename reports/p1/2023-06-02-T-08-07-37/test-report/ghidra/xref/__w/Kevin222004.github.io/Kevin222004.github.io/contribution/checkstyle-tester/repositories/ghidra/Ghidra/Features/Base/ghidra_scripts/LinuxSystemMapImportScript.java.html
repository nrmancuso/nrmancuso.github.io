16  // Apply the information from a System.map file or /proc/ksyms to the listing.
17  // Symbol names are applied. If the symbol type is available, as in the
18  // System.map file, disassembly and function creation are done for text symbols.
19  
20  //  The System.map file has the following format (it is generated by nm):
21  //  <tt>
22  //  [address] [type] [name]
23  //	-address is a hex value, say c010000
24  //	-type is the type of symbol: `man nm` for them all; T is a function
25  //	-name is the name of the symbol
26  //  </tt>
27  //  /proc/ksyms eliminates the middle column, and may add the name of the module
28  //  in square brackets at the end of the line.
29  
30  //@category CustomerSubmission.Linux
31  
32  import java.io.*;
33  
34  import ghidra.app.script.GhidraScript;
35  import ghidra.program.model.address.*;
36  import ghidra.program.model.symbol.SourceType;
37  import ghidra.program.model.symbol.SymbolTable;
38  import ghidra.util.exception.InvalidInputException;
39  
40  public class LinuxSystemMapImportScript extends GhidraScript {
41  
42  	@Override
43  	public void run() throws Exception {
44  		File file = askFile("Please specify the System.Map to apply", "Apply");
45  		BufferedReader reader = new BufferedReader(new FileReader(file));
46  		try {
47  			readSystemMap(reader);
48  		}
49  		finally {
50  			reader.close();
51  		}
52  	}
53  
54  	private void readSystemMap(BufferedReader reader) throws IOException {
55  
56  		SymbolTable st = currentProgram.getSymbolTable();
57  		AddressSpace addrspace = currentProgram.getAddressFactory().getDefaultAddressSpace();
58  		if (currentSelection != null && !currentSelection.isEmpty()) {
59  			addrspace = currentProgram.getAddressFactory().getDefaultAddressSpace();
60  		}
61  
62  		int lineno = 0;
63  		String line;
64  		while (!monitor.isCancelled() && (line = reader.readLine()) != null) {
65  			lineno++;
66  			line = line.trim();
67  			String address, type, name;
68  
69  			String[] tokens = line.split("\\s");
70  			if (tokens.length == 3) {
71  				address = tokens[0];
72  				type = tokens[1];
73  				name = tokens[2];
74  				// account for /proc/ksyms output of the type: c0100000 modulesym [module]
75  				if (name.matches("^\\[.*]$")) {
76  					name = tokens[1] + "__" + tokens[2];
77  					type = "";
78  				}
79  			}
80  			else if (tokens.length == 2) {
81  				address = tokens[0];
82  				type = "";
83  				name = tokens[1];
84  			}
85  			else {
86  				println("Error parsing line " + lineno + ": \"" + line + "\".");
87  				continue;
88  			}
89  
90  			Address addr = getAddress(addrspace, address);
91  			if (addr == null) {
92  				println("Error parsing line " + lineno + ": \"" + line +
93  					"\": could not parse address.");
94  				continue;
95  			}
96  
97  			try {
98  				st.createLabel(addr, name, SourceType.USER_DEFINED);
99  			}
100 			catch (InvalidInputException e) {
101 				println("Error adding label for line " + lineno + ": \"" + line + "\"");
102 				continue;
103 			}
104 
105 			if (type.equalsIgnoreCase("t")) {
106 				// This is a function
107 				disassemble(addr);
108 				createFunction(addr, name);
109 			}
110 		}
111 	}
112 
113 	private Address getAddress(AddressSpace addrspace, String addressString) {
114 		try {
115 			return addrspace.getAddress("0x" + addressString);
116 		}
117 		catch (AddressFormatException e) {
118 			return null;
119 		}
120 	}
121 }
