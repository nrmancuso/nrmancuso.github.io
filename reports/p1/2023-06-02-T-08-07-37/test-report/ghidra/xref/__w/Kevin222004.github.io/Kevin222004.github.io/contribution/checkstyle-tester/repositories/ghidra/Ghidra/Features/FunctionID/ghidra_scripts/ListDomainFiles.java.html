16  import java.io.File;
17  import java.io.FileWriter;
18  import java.util.*;
19  
20  import ghidra.app.script.GhidraScript;
21  import ghidra.feature.fid.db.*;
22  import ghidra.util.Msg;
23  import ghidra.util.exception.VersionException;
24  
25  public class ListDomainFiles extends GhidraScript {
26  
27  	private void accumulateDomains(Set<String> names, FidDB fidDb) {
28  		long hash = Long.MIN_VALUE;
29  		for (;;) {
30  			Long longObj = fidDb.findFullHashValueAtOrAfter(hash);
31  			if (longObj == null) {
32  				break;
33  			}
34  			hash = longObj.longValue() + 1;
35  			List<FunctionRecord> funcList = fidDb.findFunctionsByFullHash(longObj.longValue());
36  			for (FunctionRecord rec : funcList) {
37  				names.add(rec.getDomainPath());
38  			}
39  		}
40  	}
41  
42  	@Override
43  	protected void run() throws Exception {
44  		FidFileManager fidFileManager = FidFileManager.getInstance();
45  		List<FidFile> userFid = fidFileManager.getUserAddedFiles();
46  		if (userFid.isEmpty()) {
47  			return;
48  		}
49  		FidFile fidFile =
50  			askChoice("List Domain files", "Choose FID database", userFid, userFid.get(0));
51  
52  		try (FidDB fidDb = fidFile.getFidDB(true)) {
53  			File outFile = askFile("Output file", "Choose output file: ");
54  			if (outFile == null) {
55  				return;
56  			}
57  			try (FileWriter out = new FileWriter(outFile)) {
58  				TreeSet<String> domainNames = new TreeSet<>();
59  				accumulateDomains(domainNames, fidDb);
60  				for (String name : domainNames) {
61  					out.write(name);
62  					out.append('\n');
63  					println(name);
64  				}
65  			}
66  		}
67  		catch (VersionException e) {
68  			// Version upgrades are not supported
69  			Msg.showError(this, null, "Failed to open FidDb",
70  				"Failed to open incompatible FidDb (may need to regenerate with this version of Ghidra): " +
71  					fidFile.getPath());
72  			return;
73  		}
74  
75  	}
76  
77  }
