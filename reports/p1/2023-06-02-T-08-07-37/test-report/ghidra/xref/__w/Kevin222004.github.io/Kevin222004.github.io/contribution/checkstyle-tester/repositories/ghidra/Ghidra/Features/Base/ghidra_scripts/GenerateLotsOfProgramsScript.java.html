16  import java.io.IOException;
17  
18  import ghidra.app.script.GhidraScript;
19  import ghidra.framework.model.DomainFolder;
20  import ghidra.program.database.ProgramDB;
21  import ghidra.program.model.lang.Language;
22  import ghidra.program.model.lang.Processor;
23  import ghidra.program.model.listing.Program;
24  import ghidra.util.InvalidNameException;
25  import ghidra.util.exception.CancelledException;
26  
27  public class GenerateLotsOfProgramsScript extends GhidraScript {
28  
29  	@Override
30  	protected void run() throws Exception {
31  		DomainFolder root = getProjectRootFolder();
32  		DomainFolder stuff = root.createFolder("stuff");
33  		createPrograms(stuff, 30);
34  		DomainFolder a = stuff.createFolder("A");
35  		createPrograms(a, 10);
36  		DomainFolder b = stuff.createFolder("B");
37  		createPrograms(b, 10);
38  		DomainFolder c = stuff.createFolder("C");
39  		createPrograms(c, 10);
40  		DomainFolder d = stuff.createFolder("D");
41  		createPrograms(d, 10);
42  		DomainFolder e = stuff.createFolder("E");
43  		createPrograms(e, 10);
44  		DomainFolder f = stuff.createFolder("F");
45  		createPrograms(f, 10);
46  
47  		DomainFolder big = a.createFolder("Big");
48  		createPrograms(big, 200);
49  	}
50  
51  	private void createPrograms(DomainFolder parent, int count)
52  			throws IOException, InvalidNameException, CancelledException {
53  		Processor processor = Processor.toProcessor("x86");
54  		Language language = getDefaultLanguage(processor);
55  		Program program = new ProgramDB("dummy", language, language.getDefaultCompilerSpec(), this);
56  		for (int i = 0; i < count; i++) {
57  			parent.createFile("Prog_" + i, program, monitor);
58  		}
59  		program.release(this);
60  	}
61  
62  }
