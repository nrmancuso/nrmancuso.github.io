29  public class EvaluateVTMatch extends GhidraScript {
30  	// NOTE: this script is very rudimentary and makes a lot of assumptions,
31  	// such as unique symbol names in source and destination, that the
32  	// same-named symbols are actual matches, etc.
33  	//
34  	// Perhaps it should be based on manual matches as ground-truth.
35  	private static class MyFunction {
36  		public String name;
37  		public int srchits;
38  		public int desthits;
39  		public boolean matched;
40  		public boolean mismatched;
41  		public boolean srcexists;
42  		public boolean destexists;
43  		public boolean srcbody;
44  		public boolean destbody;
45  
46  		public MyFunction(String nm) {
47  			name = nm;
48  			srchits = 0;
49  			desthits = 0;
50  			matched = false;
51  			mismatched = false;
52  			srcexists = false;
53  			destexists = false;
54  			srcbody = false;
55  			destbody = false;
56  		}
57  	}
58  
59  	private static class VTScorer {
60  		private Program sourceProg;
61  		private Program destProg;
62  		private TreeMap<String, MyFunction> nameset;
63  		private FunctionManager sourceFuncMgr;
64  		private FunctionManager destFuncMgr;
65  		private int totalmatches;
66  		private int possiblematches;
67  		private int emptybodymatches;			// A pair (with same name) neither of which have a body, not considered a possiblematch
68  		private int othersrcfuncs;
69  		private int otherdestfuncs;
70  		private int conflicts;
71  		private int matchdiscovered;
72  		private int mismatch;
73  		private double falsenegative;
74  		private double falsepositive;
75  		private List<String> mismatchList;
76  		private int mismatchCountDown;
77  		
78  		public VTScorer(Program src,Program dest) {
79  			sourceProg = src;
80  			destProg = dest;
81  			nameset = new TreeMap<String,MyFunction>();
82  			sourceFuncMgr = sourceProg.getFunctionManager();
83  			destFuncMgr = destProg.getFunctionManager();
84  			totalmatches = 0;
85  			possiblematches = 0;
86  			emptybodymatches = 0;
87  			othersrcfuncs = 0;
88  			otherdestfuncs = 0;
89  			conflicts = 0;
90  			matchdiscovered = 0;
91  			mismatch = 0;
92  			mismatchList = new ArrayList<String>();
93  			mismatchCountDown = 100;
94  		}
95  
96  		private static String getName(Function func) {
97  			String name = func.getName();
98  			int pos = name.indexOf("@@GLIB");
99  			if (pos >= 0) {
100 				name = name.substring(0, pos);
101 			}
102 			return name;
103 		}
104 
105 		/**
106 		 * For every function in a manager, make sure a record exists in -mymap-
107 		 * @param funcMgr is the manager to iterate over
108 		 * @param mymap is the map to add records to
109 		 * @param isSrc is true if the srcexists field should be set to true, otherwise destexists is set
110 		 */
111 		private static void tagFunctionNames(FunctionManager funcMgr,Listing listing,
112 				TreeMap<String, MyFunction> mymap, boolean isSrc) {
113 			FunctionIterator functions = funcMgr.getFunctions(true);
114 			while (functions.hasNext()) {
115 				Function next = functions.next();
116 				if (next.isThunk()) continue;
117 				CodeUnit cu = listing.getCodeUnitAt(next.getEntryPoint());
118 				boolean hasbody = false;
119 				if ((cu != null) && (cu instanceof Instruction))
120 					hasbody = true;
121 				String funcName = getName(next);
122 				MyFunction myRec = mymap.get(funcName);
123 				if (myRec == null) {
124 					myRec = new MyFunction(funcName);
125 					mymap.put(myRec.name, myRec);
126 				}
127 				if (isSrc) {
128 					myRec.srcexists = true;
129 					myRec.srcbody = hasbody;
130 				}
131 				else {
132 					myRec.destexists = true;
133 					myRec.destbody = hasbody;
134 				}
135 			}
136 			// Make sure external functions have an entry
137 			functions = funcMgr.getExternalFunctions();
138 			while(functions.hasNext()) {
139 				Function next = functions.next();
140 				String funcName = getName(next);
141 				MyFunction myRec = mymap.get(funcName);
142 				if (myRec == null) {
143 					myRec = new MyFunction(funcName);
144 					mymap.put(myRec.name, myRec);
145 				}
146 				if (isSrc) {
147 					myRec.srcexists = true;
148 					myRec.srcbody = false;
149 				}
150 				else {
151 					myRec.destexists = true;
152 					myRec.destbody = false;
153 				}
154 			}
155 		}
156 
157 		public void tag() {
158 			tagFunctionNames(sourceFuncMgr, sourceProg.getListing(), nameset, true);
159 			tagFunctionNames(destFuncMgr, destProg.getListing(), nameset, false);			
160 		}
161 		
162 		public void registerMatch(Address srcAddr,Address destAddr) {
163 			Function sourceFunc = sourceFuncMgr.getFunctionAt(srcAddr);
164 			if (sourceFunc == null) {
165 				return;
166 			}
167 			Function destFunc = destFuncMgr.getFunctionAt(destAddr);
168 			if (destFunc == null) {
169 				return;
170 			}
171 			totalmatches += 1;			// We have a match between functions
172 
173 			String srcName = getName(sourceFunc);
174 			String destName = getName(destFunc);
175 			MyFunction srcRec = nameset.get(srcName);
176 			MyFunction destRec = nameset.get(destName);
177 			srcRec.srchits += 1;
178 			destRec.desthits += 1;
179 			if (srcRec == destRec) {
180 				srcRec.matched = true;
181 			}
182 			else {
183 				srcRec.mismatched = true;
184 				if (mismatchCountDown > 0) {
185 					mismatchList.add("Mismatch: " + srcName + " - " + destName);
186 					mismatchCountDown -= 1;
187 				}
188 			}
189 			
190 		}
191 		
192 		public void calcStats() {
193 			Iterator<MyFunction> iterator2 = nameset.values().iterator();
194 			while (iterator2.hasNext()) {
195 				MyFunction myfunc = iterator2.next();
196 				if (myfunc.srcexists && myfunc.destexists) {
197 					possiblematches += 1;
198 					if (!myfunc.srcbody || !myfunc.destbody)
199 						emptybodymatches += 1;			// One side or other does not have a body
200 				}
201 				else if (myfunc.srcexists) {
202 					othersrcfuncs += 1;
203 				}
204 				else if (myfunc.destexists) {
205 					otherdestfuncs += 1;
206 				}
207 				if (myfunc.mismatched)
208 					mismatch += 1;
209 				if (myfunc.matched) {
210 					if ((myfunc.srchits == 1) && (myfunc.desthits == 1)) {
211 						matchdiscovered += 1;
212 					}
213 					else {
214 						conflicts += 1;						// Match confused by conflicts
215 					}
216 				}
217 			}
218 			falsepositive = (double) mismatch / (double) possiblematches;		// Functions in source that were mismatched with dest
219 			falsenegative =
220 				(double) (possiblematches - matchdiscovered) / (double) possiblematches;			
221 		}
222 		
223 		public void reportResults(GhidraScript script,String msg) {
224 			script.println(msg);
225 			script.println("  False positive = " + Double.toString(falsepositive));
226 			script.println("  False negative = " + Double.toString(falsenegative));
227 			script.println("  Total reported matches = " + Integer.toString(totalmatches));
228 			script.println("  Possible valid matches = " + Integer.toString(possiblematches));
229 			script.println("  Empty body matches = " + Integer.toString(emptybodymatches));
230 			script.println("  Non-conflicting valid matches = " + Integer.toString(matchdiscovered));
231 			script.println("  Mismatches = " + Integer.toString(mismatch));
232 			script.println("  Conflicting valid matches = " + Integer.toString(conflicts));
233 			script.println("  Source unmatchable functions = " + Integer.toString(othersrcfuncs));
234 			script.println("  Destination unmatchable functions = " + Integer.toString(otherdestfuncs));			
235 		}
236 		
237 		public void reportFalseNegatives(GhidraScript script,boolean sourceSide,int max) {
238 			Iterator<MyFunction> iterator2 = nameset.values().iterator();
239 			int count = 0;
240 			while (iterator2.hasNext()) {
241 				MyFunction myfunc = iterator2.next();
242 				if ((!myfunc.srcexists)||(!myfunc.destexists)) continue;		// Make sure both sides have named function
243 				if ((!myfunc.srcbody)||(!myfunc.destbody)) continue;			// Make sure both sides have a body
244 				if (!myfunc.matched) {			// If the function pair was not matched
245 					if (sourceSide && (myfunc.srchits==0)) {
246 						script.println("Source miss: "+myfunc.name);
247 						count += 1;
248 					}
249 					else if ((!sourceSide)&&(myfunc.desthits==0)) {
250 						script.println("Dest miss: "+myfunc.name);
251 						count += 1;
252 					}
253 					if (count >= max) return;
254 				}
255 			}			
256 		}
257 
258 		public void reportUnmatchable(GhidraScript script,boolean sourceSide,int max) {
259 			Iterator<MyFunction> iterator2 = nameset.values().iterator();
260 			int count = 0;
261 			while (iterator2.hasNext()) {
262 				MyFunction myfunc = iterator2.next();
263 				if (sourceSide) {
264 					if (myfunc.srcexists && !myfunc.destexists) {
265 						script.println("Source unmatchable: "+myfunc.name);
266 						count += 1;
267 					}
268 				}
269 				else {
270 					if (myfunc.destexists && !myfunc.srcexists) {
271 						script.println("Dest unmatchable: "+myfunc.name);
272 						count += 1;
273 					}
274 					if (count >= max) break;
275 				}
276 			}
277 		}
278 
279 		public void reportMismatches(GhidraScript script) {
280 			for (String line : mismatchList) {
281 				script.println(line);
282 			}
283 		}

286 	@Override
287 	protected void run() throws Exception {
288 		DomainFile vtFile = askDomainFile("Select VT Session");
289 		openVTSessionAndDoWork(vtFile);
290 	}

292 	private void openVTSessionAndDoWork(DomainFile domainFile) {
293 
294 		DomainObject vtDomainObject = null;
295 		try {
296 			vtDomainObject = domainFile.getDomainObject(this, false, false, monitor);
297 			doWork((VTSessionDB) vtDomainObject);
298 		}
299 		catch (VersionException e) {
300 			e.printStackTrace();
301 		}
302 		catch (CancelledException e) {
303 			e.printStackTrace();
304 		}
305 		catch (IOException e) {
306 			e.printStackTrace();
307 		}
308 		finally {
309 			if (vtDomainObject != null) {
310 				vtDomainObject.release(this);
311 			}
312 		}
313 	}
314 
315 	private void doWork(VTSessionDB session) {
316 		println("Working on session: " + session);
317 
318 		List<VTMatchSet> matchSets = session.getMatchSets();
319 		Iterator<VTMatchSet> iterator = matchSets.iterator();
320 		while (iterator.hasNext()) {
321 			evaluateMatchSet(iterator.next());
322 		}
323 		evaluateAccepted(session);
324 	}
325 
326 	private void evaluateAccepted(VTSessionDB session) {
327 		VTScorer scorer = new VTScorer(session.getSourceProgram(),session.getDestinationProgram());
328 		scorer.tag();
329 
330 		VTAssociationManager associationManager = session.getAssociationManager();
331 		List<VTAssociation> associations = associationManager.getAssociations();
332 		for (VTAssociation association : associations) {
333 			if (association.getStatus() == VTAssociationStatus.ACCEPTED) {
334 				Address srcAddr = association.getSourceAddress();
335 				Address destAddr = association.getDestinationAddress();
336 				scorer.registerMatch(srcAddr, destAddr);
337 			}
338 		}
339 
340 		scorer.calcStats();
341 		scorer.reportResults(this,"OVERALL ACCEPTED RESULTS:");
342 //		scorer.reportFalseNegatives(this, true,20);
343 //		scorer.reportUnmatchable(this, true, 20);
344 		scorer.reportMismatches(this);
345 	}
346 
347 	private void evaluateMatchSet(VTMatchSet matchset) {
348 		Collection<VTMatch> matches = matchset.getMatches();
349 		if (matches.isEmpty()) {
350 			println("Empty matchset - " + matchset.getProgramCorrelatorInfo().getName() + " (" +
351 				matchset.getID() + ")");
352 			return;
353 		}
354 		VTSession vtsession = matchset.getSession();
355 		VTScorer scorer = new VTScorer(vtsession.getSourceProgram(),vtsession.getDestinationProgram());
356 		scorer.tag();
357 
358 		Iterator<VTMatch> iterator = matches.iterator();
359 		while (iterator.hasNext()) {
360 			VTMatch next = iterator.next();
361 			VTAssociation association = next.getAssociation();
362 			Address srcAddr = association.getSourceAddress();
363 			Address destAddr = association.getDestinationAddress();
364 			scorer.registerMatch(srcAddr, destAddr);
365 		}
366 		scorer.calcStats();
367 		scorer.reportResults(this,matchset.getProgramCorrelatorInfo().getName() + " (" + matchset.getID() +
368 			"):");
369 	}
