16  // Figures out computed memory references at the current cursor location or at the instruction at the
17  // start of each range in an address set.
18  // Place the cursor on a register or constant operand, and run the script.  Also, if a
19  // register has a value set at the beginning of a function, that register value is assumed
20  // to be a constant.
21  //
22  // For ease of use, attach this to a key-binding to create the reference in one keystroke.
23  //
24  // This script is very useful on the ARM, PowerPC and most RISC based processors that
25  // use multiple instructions to build up memory references, where the reference was
26  // missed by auto-analysis.  It is also useful for references that weren't created
27  // because of complex base address + offset calculations.
28  //
29  // It is very easy to use this script in conjunction with any type of search.  For
30  // example on the ARM, MOVT is used to build up and address.  Search the program text
31  // for all mnemonics MOVT and then select the ones that are creating a reference, make
32  // a selection in the listing from the search, and then execute this function.  It is
33  // best if you have already assigned a key binding.  You can also choose single items
34  // from the search table and press the key bound to this script.
35  //
36  // NOTE: Any values loaded from memory are assumed to be constant.
37  // If a reference does not make sense on an operand, then it is added to the mnemonic.
38  //
39  //@category Analysis
40

59  public class MultiInstructionMemReference extends GhidraScript {
60  
61  	Address memReferenceLocation = null;
62  	private Address curInstrloc;
63  	private Object[] inputObjects;
64  	private Object[] resultObjects;
65  	private Register singleRegister;
66  	private boolean registerInOut;
67  	private boolean targetInDelaySlot = false;
68  	private boolean loadStore = false;
69  
70  	@Override
71  	public void run() throws Exception {
72  		long numInstructions = currentProgram.getListing().getNumInstructions();
73  		monitor.initialize((int) (numInstructions));
74  		monitor.setMessage("Multi-Instruction Reference Markup");
75  		int currentOpIndex = -1;
76  
77  		Address start = currentLocation.getAddress();
78  
79  		if ((currentSelection == null || currentSelection.isEmpty()) &&
80  			currentLocation instanceof OperandFieldLocation) {
81  			OperandFieldLocation operandLocation = (OperandFieldLocation) currentLocation;
82  			currentOpIndex = operandLocation.getOperandIndex();
83  			int subOpIndex = operandLocation.getSubOperandIndex();
84  			singleRegister = getRegister(start, currentOpIndex, subOpIndex);
85  		}
86  
87  		// set up the address set to restrict processing
88  		AddressSet refLocationsSet = new AddressSet(currentSelection);
89  		if (refLocationsSet.isEmpty()) {
90  			refLocationsSet.addRange(start, start);
91  		}
92  
93  		findMemRefAtOperand(currentOpIndex, refLocationsSet);
94  	}
95  
96  	/**
97  	 * Get the register at the location
98  	 * 
99  	 * @param opIndex index into operands for instruction
100 	 * @param subOpIndex index into operands for an operand location
101 	 * 
102 	 * @return register if there is one at the location
103 	 */
104 	private Register getRegister(Address addr, int opIndex, int subOpIndex) {
105 		if (addr == null) {
106 			return null;
107 		}
108 
109 		Instruction instr = currentProgram.getListing().getInstructionContaining(addr);
110 		if (instr == null) {
111 			return null;
112 		}
113 
114 		List<Object> defOpRep = instr.getDefaultOperandRepresentationList(opIndex);
115 		if (subOpIndex >= 0 && subOpIndex < defOpRep.size()) {
116 			Object obj = defOpRep.get(subOpIndex);
117 			if (obj instanceof Register) {
118 				return (Register) obj;
119 			}
120 		}
121 		return instr.getRegister(opIndex);
122 	}

124 	@SuppressWarnings("unused")
125 	private boolean isSingleInstructions(AddressSet restrictedSet) {
126 		if (restrictedSet.isEmpty()) {
127 			return false;
128 		}
129 		AddressRangeIterator riter = restrictedSet.getAddressRanges();
130 		restrictedSet = new AddressSet(restrictedSet);
131 
132 		while (riter.hasNext()) {
133 			AddressRange addressRange = riter.next();
134 			Instruction instr =
135 				currentProgram.getListing().getInstructionAt(addressRange.getMinAddress());
136 			if (instr != null) {
137 				addressRange = new AddressRangeImpl(instr.getMinAddress(), instr.getMaxAddress());
138 			}
139 			restrictedSet.delete(addressRange);
140 		}
141 		return restrictedSet.isEmpty();
142 	}

144 	private void findMemRefAtOperand(final int opIndex, AddressSetView set) {
145 		// follow all flows building up context
146 		// use context to fill out addresses on certain instructions 
147 		ContextEvaluator eval = new ContextEvaluatorAdapter() {
148 
149 			@Override
150 			public boolean evaluateContextBefore(VarnodeContext context, Instruction instr) {
151 				// if the requested reference was on an input op-object, get context before exec
152 				return checkContext(true, opIndex, context, instr);
153 			}
154 
155 			@Override
156 			public boolean evaluateContext(VarnodeContext context, Instruction instr) {
157 				// if the requested reference was on an output op-object, get context after exec
158 				return checkContext(false, opIndex, context, instr);
159 			}
160 
161 			private boolean checkContext(boolean input, final int opIndex, VarnodeContext context,
162 					Instruction instr) {
163 				if (instr.getMinAddress().equals(curInstrloc)) {
164 					if (targetInDelaySlot && instr.getDelaySlotDepth() > 0) {
165 						instr = instr.getNext();
166 					}
167 					if (checkInstructionMatch(opIndex, input, context, instr)) {
168 						return true;
169 					}
170 					// if instruction is in delayslot, assume reference is good.
171 					if (instr.getDelaySlotDepth() > 0) {
172 						instr = instr.getNext();
173 						return checkInstructionMatch(opIndex, input, context, instr);
174 					}
175 				}
176 				return false;
177 			}
178 
179 			@Override
180 			public boolean evaluateReference(VarnodeContext context, Instruction instr, int pcodeop,
181 					Address address, int size, DataType dataType, RefType refType) {
182 
183 				return super.evaluateReference(context, instr, pcodeop, address, size, dataType, refType);
184 			}
185 
186 			private boolean checkInstructionMatch(final int opIdx, boolean input,
187 					VarnodeContext context, Instruction instr) {
188 				List<Object> list = Arrays.asList(input ? inputObjects : resultObjects);
189 
190 				// if the register appeared as an input and output varnode, and not a load or store, then
191 				// is just a calculation, so don't consider the an input values
192 				if (input && !loadStore && registerInOut) {
193 					return false;
194 				}
195 
196 				for (int index = opIdx; index < instr.getNumOperands(); index++) {
197 					if (getRefsForOperand(context, instr, list, index)) {
198 						// register is both an in/out check if symbolic on out
199 						if (registerInOut) {
200 							break;
201 						}
202 						return true;
203 					}
204 				}
205 				if (addSymbolicRefs(input, context, instr, list)) {
206 					return true;
207 				}
208 				return false;
209 			}
210 
211 			/**
212 			 * Check the current operand for references based on input/outputs
213 			 * 
214 			 * @param context - context holding values
215 			 * @param instr - instruction under consideration
216 			 * @param list - input/output lists
217 			 * @param opIndex - index of operand to check
218 			 * 
219 			 * @return true if a reference was found
220 			 */
221 			private boolean getRefsForOperand(VarnodeContext context, Instruction instr,
222 					List<Object> list, int opIndex) {
223 				Object[] opObjects = instr.getOpObjects(opIndex);
224 				for (Object opObject : opObjects) {
225 					if (!(opObject instanceof Register)) {
226 						continue;
227 					}
228 					Register reg = (Register) opObject;
229 
230 					// if operand has a single register and this isn't it
231 					if (singleRegister != null && !reg.equals(singleRegister)) {
232 						continue;
233 					}
234 
235 					// check that the register is on the correct input/output list
236 					if (!list.contains(reg)) {
237 						continue;
238 					}
239 					RegisterValue rval = context.getRegisterValue(reg);
240 					if (rval == null) {
241 						continue;
242 					}
243 					BigInteger uval = rval.getUnsignedValue();
244 					if (uval == null) {
245 						continue;
246 					}
247 					long offset = uval.longValue();
248 
249 					AddressSpace space = instr.getMinAddress().getAddressSpace();
250 					Address addr = space.getTruncatedAddress(offset, true);
251 
252 					// assume that they want the reference, don't worry it isn't in memory
253 					makeReference(instr, opIndex, addr);
254 					return true;
255 				}
256 				return false;
257 			}
258 
259 			private boolean addSymbolicRefs(boolean input, VarnodeContext context,
260 					Instruction instr, List<Object> list) {
261 				// get the value of the single register to see if this is the value desired
262 				if (singleRegister == null) {
263 					return false;
264 				}
265 				// check that the register is on the correct input/output list
266 				if (!list.contains(singleRegister)) {
267 					return false;
268 				}
269 				Varnode registerVarnodeValue = context.getRegisterVarnodeValue(singleRegister);
270 				if (!context.isSymbol(registerVarnodeValue) && !registerVarnodeValue.isRegister()) {
271 					return false;
272 				}
273 				Address symAddr = registerVarnodeValue.getAddress();
274 				if (symAddr == context.BAD_ADDRESS) {
275 					return false;
276 				}
277 
278 				String valStr = "";
279 				if (registerVarnodeValue.isRegister()) {
280 					valStr = context.getRegister(registerVarnodeValue).toString();
281 				}
282 				else {
283 					// is an offset from a space
284 					String name = symAddr.getAddressSpace().getName();
285 					BigInteger offset = symAddr.getOffsetAsBigInteger();
286 					valStr = name + " + 0x" + offset.toString(16);
287 				}
288 				Address lastSetLocation = context.getLastSetLocation(singleRegister, null);
289 
290 				String comment = instr.getComment(Instruction.EOL_COMMENT);
291 				if (comment == null) {
292 					comment = "";
293 				}
294 
295 				String inoutChar = (input ? " " : "\'");
296 				String lastStr = (lastSetLocation != null ? " @" + lastSetLocation : "");
297 
298 				String markup = singleRegister + inoutChar + "= " + valStr + lastStr;
299 				if (comment.replace('\'', ' ').contains(markup.replace('\'', ' '))) {
300 					return false;
301 				}
302 				comment = (comment.trim().length() == 0 ? markup : comment + "\n" + markup);
303 				instr.setComment(Instruction.EOL_COMMENT, comment);
304 				return false;
305 			}

307 			@Override
308 			public boolean allowAccess(VarnodeContext context, Address addr) {
309 				// allow values to be read from writable memory
310 				return true;
311 			}
312 		};
313 
314 		try {
315 			AddressRangeIterator riter = set.getAddressRanges();
316 			while (riter.hasNext() && !monitor.isCancelled()) {
317 				AddressRange addressRange = riter.next();
318 
319 				curInstrloc = addressRange.getMinAddress();
320 				AddressSet body = null;
321 				Address start = curInstrloc;
322 
323 				Function curFunc =
324 					currentProgram.getFunctionManager().getFunctionContaining(curInstrloc);
325 				if (curFunc != null) {
326 					start = curFunc.getEntryPoint();
327 					body = new AddressSet(curFunc.getBody());
328 				}
329 				else {
330 					body = new AddressSet(curInstrloc);
331 					PartitionCodeSubModel model = new PartitionCodeSubModel(currentProgram);
332 					CodeBlock block = model.getFirstCodeBlockContaining(curInstrloc, monitor);
333 					if (block != null) {
334 						start = block.getFirstStartAddress();
335 						body.add(block);
336 					}
337 				}
338 
339 				Instruction instr = currentProgram.getListing().getInstructionAt(curInstrloc);
340 				if (instr != null) {
341 					inputObjects = instr.getInputObjects();
342 					resultObjects = instr.getResultObjects();
343 					registerInOut = checkRegisterInOut(singleRegister, inputObjects, resultObjects);
344 					loadStore = checkLoadStore(instr.getPcode());
345 				}
346 
347 				// if the instruction attempting to markup is in the delayslot, backup an instruction
348 				if (instr != null && instr.isInDelaySlot()) {
349 					instr = instr.getPrevious();
350 					if (instr != null) {
351 						curInstrloc = instr.getMinAddress();
352 						targetInDelaySlot = true;
353 					}
354 				}
355 
356 				SymbolicPropogator symEval = new SymbolicPropogator(currentProgram);
357 				symEval.setParamRefCheck(false);
358 				symEval.setReturnRefCheck(false);
359 				symEval.setStoredRefCheck(false);
360 
361 				symEval.flowConstants(start, body, eval, true, monitor);
362 			}
363 		}
364 		catch (CancelledException e) {
365 		}
366 	}
367 
368 	private boolean checkLoadStore(PcodeOp[] pcode) {
369 		for (PcodeOp pcodeOp : pcode) {
370 			int opcode = pcodeOp.getOpcode();
371 			if (opcode == PcodeOp.LOAD || opcode == PcodeOp.STORE) {
372 				return true;
373 			}
374 		}
375 		return false;
376 	}
377 
378 	private boolean checkRegisterInOut(Register reg, Object[] in, Object[] out) {
379 		if (reg == null || in == null || out == null) {
380 			return false;
381 		}
382 
383 		List<Object> inList = Arrays.asList(in);
384 		List<Object> outList = Arrays.asList(out);
385 
386 		return inList.contains(reg) && outList.contains(reg);
387 	}

389 	/** Make the reference on the instruction at the correct location.
390 	 * 
391 	 * @param instruction to receive reference
392 	 * @param space reference created in this space
393 	 * @param scalar used as offset into address space
394 	 */
395 	private void makeReference(Instruction instruction, int opIndex, Address addr) {
396 		if (targetInDelaySlot && instruction.getPrototype().hasDelaySlots()) {
397 			instruction = instruction.getNext();
398 			if (instruction == null) {
399 				return;
400 			}
401 		}
402 		if (opIndex == -1) {
403 			for (int i = 0; i < instruction.getNumOperands(); i++) {
404 				int opType = instruction.getOperandType(i);
405 				// markup the program counter for any flow
406 				if ((opType & OperandType.DYNAMIC) != 0) {
407 					opIndex = i;
408 					break;
409 				}
410 			}
411 		}
412 		if (opIndex == -1) {
413 			opIndex = instruction.getNumOperands() - 1;
414 		}
415 
416 		// check if it already has the reference
417 		Reference[] referencesFrom = instruction.getReferencesFrom();
418 		boolean hasRef = Arrays.stream(referencesFrom).anyMatch(p -> p.getToAddress().equals(addr));
419 		if (hasRef) {
420 			return;
421 		}
422 
423 		if (opIndex == -1) {
424 			instruction.addMnemonicReference(addr, RefType.DATA, SourceType.ANALYSIS);
425 		}
426 		else {
427 			instruction.addOperandReference(opIndex, addr, RefType.DATA, SourceType.ANALYSIS);
428 		}
429 	}
