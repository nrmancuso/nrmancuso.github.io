22  import java.util.Arrays;
23  import java.util.LinkedList;
24  import java.util.List;

26  import com.google.common.collect.Lists;
27  import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
28  import com.puppycrawl.tools.checkstyle.api.DetailAST;

32  /**
33   * This check detects the child blocks, which length is more then 80% of parent
34   * block length. <br>
35   * <p>
36   * Supported keywords are used to detect blocks: <br>
37   * "if", "else", "for", "switch", "do", "while", "try", "catch".
38   * </p>
39   * <p>
40   * <i>Rationale:</i>
41   * </p>
42   * <p>
43   * Length of child block that is more then 80% of parent block is usually hard
44   * to read in case child block is long(few display screens). Such child blocks
45   * should be refactored or moved to separate method.
46   * </p>
47   *
48   * @author <a href="mailto:Daniil.Yaroslavtsev@gmail.com"> Daniil
49   *         Yaroslavtsev</a>
50   * @since 1.8.0
51   */
52  public class ChildBlockLengthCheck extends AbstractCheck {
53  
54      /**
55       * A key is pointing to the warning message text in "messages.properties"
56       * file.
57       */
58      public static final String MSG_KEY = "child.block.length";
59  
60      /**
61       * The constant is used in percentage arithmetic operations. Represents
62       * '100%'
63       */
64      private static final double PERCENTS_FACTOR = 100.0;
65  
66      /**
67       * Default value of maximum percentage ratio between the child and the
68       * parent block.
69       */
70      private static final double DEFAULT_MAX_CHILD_BLOCK_PERCENTAGE = 80.0;
71  
72      /**
73       * Default number of lines of which block body may consist to be skipped by
74       * check.
75       */
76      private static final int DEFAULT_IGNORE_BLOCK_LINESCOUNT = 50;
77  
78      /**
79       * Array contains all allowed block types to be checked. Supported block
80       * types: LITERAL_IF, LITERAL_SWITCH, LITERAL_FOR, LITERAL_DO,
81       * LITERAL_WHILE, LITERAL_TRY, LITERAL_ELSE, LITERAL_CATCH.
82       */
83      private int[] blockTypes;
84  
85      /**
86       * Maximum percentage ratio between the child block and the parent block.
87       */
88      private double maxChildBlockPercentage = DEFAULT_MAX_CHILD_BLOCK_PERCENTAGE;
89  
90      /**
91       * Maximum number of lines of which block body may consist to be skipped by
92       * check.
93       */
94      private int ignoreBlockLinesCount = DEFAULT_IGNORE_BLOCK_LINESCOUNT;
95  
96      /**
97       * Sets allowed types of blocks to be checked. Supported block types:
98       * LITERAL_IF, LITERAL_SWITCH, LITERAL_FOR, LITERAL_DO, LITERAL_WHILE,
99       * LITERAL_TRY, LITERAL_ELSE, LITERAL_CATCH.
100      *
101      * @param blockTypes
102      *        - DetailAST tokenTypes that are related to the types which are
103      *        allowed by user in check preferences.
104      **/
105     public void setBlockTypes(final String... blockTypes) {
106         this.blockTypes = new int[blockTypes.length];
107         for (int i = 0; i < blockTypes.length; i++) {
108             this.blockTypes[i] = TokenUtil.getTokenId(blockTypes[i]);
109         }
110     }
111 
112     /**
113      * Sets the maximum percentage ratio between child and parent block. (sets
114      * "maxChildBlockPercentage" option value)
115      *
116      * @param maxChildBlockPercentage
117      *        the new "maxChildBlockPercentage" option value.
118      */
119     public void setMaxChildBlockPercentage(int maxChildBlockPercentage) {
120         this.maxChildBlockPercentage = maxChildBlockPercentage;
121     }
122 
123     /**
124      * Sets the maximum linelength of blocks that will be ignored by check.
125      *
126      * @param ignoreBlockLinesCount
127      *        the maximum linelength of the block to be ignored.
128      */
129     public void setIgnoreBlockLinesCount(int ignoreBlockLinesCount) {
130         this.ignoreBlockLinesCount = ignoreBlockLinesCount;
131     }
132 
133     // -@cs[SimpleAccessorNameNotation] Overrides method from the base class.
134     // Issue: https://github.com/sevntu-checkstyle/sevntu.checkstyle/issues/166
135     @Override
136     public int[] getDefaultTokens() {
137         final int[] result;
138         if (blockTypes == null) {
139             result = null;
140         }
141         else {
142             result = Arrays.copyOf(blockTypes, blockTypes.length);
143         }
144         return result;
145     }
146 
147     @Override
148     public int[] getAcceptableTokens() {
149         return getDefaultTokens();
150     }
151 
152     @Override
153     public int[] getRequiredTokens() {
154         return getDefaultTokens();
155     }
156 
157     @Override
158     public void visitToken(DetailAST ast) {
159         final DetailAST aOpeningBrace = openingBrace(ast);
160 
161         // if the block has braces at all
162         if (aOpeningBrace != null) {
163             final DetailAST aClosingBrace = closingBrace(ast);
164             final int parentBlockSize =
165                     linesCount(aOpeningBrace, aClosingBrace);
166 
167             if (parentBlockSize > ignoreBlockLinesCount) {
168                 final List<DetailAST> childBlocks = getChildBlocks(
169                         aOpeningBrace, aClosingBrace);
170 
171                 final List<DetailAST> badChildBlocks = getBadChildBlocks(
172                         childBlocks, parentBlockSize);
173 
174                 if (badChildBlocks.isEmpty()) {
175                     for (DetailAST childBlock : childBlocks) {
176                         visitToken(childBlock);
177                     }
178                 }
179                 else {
180                     for (DetailAST badBlock : badChildBlocks) {
181                         final int blockSize = linesCount(badBlock);
182 
183                         final double allowedBlockSize = (int) (parentBlockSize
184                                 * maxChildBlockPercentage / PERCENTS_FACTOR);
185 
186                         log(badBlock, MSG_KEY, blockSize, allowedBlockSize);
187                     }
188                 }
189             }
190         }
191     }
192 
193     /**
194      * Gets all the child blocks for given parent block. Uses an iterative
195      * algorithm.
196      *
197      * @param blockOpeningBrace
198      *        a DetailAST node that points to the current method`s opening
199      *        brace.
200      * @param blockClosingBrace
201      *        the a block closing brace
202      * @return all child blocks that have braces.
203      */
204     private List<DetailAST> getChildBlocks(DetailAST blockOpeningBrace,
205             DetailAST blockClosingBrace) {
206         final List<DetailAST> childBlocks = Lists.newLinkedList();
207 
208         DetailAST curNode = blockOpeningBrace;
209 
210         while (curNode != blockClosingBrace) {
211             if (isAllowedBlockType(curNode.getType())) {
212                 childBlocks.add(curNode);
213             }
214             // before node leaving
215             DetailAST nextNode = curNode.getFirstChild();
216 
217             final int type = curNode.getType();
218             // skip anonymous classes and nested methods
219             if (type == TokenTypes.METHOD_DEF
220                     || type == TokenTypes.CLASS_DEF) {
221                 nextNode = curNode.getNextSibling();
222             }
223 
224             while (nextNode == null) {
225                 // leave the visited node
226                 nextNode = curNode.getNextSibling();
227                 if (nextNode == null) {
228                     curNode = curNode.getParent();
229                 }
230             }
231             curNode = nextNode;
232         }
233         return childBlocks;
234     }
235 
236     /**
237      * Checks that given child block type is allowed.
238      *
239      * @param blockType
240      *        the token type ID for the given block.
241      * @return true, if the given child block type is allowed.
242      */
243     private boolean isAllowedBlockType(int blockType) {
244         boolean result = false;
245         for (int type : blockTypes) {
246             if (type == blockType) {
247                 result = true;
248                 break;
249             }
250         }
251         return result;
252     }
253 
254     /**
255      * Gets the child blocks which occupies too much size (in percentage) of
256      * given parent block size.
257      *
258      * @param blocksList
259      *        the blocks
260      * @param parentBlockSize
261      *        the a parent block size
262      * @return the wrong child blocks
263      */
264     private List<DetailAST> getBadChildBlocks(List<DetailAST> blocksList,
265             int parentBlockSize) {
266         final List<DetailAST> result = new LinkedList<>();
267         for (DetailAST block : blocksList) {
268             if (isChildBlockBad(block, parentBlockSize)) {
269                 result.add(block);
270             }
271         }
272         return result;
273     }
274 
275     /**
276      * Checks if the child block size percentage from parent block is greater
277      * than.
278      *
279      * @param childBlock
280      *        the a child block node
281      * @param parentBlockSize
282      *        the a parent block size
283      * @return true, if is child block wrong
284      */
285     private boolean isChildBlockBad(DetailAST childBlock,
286             int parentBlockSize) {
287         boolean result = false;
288         final DetailAST openingBrace = openingBrace(childBlock);
289         if (openingBrace != null) {
290             final DetailAST closingBrace = closingBrace(childBlock);
291             final int childBlockSize = linesCount(openingBrace, closingBrace);
292             result = getPercentage(parentBlockSize, childBlockSize);
293         }
294         return result;
295     }
296 
297     /**
298      * Gets the percentage which the child block occupies inside the parent
299      * block.
300      *
301      * @param parentBlockSize
302      *        the parent block size in lines
303      * @param childBlockSize
304      *        the child block size in lines
305      * @return the percentage value.
306      */
307     private boolean getPercentage(int parentBlockSize,
308             final int childBlockSize) {
309         final double percentage =
310                 ((double) childBlockSize / (double) parentBlockSize) * 100.0;
311         return percentage > maxChildBlockPercentage;
312     }
313 
314     /**
315      * Gets the opening brace for the given block.
316      *
317      * @param parentBlock
318      *        the DetailAST node is related to the given parent block.
319      * @return the DetailAST node is related to the given block opening brace
320      */
321     private static DetailAST openingBrace(final DetailAST parentBlock) {
322         final int searchType;
323         if (parentBlock.getType() == TokenTypes.LITERAL_SWITCH) {
324             searchType = TokenTypes.LCURLY;
325         }
326         else {
327             searchType = TokenTypes.SLIST;
328         }
329         return parentBlock.findFirstToken(searchType);
330     }
331 
332     /**
333      * Gets the closing brace for the given block.
334      *
335      * @param parentBlockNode
336      *        the DetailAST node is related to the given parent block.
337      * @return the DetailAST node is related to the given block closing brace
338      */
339     private static DetailAST closingBrace(DetailAST parentBlockNode) {
340         final int aParentBlockType = parentBlockNode.getType();
341         final DetailAST result;
342         if (aParentBlockType == TokenTypes.LITERAL_SWITCH) {
343             result = parentBlockNode.getLastChild();
344         }
345         else {
346             result = openingBrace(parentBlockNode).getLastChild();
347         }
348         return result;
349     }
350 
351     /**
352      * Gets the lines count between braces of the given block.
353      *
354      * @param blockAst
355      *        - the DetailAST node is related to the given block (the block
356      *        should have braces!).
357      * @return the lines count between the given block braces.
358      */
359     private static int linesCount(DetailAST blockAst) {
360         return linesCount(openingBrace(blockAst), closingBrace(blockAst));
361     }
362 
363     /**
364      * Gets the lines count between the given block opening and closing braces.
365      *
366      * @param openingBrace
367      *        the block opening brace DetailAST (LCURLY or SLIST type)
368      * @param closingBrace
369      *        the block closing brace DetailAST (RCURLY type)
370      * @return the lines count between the given block braces.
371      */
372     private static int linesCount(DetailAST openingBrace,
373             DetailAST closingBrace) {
374         int result = closingBrace.getLineNo() - openingBrace.getLineNo();
375         if (result != 0) {
376             result--;
377         }
378         return result;
379     }
