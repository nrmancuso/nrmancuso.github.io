55  /**
56   * {@link org.springframework.integration.core.MessageSource} that creates messages
57   * from a file system directory.
58   * To prevent messages for certain files, you may supply a {@link FileListFilter}.
59   * By default, when configuring with XML or the DSL,
60   * an {@link org.springframework.integration.file.filters.AcceptOnceFileListFilter} is used.
61   * It ensures files are picked up only once from the directory.
62   * <p>
63   * A common problem with reading files is that a file may be detected before it
64   * is ready. The default
65   * {@link org.springframework.integration.file.filters.AcceptOnceFileListFilter}
66   * does not prevent this. In most cases, this can be prevented if the
67   * file-writing process renames each file as soon as it is ready for reading. A
68   * pattern-matching filter that accepts only files that are ready (e.g. based on
69   * a known suffix), composed with the default
70   * {@link org.springframework.integration.file.filters.AcceptOnceFileListFilter}
71   * would allow for this.
72   * <p>
73   * If a external {@link DirectoryScanner} is used, then the {@link FileLocker}
74   * and {@link FileListFilter} objects should be set on the external
75   * {@link DirectoryScanner}, not the instance of FileReadingMessageSource. An
76   * {@link IllegalStateException} will result otherwise.
77   * <p>
78   * A {@link Comparator} can be used to ensure internal ordering of the Files in
79   * a {@link PriorityBlockingQueue}. This does not provide the same guarantees as
80   * a {@link org.springframework.integration.aggregator.ResequencingMessageGroupProcessor},
81   * but in cases where writing files
82   * and failure downstream are rare it might be sufficient.
83   * <p>
84   * FileReadingMessageSource is fully thread-safe under concurrent
85   * <code>receive()</code> invocations and message delivery callbacks.
86   *
87   * @author Iwein Fuld
88   * @author Mark Fisher
89   * @author Oleg Zhurakousky
90   * @author Gary Russell
91   * @author Artem Bilan
92   * @author Steven Pearce
93   */
94  public class FileReadingMessageSource extends AbstractMessageSource<File> implements ManageableLifecycle {

132 	/**
133 	 * Create a FileReadingMessageSource with a bounded queue of the given
134 	 * capacity. This can be used to reduce the memory footprint of this
135 	 * component when reading from a large directory.
136 	 * @param internalQueueCapacity
137 	 *            the size of the queue used to cache files to be received
138 	 *            internally. This queue can be made larger to optimize the
139 	 *            directory scanning. With scanEachPoll set to false and the
140 	 *            queue to a large size, it will be filled once and then
141 	 *            completely emptied before a new directory listing is done.
142 	 *            This is particularly useful to reduce scans of large numbers
143 	 *            of files in a directory.
144 	 */
145 	public FileReadingMessageSource(int internalQueueCapacity) {
146 		this(null);
147 		Assert.isTrue(internalQueueCapacity > 0, "Cannot create a queue with non positive capacity");
148 		this.scanner = new HeadDirectoryScanner(internalQueueCapacity);
149 	}

151 	/**
152 	 * Create a FileReadingMessageSource with a {@link PriorityBlockingQueue}
153 	 * ordered with the passed in {@link Comparator}.
154 	 * <p> The size of the queue used should be large enough to hold all the files
155 	 * in the input directory in order to sort all of them, so restricting the
156 	 * size of the queue is mutually exclusive with ordering. No guarantees
157 	 * about file delivery order can be made under concurrent access.
158 	 * @param receptionOrderComparator the comparator to be used to order the files in the internal queue
159 	 */
160 	public FileReadingMessageSource(@Nullable Comparator<File> receptionOrderComparator) {
161 		this.toBeReceived = new PriorityBlockingQueue<>(DEFAULT_INTERNAL_QUEUE_CAPACITY, receptionOrderComparator);
162 	}

237 	 * Set this flag if you want to make sure the internal queue is
238 	 * refreshed with the latest content of the input directory on each poll.
239 	 * <p>

241 	 * directory again. In cases where order is relevant it is important to
242 	 * consider the effects of setting this flag. The internal

246 	 * @param scanEachPoll

290 	@Override
291 	public void start() {
292 		if (!this.running.getAndSet(true)) {
293 			if (!this.directory.exists() && this.autoCreateDirectory && !this.directory.mkdirs()) {
294 				throw new IllegalStateException("Cannot create directory or its parents: " + this.directory);
295 			}
296 			Assert.isTrue(this.directory.exists(),
297 					() -> "Source directory [" + this.directory + "] does not exist.");
298 			Assert.isTrue(this.directory.isDirectory(),
299 					() -> "Source path [" + this.directory + "] does not point to a directory.");
300 			Assert.isTrue(this.directory.canRead(),
301 					() -> "Source directory [" + this.directory + "] is not readable.");
302 			if (this.scanner instanceof Lifecycle) {
303 				((Lifecycle) this.scanner).start();
304 			}
305 		}
306 	}
307 
308 	@Override
309 	public void stop() {
310 		if (this.running.getAndSet(false) && this.scanner instanceof Lifecycle) {
311 			((Lifecycle) this.scanner).stop();
312 		}
313 	}

320 	@Override
321 	protected void onInit() {
322 		Assert.notNull(this.directory, "'directory' must not be null");
323 
324 		Assert.state(!(this.scannerExplicitlySet && this.useWatchService),
325 				() -> "The 'scanner' and 'useWatchService' options are mutually exclusive: " + this.scanner);
326 
327 		if (this.useWatchService) {
328 			this.scanner = new WatchServiceDirectoryScanner();
329 		}
330 
331 		// Check that the filter and locker options are _NOT_ set if an external scanner has been set.
332 		// The external scanner is responsible for the filter and locker options in that case.
333 		Assert.state(!(this.scannerExplicitlySet && (this.filter != null || this.locker != null)),
334 				() -> "When using an external scanner the 'filter' and 'locker' options should not be used. " +
335 						"Instead, set these options on the external DirectoryScanner: " + this.scanner);
336 		if (this.filter != null) {
337 			this.scanner.setFilter(this.filter);
338 		}
339 		if (this.locker != null) {
340 			this.scanner.setLocker(this.locker);
341 		}

345 	@Override
346 	protected AbstractIntegrationMessageBuilder<File> doReceive() {
347 		// rescan only if needed or explicitly configured
348 		if (this.scanEachPoll || this.toBeReceived.isEmpty()) {
349 			scanInputDirectory();
350 		}
351 
352 		File file = this.toBeReceived.poll();
353 
354 		// file == null means the queue was empty
355 		// we can't rely on isEmpty for concurrency reasons
356 		while ((file != null) && !this.scanner.tryClaim(file)) {
357 			file = this.toBeReceived.poll();
358 		}
359 
360 		if (file != null) {
361 			return getMessageBuilderFactory()
362 					.withPayload(file)
363 					.setHeader(FileHeaders.RELATIVE_PATH,
364 							file.getAbsolutePath()
365 									.replaceFirst(Matcher.quoteReplacement(
366 											this.directory.getAbsolutePath() + File.separator), ""))
367 					.setHeader(FileHeaders.FILENAME, file.getName())
368 					.setHeader(FileHeaders.ORIGINAL_FILE, file);
369 		}
370 
371 		return null;
372 	}
373 
374 	private void scanInputDirectory() {
375 		List<File> filteredFiles = this.scanner.listFiles(this.directory);
376 		Set<File> freshFiles = new LinkedHashSet<>(filteredFiles);
377 		if (!freshFiles.isEmpty()) {
378 			this.toBeReceived.addAll(freshFiles);
379 			logger.debug(() -> "Added to queue: " + freshFiles);
380 		}
381 	}

383 	/**
384 	 * Adds the failed message back to the 'toBeReceived' queue if there is room.
385 	 * @param failedMessage the {@link Message} that failed
386 	 */
387 	public void onFailure(Message<File> failedMessage) {
388 		logger.warn(() -> "Failed to send: " + failedMessage);
389 		this.toBeReceived.offer(failedMessage.getPayload());
390 	}

409 	private class WatchServiceDirectoryScanner extends DefaultDirectoryScanner implements ManageableLifecycle {
410 
411 		private final ConcurrentMap<Path, WatchKey> pathKeys = new ConcurrentHashMap<>();
412 
413 		private final Set<File> filesToPoll = ConcurrentHashMap.newKeySet();
414 
415 		private WatchService watcher;
416 
417 		private WatchEvent.Kind<?>[] kinds;
418 
419 		@Override
420 		public void setFilter(FileListFilter<File> filter) {
421 			if (filter instanceof DiscardAwareFileListFilter) {
422 				((DiscardAwareFileListFilter<File>) filter).addDiscardCallback(this.filesToPoll::add);
423 			}
424 			super.setFilter(filter);
425 		}
426 
427 		@Override
428 		public void start() {
429 			try {
430 				this.watcher = FileSystems.getDefault().newWatchService();
431 			}
432 			catch (IOException ex) {
433 				logger.error(ex, () -> "Failed to create watcher for " + FileReadingMessageSource.this.directory);
434 			}
435 
436 			this.kinds = new WatchEvent.Kind<?>[FileReadingMessageSource.this.watchEvents.length];
437 
438 			for (int i = 0; i < FileReadingMessageSource.this.watchEvents.length; i++) {
439 				this.kinds[i] = FileReadingMessageSource.this.watchEvents[i].kind;
440 			}
441 
442 			Set<File> initialFiles = walkDirectory(FileReadingMessageSource.this.directory.toPath(), null);
443 			initialFiles.addAll(filesFromEvents());
444 			this.filesToPoll.addAll(initialFiles);
445 		}
446 
447 		@Override
448 		public void stop() {
449 			try {
450 				this.watcher.close();
451 				this.watcher = null;
452 				this.pathKeys.clear();
453 			}
454 			catch (IOException ex) {
455 				logger.error(ex, () -> "Failed to close watcher for " + FileReadingMessageSource.this.directory);
456 			}
457 		}
458 
459 		@Override
460 		public boolean isRunning() {
461 			return true;
462 		}
463 
464 		@Override
465 		protected File[] listEligibleFiles(File directory) {
466 			Assert.state(this.watcher != null, "The WatchService hasn't been started");
467 
468 			Set<File> files = new LinkedHashSet<>();
469 
470 			for (Iterator<File> iterator = this.filesToPoll.iterator(); iterator.hasNext(); ) {
471 				files.add(iterator.next());
472 				iterator.remove();
473 			}
474 
475 			files.addAll(filesFromEvents());
476 
477 			return files.toArray(new File[0]);
478 		}
479 
480 		private Set<File> filesFromEvents() {
481 			WatchKey key = this.watcher.poll();
482 			Set<File> files = new LinkedHashSet<>();
483 			while (key != null) {
484 				File parentDir = ((Path) key.watchable()).toAbsolutePath().toFile();
485 				for (WatchEvent<?> event : key.pollEvents()) {
486 					if (StandardWatchEventKinds.ENTRY_CREATE.equals(event.kind()) ||
487 							StandardWatchEventKinds.ENTRY_MODIFY.equals(event.kind()) ||
488 							StandardWatchEventKinds.ENTRY_DELETE.equals(event.kind())) {
489 
490 						processFilesFromNormalEvent(files, parentDir, event);
491 					}
492 					else if (StandardWatchEventKinds.OVERFLOW.equals(event.kind())) {
493 						processFilesFromOverflowEvent(files, event);
494 					}
495 				}
496 				key.reset();
497 				key = this.watcher.poll();
498 			}
499 			return files;
500 		}
501 
502 		private void processFilesFromNormalEvent(Set<File> files, File parentDir, WatchEvent<?> event) {
503 			Path item = (Path) event.context();
504 			File file = new File(parentDir, item.toFile().getName());
505 			logger.debug(() -> "Watch event [" + event.kind() + "] for file [" + file + "]");
506 
507 			if (StandardWatchEventKinds.ENTRY_DELETE.equals(event.kind())) {
508 				if (getFilter() instanceof ResettableFileListFilter) {
509 					((ResettableFileListFilter<File>) getFilter()).remove(file);
510 				}
511 				boolean fileRemoved = files.remove(file);
512 				if (fileRemoved) {
513 					logger.debug(() -> "The file [" + file +
514 							"] has been removed from the queue because of DELETE event.");
515 				}
516 			}
517 			else {
518 				if (file.exists()) {
519 					if (file.isDirectory()) {
520 						files.addAll(walkDirectory(file.toPath(), event.kind()));
521 					}
522 					else {
523 						files.remove(file);
524 						files.add(file);
525 					}
526 				}
527 				else {
528 					logger.debug(() -> "A file [" + file + "] for the event [" + event.kind() +
529 							"] doesn't exist. Ignored.");
530 				}
531 			}
532 		}
533 
534 		private void processFilesFromOverflowEvent(Set<File> files, WatchEvent<?> event) {
535 			logger.debug(() -> "Watch event [" + StandardWatchEventKinds.OVERFLOW +
536 					"] with context [" + event.context() + "]");
537 
538 			for (WatchKey watchKey : this.pathKeys.values()) {
539 				watchKey.cancel();
540 			}
541 			this.pathKeys.clear();
542 
543 			if (event.context() != null && event.context() instanceof Path) {
544 				files.addAll(walkDirectory((Path) event.context(), event.kind()));
545 			}
546 			else {
547 				files.addAll(walkDirectory(FileReadingMessageSource.this.directory.toPath(), event.kind()));
548 			}
549 		}
550 
551 		private Set<File> walkDirectory(Path directory, final WatchEvent.Kind<?> kind) {
552 			final Set<File> walkedFiles = new LinkedHashSet<>();
553 			try {
554 				registerWatch(directory);
555 				Files.walkFileTree(directory, new SimpleFileVisitor<Path>() {
556 
557 					@Override
558 					public FileVisitResult preVisitDirectory(Path dir, BasicFileAttributes attrs) throws IOException {
559 						FileVisitResult fileVisitResult = super.preVisitDirectory(dir, attrs);
560 						registerWatch(dir);
561 						return fileVisitResult;
562 					}
563 
564 					@Override
565 					public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
566 						FileVisitResult fileVisitResult = super.visitFile(file, attrs);
567 						if (!StandardWatchEventKinds.ENTRY_MODIFY.equals(kind)) {
568 							walkedFiles.add(file.toFile());
569 						}
570 						return fileVisitResult;
571 					}
572 
573 				});
574 			}
575 			catch (IOException ex) {
576 				logger.error(ex, () -> "Failed to walk directory: " + directory.toString());
577 			}
578 			return walkedFiles;
579 		}
580 
581 		private void registerWatch(Path dir) throws IOException {
582 			if (!this.pathKeys.containsKey(dir)) {
583 				logger.debug(() -> "registering: " + dir + " for file events");
584 				WatchKey watchKey = dir.register(this.watcher, this.kinds);
585 				this.pathKeys.putIfAbsent(dir, watchKey);
586 			}
587 		}
588 
589 	}
