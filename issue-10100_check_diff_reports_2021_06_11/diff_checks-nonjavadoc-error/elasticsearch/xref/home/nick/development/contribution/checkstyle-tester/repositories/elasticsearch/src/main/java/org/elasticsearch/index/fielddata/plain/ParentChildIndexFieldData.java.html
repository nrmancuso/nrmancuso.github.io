61  /**
62   * ParentChildIndexFieldData is responsible for loading the id cache mapping
63   * needed for has_child and has_parent queries into memory.
64   */
65  public class ParentChildIndexFieldData extends AbstractIndexFieldData<AtomicParentChildFieldData> implements IndexParentChildFieldData, DocumentTypeListener {
66  
67      private final NavigableSet<BytesRef> parentTypes;
68      private final CircuitBreakerService breakerService;
69  
70      // If child type (a type with _parent field) is added or removed, we want to make sure modifications don't happen
71      // while loading.
72      private final Object lock = new Object();
73  
74      public ParentChildIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames,
75                                       FieldDataType fieldDataType, IndexFieldDataCache cache, MapperService mapperService,
76                                       CircuitBreakerService breakerService) {
77          super(index, indexSettings, fieldNames, fieldDataType, cache);
78          parentTypes = new TreeSet<>(BytesRef.getUTF8SortedAsUnicodeComparator());
79          this.breakerService = breakerService;
80          for (DocumentMapper documentMapper : mapperService.docMappers(false)) {
81              beforeCreate(documentMapper);
82          }
83          mapperService.addTypeListener(this);
84      }
85  
86      @Override
87      public XFieldComparatorSource comparatorSource(@Nullable Object missingValue, MultiValueMode sortMode, Nested nested) {
88          return new BytesRefFieldComparatorSource(this, missingValue, sortMode, nested);
89      }

91      @Override
92      public ParentChildAtomicFieldData loadDirect(AtomicReaderContext context) throws Exception {
93          AtomicReader reader = context.reader();
94          final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat(
95                  "acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO
96          );
97  
98          final NavigableSet<BytesRef> parentTypes;
99          synchronized (lock) {
100             parentTypes = ImmutableSortedSet.copyOf(BytesRef.getUTF8SortedAsUnicodeComparator(), this.parentTypes);
101         }
102         boolean success = false;
103         ParentChildAtomicFieldData data = null;
104         ParentChildFilteredTermsEnum termsEnum = new ParentChildFilteredTermsEnum(
105                 new ParentChildIntersectTermsEnum(reader, UidFieldMapper.NAME, ParentFieldMapper.NAME),
106                 parentTypes
107         );
108         ParentChildEstimator estimator = new ParentChildEstimator(breakerService.getBreaker(CircuitBreaker.Name.FIELDDATA), termsEnum);
109         TermsEnum estimatedTermsEnum = estimator.beforeLoad(null);
110         ObjectObjectOpenHashMap<String, TypeBuilder> typeBuilders = ObjectObjectOpenHashMap.newInstance();
111         try {
112             try {
113                 DocsEnum docsEnum = null;
114                 for (BytesRef term = estimatedTermsEnum.next(); term != null; term = estimatedTermsEnum.next()) {
115                     // Usually this would be estimatedTermsEnum, but the
116                     // abstract TermsEnum class does not support the .type()
117                     // and .id() methods, so we skip using the wrapped
118                     // TermsEnum and delegate directly to the
119                     // ParentChildFilteredTermsEnum that was originally wrapped
120                     String type = termsEnum.type();
121                     TypeBuilder typeBuilder = typeBuilders.get(type);
122                     if (typeBuilder == null) {
123                         typeBuilders.put(type, typeBuilder = new TypeBuilder(acceptableTransientOverheadRatio, reader));
124                     }
125 
126                     BytesRef id = termsEnum.id();
127                     final long termOrd = typeBuilder.builder.nextOrdinal();
128                     assert termOrd == typeBuilder.termOrdToBytesOffset.size();
129                     typeBuilder.termOrdToBytesOffset.add(typeBuilder.bytes.copyUsingLengthPrefix(id));
130                     docsEnum = estimatedTermsEnum.docs(null, docsEnum, DocsEnum.FLAG_NONE);
131                     for (int docId = docsEnum.nextDoc(); docId != DocsEnum.NO_MORE_DOCS; docId = docsEnum.nextDoc()) {
132                         typeBuilder.builder.addDoc(docId);
133                     }
134                 }
135 
136                 ImmutableOpenMap.Builder<String, AtomicOrdinalsFieldData> typeToAtomicFieldData = ImmutableOpenMap.builder(typeBuilders.size());
137                 for (ObjectObjectCursor<String, TypeBuilder> cursor : typeBuilders) {
138                     PagedBytes.Reader bytesReader = cursor.value.bytes.freeze(true);
139                     final Ordinals ordinals = cursor.value.builder.build(fieldDataType.getSettings());
140 
141                     typeToAtomicFieldData.put(
142                             cursor.key,
143                             new PagedBytesAtomicFieldData(bytesReader, cursor.value.termOrdToBytesOffset.build(), ordinals)
144                     );
145                 }
146                 data = new ParentChildAtomicFieldData(typeToAtomicFieldData.build());
147             } finally {
148                 for (ObjectObjectCursor<String, TypeBuilder> cursor : typeBuilders) {
149                     cursor.value.builder.close();
150                 }
151             }
152             success = true;
153             return data;
154         } finally {
155             if (success) {
156                 estimator.afterLoad(estimatedTermsEnum, data.ramBytesUsed());
157             } else {
158                 estimator.afterLoad(estimatedTermsEnum, 0);
159             }
160         }
161     }
162 
163     @Override
164     public void beforeCreate(DocumentMapper mapper) {
165         synchronized (lock) {
166             ParentFieldMapper parentFieldMapper = mapper.parentFieldMapper();
167             if (parentFieldMapper.active()) {
168                 // A _parent field can never be added to an existing mapping, so a _parent field either exists on
169                 // a new created or doesn't exists. This is why we can update the known parent types via DocumentTypeListener
170                 if (parentTypes.add(new BytesRef(parentFieldMapper.type()))) {
171                     clear();
172                 }
173             }
174         }
175     }
176 
177     @Override
178     public void afterRemove(DocumentMapper mapper) {
179         synchronized (lock) {
180             ParentFieldMapper parentFieldMapper = mapper.parentFieldMapper();
181             if (parentFieldMapper.active()) {
182                 parentTypes.remove(new BytesRef(parentFieldMapper.type()));
183             }
184         }
185     }

187     class TypeBuilder {
188 
189         final PagedBytes bytes;
190         final PackedLongValues.Builder termOrdToBytesOffset;
191         final OrdinalsBuilder builder;
192 
193         TypeBuilder(float acceptableTransientOverheadRatio, AtomicReader reader) throws IOException {
194             bytes = new PagedBytes(15);
195             termOrdToBytesOffset = PackedLongValues.monotonicBuilder(PackedInts.COMPACT);
196             builder = new OrdinalsBuilder(-1, reader.maxDoc(), acceptableTransientOverheadRatio);
197         }
198     }

211     /**
212      * Estimator that wraps parent/child id field data by wrapping the data
213      * in a RamAccountingTermsEnum.
214      */
215     public class ParentChildEstimator implements PerValueEstimator {
216 
217         private final CircuitBreaker breaker;
218         private final TermsEnum filteredEnum;
219 
220         // The TermsEnum is passed in here instead of being generated in the
221         // beforeLoad() function since it's filtered inside the previous
222         // TermsEnum wrappers
223         public ParentChildEstimator(CircuitBreaker breaker, TermsEnum filteredEnum) {
224             this.breaker = breaker;
225             this.filteredEnum = filteredEnum;
226         }
227 
228         /**
229          * General overhead for ids is 2 times the length of the ID
230          */
231         @Override
232         public long bytesPerValue(BytesRef term) {
233             if (term == null) {
234                 return 0;
235             }
236             return 2 * term.length;
237         }
238 
239         /**
240          * Wraps the already filtered {@link TermsEnum} in a
241          * {@link RamAccountingTermsEnum} and returns it
242          */
243         @Override
244         public TermsEnum beforeLoad(Terms terms) throws IOException {
245             return new RamAccountingTermsEnum(filteredEnum, breaker, this, "parent/child id cache");
246         }
247 
248         /**
249          * Adjusts the breaker based on the difference between the actual usage
250          * and the aggregated estimations.
251          */
252         @Override
253         public void afterLoad(TermsEnum termsEnum, long actualUsed) {
254             assert termsEnum instanceof RamAccountingTermsEnum;
255             long estimatedBytes = ((RamAccountingTermsEnum) termsEnum).getTotalBytes();
256             breaker.addWithoutBreaking(-(estimatedBytes - actualUsed));
257         }
258     }

260     @Override
261     public IndexParentChildFieldData loadGlobal(IndexReader indexReader) {
262         if (indexReader.leaves().size() <= 1) {
263             // ordinals are already global
264             return this;
265         }
266         try {
267             return cache.load(indexReader, this);
268         } catch (Throwable e) {
269             if (e instanceof ElasticsearchException) {
270                 throw (ElasticsearchException) e;
271             } else {
272                 throw new ElasticsearchException(e.getMessage(), e);
273             }
274         }
275     }

277     private static OrdinalMap buildOrdinalMap(AtomicParentChildFieldData[] atomicFD, String parentType) throws IOException {
278         final SortedDocValues[] ordinals = new SortedDocValues[atomicFD.length];
279         for (int i = 0; i < ordinals.length; ++i) {
280             ordinals[i] = atomicFD[i].getOrdinalsValues(parentType);
281         }
282         return OrdinalMap.build(null, ordinals, PackedInts.DEFAULT);
283     }

285     private static class OrdinalMapAndAtomicFieldData {
286         final OrdinalMap ordMap;
287         final AtomicParentChildFieldData[] fieldData;
288 
289         public OrdinalMapAndAtomicFieldData(OrdinalMap ordMap, AtomicParentChildFieldData[] fieldData) {
290             this.ordMap = ordMap;
291             this.fieldData = fieldData;
292         }
293     }

295     @Override
296     public IndexParentChildFieldData localGlobalDirect(IndexReader indexReader) throws Exception {
297         final long startTime = System.nanoTime();
298         final Set<String> parentTypes = new HashSet<>();
299         synchronized (lock) {
300             for (BytesRef type : this.parentTypes) {
301                 parentTypes.add(type.utf8ToString());
302             }
303         }
304 
305         long ramBytesUsed = 0;
306         final Map<String, OrdinalMapAndAtomicFieldData> perType = new HashMap<>();
307         for (String type : parentTypes) {
308             final AtomicParentChildFieldData[] fieldData = new AtomicParentChildFieldData[indexReader.leaves().size()];
309             for (AtomicReaderContext context : indexReader.leaves()) {
310                 fieldData[context.ord] = load(context);
311             }
312             final OrdinalMap ordMap = buildOrdinalMap(fieldData, type);
313             ramBytesUsed += ordMap.ramBytesUsed();
314             perType.put(type, new OrdinalMapAndAtomicFieldData(ordMap, fieldData));
315         }
316 
317         final AtomicParentChildFieldData[] fielddata = new AtomicParentChildFieldData[indexReader.leaves().size()];
318         for (int i = 0; i < fielddata.length; ++i) {
319             fielddata[i] = new GlobalAtomicFieldData(parentTypes, perType, i);
320         }
321 
322         breakerService.getBreaker(CircuitBreaker.Name.FIELDDATA).addWithoutBreaking(ramBytesUsed);
323         if (logger.isDebugEnabled()) {
324             logger.debug(
325                     "Global-ordinals[_parent] took {}",
326                     new TimeValue(System.nanoTime() - startTime, TimeUnit.NANOSECONDS)
327             );
328         }
329 
330         return new GlobalFieldData(indexReader, fielddata, ramBytesUsed);
331     }

333     private static class GlobalAtomicFieldData extends AbstractAtomicParentChildFieldData {
334 
335         private final Set<String> types;
336         private final Map<String, OrdinalMapAndAtomicFieldData> atomicFD;
337         private final int segmentIndex;
338 
339         public GlobalAtomicFieldData(Set<String> types, Map<String, OrdinalMapAndAtomicFieldData> atomicFD, int segmentIndex) {
340             this.types = types;
341             this.atomicFD = atomicFD;
342             this.segmentIndex = segmentIndex;
343         }
344 
345         @Override
346         public Set<String> types() {
347             return types;
348         }
349 
350         @Override
351         public SortedDocValues getOrdinalsValues(String type) {
352             final OrdinalMapAndAtomicFieldData atomicFD = this.atomicFD.get(type);
353             final OrdinalMap ordMap = atomicFD.ordMap;
354             final SortedDocValues[] allSegmentValues = new SortedDocValues[atomicFD.fieldData.length];
355             for (int i = 0; i < allSegmentValues.length; ++i) {
356                 allSegmentValues[i] = atomicFD.fieldData[i].getOrdinalsValues(type);
357             }
358             final SortedDocValues segmentValues = allSegmentValues[segmentIndex];
359             if (segmentValues.getValueCount() == ordMap.getValueCount()) {
360                 // ords are already global
361                 return segmentValues;
362             }
363             final LongValues globalOrds = ordMap.getGlobalOrds(segmentIndex);
364             return new SortedDocValues() {
365 
366                 @Override
367                 public BytesRef lookupOrd(int ord) {
368                     final int segmentIndex = ordMap.getFirstSegmentNumber(ord);
369                     final int segmentOrd = (int) ordMap.getFirstSegmentOrd(ord);
370                     return allSegmentValues[segmentIndex].lookupOrd(segmentOrd);
371                 }
372 
373                 @Override
374                 public int getValueCount() {
375                     return (int) ordMap.getValueCount();
376                 }
377 
378                 @Override
379                 public int getOrd(int docID) {
380                     final int segmentOrd = segmentValues.getOrd(docID);
381                     // TODO: is there a way we can get rid of this branch?
382                     if (segmentOrd >= 0) {
383                         return (int) globalOrds.get(segmentOrd);
384                     } else {
385                         return segmentOrd;
386                     }
387                 }
388             };
389         }
390 
391         @Override
392         public long ramBytesUsed() {
393             // this class does not take memory on its own, the index-level field data does
394             // it through the use of ordinal maps
395             return 0;
396         }
397 
398         @Override
399         public void close() throws ElasticsearchException {
400             List<Releasable> closeables = new ArrayList<>();
401             for (OrdinalMapAndAtomicFieldData fds : atomicFD.values()) {
402                 closeables.addAll(Arrays.asList(fds.fieldData));
403             }
404             Releasables.close(closeables);
405         }
406 
407     }

409     private class GlobalFieldData implements IndexParentChildFieldData, Accountable {
410 
411         private final AtomicParentChildFieldData[] fielddata;
412         private final IndexReader reader;
413         private final long ramBytesUsed;
414 
415         GlobalFieldData(IndexReader reader, AtomicParentChildFieldData[] fielddata, long ramBytesUsed) {
416             this.reader = reader;
417             this.ramBytesUsed = ramBytesUsed;
418             this.fielddata = fielddata;
419         }
420 
421         @Override
422         public Names getFieldNames() {
423             return ParentChildIndexFieldData.this.getFieldNames();
424         }
425 
426         @Override
427         public FieldDataType getFieldDataType() {
428             return ParentChildIndexFieldData.this.getFieldDataType();
429         }
430 
431         @Override
432         public AtomicParentChildFieldData load(AtomicReaderContext context) {
433             assert context.reader().getCoreCacheKey() == reader.leaves().get(context.ord).reader().getCoreCacheKey();
434             return fielddata[context.ord];
435         }
436 
437         @Override
438         public AtomicParentChildFieldData loadDirect(AtomicReaderContext context) throws Exception {
439             return load(context);
440         }
441 
442         @Override
443         public XFieldComparatorSource comparatorSource(Object missingValue, MultiValueMode sortMode, Nested nested) {
444             throw new UnsupportedOperationException("No sorting on global ords");
445         }
446 
447         @Override
448         public void clear() {
449             ParentChildIndexFieldData.this.clear();
450         }
451 
452         @Override
453         public void clear(IndexReader reader) {
454             ParentChildIndexFieldData.this.clear(reader);
455         }
456 
457         @Override
458         public Index index() {
459             return ParentChildIndexFieldData.this.index();
460         }
461 
462         @Override
463         public long ramBytesUsed() {
464             return ramBytesUsed;
465         }
466 
467         @Override
468         public IndexParentChildFieldData loadGlobal(IndexReader indexReader) {
469             if (indexReader.getCoreCacheKey() == reader.getCoreCacheKey()) {
470                 return this;
471             }
472             throw new ElasticsearchIllegalStateException();
473         }
474 
475         @Override
476         public IndexParentChildFieldData localGlobalDirect(IndexReader indexReader) throws Exception {
477             return loadGlobal(indexReader);
478         }
