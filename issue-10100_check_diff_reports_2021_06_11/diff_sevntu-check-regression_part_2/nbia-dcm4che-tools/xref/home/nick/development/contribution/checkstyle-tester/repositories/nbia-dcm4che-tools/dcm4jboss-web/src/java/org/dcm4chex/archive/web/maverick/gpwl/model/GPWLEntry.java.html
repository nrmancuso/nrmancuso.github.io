40  package org.dcm4chex.archive.web.maverick.gpwl.model;
41  
42  import java.text.SimpleDateFormat;
43  import java.util.Date;
44  
45  import org.dcm4che.data.Dataset;
46  import org.dcm4che.data.DcmElement;
47  import org.dcm4che.dict.Tags;
48  
49  /**
50   * @author franz.willer
51   *
52   * Holds a single entry of the MWLList from the model.
53   * <p>
54   * This class is a wrapper over a Dataset to get values with getter methods for the view. 
55   */
56  public class GPWLEntry {
57  
58      private Dataset ds;
59      private Dataset swcItem;
60      private Dataset refReqItem;
61  
62      /** The Date/Time formatter to format date/time values. */
63      private static final SimpleDateFormat dtformatter = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");
64      /** The Date/Time formatter to format date/time values. */
65      private static final SimpleDateFormat shortDTformatter = new SimpleDateFormat("yyyy/MM/dd HH:mm");
66      /** The Date formatter to format date values. */
67      private static final SimpleDateFormat dformatter = new SimpleDateFormat("yyyy/MM/dd");
68  
69  
70      public GPWLEntry( Dataset ds ) {
71          this.ds = ds;
72          swcItem = ds.getItem( Tags.ScheduledWorkitemCodeSeq );//scheduled workitem code sequence item.
73          refReqItem = ds.getItem(Tags.RefRequestSeq);
74      }
75  
76      public Dataset toDataset() {
77          return ds;
78      }
79  
80      public String getIUID() {
81          return ds.getString( Tags.SOPInstanceUID );
82      }
83      public String getGPSPSStatus() {
84          return ds.getString( Tags.GPSPSStatus );
85      }
86      public String getGPSPSPriority() {
87          return ds.getString( Tags.GPSPSPriority );
88      }
89      public String getInputAvailabilityFlag() {
90          return ds.getString( Tags.InputAvailabilityFlag );
91      }
92  
93      /**
94       * Returns the Scheduled Step Procedure ID.
95       * <p>
96       * (This is optional in GP; Instance UID as unique identifier) 
97       * @return
98       */
99      public String getSpsID() {
100         return ds.getString( Tags.SPSID );
101     }
102 
103     /**
104      * @return Returns the spsStartDateTime.
105      */
106     public String getSpsStartDateTime() {
107         Date d = ds.getDate( Tags.SPSStartDateAndTime );
108         if ( d == null ) {
109             d = ds.getDateTime( Tags.SPSStartDate, Tags.SPSStartTime );
110         }
111         if ( d == null ) return "";
112 
113         return shortDTformatter.format( d );
114     }
115 
116 
117     public String getAccessionNumber() {
118         return refReqItem==null ? null : refReqItem.getString( Tags.AccessionNumber );
119     }
120 
121     public String getPatientName() {
122         return ds.getString( Tags.PatientName );
123     }
124     public String getPatientID() {
125         return ds.getString( Tags.PatientID );
126     }
127 
128     public String getWorkItemCode() {
129         return getCodeValues( ds.get( Tags.ScheduledWorkitemCodeSeq ) );
130     }
131     public String getStationName() {
132         return getCodeValues( ds.get( Tags.ScheduledStationNameCodeSeq ) );
133     }
134     public String getStationClass() {
135         return getCodeValues( ds.get( Tags.ScheduledStationClassCodeSeq ) );
136     }
137     public String getStationGeoLocation() {
138         return getCodeValues( ds.get( Tags.ScheduledStationGeographicLocationCodeSeq ) );
139     }
140 
141     public String getExpectedCompletionDate() {
142         Date d = ds.getDate(Tags.ExpectedCompletionDateAndTime);
143         return d == null ? "" : shortDTformatter.format( d );
144     }
145 
146     public String[] getHumanPerformers() {
147         DcmElement elem = ds.get(Tags.ScheduledHumanPerformersSeq);
148         if ( elem != null ) {
149             String[] performers = new String[elem.countItems()];
150             for ( int i = 0 ; i < elem.countItems() ; i++) {
151                 performers[i] = elem.getItem(i).getItem(Tags.HumanPerformerCodeSeq).getString(Tags.CodeMeaning);
152             }
153             return performers;
154         }
155         return null;
156     }
157 
158     public String getBirthdate() {
159         if ( !ds.containsValue(Tags.PatientBirthDate)) return "";
160         Date d = ds.getDate(Tags.PatientBirthDate);
161         return d == null ? "" : dformatter.format(d);
162     }
163 
164     public String getPatientSex() {
165         return ds.getString(Tags.PatientSex);
166     }
167 
168     private String getCodeValues( DcmElement elem ) {
169         int len = elem.countItems();
170         Dataset dsCode;
171         StringBuffer sb = new StringBuffer();
172         for ( int i = 0 ; i < len ; i++ ) {
173             dsCode = elem.getItem( i );
174             if ( i > 0 ) sb.append("/");
175             sb.append( dsCode.getString( Tags.CodeMeaning ) ).append("[");
176             sb.append( dsCode.getString( Tags.CodeValue ) ).append("]");
177         }
178         return sb.toString();
179 
180     }
181 
182     public String getReqProcedureDescription() {
183         String desc = null;
184         DcmElement rpcs = ds.get( Tags.RequestedProcedureCodeSeq );
185         if ( rpcs != null ) {
186             desc = getCodeValues( rpcs );
187         } else {
188             desc = ds.getString( Tags.RequestedProcedureDescription );
189         }
190         return desc;
191     }
192 }
