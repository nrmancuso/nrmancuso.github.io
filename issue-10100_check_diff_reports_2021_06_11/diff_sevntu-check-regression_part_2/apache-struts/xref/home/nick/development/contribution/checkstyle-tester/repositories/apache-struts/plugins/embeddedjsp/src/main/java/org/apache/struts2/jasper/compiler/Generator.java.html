108     /**
109      * @param s
110      *            the input string
111      * @return quoted and escaped string, per Java rule
112      */
113     static String quote(String s) {
114 
115         if (s == null)
116             return "null";
117 
118         return '"' + escape(s) + '"';
119     }

121     /**
122      * @param s
123      *            the input string
124      * @return escaped string, per Java rule
125      */
126     static String escape(String s) {
127 
128         if (s == null)
129             return "";
130 
131         StringBuffer b = new StringBuffer();
132         for (int i = 0; i < s.length(); i++) {
133             char c = s.charAt(i);
134             if (c == '"')
135                 b.append('\\').append('"');
136             else if (c == '\\')
137                 b.append('\\').append('\\');
138             else if (c == '\n')
139                 b.append('\\').append('n');
140             else if (c == '\r')
141                 b.append('\\').append('r');
142             else
143                 b.append(c);
144         }
145         return b.toString();
146     }
147 
148     /**
149      * Single quote and escape a character
150      */
151     static String quote(char c) {
152 
153         StringBuffer b = new StringBuffer();
154         b.append('\'');
155         if (c == '\'')
156             b.append('\\').append('\'');
157         else if (c == '\\')
158             b.append('\\').append('\\');
159         else if (c == '\n')
160             b.append('\\').append('n');
161         else if (c == '\r')
162             b.append('\\').append('r');
163         else
164             b.append(c);
165         b.append('\'');
166         return b.toString();
167     }

169     private String createJspId() throws JasperException {
170         if (this.jspIdPrefix == null) {
171             StringBuffer sb = new StringBuffer(32);
172             String name = ctxt.getServletJavaFileName();
173             sb.append("jsp_").append(Math.abs(name.hashCode())).append('_');
174             this.jspIdPrefix = sb.toString();
175         }
176         return this.jspIdPrefix + (this.jspId++);
177     }

179     /**
180      * Generates declarations. This includes "info" of the page directive, and
181      * scriptlet declarations.
182      */
183     private void generateDeclarations(Node.Nodes page) throws JasperException {
184 
185         class DeclarationVisitor extends Node.Visitor {
186 
187             private boolean getServletInfoGenerated = false;
188 
189             /*
190              * Generates getServletInfo() method that returns the value of the
191              * page directive's 'info' attribute, if present.
192              * 
193              * The Validator has already ensured that if the translation unit
194              * contains more than one page directive with an 'info' attribute,
195              * their values match.
196              */
197             public void visit(Node.PageDirective n) throws JasperException {
198 
199                 if (getServletInfoGenerated) {
200                     return;
201                 }
202 
203                 String info = n.getAttributeValue("info");
204                 if (info == null)
205                     return;
206 
207                 getServletInfoGenerated = true;
208                 out.printil("public String getServletInfo() {");
209                 out.pushIndent();
210                 out.printin("return ");
211                 out.print(quote(info));
212                 out.println(";");
213                 out.popIndent();
214                 out.printil("}");
215                 out.println();
216             }
217 
218             public void visit(Node.Declaration n) throws JasperException {
219                 n.setBeginJavaLine(out.getJavaLine());
220                 out.printMultiLn(new String(n.getText()));
221                 out.println();
222                 n.setEndJavaLine(out.getJavaLine());
223             }
224 
225             // Custom Tags may contain declarations from tag plugins.
226             public void visit(Node.CustomTag n) throws JasperException {
227                 if (n.useTagPlugin()) {
228                     if (n.getAtSTag() != null) {
229                         n.getAtSTag().visit(this);
230                     }
231                     visitBody(n);
232                     if (n.getAtETag() != null) {
233                         n.getAtETag().visit(this);
234                     }
235                 } else {
236                     visitBody(n);
237                 }
238             }
239         }
240 
241         out.println();
242         page.visit(new DeclarationVisitor());
243     }

245     /**
246      * Compiles list of tag handler pool names.
247      */
248     private void compileTagHandlerPoolList(Node.Nodes page)
249             throws JasperException {
250 
251         class TagHandlerPoolVisitor extends Node.Visitor {
252 
253             private Vector names;
254 
255             /*
256              * Constructor
257              * 
258              * @param v Vector of tag handler pool names to populate
259              */
260             TagHandlerPoolVisitor(Vector v) {
261                 names = v;
262             }
263 
264             /*
265              * Gets the name of the tag handler pool for the given custom tag
266              * and adds it to the list of tag handler pool names unless it is
267              * already contained in it.
268              */
269             public void visit(Node.CustomTag n) throws JasperException {
270 
271                 if (!n.implementsSimpleTag()) {
272                     String name = createTagHandlerPoolName(n.getPrefix(), n
273                             .getLocalName(), n.getAttributes(), 
274                             n.getNamedAttributeNodes(), n.hasEmptyBody());
275                     n.setTagHandlerPoolName(name);
276                     if (!names.contains(name)) {
277                         names.add(name);
278                     }
279                 }
280                 visitBody(n);
281             }
282 
283             /*
284              * Creates the name of the tag handler pool whose tag handlers may
285              * be (re)used to service this action.
286              * 
287              * @return The name of the tag handler pool
288              */
289             private String createTagHandlerPoolName(String prefix,
290                     String shortName, Attributes attrs, Node.Nodes namedAttrs,
291                     boolean hasEmptyBody) {
292                 String poolName = null;
293 
294                 poolName = "_jspx_tagPool_" + prefix + "_" + shortName;
295                 if (attrs != null) {
296                     String[] attrNames =
297                         new String[attrs.getLength() + namedAttrs.size()];
298                     for (int i = 0; i < attrNames.length; i++) {
299                         attrNames[i] = attrs.getQName(i);
300                     }
301                     for (int i = 0; i < namedAttrs.size(); i++) {
302                         attrNames[attrs.getLength() + i] =
303                             ((NamedAttribute) namedAttrs.getNode(i)).getQName();
304                     }
305                     Arrays.sort(attrNames, Collections.reverseOrder());
306                     if (attrNames.length > 0) {
307                         poolName = poolName + "&";
308                     }
309                     for (int i = 0; i < attrNames.length; i++) {
310                         poolName = poolName + "_" + attrNames[i];
311                     }
312                 }
313                 if (hasEmptyBody) {
314                     poolName = poolName + "_nobody";
315                 }
316                 return JspUtil.makeJavaIdentifier(poolName);
317             }
318         }
319 
320         page.visit(new TagHandlerPoolVisitor(tagHandlerPoolNames));
321     }

323     private void declareTemporaryScriptingVars(Node.Nodes page)
324             throws JasperException {
325 
326         class ScriptingVarVisitor extends Node.Visitor {
327 
328             private Vector vars;
329 
330             ScriptingVarVisitor() {
331                 vars = new Vector();
332             }
333 
334             public void visit(Node.CustomTag n) throws JasperException {
335 
336                 if (n.getCustomNestingLevel() > 0) {
337                     TagVariableInfo[] tagVarInfos = n.getTagVariableInfos();
338                     VariableInfo[] varInfos = n.getVariableInfos();
339 
340                     if (varInfos.length > 0) {
341                         for (int i = 0; i < varInfos.length; i++) {
342                             String varName = varInfos[i].getVarName();
343                             String tmpVarName = "_jspx_" + varName + "_"
344                                     + n.getCustomNestingLevel();
345                             if (!vars.contains(tmpVarName)) {
346                                 vars.add(tmpVarName);
347                                 out.printin(varInfos[i].getClassName());
348                                 out.print(" ");
349                                 out.print(tmpVarName);
350                                 out.print(" = ");
351                                 out.print(null);
352                                 out.println(";");
353                             }
354                         }
355                     } else {
356                         for (int i = 0; i < tagVarInfos.length; i++) {
357                             String varName = tagVarInfos[i].getNameGiven();
358                             if (varName == null) {
359                                 varName = n.getTagData().getAttributeString(
360                                         tagVarInfos[i].getNameFromAttribute());
361                             } else if (tagVarInfos[i].getNameFromAttribute() != null) {
362                                 // alias
363                                 continue;
364                             }
365                             String tmpVarName = "_jspx_" + varName + "_"
366                                     + n.getCustomNestingLevel();
367                             if (!vars.contains(tmpVarName)) {
368                                 vars.add(tmpVarName);
369                                 out.printin(tagVarInfos[i].getClassName());
370                                 out.print(" ");
371                                 out.print(tmpVarName);
372                                 out.print(" = ");
373                                 out.print(null);
374                                 out.println(";");
375                             }
376                         }
377                     }
378                 }
379 
380                 visitBody(n);
381             }
382         }
383 
384         page.visit(new ScriptingVarVisitor());
385     }

562

565         String serviceMethodName = Constants.SERVICE_METHOD_NAME;

587         generateDeclarations(page);
588 
589         // Static initializations here
590         genPreambleStaticInitializers();
591 
592         // Class variable declarations
593         genPreambleClassVariableDeclarations(servletClassName);
594 
595         // Constructor
596         // generateConstructor(className);

598         // Methods here
599         genPreambleMethods();

611         out.printil("PageContext pageContext = null;");

617             out.printil("Throwable exception = org.apache.struts2.jasper.runtime.JspRuntimeLibrary.getThrowable(request);");

619             out.pushIndent();

626         out.printil("ServletConfig config = null;");
627         out.printil("JspWriter out = null;");
628         out.printil("Object page = this;");
629

631         out.printil("PageContext _jspx_page_context = null;");
632         out.println();
633 
634         declareTemporaryScriptingVars(page);
635         out.println();
636 
637         out.printil("try {");
638         out.pushIndent();
639 
640         out.printin("response.setContentType(");
641         out.print(quote(pageInfo.getContentType()));
642         out.println(");");
643 
644         if (ctxt.getOptions().isXpoweredBy()) {
645             out.printil("response.addHeader(\"X-Powered-By\", \"JSP/2.1\");");
646         }

648         out
649                 .printil("pageContext = _jspxFactory.getPageContext(this, request, response,");
650         out.printin("\t\t\t");
651         out.print(quote(pageInfo.getErrorPage()));
652         out.print(", " + pageInfo.isSession());
653         out.print(", " + pageInfo.getBuffer());
654         out.print(", " + pageInfo.isAutoFlush());
655         out.println(");");
656         out.printil("_jspx_page_context = pageContext;");
657 
658         out.printil("application = pageContext.getServletContext();");
659         out.printil("config = pageContext.getServletConfig();");
660 
661         if (pageInfo.isSession())
662             out.printil("session = pageContext.getSession();");
663         out.printil("out = pageContext.getOut();");
664         out.printil("_jspx_out = out;");
665         out.println();
666     }

668     /**
669      * Generates an XML Prolog, which includes an XML declaration and an XML
670      * doctype declaration.
671      */
672     private void generateXmlProlog(Node.Nodes page) {
673 
674         /*
675          * An XML declaration is generated under the following conditions: -
676          * 'omit-xml-declaration' attribute of <jsp:output> action is set to
677          * "no" or "false" - JSP document without a <jsp:root>
678          */
679         String omitXmlDecl = pageInfo.getOmitXmlDecl();
680         if ((omitXmlDecl != null && !JspUtil.booleanValue(omitXmlDecl))
681                 || (omitXmlDecl == null && page.getRoot().isXmlSyntax()
682                         && !pageInfo.hasJspRoot() && !ctxt.isTagFile())) {
683             String cType = pageInfo.getContentType();
684             String charSet = cType.substring(cType.indexOf("charset=") + 8);
685             out.printil("out.write(\"<?xml version=\\\"1.0\\\" encoding=\\\""
686                     + charSet + "\\\"?>\\n\");");
687         }
688 
689         /*
690          * Output a DOCTYPE declaration if the doctype-root-element appears. If
691          * doctype-public appears: <!DOCTYPE name PUBLIC "doctypePublic"
692          * "doctypeSystem"> else <!DOCTYPE name SYSTEM "doctypeSystem" >
693          */
694 
695         String doctypeName = pageInfo.getDoctypeName();
696         if (doctypeName != null) {
697             String doctypePublic = pageInfo.getDoctypePublic();
698             String doctypeSystem = pageInfo.getDoctypeSystem();
699             out.printin("out.write(\"<!DOCTYPE ");
700             out.print(doctypeName);
701             if (doctypePublic == null) {
702                 out.print(" SYSTEM \\\"");
703             } else {
704                 out.print(" PUBLIC \\\"");
705                 out.print(doctypePublic);
706                 out.print("\\\" \\\"");
707             }
708             out.print(doctypeSystem);
709             out.println("\\\">\\n\");");
710         }
711     }

778         /**
779          * Returns an attribute value, optionally URL encoded. If the value is a
780          * runtime expression, the result is the expression itself, as a string.
781          * If the result is an EL expression, we insert a call to the
782          * interpreter. If the result is a Named Attribute we insert the
783          * generated variable name. Otherwise the result is a string literal,
784          * quoted and escaped.
785          * 
786          * @param attr
787          *            An JspAttribute object
788          * @param encode
789          *            true if to be URL encoded
790          * @param expectedType
791          *            the expected type for an EL evaluation (ignored for
792          *            attributes that aren't EL expressions)
793          */
794         private String attributeValue(Node.JspAttribute attr, boolean encode,
795                 Class expectedType) {
796             String v = attr.getValue();
797             if (!attr.isNamedAttribute() && (v == null))
798                 return "";
799 
800             if (attr.isExpression()) {
801                 if (encode) {
802                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode(String.valueOf("
803                             + v + "), request.getCharacterEncoding())";
804                 }
805                 return v;
806             } else if (attr.isELInterpreterInput()) {
807                 v = attributeValueWithEL(this.isTagFile, v, expectedType,
808                         attr.getEL().getMapName());
809                 if (encode) {
810                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode("
811                             + v + ", request.getCharacterEncoding())";
812                 }
813                 return v;
814             } else if (attr.isNamedAttribute()) {
815                 return attr.getNamedAttributeNode().getTemporaryVariableName();
816             } else {
817                 if (encode) {
818                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode("
819                             + quote(v) + ", request.getCharacterEncoding())";
820                 }
821                 return quote(v);
822             }
823         }

826         /*
827          * When interpreting the EL attribute value, literals outside the EL
828          * must not be unescaped but the EL processor will unescape them.
829          * Therefore, make sure only the EL expressions are processed by the EL
830          * processor.
831          */
832         private String attributeValueWithEL(boolean isTag, String tx,
833                 Class<?> expectedType, String mapName) {
834             if (tx==null) return null;
835             Class<?> type = expectedType;
836             int size = tx.length();
837             StringBuffer output = new StringBuffer(size);
838             boolean el = false;
839             int i = 0;
840             int mark = 0;
841             char ch;
842             
843             while(i < size){
844                 ch = tx.charAt(i);
845                 
846                 // Start of an EL expression
847                 if (!el && i+1 < size && ch == '$' && tx.charAt(i+1)=='{') {
848                     if (mark < i) {
849                         if (output.length() > 0) {
850                             output.append(" + ");
851                             // Composite expression - must coerce to String
852                             type = String.class;
853                         }
854                         output.append(quote(tx.substring(mark, i)));
855                     }
856                     mark = i;
857                     el = true;
858                     i += 2;
859                 } else if (ch=='\\' && i+1 < size &&
860                         (tx.charAt(i+1)=='$' || tx.charAt(i+1)=='}')) { 
861                     // Skip an escaped $ or }
862                     i += 2;
863                 } else if (el && ch=='}') {
864                     // End of an EL expression
865                     if (output.length() > 0) {
866                         output.append(" + ");
867                         // Composite expression - must coerce to String
868                         type = String.class;
869                     }
870                     output.append(
871                             JspUtil.interpreterCall(isTag,
872                                     tx.substring(mark, i+1), type,
873                                     mapName, false));
874                     mark = i + 1;
875                     el = false;
876                     ++i;
877                 } else {
878                     // Nothing to see here - move to next character
879                     ++i;
880                 }
881             }
882             if (!el && mark < i) {
883                 if (output.length() > 0) {
884                     output.append(" + ");
885                 }
886                 output.append(quote(tx.substring(mark, i)));
887             }
888             return output.toString();
889         }

1238                     Class bean = ctxt.getClassLoader().loadClass(klass);
1239                     if (klass.indexOf('$') >= 0) {
1240                         // Obtain the canonical type name
1241                         canonicalName = JspUtil.getCanonicalName(bean);
1242                     } else {
1243                         canonicalName = klass;
1244                     }
1245                     int modifiers = bean.getModifiers();
1246                     if (!Modifier.isPublic(modifiers)
1247                             || Modifier.isInterface(modifiers)
1248                             || Modifier.isAbstract(modifiers)) {
1249                         throw new Exception("Invalid bean class modifier");
1250                     }
1251                     // Check that there is a 0 arg constructor
1252                     bean.getConstructor(new Class[] {});
1253                     // At compile time, we have determined that the bean class
1254                     // exists, with a public zero constructor, new() can be
1255                     // used for bean instantiation.
1256                     generateNew = true;

1259                     // compilation error or a runtime error will be raised,

1262                             .getErrorOnUseBeanInvalidClassAttribute()) {
1263                         err.jspError(n, "jsp.error.invalid.bean", klass);

1266                         // Doing our best here to get a canonical name
1267                         // from the binary name, should work 99.99% of time.
1268                         canonicalName = klass.replace('$', '.');
1269                     }
1270                 }
1271                 if (type == null) {
1272                     // if type is unspecified, use "class" as type of bean
1273                     type = canonicalName;
1274                 }
1275             }

1284                 scopename = "PageContext.SESSION_SCOPE";
1285                 lock = "session";
1286             } else if ("application".equals(scope)) {

1288                 lock = "application";

1312             out.print(scopename);

1318              */

1324                 /*
1325                  * If both class name and beanName is not specified, the bean
1326                  * must be found locally, otherwise it's an error
1327                  */

1329                         .printin("throw new java.lang.InstantiationException(\"bean ");
1330                 out.print(name);
1331                 out.println(" not found within scope\");");

1333                 /*
1334                  * Instantiate the bean if it is not in the specified scope.
1335                  */
1336                 if (!generateNew) {
1337                     String binaryName;
1338                     if (beanName != null) {
1339                         if (beanName.isNamedAttribute()) {
1340                             // If the value for beanName was specified via
1341                             // jsp:attribute, first generate code to evaluate
1342                             // that body.
1343                             binaryName = generateNamedAttributeValue(beanName
1344                                     .getNamedAttributeNode());
1345                         } else {
1346                             binaryName = attributeValue(beanName, false,
1347                                     String.class);
1348                         }
1349                     } else {
1350                         // Implies klass is not null
1351                         binaryName = quote(klass);
1352                     }
1353                     out.printil("try {");
1354                     out.pushIndent();

1356                     out.print(" = (");

1396                 out.print(scopename);
