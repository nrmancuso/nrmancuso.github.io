101     /**
102      * Find the list of environment variables for this process.
103      *
104      * @return a map containing the environment variables.
105      * @since Ant 1.8.2
106      */
107     public static synchronized Map<String,String> getEnvironmentVariables() {
108         if (procEnvironment != null) {
109             return procEnvironment;
110         }
111         if (!Os.isFamily("openvms")) {
112             try {
113                 procEnvironment = System.getenv();
114                 return procEnvironment;
115             } catch (Exception x) {
116                 x.printStackTrace();
117             }
118         }
119 
120         procEnvironment = new LinkedHashMap<String, String>();
121         try {
122             ByteArrayOutputStream out = new ByteArrayOutputStream();
123             Execute exe = new Execute(new PumpStreamHandler(out));
124             exe.setCommandline(getProcEnvCommand());
125             // Make sure we do not recurse forever
126             exe.setNewenvironment(true);
127             int retval = exe.execute();
128             if (retval != 0) {
129                 // Just try to use what we got
130             }
131             BufferedReader in =
132                 new BufferedReader(new StringReader(toString(out)));
133 
134             if (Os.isFamily("openvms")) {
135                 procEnvironment = getVMSLogicals(in);
136                 return procEnvironment;
137             }
138             String var = null;
139             String line, lineSep = StringUtils.LINE_SEP;
140             while ((line = in.readLine()) != null) {
141                 if (line.indexOf('=') == -1) {
142                     // Chunk part of previous env var (UNIX env vars can
143                     // contain embedded new lines).
144                     if (var == null) {
145                         var = lineSep + line;
146                     } else {
147                         var += lineSep + line;
148                     }
149                 } else {
150                     // New env var...append the previous one if we have it.
151                     if (var != null) {
152                         int eq = var.indexOf("=");
153                         procEnvironment.put(var.substring(0, eq),
154                                             var.substring(eq + 1));
155                     }
156                     var = line;
157                 }
158             }
159             // Since we "look ahead" before adding, there's one last env var.
160             if (var != null) {
161                 int eq = var.indexOf("=");
162                 procEnvironment.put(var.substring(0, eq), var.substring(eq + 1));
163             }
164         } catch (java.io.IOException exc) {
165             exc.printStackTrace();
166             // Just try to see how much we got
167         }
168         return procEnvironment;
169     }

187     /**
188      * This is the operation to get our environment.
189      * It is a notorious troublespot pre-Java1.5, and should be approached
190      * with extreme caution.
191      *
192      * @return
193      */
194     private static String[] getProcEnvCommand() {
195         if (Os.isFamily("os/2")) {
196             // OS/2 - use same mechanism as Windows 2000
197             return new String[] {"cmd", "/c", "set"};
198         } else if (Os.isFamily("windows")) {
199             // Determine if we're running under XP/2000/NT or 98/95
200             if (Os.isFamily("win9x")) {
201                 // Windows 98/95
202                 return new String[] {"command.com", "/c", "set"};
203             } else {
204                 // Windows XP/2000/NT/2003
205                 return new String[] {"cmd", "/c", "set"};
206             }
207         } else if (Os.isFamily("z/os") || Os.isFamily("unix")) {
208             // On most systems one could use: /bin/sh -c env
209 
210             // Some systems have /bin/env, others /usr/bin/env, just try
211             String[] cmd = new String[1];
212             if (new File("/bin/env").canRead()) {
213                 cmd[0] = "/bin/env";
214             } else if (new File("/usr/bin/env").canRead()) {
215                 cmd[0] = "/usr/bin/env";
216             } else {
217                 // rely on PATH
218                 cmd[0] = "env";
219             }
220             return cmd;
221         } else if (Os.isFamily("netware") || Os.isFamily("os/400")) {
222             // rely on PATH
223             return new String[] {"env"};
224         } else if (Os.isFamily("openvms")) {
225             return new String[] {"show", "logical"};
226         } else {
227             // MAC OS 9 and previous
228             // TODO: I have no idea how to get it, someone must fix it
229             return null;
230         }
231     }

233     /**
234      * ByteArrayOutputStream#toString doesn't seem to work reliably on
235      * OS/390, at least not the way we use it in the execution
236      * context.
237      *
238      * @param bos the output stream that one wants to read.
239      * @return the output stream as a string, read with
240      * special encodings in the case of z/os and os/400.
241      * @since Ant 1.5
242      */
243     public static String toString(ByteArrayOutputStream bos) {
244         if (Os.isFamily("z/os")) {
245             try {
246                 return bos.toString("Cp1047");
247             } catch (java.io.UnsupportedEncodingException e) {
248                 // noop default encoding used
249             }
250         } else if (Os.isFamily("os/400")) {
251             try {
252                 return bos.toString("Cp500");
253             } catch (java.io.UnsupportedEncodingException e) {
254                 // noop default encoding used
255             }
256         }
257         return bos.toString();
258     }

406     /**
407      * Creates a process that runs a command.
408      *
409      * @param project the Project, only used for logging purposes, may be null.
410      * @param command the command to run.
411      * @param env the environment for the command.
412      * @param dir the working directory for the command.
413      * @param useVM use the built-in exec command for JDK 1.3 if available.
414      * @return the process started.
415      * @throws IOException forwarded from the particular launcher used.
416      * @since Ant 1.5
417      */
418     public static Process launch(Project project, String[] command,
419                                  String[] env, File dir, boolean useVM)
420         throws IOException {
421         if (dir != null && !dir.exists()) {
422             throw new BuildException(dir + " doesn't exist.");
423         }
424 
425         CommandLauncher vmLauncher = CommandLauncher.getVMLauncher(project);
426         CommandLauncher launcher = (useVM && vmLauncher != null)
427             ? vmLauncher : CommandLauncher.getShellLauncher(project);
428         return launcher.exec(project, command, env, dir);
429     }

431     /**
432      * Runs a process defined by the command line and returns its exit status.
433      *
434      * @return the exit status of the subprocess or <code>INVALID</code>.
435      * @exception java.io.IOException The exception is thrown, if launching
436      *            of the subprocess failed.
437      */
438     public int execute() throws IOException {
439         if (workingDirectory != null && !workingDirectory.exists()) {
440             throw new BuildException(workingDirectory + " doesn't exist.");
441         }
442         final Process process = launch(project, getCommandline(),
443                                        getEnvironment(), workingDirectory,
444                                        useVMLauncher);
445         try {
446             streamHandler.setProcessInputStream(process.getOutputStream());
447             streamHandler.setProcessOutputStream(process.getInputStream());
448             streamHandler.setProcessErrorStream(process.getErrorStream());
449         } catch (IOException e) {
450             process.destroy();
451             throw e;
452         }
453         streamHandler.start();
454 
455         try {
456             // add the process to the list of those to destroy if the VM exits
457             //
458             processDestroyer.add(process);
459 
460             if (watchdog != null) {
461                 watchdog.start(process);
462             }
463             waitFor(process);
464 
465             if (watchdog != null) {
466                 watchdog.stop();
467             }
468             streamHandler.stop();
469             closeStreams(process);
470 
471             if (watchdog != null) {
472                 watchdog.checkException();
473             }
474             return getExitValue();
475         } catch (ThreadDeath t) {
476             // #31928: forcibly kill it before continuing.
477             process.destroy();
478             throw t;
479         } finally {
480             // remove the process to the list of those to destroy if
481             // the VM exits
482             //
483             processDestroyer.remove(process);
484         }
485     }

487     /**
488      * Starts a process defined by the command line.
489      * Ant will not wait for this process, nor log its output.
490      *
491      * @throws java.io.IOException The exception is thrown, if launching
492      *            of the subprocess failed.
493      * @since Ant 1.6
494      */
495     public void spawn() throws IOException {
496         if (workingDirectory != null && !workingDirectory.exists()) {
497             throw new BuildException(workingDirectory + " doesn't exist.");
498         }
499         final Process process = launch(project, getCommandline(),
500                                        getEnvironment(), workingDirectory,
501                                        useVMLauncher);
502         if (Os.isFamily("windows")) {
503             try {
504                 Thread.sleep(ONE_SECOND);
505             } catch (InterruptedException e) {
506                 project.log("interruption in the sleep after having spawned a"
507                             + " process", Project.MSG_VERBOSE);
508             }
509         }
510         OutputStream dummyOut = new OutputStream() {
511             @Override
512             public void write(int b) throws IOException {
513                 // Method intended to swallow whatever comes at it
514             }
515         };
516 
517         ExecuteStreamHandler handler = new PumpStreamHandler(dummyOut);
518         handler.setProcessErrorStream(process.getErrorStream());
519         handler.setProcessOutputStream(process.getInputStream());
520         handler.start();
521         process.getOutputStream().close();
522 
523         project.log("spawned process " + process.toString(),
524                     Project.MSG_VERBOSE);
525     }

560     /**
561      * Checks whether <code>exitValue</code> signals a failure on the current
562      * system (OS specific).
563      *
564      * <p><b>Note</b> that this method relies on the conventions of
565      * the OS, it will return false results if the application you are
566      * running doesn't follow these conventions. One notable
567      * exception is the Java VM provided by HP for OpenVMS - it will
568      * return 0 if successful (like on any other platform), but this
569      * signals a failure on OpenVMS. So if you execute a new Java VM
570      * on OpenVMS, you cannot trust this method.</p>
571      *
572      * @param exitValue the exit value (return code) to be checked.
573      * @return <code>true</code> if <code>exitValue</code> signals a failure.
574      */
575     public static boolean isFailure(int exitValue) {
576         // on openvms even exit value signals failure;
577         // for other platforms nonzero exit value signals failure
578         return Os.isFamily("openvms")
579             ? (exitValue % 2 == 0) : (exitValue != 0);
580     }

603     /**
604      * Patch the current environment with the new values from the user.
605      *
606      * @return the patched environment.
607      */
608     private String[] patchEnvironment() {
609         // On OpenVMS Runtime#exec() doesn't support the environment array,
610         // so we only return the new values which then will be set in
611         // the generated DCL script, inheriting the parent process environment
612         if (Os.isFamily("openvms")) {
613             return env;
614         }
615         Map<String, String> osEnv =
616             new LinkedHashMap<String, String>(getEnvironmentVariables());
617         for (int i = 0; i < env.length; i++) {
618             String keyValue = env[i];
619             String key = keyValue.substring(0, keyValue.indexOf('='));
620             // Find the key in the current enviroment copy
621             // and remove it.
622 
623             // Try without changing case first
624             if (osEnv.remove(key) == null && environmentCaseInSensitive) {
625                 // not found, maybe perform a case insensitive search
626 
627                 for (String osEnvItem : osEnv.keySet()) {
628                     // Nb: using default locale as key is a env name
629                     if (osEnvItem.toLowerCase().equals(key.toLowerCase())) {
630                         // Use the original casiness of the key
631                         key = osEnvItem;
632                         break;
633                     }
634                 }
635             }
636 
637             // Add the key to the enviromnent copy
638             osEnv.put(key, keyValue.substring(key.length() + 1));
639         }
640 
641         ArrayList<String> l = new ArrayList<String>();
642         for (Entry<String, String> entry : osEnv.entrySet()) {
643             l.add(entry.getKey() + "=" + entry.getValue());
644         }
645         return l.toArray(new String[osEnv.size()]);
646     }

648     /**
649      * A utility method that runs an external command. Writes the output and
650      * error streams of the command to the project log.
651      *
652      * @param task The task that the command is part of. Used for logging
653      * @param cmdline The command to execute.
654      * @throws BuildException if the command does not exit successfully.
655      */
656     public static void runCommand(Task task, String[] cmdline)
657         throws BuildException {
658         try {
659             task.log(Commandline.describeCommand(cmdline),
660                      Project.MSG_VERBOSE);
661             Execute exe = new Execute(
662                 new LogStreamHandler(task, Project.MSG_INFO, Project.MSG_ERR));
663             exe.setAntRun(task.getProject());
664             exe.setCommandline(cmdline);
665             int retval = exe.execute();
666             if (isFailure(retval)) {
667                 throw new BuildException(cmdline[0]
668                     + " failed with return code " + retval, task.getLocation());
669             }
670         } catch (java.io.IOException exc) {
671             throw new BuildException("Could not launch " + cmdline[0] + ": "
672                 + exc, task.getLocation());
673         }
674     }

687     /**
688      * This method is VMS specific and used by getEnvironmentVariables().
689      *
690      * Parses VMS logicals from <code>in</code> and returns them as a Map.
691      * <code>in</code> is expected to be the
692      * output of "SHOW LOGICAL". The method takes care of parsing the output
693      * correctly as well as making sure that a logical defined in multiple
694      * tables only gets added from the highest order table. Logicals with
695      * multiple equivalence names are mapped to a variable with multiple
696      * values separated by a comma (,).
697      */
698     private static Map<String, String> getVMSLogicals(BufferedReader in)
699         throws IOException {
700         HashMap<String, String> logicals = new HashMap<String, String>();
701         String logName = null, logValue = null, newLogName;
702         String line = null;
703         // CheckStyle:MagicNumber OFF
704         while ((line = in.readLine()) != null) {
705             // parse the VMS logicals into required format ("VAR=VAL[,VAL2]")
706             if (line.startsWith("\t=")) {
707                 // further equivalence name of previous logical
708                 if (logName != null) {
709                     logValue += "," + line.substring(4, line.length() - 1);
710                 }
711             } else if (line.startsWith("  \"")) {
712                 // new logical?
713                 if (logName != null) {
714                     logicals.put(logName, logValue);
715                 }
716                 int eqIndex = line.indexOf('=');
717                 newLogName = line.substring(3, eqIndex - 2);
718                 if (logicals.containsKey(newLogName)) {
719                     // already got this logical from a higher order table
720                     logName = null;
721                 } else {
722                     logName = newLogName;
723                     logValue = line.substring(eqIndex + 3, line.length() - 1);
724                 }
725             }
726         }
727         // CheckStyle:MagicNumber ON
728         // Since we "look ahead" before adding, there's one last env var.
729         if (logName != null) {
730             logicals.put(logName, logValue);
731         }
732         return logicals;
733     }
