348         /**
349          * scans the remote directory,
350          * storing internally the included files, directories, ...
351          */
352         public void scan() {
353             if (includes == null) {
354                 // No includes supplied, so set it to 'matches all'
355                 includes = new String[1];
356                 includes[0] = "**";
357             }
358             if (excludes == null) {
359                 excludes = new String[0];
360             }
361 
362             filesIncluded = new VectorSet();
363             filesNotIncluded = new Vector();
364             filesExcluded = new VectorSet();
365             dirsIncluded = new VectorSet();
366             dirsNotIncluded = new Vector();
367             dirsExcluded = new VectorSet();
368 
369             try {
370                 String cwd = ftp.printWorkingDirectory();
371                 // always start from the current ftp working dir
372                 forceRemoteSensitivityCheck();
373 
374                 checkIncludePatterns();
375                 clearCaches();
376                 ftp.changeWorkingDirectory(cwd);
377             } catch (IOException e) {
378                 throw new BuildException("Unable to scan FTP server: ", e);
379             }
380         }

383         /**
384          * this routine is actually checking all the include patterns in
385          * order to avoid scanning everything under base dir
386          * @since ant1.6
387          */
388         private void checkIncludePatterns() {
389 
390             Hashtable newroots = new Hashtable();
391             // put in the newroots vector the include patterns without
392             // wildcard tokens
393             for (int icounter = 0; icounter < includes.length; icounter++) {
394                 String newpattern =
395                     SelectorUtils.rtrimWildcardTokens(includes[icounter]);
396                 newroots.put(newpattern, includes[icounter]);
397             }
398             if (remotedir == null) {
399                 try {
400                     remotedir = ftp.printWorkingDirectory();
401                 } catch (IOException e) {
402                     throw new BuildException("could not read current ftp directory",
403                                              getLocation());
404                 }
405             }
406             AntFTPFile baseFTPFile = new AntFTPRootFile(ftp, remotedir);
407             rootPath = baseFTPFile.getAbsolutePath();
408             // construct it
409             if (newroots.containsKey("")) {
410                 // we are going to scan everything anyway
411                 scandir(rootPath, "", true);
412             } else {
413                 // only scan directories that can include matched files or
414                 // directories
415                 Enumeration enum2 = newroots.keys();
416 
417                 while (enum2.hasMoreElements()) {
418                     String currentelement = (String) enum2.nextElement();
419                     String originalpattern = (String) newroots.get(currentelement);
420                     AntFTPFile myfile = new AntFTPFile(baseFTPFile, currentelement);
421                     boolean isOK = true;
422                     boolean traversesSymlinks = false;
423                     String path = null;
424 
425                     if (myfile.exists()) {
426                         forceRemoteSensitivityCheck();
427                         if (remoteSensitivityChecked
428                             && remoteSystemCaseSensitive && isFollowSymlinks()) {
429                             // cool case,
430                             //we do not need to scan all the subdirs in the relative path
431                             path = myfile.getFastRelativePath();
432                         } else {
433                             // may be on a case insensitive file system.  We want
434                             // the results to show what's really on the disk, so
435                             // we need to double check.
436                             try {
437                                 path = myfile.getRelativePath();
438                                 traversesSymlinks = myfile.isTraverseSymlinks();
439                             }  catch (IOException be) {
440                                 throw new BuildException(be, getLocation());
441                             } catch (BuildException be) {
442                                 isOK = false;
443                             }
444                         }
445                     } else {
446                         isOK = false;
447                     }
448                     if (isOK) {
449                         currentelement = path.replace(remoteFileSep.charAt(0), File.separatorChar);
450                         if (!isFollowSymlinks()
451                             && traversesSymlinks) {
452                             continue;
453                         }
454 
455                         if (myfile.isDirectory()) {
456                             if (isIncluded(currentelement)
457                                 && currentelement.length() > 0) {
458                                 accountForIncludedDir(currentelement, myfile, true);
459                             }  else {
460                                 if (currentelement.length() > 0) {
461                                     if (currentelement.charAt(currentelement
462                                                               .length() - 1)
463                                         != File.separatorChar) {
464                                         currentelement =
465                                             currentelement + File.separatorChar;
466                                     }
467                                 }
468                                 scandir(myfile.getAbsolutePath(), currentelement, true);
469                             }
470                         } else {
471                             if (isCaseSensitive
472                                 && originalpattern.equals(currentelement)) {
473                                 accountForIncludedFile(currentelement);
474                             } else if (!isCaseSensitive
475                                        && originalpattern
476                                        .equalsIgnoreCase(currentelement)) {
477                                 accountForIncludedFile(currentelement);
478                             }
479                         }
480                     }
481                 }
482             }
483         }
484         /**
485          * scans a particular directory. populates the scannedDirs cache.
486          *
487          * @param dir directory to scan
488          * @param vpath  relative path to the base directory of the remote fileset
489          * always ended with a File.separator
490          * @param fast seems to be always true in practice
491          */
492         protected void scandir(String dir, String vpath, boolean fast) {
493             // avoid double scanning of directories, can only happen in fast mode
494             if (fast && hasBeenScanned(vpath)) {
495                 return;
496             }
497             try {
498                 if (!ftp.changeWorkingDirectory(dir)) {
499                     return;
500                 }
501                 String completePath = null;
502                 if (!vpath.equals("")) {
503                     completePath = rootPath + remoteFileSep
504                         + vpath.replace(File.separatorChar, remoteFileSep.charAt(0));
505                 } else {
506                     completePath = rootPath;
507                 }
508                 FTPFile[] newfiles = listFiles(completePath, false);
509 
510                 if (newfiles == null) {
511                     ftp.changeToParentDirectory();
512                     return;
513                 }
514                 for (int i = 0; i < newfiles.length; i++) {
515                     FTPFile file = newfiles[i];
516                     if (file != null
517                         && !file.getName().equals(".")
518                         && !file.getName().equals("..")) {
519                         String name = vpath + file.getName();
520                         scannedDirs.put(name, new FTPFileProxy(file));
521                         if (isFunctioningAsDirectory(ftp, dir, file)) {
522                             boolean slowScanAllowed = true;
523                             if (!isFollowSymlinks() && file.isSymbolicLink()) {
524                                 dirsExcluded.addElement(name);
525                                 slowScanAllowed = false;
526                             } else if (isIncluded(name)) {
527                                 accountForIncludedDir(name,
528                                                       new AntFTPFile(ftp, file, completePath) , fast);
529                             } else {
530                                 dirsNotIncluded.addElement(name);
531                                 if (fast && couldHoldIncluded(name)) {
532                                     scandir(file.getName(),
533                                             name + File.separator, fast);
534                                 }
535                             }
536                             if (!fast && slowScanAllowed) {
537                                 scandir(file.getName(),
538                                         name + File.separator, fast);
539                             }
540                         } else {
541                             if (!isFollowSymlinks() && file.isSymbolicLink()) {
542                                 filesExcluded.addElement(name);
543                             } else if (isFunctioningAsFile(ftp, dir, file)) {
544                                 accountForIncludedFile(name);
545                             }
546                         }
547                     }
548                 }
549                 ftp.changeToParentDirectory();
550             } catch (IOException e) {
551                 throw new BuildException("Error while communicating with FTP "
552                                          + "server: ", e);
553             }
554         }
555         /**
556          * process included file
557          * @param name  path of the file relative to the directory of the fileset
558          */
559         private void accountForIncludedFile(String name) {
560             if (!filesIncluded.contains(name)
561                 && !filesExcluded.contains(name)) {
562 
563                 if (isIncluded(name)) {
564                     if (!isExcluded(name)
565                         && isSelected(name, (File) scannedDirs.get(name))) {
566                         filesIncluded.addElement(name);
567                     } else {
568                         filesExcluded.addElement(name);
569                     }
570                 } else {
571                     filesNotIncluded.addElement(name);
572                 }
573             }
574         }

577          *
578          * @param name path of the directory relative to the directory of
579          * the fileset
580          * @param file directory as file
581          * @param fast
582          */
583         private void accountForIncludedDir(String name, AntFTPFile file, boolean fast) {
584             if (!dirsIncluded.contains(name)
585                 && !dirsExcluded.contains(name)) {
586 
587                 if (!isExcluded(name)) {
588                     if (fast) {
589                         if (file.isSymbolicLink()) {
590                             try {
591                                 file.getClient().changeWorkingDirectory(file.curpwd);
592                             } catch (IOException ioe) {
593                                 throw new BuildException("could not change directory to curpwd");
594                             }
595                             scandir(file.getLink(),
596                                     name + File.separator, fast);
597                         } else {
598                             try {
599                                 file.getClient().changeWorkingDirectory(file.curpwd);
600                             } catch (IOException ioe) {
601                                 throw new BuildException("could not change directory to curpwd");
602                             }
603                             scandir(file.getName(),
604                                     name + File.separator, fast);
605                         }
606                     }
607                     dirsIncluded.addElement(name);
608                 } else {
609                     dirsExcluded.addElement(name);
610                     if (fast && couldHoldIncluded(name)) {
611                         try {
612                             file.getClient().changeWorkingDirectory(file.curpwd);
613                         } catch (IOException ioe) {
614                             throw new BuildException("could not change directory to curpwd");
615                         }
616                         scandir(file.getName(),
617                                 name + File.separator, fast);
618                     }
619                 }
620             }

663             String currentPath = directory;
664             if (changedir) {
665                 try {
666                     boolean result = ftp.changeWorkingDirectory(directory);

669                     }
670                     currentPath = ftp.printWorkingDirectory();

673                 }
674             }
675             if (fileListMap.containsKey(currentPath)) {
676                 getProject().log("filelist map used in listing files", Project.MSG_DEBUG);
677                 return ((FTPFile[]) fileListMap.get(currentPath));
678             }
679             FTPFile[] result = null;
680             try {
681                 result = ftp.listFiles();
682             } catch (IOException ioe) {
683                 throw new BuildException(ioe, getLocation());
684             }
685             fileListMap.put(currentPath, result);
686             if (!remoteSensitivityChecked) {
687                 checkRemoteSensitivity(result, directory);
688             }
689             return result;
690         }

710         private void checkRemoteSensitivity(FTPFile[] array, String directory) {
711             if (array == null) {
712                 return;
713             }
714             boolean candidateFound = false;
715             String target = null;
716             for (int icounter = 0; icounter < array.length; icounter++) {
717                 if (array[icounter] != null && array[icounter].isDirectory()) {
718                     if (!array[icounter].getName().equals(".")
719                         && !array[icounter].getName().equals("..")) {
720                         candidateFound = true;
721                         target = fiddleName(array[icounter].getName());
722                         getProject().log("will try to cd to "
723                                          + target + " where a directory called " + array[icounter].getName()
724                                          + " exists", Project.MSG_DEBUG);
725                         for (int pcounter = 0; pcounter < array.length; pcounter++) {
726                             if (array[pcounter] != null
727                                 && pcounter != icounter
728                                 && target.equals(array[pcounter].getName())) {
729                                 candidateFound = false;
730                                 break;
731                             }
732                         }
733                         if (candidateFound) {
734                             break;
735                         }
736                     }
737                 }
738             }
739             if (candidateFound) {
740                 try {
741                     getProject().log("testing case sensitivity, attempting to cd to "
742                                      + target, Project.MSG_DEBUG);
743                     remoteSystemCaseSensitive  = !ftp.changeWorkingDirectory(target);
744                 } catch (IOException ioe) {
745                     remoteSystemCaseSensitive = true;
746                 } finally {
747                     try {
748                         ftp.changeWorkingDirectory(directory);
749                     } catch (IOException ioe) {
750                         throw new BuildException(ioe, getLocation());
751                     }
752                 }

814                 Vector pathElements = SelectorUtils.tokenizePath(path);
815                 try {
816                     boolean result = this.client.changeWorkingDirectory(parent.getAbsolutePath());

828                     String currentPathElement = (String) pathElements.elementAt(fcount);
829                     try {
830                         boolean result = this.client.changeWorkingDirectory(currentPathElement);
831                         if (!result && !isCaseSensitive()
832                             && (remoteSystemCaseSensitive || !remoteSensitivityChecked)) {

834                                                                                 currentPathElement);

839                             return;

844                         throw new BuildException("could not change working dir to "
845                                                  + (String) pathElements.elementAt(fcount)
846                                                  + " from " + this.curpwd);
847                     }
848 
849                 }
850                 String lastpathelement = (String) pathElements.elementAt(size - 1);
851                 FTPFile [] theFiles = listFiles(this.curpwd);

922              * If a symbolic link is encountered, it is followed, but the name of the link

924              * (ie does not behave like File.getCanonicalPath())
925              * @return                relative path, separated by remoteFileSep

935                                                        parent.getRelativePath());

943             /**
944              * get thge relative path of this file
945              * @param currentPath          base path
946              * @param currentRelativePath  relative path of the base path with regards to remote dir
947              * @return relative path
948              */
949             private String getRelativePath(String currentPath, String currentRelativePath) {
950                 Vector pathElements = SelectorUtils.tokenizePath(getAbsolutePath(), remoteFileSep);
951                 Vector pathElements2 = SelectorUtils.tokenizePath(currentPath, remoteFileSep);
952                 String relPath = currentRelativePath;
953                 final int size = pathElements.size();
954                 for (int pcount = pathElements2.size(); pcount < size; pcount++) {
955                     String currentElement = (String) pathElements.elementAt(pcount);
956                     FTPFile[] theFiles = listFiles(currentPath);
957                     FTPFile theFile = null;
958                     if (theFiles != null) {
959                         theFile = getFile(theFiles, currentElement);
960                     }
961                     if (!relPath.equals("")) {
962                         relPath = relPath + remoteFileSep;
963                     }
964                     if (theFile == null) {
965                         // hit a hidden file assume not a symlink
966                         relPath = relPath + currentElement;
967                         currentPath = currentPath + remoteFileSep + currentElement;
968                         log("Hidden file " + relPath
969                             + " assumed to not be a symlink.",
970                             Project.MSG_VERBOSE);
971                     } else {
972                         traversesSymlinks = traversesSymlinks || theFile.isSymbolicLink();
973                         relPath = relPath + theFile.getName();
974                         currentPath = currentPath + remoteFileSep + theFile.getName();
975                     }
976                 }
977                 return relPath;
978             }

1118     /**
1119      * check FTPFiles to check whether they function as directories too
1120      * the FTPFile API seem to make directory and symbolic links incompatible
1121      * we want to find out if we can cd to a symbolic link
1122      * @param dir  the parent directory of the file to test
1123      * @param file the file to test
1124      * @return true if it is possible to cd to this directory
1125      * @since ant 1.6
1126      */
1127     private boolean isFunctioningAsDirectory(FTPClient ftp, String dir, FTPFile file) {
1128         boolean result = false;
1129         String currentWorkingDir = null;
1130         if (file.isDirectory()) {
1131             return true;
1132         } else if (file.isFile()) {
1133             return false;
1134         }
1135         try {
1136             currentWorkingDir = ftp.printWorkingDirectory();
1137         } catch (IOException ioe) {
1138             getProject().log("could not find current working directory " + dir
1139                              + " while checking a symlink",
1140                              Project.MSG_DEBUG);
1141         }
1142         if (currentWorkingDir != null) {
1143             try {
1144                 result = ftp.changeWorkingDirectory(file.getLink());
1145             } catch (IOException ioe) {
1146                 getProject().log("could not cd to " + file.getLink() + " while checking a symlink",
1147                                  Project.MSG_DEBUG);
1148             }
1149             if (result) {
1150                 boolean comeback = false;
1151                 try {
1152                     comeback = ftp.changeWorkingDirectory(currentWorkingDir);
1153                 } catch (IOException ioe) {
1154                     getProject().log("could not cd back to " + dir + " while checking a symlink",
1155                                      Project.MSG_ERR);
1156                 } finally {
1157                     if (!comeback) {
1158                         throw new BuildException("could not cd back to " + dir
1159                                                  + " while checking a symlink");
1160                     }
1161                 }
1162             }
1163         }
1164         return result;
1165     }

1718     /**
1719      * For each file in the fileset, do the appropriate action: send, get,
1720      * delete, or list.
1721      *
1722      * @param ftp the FTPClient instance used to perform FTP actions
1723      * @param fs the fileset on which the actions are performed.
1724      *
1725      * @return the number of files to be transferred.
1726      *
1727      * @throws IOException if there is a problem reading a file
1728      * @throws BuildException if there is a problem in the configuration.
1729      */
1730     protected int transferFiles(final FTPClient ftp, FileSet fs)
1731         throws IOException, BuildException {
1732         DirectoryScanner ds;
1733         if (action == SEND_FILES) {
1734             ds = fs.getDirectoryScanner(getProject());
1735         } else {
1736             ds = new FTPDirectoryScanner(ftp);
1737             fs.setupDirectoryScanner(ds, getProject());
1738             ds.setFollowSymlinks(fs.isFollowSymlinks());
1739             ds.scan();
1740         }
1741 
1742         String[] dsfiles = null;
1743         if (action == RM_DIR) {
1744             dsfiles = ds.getIncludedDirectories();
1745         } else {
1746             dsfiles = ds.getIncludedFiles();
1747         }
1748         String dir = null;
1749 
1750         if ((ds.getBasedir() == null)
1751             && ((action == SEND_FILES) || (action == GET_FILES))) {
1752             throw new BuildException("the dir attribute must be set for send "
1753                                      + "and get actions");
1754         } else {
1755             if ((action == SEND_FILES) || (action == GET_FILES)) {
1756                 dir = ds.getBasedir().getAbsolutePath();
1757             }
1758         }
1759 
1760         // If we are doing a listing, we need the output stream created now.
1761         BufferedWriter bw = null;
1762 
1763         try {
1764             if (action == LIST_FILES) {
1765                 File pd = listing.getParentFile();
1766 
1767                 if (!pd.exists()) {
1768                     pd.mkdirs();
1769                 }
1770                 bw = new BufferedWriter(new FileWriter(listing));
1771             }
1772             RetryHandler h = new RetryHandler(this.retriesAllowed, this);
1773             if (action == RM_DIR) {
1774                 // to remove directories, start by the end of the list
1775                 // the trunk does not let itself be removed before the leaves
1776                 for (int i = dsfiles.length - 1; i >= 0; i--) {
1777                     final String dsfile = dsfiles[i];
1778                     executeRetryable(h, new Retryable() {
1779                             public void execute() throws IOException {
1780                                 rmDir(ftp, dsfile);
1781                             }
1782                         }, dsfile);
1783                 }
1784             } else {
1785                 final BufferedWriter fbw = bw;
1786                 final String fdir = dir;
1787                 if (this.newerOnly) {
1788                     this.granularityMillis =
1789                         this.timestampGranularity.getMilliseconds(action);
1790                 }
1791                 for (int i = 0; i < dsfiles.length; i++) {
1792                     final String dsfile = dsfiles[i];
1793                     executeRetryable(h, new Retryable() {
1794                             public void execute() throws IOException {
1795                                 switch (action) {
1796                                 case SEND_FILES:
1797                                     sendFile(ftp, fdir, dsfile);
1798                                     break;
1799                                 case GET_FILES:
1800                                     getFile(ftp, fdir, dsfile);
1801                                     break;
1802                                 case DEL_FILES:
1803                                     delFile(ftp, dsfile);
1804                                     break;
1805                                 case LIST_FILES:
1806                                     listFile(ftp, fbw, dsfile);
1807                                     break;
1808                                 case CHMOD:
1809                                     doSiteCommand(ftp, "chmod " + chmod
1810                                                   + " " + resolveFile(dsfile));
1811                                     transferred++;
1812                                     break;
1813                                 default:
1814                                     throw new BuildException("unknown ftp action " + action);
1815                                 }
1816                             }
1817                         }, dsfile);
1818                 }
1819             }
1820         } finally {
1821             FileUtils.close(bw);
1822         }
1823 
1824         return dsfiles.length;
1825     }

1883      * Creates all parent directories specified in a complete relative
1884      * pathname. Attempts to create existing directories will not cause
1885      * errors.
1886      *
1887      * @param ftp the FTP client instance to use to execute FTP actions on
1888      *        the remote server.
1889      * @param filename the name of the file whose parents should be created.
1890      * @throws IOException under non documented circumstances
1891      * @throws BuildException if it is impossible to cd to a remote directory
1892      *
1893      */
1894     protected void createParents(FTPClient ftp, String filename)
1895         throws IOException, BuildException {
1896 
1897         File dir = new File(filename);
1898         if (dirCache.contains(dir)) {
1899             return;
1900         }
1901 
1902         Vector parents = new Vector();
1903         String dirname;
1904 
1905         while ((dirname = dir.getParent()) != null) {
1906             File checkDir = new File(dirname);
1907             if (dirCache.contains(checkDir)) {
1908                 break;
1909             }
1910             dir = checkDir;
1911             parents.addElement(dir);
1912         }
1913 
1914         // find first non cached dir
1915         int i = parents.size() - 1;
1916 
1917         if (i >= 0) {
1918             String cwd = ftp.printWorkingDirectory();
1919             String parent = dir.getParent();
1920             if (parent != null) {
1921                 if (!ftp.changeWorkingDirectory(resolveFile(parent))) {
1922                     throw new BuildException("could not change to "
1923                                              + "directory: " + ftp.getReplyString());
1924                 }
1925             }
1926 
1927             while (i >= 0) {
1928                 dir = (File) parents.elementAt(i--);
1929                 // check if dir exists by trying to change into it.
1930                 if (!ftp.changeWorkingDirectory(dir.getName())) {
1931                     // could not change to it - try to create it
1932                     log("creating remote directory "
1933                         + resolveFile(dir.getPath()), Project.MSG_VERBOSE);
1934                     if (!ftp.makeDirectory(dir.getName())) {
1935                         handleMkDirFailure(ftp);
1936                     }
1937                     if (!ftp.changeWorkingDirectory(dir.getName())) {
1938                         throw new BuildException("could not change to "
1939                                                  + "directory: " + ftp.getReplyString());
1940                     }
1941                 }
1942                 dirCache.add(dir);
1943             }
1944             ftp.changeWorkingDirectory(cwd);
1945         }
1946     }

2357         throws IOException, BuildException {
2358         String workingDirectory = ftp.printWorkingDirectory();

2372             subdir = st.nextToken();

2379                     int rc = ftp.getReplyCode();
