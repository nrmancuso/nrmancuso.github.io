1   package org.infinispan.objectfilter.impl.predicateindex.be;
2   
3   import org.infinispan.objectfilter.impl.predicateindex.FilterEvalContext;
4   
5   /**
6    * @author anistor@redhat.com
7    * @since 7.0
8    */
9   public final class OrNode extends BENode {
10  
11     public OrNode(BENode parent) {
12        super(parent);
13     }
14  
15     @Override
16     public void handleChildValue(BENode child, boolean childValue, FilterEvalContext evalContext) {
17        if (isEvaluationComplete(evalContext)) {
18           throw new IllegalStateException("This should never be called again because the state of this node has been decided already.");
19        }
20  
21        if (childValue) {
22           // value of this node is decided: TRUE
23           if (parent != null) {
24              // propagate to the parent, if we have a parent
25              parent.handleChildValue(this, true, evalContext);
26           } else {
27              // mark this node as satisfied
28              setState(BETree.EXPR_TRUE, evalContext);
29           }
30        } else {
31           if (--evalContext.treeCounters[startIndex] == 0) {
32              // value of this node is decided: FALSE
33              if (parent != null) {
34                 // propagate to the parent, if we have a parent
35                 parent.handleChildValue(this, false, evalContext);
36              } else {
37                 // mark this node as 'unsatisfied'
38                 setState(BETree.EXPR_FALSE, evalContext);
39              }
40           } else {
41              // value of this node cannot be decided yet, so we cannot propagate to the parent anything yet but let's at least mark down the child as 'unsatisfied'
42              child.setState(BETree.EXPR_FALSE, evalContext);
43           }
44        }
45     }
46  }
