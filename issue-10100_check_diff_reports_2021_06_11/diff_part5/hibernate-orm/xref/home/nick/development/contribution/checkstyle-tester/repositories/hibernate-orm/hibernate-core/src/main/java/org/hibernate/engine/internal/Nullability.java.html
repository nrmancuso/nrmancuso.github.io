24  package org.hibernate.engine.internal;
25  
26  import java.util.Iterator;
27  
28  import org.hibernate.EntityMode;
29  import org.hibernate.HibernateException;
30  import org.hibernate.PropertyValueException;
31  import org.hibernate.bytecode.instrumentation.spi.LazyPropertyInitializer;
32  import org.hibernate.engine.spi.CascadingAction;
33  import org.hibernate.engine.spi.SessionImplementor;
34  import org.hibernate.persister.entity.EntityPersister;
35  import org.hibernate.type.CollectionType;
36  import org.hibernate.type.CompositeType;
37  import org.hibernate.type.Type;
38  
39  /**
40   * Implements the algorithm for validating property values for illegal null values
41   * 
42   * @author Gavin King
43   */
44  public final class Nullability {
45  	
46  	private final SessionImplementor session;
47  	private final boolean checkNullability;
48  
49  	public Nullability(SessionImplementor session) {
50  		this.session = session;
51  		this.checkNullability = session.getFactory().getSettings().isCheckNullability();
52  	}
53  	/**
54  	 * Check nullability of the class persister properties
55  	 *
56  	 * @param values entity properties
57  	 * @param persister class persister
58  	 * @param isUpdate whether it is intended to be updated or saved
59  	 * @throws org.hibernate.PropertyValueException Break the nullability of one property
60  	 * @throws HibernateException error while getting Component values
61  	 */
62  	public void checkNullability(
63  			final Object[] values,
64  			final EntityPersister persister,
65  			final boolean isUpdate) 
66  	throws PropertyValueException, HibernateException {
67  		/*
68  		 * Typically when Bean Validation is on, we don't want to validate null values
69  		 * at the Hibernate Core level. Hence the checkNullability setting.
70  		 */
71  		if ( checkNullability ) {
72  			/*
73  			  * Algorithm
74  			  * Check for any level one nullability breaks
75  			  * Look at non null components to
76  			  *   recursively check next level of nullability breaks
77  			  * Look at Collections contraining component to
78  			  *   recursively check next level of nullability breaks
79  			  *
80  			  *
81  			  * In the previous implementation, not-null stuffs where checked
82  			  * filtering by level one only updateable
83  			  * or insertable columns. So setting a sub component as update="false"
84  			  * has no effect on not-null check if the main component had good checkeability
85  			  * In this implementation, we keep this feature.
86  			  * However, I never see any documentation mentioning that, but it's for
87  			  * sure a limitation.
88  			  */
89  
90  			final boolean[] nullability = persister.getPropertyNullability();
91  			final boolean[] checkability = isUpdate ?
92  				persister.getPropertyUpdateability() :
93  				persister.getPropertyInsertability();
94  			final Type[] propertyTypes = persister.getPropertyTypes();
95  
96  			for ( int i = 0; i < values.length; i++ ) {
97  
98  				if ( checkability[i] && values[i]!= LazyPropertyInitializer.UNFETCHED_PROPERTY ) {
99  					final Object value = values[i];
100 					if ( !nullability[i] && value == null ) {
101 
102 						//check basic level one nullablilty
103 						throw new PropertyValueException(
104 								"not-null property references a null or transient value",
105 								persister.getEntityName(),
106 								persister.getPropertyNames()[i]
107 							);
108 
109 					}
110 					else if ( value != null ) {
111 
112 						//values is not null and is checkable, we'll look deeper
113 						String breakProperties = checkSubElementsNullability( propertyTypes[i], value );
114 						if ( breakProperties != null ) {
115 							throw new PropertyValueException(
116 								"not-null property references a null or transient value",
117 								persister.getEntityName(),
118 								buildPropertyPath( persister.getPropertyNames()[i], breakProperties )
119 							);
120 						}
121 
122 					}
123 				}
124 
125 			}
126 		}
127 	}
128 
129 	/**
130 	 * check sub elements-nullability. Returns property path that break
131 	 * nullability or null if none
132 	 *
133 	 * @param propertyType type to check
134 	 * @param value value to check
135 	 *
136 	 * @return property path
137 	 * @throws HibernateException error while getting subcomponent values
138 	 */
139 	private String checkSubElementsNullability(final Type propertyType, final Object value) 
140 	throws HibernateException {
141 		//for non null args, check for components and elements containing components
142 		if ( propertyType.isComponentType() ) {
143 			return checkComponentNullability( value, (CompositeType) propertyType );
144 		}
145 		else if ( propertyType.isCollectionType() ) {
146 
147 			//persistent collections may have components
148 			CollectionType collectionType = (CollectionType) propertyType;
149 			Type collectionElementType = collectionType.getElementType( session.getFactory() );
150 			if ( collectionElementType.isComponentType() ) {
151 				//check for all components values in the collection
152 
153 				CompositeType componentType = (CompositeType) collectionElementType;
154 				Iterator iter = CascadingAction.getLoadedElementsIterator( session, collectionType, value );
155 				while ( iter.hasNext() ) {
156 					Object compValue = iter.next();
157 					if (compValue != null) {
158 						return checkComponentNullability(compValue, componentType);
159 					}
160 				}
161 			}
162 		}
163 		return null;
164 	}
165 
166 	/**
167 	 * check component nullability. Returns property path that break
168 	 * nullability or null if none
169 	 *
170 	 * @param value component properties
171 	 * @param compType component not-nullable type
172 	 *
173 	 * @return property path
174 	 * @throws HibernateException error while getting subcomponent values
175 	 */
176 	private String checkComponentNullability(final Object value, final CompositeType compType)
177 	throws HibernateException {
178 		/* will check current level if some of them are not null
179 		 * or sublevels if they exist
180 		 */
181 		boolean[] nullability = compType.getPropertyNullability();
182 		if ( nullability!=null ) {
183 			//do the test
184 			final Object[] values = compType.getPropertyValues( value, EntityMode.POJO );
185 			final Type[] propertyTypes = compType.getSubtypes();
186 			for ( int i=0; i<values.length; i++ ) {
187 				final Object subvalue = values[i];
188 				if ( !nullability[i] && subvalue==null ) {
189 					return compType.getPropertyNames()[i];
190 				}
191 				else if ( subvalue != null ) {
192 					String breakProperties = checkSubElementsNullability( propertyTypes[i], subvalue );
193 					if ( breakProperties != null ) {
194 						return buildPropertyPath( compType.getPropertyNames()[i], breakProperties );
195 					}
196 	 			}
197 	 		}
198 		}
199 		return null;
200 	}
201 
202 	/**
203 	 * Return a well formed property path.
204 	 * Basically, it will return parent.child
205 	 *
206 	 * @param parent parent in path
207 	 * @param child child in path
208 	 * @return parent-child path
209 	 */
210 	private static String buildPropertyPath(String parent, String child) {
211 		return new StringBuilder( parent.length() + child.length() + 1 )
212 			.append(parent).append('.').append(child).toString();
213 	}
