87      @Override
88      public void parse(ClassNameAndSuperclassInfo.Builder builder) throws InvalidClassFileFormatException {
89          try {
90              int magic = in.readInt();
91              if (magic != 0xcafebabe) {
92                  throw new InvalidClassFileFormatException("Classfile header isn't 0xCAFEBABE", expectedClassDescriptor,
93                          codeBaseEntry);
94              }
95              int major_version = in.readUnsignedShort();
96              int minor_version = in.readUnsignedShort();
97              int constant_pool_count = in.readUnsignedShort();
98  
99              constantPool = new Constant[constant_pool_count];
100             for (int i = 1; i < constantPool.length; i++) {
101                 constantPool[i] = readConstant();
102                 if (constantPool[i].tag == IClassConstants.CONSTANT_Double
103                         || constantPool[i].tag == IClassConstants.CONSTANT_Long) {
104                     // Double and Long constants take up two constant pool
105                     // entries
106                     ++i;
107                 }
108             }
109 
110             int access_flags = in.readUnsignedShort();
111 
112             int this_class = in.readUnsignedShort();
113             ClassDescriptor thisClassDescriptor = getClassDescriptor(this_class);
114 
115             int super_class = in.readUnsignedShort();
116             ClassDescriptor superClassDescriptor = getClassDescriptor(super_class);
117 
118             int interfaces_count = in.readUnsignedShort();
119             if (interfaces_count < 0) {
120                 throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
121             }
122             ClassDescriptor[] interfaceDescriptorList;
123             if (interfaces_count == 0) {
124                 interfaceDescriptorList = ClassDescriptor.EMPTY_ARRAY;
125             } else {
126                 interfaceDescriptorList = new ClassDescriptor[interfaces_count];
127                 for (int i = 0; i < interfaceDescriptorList.length; i++) {
128                     interfaceDescriptorList[i] = getClassDescriptor(in.readUnsignedShort());
129                 }
130             }
131             // Extract all references to other classes,
132             // both CONSTANT_Class entries and also referenced method
133             // signatures.
134             Collection<ClassDescriptor> referencedClassDescriptorList = extractReferencedClasses();
135 
136             builder.setClassDescriptor(thisClassDescriptor);
137             builder.setSuperclassDescriptor(superClassDescriptor);
138             builder.setInterfaceDescriptorList(interfaceDescriptorList);
139             builder.setCodeBaseEntry(codeBaseEntry);
140             builder.setAccessFlags(access_flags);
141             builder.setReferencedClassDescriptors(referencedClassDescriptorList);
142             builder.setClassfileVersion(major_version, minor_version);
143         } catch (IOException e) {
144             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry, e);
145         }
146     }

153     /**
154      * Extract references to other classes.
155      *
156      * @return array of ClassDescriptors of referenced classes
157      * @throws InvalidClassFileFormatException
158      */
159     private Collection<ClassDescriptor> extractReferencedClasses() throws InvalidClassFileFormatException {
160         Set<ClassDescriptor> referencedClassSet = new HashSet<>();
161         for (Constant constant : constantPool) {
162             if (constant == null) {
163                 continue;
164             }
165             if (constant.tag == IClassConstants.CONSTANT_Class) {
166                 @SlashedClassName
167                 String className = getUtf8String((Integer) constant.data[0]);
168                 if (className.indexOf('[') >= 0) {
169                     extractReferencedClassesFromSignature(referencedClassSet, className);
170                 } else if (ClassName.isValidClassName(className)) {
171                     referencedClassSet.add(DescriptorFactory.instance().getClassDescriptor(className));
172                 }
173             } else if (constant.tag == IClassConstants.CONSTANT_Methodref || constant.tag == IClassConstants.CONSTANT_Fieldref
174                     || constant.tag == IClassConstants.CONSTANT_InterfaceMethodref) {
175                 // Get the target class name
176                 String className = getClassName((Integer) constant.data[0]);
177                 extractReferencedClassesFromSignature(referencedClassSet, className);
178 
179                 // Parse signature to extract class names
180                 String signature = getSignatureFromNameAndType((Integer) constant.data[1]);
181                 extractReferencedClassesFromSignature(referencedClassSet, signature);
182             }
183         }
184         return referencedClassSet;
185     }
186 
187     public static void extractReferencedClassesFromSignature(Set<ClassDescriptor> referencedClassSet, String signature) {
188         while (signature.length() > 0) {
189             int start = signature.indexOf('L');
190             if (start < 0) {
191                 break;
192             }
193             int end = signature.indexOf(';', start);
194             if (end < 0) {
195                 break;
196             }
197             @SlashedClassName
198             String className = signature.substring(start + 1, end);
199             if (ClassName.isValidClassName(className)) {
200                 referencedClassSet.add(DescriptorFactory.instance().getClassDescriptor(className));
201             }
202             signature = signature.substring(end + 1);
203         }
204     }

206     // 8: UTF-8 string
207     // I: int
208     // F: float
209     // L: long
210     // D: double
211     // i: 2-byte constant pool index
212     // See https://docs.oracle.com/javase/specs/jvms/se9/html/jvms-4.html#jvms-4.4
213     private static final String[] CONSTANT_FORMAT_MAP = { null,
214         "8", // 1:CONSTANT_Utf8
215         null, // 2:
216         "I", // 3: CONSTANT_Integer
217         "F", // 4: CONSTANT_Float
218         "L", // 5: CONSTANT_Long
219         "D", // 6: CONSTANT_Double
220         "i", // 7: CONSTANT_Class
221         "i", // 8: CONSTANT_String
222         "ii", // 9: CONSTANT_Fieldref
223         "ii", // 10: CONSTANT_Methodref
224         "ii", // 11: CONSTANT_InterfaceMethodref
225         "ii", // 12: CONSTANT_NameAndType
226         null, // 13:
227         null, // 14:
228         "bi", // 15: CONSTANT_MethodHandle
229         "i", // 16: CONSTANT_MethodType
230         null, // 17:
231         "ii", // 18: CONSTANT_InvokeDynamic
232         "i", // 19: CONSTANT_Module
233         "i", // 20: CONSTANT_Package
234     };

236     /**
237      * Read a constant from the constant pool. Return null for
238      *
239      * @return a StaticConstant
240      * @throws InvalidClassFileFormatException
241      * @throws IOException
242      */
243     private Constant readConstant() throws InvalidClassFileFormatException, IOException {
244         int tag = in.readUnsignedByte();
245         if (tag < 0 || tag >= CONSTANT_FORMAT_MAP.length) {
246             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
247         }
248         String format = CONSTANT_FORMAT_MAP[tag];
249         if (format == null) {
250             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
251         }
252 
253         Object[] data = new Object[format.length()];
254         for (int i = 0; i < format.length(); i++) {
255             char spec = format.charAt(i);
256             switch (spec) {
257             case '8':
258                 data[i] = in.readUTF();
259                 break;
260             case 'I':
261                 data[i] = in.readInt();
262                 break;
263             case 'F':
264                 data[i] = Float.valueOf(in.readFloat());
265                 break;
266             case 'L':
267                 data[i] = in.readLong();
268                 break;
269             case 'D':
270                 data[i] = Double.valueOf(in.readDouble());
271                 break;
272             case 'i':
273                 data[i] = in.readUnsignedShort();
274                 break;
275             case 'b':
276                 data[i] = in.readUnsignedByte();
277                 break;
278             default:
279                 throw new IllegalStateException();
280             }
281         }
282 
283         return new Constant(tag, data);
284     }

286     /**
287      * Get a class name from a CONSTANT_Class. Note that this may be an array
288      * (e.g., "[Ljava/lang/String;").
289      *
290      * @param index
291      *            index of the constant
292      * @return the class name
293      * @throws InvalidClassFileFormatException
294      */
295     private @SlashedClassName
296     String getClassName(int index) throws InvalidClassFileFormatException {
297         if (index == 0) {
298             return null;
299         }
300 
301         checkConstantPoolIndex(index);
302         Constant constant = constantPool[index];
303         checkConstantTag(constant, IClassConstants.CONSTANT_Class);
304 
305         int refIndex = ((Integer) constant.data[0]).intValue();
306         String stringValue = getUtf8String(refIndex);
307 
308         return stringValue;
309     }

311     /**
312      * Get the ClassDescriptor of a class referenced in the constant pool.
313      *
314      * @param index
315      *            index of the referenced class in the constant pool
316      * @return the ClassDescriptor of the referenced class
317      * @throws InvalidClassFileFormatException
318      */
319     private ClassDescriptor getClassDescriptor(int index) throws InvalidClassFileFormatException {
320         @SlashedClassName
321         String className = getClassName(index);
322         return className != null ? DescriptorFactory.instance().getClassDescriptor(className) : null;
323     }

325     /**
326      * Get the UTF-8 string constant at given constant pool index.
327      *
328      * @param refIndex
329      *            the constant pool index
330      * @return the String at that index
331      * @throws InvalidClassFileFormatException
332      */
333     private String getUtf8String(int refIndex) throws InvalidClassFileFormatException {
334         checkConstantPoolIndex(refIndex);
335         Constant refConstant = constantPool[refIndex];
336         checkConstantTag(refConstant, IClassConstants.CONSTANT_Utf8);
337         return (String) refConstant.data[0];
338     }

340     /**
341      * Check that a constant pool index is valid.
342      *
343      * @param index
344      *            the index to check
345      * @throws InvalidClassFileFormatException
346      *             if the index is not valid
347      */
348     private void checkConstantPoolIndex(int index) throws InvalidClassFileFormatException {
349         if (index < 0 || index >= constantPool.length || constantPool[index] == null) {
350             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
351         }
352     }

354     /**
355      * Check that a constant has the expected tag.
356      *
357      * @param constant
358      *            the constant to check
359      * @param expectedTag
360      *            the expected constant tag
361      * @throws InvalidClassFileFormatException
362      *             if the constant's tag does not match the expected tag
363      */
364     private void checkConstantTag(Constant constant, int expectedTag) throws InvalidClassFileFormatException {
365         if (constant.tag != expectedTag) {
366             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
367         }
368     }

407     /**
408      * Read field_info or method_info. They have the same format.
409      *
410      * @param <E>
411      *            descriptor type to return
412      * @param thisClassDescriptor
413      *            class descriptor of class being parsed
414      * @param creator
415      *            callback to create the FieldDescriptor or MethodDescriptor
416      * @return the parsed descriptor
417      * @throws IOException
418      * @throws InvalidClassFileFormatException
419      *
420     private <E> E readFieldOrMethod(ClassDescriptor thisClassDescriptor, FieldOrMethodDescriptorCreator<E> creator)
421             throws IOException, InvalidClassFileFormatException {
422         int access_flags = in.readUnsignedShort();
423         int name_index = in.readUnsignedShort();
424         int descriptor_index = in.readUnsignedShort();
425         int attributes_count = in.readUnsignedShort();
426 
427         String name = getUtf8String(name_index);
428         String signature = getUtf8String(descriptor_index);
429         if (attributes_count < 0) {
430             throw new InvalidClassFileFormatException(expectedClassDescriptor, codeBaseEntry);
431         }
432         for (int i = 0; i < attributes_count; i++) {
433             readAttribute();
434         }
435 
436         return creator.create(thisClassDescriptor.getClassName(), name, signature, access_flags);
437     }*/
