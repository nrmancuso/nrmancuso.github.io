110     public InternalEngine(EngineConfig engineConfig) throws EngineException {
111         super(engineConfig);
112         this.versionMap = new LiveVersionMap();
113         store.incRef();
114         IndexWriter writer = null;
115         SearcherManager manager = null;
116         boolean success = false;
117         try {
118             try {
119                 upgrade3xSegments(store);
120             } catch (IOException ex) {
121                 throw new EngineCreationFailureException(shardId, "failed to upgrade 3x segments", ex);
122             }
123             this.onGoingRecoveries = new FlushingRecoveryCounter(this, store, logger);
124             this.lastDeleteVersionPruneTimeMSec = engineConfig.getThreadPool().estimatedTimeInMillis();
125             this.indexingService = engineConfig.getIndexingService();
126             this.warmer = engineConfig.getWarmer();
127             this.translog = engineConfig.getTranslog();
128             this.mergePolicyProvider = engineConfig.getMergePolicyProvider();
129             this.mergeScheduler = engineConfig.getMergeScheduler();
130             this.dirtyLocks = new Object[engineConfig.getIndexConcurrency() * 50]; // we multiply it to have enough...
131             for (int i = 0; i < dirtyLocks.length; i++) {
132                 dirtyLocks[i] = new Object();
133             }
134 
135             throttle = new IndexThrottle();
136             this.searcherFactory = new SearchFactory(engineConfig);
137             try {
138                 writer = createWriter();
139             } catch (IOException e) {
140                 throw new EngineCreationFailureException(shardId, "failed to create engine", e);
141             }
142             indexWriter = writer;
143             manager = createSearcherManager();
144             this.searcherManager = manager;
145             this.mergeSchedulerFailureListener = new FailEngineOnMergeFailure();
146             this.mergeSchedulerListener = new MergeSchedulerListener();
147             this.mergeScheduler.addListener(mergeSchedulerListener);
148             this.mergeScheduler.addFailureListener(mergeSchedulerFailureListener);
149             success = true;
150         } finally {
151             if (success == false) {
152                 IOUtils.closeWhileHandlingException(writer, manager);
153                 versionMap.clear();
154                 if (isClosed.get() == false) {
155                     // failure we need to dec the store reference
156                     store.decRef();
157                 }
158             }
159         }
160     }

162     private SearcherManager createSearcherManager() throws EngineException {
163         boolean success = false;
164         SearcherManager searcherManager = null;
165         try {
166             try {
167                 // commit on a just opened writer will commit even if there are no changes done to it
168                 // we rely on that for the commit data translog id key
169                 final long translogId;
170                 if (Lucene.indexExists(store.directory())) {
171                     Map<String, String> commitUserData = Lucene.readSegmentInfos(store.directory()).getUserData();
172                     if (commitUserData.containsKey(Translog.TRANSLOG_ID_KEY)) {
173                         translogId = Long.parseLong(commitUserData.get(Translog.TRANSLOG_ID_KEY));
174                     } else {
175                         translogId = System.currentTimeMillis();
176                         indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
177                         indexWriter.commit();
178                     }
179                 } else {
180                     translogId = System.currentTimeMillis();
181                     indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
182                     commitIndexWriter(indexWriter);
183                 }
184                 searcherManager = new SearcherManager(indexWriter, true, searcherFactory);
185                 lastCommittedSegmentInfos = store.readLastCommittedSegmentsInfo();
186                 translog.newTranslog(translogId);
187                 versionMap.setManager(searcherManager);
188                 translogIdGenerator.set(translogId);
189                 success = true;
190                 return searcherManager;
191             } catch (IOException e) {
192                 maybeFailEngine("start", e);
193                 try {
194                     indexWriter.rollback();
195                 } catch (IOException e1) { // iw is closed below
196                     e.addSuppressed(e1);
197                 }
198                 throw new EngineCreationFailureException(shardId, "failed to open reader on writer", e);
199             }
200         } finally {
201             if (success == false) { // release everything we created on a failure
202                 IOUtils.closeWhileHandlingException(searcherManager, indexWriter);
203             }
204         }
205     }

218     @Override
219     public GetResult get(Get get) throws EngineException {
220         try (ReleasableLock lock = readLock.acquire()) {
221             ensureOpen();
222             if (get.realtime()) {
223                 VersionValue versionValue = versionMap.getUnderLock(get.uid().bytes());
224                 if (versionValue != null) {
225                     if (versionValue.delete()) {
226                         return GetResult.NOT_EXISTS;
227                     }
228                     if (get.versionType().isVersionConflictForReads(versionValue.version(), get.version())) {
229                         Uid uid = Uid.createUid(get.uid().text());
230                         throw new VersionConflictEngineException(shardId, uid.type(), uid.id(), versionValue.version(), get.version());
231                     }
232                     if (!get.loadSource()) {
233                         return new GetResult(true, versionValue.version(), null);
234                     }
235                     Translog.Operation op = translog.read(versionValue.translogLocation());
236                     if (op != null) {
237                         return new GetResult(true, versionValue.version(), op.getSource());
238                     }
239                 }
240             }
241 
242             // no version, get the version from the index, we know that we refresh on flush
243             return getFromSearcher(get);
244         }
245     }

247     @Override
248     public void create(Create create) throws EngineException {
249         try (ReleasableLock lock = readLock.acquire()) {
250             ensureOpen();
251             if (create.origin() == Operation.Origin.RECOVERY) {
252                 // Don't throttle recovery operations
253                 innerCreate(create);
254             } else {
255                 try (Releasable r = throttle.acquireThrottle()) {
256                     innerCreate(create);
257                 }
258             }
259             flushNeeded = true;
260         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
261             maybeFailEngine("create", t);
262             throw new CreateFailedEngineException(shardId, create, t);
263         }
264         checkVersionMapRefresh();
265     }

267     private void innerCreate(Create create) throws IOException {
268         if (engineConfig.isOptimizeAutoGenerateId() && create.autoGeneratedId() && !create.canHaveDuplicates()) {
269             // We don't need to lock because this ID cannot be concurrently updated:
270             innerCreateNoLock(create, Versions.NOT_FOUND, null);
271         } else {
272             synchronized (dirtyLock(create.uid())) {
273                 final long currentVersion;
274                 final VersionValue versionValue;
275                 versionValue = versionMap.getUnderLock(create.uid().bytes());
276                 if (versionValue == null) {
277                     currentVersion = loadCurrentVersionFromIndex(create.uid());
278                 } else {
279                     if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
280                         currentVersion = Versions.NOT_FOUND; // deleted, and GC
281                     } else {
282                         currentVersion = versionValue.version();
283                     }
284                 }
285                 innerCreateNoLock(create, currentVersion, versionValue);
286             }
287         }
288     }

290     private void innerCreateNoLock(Create create, long currentVersion, VersionValue versionValue) throws IOException {
291 
292         // same logic as index
293         long updatedVersion;
294         long expectedVersion = create.version();
295         if (create.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
296             if (create.origin() == Operation.Origin.RECOVERY) {
297                 return;
298             } else {
299                 throw new VersionConflictEngineException(shardId, create.type(), create.id(), currentVersion, expectedVersion);
300             }
301         }
302         updatedVersion = create.versionType().updateVersion(currentVersion, expectedVersion);
303 
304         // if the doc exists
305         boolean doUpdate = false;
306         if ((versionValue != null && versionValue.delete() == false) || (versionValue == null && currentVersion != Versions.NOT_FOUND)) {
307             if (create.origin() == Operation.Origin.RECOVERY) {
308                 return;
309             } else if (create.origin() == Operation.Origin.REPLICA) {
310                 // #7142: the primary already determined it's OK to index this document, and we confirmed above that the version doesn't
311                 // conflict, so we must also update here on the replica to remain consistent:
312                 doUpdate = true;
313             } else if (create.origin() == Operation.Origin.PRIMARY && create.autoGeneratedId() && create.canHaveDuplicates() && currentVersion == 1 && create.version() == Versions.MATCH_ANY) {
314                 /**
315                  * If bulk index request fails due to a disconnect, unavailable shard etc. then the request is
316                  * retried before it actually fails. However, the documents might already be indexed.
317                  * For autogenerated ids this means that a version conflict will be reported in the bulk request
318                  * although the document was indexed properly.
319                  * To avoid this we have to make sure that the index request is treated as an update and set updatedVersion to 1.
320                  * See also discussion on https://github.com/elasticsearch/elasticsearch/pull/9125
321                  */
322                 doUpdate = true;
323                 updatedVersion = 1;
324             } else {
325                 // On primary, we throw DAEE if the _uid is already in the index with an older version:
326                 assert create.origin() == Operation.Origin.PRIMARY;
327                 throw new DocumentAlreadyExistsException(shardId, create.type(), create.id());
328             }
329         }
330 
331         create.updateVersion(updatedVersion);
332 
333         if (doUpdate) {
334             if (create.docs().size() > 1) {
335                 indexWriter.updateDocuments(create.uid(), create.docs(), create.analyzer());
336             } else {
337                 indexWriter.updateDocument(create.uid(), create.docs().get(0), create.analyzer());
338             }
339         } else {
340             if (create.docs().size() > 1) {
341                 indexWriter.addDocuments(create.docs(), create.analyzer());
342             } else {
343                 indexWriter.addDocument(create.docs().get(0), create.analyzer());
344             }
345         }
346         Translog.Location translogLocation = translog.add(new Translog.Create(create));
347 
348         versionMap.putUnderLock(create.uid().bytes(), new VersionValue(updatedVersion, translogLocation));
349 
350         indexingService.postCreateUnderLock(create);
351     }

353     @Override
354     public void index(Index index) throws EngineException {
355         try (ReleasableLock lock = readLock.acquire()) {
356             ensureOpen();
357             if (index.origin() == Operation.Origin.RECOVERY) {
358                 // Don't throttle recovery operations
359                 innerIndex(index);
360             } else {
361                 try (Releasable r = throttle.acquireThrottle()) {
362                     innerIndex(index);
363                 }
364             }
365             flushNeeded = true;
366         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
367             maybeFailEngine("index", t);
368             throw new IndexFailedEngineException(shardId, index, t);
369         }
370         checkVersionMapRefresh();
371     }

373     /**
374      * Forces a refresh if the versionMap is using too much RAM
375      */
376     private void checkVersionMapRefresh() {
377         if (versionMap.ramBytesUsedForRefresh() > config().getVersionMapSize().bytes() && versionMapRefreshPending.getAndSet(true) == false) {
378             try {
379                 if (isClosed.get()) {
380                     // no point...
381                     return;
382                 }
383                 // Now refresh to clear versionMap:
384                 engineConfig.getThreadPool().executor(ThreadPool.Names.REFRESH).execute(new Runnable() {
385                     public void run() {
386                         try {
387                             refresh("version_table_full");
388                         } catch (EngineClosedException ex) {
389                             // ignore
390                         }
391                     }
392                 });
393             } catch (EsRejectedExecutionException ex) {
394                 // that is fine too.. we might be shutting down
395             }
396         }
397     }

399     private void innerIndex(Index index) throws IOException {
400         synchronized (dirtyLock(index.uid())) {
401             final long currentVersion;
402             VersionValue versionValue = versionMap.getUnderLock(index.uid().bytes());
403             if (versionValue == null) {
404                 currentVersion = loadCurrentVersionFromIndex(index.uid());
405             } else {
406                 if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
407                     currentVersion = Versions.NOT_FOUND; // deleted, and GC
408                 } else {
409                     currentVersion = versionValue.version();
410                 }
411             }
412 
413             long updatedVersion;
414             long expectedVersion = index.version();
415             if (index.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
416                 if (index.origin() == Operation.Origin.RECOVERY) {
417                     return;
418                 } else {
419                     throw new VersionConflictEngineException(shardId, index.type(), index.id(), currentVersion, expectedVersion);
420                 }
421             }
422             updatedVersion = index.versionType().updateVersion(currentVersion, expectedVersion);
423 
424             index.updateVersion(updatedVersion);
425             if (currentVersion == Versions.NOT_FOUND) {
426                 // document does not exists, we can optimize for create
427                 index.created(true);
428                 if (index.docs().size() > 1) {
429                     indexWriter.addDocuments(index.docs(), index.analyzer());
430                 } else {
431                     indexWriter.addDocument(index.docs().get(0), index.analyzer());
432                 }
433             } else {
434                 if (versionValue != null) {
435                     index.created(versionValue.delete()); // we have a delete which is not GC'ed...
436                 }
437                 if (index.docs().size() > 1) {
438                     indexWriter.updateDocuments(index.uid(), index.docs(), index.analyzer());
439                 } else {
440                     indexWriter.updateDocument(index.uid(), index.docs().get(0), index.analyzer());
441                 }
442             }
443             Translog.Location translogLocation = translog.add(new Translog.Index(index));
444 
445             versionMap.putUnderLock(index.uid().bytes(), new VersionValue(updatedVersion, translogLocation));
446 
447             indexingService.postIndexUnderLock(index);
448         }
449     }

451     @Override
452     public void delete(Delete delete) throws EngineException {
453         try (ReleasableLock lock = readLock.acquire()) {
454             ensureOpen();
455             // NOTE: we don't throttle this when merges fall behind because delete-by-id does not create new segments:
456             innerDelete(delete);
457             flushNeeded = true;
458         } catch (OutOfMemoryError | IllegalStateException | IOException t) {
459             maybeFailEngine("delete", t);
460             throw new DeleteFailedEngineException(shardId, delete, t);
461         }
462 
463         maybePruneDeletedTombstones();
464         checkVersionMapRefresh();
465     }

467     private void maybePruneDeletedTombstones() {
468         // It's expensive to prune because we walk the deletes map acquiring dirtyLock for each uid so we only do it
469         // every 1/4 of gcDeletesInMillis:
470         if (engineConfig.isEnableGcDeletes() && engineConfig.getThreadPool().estimatedTimeInMillis() - lastDeleteVersionPruneTimeMSec > engineConfig.getGcDeletesInMillis() * 0.25) {
471             pruneDeletedTombstones();
472         }
473     }

475     private void innerDelete(Delete delete) throws IOException {
476         synchronized (dirtyLock(delete.uid())) {
477             final long currentVersion;
478             VersionValue versionValue = versionMap.getUnderLock(delete.uid().bytes());
479             if (versionValue == null) {
480                 currentVersion = loadCurrentVersionFromIndex(delete.uid());
481             } else {
482                 if (engineConfig.isEnableGcDeletes() && versionValue.delete() && (engineConfig.getThreadPool().estimatedTimeInMillis() - versionValue.time()) > engineConfig.getGcDeletesInMillis()) {
483                     currentVersion = Versions.NOT_FOUND; // deleted, and GC
484                 } else {
485                     currentVersion = versionValue.version();
486                 }
487             }
488 
489             long updatedVersion;
490             long expectedVersion = delete.version();
491             if (delete.versionType().isVersionConflictForWrites(currentVersion, expectedVersion)) {
492                 if (delete.origin() == Operation.Origin.RECOVERY) {
493                     return;
494                 } else {
495                     throw new VersionConflictEngineException(shardId, delete.type(), delete.id(), currentVersion, expectedVersion);
496                 }
497             }
498             updatedVersion = delete.versionType().updateVersion(currentVersion, expectedVersion);
499             final boolean found;
500             if (currentVersion == Versions.NOT_FOUND) {
501                 // doc does not exist and no prior deletes
502                 found = false;
503             } else if (versionValue != null && versionValue.delete()) {
504                 // a "delete on delete", in this case, we still increment the version, log it, and return that version
505                 found = false;
506             } else {
507                 // we deleted a currently existing document
508                 indexWriter.deleteDocuments(delete.uid());
509                 found = true;
510             }
511 
512             delete.updateVersion(updatedVersion, found);
513             Translog.Location translogLocation = translog.add(new Translog.Delete(delete));
514             versionMap.putUnderLock(delete.uid().bytes(), new DeleteVersionValue(updatedVersion, engineConfig.getThreadPool().estimatedTimeInMillis(), translogLocation));
515 
516             indexingService.postDeleteUnderLock(delete);
517         }
518     }

520     @Override
521     public void delete(DeleteByQuery delete) throws EngineException {
522         try (ReleasableLock lock = readLock.acquire()) {
523             ensureOpen();
524             if (delete.origin() == Operation.Origin.RECOVERY) {
525                 // Don't throttle recovery operations
526                 innerDelete(delete);
527             } else {
528                 try (Releasable r = throttle.acquireThrottle()) {
529                     innerDelete(delete);
530                 }
531             }
532         }
533     }

535     private void innerDelete(DeleteByQuery delete) throws EngineException {
536         try {
537             Query query;
538             if (delete.nested() && delete.aliasFilter() != null) {
539                 query = new IncludeNestedDocsQuery(new XFilteredQuery(delete.query(), delete.aliasFilter()), delete.parentFilter());
540             } else if (delete.nested()) {
541                 query = new IncludeNestedDocsQuery(delete.query(), delete.parentFilter());
542             } else if (delete.aliasFilter() != null) {
543                 query = new XFilteredQuery(delete.query(), delete.aliasFilter());
544             } else {
545                 query = delete.query();
546             }
547 
548             indexWriter.deleteDocuments(query);
549             translog.add(new Translog.DeleteByQuery(delete));
550             flushNeeded = true;
551         } catch (Throwable t) {
552             maybeFailEngine("delete_by_query", t);
553             throw new DeleteByQueryFailedEngineException(shardId, delete, t);
554         }
555 
556         // TODO: This is heavy, since we refresh, but we must do this because we don't know which documents were in fact deleted (i.e., our
557         // versionMap isn't updated), so we must force a cutover to a new reader to "see" the deletions:
558         refresh("delete_by_query");
559     }

561     @Override
562     public void refresh(String source) throws EngineException {
563         // we obtain a read lock here, since we don't want a flush to happen while we are refreshing
564         // since it flushes the index as well (though, in terms of concurrency, we are allowed to do it)
565         try (ReleasableLock lock = readLock.acquire()) {
566             ensureOpen();
567             updateIndexWriterSettings();
568             searcherManager.maybeRefreshBlocking();
569         } catch (AlreadyClosedException e) {
570             ensureOpen();
571             maybeFailEngine("refresh", e);
572         } catch (EngineClosedException e) {
573             throw e;
574         } catch (Throwable t) {
575             failEngine("refresh failed", t);
576             throw new RefreshFailedEngineException(shardId, t);
577         }
578 
579         // TODO: maybe we should just put a scheduled job in threadPool?
580         // We check for pruning in each delete request, but we also prune here e.g. in case a delete burst comes in and then no more deletes
581         // for a long time:
582         maybePruneDeletedTombstones();
583         versionMapRefreshPending.set(false);
584     }

596     private void flush(boolean commitTranslog, boolean force, boolean waitIfOngoing) throws EngineException {
597         ensureOpen();
598         if (commitTranslog) {
599             // check outside the lock as well so we can check without blocking on the write lock
600             if (onGoingRecoveries.get() > 0) {
601                 throw new FlushNotAllowedEngineException(shardId, "recovery is in progress, flush with committing translog is not allowed");
602             }
603         }
604         /*
605          * Unfortunately the lock order is important here. We have to acquire the readlock fist otherwise
606          * if we are flushing at the end of the recovery while holding the write lock we can deadlock if:
607          *  Thread 1: flushes via API and gets the flush lock but blocks on the readlock since Thread 2 has the writeLock
608          *  Thread 2: flushes at the end of the recovery holding the writeLock and blocks on the flushLock owned by Thread 1
609          */
610         try (ReleasableLock lock = readLock.acquire()) {
611             ensureOpen();
612             updateIndexWriterSettings();
613             if (flushLock.tryLock() == false) {
614                 // if we can't get the lock right away we block if needed otherwise barf
615                 if (waitIfOngoing) {
616                     logger.trace("waiting for in-flight flush to finish");
617                     flushLock.lock();
618                     logger.trace("acquired flush lock after blocking");
619                 } else {
620                     throw new FlushNotAllowedEngineException(shardId, "already flushing...");
621                 }
622             } else {
623                 logger.trace("acquired flush lock immediately");
624             }
625             try {
626                 if (commitTranslog) {
627                     if (onGoingRecoveries.get() > 0) {
628                         throw new FlushNotAllowedEngineException(shardId, "recovery is in progress, flush is not allowed");
629                     }
630 
631                     if (flushNeeded || force) {
632                         flushNeeded = false;
633                         try {
634                             long translogId = translogIdGenerator.incrementAndGet();
635                             translog.newTransientTranslog(translogId);
636                             indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
637                             commitIndexWriter(indexWriter);
638                             // we need to refresh in order to clear older version values
639                             refresh("version_table_flush");
640                             // we need to move transient to current only after we refresh
641                             // so items added to current will still be around for realtime get
642                             // when tans overrides it
643                             translog.makeTransientCurrent();
644 
645                         } catch (Throwable e) {
646                             translog.revertTransient();
647                             throw new FlushFailedEngineException(shardId, e);
648                         }
649                     }
650                 } else {
651                     // note, its ok to just commit without cleaning the translog, its perfectly fine to replay a
652                     // translog on an index that was opened on a committed point in time that is "in the future"
653                     // of that translog
654                     // we allow to *just* commit if there is an ongoing recovery happening...
655                     // its ok to use this, only a flush will cause a new translogId, and we are locked here from
656                     // other flushes use flushLock
657                     try {
658                         long translogId = translog.currentId();
659                         indexWriter.setCommitData(Collections.singletonMap(Translog.TRANSLOG_ID_KEY, Long.toString(translogId)));
660                         commitIndexWriter(indexWriter);
661                     } catch (Throwable e) {
662                         throw new FlushFailedEngineException(shardId, e);
663                     }
664 
665                 }
666                 /*
667                  * we have to inc-ref the store here since if the engine is closed by a tragic event
668                  * we don't acquire the write lock and wait until we have exclusive access. This might also
669                  * dec the store reference which can essentially close the store and unless we can inc the reference
670                  * we can't use it.
671                  */
672                 store.incRef();
673                 try {
674                     // reread the last committed segment infos
675                     lastCommittedSegmentInfos = store.readLastCommittedSegmentsInfo();
676                 } catch (Throwable e) {
677                     if (isClosed.get() == false) {
678                         logger.warn("failed to read latest segment infos on flush", e);
679                         if (Lucene.isCorruptionException(e)) {
680                             throw new FlushFailedEngineException(shardId, e);
681                         }
682                     }
683                 } finally {
684                     store.decRef();
685                 }
686             } catch (FlushFailedEngineException ex) {
687                 maybeFailEngine("flush", ex);
688                 throw ex;
689             } finally {
690                 flushLock.unlock();
691             }
692         }
693         // We don't have to do this here; we do it defensively to make sure that even if wall clock time is misbehaving
694         // (e.g., moves backwards) we will at least still sometimes prune deleted tombstones:
695         if (engineConfig.isEnableGcDeletes()) {
696             pruneDeletedTombstones();
697         }
698     }
699 
700     private void pruneDeletedTombstones() {
701         long timeMSec = engineConfig.getThreadPool().estimatedTimeInMillis();
702 
703         // TODO: not good that we reach into LiveVersionMap here; can we move this inside VersionMap instead?  problem is the dirtyLock...
704 
705         // we only need to prune the deletes map; the current/old version maps are cleared on refresh:
706         for (Map.Entry<BytesRef, VersionValue> entry : versionMap.getAllTombstones()) {
707             BytesRef uid = entry.getKey();
708             synchronized (dirtyLock(uid)) { // can we do it without this lock on each value? maybe batch to a set and get the lock once per set?
709 
710                 // Must re-get it here, vs using entry.getValue(), in case the uid was indexed/deleted since we pulled the iterator:
711                 VersionValue versionValue = versionMap.getTombstoneUnderLock(uid);
712                 if (versionValue != null) {
713                     if (timeMSec - versionValue.time() > engineConfig.getGcDeletesInMillis()) {
714                         versionMap.removeTombstoneUnderLock(uid);
715                     }
716                 }
717             }
718         }
719 
720         lastDeleteVersionPruneTimeMSec = timeMSec;
721     }

748     @Override
749     public void forceMerge(final boolean flush, int maxNumSegments, boolean onlyExpungeDeletes, final boolean upgrade) throws EngineException {
750         /*
751          * We do NOT acquire the readlock here since we are waiting on the merges to finish
752          * that's fine since the IW.rollback should stop all the threads and trigger an IOException
753          * causing us to fail the forceMerge
754          *
755          * The way we implement upgrades is a bit hackish in the sense that we set an instance
756          * variable and that this setting will thus apply to the next forced merge that will be run.
757          * This is ok because (1) this is the only place we call forceMerge, (2) we have a single
758          * thread for optimize, and the 'optimizeLock' guarding this code, and (3) ConcurrentMergeScheduler
759          * syncs calls to findForcedMerges.
760          */
761         assert indexWriter.getConfig().getMergePolicy() instanceof ElasticsearchMergePolicy : "MergePolicy is " + indexWriter.getConfig().getMergePolicy().getClass().getName();
762         ElasticsearchMergePolicy mp = (ElasticsearchMergePolicy) indexWriter.getConfig().getMergePolicy();
763         optimizeLock.lock();
764         try {
765             ensureOpen();
766             if (upgrade) {
767                 logger.info("starting segment upgrade");
768                 mp.setUpgradeInProgress(true);
769             }
770             store.incRef(); // increment the ref just to ensure nobody closes the store while we optimize
771             try {
772                 if (onlyExpungeDeletes) {
773                     assert upgrade == false;
774                     indexWriter.forceMergeDeletes(true /* blocks and waits for merges*/);
775                 } else if (maxNumSegments <= 0) {
776                     assert upgrade == false;
777                     indexWriter.maybeMerge();
778                 } else {
779                     indexWriter.forceMerge(maxNumSegments, true /* blocks and waits for merges*/);
780                 }
781                 if (flush) {
782                     flush(true, true, true);
783                 }
784                 if (upgrade) {
785                     logger.info("finished segment upgrade");
786                 }
787             } finally {
788                 store.decRef();
789             }
790         } catch (Throwable t) {
791             OptimizeFailedEngineException ex = new OptimizeFailedEngineException(shardId, t);
792             maybeFailEngine("force merge", ex);
793             throw ex;
794         } finally {
795             try {
796                 mp.setUpgradeInProgress(false); // reset it just to make sure we reset it in a case of an error
797             } finally {
798                 optimizeLock.unlock();
799             }
800         }
801     }

816     @Override
817     public void recover(RecoveryHandler recoveryHandler) throws EngineException {
818         // take a write lock here so it won't happen while a flush is in progress
819         // this means that next commits will not be allowed once the lock is released
820         try (ReleasableLock lock = writeLock.acquire()) {
821             ensureOpen();
822             onGoingRecoveries.startRecovery();
823         }
824 
825         SnapshotIndexCommit phase1Snapshot;
826         try {
827             logger.trace("[pre-phase1] performing deletion policy snapshot");
828             phase1Snapshot = deletionPolicy.snapshot();
829         } catch (Throwable e) {
830             maybeFailEngine("recovery", e);
831             Releasables.closeWhileHandlingException(onGoingRecoveries);
832             throw new RecoveryEngineException(shardId, 1, "Snapshot failed", e);
833         }
834 
835         try {
836             logger.trace("[phase1] performing phase 1 recovery (file recovery)");
837             recoveryHandler.phase1(phase1Snapshot);
838         } catch (Throwable e) {
839             maybeFailEngine("recovery phase 1 (file transfer)", e);
840             // close the snapshot first to release the reference to the translog file, so a flush post recovery can delete it
841             Releasables.closeWhileHandlingException(phase1Snapshot, onGoingRecoveries);
842             throw new RecoveryEngineException(shardId, 1, "Execution failed", wrapIfClosed(e));
843         }
844 
845         Translog.Snapshot phase2Snapshot;
846         try {
847             logger.trace("[pre-phase2] performing translog snapshot");
848             phase2Snapshot = translog.snapshot();
849         } catch (Throwable e) {
850             maybeFailEngine("translog snapshot", e);
851             Releasables.closeWhileHandlingException(phase1Snapshot, onGoingRecoveries);
852             throw new RecoveryEngineException(shardId, 2, "Snapshot failed", wrapIfClosed(e));
853         }
854         try {
855             logger.trace("[phase2] performing phase 2 recovery (translog replay)");
856             recoveryHandler.phase2(phase2Snapshot);
857         } catch (Throwable e) {
858             maybeFailEngine("recovery phase 2 (snapshot transfer)", e);
859             // close the snapshots first to release the reference to the translog file, so a flush post recovery can delete it
860             Releasables.closeWhileHandlingException(phase1Snapshot, phase2Snapshot, onGoingRecoveries);
861             throw new RecoveryEngineException(shardId, 2, "Execution failed", wrapIfClosed(e));
862         }
863 
864         writeLock.acquire();
865         Translog.Snapshot phase3Snapshot = null;
866         boolean success = false;
867         try {
868             ensureOpen();
869             logger.trace("[pre-phase3] performing translog snapshot");
870             phase3Snapshot = translog.snapshot(phase2Snapshot);
871             logger.trace("[phase3] performing phase 3 recovery (translog replay under write lock)");
872             recoveryHandler.phase3(phase3Snapshot);
873             success = true;
874         } catch (Throwable e) {
875             maybeFailEngine("recovery phase 3 (snapshot transfer)", e);
876             throw new RecoveryEngineException(shardId, 3, "Execution failed", wrapIfClosed(e));
877         } finally {
878             // close the snapshots first to release the reference to the translog file, so a flush post recovery can delete it
879             Releasables.close(success, phase1Snapshot, phase2Snapshot, phase3Snapshot,
880                     onGoingRecoveries, writeLock); // hmm why can't we use try-with here?
881         }
882         logger.trace("[post-recovery] recovery complete");
883     }

885     @Override
886     protected boolean maybeFailEngine(String source, Throwable t) {
887         boolean shouldFail = super.maybeFailEngine(source, t);
888         if (shouldFail) {
889             return true;
890         }
891 
892         // Check for AlreadyClosedException
893         if (t instanceof AlreadyClosedException) {
894             // if we are already closed due to some tragic exception
895             // we need to fail the engine. it might have already been failed before
896             // but we are double-checking it's failed and closed
897             if (isOpen(indexWriter) == false && getTragicException(indexWriter) != null) {
898                 failEngine("already closed by tragic event", getTragicException(indexWriter));
899             }
900             return true;
901         } else if (t != null && isOpen(indexWriter) == false && getTragicException(indexWriter) == t) {
902             // this spot on - we are handling the tragic event exception here so we have to fail the engine
903             // right away
904             failEngine(source, t);
905             return true;
906         }
907         return false;
908     }

959     /**
960      * Closes the engine without acquiring the write lock. This should only be
961      * called while the write lock is hold or in a disaster condition ie. if the engine
962      * is failed.
963      */
964     protected final void closeNoLock(String reason) throws ElasticsearchException {
965         if (isClosed.compareAndSet(false, true)) {
966             assert rwl.isWriteLockedByCurrentThread() || failEngineLock.isHeldByCurrentThread() : "Either the write lock must be held or the engine must be currently be failing itself";
967             try {
968                 try {
969                     translog.sync();
970                 } catch (IOException ex) {
971                     logger.warn("failed to sync translog");
972                 }
973                 this.versionMap.clear();
974                 logger.trace("close searcherManager");
975                 try {
976                     IOUtils.close(searcherManager);
977                 } catch (Throwable t) {
978                     logger.warn("Failed to close SearcherManager", t);
979                 }
980                 // no need to commit in this case!, we snapshot before we close the shard, so translog and all sync'ed
981                 logger.trace("rollback indexWriter");
982                 try {
983                     indexWriter.rollback();
984                 } catch (AlreadyClosedException e) {
985                     // ignore
986                 }
987                 logger.trace("rollback indexWriter done");
988             } catch (Throwable e) {
989                 logger.warn("failed to rollback writer on close", e);
990             } finally {
991                 store.decRef();
992                 this.mergeScheduler.removeListener(mergeSchedulerListener);
993                 this.mergeScheduler.removeFailureListener(mergeSchedulerFailureListener);
994                 logger.debug("engine closed [{}]", reason);
995             }
996         }
997     }

1024     private IndexWriter createWriter() throws IOException {
1025         try {
1026             boolean create = !Lucene.indexExists(store.directory());
1027             final IndexWriterConfig iwc = new IndexWriterConfig(Lucene.VERSION, engineConfig.getAnalyzer());
1028             iwc.setOpenMode(create ? IndexWriterConfig.OpenMode.CREATE : IndexWriterConfig.OpenMode.APPEND);
1029             iwc.setIndexDeletionPolicy(deletionPolicy);
1030             iwc.setInfoStream(new LoggerInfoStream(logger));
1031             iwc.setMergeScheduler(mergeScheduler.newMergeScheduler());
1032             MergePolicy mergePolicy = mergePolicyProvider.getMergePolicy();
1033             // Give us the opportunity to upgrade old segments while performing
1034             // background merges
1035             mergePolicy = new ElasticsearchMergePolicy(mergePolicy);
1036             iwc.setMergePolicy(mergePolicy);
1037             iwc.setSimilarity(engineConfig.getSimilarity());
1038             iwc.setRAMBufferSizeMB(engineConfig.getIndexingBufferSize().mbFrac());
1039             iwc.setMaxThreadStates(engineConfig.getIndexConcurrency());
1040             iwc.setCodec(engineConfig.getCodec());
1041             /* We set this timeout to a highish value to work around
1042              * the default poll interval in the Lucene lock that is
1043              * 1000ms by default. We might need to poll multiple times
1044              * here but with 1s poll this is only executed twice at most
1045              * in combination with the default writelock timeout*/
1046             iwc.setWriteLockTimeout(5000);
1047             iwc.setUseCompoundFile(this.engineConfig.isCompoundOnFlush());
1048             iwc.setCheckIntegrityAtMerge(engineConfig.isChecksumOnMerge());
1049             // Warm-up hook for newly-merged segments. Warming up segments here is better since it will be performed at the end
1050             // of the merge operation and won't slow down _refresh
1051             iwc.setMergedSegmentWarmer(new IndexReaderWarmer() {
1052                 @Override
1053                 public void warm(AtomicReader reader) throws IOException {
1054                     try {
1055                         assert isMergedSegment(reader);
1056                         if (warmer != null) {
1057                             final Engine.Searcher searcher = new Searcher("warmer", new IndexSearcher(reader));
1058                             final IndicesWarmer.WarmerContext context = new IndicesWarmer.WarmerContext(shardId, searcher);
1059                             warmer.warmNewReaders(context);
1060                         }
1061                     } catch (Throwable t) {
1062                         // Don't fail a merge if the warm-up failed
1063                         if (isClosed.get() == false) {
1064                             logger.warn("Warm-up failed", t);
1065                         }
1066                         if (t instanceof Error) {
1067                             // assertion/out-of-memory error, don't ignore those
1068                             throw (Error) t;
1069                         }
1070                     }
1071                 }
1072             });
1073             return new IndexWriter(store.directory(), iwc);
1074         } catch (LockObtainFailedException ex) {
1075             boolean isLocked = IndexWriter.isLocked(store.directory());
1076             logger.warn("Could not lock IndexWriter isLocked [{}]", ex, isLocked);
1077             throw ex;
1078         }
1079     }

1101         @Override
1102         public IndexSearcher newSearcher(IndexReader reader) throws IOException {
1103             IndexSearcher searcher = new IndexSearcher(reader);
1104             searcher.setSimilarity(engineConfig.getSimilarity());
1105             if (warmer != null) {
1106                 // we need to pass a custom searcher that does not release anything on Engine.Search Release,
1107                 // we will release explicitly
1108                 IndexSearcher newSearcher = null;
1109                 boolean closeNewSearcher = false;
1110                 try {
1111                     if (searcherManager == null) {
1112                         // we are starting up - no writer active so we can't acquire a searcher.
1113                         newSearcher = searcher;
1114                     } else {
1115                         try (final Searcher currentSearcher = acquireSearcher("search_factory")) {
1116                             // figure out the newSearcher, with only the new readers that are relevant for us
1117                             List<IndexReader> readers = Lists.newArrayList();
1118                             for (AtomicReaderContext newReaderContext : searcher.getIndexReader().leaves()) {
1119                                 if (isMergedSegment(newReaderContext.reader())) {
1120                                     // merged segments are already handled by IndexWriterConfig.setMergedSegmentWarmer
1121                                     continue;
1122                                 }
1123                                 boolean found = false;
1124                                 for (AtomicReaderContext currentReaderContext : currentSearcher.reader().leaves()) {
1125                                     if (currentReaderContext.reader().getCoreCacheKey().equals(newReaderContext.reader().getCoreCacheKey())) {
1126                                         found = true;
1127                                         break;
1128                                     }
1129                                 }
1130                                 if (!found) {
1131                                     readers.add(newReaderContext.reader());
1132                                 }
1133                             }
1134                             if (!readers.isEmpty()) {
1135                                 // we don't want to close the inner readers, just increase ref on them
1136                                 newSearcher = new IndexSearcher(new MultiReader(readers.toArray(new IndexReader[readers.size()]), false));
1137                                 closeNewSearcher = true;
1138                             }
1139                         }
1140                     }
1141 
1142                     if (newSearcher != null) {
1143                         IndicesWarmer.WarmerContext context = new IndicesWarmer.WarmerContext(shardId, new Searcher("warmer", newSearcher));
1144                         warmer.warmNewReaders(context);
1145                     }
1146                     warmer.warmTopReader(new IndicesWarmer.WarmerContext(shardId, new Searcher("warmer", searcher)));
1147                 } catch (Throwable e) {
1148                     if (isClosed.get() == false) {
1149                         logger.warn("failed to prepare/warm", e);
1150                     }
1151                 } finally {
1152                     // no need to release the fullSearcher, nothing really is done...
1153                     if (newSearcher != null && closeNewSearcher) {
1154                         IOUtils.closeWhileHandlingException(newSearcher.getIndexReader()); // ignore
1155                     }
1156                 }
1157             }
1158             return searcher;
1159         }

1194     class MergeSchedulerListener implements MergeSchedulerProvider.Listener {
1195         private final AtomicInteger numMergesInFlight = new AtomicInteger(0);
1196         private final AtomicBoolean isThrottling = new AtomicBoolean();
1197 
1198         @Override
1199         public synchronized void beforeMerge(OnGoingMerge merge) {
1200             int maxNumMerges = mergeScheduler.getMaxMerges();
1201             if (numMergesInFlight.incrementAndGet() > maxNumMerges) {
1202                 if (isThrottling.getAndSet(true) == false) {
1203                     logger.info("now throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
1204                     indexingService.throttlingActivated();
1205                     activateThrottling();
1206                 }
1207             }
1208         }
1209 
1210         @Override
1211         public synchronized void afterMerge(OnGoingMerge merge) {
1212             int maxNumMerges = mergeScheduler.getMaxMerges();
1213             if (numMergesInFlight.decrementAndGet() < maxNumMerges) {
1214                 if (isThrottling.getAndSet(false)) {
1215                     logger.info("stop throttling indexing: numMergesInFlight={}, maxNumMerges={}", numMergesInFlight, maxNumMerges);
1216                     indexingService.throttlingDeactivated();
1217                     deactivateThrottling();
1218                 }
1219             }
1220         }
1221     }
