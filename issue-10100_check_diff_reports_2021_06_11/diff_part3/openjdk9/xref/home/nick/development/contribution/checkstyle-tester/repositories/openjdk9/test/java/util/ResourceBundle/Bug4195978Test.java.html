23  /**
24   * @test
25   * @bug 4195978
26   * @summary Verifies that resource bundle names have case distinction.
27   * @author joconner
28   */
29  import java.util.*;
30  
31  public class Bug4195978Test extends ListResourceBundle {
32  
33      public static void main(final String args[]) throws Exception {
34          new Bug4195978Test().test();
35      }
36  
37      public void test() throws Exception {
38          try {
39              // load a property resourcebundle
40              final ResourceBundle bundle = ResourceBundle.getBundle("bug4195978Test");
41              // load this file as a ListResourceBundle
42              final ResourceBundle bundle2 = ResourceBundle.getBundle("Bug4195978Test");
43  
44              // get the "test" keyid from both bundles
45              String b1 = bundle.getString("test");
46              String b2 = bundle2.getString("test");
47  
48              // one should be lowercase, the other is uppercase
49              // each bundle can be used seperately because their names
50              // are distinguished by case.
51              if (b1.equals("test") && b2.equals("TEST")) {
52                  System.out.println("Passed");
53              }
54          } catch (Exception e) {
55              System.err.println("Failed");
56              System.err.println(e);
57              throw e;
58          }
59  
60  
61      }
62  
63      public Object[][] getContents() {
64          return contents;
65      }
66  
67      Object[][] contents = {
68          {"test", "TEST"},
69      };
70  
71  
72  }
