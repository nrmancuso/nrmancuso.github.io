19  package org.jsecurity.authc.pam;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  import org.jsecurity.authc.*;
24  import org.jsecurity.realm.Realm;
25  
26  /**
27   * <tt>ModularAuthenticationStrategy</tt> implementation that requires <em>all</em> configured realms to
28   * <b>successfully</b> process the submitted <tt>AuthenticationToken</tt> during the log-in attempt.
29   *
30   * <p>If one or more realms do not support the submitted token, or one or more are unable to acquire
31   * <tt>AuthenticationInfo</tt> for the token, this implementation will immediately fail the log-in attempt for the
32   * associated subject (user).
33   *
34   * @author Les Hazlewood
35   * @since 0.2
36   */
37  public class AllSuccessfulModularAuthenticationStrategy extends AbstractAuthenticationStrategy {
38  
39      private static final Log log = LogFactory.getLog(AllSuccessfulModularAuthenticationStrategy.class);    
40  
41      public AuthenticationInfo beforeAttempt(Realm realm, AuthenticationToken token, AuthenticationInfo info) throws AuthenticationException {
42          if (!realm.supports(token)) {
43              String msg = "Realm [" + realm + "] of type [" + realm.getClass().getName() + "] does not support " +
44                      " the submitted AuthenticationToken [" + token + "].  The [" + getClass().getName() +
45                      "] implementation requires all configured realm(s) to support and be able to process the submitted " +
46                      "AuthenticationToken.";
47              throw new UnsupportedTokenException(msg);
48          }
49  
50          return info;
51      }
52  
53      public AuthenticationInfo afterAttempt(Realm realm, AuthenticationToken token, AuthenticationInfo info, AuthenticationInfo aggregate, Throwable t)
54              throws AuthenticationException {
55          if (t != null) {
56              if (t instanceof AuthenticationException) {
57                  //propagate:
58                  throw ((AuthenticationException) t);
59              } else {
60                  String msg = "Unable to acquire account data from realm [" + realm + "].  The [" +
61                          getClass().getName() + " implementation requires all configured realm(s) to operate successfully " +
62                          "for a successful authentication.";
63                  throw new AuthenticationException(msg, t);
64              }
65          }
66          if (info == null) {
67              String msg = "Realm [" + realm + "] could not find any associated account data for the submitted " +
68                      "AuthenticationToken [" + token + "].  The [" + getClass().getName() + "] implementation requires " +
69                      "all configured realm(s) to acquire valid account data for a submitted token during the " +
70                      "log-in process.";
71              throw new UnknownAccountException(msg);
72          }
73  
74          // If non-null account is returned, then the realm was able to authenticate the
75          // user - so merge the account with any accumulated before:
76          if (log.isDebugEnabled()) {
77              log.debug("Account successfully authenticated using realm of type [" + realm.getClass().getName() + "]");
78          }
79  
80          if( aggregate instanceof MergableAuthenticationInfo ) {
81              ((MergableAuthenticationInfo)aggregate).merge(info);
82              return aggregate;
83          } else {
84              throw new IllegalArgumentException( "Attempt to merge authentication info from multiple realms, but aggreagate " +
85                        "AuthenticationInfo is not of type MergableAuthenticationInfo." );
86          }
87      }
88  }
