24  package org.hibernate.test.annotations.lob;
25  
26  import java.util.Arrays;
27  
28  import org.junit.Test;
29  
30  import org.hibernate.Session;
31  import org.hibernate.testing.DialectChecks;
32  import org.hibernate.testing.RequiresDialectFeature;
33  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
34  import org.hibernate.type.MaterializedBlobType;
35  import org.hibernate.type.Type;
36  
37  import static org.junit.Assert.assertEquals;
38  import static org.junit.Assert.assertTrue;
39  
40  /**
41   * @author Steve Ebersole
42   */
43  @RequiresDialectFeature(DialectChecks.SupportsExpectedLobUsagePattern.class)
44  public class MaterializedBlobTest extends BaseCoreFunctionalTestCase {
45  	@Override
46  	protected Class<?>[] getAnnotatedClasses() {
47  		return new Class<?>[] { MaterializedBlobEntity.class };
48  	}
49  
50  	@Test
51  	public void testTypeSelection() {
52  		int index = sessionFactory().getEntityPersister( MaterializedBlobEntity.class.getName() ).getEntityMetamodel().getPropertyIndex( "theBytes" );
53  		Type  type = sessionFactory().getEntityPersister( MaterializedBlobEntity.class.getName() ).getEntityMetamodel().getProperties()[index].getType();
54  		assertEquals( MaterializedBlobType.INSTANCE, type );
55  	}
56  
57  	@Test
58  	public void testSaving() {
59  		byte[] testData = "test data".getBytes();
60  
61  		Session session = openSession();
62  		session.beginTransaction();
63  		MaterializedBlobEntity entity = new MaterializedBlobEntity( "test", testData );
64  		session.save( entity );
65  		session.getTransaction().commit();
66  		session.close();
67  
68  		session = openSession();
69  		session.beginTransaction();
70  		entity = ( MaterializedBlobEntity ) session.get( MaterializedBlobEntity.class, entity.getId() );
71  		assertTrue( Arrays.equals( testData, entity.getTheBytes() ) );
72  		session.delete( entity );
73  		session.getTransaction().commit();
74  		session.close();
75  	}
76  }
