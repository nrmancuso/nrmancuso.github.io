66  class UPSFindScp extends DcmServiceBase {
67  
68      private static final int FUZZY_MATCHING = 2;
69  
70      private final UPSScpService service;
71  
72      private Logger log;
73  
74      public UPSFindScp(UPSScpService service) {
75          this.service = service;
76          this.log = service.getLog();
77      }
78  
79      @Override
80      protected MultiDimseRsp doCFind(ActiveAssociation assoc, Dimse rq,
81              Command rspCmd) throws IOException, DcmServiceException {
82          Association a = assoc.getAssociation();
83          Command rqCmd = rq.getCommand();
84          Dataset rqData = rq.getDataset();
85          if (rqData == null)
86              throw new DcmServiceException(Status.MistypedArgument,
87                      "C-FIND-RQ without Data Set");
88          log.debug("Identifier:\n");
89          log.debug(rqData);
90          checkCFindRQ(rqData);
91          try {
92              service.logDicomQuery(a, rqCmd.getAffectedSOPClassUID(),
93                      rqData);
94              boolean fuzzyMatchingOfPN = fuzzyMatchingOfPN(
95                      a.getAcceptedExtNegotiation(
96                              rqCmd.getAffectedSOPClassUID()));
97              UPSQueryCmd queryCmd = 
98                      new UPSQueryCmd(rqData, fuzzyMatchingOfPN,
99                              service.isNoMatchForNoValue());
100             queryCmd.setFetchSize(service.getFetchSize()).execute();
101             return new MultiCFindRsp(queryCmd);
102         } catch (DcmServiceException e) {
103             throw e;
104         } catch (Exception e) {
105             service.getLog().error("Query DB failed:", e);
106             throw new DcmServiceException(Status.ProcessingFailure, e);
107         }
108     }
109 
110     private boolean fuzzyMatchingOfPN(ExtNegotiation extNeg) {
111         byte[] info;
112         return extNeg != null 
113                 && (info = extNeg.info()).length > FUZZY_MATCHING
114                 && info[FUZZY_MATCHING] != 0;
115     }
116 
117    private void checkCFindRQ(Dataset rqData) throws DcmServiceException {
118         String s = rqData.getString(Tags.UPSState);
119         if (s != null)
120             UPSScpService.upsStateAsInt(s);
121     }
122 
123     private class MultiCFindRsp implements MultiDimseRsp {
124         private final UPSQueryCmd queryCmd;
125         private boolean canceled = false;
126 
127         public MultiCFindRsp(UPSQueryCmd queryCmd) {
128             this.queryCmd = queryCmd;
129         }
130 
131         public DimseListener getCancelListener() {
132             return new DimseListener() {
133                 public void dimseReceived(Association assoc, Dimse dimse) {
134                     canceled = true;
135                 }
136             };
137         }
138 
139         public Dataset next(ActiveAssociation assoc, Dimse rq, Command rspCmd)
140             throws DcmServiceException
141         {
142             if (canceled) {
143                 rspCmd.putUS(Tags.Status, Status.Cancel);
144                 return null;
145             }
146             try {
147                 if (!queryCmd.next()) {
148                     rspCmd.putUS(Tags.Status, Status.Success);
149                     return null;
150                 }
151                 rspCmd.putUS(Tags.Status, Status.Pending);
152                 Dataset rspData = queryCmd.getDataset();
153                                 log.debug("Identifier:\n");
154                                 log.debug(rspData);
155                 return rspData;
156             } catch (SQLException e) {
157                 service.getLog().error("Retrieve DB record failed:", e);
158                 throw new DcmServiceException(Status.ProcessingFailure, e);                
159             } catch (Exception e) {
160                 service.getLog().error("Corrupted DB record:", e);
161                 throw new DcmServiceException(Status.ProcessingFailure, e);                
162             }
163         }
164 
165         public void release() {
166             queryCmd.close();
167         }
168     }
