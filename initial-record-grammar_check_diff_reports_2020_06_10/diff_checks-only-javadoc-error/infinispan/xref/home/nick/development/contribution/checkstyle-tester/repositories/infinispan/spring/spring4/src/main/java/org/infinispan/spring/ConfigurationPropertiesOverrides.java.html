1   package org.infinispan.spring;
2   
3   import java.net.InetSocketAddress;
4   import java.util.Collection;
5   import java.util.Map;
6   import java.util.Properties;
7   
8   import static org.infinispan.client.hotrod.impl.ConfigurationProperties.*;
9   
10  /**
11   * <p>
12   * Provides a mechanism to override selected configuration properties using explicit setters for
13   * each configuration setting.
14   * </p>
15   *
16   * @author <a href="mailto:olaf DOT bergner AT gmx DOT de">Olaf Bergner</a>
17   *
18   */
19  public class ConfigurationPropertiesOverrides {
20  
21     private final Properties overridingProperties = new Properties();
22  
23     /**
24      * @return
25      * @see java.util.Hashtable#isEmpty()
26      */
27     public boolean isEmpty() {
28        return this.overridingProperties.isEmpty();
29     }
30  
31     /**
32      * @param TransportFactory
33      */
34     public void setTransportFactory(final String TransportFactory) {
35        this.overridingProperties.setProperty(TRANSPORT_FACTORY, TransportFactory);
36     }
37  
38     /**
39      * @param serverList
40      */
41     public void setServerList(final Collection<InetSocketAddress> serverList) {
42        final StringBuilder serverListStr = new StringBuilder();
43        for (final InetSocketAddress server : serverList) {
44           serverListStr.append(server.getHostName()).append(":").append(server.getPort())
45                 .append(";");
46        }
47        serverListStr.deleteCharAt(serverListStr.length() - 1);
48        this.overridingProperties.setProperty(SERVER_LIST, serverListStr.toString());
49     }
50  
51     /**
52      * @param marshaller
53      */
54     public void setMarshaller(final String marshaller) {
55        this.overridingProperties.setProperty(MARSHALLER, marshaller);
56     }
57  
58     /**
59      * @param asyncExecutorFactory
60      */
61     public void setAsyncExecutorFactory(final String asyncExecutorFactory) {
62        this.overridingProperties.setProperty(ASYNC_EXECUTOR_FACTORY, asyncExecutorFactory);
63     }
64  
65     /**
66      * @param tcpNoDelay
67      */
68     public void setTcpNoDelay(final boolean tcpNoDelay) {
69        this.overridingProperties.setProperty(TCP_NO_DELAY, Boolean.toString(tcpNoDelay));
70     }
71  
72     public void setTcpKeepAlive(final boolean tcpKeepAlive) {
73        this.overridingProperties.setProperty(TCP_KEEP_ALIVE, Boolean.toString(tcpKeepAlive));
74     }
75  
76     /**
77      * @param pingOnStartup
78      */
79     public void setPingOnStartup(final boolean pingOnStartup) {
80        this.overridingProperties.setProperty(PING_ON_STARTUP, Boolean.toString(pingOnStartup));
81     }
82  
83     /**
84      * @param requestBalancingStrategy
85      */
86     public void setRequestBalancingStrategy(final String requestBalancingStrategy) {
87        this.overridingProperties.setProperty(REQUEST_BALANCING_STRATEGY, requestBalancingStrategy);
88     }
89  
90     /**
91      * @param keySizeEstimate
92      */
93     public void setKeySizeEstimate(final int keySizeEstimate) {
94        this.overridingProperties.setProperty(KEY_SIZE_ESTIMATE, Integer.toString(keySizeEstimate));
95     }
96  
97     /**
98      * @param valueSizeEstimate
99      */
100    public void setValueSizeEstimate(final int valueSizeEstimate) {
101       this.overridingProperties.setProperty(VALUE_SIZE_ESTIMATE,
102                                             Integer.toString(valueSizeEstimate));
103    }
104 
105    /**
106     * @param forceReturnValues
107     */
108    public void setForceReturnValues(final boolean forceReturnValues) {
109       this.overridingProperties.setProperty(FORCE_RETURN_VALUES,
110                                             Boolean.toString(forceReturnValues));
111    }
112 
113    /**
114     * @param configurationPropertiesToOverride
115     * @return
116     */
117    public Properties override(final Properties configurationPropertiesToOverride) {
118       final Properties answer = Properties.class.cast(configurationPropertiesToOverride.clone());
119       for (final Map.Entry<Object, Object> prop : this.overridingProperties.entrySet()) {
120          answer.setProperty(String.class.cast(prop.getKey()), String.class.cast(prop.getValue()));
121       }
122       return answer;
123    }
124 }
