4   import static java.lang.annotation.RetentionPolicy.RUNTIME;

6   import java.lang.annotation.Retention;
7   import java.lang.annotation.Target;
8   
9   /**
10   * <p>
11   * Identifies the key for a group.
12   * </p>
13   * 
14   * <p>
15   * <code>@Group</code> should be used when you have control over the key class. For example:
16   * </p>
17   * 
18   * <pre>
19   * class User {
20   * 
21   *    ...
22   *    String office;
23   *    ...
24   *    
25   *    int hashCode() {
26   *       // Defines the hash for the key, normally used to determine location
27   *       ...
28   *    }
29   *    
30   *    // Override the location by specifying a group, all keys in the same 
31   *    // group end up with the same owner
32   *    @Group
33   *    String getOffice() {
34   *       return office;
35   *    }
36   *    
37   * }
38   * </pre>
39   * 
40   * <p>
41   * If you don't have control over the key class, you can specify a {@link Grouper} (in your configuration) which can be used to
42   * specify the group externally.
43   * </p>
44   * 
45   * <p>
46   * You must set the <code>groupsEnabled<code> property to true in your configuration in order to use groups.
47   * </p>
48   * 
49   * @see Grouper
50   * 
51   * @author Pete Muir
52   * 
53   */
54  @Target(METHOD)
55  @Retention(RUNTIME)
56  public @interface Group {
57  
58  }
