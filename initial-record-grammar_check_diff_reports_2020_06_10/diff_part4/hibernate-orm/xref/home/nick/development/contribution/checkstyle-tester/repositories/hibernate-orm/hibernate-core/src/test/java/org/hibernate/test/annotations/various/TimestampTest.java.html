24  package org.hibernate.test.annotations.various;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.mapping.PersistentClass;
29  import org.hibernate.mapping.Property;
30  import org.hibernate.metadata.ClassMetadata;
31  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
32  import org.hibernate.type.DbTimestampType;
33  import org.hibernate.type.TimestampType;
34  
35  import static org.junit.Assert.assertEquals;
36  import static org.junit.Assert.assertNotNull;
37  import static org.junit.Assert.assertTrue;
38  
39  /**
40   * Test for the @Timestamp annotation.
41   *
42   * @author Hardy Ferentschik
43   */
44  public class TimestampTest extends BaseCoreFunctionalTestCase {
45  	@Test
46  	public void testTimestampSourceIsVM() throws Exception {
47  		assertTimestampSource( VMTimestamped.class, TimestampType.class );
48  	}
49  
50  	@Test
51  	public void testTimestampSourceIsDB() throws Exception {
52  		assertTimestampSource( DBTimestamped.class, DbTimestampType.class );
53  	}
54  
55  	private void assertTimestampSource(Class<?> clazz, Class<?> expectedTypeClass) throws Exception {
56  		buildConfiguration();
57  		ClassMetadata meta = sessionFactory().getClassMetadata( clazz );
58  		assertTrue( "Entity is annotated with @Timestamp and should hence be versioned", meta.isVersioned() );
59  
60  		PersistentClass persistentClass = configuration().getClassMapping( clazz.getName() );
61  		assertNotNull( persistentClass );
62  		Property versionProperty = persistentClass.getVersion();
63  		assertNotNull( versionProperty );
64  		assertEquals( "Wrong timestamp type", expectedTypeClass, versionProperty.getType().getClass() );
65  	}
66  
67  	@Override
68  	protected Class[] getAnnotatedClasses() {
69  		return new Class[] {
70  				VMTimestamped.class, DBTimestamped.class
71  		};
72  	}
73  }
