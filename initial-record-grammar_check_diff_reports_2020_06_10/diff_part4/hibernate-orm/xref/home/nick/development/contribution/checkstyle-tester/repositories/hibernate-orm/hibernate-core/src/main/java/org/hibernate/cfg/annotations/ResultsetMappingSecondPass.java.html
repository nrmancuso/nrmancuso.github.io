59  public class ResultsetMappingSecondPass implements QuerySecondPass {
60      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class,
61                                                                         ResultsetMappingSecondPass.class.getName());
62  
63  	private SqlResultSetMapping ann;
64  	private Mappings mappings;
65  	private boolean isDefault;
66  
67  	public ResultsetMappingSecondPass(SqlResultSetMapping ann, Mappings mappings, boolean isDefault) {
68  		this.ann = ann;
69  		this.mappings = mappings;
70  		this.isDefault = isDefault;
71  	}
72  
73  	public void doSecondPass(Map persistentClasses) throws MappingException {
74  		//TODO add parameters checkings
75  		if ( ann == null ) return;
76  		ResultSetMappingDefinition definition = new ResultSetMappingDefinition( ann.name() );
77  		LOG.debugf( "Binding result set mapping: %s", definition.getName() );
78  
79  		int entityAliasIndex = 0;
80  
81  		for (EntityResult entity : ann.entities()) {
82  			//TODO parameterize lock mode?
83  			List<FieldResult> properties = new ArrayList<FieldResult>();
84  			List<String> propertyNames = new ArrayList<String>();
85  			for (FieldResult field : entity.fields()) {
86  				//use an ArrayList cause we might have several columns per root property
87  				String name = field.name();
88  				if ( name.indexOf( '.' ) == -1 ) {
89  					//regular property
90  					properties.add( field );
91  					propertyNames.add( name );
92  				}
93  				else {
94  					/**
95  					 * Reorder properties
96  					 * 1. get the parent property
97  					 * 2. list all the properties following the expected one in the parent property
98  					 * 3. calculate the lowest index and insert the property
99  					 */
100 					PersistentClass pc = mappings.getClass( entity.entityClass().getName() );
101 					if ( pc == null ) {
102 						throw new MappingException(
103 								"Entity not found " + entity.entityClass().getName()
104 										+ " in SqlResultsetMapping " + ann.name()
105 						);
106 					}
107 					int dotIndex = name.lastIndexOf( '.' );
108 					String reducedName = name.substring( 0, dotIndex );
109 					Iterator parentPropIter = getSubPropertyIterator( pc, reducedName );
110 					List followers = getFollowers( parentPropIter, reducedName, name );
111 
112 					int index = propertyNames.size();
113 					int followersSize = followers.size();
114 					for (int loop = 0; loop < followersSize; loop++) {
115 						String follower = (String) followers.get( loop );
116 						int currentIndex = getIndexOfFirstMatchingProperty( propertyNames, follower );
117 						index = currentIndex != -1 && currentIndex < index ? currentIndex : index;
118 					}
119 					propertyNames.add( index, name );
120 					properties.add( index, field );
121 				}
122 			}
123 
124 			Set<String> uniqueReturnProperty = new HashSet<String>();
125 			Map<String, ArrayList<String>> propertyResultsTmp = new HashMap<String, ArrayList<String>>();
126 			for ( Object property : properties ) {
127 				final FieldResult propertyresult = ( FieldResult ) property;
128 				final String name = propertyresult.name();
129 				if ( "class".equals( name ) ) {
130 					throw new MappingException(
131 							"class is not a valid property name to use in a @FieldResult, use @Entity(discriminatorColumn) instead"
132 					);
133 				}
134 
135 				if ( uniqueReturnProperty.contains( name ) ) {
136 					throw new MappingException(
137 							"duplicate @FieldResult for property " + name +
138 									" on @Entity " + entity.entityClass().getName() + " in " + ann.name()
139 					);
140 				}
141 				uniqueReturnProperty.add( name );
142 
143 				final String quotingNormalizedColumnName = mappings.getObjectNameNormalizer()
144 						.normalizeIdentifierQuoting( propertyresult.column() );
145 
146 				String key = StringHelper.root( name );
147 				ArrayList<String> intermediateResults = propertyResultsTmp.get( key );
148 				if ( intermediateResults == null ) {
149 					intermediateResults = new ArrayList<String>();
150 					propertyResultsTmp.put( key, intermediateResults );
151 				}
152 				intermediateResults.add( quotingNormalizedColumnName );
153 			}
154 
155 			Map<String, String[]> propertyResults = new HashMap<String,String[]>();
156 			for ( Map.Entry<String, ArrayList<String>> entry : propertyResultsTmp.entrySet() ) {
157 				propertyResults.put(
158 						entry.getKey(),
159 						entry.getValue().toArray( new String[ entry.getValue().size() ] )
160 				);
161 			}
162 
163 			if ( !BinderHelper.isEmptyAnnotationValue( entity.discriminatorColumn() ) ) {
164 				final String quotingNormalizedName = mappings.getObjectNameNormalizer().normalizeIdentifierQuoting(
165 						entity.discriminatorColumn()
166 				);
167 				propertyResults.put( "class", new String[] { quotingNormalizedName } );
168 			}
169 
170 			if ( propertyResults.isEmpty() ) {
171 				propertyResults = java.util.Collections.emptyMap();
172 			}
173 
174 			NativeSQLQueryRootReturn result = new NativeSQLQueryRootReturn(
175 					"alias" + entityAliasIndex++,
176 					entity.entityClass().getName(),
177 					propertyResults,
178 					LockMode.READ
179 			);
180 			definition.addQueryReturn( result );
181 		}
182 
183 		for ( ColumnResult column : ann.columns() ) {
184 			definition.addQueryReturn(
185 					new NativeSQLQueryScalarReturn(
186 							mappings.getObjectNameNormalizer().normalizeIdentifierQuoting(
187 									column.name()
188 							),
189 							null
190 					)
191 			);
192 		}
193 
194 		if ( isDefault ) {
195 			mappings.addDefaultResultSetMapping( definition );
196 		}
197 		else {
198 			mappings.addResultSetMapping( definition );
199 		}
200 	}
201 
202 	@SuppressWarnings({ "unchecked" })
203 	private List getFollowers(Iterator parentPropIter, String reducedName, String name) {
204 		boolean hasFollowers = false;
205 		List followers = new ArrayList();
206 		while ( parentPropIter.hasNext() ) {
207 			String currentPropertyName = ( (Property) parentPropIter.next() ).getName();
208 			String currentName = reducedName + '.' + currentPropertyName;
209 			if ( hasFollowers ) {
210 				followers.add( currentName );
211 			}
212 			if ( name.equals( currentName ) ) hasFollowers = true;
213 		}
214 		return followers;
215 	}
216 
217 	private Iterator getSubPropertyIterator(PersistentClass pc, String reducedName) {
218 		Value value = pc.getRecursiveProperty( reducedName ).getValue();
219 		Iterator parentPropIter;
220 		if ( value instanceof Component ) {
221 			Component comp = (Component) value;
222 			parentPropIter = comp.getPropertyIterator();
223 		}
224 		else if ( value instanceof ToOne ) {
225 			ToOne toOne = (ToOne) value;
226 			PersistentClass referencedPc = mappings.getClass( toOne.getReferencedEntityName() );
227 			if ( toOne.getReferencedPropertyName() != null ) {
228 				try {
229 					parentPropIter = ( (Component) referencedPc.getRecursiveProperty(
230 							toOne.getReferencedPropertyName()
231 					).getValue() ).getPropertyIterator();
232 				}
233 				catch (ClassCastException e) {
234 					throw new MappingException(
235 							"dotted notation reference neither a component nor a many/one to one", e
236 					);
237 				}
238 			}
239 			else {
240 				try {
241 					if ( referencedPc.getIdentifierMapper() == null ) {
242 						parentPropIter = ( (Component) referencedPc.getIdentifierProperty()
243 								.getValue() ).getPropertyIterator();
244 					}
245 					else {
246 						parentPropIter = referencedPc.getIdentifierMapper().getPropertyIterator();
247 					}
248 				}
249 				catch (ClassCastException e) {
250 					throw new MappingException(
251 							"dotted notation reference neither a component nor a many/one to one", e
252 					);
253 				}
254 			}
255 		}
256 		else {
257 			throw new MappingException( "dotted notation reference neither a component nor a many/one to one" );
258 		}
259 		return parentPropIter;
260 	}
261 
262 	private static int getIndexOfFirstMatchingProperty(List propertyNames, String follower) {
263 		int propertySize = propertyNames.size();
264 		for (int propIndex = 0; propIndex < propertySize; propIndex++) {
265 			if ( ( (String) propertyNames.get( propIndex ) ).startsWith( follower ) ) {
266 				return propIndex;
267 			}
268 		}
269 		return -1;
270 	}
