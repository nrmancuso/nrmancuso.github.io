17  package org.springframework.integration.json;
18  
19  import static org.assertj.core.api.Assertions.assertThat;
20  
21  import java.util.List;
22  
23  import org.junit.jupiter.api.Test;
24  
25  import org.springframework.core.ParameterizedTypeReference;
26  import org.springframework.core.ResolvableType;
27  import org.springframework.integration.expression.ValueExpression;
28  import org.springframework.integration.support.json.Jackson2JsonObjectMapper;
29  import org.springframework.messaging.Message;
30  import org.springframework.messaging.support.GenericMessage;
31  
32  import com.fasterxml.jackson.core.JsonParser;
33  import com.fasterxml.jackson.databind.ObjectMapper;
34  
35  /**
36   * @author Mark Fisher
37   * @author Artem Bilan
38   * @author Gary Russell
39   *
40   * @since 2.0
41   */
42  public class JsonToObjectTransformerTests {
43  
44  	@Test
45  	public void objectPayload() {
46  		JsonToObjectTransformer transformer =
47  				new JsonToObjectTransformer(
48  						ResolvableType.forType(new ParameterizedTypeReference<List<TestPerson>>() { }));
49  		// Since DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES is disabled by default
50  		// (see Jackson2JsonObjectMapper)
51  		// the extra "foo" property is ignored.
52  		String jsonString = "[{\"firstName\":\"John\",\"lastName\":\"Doe\",\"age\":42," +
53  				"\"address\":{\"number\":123,\"street\":\"Main Street\"}, \"foo\":\"bar\"}]";
54  		Message<?> message = transformer.transform(new GenericMessage<>(jsonString));
55  		assertThat(message)
56  				.extracting(Message::getPayload)
57  				.isInstanceOf(List.class)
58  				.asList()
59  				.hasSize(1)
60  				.element(0)
61  				.isInstanceOf(TestPerson.class)
62  				.satisfies((actual) -> {
63  					TestPerson bean = (TestPerson) actual;
64  					assertThat(bean).extracting(TestPerson::getFirstName).isEqualTo("John");
65  					assertThat(bean).extracting(TestPerson::getLastName).isEqualTo("Doe");
66  					assertThat(bean).extracting(TestPerson::getAge).isEqualTo(42);
67  					assertThat(bean).extracting(TestPerson::getAddress).asString().isEqualTo("123 Main Street");
68  				});
69  	}
70  
71  	@Test
72  	public void objectPayloadWithCustomMapper() {
73  		ObjectMapper customMapper = new ObjectMapper();
74  		customMapper.configure(JsonParser.Feature.ALLOW_UNQUOTED_FIELD_NAMES, Boolean.TRUE);
75  		customMapper.configure(JsonParser.Feature.ALLOW_SINGLE_QUOTES, Boolean.TRUE);
76  		JsonToObjectTransformer transformer = new JsonToObjectTransformer(new Jackson2JsonObjectMapper(customMapper));
77  		transformer.setValueTypeExpression(new ValueExpression<>(ResolvableType.forClass(TestPerson.class)));
78  		String jsonString = "{firstName:'John', lastName:'Doe', age:42, address:{number:123, street:'Main Street'}}";
79  		Message<?> message = transformer.transform(new GenericMessage<>(jsonString));
80  		TestPerson person = (TestPerson) message.getPayload();
81  		assertThat(person.getFirstName()).isEqualTo("John");
82  		assertThat(person.getLastName()).isEqualTo("Doe");
83  		assertThat(person.getAge()).isEqualTo(42);
84  		assertThat(person.getAddress().toString()).isEqualTo("123 Main Street");
85  	}
