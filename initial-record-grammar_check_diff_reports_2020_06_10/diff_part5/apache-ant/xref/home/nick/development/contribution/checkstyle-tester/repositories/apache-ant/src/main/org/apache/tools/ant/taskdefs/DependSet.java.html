41  /**
42   * Examines and removes out of date target files.  If any of the target files
43   * are out of date with respect to any of the source files, all target
44   * files are removed.  This is useful where dependencies cannot be
45   * computed (for example, dynamically interpreted parameters or files
46   * that need to stay in synch but are not directly linked) or where
47   * the ant task in question could compute them but does not (for
48   * example, the linked DTD for an XML file using the XSLT task).
49   *
50   * nested arguments:
51   * <ul>
52   * <li>sources        (resource union describing the source resources to examine)
53   * <li>srcfileset     (fileset describing the source files to examine)
54   * <li>srcfilelist    (filelist describing the source files to examine)
55   * <li>targets        (path describing the target files to examine)
56   * <li>targetfileset  (fileset describing the target files to examine)
57   * <li>targetfilelist (filelist describing the target files to examine)
58   * </ul>
59   * At least one of both source and target entities is required.
60   * <p>
61   * This task will examine each of the sources against each of the target files. If
62   * any target files are out of date with respect to any of the sources, all targets
63   * are removed. If any sources or targets do not exist, all targets are removed.
64   * Hint: If missing files should be ignored, specify them as include patterns
65   * in filesets, rather than using filelists.
66   * </p><p>
67   * This task attempts to optimize speed of dependency checking
68   * by comparing only the dates of the oldest target file and the newest source.
69   * </p><p>
70   * Example uses:
71   * <ul><li>
72   * Record the fact that an XML file must be up to date with respect to its XSD
73   * (Schema file), even though the XML file itself includes no reference to its XSD.
74   * </li><li>
75   * Record the fact that an XSL stylesheet includes other sub-stylesheets
76   * </li><li>
77   * Record the fact that java files must be recompiled if the ant build file changes
78   * </li></ul>
79   *
80   * @ant.task category="filesystem"
81   * @since Ant 1.4
82   */
83  public class DependSet extends MatchingTask {
84  
85      private static final ResourceSelector NOT_EXISTS = new Not(new Exists());
86      private static final ResourceComparator DATE
87          = new org.apache.tools.ant.types.resources.comparators.Date();
88      private static final ResourceComparator REVERSE_DATE = new Reverse(DATE);
89  
90      private static final class NonExistent extends Restrict {
91          private NonExistent(ResourceCollection rc) {
92              super.add(rc);
93              super.add(NOT_EXISTS);
94          }
95      }
96  
97      private static final class HideMissingBasedir
98          implements ResourceCollection {
99          private FileSet fs;
100 
101         private HideMissingBasedir(FileSet fs) {
102             this.fs = fs;
103         }
104         public Iterator<Resource> iterator() {
105             return basedirExists() ? fs.iterator() : Resources.EMPTY_ITERATOR;
106         }
107         public int size() {
108             return basedirExists() ? fs.size() : 0;
109         }
110         public boolean isFilesystemOnly() {
111             return true;
112         }
113         private boolean basedirExists() {
114             File basedir = fs.getDir();
115             //trick to evoke "basedir not set" if null:
116             return basedir == null || basedir.exists();
117         }
118     }
119 
120     private Union sources = null;
121     private Path targets = null;
122 
123     private boolean verbose;
124 
125     /**
126      * Create a nested sources element.
127      * @return a Union instance.
128      */
129     public synchronized Union createSources() {
130         sources = (sources == null) ? new Union() : sources;
131         return sources;
132     }
133 
134     /**
135      * Add a set of source files.
136      * @param fs the FileSet to add.
137      */
138     public void addSrcfileset(FileSet fs) {
139         createSources().add(fs);
140     }
141 
142     /**
143      * Add a list of source files.
144      * @param fl the FileList to add.
145      */
146     public void addSrcfilelist(FileList fl) {
147         createSources().add(fl);
148     }
149 
150     /**
151      * Create a nested targets element.
152      * @return a Union instance.
153      */
154     public synchronized Path createTargets() {
155         targets = (targets == null) ? new Path(getProject()) : targets;
156         return targets;
157     }
158 
159     /**
160      * Add a set of target files.
161      * @param fs the FileSet to add.
162      */
163     public void addTargetfileset(FileSet fs) {
164         createTargets().add(new HideMissingBasedir(fs));
165     }
166 
167     /**
168      * Add a list of target files.
169      * @param fl the FileList to add.
170      */
171     public void addTargetfilelist(FileList fl) {
172         createTargets().add(fl);
173     }
174 
175     /**
176      * In verbose mode missing targets and sources as well as the
177      * modification times of the newest source and latest target will
178      * be logged as info.
179      *
180      * <p>All deleted files will be logged as well.</p>
181      *
182      * @since Ant 1.8.0
183      */
184     public void setVerbose(boolean b) {
185         verbose = b;
186     }
187 
188     /**
189      * Execute the task.
190      * @throws BuildException if errors occur.
191      */
192     public void execute() throws BuildException {
193         if (sources == null) {
194           throw new BuildException(
195               "At least one set of source resources must be specified");
196         }
197         if (targets == null) {
198           throw new BuildException(
199               "At least one set of target files must be specified");
200         }
201         //no sources = nothing to compare; no targets = nothing to delete:
202         if (sources.size() > 0 && targets.size() > 0 && !uptodate(sources, targets)) {
203            log("Deleting all target files.", Project.MSG_VERBOSE);
204            if (verbose) {
205                String[] t = targets.list();
206                for (int i = 0; i < t.length; i++) {
207                    log("Deleting " + t[i]);
208                }
209            }
210            Delete delete = new Delete();
211            delete.bindToOwner(this);
212            delete.add(targets);
213            delete.perform();
214         }
215     }
216 
217     private boolean uptodate(ResourceCollection src, ResourceCollection target) {
218         org.apache.tools.ant.types.resources.selectors.Date datesel
219             = new org.apache.tools.ant.types.resources.selectors.Date();
220         datesel.setMillis(System.currentTimeMillis());
221         datesel.setWhen(TimeComparison.AFTER);
222         // don't whine because a file has changed during the last
223         // second (or whathever our current granularity may be)
224         datesel.setGranularity(0);
225         logFuture(targets, datesel);
226 
227         NonExistent missingTargets = new NonExistent(targets);
228         int neTargets = missingTargets.size();
229         if (neTargets > 0) {
230             log(neTargets + " nonexistent targets", Project.MSG_VERBOSE);
231             logMissing(missingTargets, "target");
232             return false;
233         }
234         Resource oldestTarget = getOldest(targets);
235         logWithModificationTime(oldestTarget, "oldest target file");
236 
237         logFuture(sources, datesel);
238 
239         NonExistent missingSources = new NonExistent(sources);
240         int neSources = missingSources.size();
241         if (neSources > 0) {
242             log(neSources + " nonexistent sources", Project.MSG_VERBOSE);
243             logMissing(missingSources, "source");
244             return false;
245         }
246         Resource newestSource = (Resource) getNewest(sources);
247         logWithModificationTime(newestSource, "newest source");
248         return oldestTarget.getLastModified() >= newestSource.getLastModified();
249     }
250 
251     private void logFuture(ResourceCollection rc, ResourceSelector rsel) {
252         Restrict r = new Restrict();
253         r.add(rsel);
254         r.add(rc);
255         for (Resource res : r) {
256             log("Warning: " + res + " modified in the future.", Project.MSG_WARN);
257         }
258     }
259 
260     private Resource getXest(ResourceCollection rc, ResourceComparator c) {
261         Iterator<Resource> i = rc.iterator();
262         if (!i.hasNext()) {
263             return null;
264 
265         }
266         Resource xest = i.next();
267         while (i.hasNext()) {
268             Resource next = i.next();
269             if (c.compare(xest, next) < 0) {
270                 xest = next;
271             }
272         }
273         return xest;
274     }
275 
276     private Resource getOldest(ResourceCollection rc) {
277         return getXest(rc, REVERSE_DATE);
278     }
279 
280     private Resource getNewest(ResourceCollection rc) {
281         return getXest(rc, DATE);
282     }
283 
284     private void logWithModificationTime(Resource r, String what) {
285         log(r.toLongString() + " is " + what + ", modified at "
286             + new Date(r.getLastModified()),
287             verbose ? Project.MSG_INFO : Project.MSG_VERBOSE);
288     }
289 
290     private void logMissing(ResourceCollection missing, String what) {
291         if (verbose) {
292             for (Resource r : missing) {
293                 log("Expected " + what + " " + r.toLongString()
294                     + " is missing.");
295             }
296         }
297     }
