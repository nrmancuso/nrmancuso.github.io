36  /**
37   * Converts files from native encodings to ASCII.
38   *
39   * @since Ant 1.2
40   */
41  public class Native2Ascii extends MatchingTask {
42  
43      private boolean reverse = false;  // convert from ascii back to native
44      private String encoding = null;   // encoding to convert to/from
45      private File srcDir = null;       // Where to find input files
46      private File destDir = null;      // Where to put output files
47      private String extension = null;  // Extension of output files if different
48  
49      private Mapper mapper;
50      private FacadeTaskHelper facade = null;
51      private Native2AsciiAdapter nestedAdapter = null;
52  
53      /** No args constructor */
54      public Native2Ascii() {
55          facade = new FacadeTaskHelper(Native2AsciiAdapterFactory.getDefault());
56      }
57  
58      /**
59       * Flag the conversion to run in the reverse sense,
60       * that is Ascii to Native encoding.
61       *
62       * @param reverse True if the conversion is to be reversed,
63       *                otherwise false;
64       */
65      public void setReverse(boolean reverse) {
66          this.reverse = reverse;
67      }
68  
69      /**
70       * The value of the reverse attribute.
71       * @return the reverse attribute.
72       * @since Ant 1.6.3
73       */
74      public boolean getReverse() {
75          return reverse;
76      }

78      /**
79       * Set the encoding to translate to/from.
80       * If unset, the default encoding for the JVM is used.
81       *
82       * @param encoding String containing the name of the Native
83       *                 encoding to convert from or to.
84       */
85      public void setEncoding(String encoding) {
86          this.encoding = encoding;
87      }

90       * The value of the encoding attribute.

96      }

188     /**
189      * Set the adapter explicitly.
190      * @since Ant 1.8.0

192     public void add(Native2AsciiAdapter adapter) {
193         if (nestedAdapter != null) {
194             throw new BuildException("Can't have more than one native2ascii"

196         }
197         nestedAdapter = adapter;

200     /**
201      * Execute the task
202      *
203      * @throws BuildException is there is a problem in the task execution.
204      */
205     public void execute() throws BuildException {
206 
207         DirectoryScanner scanner = null; // Scanner to find our inputs
208         String[] files;                  // list of files to process
209 
210         // default srcDir to basedir
211         if (srcDir == null) {
212             srcDir = getProject().resolveFile(".");
213         }
214 
215         // Require destDir
216         if (destDir == null) {
217             throw new BuildException("The dest attribute must be set.");
218         }
219 
220         // if src and dest dirs are the same, require the extension
221         // to be set, so we don't stomp every file.  One could still
222         // include a file with the same extension, but ....
223         if (srcDir.equals(destDir) && extension == null && mapper == null) {
224             throw new BuildException("The ext attribute or a mapper must be set if"
225                                      + " src and dest dirs are the same.");
226         }
227 
228         FileNameMapper m = null;
229         if (mapper == null) {
230             if (extension == null) {
231                 m = new IdentityMapper();
232             } else {
233                 m = new ExtMapper();
234             }
235         } else {
236             m = mapper.getImplementation();
237         }
238 
239         scanner = getDirectoryScanner(srcDir);
240         files = scanner.getIncludedFiles();
241         SourceFileScanner sfs = new SourceFileScanner(this);
242         files = sfs.restrict(files, srcDir, destDir, m);
243         int count = files.length;
244         if (count == 0) {
245             return;
246         }
247         String message = "Converting " + count + " file"
248             + (count != 1 ? "s" : "") + " from ";
249         log(message + srcDir + " to " + destDir);
250         for (int i = 0; i < files.length; i++) {
251             convert(files[i], m.mapFileName(files[i])[0]);
252         }
253     }
254 
255     /**
256      * Convert a single file.
257      *
258      * @param srcName name of the input file.
259      * @param destName name of the input file.
260      */
261     private void convert(String srcName, String destName)
262         throws BuildException {
263         File srcFile;                         // File to convert
264         File destFile;                        // where to put the results
265 
266         // Build the full file names
267         srcFile = new File(srcDir, srcName);
268         destFile = new File(destDir, destName);
269 
270         // Make sure we're not about to clobber something
271         if (srcFile.equals(destFile)) {
272             throw new BuildException("file " + srcFile
273                                      + " would overwrite its self");
274         }
275 
276         // Make intermediate directories if needed
277         // TODO JDK 1.1 doesn't have File.getParentFile,
278         String parentName = destFile.getParent();
279         if (parentName != null) {
280             File parentFile = new File(parentName);
281 
282             if (!parentFile.exists()
283                 && !(parentFile.mkdirs() || parentFile.isDirectory())) {
284                 throw new BuildException("cannot create parent directory "
285                                          + parentName);
286             }
287         }
288 
289         log("converting " + srcName, Project.MSG_VERBOSE);
290         Native2AsciiAdapter ad =
291             nestedAdapter != null ? nestedAdapter :
292             Native2AsciiAdapterFactory.getAdapter(facade.getImplementation(),
293                                                   this,
294                                                   createImplementationClasspath());
295         if (!ad.convert(this, srcFile, destFile)) {
296             throw new BuildException("conversion failed");
297         }
298     }

309 
310     private class ExtMapper implements FileNameMapper {
311 
312         public void setFrom(String s) {
313         }
314         public void setTo(String s) {
315         }
316 
317         public String[] mapFileName(String fileName) {
318             int lastDot = fileName.lastIndexOf('.');
319             if (lastDot >= 0) {
320                 return new String[] {fileName.substring(0, lastDot)
321                                          + extension};
322             } else {
323                 return new String[] {fileName + extension};
324             }
325         }
326     }
