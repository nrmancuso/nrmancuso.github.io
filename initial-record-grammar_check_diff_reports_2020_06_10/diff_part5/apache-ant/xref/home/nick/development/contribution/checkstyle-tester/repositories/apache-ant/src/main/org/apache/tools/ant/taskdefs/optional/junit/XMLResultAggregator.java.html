44  /**
45   * Aggregates all &lt;junit&gt; XML formatter testsuite data under
46   * a specific directory and transforms the results via XSLT.
47   * It is not particularly clean but
48   * should be helpful while I am thinking about another technique.
49   *
50   * <p> The main problem is due to the fact that a JVM can be forked for a testcase
51   * thus making it impossible to aggregate all testcases since the listener is
52   * (obviously) in the forked JVM. A solution could be to write a
53   * TestListener that will receive events from the TestRunner via sockets. This
54   * is IMHO the simplest way to do it to avoid this file hacking thing.
55   *
56   * @ant.task name="junitreport" category="testing"
57   */
58  public class XMLResultAggregator extends Task implements XMLConstants {
59  
60      // CheckStyle:VisibilityModifier OFF - bc
61      /** the list of all filesets, that should contains the xml to aggregate */
62      protected Vector filesets = new Vector();
63  
64      /** the name of the result file */
65      protected String toFile;
66  
67      /** the directory to write the file to */
68      protected File toDir;
69  
70      protected Vector transformers = new Vector();
71  
72      /** The default directory: <tt>&#046;</tt>. It is resolved from the project directory */
73      public static final String DEFAULT_DIR = ".";
74  
75      /** the default file name: <tt>TESTS-TestSuites.xml</tt> */
76      public static final String DEFAULT_FILENAME = "TESTS-TestSuites.xml";
77  
78      /** the current generated id */
79      protected int generatedId = 0;
80  
81      /**
82       * text checked for in tests, {@value}
83       */
84      static final String WARNING_IS_POSSIBLY_CORRUPTED
85          = " is not a valid XML document. It is possibly corrupted.";
86      /**
87       * text checked for in tests, {@value}
88       */
89      static final String WARNING_INVALID_ROOT_ELEMENT
90          = " is not a valid testsuite XML document";
91      /**
92       * text checked for in tests, {@value}
93       */
94      static final String WARNING_EMPTY_FILE
95          = " is empty.\nThis can be caused by the test JVM exiting unexpectedly";
96      // CheckStyle:VisibilityModifier ON

98      /**
99       * Generate a report based on the document created by the merge.
100      * @return the report
101      */
102     public AggregateTransformer createReport() {
103         AggregateTransformer transformer = new AggregateTransformer(this);
104         transformers.addElement(transformer);
105         return transformer;
106     }

134         filesets.addElement(fs);

137     /**
138      * Aggregate all testsuites into a single document and write it to the
139      * specified directory and file.
140      * @throws  BuildException  thrown if there is a serious error while writing
141      *          the document.
142      */
143     public void execute() throws BuildException {
144         Element rootElement = createDocument();
145         File destFile = getDestinationFile();
146         // write the document
147         try {
148             writeDOMTree(rootElement.getOwnerDocument(), destFile);
149         } catch (IOException e) {
150             throw new BuildException("Unable to write test aggregate to '" + destFile + "'", e);
151         }
152         // apply transformation
153         Enumeration e = transformers.elements();
154         while (e.hasMoreElements()) {
155             AggregateTransformer transformer =
156                 (AggregateTransformer) e.nextElement();
157             transformer.setXmlDocument(rootElement.getOwnerDocument());
158             transformer.transform();
159         }
160     }
161 
162     /**
163      * Get the full destination file where to write the result. It is made of
164      * the <tt>todir</tt> and <tt>tofile</tt> attributes.
165      * @return the destination file where should be written the result file.
166      */
167     public File getDestinationFile() {
168         if (toFile == null) {
169             toFile = DEFAULT_FILENAME;
170         }
171         if (toDir == null) {
172             toDir = getProject().resolveFile(DEFAULT_DIR);
173         }
174         return new File(toDir, toFile);
175     }
176 
177     /**
178      * Get all <code>.xml</code> files in the fileset.
179      *
180      * @return all files in the fileset that end with a '.xml'.
181      */
182     protected File[] getFiles() {
183         Vector v = new Vector();
184         final int size = filesets.size();
185         for (int i = 0; i < size; i++) {
186             FileSet fs = (FileSet) filesets.elementAt(i);
187             DirectoryScanner ds = fs.getDirectoryScanner(getProject());
188             ds.scan();
189             String[] f = ds.getIncludedFiles();
190             for (int j = 0; j < f.length; j++) {
191                 String pathname = f[j];
192                 if (pathname.endsWith(".xml")) {
193                     File file = new File(ds.getBasedir(), pathname);
194                     file = getProject().resolveFile(file.getPath());
195                     v.addElement(file);
196                 }
197             }
198         }
199 
200         File[] files = new File[v.size()];
201         v.copyInto(files);
202         return files;
203     }

207     /**
208      * Write the DOM tree to a file.
209      * @param doc the XML document to dump to disk.
210      * @param file the filename to write the document to. Should obviously be a .xml file.
211      * @throws IOException thrown if there is an error while writing the content.
212      */
213     protected void writeDOMTree(Document doc, File file) throws IOException {
214         OutputStream os = new FileOutputStream(file);
215         try {
216             PrintWriter wri = new PrintWriter(new OutputStreamWriter(new BufferedOutputStream(os), "UTF8"));
217             wri.write("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n");
218             (new DOMElementWriter()).write(doc.getDocumentElement(), wri, 0, "  ");
219             wri.flush();
220             // writers do not throw exceptions, so check for them.
221             if (wri.checkError()) {
222                 throw new IOException("Error while writing DOM content");
223             }
224         } finally {
225             os.close();
226         }
227     }

229     /**
230      * <p> Create a DOM tree.
231      * Has 'testsuites' as firstchild and aggregates all
232      * testsuite results that exists in the base directory.
233      * @return  the root element of DOM tree that aggregates all testsuites.
234      */
235     protected Element createDocument() {
236         // create the dom tree
237         DocumentBuilder builder = getDocumentBuilder();
238         Document doc = builder.newDocument();
239         Element rootElement = doc.createElement(TESTSUITES);
240         doc.appendChild(rootElement);
241 
242         generatedId = 0;
243 
244         // get all files and add them to the document
245         File[] files = getFiles();
246         for (int i = 0; i < files.length; i++) {
247             File file = files[i];
248             try {
249                 log("Parsing file: '" + file + "'", Project.MSG_VERBOSE);
250                 if (file.length() > 0) {
251                     Document testsuiteDoc
252                             = builder.parse(
253                                 FileUtils.getFileUtils().toURI(files[i].getAbsolutePath()));
254                     Element elem = testsuiteDoc.getDocumentElement();
255                     // make sure that this is REALLY a testsuite.
256                     if (TESTSUITE.equals(elem.getNodeName())) {
257                         addTestSuite(rootElement, elem);
258                         generatedId++;
259                     } else {
260                         //wrong root element name
261                         // issue a warning.
262                         log("the file " + file
263                                 + WARNING_INVALID_ROOT_ELEMENT,
264                                 Project.MSG_WARN);
265                     }
266                 } else {
267                     log("the file " + file
268                             + WARNING_EMPTY_FILE,
269                             Project.MSG_WARN);
270                 }
271             } catch (SAXException e) {
272                 // a testcase might have failed and write a zero-length document,
273                 // It has already failed, but hey.... mm. just put a warning
274                 log("The file " + file + WARNING_IS_POSSIBLY_CORRUPTED, Project.MSG_WARN);
275                 log(StringUtils.getStackTrace(e), Project.MSG_DEBUG);
276             } catch (IOException e) {
277                 log("Error while accessing file " + file + ": "
278                     + e.getMessage(), Project.MSG_ERR);
279                 log("Error while accessing file " + file + ": "
280                     + e.getMessage(), e, Project.MSG_VERBOSE);
281             }
282         }
283         return rootElement;
284     }

286     /**
287      * <p> Add a new testsuite node to the document.
288      * The main difference is that it
289      * split the previous fully qualified name into a package and a name.
290      * <p> For example: <tt>org.apache.Whatever</tt> will be split into
291      * <tt>org.apache</tt> and <tt>Whatever</tt>.
292      * @param root the root element to which the <tt>testsuite</tt> node should
293      *        be appended.
294      * @param testsuite the element to append to the given root. It will slightly
295      *        modify the original node to change the name attribute and add
296      *        a package one.
297      */
298     protected void addTestSuite(Element root, Element testsuite) {
299         String fullclassname = testsuite.getAttribute(ATTR_NAME);
300         int pos = fullclassname.lastIndexOf('.');
301 
302         // a missing . might imply no package at all. Don't get fooled.
303         String pkgName = (pos == -1) ? "" : fullclassname.substring(0, pos);
304         String classname = (pos == -1) ? fullclassname : fullclassname.substring(pos + 1);
305         Element copy = (Element) DOMUtil.importNode(root, testsuite);
306 
307         // modify the name attribute and set the package
308         copy.setAttribute(ATTR_NAME, classname);
309         copy.setAttribute(ATTR_PACKAGE, pkgName);
310         copy.setAttribute(ATTR_ID, Integer.toString(generatedId));
311     }

313     /**
314      * Create a new document builder. Will issue an <tt>ExceptionInitializerError</tt>
315      * if something is going wrong. It is fatal anyway.
316      * @todo factorize this somewhere else. It is duplicated code.
317      * @return a new document builder to create a DOM
318      */
319     private static DocumentBuilder getDocumentBuilder() {
320         try {
321             return DocumentBuilderFactory.newInstance().newDocumentBuilder();
322         } catch (Exception exc) {
323             throw new ExceptionInInitializerError(exc);
324         }
325     }
