296      * Check the configuration of this ExecuteOn instance.
297      */
298     protected void checkConfiguration() {
299 //     * @TODO using taskName here is brittle, as a user could override it.
300 //     *       this should probably be modified to use the classname instead.
301         if ("execon".equals(getTaskName())) {
302             log("!! execon is deprecated. Use apply instead. !!");
303         }
304         super.checkConfiguration();
305         if (filesets.size() == 0 && resources == null) {
306             throw new BuildException("no resources specified",
307                                      getLocation());
308         }
309         if (targetFilePos != null && mapperElement == null) {
310             throw new BuildException("targetfile specified without mapper",
311                                      getLocation());
312         }
313         if (destDir != null && mapperElement == null) {
314             throw new BuildException("dest specified without mapper",
315                                      getLocation());
316         }
317         if (mapperElement != null) {
318             mapper = mapperElement.getImplementation();
319         }
320     }

342     /**
343      * Run the specified Execute object.
344      * @param exe the Execute instance representing the external process.
345      * @throws BuildException on error
346      */
347     protected void runExec(Execute exe) throws BuildException {
348         int totalFiles = 0;
349         int totalDirs = 0;
350         boolean haveExecuted = false;
351         try {
352             Vector<String> fileNames = new Vector<String>();
353             Vector<File> baseDirs = new Vector<File>();
354             final int size = filesets.size();
355             for (int i = 0; i < size; i++) {
356                 String currentType = type;
357                 AbstractFileSet fs = filesets.elementAt(i);
358                 if (fs instanceof DirSet) {
359                     if (!FileDirBoth.DIR.equals(type)) {
360                         log("Found a nested dirset but type is " + type + ". "
361                             + "Temporarily switching to type=\"dir\" on the"
362                             + " assumption that you really did mean"
363                             + " <dirset> not <fileset>.", Project.MSG_DEBUG);
364                         currentType = FileDirBoth.DIR;
365                     }
366                 }
367                 File base = fs.getDir(getProject());
368 
369                 DirectoryScanner ds = fs.getDirectoryScanner(getProject());
370 
371                 if (!FileDirBoth.DIR.equals(currentType)) {
372                     String[] s = getFiles(base, ds);
373                     for (int j = 0; j < s.length; j++) {
374                         totalFiles++;
375                         fileNames.addElement(s[j]);
376                         baseDirs.addElement(base);
377                     }
378                 }
379                 if (!FileDirBoth.FILE.equals(currentType)) {
380                     String[] s = getDirs(base, ds);
381                     for (int j = 0; j < s.length; j++) {
382                         totalDirs++;
383                         fileNames.addElement(s[j]);
384                         baseDirs.addElement(base);
385                     }
386                 }
387                 if (fileNames.size() == 0 && skipEmpty) {
388                     logSkippingFileset(currentType, ds, base);
389                     continue;
390                 }
391                 if (!parallel) {
392                     String[] s = new String[fileNames.size()];
393                     fileNames.copyInto(s);
394                     for (int j = 0; j < s.length; j++) {
395                         String[] command = getCommandline(s[j], base);
396                         log(Commandline.describeCommand(command),
397                             Project.MSG_VERBOSE);
398                         exe.setCommandline(command);
399 
400                         if (redirectorElement != null) {
401                             setupRedirector();
402                             redirectorElement.configure(redirector, s[j]);
403                         }
404                         if (redirectorElement != null || haveExecuted) {
405                             // need to reset the stream handler to restart
406                             // reading of pipes;
407                             // go ahead and do it always w/ nested redirectors
408                             exe.setStreamHandler(redirector.createHandler());
409                         }
410                         runExecute(exe);
411                         haveExecuted = true;
412                     }
413                     fileNames.removeAllElements();
414                     baseDirs.removeAllElements();
415                 }
416             }
417 
418             if (resources != null) {
419                 for (Resource res : resources) {
420 
421                     if (!res.isExists() && ignoreMissing) {
422                         continue;
423                     }
424 
425                     File base = null;
426                     String name = res.getName();
427                     FileProvider fp = res.as(FileProvider.class);
428                     if (fp != null) {
429                         FileResource fr = ResourceUtils.asFileResource(fp);
430                         base = fr.getBaseDir();
431                         if (base == null) {
432                             name = fr.getFile().getAbsolutePath();
433                         }
434                     }
435 
436                     if (restrict(new String[] {name}, base).length == 0) {
437                         continue;
438                     }
439 
440                     if ((!res.isDirectory() || !res.isExists())
441                         && !FileDirBoth.DIR.equals(type)) {
442                         totalFiles++;
443                     } else if (res.isDirectory()
444                                && !FileDirBoth.FILE.equals(type)) {
445                         totalDirs++;
446                     } else {
447                         continue;
448                     }
449 
450                     baseDirs.add(base);
451                     fileNames.add(name);
452 
453                     if (!parallel) {
454                         String[] command = getCommandline(name, base);
455                         log(Commandline.describeCommand(command),
456                             Project.MSG_VERBOSE);
457                         exe.setCommandline(command);
458 
459                         if (redirectorElement != null) {
460                             setupRedirector();
461                             redirectorElement.configure(redirector, name);
462                         }
463                         if (redirectorElement != null || haveExecuted) {
464                             // need to reset the stream handler to restart
465                             // reading of pipes;
466                             // go ahead and do it always w/ nested redirectors
467                             exe.setStreamHandler(redirector.createHandler());
468                         }
469                         runExecute(exe);
470                         haveExecuted = true;
471                         fileNames.removeAllElements();
472                         baseDirs.removeAllElements();
473                     }
474                 }
475             }
476             if (parallel && (fileNames.size() > 0 || !skipEmpty)) {
477                 runParallel(exe, fileNames, baseDirs);
478                 haveExecuted = true;
479             }
480             if (haveExecuted) {
481                 log("Applied " + cmdl.getExecutable() + " to "
482                     + totalFiles + " file"
483                     + (totalFiles != 1 ? "s" : "") + " and "
484                     + totalDirs + " director"
485                     + (totalDirs != 1 ? "ies" : "y") + ".",
486                     verbose ? Project.MSG_INFO : Project.MSG_VERBOSE);
487             }
488         } catch (IOException e) {
489             throw new BuildException("Execute failed: " + e, e, getLocation());
490         } finally {
491             // close the output file if required
492             logFlush();
493             redirector.setAppendProperties(false);
494             redirector.setProperties();
495         }
496     }

498     /**
499      * log a message for skipping a fileset.
500      * @param currentType the current type.
501      * @param ds the directory scanner.
502      * @param base the dir base
503      */
504     private void logSkippingFileset(
505         String currentType, DirectoryScanner ds, File base) {
506         int includedCount
507             = ((!FileDirBoth.DIR.equals(currentType))
508                ? ds.getIncludedFilesCount() : 0)
509             + ((!FileDirBoth.FILE.equals(currentType))
510                ? ds.getIncludedDirsCount() : 0);
511 
512         log("Skipping fileset for directory " + base + ". It is "
513             + ((includedCount > 0) ? "up to date." : "empty."),
514              verbose ? Project.MSG_INFO : Project.MSG_VERBOSE);
515     }

517     /**
518      * Construct the command line for parallel execution.
519      *
520      * @param srcFiles The filenames to add to the commandline.
521      * @param baseDirs filenames are relative to this dir.
522      * @return the command line in the form of a String[].
523      */
524     protected String[] getCommandline(String[] srcFiles, File[] baseDirs) {
525         final char fileSeparator = File.separatorChar;
526         Vector<String> targets = new Vector<String>();
527         if (targetFilePos != null) {
528             HashSet<String> addedFiles = new HashSet<String>();
529             for (int i = 0; i < srcFiles.length; i++) {
530                 String[] subTargets = mapper.mapFileName(srcFiles[i]);
531                 if (subTargets != null) {
532                     for (int j = 0; j < subTargets.length; j++) {
533                         String name = null;
534                         if (!relative) {
535                             name = new File(destDir, subTargets[j]).getAbsolutePath();
536                         } else {
537                             name = subTargets[j];
538                         }
539                         if (forwardSlash && fileSeparator != '/') {
540                             name = name.replace(fileSeparator, '/');
541                         }
542                         if (!addedFiles.contains(name)) {
543                             targets.addElement(name);
544                             addedFiles.add(name);
545                         }
546                     }
547                 }
548             }
549         }
550         String[] targetFiles = (String[]) targets.toArray(new String[targets.size()]);
551 
552         if (!addSourceFile) {
553             srcFiles = new String[0];
554         }
555         String[] orig = cmdl.getCommandline();
556         String[] result
557             = new String[orig.length + srcFiles.length + targetFiles.length];
558 
559         int srcIndex = orig.length;
560         if (srcFilePos != null) {
561             srcIndex = srcFilePos.getPosition();
562         }
563         if (targetFilePos != null) {
564             int targetIndex = targetFilePos.getPosition();
565 
566             if (srcIndex < targetIndex
567                 || (srcIndex == targetIndex && srcIsFirst)) {
568 
569                 // 0 --> srcIndex
570                 System.arraycopy(orig, 0, result, 0, srcIndex);
571 
572                 // srcIndex --> targetIndex
573                 System.arraycopy(orig, srcIndex, result,
574                                  srcIndex + srcFiles.length,
575                                  targetIndex - srcIndex);
576 
577                 insertTargetFiles(targetFiles, result,
578                                   targetIndex + srcFiles.length,
579                                   targetFilePos.getPrefix(),
580                                   targetFilePos.getSuffix());
581 
582                 // targetIndex --> end
583                 System.arraycopy(orig, targetIndex, result,
584                     targetIndex + srcFiles.length + targetFiles.length,
585                     orig.length - targetIndex);
586             } else {
587                 // 0 --> targetIndex
588                 System.arraycopy(orig, 0, result, 0, targetIndex);
589 
590                 insertTargetFiles(targetFiles, result, targetIndex,
591                                   targetFilePos.getPrefix(),
592                                   targetFilePos.getSuffix());
593 
594                 // targetIndex --> srcIndex
595                 System.arraycopy(orig, targetIndex, result,
596                                  targetIndex + targetFiles.length,
597                                  srcIndex - targetIndex);
598 
599                 // srcIndex --> end
600                 System.arraycopy(orig, srcIndex, result,
601                     srcIndex + srcFiles.length + targetFiles.length,
602                     orig.length - srcIndex);
603                 srcIndex += targetFiles.length;
604             }
605 
606         } else { // no targetFilePos
607 
608             // 0 --> srcIndex
609             System.arraycopy(orig, 0, result, 0, srcIndex);
610             // srcIndex --> end
611             System.arraycopy(orig, srcIndex, result,
612                              srcIndex + srcFiles.length,
613                              orig.length - srcIndex);
614         }
615         // fill in source file names
616         for (int i = 0; i < srcFiles.length; i++) {
617             String src;
618             if (relative) {
619                 src = srcFiles[i];
620             } else {
621                 src = new File(baseDirs[i], srcFiles[i]).getAbsolutePath();
622             }
623             if (forwardSlash && fileSeparator != '/') {
624                 src = src.replace(fileSeparator, '/');
625             }
626             if (srcFilePos != null &&
627                 (srcFilePos.getPrefix().length() > 0
628                  || srcFilePos.getSuffix().length() > 0)) {
629                 src = srcFilePos.getPrefix() + src + srcFilePos.getSuffix();
630             }
631             result[srcIndex + i] = src;
632         }
633         return result;
634     }

686     /**
687      * Run the command in "parallel" mode, making sure that at most
688      * maxParallel sourcefiles get passed on the command line.
689      * @param exe the Executable to use.
690      * @param fileNames the Vector of filenames.
691      * @param baseDirs the Vector of base directories corresponding to fileNames.
692      * @throws IOException  on I/O errors.
693      * @throws BuildException on other errors.
694      * @since Ant 1.6
695      */
696     protected void runParallel(Execute exe, Vector<String> fileNames,
697                                Vector<File> baseDirs)
698         throws IOException, BuildException {
699         String[] s = new String[fileNames.size()];
700         fileNames.copyInto(s);
701         File[] b = new File[baseDirs.size()];
702         baseDirs.copyInto(b);
703 
704         if (maxParallel <= 0
705             || s.length == 0 /* this is skipEmpty == false */) {
706             String[] command = getCommandline(s, b);
707             log(Commandline.describeCommand(command), Project.MSG_VERBOSE);
708             exe.setCommandline(command);
709             if (redirectorElement != null) {
710                 setupRedirector();
711                 redirectorElement.configure(redirector, null);
712                 exe.setStreamHandler(redirector.createHandler());
713             }
714             runExecute(exe);
715         } else {
716             int stillToDo = fileNames.size();
717             int currentOffset = 0;
718             while (stillToDo > 0) {
719                 int currentAmount = Math.min(stillToDo, maxParallel);
720                 String[] cs = new String[currentAmount];
721                 System.arraycopy(s, currentOffset, cs, 0, currentAmount);
722                 File[] cb = new File[currentAmount];
723                 System.arraycopy(b, currentOffset, cb, 0, currentAmount);
724                 String[] command = getCommandline(cs, cb);
725                 log(Commandline.describeCommand(command), Project.MSG_VERBOSE);
726                 exe.setCommandline(command);
727                 if (redirectorElement != null) {
728                     setupRedirector();
729                     redirectorElement.configure(redirector, null);
730                 }
731                 if (redirectorElement != null || currentOffset > 0) {
732                     // need to reset the stream handler to restart
733                     // reading of pipes;
734                     // go ahead and do it always w/ nested redirectors
735                     exe.setStreamHandler(redirector.createHandler());
736                 }
737                 runExecute(exe);
738 
739                 stillToDo -= currentAmount;
740                 currentOffset += currentAmount;
741             }
742         }
743     }

745     /**
746      * Inserts target file names (which are already absolute paths)
747      * into the list of arguments, taking prefix and postfix into
748      * account.
749      */
750     private static void insertTargetFiles(String[] targetFiles,
751                                           String[] arguments,
752                                           int insertPosition,
753                                           String prefix, String suffix) {
754         if (prefix.length() == 0 && suffix.length() == 0) {
755             System.arraycopy(targetFiles, 0, arguments, insertPosition,
756                              targetFiles.length);
757         } else {
758             for (int i = 0; i < targetFiles.length; i++) {
759                 arguments[insertPosition + i] =
760                     prefix + targetFiles[i] + suffix;
761             }
762         }
763     }
