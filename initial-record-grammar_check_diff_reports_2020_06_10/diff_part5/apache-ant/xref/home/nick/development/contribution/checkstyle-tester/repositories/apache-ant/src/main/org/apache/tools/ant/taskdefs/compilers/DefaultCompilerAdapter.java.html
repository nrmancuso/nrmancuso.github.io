150     /**
151      * Builds the compilation classpath.
152      * @return the compilation class path
153      */
154     protected Path getCompileClasspath() {
155         Path classpath = new Path(project);
156 
157         // add dest dir to classpath so that previously compiled and
158         // untouched classes are on classpath
159 
160         if (destDir != null && getJavac().isIncludeDestClasses()) {
161             classpath.setLocation(destDir);
162         }
163 
164         // Combine the build classpath with the system classpath, in an
165         // order determined by the value of build.sysclasspath
166 
167         Path cp = compileClasspath;
168         if (cp == null) {
169             cp = new Path(project);
170         }
171         if (includeAntRuntime) {
172             classpath.addExisting(cp.concatSystemClasspath("last"));
173         } else {
174             classpath.addExisting(cp.concatSystemClasspath("ignore"));
175         }
176 
177         if (includeJavaRuntime) {
178             classpath.addJavaRuntime();
179         }
180 
181         return classpath;
182     }

193     /**
194      * Does the command line argument processing common to classic and
195      * modern.  Doesn't add the files to compile.
196      * @param cmd the command line
197      * @param useDebugLevel if true set set the debug level with the -g switch
198      * @return the command line
199      */
200     protected Commandline setupJavacCommandlineSwitches(Commandline cmd,
201                                                         boolean useDebugLevel) {
202         Path classpath = getCompileClasspath();
203         // For -sourcepath, use the "sourcepath" value if present.
204         // Otherwise default to the "srcdir" value.
205         Path sourcepath = null;
206         if (compileSourcepath != null) {
207             sourcepath = compileSourcepath;
208         } else {
209             sourcepath = src;
210         }
211 
212         String memoryParameterPrefix = assumeJava11() ? "-J-" : "-J-X";
213         if (memoryInitialSize != null) {
214             if (!attributes.isForkedJavac()) {
215                 attributes.log("Since fork is false, ignoring "
216                                + "memoryInitialSize setting.",
217                                Project.MSG_WARN);
218             } else {
219                 cmd.createArgument().setValue(memoryParameterPrefix
220                                               + "ms" + memoryInitialSize);
221             }
222         }
223 
224         if (memoryMaximumSize != null) {
225             if (!attributes.isForkedJavac()) {
226                 attributes.log("Since fork is false, ignoring "
227                                + "memoryMaximumSize setting.",
228                                Project.MSG_WARN);
229             } else {
230                 cmd.createArgument().setValue(memoryParameterPrefix
231                                               + "mx" + memoryMaximumSize);
232             }
233         }
234 
235         if (attributes.getNowarn()) {
236             cmd.createArgument().setValue("-nowarn");
237         }
238 
239         if (deprecation) {
240             cmd.createArgument().setValue("-deprecation");
241         }
242 
243         if (destDir != null) {
244             cmd.createArgument().setValue("-d");
245             cmd.createArgument().setFile(destDir);
246         }
247 
248         cmd.createArgument().setValue("-classpath");
249 
250         // Just add "sourcepath" to classpath ( for JDK1.1 )
251         // as well as "bootclasspath" and "extdirs"
252         if (assumeJava11()) {
253             Path cp = new Path(project);
254 
255             Path bp = getBootClassPath();
256             if (bp.size() > 0) {
257                 cp.append(bp);
258             }
259 
260             if (extdirs != null) {
261                 cp.addExtdirs(extdirs);
262             }
263             cp.append(classpath);
264             cp.append(sourcepath);
265             cmd.createArgument().setPath(cp);
266         } else {
267             cmd.createArgument().setPath(classpath);
268             // If the buildfile specifies sourcepath="", then don't
269             // output any sourcepath.
270             if (sourcepath.size() > 0) {
271                 cmd.createArgument().setValue("-sourcepath");
272                 cmd.createArgument().setPath(sourcepath);
273             }
274             if (target != null) {
275                 cmd.createArgument().setValue("-target");
276                 cmd.createArgument().setValue(target);
277             }
278 
279             Path bp = getBootClassPath();
280             if (bp.size() > 0) {
281                 cmd.createArgument().setValue("-bootclasspath");
282                 cmd.createArgument().setPath(bp);
283             }
284 
285             if (extdirs != null && extdirs.size() > 0) {
286                 cmd.createArgument().setValue("-extdirs");
287                 cmd.createArgument().setPath(extdirs);
288             }
289         }
290 
291         if (encoding != null) {
292             cmd.createArgument().setValue("-encoding");
293             cmd.createArgument().setValue(encoding);
294         }
295         if (debug) {
296             if (useDebugLevel && !assumeJava11()) {
297                 String debugLevel = attributes.getDebugLevel();
298                 if (debugLevel != null) {
299                     cmd.createArgument().setValue("-g:" + debugLevel);
300                 } else {
301                     cmd.createArgument().setValue("-g");
302                 }
303             } else {
304                 cmd.createArgument().setValue("-g");
305             }
306         } else if (getNoDebugArgument() != null) {
307             cmd.createArgument().setValue(getNoDebugArgument());
308         }
309         if (optimize) {
310             cmd.createArgument().setValue("-O");
311         }
312 
313         if (depend) {
314             if (assumeJava11()) {
315                 cmd.createArgument().setValue("-depend");
316             } else if (assumeJava12()) {
317                 cmd.createArgument().setValue("-Xdepend");
318             } else {
319                 attributes.log("depend attribute is not supported by the "
320                                + "modern compiler", Project.MSG_WARN);
321             }
322         }
323 
324         if (verbose) {
325             cmd.createArgument().setValue("-verbose");
326         }
327 
328         addCurrentCompilerArgs(cmd);
329 
330         return cmd;
331     }

333     /**
334      * Does the command line argument processing for modern.  Doesn't
335      * add the files to compile.
336      * @param cmd the command line
337      * @return the command line
338      */
339     protected Commandline setupModernJavacCommandlineSwitches(Commandline cmd) {
340         setupJavacCommandlineSwitches(cmd, true);
341         if (!assumeJava13()) { // -source added with JDK 1.4
342             final String t = attributes.getTarget();
343             if (attributes.getSource() != null) {
344                 cmd.createArgument().setValue("-source");
345                 cmd.createArgument()
346                     .setValue(adjustSourceValue(attributes.getSource()));
347 
348             } else if (t != null && mustSetSourceForTarget(t)) {
349                 setImplicitSourceSwitch(cmd, t, adjustSourceValue(t));
350             }
351         }
352         return cmd;
353     }

389     /**
390      * Logs the compilation parameters, adds the files to compile and logs the
391      * &quot;niceSourceList&quot;
392      * @param cmd the command line
393      */
394     protected void logAndAddFilesToCompile(Commandline cmd) {
395         attributes.log("Compilation " + cmd.describeArguments(),
396                        Project.MSG_VERBOSE);
397 
398         StringBuffer niceSourceList = new StringBuffer("File");
399         if (compileList.length != 1) {
400             niceSourceList.append("s");
401         }
402         niceSourceList.append(" to be compiled:");
403 
404         niceSourceList.append(StringUtils.LINE_SEP);
405 
406         for (int i = 0; i < compileList.length; i++) {
407             String arg = compileList[i].getAbsolutePath();
408             cmd.createArgument().setValue(arg);
409             niceSourceList.append("    ");
410             niceSourceList.append(arg);
411             niceSourceList.append(StringUtils.LINE_SEP);
412         }
413 
414         attributes.log(niceSourceList.toString(), Project.MSG_VERBOSE);
415     }

430     /**
431      * Do the compile with the specified arguments.
432      *
433      * <p>The working directory if the executed process will be the
434      * project's base directory.</p>
435      *
436      * @param args - arguments to pass to process on command line
437      * @param firstFileName - index of the first source file in args,
438      * if the index is negative, no temporary file will ever be
439      * created, but this may hit the command line length limit on your
440      * system.
441      * @param quoteFiles - if set to true, filenames containing
442      * spaces will be quoted when they appear in the external file.
443      * This is necessary when running JDK 1.4's javac and probably
444      * others.
445      * @return the exit code of the compilation
446      *
447      * @since Ant 1.6
448      */
449     protected int executeExternalCompile(String[] args, int firstFileName,
450                                          boolean quoteFiles) {
451         String[] commandArray = null;
452         File tmpFile = null;
453 
454         try {
455             /*
456              * Many system have been reported to get into trouble with
457              * long command lines - no, not only Windows ;-).
458              *
459              * POSIX seems to define a lower limit of 4k, so use a temporary
460              * file if the total length of the command line exceeds this limit.
461              */
462             if (Commandline.toString(args).length() > COMMAND_LINE_LIMIT
463                 && firstFileName >= 0) {
464                 BufferedWriter out = null;
465                 try {
466                     tmpFile = FILE_UTILS.createTempFile(
467                         "files", "", getJavac().getTempdir(), true, true);
468                     out = new BufferedWriter(new FileWriter(tmpFile));
469                     for (int i = firstFileName; i < args.length; i++) {
470                         if (quoteFiles && args[i].indexOf(" ") > -1) {
471                             args[i] = args[i].replace(File.separatorChar, '/');
472                             out.write("\"" + args[i] + "\"");
473                         } else {
474                             out.write(args[i]);
475                         }
476                         out.newLine();
477                     }
478                     out.flush();
479                     commandArray = new String[firstFileName + 1];
480                     System.arraycopy(args, 0, commandArray, 0, firstFileName);
481                     commandArray[firstFileName] = "@" + tmpFile;
482                 } catch (IOException e) {
483                     throw new BuildException("Error creating temporary file",
484                                              e, location);
485                 } finally {
486                     FileUtils.close(out);
487                 }
488             } else {
489                 commandArray = args;
490             }
491 
492             try {
493                 Execute exe = new Execute(
494                                   new LogStreamHandler(attributes,
495                                                        Project.MSG_INFO,
496                                                        Project.MSG_WARN));
497                 if (Os.isFamily("openvms")) {
498                     //Use the VM launcher instead of shell launcher on VMS
499                     //for java
500                     exe.setVMLauncher(true);
501                 }
502                 exe.setAntRun(project);
503                 exe.setWorkingDirectory(project.getBaseDir());
504                 exe.setCommandline(commandArray);
505                 exe.execute();
506                 return exe.getExitValue();
507             } catch (IOException e) {
508                 throw new BuildException("Error running " + args[0]
509                         + " compiler", e, location);
510             }
511         } finally {
512             if (tmpFile != null) {
513                 tmpFile.delete();
514             }
515         }
516     }

617     /**
618      * Shall we assume command line switches for the given version of Java?
619      * @since Ant 1.8.3
620      */
621     private boolean assumeJavaXY(String javacXY, String javaEnvVersionXY) {
622         return javacXY.equals(attributes.getCompilerVersion())
623             || ("classic".equals(attributes.getCompilerVersion())
624                 && JavaEnvUtils.isJavaVersion(javaEnvVersionXY))
625             || ("modern".equals(attributes.getCompilerVersion())
626                 && JavaEnvUtils.isJavaVersion(javaEnvVersionXY))
627             || ("extJavac".equals(attributes.getCompilerVersion())
628                 && JavaEnvUtils.isJavaVersion(javaEnvVersionXY));
629     }

660     private void setImplicitSourceSwitch(Commandline cmd,
661                                          String target, String source) {
662         attributes.log("", Project.MSG_WARN);
663         attributes.log("          WARNING", Project.MSG_WARN);
664         attributes.log("", Project.MSG_WARN);
665         attributes.log("The -source switch defaults to " + getDefaultSource()
666                        + ".",
667                        Project.MSG_WARN);
668         attributes.log("If you specify -target " + target
669                        + " you now must also specify -source " + source
670                        + ".", Project.MSG_WARN);
671         attributes.log("Ant will implicitly add -source " + source
672                        + " for you.  Please change your build file.",
673                        Project.MSG_WARN);
674         cmd.createArgument().setValue("-source");
675         cmd.createArgument().setValue(source);
676     }

678     /**
679      * A string that describes the default value for -source of the
680      * selected JDK's javac.
681      */
682     private String getDefaultSource() {
683         if (assumeJava15() || assumeJava16()) {
684             return "1.5 in JDK 1.5 and 1.6";
685         }
686         if (assumeJava17()) {
687             return "1.7 in JDK 1.7";
688         }
689         if (assumeJava18()) {
690             return "1.8 in JDK 1.8";
691         }
692         return "";
693     }

695     /**
696      * Whether the selected -target is known to be incompatible with
697      * the default -source value of the selected JDK's javac.
698      *
699      * <p>Assumes it will never be called unless the selected JDK is
700      * at least Java 1.5.</p>
701      *
702      * @param t the -target value, must not be null
703      */
704     private boolean mustSetSourceForTarget(String t) {
705         if (assumeJava14()) {
706             return false;
707         }
708         if (t.startsWith("1.")) {
709             t = t.substring(2);
710         }
711         return t.equals("1") || t.equals("2") || t.equals("3") || t.equals("4")
712             || ((t.equals("5") || t.equals("6"))
713                 && !assumeJava15() && !assumeJava16())
714             || (t.equals("7") && !assumeJava17());
715     }

718     /**
719      * Turn the task's attribute for -source into soemthing that is
720      * understood by all javac's after 1.4.
721      *
722      * <p>support for -source 1.1 and -source 1.2 has been added with
723      * JDK 1.4.2 but isn't present in 1.5.0+</p>
724      */
725     private String adjustSourceValue(String source) {
726         return (source.equals("1.1") || source.equals("1.2")) ? "1.3" : source;
727     }
