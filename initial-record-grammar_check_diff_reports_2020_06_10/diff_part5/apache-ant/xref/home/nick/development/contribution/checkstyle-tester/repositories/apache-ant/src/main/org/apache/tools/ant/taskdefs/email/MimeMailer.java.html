59  /**
60   * Uses the JavaMail classes to send Mime format email.
61   *
62   * @since Ant 1.5
63   */
64  public class MimeMailer extends Mailer {
65      private static final String SSL_FACTORY = "javax.net.ssl.SSLSocketFactory";
66  
67      private static final String GENERIC_ERROR =
68          "Problem while sending mime mail:";
69  
70      /** Default character set */
71      private static final String DEFAULT_CHARSET
72          = System.getProperty("file.encoding");
73  
74      // To work properly with national charsets we have to use
75      // implementation of interface javax.activation.DataSource
76      /**
77       * String data source implementation.
78       * @since Ant 1.6
79       */
80      class StringDataSource implements javax.activation.DataSource {
81          private String data = null;
82          private String type = null;
83          private String charset = null;
84          private ByteArrayOutputStream out;
85  
86          public InputStream getInputStream() throws IOException {
87              if (data == null && out == null) {
88                  throw new IOException("No data");
89              }
90              if (out != null) {
91                  String encodedOut = out.toString(charset);
92                  data = (data != null) ? data.concat(encodedOut) : encodedOut;
93                  out = null;
94              }
95              return new ByteArrayInputStream(data.getBytes(charset));
96          }
97  
98          public OutputStream getOutputStream() throws IOException {
99              out = (out == null) ? new ByteArrayOutputStream() : out;
100             return out;
101         }
102 
103         public void setContentType(String type) {
104             this.type = type.toLowerCase(Locale.ENGLISH);
105         }
106 
107         public String getContentType() {
108             if (type != null && type.indexOf("charset") > 0
109                 && type.startsWith("text/")) {
110                 return type;
111             }
112             // Must be like "text/plain; charset=windows-1251"
113             return new StringBuffer(type != null ? type : "text/plain").append(
114                 "; charset=").append(charset).toString();
115         }
116 
117         public String getName() {
118             return "StringDataSource";
119         }
120 
121         public void setCharset(String charset) {
122             this.charset = charset;
123         }
124 
125         public String getCharset() {
126             return charset;
127         }
128     }

130     /**
131      * Send the email.
132      *
133      * @throws BuildException if the email can't be sent.
134      */
135     public void send() {
136         try {
137             Properties props = new Properties();
138 
139             props.put("mail.smtp.host", host);
140             props.put("mail.smtp.port", String.valueOf(port));
141 
142             // Aside, the JDK is clearly unaware of the Scottish
143             // 'session', which involves excessive quantities of
144             // alcohol :-)
145             Session sesh;
146             Authenticator auth = null;
147             if (SSL) {
148                 try {
149                     Provider p = (Provider) Class.forName(
150                         "com.sun.net.ssl.internal.ssl.Provider").newInstance();
151                     Security.addProvider(p);
152                 } catch (Exception e) {
153                     throw new BuildException("could not instantiate ssl "
154                         + "security provider, check that you have JSSE in "
155                         + "your classpath");
156                 }
157                 // SMTP provider
158                 props.put("mail.smtp.socketFactory.class", SSL_FACTORY);
159                 props.put("mail.smtp.socketFactory.fallback", "false");
160                 props.put("mail.smtps.host", host);
161                 if (isPortExplicitlySpecified()) {
162                     props.put("mail.smtps.port", String.valueOf(port));
163                     props.put("mail.smtp.socketFactory.port",
164                               String.valueOf(port));
165                 }
166             }
167             if (user != null || password != null) {
168                 props.put("mail.smtp.auth", "true");
169                 auth = new SimpleAuthenticator(user, password);
170             }
171             if (isStartTLSEnabled()) {
172                 props.put("mail.smtp.starttls.enable", "true");
173             }
174             sesh = Session.getInstance(props, auth);
175 
176             //create the message
177             MimeMessage msg = new MimeMessage(sesh);
178             MimeMultipart attachments = new MimeMultipart();
179 
180             //set the sender
181             if (from.getName() == null) {
182                 msg.setFrom(new InternetAddress(from.getAddress()));
183             } else {
184                 msg.setFrom(new InternetAddress(from.getAddress(),
185                     from.getName()));
186             }
187             // set the reply to addresses
188             msg.setReplyTo(internetAddresses(replyToList));
189             msg.setRecipients(Message.RecipientType.TO,
190                 internetAddresses(toList));
191             msg.setRecipients(Message.RecipientType.CC,
192                 internetAddresses(ccList));
193             msg.setRecipients(Message.RecipientType.BCC,
194                 internetAddresses(bccList));
195 
196             // Choosing character set of the mail message
197             // First: looking it from MimeType
198             String charset = parseCharSetFromMimeType(message.getMimeType());
199             if (charset != null) {
200                 // Assign/reassign message charset from MimeType
201                 message.setCharset(charset);
202             } else {
203                 // Next: looking if charset having explicit definition
204                 charset = message.getCharset();
205                 if (charset == null) {
206                     // Using default
207                     charset = DEFAULT_CHARSET;
208                     message.setCharset(charset);
209                 }
210             }
211             // Using javax.activation.DataSource paradigm
212             StringDataSource sds = new StringDataSource();
213             sds.setContentType(message.getMimeType());
214             sds.setCharset(charset);
215 
216             if (subject != null) {
217                 msg.setSubject(subject, charset);
218             }
219             msg.addHeader("Date", getDate());
220 
221             if (headers != null) {
222                 for (Iterator iter = headers.iterator(); iter.hasNext();) {
223                     Header h = (Header) iter.next();
224                     msg.addHeader(h.getName(), h.getValue());
225                 }
226             }
227             PrintStream out = new PrintStream(sds.getOutputStream());
228             message.print(out);
229             out.close();
230 
231             MimeBodyPart textbody = new MimeBodyPart();
232             textbody.setDataHandler(new DataHandler(sds));
233             attachments.addBodyPart(textbody);
234 
235             Enumeration e = files.elements();
236 
237             while (e.hasMoreElements()) {
238                 File file = (File) e.nextElement();
239 
240                 MimeBodyPart body;
241 
242                 body = new MimeBodyPart();
243                 if (!file.exists() || !file.canRead()) {
244                     throw new BuildException("File \"" + file.getAbsolutePath()
245                          + "\" does not exist or is not "
246                          + "readable.");
247                 }
248                 FileDataSource fileData = new FileDataSource(file);
249                 DataHandler fileDataHandler = new DataHandler(fileData);
250 
251                 body.setDataHandler(fileDataHandler);
252                 body.setFileName(file.getName());
253                 attachments.addBodyPart(body);
254             }
255             msg.setContent(attachments);
256             try {
257                 // Send the message using SMTP, or SMTPS if the host uses SSL
258                 Transport transport = sesh.getTransport(SSL ? "smtps" : "smtp");
259                 transport.connect(host, user, password);
260                 transport.sendMessage(msg, msg.getAllRecipients());
261             } catch (SendFailedException sfe) {
262                 if (!shouldIgnoreInvalidRecipients()) {
263                     throw new BuildException(GENERIC_ERROR, sfe);
264                 } else if (sfe.getValidSentAddresses() == null
265                            || sfe.getValidSentAddresses().length == 0) {
266                     throw new BuildException("Couldn't reach any recipient",
267                                              sfe);
268                 } else {
269                     Address[] invalid = sfe.getInvalidAddresses();
270                     if (invalid == null) {
271                         invalid = new Address[0];
272                     }
273                     for (int i = 0; i < invalid.length; i++) {
274                         didntReach(invalid[i], "invalid", sfe);
275                     }
276                     Address[] validUnsent = sfe.getValidUnsentAddresses();
277                     if (validUnsent == null) {
278                         validUnsent = new Address[0];
279                     }
280                     for (int i = 0; i < validUnsent.length; i++) {
281                         didntReach(validUnsent[i], "valid", sfe);
282                     }
283                 }
284             }
285         } catch (MessagingException e) {
286             throw new BuildException(GENERIC_ERROR, e);
287         } catch (IOException e) {
288             throw new BuildException(GENERIC_ERROR, e);
289         }
290     }

292     private static InternetAddress[] internetAddresses(Vector list)
293         throws AddressException, UnsupportedEncodingException {
294         final int size = list.size();
295         InternetAddress[] addrs = new InternetAddress[size];
296 
297         for (int i = 0; i < size; ++i) {
298             EmailAddress addr = (EmailAddress) list.elementAt(i);
299 
300             String name = addr.getName();
301             addrs[i] = (name == null)
302                 ? new InternetAddress(addr.getAddress())
303                 : new InternetAddress(addr.getAddress(), name);
304         }
305         return addrs;
306     }
307 
308     private String parseCharSetFromMimeType(String type) {
309         if (type == null) {
310             return null;
311         }
312         int pos = type.indexOf("charset");
313         if (pos < 0) {
314           return null;
315         }
316         // Assuming mime type in form "text/XXXX; charset=XXXXXX"
317         StringTokenizer token = new StringTokenizer(type.substring(pos), "=; ");
318         token.nextToken(); // Skip 'charset='
319         return token.nextToken();
320     }
321 
322     private void didntReach(Address addr, String category,
323                             MessagingException ex) {
324         String msg = "Failed to send mail to " + category + " address "
325             + addr + " because of " + ex.getMessage();
326         if (task != null) {
327             task.log(msg, Project.MSG_WARN);
328         } else {
329             System.err.println(msg);
330         }
331     }

333     static class SimpleAuthenticator extends Authenticator {
334         private String user = null;
335         private String password = null;
336         public SimpleAuthenticator(String user, String password) {
337             this.user = user;
338             this.password = password;
339         }
340         public PasswordAuthentication getPasswordAuthentication() {
341 
342             return new PasswordAuthentication(user, password);
343         }
