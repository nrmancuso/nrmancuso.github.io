102 	public String getContent() {
103 		String result = printer.content.toString();
104 		for (val entry : stringReplacements.entrySet()) {
105 			result = result.replace(entry.getKey(), entry.getValue());
106 		}
107 		return result;
108 	}

115 		propertySkipList.add(type.getSimpleName() + "/" + propertyName);

128 	@Data
129 	private static class ReferenceTrackingSkip {
130 		private final Class<? extends ASTNode> parent;
131 		private final Class<?> type;
132 	}
133 	
134 	public EcjTreePrinter skipReferenceTracking(Class<? extends ASTNode> parent, Class<?> type) {
135 		referenceTrackingSkipList.add(new ReferenceTrackingSkip(parent, type));
136 		return this;
137 	}

139 	private final EcjTreeVisitor visitor = new EcjTreeVisitor() {
140 		@Override public void visitAny(ASTNode node) {
141 			Collection<ComponentField> fields = findFields(node);
142 			for (ComponentField f : fields) {
143 				String skipListKey = node.getClass().getSimpleName() + "/" + f.field.getName();
144 				if (propertySkipList.contains(skipListKey)) continue;
145 				
146 				Object value;
147 				
148 				if (node instanceof ConditionalExpression) ((ConditionalExpression)node).valueIfTrue.sourceEnd = -2;
149 				if ("originalSourceEnd".equals(f.field.getName()) && node instanceof ArrayTypeReference) {
150 					//workaround for eclipse arbitrarily skipping this field and setting it.
151 					value = -2;
152 				} else {
153 					value = readField(f.field, node);
154 				}
155 				if (value == null) {
156 					continue;
157 				}
158 				
159 				if (propertyIfValueSkipList.get(skipListKey).contains(value)) continue;
160 				
161 				boolean trackRef = true;
162 				for (ReferenceTrackingSkip skip : referenceTrackingSkipList) {
163 					if (skip.getParent() != null && !skip.getParent().isInstance(node)) continue;
164 					if (skip.getType() != null && !skip.getType().isInstance(value)) continue;
165 					trackRef = false;
166 					break;
167 				}
168 				f.print(printer, this, value, trackRef);
169 			}
170 		}
171 		
172 		//TODO all the javadocy nodes need to be as methods in EcjTreeVisitor.
173 		
174 		@Override
175 		public void visitOther(ASTNode node) {
176 			visitAny(node);
177 		}
178 	};

180 	@SneakyThrows(IllegalAccessException.class)
181 	private Object readField(Field field, ASTNode node) {
182 		return field.get(node);
183 	}
184 	
185 	private static Collection<ComponentField> findFields(ASTNode node) {
186 		Class<? extends ASTNode> clazz = node.getClass();
187 		if (visitedClasses.containsKey(clazz)) {
188 			return visitedClasses.get(clazz);
189 		}
190 		List<ComponentField> fields = Lists.newArrayList();
191 		for (Field f : findAllFields(clazz)) {
192 			if ((f.getModifiers() & Modifier.STATIC) != 0) continue;
193 			fields.add(new ComponentField(f));
194 		}
195 		Collections.sort(fields);
196 		visitedClasses.putAll(clazz, fields);
197 		return fields;
198 	}
199 	
200 	
201 	private static List<Field> findAllFields(Class<?> clazz) {
202 		List<Field> allFields = Lists.newArrayList();
203 		findAllFieldsRecursively(allFields, clazz);
204 		return allFields;
205 	}
206 
207 	private static void findAllFieldsRecursively(List<Field> allFields, Class<?> clazz) {
208 		if (clazz == Object.class) {
209 			return;
210 		}
211 		allFields.addAll(Arrays.asList(clazz.getDeclaredFields()));
212 		findAllFieldsRecursively(allFields, clazz.getSuperclass());
213 	}

215 	static class Printer {
216 		final Map<Object, Integer> visited = new MapMaker().weakKeys().makeMap();
217 		int objectCounter = 0;
218 		
219 		private final StringBuilder content = new StringBuilder();
220 		private int indent;
221 		
222 		private final boolean printPositions;
223 		
224 		public Printer(boolean printPositions) {
225 			this.printPositions = printPositions;
226 		}
227 		
228 		void begin(String typeName, String description, Class<?> clazz, int id) {
229 			printIndent();
230 			content.append(typeName).append(" ").append(description)
231 					.append(" = ").append(clazz.getSimpleName());
232 			if (id != -1) content.append(" (id:").append(id).append(")");
233 			content.append("\n");
234 			indent++;
235 		}
236 		
237 		void end() {
238 			indent--;
239 		}
240 		
241 		public void printProperty(String typeName, String name, Object value, boolean posField) {
242 			printIndent();
243 			String stringValue;
244 			if (posField) {
245 				if (value instanceof Long) {
246 					long longValue = (Long)value;
247 					stringValue = String.format("(%d, %d)", (int)(longValue >> 32), (int)(longValue & 0xFFFFFFFFL));
248 				} else {
249 					stringValue = String.valueOf(value);
250 				}
251 			} else if ("bits".equals(name) && value instanceof Integer) {
252 				stringValue = formatBits((Integer)value);
253 			} else if (value instanceof Long) {
254 				long longValue = ((Long)value).longValue();
255 				stringValue = String.format("0x%1$016x (%1$d)  %2$d<<32 | %3$d", value, (int)(longValue>>32), (int)(longValue & 0xFFFFFFFFL));
256 			} else if (value instanceof Integer) {
257 				stringValue = String.format("0x%1$08x (%1$d)", value);
258 			} else if (value instanceof char[]) {
259 				stringValue = new lombok.ast.StringLiteral().astValue(new String((char[])value)).rawValue();
260 			} else if (value instanceof char[][]) {
261 				StringBuilder sb = new StringBuilder();
262 				for (char[] single : ((char[][])value)) {
263 					if (sb.length() != 0) {
264 						sb.append(", ");
265 					}
266 					sb.append(new lombok.ast.StringLiteral().astValue(new String(single)).rawValue());
267 				}
268 				stringValue = "{" + sb.toString() + "}";
269 			} else if ("compilationResult".equals(name)) {
270 				stringValue = value == null ? "[NULL]" : "[SET]";
271 			} else if ("problemReporter".equals(name)) {
272 				return;
273 			} else {
274 				stringValue = String.valueOf(value);
275 			}
276 			content.append(typeName).append(" ").append(name).append(" = ").append(stringValue).append("\n");
277 		}
278 		
279 		private static String formatBits(int value) {
280 			List<Integer> elems = Lists.newArrayList();
281 			int pos = 0;
282 			while (value != 0) {
283 				pos++;
284 				if ((value & 1) != 0) elems.add(pos);
285 				value >>>= 1;
286 			}
287 			return elems.isEmpty() ? "NONE" : "[" + Joiner.on(",").join(elems) + "]";
288 		}
289 		
290 		private void printIndent() {
291 			for (int i = 0; i < indent; i++) {
292 				content.append("  ");
293 			}
294 		}
295 	}

297 	static class ComponentField implements Comparable<ComponentField>{
298 		private static final List<Package> KNOWN_PACKAGES = ImmutableList.of(
299 				String.class.getPackage(),
300 				ASTNode.class.getPackage(),
301 				TypeBinding.class.getPackage(),
302 				Constant.class.getPackage()
303 		);
304 		
305 		private ImmutableMap<Class<?>, Object> DEFAULTS = 
306 				ImmutableMap.<Class<?>, Object>builder()
307 						.put(boolean.class, false)
308 						.put(byte.class, (byte)0)
309 						.put(short.class, (short)0)
310 						.put(int.class, 0)
311 						.put(char.class, '\0')
312 						.put(long.class, 0L)
313 						.put(float.class, 0f)
314 						.put(double.class, .0)
315 						.build();
316 		
317 		private final Field field;
318 		private final int dimensions;
319 		private final Class<?> type;
320 		
321 		public ComponentField(Field field) {
322 			this.field = field;
323 			field.setAccessible(true);
324 			Class<?> type = field.getType();
325 			int dimensions = 0;
326 			while (type.isArray()) {
327 				dimensions++;
328 				type = type.getComponentType();
329 			}
330 			this.dimensions = dimensions;
331 			this.type = type;
332 		}
333 		
334 		@Override
335 		public String toString() {
336 			return createDescription();
337 		}
338 		
339 		private String createDescription() {
340 			StringBuilder result = new StringBuilder();
341 			result.append(typeName());
342 			for (int dim = 0; dim < dimensions; dim++) {
343 				result.append("[]");
344 			}
345 			result.append(" ").append(field.getName());
346 			return result.toString();
347 		}
348 		
349 		private String typeName() {
350 			if (type.isPrimitive() || KNOWN_PACKAGES.contains(type.getPackage())) {
351 				return type.getSimpleName(); 
352 			}
353 			return type.getName();
354 		}
355 		
356 		boolean isPositionField() {
357 			return POSITION_FIELDNAMES.contains(field.getName());
358 		}
359 		
360 		@Override
361 		public int compareTo(ComponentField o) {
362 			Class<?> otherType = o.type;
363 			
364 			if (isPositionField() && o.isPositionField()) {
365 				return POSITION_FIELDNAMES.indexOf(field.getName()) - POSITION_FIELDNAMES.indexOf(o.field.getName());
366 			}
367 			
368 			if (isPositionField() || o.isPositionField()) return isPositionField() ? -1 : 1;
369 			
370 			if (type.isPrimitive() == otherType.isPrimitive()) {
371 				return String.CASE_INSENSITIVE_ORDER.compare(field.getName(), o.field.getName());
372 			}
373 			return type.isPrimitive() ? -1 : 1;
374 		}
375 		
376 		public void printVisited(Printer printer, Integer id) {
377 			printer.printProperty(typeName(), field.getName(), "reference to " + id, false);
378 		}
379 		
380 		public void print(Printer printer, EcjTreeVisitor visitor, Object value, boolean trackRef) {
381 			boolean posField = isPositionField();
382 			
383 			if (!printer.printPositions && posField) return;
384 			if (!posField && isDefault(value)) return;
385 			
386 			unroll(printer, visitor, value, 0, field.getName(), posField, trackRef);
387 		}
388 		
389 		private void unroll(Printer printer, EcjTreeVisitor visitor, Object value, int dim, String description, boolean posField, boolean trackRef) {
390 			if (dim == dimensions) {
391 				if (value instanceof ASTNode) {
392 					if (!trackRef) {
393 						printer.begin(typeName(), description, value.getClass(), printer.objectCounter++);
394 						visitor.visitEcjNode((ASTNode)value);
395 						printer.end();
396 					} else if (printer.visited.containsKey(value)) {
397 						printVisited(printer, printer.visited.get(value));
398 					} else {
399 						printer.visited.put(value, printer.objectCounter);
400 						printer.begin(typeName(), description, value.getClass(), printer.objectCounter++);
401 						visitor.visitEcjNode((ASTNode)value);
402 						printer.end();
403 					}
404 				} else {
405 					printer.printProperty(typeName(), description, value, posField);
406 				}
407 			} else {
408 				if (value == null) {
409 					printer.printProperty(typeName(), description,  "NULL", posField);
410 				} else {
411 					if (type == char.class && dimensions - dim <= 2) {
412 						if (dimensions - dim == 1) {
413 							printer.printProperty(typeName(), description + "[]", value, posField);
414 						} else {
415 							printer.printProperty(typeName(), description + "[][]", value, posField);
416 						}
417 						return;
418 					}
419 					int length = Array.getLength(value);
420 					for (int i = 0; i < length; i++) {
421 						unroll(printer, visitor, Array.get(value, i), dim + 1, description + "[" + i + "]", posField, trackRef);
422 					}
423 				}
424 			}
425 		}
426 		
427 		private boolean isDefault(Object value) {
428 			if (value == null) return true;
429 			
430 			if (type.isPrimitive() && dimensions == 0) {
431 				return DEFAULTS.get(type).equals(value);
432 			}
433 			return false;
434 		}
