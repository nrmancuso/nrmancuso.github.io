56  	public Node createLeftAssociativeBinaryExpression(
57  			org.parboiled.Node<Node> head,
58  			List<org.parboiled.Node<Node>> operatorsNodes, 
59  			List<String> operators,
60  			List<org.parboiled.Node<Node>> tail) {
61  		
62  		Node currentLeft = head.getValue();
63  		
64  		for (int i = 0; i < operators.size(); i++) {
65  			currentLeft = new BinaryExpression()
66  					.rawLeft(currentLeft)
67  					.rawRight(tail.get(i).getValue()).rawOperator(operators.get(i));
68  			source.registerStructure(currentLeft, operatorsNodes.get(i));
69  			positionSpan(currentLeft, head, tail.get(i));
70  		}
71  		
72  		return currentLeft;
73  	}

75  	public Node createAssignmentExpression(Node lhs, String operator, Node rhs) {
76  		if (rhs == null && operator == null) return lhs;
77  		return posify(new BinaryExpression().rawLeft(lhs).rawRight(rhs).rawOperator(operator));
78  	}

80  	public Node createInlineIfExpression(
81  			Node head, org.parboiled.Node<Node> operator1Node,
82  			org.parboiled.Node<Node> operator2Node, Node tail1, Node tail2) {
83  		
84  		if (tail1 == null || tail2 == null) return head;
85  		
86  		InlineIfExpression result = new InlineIfExpression()
87  				.rawCondition(head).rawIfTrue(tail1).rawIfFalse(tail2);
88  		source.registerStructure(result, operator1Node);
89  		source.registerStructure(result, operator2Node);
90  		return posify(result);
91  	}

93  	public Node createUnaryPrefixExpression(Node operand, org.parboiled.Node<Node> opNode, String symbol) {
94  		if (opNode == null) return operand;
95  		
96  		if (!opNode.getChildren().isEmpty() && "cast".equals(opNode.getChildren().get(0).getLabel())) {
97  			return posify(new Cast().rawOperand(operand).rawTypeReference(opNode.getValue()));
98  		} else {
99  			if (symbol != null) symbol = symbol.trim();
100 			if (!symbol.isEmpty()) {
101 				UnaryOperator op = UnaryOperator.fromSymbol(symbol, false);
102 				UnaryExpression expr = new UnaryExpression().rawOperand(operand);
103 				if (op != null) expr.astOperator(op);
104 				return posify(expr);
105 			}
106 		}
107 		
108 		return operand;
109 	}

111 	public Node createUnaryPrefixExpressions(
112 			org.parboiled.Node<Node> operand,
113 			List<org.parboiled.Node<Node>> operators,
114 			List<String> operatorTexts) {
115 		
116 		if (operators == null || operators.isEmpty()) return operand.getValue();
117 		
118 		Node current = operand.getValue();
119 		for (int i = operators.size() - 1; i >= 0; i--) {
120 			org.parboiled.Node<Node> operator = operators.get(i);
121 			Node prev = current;
122 			if (operator == null) continue;
123 			if (!operator.getChildren().isEmpty() && "cast".equals(operator.getChildren().get(0).getLabel())) {
124 				current = new Cast().rawOperand(current).rawTypeReference(operator.getValue());
125 			} else {
126 				String symbol = operatorTexts.get(i);
127 				if (symbol == null) continue;
128 				symbol = symbol.trim();
129 				if (symbol.isEmpty()) continue;
130 				
131 				UnaryOperator op = UnaryOperator.fromSymbol(symbol, false);
132 				UnaryExpression expr = new UnaryExpression().rawOperand(current);
133 				if (op != null) expr.astOperator(op);
134 				current = expr;
135 			}
136 			
137 			if (prev != null && !prev.getPosition().isUnplaced() && prev != current && current != null) {
138 				positionSpan(current, operator, operand);
139 			}
140 		}
141 		
142 		return current;
143 	}

145 	public Node createUnaryPostfixExpression(Node operand, List<org.parboiled.Node<Node>> nodes, List<String> operators) {
146 		if (operators == null) return operand;
147 		
148 		Node current = operand;
149 		for (int i = 0; i < operators.size(); i++) {
150 			String op = operators.get(i);
151 			if (op == null) continue;
152 			op = op.trim();
153 			Node prev = current;
154 			if (op.equals("++")) current = new UnaryExpression().rawOperand(current).astOperator(UnaryOperator.POSTFIX_INCREMENT);
155 			else if (op.equals("--")) current = new UnaryExpression().rawOperand(current).astOperator(UnaryOperator.POSTFIX_DECREMENT);
156 			org.parboiled.Node<Node> p = nodes.get(i);
157 			if (prev != null && !prev.getPosition().isUnplaced() && p != null) {
158 				current.setPosition(new Position(prev.getPosition().getStart(), p.getEndIndex()));
159 			}
160 		}
161 		return current;
162 	}

169 	public Node createQualifiedConstructorInvocation(
170 			Node constructorTypeArgs,
171 			org.parboiled.Node<Node> identifier, org.parboiled.Node<Node> classTypeArgs,
172 			Node methodArguments, Node classBody) {
173 		
174 		TypeReferencePart classTypeArgs0;
175 		boolean classTypeArgsCorrect = false;
176 		Node identifierNode = identifier == null ? null : identifier.getValue();
177 		if (classTypeArgs != null && classTypeArgs.getValue() instanceof TypeReferencePart) {
178 			classTypeArgs0 = (TypeReferencePart)classTypeArgs.getValue();
179 			classTypeArgsCorrect = true;
180 		} else {
181 			classTypeArgs0 = new TypeReferencePart();
182 			if (identifierNode != null) {
183 				classTypeArgs0.setPosition(identifierNode.getPosition());
184 			}
185 		}
186 		
187 		TypeReference typeReference = new TypeReference().astParts().addToEnd(
188 				classTypeArgs0.astIdentifier(createIdentifierIfNeeded(identifierNode, currentPos())));
189 		if (!classTypeArgsCorrect) {
190 			if (identifier != null && identifier.getValue() != null) {
191 				typeReference.setPosition(identifier.getValue().getPosition());
192 			}
193 		} else {
194 			positionSpan(typeReference, identifier, classTypeArgs);
195 		}
196 		
197 		ConstructorInvocation constructorInvocation = new ConstructorInvocation()
198 				.rawTypeReference(typeReference)
199 				.rawAnonymousClassBody(classBody);
200 		
201 		if (constructorTypeArgs instanceof TemporaryNode.TypeArguments) {
202 			for (Node arg : ((TemporaryNode.TypeArguments)constructorTypeArgs).arguments) {
203 				constructorInvocation.rawConstructorTypeArguments().addToEnd(arg);
204 			}
205 		}
206 		
207 		if (methodArguments instanceof TemporaryNode.MethodArguments) {
208 			for (Node arg : ((TemporaryNode.MethodArguments)methodArguments).arguments) {
209 				constructorInvocation.rawArguments().addToEnd(arg);
210 			}
211 		}
212 		
213 		return posify(constructorInvocation);
214 	}

216 	public Node createChainOfQualifiedConstructorInvocations(
217 			org.parboiled.Node<Node> qualifier,
218 			List<org.parboiled.Node<Node>> constructorInvocations) {
219 		Node current = qualifier.getValue();
220 		
221 		if (constructorInvocations == null) return current;
222 		
223 		for (org.parboiled.Node<Node> pNode : constructorInvocations) {
224 			Node n = pNode.getValue();
225 			if (n instanceof ConstructorInvocation) {
226 				current = ((ConstructorInvocation)n).rawQualifier(current);
227 				positionSpan(current, qualifier, pNode);
228 			} else DanglingNodes.addDanglingNode(current, n);
229 		}
230 		
231 		return current;
232 	}

234 	public Node createMethodInvocationOperation(org.parboiled.Node<Node> dot, Node typeArguments, Node name, Node arguments) {
235 		MethodInvocation mi = new MethodInvocation().astName(createIdentifierIfNeeded(name, currentPos()));
236 		
237 		if (typeArguments instanceof TemporaryNode.TypeArguments) {
238 			for (Node arg : ((TemporaryNode.TypeArguments)typeArguments).arguments) {
239 				mi.rawMethodTypeArguments().addToEnd(arg);
240 			}
241 		} else DanglingNodes.addDanglingNode(mi, typeArguments);
242 		
243 		if (arguments instanceof TemporaryNode.MethodArguments) {
244 			for (Node arg : ((TemporaryNode.MethodArguments)arguments).arguments) {
245 				mi.rawArguments().addToEnd(arg);
246 			}
247 		} else DanglingNodes.addDanglingNode(mi, arguments);
248 		
249 		source.registerStructure(mi, dot);
250 		
251 		return posify(mi);
252 	}

262 	public Node createLevel1Expression(org.parboiled.Node<Node> operand, List<org.parboiled.Node<Node>> operations) {
263 		Node current = operand.getValue();
264 		if (operations == null) return current;
265 		
266 		for (org.parboiled.Node<Node> pNode : operations) {
267 			Node o = pNode.getValue();
268 			
269 			if (o instanceof ArrayAccess) {
270 				current = ((ArrayAccess)o).rawOperand(current);
271 			} else if (o instanceof MethodInvocation) {
272 				current = ((MethodInvocation)o).rawOperand(current);
273 			} else if (o instanceof Select) {
274 				current = ((Select)o).rawOperand(current);
275 			} else {
276 				DanglingNodes.addDanglingNode(current, o);
277 			}
278 			
279 			positionSpan(o, operand, pNode);
280 		}
281 		return current;
282 	}

284 	public Node createPrimary(Node identifier, Node methodArguments) {
285 		Identifier id = createIdentifierIfNeeded(identifier, currentPos());
286 		
287 		if (methodArguments instanceof TemporaryNode.MethodArguments) {
288 			MethodInvocation invoke = new MethodInvocation().astName(id);
289 			for (Node arg : ((TemporaryNode.MethodArguments)methodArguments).arguments) {
290 				invoke.rawArguments().addToEnd(arg);
291 			}
292 			return posify(invoke);
293 		} else {
294 			VariableReference ref = new VariableReference().astIdentifier(id);
295 			DanglingNodes.addDanglingNode(ref, methodArguments);
296 			return posify(ref);
297 		}
298 	}

301 		ConstructorInvocation result = new ConstructorInvocation()
302 				.rawTypeReference(type)
303 				.rawAnonymousClassBody(anonymousClassBody);
304 		
305 		if (constructorTypeArgs instanceof TemporaryNode.TypeArguments) {
306 			for (Node arg : ((TemporaryNode.TypeArguments)constructorTypeArgs).arguments) {
307 				result.rawConstructorTypeArguments().addToEnd(arg);
308 			}
309 		}
310 		
311 		if (args instanceof TemporaryNode.MethodArguments) {
312 			for (Node arg : ((TemporaryNode.MethodArguments)args).arguments) {
313 				result.rawArguments().addToEnd(arg);
314 			}
315 		} else DanglingNodes.addDanglingNode(result, args);
316 		
317 		return posify(result);
318 	}

320 	public Node createArrayInitializerExpression(Node head, List<Node> tail) {
321 		ArrayInitializer ai = new ArrayInitializer();
322 		if (head != null) ai.rawExpressions().addToEnd(head);
323 		if (tail != null) for (Node n : tail) if (n != null) ai.rawExpressions().addToEnd(n);
324 		return posify(ai);
325 	}

334 		ArrayCreation ac = new ArrayCreation().rawComponentTypeReference(type).rawInitializer(initializer);
335 		if (dimensions != null) for (Node d : dimensions) {

345 		}

349 	public Node createThisOrSuperOrClass(org.parboiled.Node<Node> dot, String text, Node qualifier) {
350 		Node result;
351 		if ("super".equals(text)) result = new Super().rawQualifier(qualifier);
352 		else if ("class".equals(text)) result = new ClassLiteral().rawTypeReference(qualifier);
353 		else result = new This().rawQualifier(qualifier);
354 		if (dot != null) source.registerStructure(result, dot);
355 		return posify(result);
356 	}

359 		return node instanceof VariableReference || node instanceof Select || node instanceof ArrayAccess;

363 		return node instanceof MethodInvocation || node instanceof ConstructorInvocation;

366 	public boolean typeIsAlsoLegalAsExpression(Node type) {
367 		if (!(type instanceof TypeReference)) return true;
368 		TypeReference tr = (TypeReference)type;
369 		if (tr.astArrayDimensions() > 0) return false;
370 		if (tr.isPrimitive() || tr.isVoid()) return false;
371 		for (Node part : tr.rawParts()) {
372 			if (part instanceof TypeReferencePart) {
373 				if (!((TypeReferencePart)part).rawTypeArguments().isEmpty()) return false;
374 			}
375 		}
376 		
377 		return true;
378 	}
