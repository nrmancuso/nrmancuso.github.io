1   package org.infinispan.distribution;

3   import org.infinispan.Cache;
4   import org.infinispan.commands.VisitableCommand;
5   import org.infinispan.commands.write.InvalidateL1Command;
6   import org.infinispan.test.ReplListener;

9   
10  import java.util.HashMap;
11  import java.util.Map;
12  
13  @Test(groups = {"functional", "smoke"}, testName = "distribution.DistAsyncFuncTest")
14  public class DistAsyncFuncTest extends DistSyncFuncTest {
15  
16     ReplListener r1, r2, r3, r4;
17     ReplListener[] r;
18     Map<Cache<?, ?>, ReplListener> listenerLookup;
19  
20     public DistAsyncFuncTest() {
21        sync = false;
22        tx = false;
23        testRetVals = false;
24     }
25  
26     @Override
27     protected void createCacheManagers() throws Throwable {
28        super.createCacheManagers();
29        r1 = new ReplListener(c1, true, true);
30        r2 = new ReplListener(c2, true, true);
31        r3 = new ReplListener(c3, true, true);
32        r4 = new ReplListener(c4, true, true);
33        r = new ReplListener[]{r1, r2, r3, r4};
34        listenerLookup = new HashMap<Cache<?, ?>, ReplListener>();
35        for (ReplListener rl : r) listenerLookup.put(rl.getCache(), rl);
36     }
37  
38     @Override
39     protected void asyncWait(Object key, Class<? extends VisitableCommand> command, Cache<?, ?>... cachesOnWhichKeyShouldInval) {
40        if (key == null) {
41           // test all caches.
42           for (ReplListener rl : r) rl.expect(command);
43           for (ReplListener rl : r) rl.waitForRpc();
44        } else {
45           for (Cache<?, ?> c : getOwners(key)) {
46              listenerLookup.get(c).expect(command);
47              listenerLookup.get(c).waitForRpc();
48           }
49  
50           if (cachesOnWhichKeyShouldInval != null) {
51              for (Cache<?, ?> c : cachesOnWhichKeyShouldInval) {
52                 listenerLookup.get(c).expect(InvalidateL1Command.class);
53                 listenerLookup.get(c).waitForRpc();
54              }
55           }
56        }
57        // This sucks but for async transactions we still need this!!
58        TestingUtil.sleepThread(1000);
59     }
60  }
