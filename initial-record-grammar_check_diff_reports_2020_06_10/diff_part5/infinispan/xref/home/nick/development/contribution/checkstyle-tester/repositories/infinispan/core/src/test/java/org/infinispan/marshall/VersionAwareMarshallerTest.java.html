142       Map<Integer, GlobalTransaction> m1 = new HashMap<Integer, GlobalTransaction>();
143       Map<Integer, GlobalTransaction> m2 = new TreeMap<Integer, GlobalTransaction>();
144       Map<Integer, GlobalTransaction> m3 = new HashMap<Integer, GlobalTransaction>();
145       Map<Integer, GlobalTransaction> m4 = new FastCopyHashMap<Integer, GlobalTransaction>();
146       for (int i = 0; i < 10; i++) {
147          JGroupsAddress jGroupsAddress = new JGroupsAddress(new IpAddress(1000 * i));
148          GlobalTransaction gtx = gtf.newGlobalTransaction(jGroupsAddress, false);
149          m1.put(1000 * i, gtx);
150          m2.put(1000 * i, gtx);
151          m4.put(1000 * i, gtx);
152       }
153       Map m5 = Immutables.immutableMapWrap(m3);
154       marshallAndAssertEquality(m1);
155       marshallAndAssertEquality(m2);
156       byte[] bytes = marshaller.objectToByteBuffer(m4);
157       Map<Integer, GlobalTransaction> m4Read = (Map<Integer, GlobalTransaction>) marshaller.objectFromByteBuffer(bytes);
158       for (Map.Entry<Integer, GlobalTransaction> entry : m4.entrySet()) {
159          assert m4Read.get(entry.getKey()).equals(entry.getValue()) : "Writen[" + entry.getValue() + "] and read[" + m4Read.get(entry.getKey()) + "] objects should be the same";
160       }
161 
162       marshallAndAssertEquality(m5);
163    }

597       private static final long serialVersionUID = 9032309454840083326L;
598 
599       @Override
600       public boolean equals(Object o) {
601          if (this == o) {
602             return true;
603          }
604          if (o == null || getClass() != o.getClass()) {
605             return false;
606          }
607 
608          Pojo pojo = (Pojo) o;
609 
610          if (b != pojo.b) {
611             return false;
612          }
613          if (i != pojo.i) {
614             return false;
615          }
616 
617          return true;
618       }
619 
620       @Override
621       public int hashCode() {
622          int result;
623          result = i;
624          result = 31 * result + (b ? 1 : 0);
625          return result;
626       }

630          out.writeInt(i);
631          out.writeBoolean(b);
632          serializationCount++;
633       }

637          i = in.readInt();
638          b = in.readBoolean();
