1   package org.infinispan.lock;
2   
3   import org.infinispan.Cache;
4   import org.infinispan.configuration.cache.CacheMode;
5   import org.infinispan.configuration.cache.ConfigurationBuilder;
6   import org.infinispan.manager.EmbeddedCacheManager;
7   import org.infinispan.test.MultipleCacheManagersTest;
8   import org.infinispan.test.fwk.CleanupAfterMethod;
9   import org.infinispan.test.fwk.TestCacheManagerFactory;
10  import org.infinispan.transaction.LockingMode;
11  import org.testng.annotations.Test;
12  
13  @Test(testName = "lock.StaleLocksTransactionTest", groups = "functional")
14  @CleanupAfterMethod
15  public class StaleLocksTransactionTest extends MultipleCacheManagersTest {
16  
17     Cache<String, String> c1, c2;
18  
19     @Override
20     protected void createCacheManagers() throws Throwable {
21        ConfigurationBuilder cfg = getDefaultClusteredCacheConfig(CacheMode.DIST_SYNC, true);
22        cfg
23           .locking().lockAcquisitionTimeout(100)
24           .transaction().lockingMode(LockingMode.PESSIMISTIC);
25        EmbeddedCacheManager cm1 = TestCacheManagerFactory.createClusteredCacheManager(cfg);
26        EmbeddedCacheManager cm2 = TestCacheManagerFactory.createClusteredCacheManager(cfg);
27        registerCacheManager(cm1, cm2);
28        c1 = cm1.getCache();
29        c2 = cm2.getCache();
30     }
31  
32     public void testNoModsCommit() throws Exception {
33        doTest(false, true);
34     }
35  
36     public void testModsRollback() throws Exception {
37        doTest(true, false);
38     }
39  
40     public void testNoModsRollback() throws Exception {
41        doTest(false, false);
42     }
43  
44     public void testModsCommit() throws Exception {
45        doTest(true, true);
46     }
47  
48     private void doTest(boolean mods, boolean commit) throws Exception {
49        tm(c1).begin();
50        c1.getAdvancedCache().lock("k");
51        assert c1.get("k") == null;
52        if (mods) c1.put("k", "v");
53        if (commit)
54           tm(c1).commit();
55        else
56           tm(c1).rollback();
57  
58        assertNotLocked(c1, "k");
59        assertNotLocked(c2, "k");
60     }
61  }
