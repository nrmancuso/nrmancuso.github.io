92  	public FOPCreator(WaveformGroup[] wfgrps, WaveformInfo wfInfo, Float width, Float height) {
93  		waveformGroups = wfgrps;

101 	}

103 	public void toXML( OutputStream out ) throws TransformerConfigurationException, SAXException {
104         SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
105 
106 		th = tf.newTransformerHandler();
107 		th.getTransformer().setOutputProperty(OutputKeys.INDENT, "yes");
108 		th.setResult( new StreamResult(out) );
109 		th.startDocument();
110 
111 		toXML();
112 		
113 		th.endDocument();
114 	    try {
115 			out.flush();
116 			out.close();
117 		} catch (Exception e) {
118 			// TODO Auto-generated catch block

120 		}
121 		
122 	}

130 	private void toXML() throws SAXException, TransformerConfigurationException {
131 		util = new XMLUtil( th );
132 		util.startElement( "fo:root", "xmlns:fo", "http://www.w3.org/1999/XSL/Format" );
133 		{
134 			addLayoutMasterSet();
135 			util.startElement( "fo:page-sequence", "master-reference", "page" );
136 			{
137 				addHeader();
138 				util.startElement( "fo:flow", "flow-name", "xsl-region-body");
139 				int len = waveformGroups.length;
140 				for ( int i = 0 ; i < len ; i++ ) {
141 					addGraphicHeader( waveformGroups[i] );
142 					addGraphic( waveformGroups[i] );
143 					addGraphicFooter( waveformGroups[i] );
144 				}
145 				addFooter();
146 				util.endElement( "fo:flow" );
147 			}
148 			util.endElement( "fo:page-sequence" );
149 			//addTestPage();
150 		}
151 		util.endElement( "fo:root" );
152 	}

164 			creator.setXMLUtil( new XMLUtil( th, "svg" ) );
165 			creator.addSVGStart();
166 			creator.addDefs();
167 			creator.addGrid(pageWidth.intValue(), pageHeight.intValue());
168 			util.endElement("svg:svg");
169 		util.endElement( "fo:instream-foreign-object" );
170 	util.endElement( "fo:block" );
171 		util.endElement( "fo:flow" );
172 		util.endElement( "fo:page-sequence" );
173 	}

290 	private void addTable( String border, String[] columnWidths, String[] textAligns, String[][] data) throws SAXException {
291 		AttributesImpl tblAttr = util.newAttribute( "border-style", border);
292 		util.addAttribute( tblAttr, "table-layout", "fixed" );
293 		util.startElement( "fo:table", tblAttr);
294 		AttributesImpl colAttr = util.newAttribute( "column-number", border);
295 		for ( int i = 0 ; i < columnWidths.length ; i++ ) {
296 			util.singleElement("fo:table-column", util.newAttribute( "column-width", columnWidths[i]), null );
297 		}
298 		util.startElement( "fo:table-body", null);
299 		int rows = data.length;
300 		int cols = data[0].length;
301 		AttributesImpl cellAttr = util.newAttribute( "border-style", "none");
302 		for ( int row = 0 ; row < rows ; row++ ) {
303 			util.startElement( "fo:table-row", null);
304 			for ( int col = 0 ; col < cols ; col++ ) {
305 				util.startElement( "fo:table-cell", cellAttr );
306 				util.singleElement("fo:block", util.newAttribute( "text-align", textAligns[col]), data[row][col] );
307 				util.endElement( "fo:table-cell");
308 			}
309 			util.endElement( "fo:table-row");
310 		}
311 		util.endElement( "fo:table-body");
312 		util.endElement( "fo:table");
313 	}
