57  public class SeriesModel extends AbstractModel {
58  
59      private long pk = -1l;
60  
61      private boolean incorrectWLEntry = false;
62      private String drCode = null;
63      private String drCodeMeaning = null;
64      private String drCodeDesignator = null;
65  
66  
67  
68      public SeriesModel() {
69      }
70  
71      public SeriesModel(Dataset ds) {
72          super(ds);
73          ds.setPrivateCreatorID(PrivateTags.CreatorID);
74          Dataset item = ds.getItem(Tags.PPSDiscontinuationReasonCodeSeq);
75          if ( item != null ) {
76              drCode = item.getString(Tags.CodeValue);
77              drCodeMeaning = item.getString(Tags.CodeMeaning);
78              drCodeDesignator = item.getString(Tags.CodingSchemeDesignator);
79              incorrectWLEntry = "110514".equals(drCode) && "DCM".equals(drCodeDesignator);
80          }
81          ByteBuffer bb = ds.getByteBuffer(PrivateTags.SeriesPk);
82          this.pk = bb == null ? -1 : Convert.toLong(bb.array());
83      }
84  
85      public final long getPk() {
86          return pk;
87      }
88  
89      public final void setPk(long pk) {
90          ds.setPrivateCreatorID(PrivateTags.CreatorID);
91          ds.putOB(PrivateTags.SeriesPk, Convert.toBytes(pk));
92          this.pk = pk;
93      }
94  
95      public int hashCode() {
96          return (int)( pk ^ pk >>> 32);//like Long.hashCode()
97      }
98  
99      public boolean equals(Object o) {
100         if (this == o) return true;
101         if (!(o instanceof SeriesModel)) return false;
102         SeriesModel other = (SeriesModel) o;
103         return pk == other.pk;
104     }
