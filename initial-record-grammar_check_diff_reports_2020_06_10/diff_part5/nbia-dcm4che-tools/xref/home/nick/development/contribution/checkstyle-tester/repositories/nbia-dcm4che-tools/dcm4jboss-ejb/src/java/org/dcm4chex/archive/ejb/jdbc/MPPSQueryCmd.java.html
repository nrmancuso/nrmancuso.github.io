42  import java.sql.SQLException;
43  import java.sql.Types;
44  import java.util.ArrayList;
45  import java.util.List;
46  
47  import org.dcm4che.data.Dataset;
48  import org.dcm4che.data.DcmObjectFactory;
49  import org.dcm4che.dict.Tags;
50  import org.dcm4che.net.DcmServiceException;
51  import org.dcm4chex.archive.common.DatasetUtils;
52  import org.dcm4chex.archive.ejb.conf.AttributeFilter;
53  
54  /**
55   * @author gunter.zeilinger@tiani.com
56   * @version $Revision: 16583 $ $Date: 2012-02-08 13:36:32 +0100 (Mi, 08 Feb 2012) $
57   * @since 10.02.2004
58   */
59  public class MPPSQueryCmd extends BaseReadCmd {
60  
61      public static int transactionIsolationLevel = 0;
62      public static int blobAccessType = Types.LONGVARBINARY;
63  
64      private static final String[] FROM = { "Patient", "MPPS"};
65  
66      private static final String[] SELECT = { "Patient.encodedAttributes",
67      "MPPS.encodedAttributes"};
68  
69      private static final String[] RELATIONS = { "Patient.pk",
70      "MPPS.patient_fk"};
71  
72      private static final int[] PATIENT_ATTRS = { Tags.RefPatientSeq,
73          Tags.PatientName, Tags.PatientID, Tags.IssuerOfPatientID,
74          Tags.PatientBirthDate, Tags.PatientSex, };
75  
76      private final SqlBuilder sqlBuilder = new SqlBuilder();
77  
78      /**
79       * @param ds
80       * @throws SQLException
81       */
82      public MPPSQueryCmd(Dataset keys, boolean emptyAccNo) throws SQLException {
83          super(JdbcProperties.getInstance().getDataSource(),
84                  transactionIsolationLevel);
85          try {
86              AttributeFilter patAttrFilter = AttributeFilter.getPatientAttributeFilter();
87              defineColumnTypes(new int[] { blobAccessType, blobAccessType });
88              // ensure keys contains (8,0005) for use as result filter
89              sqlBuilder.setFrom(FROM);
90              sqlBuilder.setRelations(RELATIONS);
91              sqlBuilder.addListOfStringMatch(null, "MPPS.sopIuid",
92                      SqlBuilder.TYPE1,
93                      keys.getStrings(Tags.SOPInstanceUID) );
94              if ( emptyAccNo ) {
95                  sqlBuilder.addNULLValueMatch(null, "MPPS.accessionNumber", false );
96              } else {
97                  sqlBuilder.addListOfStringMatch(null, "MPPS.accessionNumber",
98                          SqlBuilder.TYPE2,
99                          keys.getStrings(Tags.AccessionNumber) );
100             }
101             sqlBuilder.addListOfStringMatch(null, "Patient.patientId",
102                     SqlBuilder.TYPE1,
103                     keys.getStrings(Tags.PatientID) );
104             sqlBuilder.addPNMatch(new String[] {
105                     "Patient.patientName",
106                     "Patient.patientIdeographicName",
107             "Patient.patientPhoneticName"},
108             SqlBuilder.TYPE2,
109             patAttrFilter.isICase(Tags.PatientName),
110             keys.getString(Tags.PatientName));
111             sqlBuilder.addListOfStringMatch(null, "MPPS.modality",
112                     SqlBuilder.TYPE1,
113                     keys.getStrings(Tags.Modality));
114             sqlBuilder.addListOfStringMatch(null, "MPPS.performedStationAET",
115                     SqlBuilder.TYPE1,
116                     keys.getStrings(Tags.PerformedStationAET));
117             sqlBuilder.addRangeMatch(null, "MPPS.ppsStartDateTime",
118                     SqlBuilder.TYPE1,
119                     keys.getDateTimeRange(Tags.PPSStartDate,Tags.PPSStartTime));
120             sqlBuilder.addListOfStringMatch(null, "MPPS.ppsStatusAsInt",
121                     SqlBuilder.TYPE1,
122                     keys.getStrings(Tags.PPSStatus));
123         } catch (SQLException x) {
124             close();
125             throw x;
126         } catch (RuntimeException x) {
127             close();
128             throw x;
129         }
130     }
131 
132     public int count() throws SQLException {
133         try {
134             sqlBuilder.setSelectCount(new String[]{"MPPS.pk"}, true);
135             execute( sqlBuilder.getSql() );
136             next();
137             return rs.getInt(1);
138         } finally {
139             close();
140         }
141     }
142     public List<Dataset> list(int offset, int limit ) {
143         sqlBuilder.setSelect(SELECT);
144         sqlBuilder.addOrderBy("MPPS.ppsStartDateTime", SqlBuilder.DESC);
145         sqlBuilder.setOffset(offset);
146         sqlBuilder.setLimit(limit);
147         setFetchSize(limit);
148         List<Dataset> resp = null;
149         try {
150             resp = new ArrayList<Dataset>();
151             execute(sqlBuilder.getSql());
152             while ( next() ) {
153                 resp.add( getDataset() );
154             }
155         } catch ( Exception x ) {
156             log.error( "Exception occured in getMWLEntries: "+x.getMessage(), x );
157         } finally {
158             close();
159         }
160         return resp;
161     }
162 
163     public Dataset getDataset() throws SQLException {
164         Dataset ds = DcmObjectFactory.getInstance().newDataset();       
165         DatasetUtils.fromByteArray( rs.getBytes(2), ds); //mpps attrs
166         DatasetUtils.fromByteArray( rs.getBytes(1), ds); //pat attrs
167         return ds;
168     }
