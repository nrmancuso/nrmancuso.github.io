123     public void forwardCancelRQ(Dimse ccancelrq) {
124         canceled = true;
125         if (assoc != null) {
126             try {
127                 assoc.getAssociation().write(ccancelrq);
128             } catch (Exception e) {
129                 log.warn("Failed to forward C-CANCEL-RQ:", e);
130             }
131         }
132     }

134     public void splitAndForwardMoveRQ(int pcid, int msgid, int priority,
135             String moveDest, Collection<String> studyIuids,
136             Collection<String> seriesIuids, Collection<String> iuids) {
137         DcmObjectFactory dof = DcmObjectFactory.getInstance();
138         Dataset ds = dof.newDataset();
139         ds.putCS(Tags.QueryRetrieveLevel, IMAGE);
140         if (studyIuids.size() == 1)
141             ds.putUI(Tags.StudyInstanceUID, studyIuids.iterator().next());
142         if (seriesIuids.size() == 1)
143             ds.putUI(Tags.SeriesInstanceUID, seriesIuids.iterator().next());
144         String[] a = (String[]) iuids.toArray(new String[iuids.size()]);
145         if (a.length <= service.getMaxUIDsPerMoveRQ()) {
146             ds.putUI(Tags.SOPInstanceUID, a);
147             forwardMoveRQ(pcid, msgid, priority, moveDest, ds, iuids);
148         } else {
149             String[] b = new String[service.getMaxUIDsPerMoveRQ()];
150             int off = 0;
151             while (off + b.length < a.length) {
152                 System.arraycopy(a, off, b, 0, b.length);
153                 ds.putUI(Tags.SOPInstanceUID, b);
154                 forwardMoveRQ(pcid, msgid, priority, moveDest, ds,
155                         Arrays.asList(b));
156                 if (canceled) {
157                     return;
158                 }
159                 off += b.length;
160             }
161             b = new String[a.length - off];
162             System.arraycopy(a, off, b, 0, b.length);
163             ds.putUI(Tags.SOPInstanceUID, b);
164             forwardMoveRQ(pcid, msgid, priority, moveDest, ds,
165                     Arrays.asList(b));
166         }
167     }

169     public void forwardMoveRQ(int pcid, int msgid, int priority,
170             String moveDest, Dataset moveRqData,
171             Collection<String> iuids) {
172         remaining -= iuids.size();
173         String sopClassUID = PATIENT.equals(
174                 moveRqData.getString(Tags.QueryRetrieveLevel)) 
175                     ? UIDs.PatientRootQueryRetrieveInformationModelMOVE
176                     : UIDs.StudyRootQueryRetrieveInformationModelMOVE;
177         try {
178             assoc = openAssociation(pcid, sopClassUID );
179         } catch (Exception e) {
180             log.info("Failed to open assocation to " + calledAET, e);
181             failedIUIDs.addAll(iuids);
182             return;
183         }
184         try {
185             moveRspCmd = null;
186             moveRspData = null;
187             DimseListener moveRspListener = new DimseListener() {
188                 public void dimseReceived(Association assoc, Dimse dimse) {
189                     moveRspCmd = dimse.getCommand();
190                     try {
191                         moveRspData = dimse.getDataset();
192                     } catch (IOException e) {
193                         log.error("Failure during receive of C-MOVE_RSP from " 
194                                 + calledAET, e);
195                     }
196                 }
197             };
198             AssociationFactory asf = AssociationFactory.getInstance();
199             Command moveRqCmd = DcmObjectFactory.getInstance().newCommand();
200             moveRqCmd.initCMoveRQ(msgid, sopClassUID, priority, moveDest);
201             assoc.invoke(asf.newDimse(pcid, moveRqCmd , moveRqData), moveRspListener);
202         } catch (Exception e) {
203             log.error("Failed to forward MOVE RQ to " + calledAET, e);
204         } finally {
205             try {
206                 assoc.release(true);
207                 // workaround to ensure that the final MOVE-RSP of forwarded
208                 // MOVE-RQ is processed before continuing
209                 Thread.sleep(10);
210             } catch (Exception e) {
211                 log.info("Exception during release of assocation to "
212                         + calledAET, e);
213             }
214          }
215         if (moveRspCmd == null || moveRspCmd.getStatus() == Status.Pending) {
216             log.error("No final MOVE RSP received from " + calledAET);
217             failedIUIDs.addAll(iuids);
218             failed += iuids.size();
219         } else {
220             completed += moveRspCmd.getInt(Tags.NumberOfCompletedSubOperations, 0);
221             warnings += moveRspCmd.getInt(Tags.NumberOfWarningSubOperations, 0);
222             failed += moveRspCmd.getInt(Tags.NumberOfFailedSubOperations, 0);
223             remaining += moveRspCmd.getInt(Tags.NumberOfRemainingSubOperations, 0);
224             if (moveRspData != null) {
225                 String[] a = moveRspData
226                         .getStrings(Tags.FailedSOPInstanceUIDList);
227                 if (a != null && a.length != 0) {
228                     failedIUIDs.addAll(Arrays.asList(a));
229                 } else {
230                     failedIUIDs.addAll(iuids);
231                 }
232             }
233         }
234         moveRspCmd = null;
235     }

237     private ActiveAssociation openAssociation(int pcid, String sopClassUID)
238             throws Exception {
239         AEDTO retrieveAEData = service.queryAEData(calledAET, null);
240         AssociationFactory asf = AssociationFactory.getInstance();
241         Association a = asf.newRequestor(
242                 service.createSocket(localAET, retrieveAEData));
243         a.setAcTimeout(service.getAcTimeout());
244         a.setDimseTimeout(service.getDimseTimeout());
245         a.setSoCloseDelay(service.getSoCloseDelay());
246         AAssociateRQ rq = asf.newAAssociateRQ();
247         rq.setCalledAET(calledAET);
248         rq.setCallingAET(callingAET);
249         rq.addPresContext(asf.newPresContext(pcid, sopClassUID));
250         rq.addExtNegotiation(asf.newExtNegotiation(sopClassUID, RELATIONAL_RETRIEVE));
251         PDU pdu = a.connect(rq);
252         if (!(pdu instanceof AAssociateAC)) {
253             throw new IOException("Association not accepted by "
254                     + calledAET + ":\n" + pdu);
255         }
256         AAssociateAC ac = (AAssociateAC) pdu;
257         ActiveAssociation assoc = asf.newActiveAssociation(a, null);
258         assoc.start();
259         if (a.getAcceptedTransferSyntaxUID(pcid) == null) {
260             try {
261                 assoc.release(false);
262             } catch (Exception e) {
263                 log.info("Exception during release of assocation to "
264                         + calledAET, e);
265             }
266             throw new IOException(
267                     QueryRetrieveScpService.uidDict.toString(sopClassUID)
268                     + " not accepted by " + calledAET);
269         }
270         ExtNegotiation extNeg = ac.getExtNegotiation(sopClassUID);
271         if (extNeg == null
272                 || !Arrays.equals(extNeg.info(), RELATIONAL_RETRIEVE)) {
273             log.warn("Relational Retrieve not supported by " + calledAET);
274         }
275         return assoc;
276     }

278     public void adjustPendingRsp(Command rspCmd) {
279         Command tmp = moveRspCmd;
280         rspCmd.putUS(Tags.NumberOfRemainingSubOperations, remaining
281                 + (tmp == null ? 0 
282                         : tmp.getInt(Tags.NumberOfRemainingSubOperations, 0)));
283         rspCmd.putUS(Tags.NumberOfCompletedSubOperations, 
284                 rspCmd.getInt(Tags.NumberOfCompletedSubOperations, 0)
285                 + completed + (tmp == null ? 0 
286                         : tmp.getInt(Tags.NumberOfCompletedSubOperations, 0)));
287         rspCmd.putUS(Tags.NumberOfWarningSubOperations, 
288                 rspCmd.getInt(Tags.NumberOfWarningSubOperations, 0)
289                 + warnings + (tmp == null ? 0 
290                         : tmp.getInt(Tags.NumberOfWarningSubOperations, 0)));
291         rspCmd.putUS(Tags.NumberOfFailedSubOperations, 
292                 rspCmd.getInt(Tags.NumberOfFailedSubOperations, 0)
293                 + failed + (tmp == null ? 0 
294                         : tmp.getInt(Tags.NumberOfFailedSubOperations, 0)));
295      }

297     public static void addStudySeriesIUIDs(Dataset moveRqData,
298             Collection<String> studyIUIDs, Collection<String> seriesIUIDs) {
299         String qrLevel = moveRqData.getString(Tags.QueryRetrieveLevel);
300         if ("STUDY".equals(qrLevel) || "PATIENT".equals(qrLevel))
301             return;
302         if (studyIUIDs.size() == 1)
303             moveRqData.putUI(Tags.StudyInstanceUID,
304                     studyIUIDs.iterator().next());
305         if ("SERIES".equals(qrLevel)) return;
306         if (seriesIUIDs.size() == 1)
307             moveRqData.putUI(Tags.SeriesInstanceUID,
308                     seriesIUIDs.iterator().next());
309     }
