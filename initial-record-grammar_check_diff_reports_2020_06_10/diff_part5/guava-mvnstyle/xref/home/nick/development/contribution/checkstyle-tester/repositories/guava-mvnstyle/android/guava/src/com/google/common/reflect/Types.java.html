67    /** Returns the array type of {@code componentType}. */
68    static Type newArrayType(Type componentType) {
69      if (componentType instanceof WildcardType) {
70        WildcardType wildcard = (WildcardType) componentType;
71        Type[] lowerBounds = wildcard.getLowerBounds();
72        checkArgument(lowerBounds.length <= 1, "Wildcard cannot have more than one lower bounds.");
73        if (lowerBounds.length == 1) {
74          return supertypeOf(newArrayType(lowerBounds[0]));
75        } else {
76          Type[] upperBounds = wildcard.getUpperBounds();
77          checkArgument(upperBounds.length == 1, "Wildcard should have only one upper bound.");
78          return subtypeOf(newArrayType(upperBounds[0]));
79        }
80      }
81      return JavaVersion.CURRENT.newArrayType(componentType);
82    }

85     * Returns a type where {@code rawType} is parameterized by {@code arguments} and is owned by
86     * {@code ownerType}.
87     */
88    static ParameterizedType newParameterizedTypeWithOwner(
89        @NullableDecl Type ownerType, Class<?> rawType, Type... arguments) {
90      if (ownerType == null) {
91        return newParameterizedType(rawType, arguments);
92      }
93      // ParameterizedTypeImpl constructor already checks, but we want to throw NPE before IAE
94      checkNotNull(arguments);
95      checkArgument(rawType.getEnclosingClass() != null, "Owner type for unenclosed %s", rawType);
96      return new ParameterizedTypeImpl(ownerType, rawType, arguments);
97    }

105   /** Decides what owner type to use for constructing {@link ParameterizedType} from a raw class. */
106   private enum ClassOwnership {
107     OWNED_BY_ENCLOSING_CLASS {
108       @Override
109       @NullableDecl
110       Class<?> getOwnerType(Class<?> rawType) {
111         return rawType.getEnclosingClass();
112       }
113     },
114     LOCAL_CLASS_HAS_NO_OWNER {
115       @Override
116       @NullableDecl
117       Class<?> getOwnerType(Class<?> rawType) {
118         if (rawType.isLocalClass()) {
119           return null;
120         } else {
121           return rawType.getEnclosingClass();
122         }
123       }
124     };
125 
126     @NullableDecl
127     abstract Class<?> getOwnerType(Class<?> rawType);
128 
129     static final ClassOwnership JVM_BEHAVIOR = detectJvmBehavior();
130 
131     private static ClassOwnership detectJvmBehavior() {
132       class LocalClass<T> {}
133       Class<?> subclass = new LocalClass<String>() {}.getClass();
134       ParameterizedType parameterizedType = (ParameterizedType) subclass.getGenericSuperclass();
135       for (ClassOwnership behavior : ClassOwnership.values()) {
136         if (behavior.getOwnerType(LocalClass.class) == parameterizedType.getOwnerType()) {
137           return behavior;
138         }
139       }
140       throw new AssertionError();
141     }
142   }

175   @NullableDecl
176   static Type getComponentType(Type type) {
177     checkNotNull(type);
178     final AtomicReference<Type> result = new AtomicReference<>();
179     new TypeVisitor() {
180       @Override
181       void visitTypeVariable(TypeVariable<?> t) {
182         result.set(subtypeOfComponentType(t.getBounds()));
183       }
184 
185       @Override
186       void visitWildcardType(WildcardType t) {
187         result.set(subtypeOfComponentType(t.getUpperBounds()));
188       }
189 
190       @Override
191       void visitGenericArrayType(GenericArrayType t) {
192         result.set(t.getGenericComponentType());
193       }
194 
195       @Override
196       void visitClass(Class<?> t) {
197         result.set(t.getComponentType());
198       }
199     }.visit(type);
200     return result.get();
201   }

203   /**
204    * Returns {@code ? extends X} if any of {@code bounds} is a subtype of {@code X[]}; or null
205    * otherwise.
206    */
207   @NullableDecl
208   private static Type subtypeOfComponentType(Type[] bounds) {
209     for (Type bound : bounds) {
210       Type componentType = getComponentType(bound);
211       if (componentType != null) {
212         // Only the first bound can be a class or array.
213         // Bounds after the first can only be interfaces.
214         if (componentType instanceof Class) {
215           Class<?> componentClass = (Class<?>) componentType;
216           if (componentClass.isPrimitive()) {
217             return componentClass;
218           }
219         }
220         return subtypeOf(componentType);
221       }
222     }
223     return null;
224   }

261   private static final class ParameterizedTypeImpl implements ParameterizedType, Serializable {
262 
263     @NullableDecl private final Type ownerType;
264     private final ImmutableList<Type> argumentsList;
265     private final Class<?> rawType;
266 
267     ParameterizedTypeImpl(@NullableDecl Type ownerType, Class<?> rawType, Type[] typeArguments) {
268       checkNotNull(rawType);
269       checkArgument(typeArguments.length == rawType.getTypeParameters().length);
270       disallowPrimitiveType(typeArguments, "type parameter");
271       this.ownerType = ownerType;
272       this.rawType = rawType;
273       this.argumentsList = JavaVersion.CURRENT.usedInGenericType(typeArguments);
274     }
275 
276     @Override
277     public Type[] getActualTypeArguments() {
278       return toArray(argumentsList);
279     }
280 
281     @Override
282     public Type getRawType() {
283       return rawType;
284     }
285 
286     @Override
287     public Type getOwnerType() {
288       return ownerType;
289     }
290 
291     @Override
292     public String toString() {
293       StringBuilder builder = new StringBuilder();
294       if (ownerType != null && JavaVersion.CURRENT.jdkTypeDuplicatesOwnerName()) {
295         builder.append(JavaVersion.CURRENT.typeName(ownerType)).append('.');
296       }
297       return builder
298           .append(rawType.getName())
299           .append('<')
300           .append(COMMA_JOINER.join(transform(argumentsList, TYPE_NAME)))
301           .append('>')
302           .toString();
303     }
304 
305     @Override
306     public int hashCode() {
307       return (ownerType == null ? 0 : ownerType.hashCode())
308           ^ argumentsList.hashCode()
309           ^ rawType.hashCode();
310     }
311 
312     @Override
313     public boolean equals(Object other) {
314       if (!(other instanceof ParameterizedType)) {
315         return false;
316       }
317       ParameterizedType that = (ParameterizedType) other;
318       return getRawType().equals(that.getRawType())
319           && Objects.equal(getOwnerType(), that.getOwnerType())
320           && Arrays.equals(getActualTypeArguments(), that.getActualTypeArguments());
321     }
322 
323     private static final long serialVersionUID = 0;
324   }

326   private static <D extends GenericDeclaration> TypeVariable<D> newTypeVariableImpl(
327       D genericDeclaration, String name, Type[] bounds) {
328     TypeVariableImpl<D> typeVariableImpl =
329         new TypeVariableImpl<D>(genericDeclaration, name, bounds);
330     @SuppressWarnings("unchecked")
331     TypeVariable<D> typeVariable =
332         Reflection.newProxy(
333             TypeVariable.class, new TypeVariableInvocationHandler(typeVariableImpl));
334     return typeVariable;
335   }

337   /**
338    * Invocation handler to work around a compatibility problem between Java 7 and Java 8.
339    *
340    * <p>Java 8 introduced a new method {@code getAnnotatedBounds()} in the {@link TypeVariable}
341    * interface, whose return type {@code AnnotatedType[]} is also new in Java 8. That means that we
342    * cannot implement that interface in source code in a way that will compile on both Java 7 and
343    * Java 8. If we include the {@code getAnnotatedBounds()} method then its return type means it
344    * won't compile on Java 7, while if we don't include the method then the compiler will complain
345    * that an abstract method is unimplemented. So instead we use a dynamic proxy to get an
346    * implementation. If the method being called on the {@code TypeVariable} instance has the same
347    * name as one of the public methods of {@link TypeVariableImpl}, the proxy calls the same method
348    * on its instance of {@code TypeVariableImpl}. Otherwise it throws {@link
349    * UnsupportedOperationException}; this should only apply to {@code getAnnotatedBounds()}. This
350    * does mean that users on Java 8 who obtain an instance of {@code TypeVariable} from {@link
351    * TypeResolver#resolveType} will not be able to call {@code getAnnotatedBounds()} on it, but that
352    * should hopefully be rare.
353    *
354    * <p>This workaround should be removed at a distant future time when we no longer support Java
355    * versions earlier than 8.
356    */
357   private static final class TypeVariableInvocationHandler implements InvocationHandler {
358     private static final ImmutableMap<String, Method> typeVariableMethods;
359 
360     static {
361       ImmutableMap.Builder<String, Method> builder = ImmutableMap.builder();
362       for (Method method : TypeVariableImpl.class.getMethods()) {
363         if (method.getDeclaringClass().equals(TypeVariableImpl.class)) {
364           try {
365             method.setAccessible(true);
366           } catch (AccessControlException e) {
367             // OK: the method is accessible to us anyway. The setAccessible call is only for
368             // unusual execution environments where that might not be true.
369           }
370           builder.put(method.getName(), method);
371         }
372       }
373       typeVariableMethods = builder.build();
374     }
375 
376     private final TypeVariableImpl<?> typeVariableImpl;
377 
378     TypeVariableInvocationHandler(TypeVariableImpl<?> typeVariableImpl) {
379       this.typeVariableImpl = typeVariableImpl;
380     }
381 
382     @Override
383     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
384       String methodName = method.getName();
385       Method typeVariableMethod = typeVariableMethods.get(methodName);
386       if (typeVariableMethod == null) {
387         throw new UnsupportedOperationException(methodName);
388       } else {
389         try {
390           return typeVariableMethod.invoke(typeVariableImpl, args);
391         } catch (InvocationTargetException e) {
392           throw e.getCause();
393         }
394       }
395     }
396   }

398   private static final class TypeVariableImpl<D extends GenericDeclaration> {
399 
400     private final D genericDeclaration;
401     private final String name;
402     private final ImmutableList<Type> bounds;
403 
404     TypeVariableImpl(D genericDeclaration, String name, Type[] bounds) {
405       disallowPrimitiveType(bounds, "bound for type variable");
406       this.genericDeclaration = checkNotNull(genericDeclaration);
407       this.name = checkNotNull(name);
408       this.bounds = ImmutableList.copyOf(bounds);
409     }
410 
411     public Type[] getBounds() {
412       return toArray(bounds);
413     }
414 
415     public D getGenericDeclaration() {
416       return genericDeclaration;
417     }
418 
419     public String getName() {
420       return name;
421     }
422 
423     public String getTypeName() {
424       return name;
425     }
426 
427     @Override
428     public String toString() {
429       return name;
430     }
431 
432     @Override
433     public int hashCode() {
434       return genericDeclaration.hashCode() ^ name.hashCode();
435     }
436 
437     @Override
438     public boolean equals(Object obj) {
439       if (NativeTypeVariableEquals.NATIVE_TYPE_VARIABLE_ONLY) {
440         // equal only to our TypeVariable implementation with identical bounds
441         if (obj != null
442             && Proxy.isProxyClass(obj.getClass())
443             && Proxy.getInvocationHandler(obj) instanceof TypeVariableInvocationHandler) {
444           TypeVariableInvocationHandler typeVariableInvocationHandler =
445               (TypeVariableInvocationHandler) Proxy.getInvocationHandler(obj);
446           TypeVariableImpl<?> that = typeVariableInvocationHandler.typeVariableImpl;
447           return name.equals(that.getName())
448               && genericDeclaration.equals(that.getGenericDeclaration())
449               && bounds.equals(that.bounds);
450         }
451         return false;
452       } else {
453         // equal to any TypeVariable implementation regardless of bounds
454         if (obj instanceof TypeVariable) {
455           TypeVariable<?> that = (TypeVariable<?>) obj;
456           return name.equals(that.getName())
457               && genericDeclaration.equals(that.getGenericDeclaration());
458         }
459         return false;
460       }
461     }
462   }

464   static final class WildcardTypeImpl implements WildcardType, Serializable {
465 
466     private final ImmutableList<Type> lowerBounds;
467     private final ImmutableList<Type> upperBounds;
468 
469     WildcardTypeImpl(Type[] lowerBounds, Type[] upperBounds) {
470       disallowPrimitiveType(lowerBounds, "lower bound for wildcard");
471       disallowPrimitiveType(upperBounds, "upper bound for wildcard");
472       this.lowerBounds = JavaVersion.CURRENT.usedInGenericType(lowerBounds);
473       this.upperBounds = JavaVersion.CURRENT.usedInGenericType(upperBounds);
474     }
475 
476     @Override
477     public Type[] getLowerBounds() {
478       return toArray(lowerBounds);
479     }
480 
481     @Override
482     public Type[] getUpperBounds() {
483       return toArray(upperBounds);
484     }
485 
486     @Override
487     public boolean equals(Object obj) {
488       if (obj instanceof WildcardType) {
489         WildcardType that = (WildcardType) obj;
490         return lowerBounds.equals(Arrays.asList(that.getLowerBounds()))
491             && upperBounds.equals(Arrays.asList(that.getUpperBounds()));
492       }
493       return false;
494     }
495 
496     @Override
497     public int hashCode() {
498       return lowerBounds.hashCode() ^ upperBounds.hashCode();
499     }
500 
501     @Override
502     public String toString() {
503       StringBuilder builder = new StringBuilder("?");
504       for (Type lowerBound : lowerBounds) {
505         builder.append(" super ").append(JavaVersion.CURRENT.typeName(lowerBound));
506       }
507       for (Type upperBound : filterUpperBounds(upperBounds)) {
508         builder.append(" extends ").append(JavaVersion.CURRENT.typeName(upperBound));
509       }
510       return builder.toString();
511     }
512 
513     private static final long serialVersionUID = 0;
514   }

516   private static Type[] toArray(Collection<Type> types) {
517     return types.toArray(new Type[0]);
518   }

520   private static Iterable<Type> filterUpperBounds(Iterable<Type> bounds) {
521     return Iterables.filter(bounds, Predicates.not(Predicates.<Type>equalTo(Object.class)));
522   }

524   private static void disallowPrimitiveType(Type[] types, String usedAs) {
525     for (Type type : types) {
526       if (type instanceof Class) {
527         Class<?> cls = (Class<?>) type;
528         checkArgument(!cls.isPrimitive(), "Primitive type '%s' used as %s", cls, usedAs);
529       }
530     }

541   // TODO(benyu): Once behavior is the same for all Java versions we support, delete this.
542   enum JavaVersion {
543     JAVA6 {
544       @Override
545       GenericArrayType newArrayType(Type componentType) {
546         return new GenericArrayTypeImpl(componentType);
547       }
548 
549       @Override
550       Type usedInGenericType(Type type) {
551         checkNotNull(type);
552         if (type instanceof Class) {
553           Class<?> cls = (Class<?>) type;
554           if (cls.isArray()) {
555             return new GenericArrayTypeImpl(cls.getComponentType());
556           }
557         }
558         return type;
559       }
560     },
561     JAVA7 {
562       @Override
563       Type newArrayType(Type componentType) {
564         if (componentType instanceof Class) {
565           return getArrayClass((Class<?>) componentType);
566         } else {
567           return new GenericArrayTypeImpl(componentType);
568         }
569       }
570 
571       @Override
572       Type usedInGenericType(Type type) {
573         return checkNotNull(type);
574       }
575     },
576     JAVA8 {
577       @Override
578       Type newArrayType(Type componentType) {
579         return JAVA7.newArrayType(componentType);
580       }
581 
582       @Override
583       Type usedInGenericType(Type type) {
584         return JAVA7.usedInGenericType(type);
585       }
586 
587       @Override
588       String typeName(Type type) {
589         try {
590           Method getTypeName = Type.class.getMethod("getTypeName");
591           return (String) getTypeName.invoke(type);
592         } catch (NoSuchMethodException e) {
593           throw new AssertionError("Type.getTypeName should be available in Java 8");
594           /*
595            * Do not merge the 2 catch blocks below. javac would infer a type of
596            * ReflectiveOperationException, which Animal Sniffer would reject. (Old versions of
597            * Android don't *seem* to mind, but there might be edge cases of which we're unaware.)
598            */
599         } catch (InvocationTargetException e) {
600           throw new RuntimeException(e);
601         } catch (IllegalAccessException e) {
602           throw new RuntimeException(e);
603         }
604       }
605     },
606     JAVA9 {
607       @Override
608       Type newArrayType(Type componentType) {
609         return JAVA8.newArrayType(componentType);
610       }
611 
612       @Override
613       Type usedInGenericType(Type type) {
614         return JAVA8.usedInGenericType(type);
615       }
616 
617       @Override
618       String typeName(Type type) {
619         return JAVA8.typeName(type);
620       }
621 
622       @Override
623       boolean jdkTypeDuplicatesOwnerName() {
624         return false;
625       }
626     };

628     static final JavaVersion CURRENT;
629 
630     static {
631       if (AnnotatedElement.class.isAssignableFrom(TypeVariable.class)) {
632         if (new TypeCapture<Entry<String, int[][]>>() {}.capture()
633             .toString()
634             .contains("java.util.Map.java.util.Map")) {
635           CURRENT = JAVA8;
636         } else {
637           CURRENT = JAVA9;
638         }
639       } else if (new TypeCapture<int[]>() {}.capture() instanceof Class) {
640         CURRENT = JAVA7;
641       } else {
642         CURRENT = JAVA6;
643       }
644     }

651       ImmutableList.Builder<Type> builder = ImmutableList.builder();
652       for (Type type : types) {

655       return builder.build();

667   /**
668    * Per <a href="https://code.google.com/p/guava-libraries/issues/detail?id=1635">issue 1635</a>,
669    * In JDK 1.7.0_51-b13, {@link TypeVariableImpl#equals(Object)} is changed to no longer be equal
670    * to custom TypeVariable implementations. As a result, we need to make sure our TypeVariable
671    * implementation respects symmetry. Moreover, we don't want to reconstruct a native type variable
672    * {@code <A>} using our implementation unless some of its bounds have changed in resolution. This
673    * avoids creating unequal TypeVariable implementation unnecessarily. When the bounds do change,
674    * however, it's fine for the synthetic TypeVariable to be unequal to any native TypeVariable
675    * anyway.
676    */
677   static final class NativeTypeVariableEquals<X> {
678     static final boolean NATIVE_TYPE_VARIABLE_ONLY =
679         !NativeTypeVariableEquals.class.getTypeParameters()[0].equals(
680             newArtificialTypeVariable(NativeTypeVariableEquals.class, "X"));
681   }
