118     /**
119      * Gets a list of regular expressions of properties to exclude from the JSON
120      * output.
121      *
122      * @return A list of compiled regular expression patterns
123      */
124     public List<Pattern> getExcludePropertiesList() {
125         return this.excludeProperties;
126     }
127 
128     /**
129      * Sets a comma-delimited list of regular expressions to match properties
130      * that should be excluded from the JSON output.
131      *
132      * @param commaDelim A comma-delimited list of regular expressions
133      */
134     public void setExcludeProperties(String commaDelim) {
135         Set<String> excludePatterns = JSONUtil.asSet(commaDelim);
136         if (excludePatterns != null) {
137             this.excludeProperties = new ArrayList<>(excludePatterns.size());
138             for (String pattern : excludePatterns) {
139                 this.excludeProperties.add(Pattern.compile(pattern));
140             }
141         }
142     }
143 
144     /**
145      * Sets a comma-delimited list of wildcard expressions to match properties
146      * that should be excluded from the JSON output.
147      *
148      * @param commaDelim A comma-delimited list of wildcard patterns
149      */
150     public void setExcludeWildcards(String commaDelim) {
151         Set<String> excludePatterns = JSONUtil.asSet(commaDelim);
152         if (excludePatterns != null) {
153             this.excludeProperties = new ArrayList<>(excludePatterns.size());
154             for (String pattern : excludePatterns) {
155                 this.excludeProperties.add(WildcardUtil.compileWildcardPattern(pattern));
156             }
157         }
158     }

187     public void execute(ActionInvocation invocation) throws Exception {
188         ActionContext actionContext = invocation.getInvocationContext();
189         HttpServletRequest request = actionContext.getServletRequest();
190         HttpServletResponse response = actionContext.getServletResponse();
191 
192         // only permit caching bean information when struts devMode = false
193         cacheBeanInfo = !devMode;
194 
195         try {
196             Object rootObject;
197             rootObject = readRootObject(invocation);
198             writeToResponse(response, createJSONString(request, rootObject), enableGzip(request));
199         } catch (IOException exception) {
200             LOG.error(exception.getMessage(), exception);
201             throw exception;
202         }
203     }
204 
205     protected Object readRootObject(ActionInvocation invocation) {
206         if (enableSMD) {
207             return buildSMDObject(invocation);
208         }
209         return findRootObject(invocation);
210     }
211 
212     protected Object findRootObject(ActionInvocation invocation) {
213         ValueStack stack = invocation.getStack();
214         Object rootObject;
215         if (this.root != null) {
216             LOG.debug("Root was defined as [{}], searching stack for it", this.root);
217             rootObject = stack.findValue(root);
218         } else {
219             LOG.debug("Root was not defined, searching for #action");
220             rootObject = stack.findValue("#action");
221             if (rootObject instanceof ModelDriven) {
222                 LOG.debug("Action is an instance of ModelDriven, assuming model is on the top of the stack and using it");
223                 rootObject = stack.peek();
224             } else if (rootObject == null) {
225                 LOG.debug("Neither #action nor ModelDriven, peeking up object from the top of the stack");
226                 rootObject = stack.peek();
227             }
228         }
229         return rootObject;
230     }
231 
232     protected String createJSONString(HttpServletRequest request, Object rootObject) throws JSONException {
233         String json = jsonUtil.serialize(rootObject, excludeProperties, includeProperties, ignoreHierarchy,
234             enumAsBean, excludeNullProperties, defaultDateFormat, cacheBeanInfo);
235         json = addCallbackIfApplicable(request, json);
236         return json;
237     }
238 
239     protected boolean enableGzip(HttpServletRequest request) {
240         return enableGZIP && JSONUtil.isGzipInRequest(request);
241     }
242 
243     protected void writeToResponse(HttpServletResponse response, String json, boolean gzip) throws IOException {
244         JSONUtil.writeJSONToResponse(new SerializationParams(response, getEncoding(), isWrapWithComments(),
245             json, false, gzip, noCache, statusCode, errorCode, prefix, contentType, wrapPrefix,
246             wrapSuffix));
247     }
248 
249     protected org.apache.struts2.json.smd.SMD buildSMDObject(ActionInvocation invocation) {
250         return new SMDGenerator(findRootObject(invocation), excludeProperties, ignoreInterfaces).generate(invocation);
251     }

253     /**
254      * Retrieve the encoding
255      *
256      * @return The encoding associated with this template (defaults to the value
257      * of param 'encoding', if empty default to 'struts.i18n.encoding' property)
258      */
259     protected String getEncoding() {
260         String encoding = this.encoding;
261 
262         if (encoding == null) {
263             encoding = this.defaultEncoding;
264         }
265 
266         if (encoding == null) {
267             encoding = System.getProperty("file.encoding");
268         }
269 
270         if (encoding == null) {
271             encoding = "UTF-8";
272         }
273 
274         return encoding;
275     }
276 
277     protected String addCallbackIfApplicable(HttpServletRequest request, String json) {
278         if ((callbackParameter != null) && (callbackParameter.length() > 0)) {
279             String callbackName = request.getParameter(callbackParameter);
280             if (StringUtils.isNotEmpty(callbackName)) {
281                 json = callbackName + "(" + json + ")";
282             }
283         }
284         return json;
285     }
