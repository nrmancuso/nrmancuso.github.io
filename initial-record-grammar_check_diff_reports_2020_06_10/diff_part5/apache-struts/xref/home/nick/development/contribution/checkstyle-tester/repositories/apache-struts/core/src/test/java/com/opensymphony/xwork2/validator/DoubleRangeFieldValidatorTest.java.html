53      public void testRangeValidationWithError() throws Exception {
54          //Explicitly set an out-of-range double for DoubleRangeValidatorTest
55          Map<String, Object> context = new HashMap<>();
56          HashMap<String, Object> params = new HashMap<>();
57          params.put("percentage", 100.12);
58          context.put(ActionContext.PARAMETERS, HttpParameters.create(params).build());
59  
60          ActionProxy proxy = actionProxyFactory.createActionProxy("", MockConfigurationProvider.VALIDATION_ACTION_NAME, null, context);
61          proxy.execute();
62          assertTrue(((ValidationAware) proxy.getAction()).hasFieldErrors());
63  
64          Map<String, List<String>> errors = ((ValidationAware) proxy.getAction()).getFieldErrors();
65  
66          List<String> errorMessages = errors.get("percentage");
67          assertNotNull("Expected double range validation error message.", errorMessages);
68          assertEquals(1, errorMessages.size());
69  
70          String errorMessage = errorMessages.get(0);
71          assertNotNull("Expecting: percentage must be between 0.1 and 10.1, current value is 100.12.", errorMessage);
72          assertEquals("percentage must be between 0.1 and 10.1, current value is 100.12.", errorMessage);
73      }

222     public void testExpressionParams() throws Exception {
223         ValueStack stack = ActionContext.getContext().getValueStack();
224         ActionSupport action = new ActionSupport() {
225 
226             public Double getMinInclusiveValue() {
227                 return 10d;
228             }
229 
230             public Double getMaxInclusiveValue() {
231                 return 11d;
232             }
233 
234             public Double getMinExclusiveValue() {
235                 return 13d;
236             }
237 
238             public Double getMaxExclusiveValue() {
239                 return 14d;
240             }
241 
242             public Double getPrice() {
243                 return 15d;
244             }
245         };
246 
247         stack.push(action);
248 
249         val.setMinInclusiveExpression("${minInclusiveValue}");
250         val.setMaxInclusiveExpression("${maxInclusiveValue}");
251         val.setMinExclusiveExpression("${minExclusiveValue}");
252         val.setMaxExclusiveExpression("${maxExclusiveValue}");
253 
254         val.setFieldName("price");
255         val.setDefaultMessage("Price is wrong!");
256 
257         DelegatingValidatorContext context = new DelegatingValidatorContext(action, tpf);
258         val.setValidatorContext(context);
259 
260         val.validate(action);
261         assertTrue(action.getFieldErrors().get("price").size() == 1);
262     }
