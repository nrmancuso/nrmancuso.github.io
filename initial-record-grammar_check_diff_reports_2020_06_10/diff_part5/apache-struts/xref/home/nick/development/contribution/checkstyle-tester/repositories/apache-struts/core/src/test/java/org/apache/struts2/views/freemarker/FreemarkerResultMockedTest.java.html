241     private void init() {
242         stringWriter = new StringWriter();
243         writer = new PrintWriter(stringWriter);
244         response = new StrutsMockHttpServletResponse();
245         response.setWriter(writer);
246         request = new MockHttpServletRequest();
247         stack = ActionContext.getContext().getValueStack();
248 
249         context = ActionContext.of(stack.getContext())
250             .withServletResponse(response)
251             .withServletRequest(request)
252             .withServletContext(servletContext)
253             .bind();
254 
255         servletContext.setAttribute(FreemarkerManager.CONFIG_SERVLET_CONTEXT_KEY, null);
256 
257         invocation = new MockActionInvocation();
258         invocation.setStack(stack);
259         invocation.setInvocationContext(context);
260 
261         //get fm config to use it in mock servlet context
262         FreemarkerManager freemarkerManager = container.getInstance(FreemarkerManager.class);
263 
264         freemarkerConfig = freemarkerManager.getConfiguration(ServletActionContext.getServletContext());
265         freemarkerConfig.setTemplateExceptionHandler(TemplateExceptionHandler.RETHROW_HANDLER);
266         freemarkerConfig.setServletContextForTemplateLoading(servletContext, null);
267     }
