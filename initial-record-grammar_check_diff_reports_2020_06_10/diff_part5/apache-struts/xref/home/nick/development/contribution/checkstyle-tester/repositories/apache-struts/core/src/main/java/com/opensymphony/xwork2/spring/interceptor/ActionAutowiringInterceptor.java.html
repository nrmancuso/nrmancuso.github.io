19  package com.opensymphony.xwork2.spring.interceptor;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.ActionInvocation;
23  import com.opensymphony.xwork2.interceptor.AbstractInterceptor;
24  import com.opensymphony.xwork2.spring.SpringObjectFactory;
25  import org.apache.logging.log4j.LogManager;
26  import org.apache.logging.log4j.Logger;
27  import org.springframework.beans.BeansException;
28  import org.springframework.context.ApplicationContext;
29  import org.springframework.context.ApplicationContextAware;
30  import org.springframework.web.context.WebApplicationContext;
31  
32  /**
33   * <!-- START SNIPPET: description -->

35   * <!-- END SNIPPET: description -->
36   *
37   * <!-- START SNIPPET: parameters -->
38   * TODO: Describe the parameters for this Interceptor.
39   * <!-- END SNIPPET: parameters -->
40   *
41   * <!-- START SNIPPET: extending -->
42   * TODO: Discuss some possible extension of the Interceptor.
43   * <!-- END SNIPPET: extending -->
44   *
45   * <pre>
46   * <!-- START SNIPPET: example -->
47   * &lt;!-- TODO: Describe how the Interceptor reference will effect execution --&gt;
48   * &lt;action name="someAction" class="com.examples.SomeAction"&gt;
49   *      TODO: fill in the interceptor reference.
50   *     &lt;interceptor-ref name=""/&gt;
51   *     &lt;result name="success"&gt;good_result.ftl&lt;/result&gt;
52   * &lt;/action&gt;
53   * <!-- END SNIPPET: example -->
54   * </pre>
55   *
56   * <p>
57   * Autowires action classes to Spring beans.  The strategy for autowiring the beans can be configured
58   * by setting the parameter on the interceptor.  Actions that need access to the <code>ActionContext</code>
59   * can implements the <code>ApplicationContextAware</code> interface.  The context will also be placed on
60   * the action context under the APPLICATION_CONTEXT attribute.
61   * </p>
62   *
63   * @author Simon Stewart
64   * @author Eric Hauser
65   */
66  public class ActionAutowiringInterceptor extends AbstractInterceptor implements ApplicationContextAware {
67  
68      private static final Logger LOG = LogManager.getLogger(ActionAutowiringInterceptor.class);
69  
70      private boolean initialized = false;
71      private ApplicationContext context;
72      private SpringObjectFactory factory;
73      private Integer autowireStrategy;
74  
75      /**
76       * @param autowireStrategy the autowire strategy
77       */
78      public void setAutowireStrategy(Integer autowireStrategy) {
79          this.autowireStrategy = autowireStrategy;
80      }
81  
82      /**
83       * <p>
84       * Looks for the <code>ApplicationContext</code> under the attribute that the Spring listener sets in
85       * the servlet context.  The configuration is done the first time here instead of in init() since the
86       * <code>ActionContext</code> is not available during <code>Interceptor</code> initialization.
87       * </p>
88       *
89       * <p>
90       * Autowires the action to Spring beans and places the <code>ApplicationContext</code>
91       * on the <code>ActionContext</code>
92       * </p>
93       *
94       * <p>
95       * TODO: Should this check to see if the <code>SpringObjectFactory</code> has already been configured instead of instantiating a new one?  Or is there a good reason for the interceptor to have it's own factory?
96       * </p>
97       *
98       * @param invocation the action invocation
99       * @throws Exception in case of any errors
100      */
101     @Override public String intercept(ActionInvocation invocation) throws Exception {
102         if (!initialized) {
103             ApplicationContext applicationContext = (ApplicationContext) ActionContext.getContext().getApplication().get(
104                     WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE);
105 
106             if (applicationContext == null) {
107                 LOG.warn("ApplicationContext could not be found.  Action classes will not be autowired.");
108             } else {
109                 setApplicationContext(applicationContext);
110                 factory = new SpringObjectFactory();
111                 factory.setContainer(ActionContext.getContext().getContainer());
112                 factory.setApplicationContext(getApplicationContext());
113                 if (autowireStrategy != null) {
114                     factory.setAutowireStrategy(autowireStrategy);
115                 }
116             }
117             initialized = true;
118         }
119 
120         if (factory != null) {
121             Object bean = invocation.getAction();
122             factory.autoWireBean(bean);
123         }
124         return invocation.invoke();
125     }
126 
127     /**
128      * @param applicationContext the application context
129      * @throws BeansException in case of errors
130      */
131     public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
132         context = applicationContext;
133     }
134 
135     /**
136      * @return the application context
137      */
138     protected ApplicationContext getApplicationContext() {
139         return context;
140     }
141 
142 }
