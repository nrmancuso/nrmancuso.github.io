19  package org.apache.struts2.osgi.host;
20  
21  import org.apache.logging.log4j.Logger;
22  import org.apache.logging.log4j.LogManager;
23  import org.apache.felix.shell.ShellService;
24  import org.osgi.framework.Bundle;
25  import org.osgi.framework.BundleContext;
26  import org.osgi.framework.BundleException;
27  import org.osgi.framework.BundleReference;
28  
29  import javax.servlet.ServletContext;
30  import java.net.URL;
31  import java.security.CodeSource;
32  import java.security.ProtectionDomain;
33  import java.util.ArrayList;
34  import java.util.Collections;
35  import java.util.HashMap;
36  import java.util.List;
37  import java.util.Map;
38  
39  /**
40   * A glassfish implementation of an OsgiHost
41   */
42  public class GlassfishOSGiHost extends BaseOsgiHost implements OsgiHost {
43  
44      private static final Logger LOG = LogManager.getLogger(GlassfishOSGiHost.class);
45  
46      /**
47       * Location inside the WAR where initial bundles are located.
48       */
49      private static final String BUNDLES_DIR = "/WEB-INF/classes/bundles/2/";
50  
51      private BundleContext bctx = null;
52  
53      @Override
54      public void init(ServletContext servletContext) {
55          this.servletContext = servletContext;
56  
57          installManagedBundles();
58  
59          addSpringOSGiSupport();
60  
61          // add the bundle context to the ServletContext
62          servletContext.setAttribute(OSGI_BUNDLE_CONTEXT, bctx);
63      }
64  
65      private void installManagedBundles() {
66          try {
67  
68              // Obtaining BundleContext from ServletContext class which is loaded
69              // by bundle class loader
70              BundleReference ref = (BundleReference) servletContext.getClass()
71                      .getClassLoader();
72              bctx = ref.getBundle().getBundleContext();
73  
74              // installing managed bundles
75              installBundles();
76  
77          } catch (Exception ex) {
78              LOG.error("Installing Managed Bundles met a problem", ex);
79          }
80      }
81  
82      private void installBundles() throws Exception {
83          ArrayList<Bundle> installed = new ArrayList<Bundle>();
84          for (URL url : findBundles()) {
85              LOG.debug("Installing bundle [" + url + "]");
86              Bundle bundle = bctx.installBundle(url.toExternalForm());
87              installed.add(bundle);
88          }
89          for (Bundle bundle : installed) {
90              try {
91                  bundle.start();
92              } catch (BundleException e) {
93                  e.printStackTrace();
94                  LOG.error("Failed to start " + bundle, e);
95              }
96          }
97  
98      }
99  
100     private List<URL> findBundles() throws Exception {
101         ArrayList<URL> list = new ArrayList<URL>();
102         for (Object o : this.servletContext.getResourcePaths(BUNDLES_DIR)) {
103             String name = (String) o;
104             if (name.endsWith(".jar")) {
105                 URL url = this.servletContext.getResource(name);
106                 if (url != null) {
107                     list.add(url);
108                 }
109             }
110         }
111 
112         ProtectionDomain protectionDomain = ShellService.class.getProtectionDomain();
113         CodeSource codeSource = protectionDomain.getCodeSource();
114         URL loc = codeSource.getLocation();
115         list.add(loc);
116 
117         return list;
118     }
119 
120     @Override
121     protected void addSpringOSGiSupport() {
122         // see the javadoc for
123         // org.springframework.osgi.web.context.support.OsgiBundleXmlWebApplicationContext
124         // for more details
125         // OsgiBundleXmlWebApplicationContext expects the the BundleContext to
126         // be set in the ServletContext under the attribute
127         // OsgiBundleXmlWebApplicationContext.BUNDLE_CONTEXT_ATTRIBUTE
128         try {
129             Class<?> clazz = Class
130                     .forName("org.springframework.osgi.web.context.support.OsgiBundleXmlWebApplicationContext");
131             String key = (String) clazz.getDeclaredField(
132                     "BUNDLE_CONTEXT_ATTRIBUTE").get(null);
133             servletContext.setAttribute(key, bctx);
134         } catch (ClassNotFoundException e) {
135             if (LOG.isDebugEnabled()) {
136                 LOG.debug("Spring OSGi support is not enabled");
137             }
138         } catch (Exception e) {
139             LOG.error(
140                         "The API of Spring OSGi has changed and the field [[}] is no longer available. The OSGi plugin needs to be updated",
141                         "org.springframework.osgi.web.context.support.OsgiBundleXmlWebApplicationContext.BUNDLE_CONTEXT_ATTRIBUTE", e);
142         }
143     }
144 
145     /**
146      * This bundle map will not change, but the status of the bundles can change
147      * over time. Use getActiveBundles() for active bundles
148      */
149     @Override
150     public Map<String, Bundle> getBundles() {
151         Map<String, Bundle> bundles = new HashMap<String, Bundle>();
152         for (Bundle bundle : bctx.getBundles()) {
153             bundles.put(bundle.getSymbolicName(), bundle);
154         }
155 
156         return Collections.unmodifiableMap(bundles);
157     }
158 
159     public Map<String, Bundle> getActiveBundles() {
160         Map<String, Bundle> bundles = new HashMap<String, Bundle>();
161         for (Bundle bundle : bctx.getBundles()) {
162             if (bundle.getState() == Bundle.ACTIVE)
163                 bundles.put(bundle.getSymbolicName(), bundle);
164         }
165 
166         return Collections.unmodifiableMap(bundles);
167     }
168 
169     public BundleContext getBundleContext() {
170         return bctx;
171     }
172 
173     public void destroy() throws Exception {
174     }
175 
176 }
