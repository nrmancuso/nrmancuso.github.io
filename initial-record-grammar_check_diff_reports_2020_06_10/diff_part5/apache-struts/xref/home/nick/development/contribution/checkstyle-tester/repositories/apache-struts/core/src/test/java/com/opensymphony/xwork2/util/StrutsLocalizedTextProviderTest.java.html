254     /**
255      * Unit test to confirm expected behaviour of "clearing methods" provided to
256      *   StrutsLocalizedTextProvider (from AbstractLocalizedTextProvider).
257      *
258      * @since 2.6
259      */
260     public void testLocalizedTextProviderClearingMethods() {
261         TestStrutsLocalizedTextProvider testStrutsLocalizedTextProvider = new TestStrutsLocalizedTextProvider();
262         assertTrue("testStrutsLocalizedTextProvider not instance of AbstractLocalizedTextProvider ?",
263                 testStrutsLocalizedTextProvider instanceof AbstractLocalizedTextProvider);
264         assertEquals("testStrutsLocalizedTextProvider starting default bundle map size not 0 before any retrievals ?",
265                 0, testStrutsLocalizedTextProvider.currentBundlesMapSize());
266 
267         // Access the two default bundles to populate their cache entries and test bundle map size.
268         ResourceBundle tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
269                 TestStrutsLocalizedTextProvider.XWORK_MESSAGES_BUNDLE, Locale.ENGLISH);
270         assertNotNull("XWORK_MESSAGES_BUNDLE retrieval null ?", tempBundle);
271         tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
272                 TestStrutsLocalizedTextProvider.STRUTS_MESSAGES_BUNDLE, Locale.ENGLISH);
273         assertNotNull("STRUTS_MESSAGES_BUNDLE retrieval null ?", tempBundle);
274         assertEquals("testStrutsLocalizedTextProvider bundle map size not 2 after retrievals ?",
275                 2, testStrutsLocalizedTextProvider.currentBundlesMapSize());
276 
277         // Add and then access four test bundles to populate their cache entries and test bundle map size.
278         testStrutsLocalizedTextProvider.addDefaultResourceBundle("com/opensymphony/xwork2/util/LocalizedTextUtilTest");
279         testStrutsLocalizedTextProvider.addDefaultResourceBundle("com/opensymphony/xwork2/util/FindMe");
280         testStrutsLocalizedTextProvider.addDefaultResourceBundle("com/opensymphony/xwork2/SimpleAction");
281         testStrutsLocalizedTextProvider.addDefaultResourceBundle("com/opensymphony/xwork2/test");
282         tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
283                 "com/opensymphony/xwork2/util/LocalizedTextUtilTest", Locale.ENGLISH);
284         assertNotNull("com/opensymphony/xwork2/util/LocalizedTextUtilTest retrieval null ?", tempBundle);
285         tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
286                 "com/opensymphony/xwork2/util/FindMe", Locale.ENGLISH);
287         assertNotNull("com/opensymphony/xwork2/util/FindMe retrieval null ?", tempBundle);
288         tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
289                 "com/opensymphony/xwork2/SimpleAction", Locale.ENGLISH);
290         assertNotNull("com/opensymphony/xwork2/SimpleAction retrieval null ?", tempBundle);
291         tempBundle = testStrutsLocalizedTextProvider.findResourceBundle(
292                 "com/opensymphony/xwork2/test", Locale.ENGLISH);
293         assertNotNull("com/opensymphony/xwork2/test retrieval null ?", tempBundle);
294         assertEquals("testStrutsLocalizedTextProvider bundle map size not 6 after retrievals ?",
295                 6, testStrutsLocalizedTextProvider.currentBundlesMapSize());
296 
297         // Expect the call to be ineffective due to deprecation and change to a "no-op" (but shouldn't throw an Exception or cause failure).
298         testStrutsLocalizedTextProvider.callClearBundleNoLocale("com/opensymphony/xwork2/test");
299         assertEquals("testStrutsLocalizedTextProvider bundle map size not 6 after non-locale clear call ?",
300                 6, testStrutsLocalizedTextProvider.currentBundlesMapSize());
301 
302         // Expect the call to function with bundle name + locale.  Remove all four of the non-default
303         //   bundles and confirm the bundle map size changes.
304         testStrutsLocalizedTextProvider.callClearBundleWithLocale("com/opensymphony/xwork2/test", Locale.ENGLISH);
305         assertEquals("testStrutsLocalizedTextProvider bundle map size not 5 after locale clear call ?",
306                 5, testStrutsLocalizedTextProvider.currentBundlesMapSize());
307         testStrutsLocalizedTextProvider.callClearBundleWithLocale("com/opensymphony/xwork2/SimpleAction", Locale.ENGLISH);
308         assertEquals("testStrutsLocalizedTextProvider bundle map size not 4 after locale clear call ?",
309                 4, testStrutsLocalizedTextProvider.currentBundlesMapSize());
310         testStrutsLocalizedTextProvider.callClearBundleWithLocale("com/opensymphony/xwork2/util/FindMe", Locale.ENGLISH);
311         assertEquals("testStrutsLocalizedTextProvider bundle map size not 3 after locale clear call ?",
312                 3, testStrutsLocalizedTextProvider.currentBundlesMapSize());
313         testStrutsLocalizedTextProvider.callClearBundleWithLocale("com/opensymphony/xwork2/util/LocalizedTextUtilTest", Locale.ENGLISH);
314         assertEquals("testStrutsLocalizedTextProvider bundle map size not 2 after locale clear call ?",
315                 2, testStrutsLocalizedTextProvider.currentBundlesMapSize());
316 
317         // Confirm the missing bundles cache clearing method does not produce any Exceptions or failures.
318         testStrutsLocalizedTextProvider.callClearMissingBundlesCache();
319     }
