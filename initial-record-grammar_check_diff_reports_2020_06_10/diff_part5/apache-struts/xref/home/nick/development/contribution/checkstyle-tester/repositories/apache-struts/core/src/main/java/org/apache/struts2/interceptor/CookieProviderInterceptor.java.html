19  package org.apache.struts2.interceptor;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.ActionInvocation;
23  import com.opensymphony.xwork2.interceptor.AbstractInterceptor;
24  import com.opensymphony.xwork2.interceptor.PreResultListener;
25  import org.apache.logging.log4j.Logger;
26  import org.apache.logging.log4j.LogManager;
27  import org.apache.struts2.StrutsStatics;
28  
29  import javax.servlet.http.Cookie;
30  import javax.servlet.http.HttpServletResponse;
31  import java.util.Set;
32  
33  /**
34   * <!-- START SNIPPET: description -->
35   * Allows actions to send cookies to client, action must implement {@link CookieProvider}
36   * You must reference this interceptor in your default stack or in action's stack, see example below.
37   *
38   * <!-- END SNIPPET: description -->
39   *
40   * <!-- START SNIPPET: parameters -->
41   *
42   * none
43   *
44   * <!-- END SNIPPET: parameters -->
45   *
46   * <!-- START SNIPPET: extending -->
47   *
48   * <ul>
49   *     <li>addCookiesToResponse - this method applies cookie created by action to response</li>
50   * </ul>
51   *
52   * <!-- END SNIPPET: extending -->
53   *
54   * <pre>
55   * <!-- START SNIPPET: example -->
56   *
57   * &lt;action ... &gt;
58   *   &lt;interceptor-ref name="defaultStack"/&gt;
59   *   &lt;interceptor-ref name="cookieProvider"/&gt;
60   *   ...
61   * &lt;/action&gt;
62   *
63   * <!-- END SNIPPET: example -->
64   * </pre>
65   */
66  public class CookieProviderInterceptor extends AbstractInterceptor implements PreResultListener {
67  
68      private static final Logger LOG = LogManager.getLogger(CookieProviderInterceptor.class);
69  
70      public String intercept(ActionInvocation invocation) throws Exception {
71          invocation.addPreResultListener(this);
72          return invocation.invoke();
73      }
74  
75      /**
76       * Do what name suggests
77       *
78       * @param action {@link CookieProvider} action
79       * @param response current {@link HttpServletResponse}
80       */
81      protected void addCookiesToResponse(CookieProvider action, HttpServletResponse response) {
82          Set<Cookie> cookies = action.getCookies();
83          if (cookies != null) {
84              for (Cookie cookie : cookies) {
85                  if (LOG.isDebugEnabled()) {
86                      LOG.debug("Sending cookie [{}] with value [{}] for domain [{}]",
87                              cookie.getName(), cookie.getValue(), (cookie.getDomain() != null ? cookie.getDomain() : "no domain"));
88                  }
89                  response.addCookie(cookie);
90              }
91          }
92      }
93  
94      public void beforeResult(ActionInvocation invocation, String resultCode) {
95          try {
96              LOG.trace("beforeResult start");
97              ActionContext ac = invocation.getInvocationContext();
98              if (invocation.getAction() instanceof CookieProvider) {
99                  HttpServletResponse response = ac.getServletResponse();
100                 addCookiesToResponse((CookieProvider) invocation.getAction(), response);
101             }
102             LOG.trace("beforeResult end");
103         } catch (Exception ex) {
104             LOG.error("Unable to setup cookies", ex);
105         }
106     }
107 
108 }
