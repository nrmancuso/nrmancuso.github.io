19  package org.apache.struts2.interceptor;
20  
21  import java.text.SimpleDateFormat;
22  import java.util.Date;
23  import java.util.HashMap;
24  import java.util.Map;
25  
26  import org.apache.struts2.StrutsInternalTestCase;
27  
28  import com.opensymphony.xwork2.ActionContext;
29  import com.opensymphony.xwork2.mock.MockActionInvocation;
30  import org.apache.struts2.dispatcher.HttpParameters;
31  
32  /**
33   * Unit test for DateTextFieldInterceptor. 
34   */
35  public class DateTextFieldInterceptorTest extends StrutsInternalTestCase {
36  
37      private DateTextFieldInterceptor interceptor;
38      private MockActionInvocation ai;
39      private Map<String, Object> param;
40      
41      protected void setUp() throws Exception {
42      	super.setUp();
43      	param = new HashMap<>();
44      	
45      	interceptor = new DateTextFieldInterceptor();
46      	ai = new MockActionInvocation();
47      	ai.setInvocationContext(ActionContext.getContext());
48      }
49  	
50  	public void testNoParam() throws Exception {
51  		ActionContext.getContext().setParameters(HttpParameters.create(param).build());
52  
53  		interceptor.init();
54  		interceptor.intercept(ai);
55  		interceptor.destroy();
56  
57  		assertEquals(0, ai.getInvocationContext().getParameters().keySet().size());
58  	}
59  
60  	public void testOneDateTextField() throws Exception {
61  		param.put("__year_name", new String[]{"2000"});
62  		param.put("__month_name", new String[]{"06"});
63  		param.put("__day_name", new String[]{"15"});
64  
65  		ActionContext.getContext().setParameters(HttpParameters.create(param).build());
66  
67  		interceptor.init();
68  		interceptor.intercept(ai);
69  		interceptor.destroy();
70  
71  		HttpParameters parameters = ai.getInvocationContext().getParameters();
72  		assertFalse(parameters.contains("__year_name"));
73  		assertFalse(parameters.contains("__month_name"));
74  		assertFalse(parameters.contains("__day_name"));
75  		assertTrue(parameters.contains("name"));
76  		assertEquals(1, parameters.keySet().size());
77  		Date date = new SimpleDateFormat("yyyy-MM-dd").parse("2000-06-15"); 
78  		assertEquals(date, parameters.get("name").getObject());
79  	}
80  
81  }
