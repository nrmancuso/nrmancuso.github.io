324

326 
327         // serialize and compare
328         List list = action.getList();
329

331         assertEquals(list.size(), 10);

334         assertNotNull(list);

370         assertEquals(action.getFoo(), "foo");
371

374         assertNotNull(map);
375         assertEquals(map.size(), 2);
376         assertEquals(map.get("a"), 1L);

378         assertNotNull(list);

513     @Override
514     protected void setUp() throws Exception {
515         super.setUp();
516 
517         this.request = new MockHttpServletRequest();
518         this.response = new MockHttpServletResponse();
519 
520         MockServletContext servletContext = new MockServletContext();
521 
522         ActionContext context = ActionContext.getContext()
523             .withServletRequest(request)
524             .withServletResponse(response)
525             .withServletContext(servletContext);
526 
527         ValueStack stack = context.getValueStack();
528 
529         this.invocation = new MockActionInvocationEx();
530         this.invocation.setInvocationContext(context);
531         this.invocation.setStack(stack);
532     }
