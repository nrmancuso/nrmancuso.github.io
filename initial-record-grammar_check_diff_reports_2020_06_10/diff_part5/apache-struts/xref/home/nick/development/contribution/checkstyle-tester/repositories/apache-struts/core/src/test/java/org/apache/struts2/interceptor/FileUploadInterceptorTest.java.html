267     public void testSuccessUploadOfATextFileMultipartRequest() throws Exception {
268         MockHttpServletRequest req = new MockHttpServletRequest();
269         req.setCharacterEncoding(StandardCharsets.UTF_8.name());
270         req.setMethod("post");
271         req.addHeader("Content-type", "multipart/form-data; boundary=---1234");
272 
273         // inspired by the unit tests for jakarta commons fileupload
274         String content = ("-----1234\r\n" +
275                 "Content-Disposition: form-data; name=\"file\"; filename=\"deleteme.txt\"\r\n" +
276                 "Content-Type: text/html\r\n" +
277                 "\r\n" +
278                 "Unit test of FileUploadInterceptor" +
279                 "\r\n" +
280                 "-----1234--\r\n");
281         req.setContent(content.getBytes("US-ASCII"));
282 
283         MyFileupAction action = new MyFileupAction();
284 
285         MockActionInvocation mai = new MockActionInvocation();
286         mai.setAction(action);
287         mai.setResultCode("success");
288         mai.setInvocationContext(ActionContext.getContext());
289         Map<String, Object> param = new HashMap<>();
290         ActionContext.getContext().setParameters(HttpParameters.create(param).build());
291         ActionContext.getContext().put(ServletActionContext.HTTP_REQUEST, createMultipartRequest(req, 2000));
292 
293         interceptor.intercept(mai);
294 
295         assertTrue(!action.hasErrors());
296 
297         HttpParameters parameters = mai.getInvocationContext().getParameters();
298         assertTrue(parameters.keySet().size() == 3);
299         UploadedFile[] files = (UploadedFile[]) parameters.get("file").getObject();
300         String[] fileContentTypes = parameters.get("fileContentType").getMultipleValues();
301         String[] fileRealFilenames = parameters.get("fileFileName").getMultipleValues();
302 
303         assertNotNull(files);
304         assertNotNull(fileContentTypes);
305         assertNotNull(fileRealFilenames);
306         assertTrue(files.length == 1);
307         assertTrue(fileContentTypes.length == 1);
308         assertTrue(fileRealFilenames.length == 1);
309         assertEquals("text/html", fileContentTypes[0]);
310         assertNotNull("deleteme.txt", fileRealFilenames[0]);
311     }

313     /**
314      * tests whether with multiple files sent with the same name, the ones with forbiddenTypes (see
315      * FileUploadInterceptor.setAllowedTypes(...) ) are sorted out.
316      *
317      * @throws Exception
318      */
319     public void testMultipleAccept() throws Exception {
320         final String htmlContent = "<html><head></head><body>html content</body></html>";
321         final String plainContent = "plain content";
322         final String bondary = "simple boundary";
323         final String endline = "\r\n";
324 
325         MockHttpServletRequest req = new MockHttpServletRequest();
326         req.setCharacterEncoding(StandardCharsets.UTF_8.name());
327         req.setMethod("POST");
328         req.addHeader("Content-type", "multipart/form-data; boundary=" + bondary);
329         StringBuilder content = new StringBuilder(128);
330         content.append(encodeTextFile(bondary, endline, "file", "test.html", "text/plain", plainContent));
331         content.append(encodeTextFile(bondary, endline, "file", "test1.html", "text/html", htmlContent));
332         content.append(encodeTextFile(bondary, endline, "file", "test2.html", "text/html", htmlContent));
333         content.append(endline);
334         content.append(endline);
335         content.append(endline);
336         content.append("--");
337         content.append(bondary);
338         content.append("--");
339         content.append(endline);
340         req.setContent(content.toString().getBytes());
341 
342         assertTrue(ServletFileUpload.isMultipartContent(req));
343 
344         MyFileupAction action = new MyFileupAction();
345         container.inject(action);
346         MockActionInvocation mai = new MockActionInvocation();
347         mai.setAction(action);
348         mai.setResultCode("success");
349         mai.setInvocationContext(ActionContext.getContext());
350         Map<String, Object> param = new HashMap<String, Object>();
351         ActionContext.getContext().setParameters(HttpParameters.create(param).build());
352         ActionContext.getContext().put(ServletActionContext.HTTP_REQUEST, createMultipartRequest(req, 2000));
353 
354         interceptor.setAllowedTypes("text/html");
355         interceptor.intercept(mai);
356 
357         HttpParameters parameters = mai.getInvocationContext().getParameters();
358         assertEquals(3, parameters.keySet().size());
359         UploadedFile[] files = (UploadedFile[]) parameters.get("file").getObject();
360         String[] fileContentTypes = parameters.get("fileContentType").getMultipleValues();
361         String[] fileRealFilenames = parameters.get("fileFileName").getMultipleValues();
362 
363         assertNotNull(files);
364         assertNotNull(fileContentTypes);
365         assertNotNull(fileRealFilenames);
366         assertEquals("files accepted ", 2, files.length);
367         assertEquals(2, fileContentTypes.length);
368         assertEquals(2, fileRealFilenames.length);
369         assertEquals("text/html", fileContentTypes[0]);
370         assertNotNull("test1.html", fileRealFilenames[0]);
371     }

412     private String encodeTextFile(String bondary, String endline, String name, String filename, String contentType, String content) {
413         final StringBuilder sb = new StringBuilder(64);
414         sb.append(endline);
415         sb.append("--");
416         sb.append(bondary);
417         sb.append(endline);
418         sb.append("Content-Disposition: form-data; name=\"");
419         sb.append(name);
420         sb.append("\"; filename=\"");
421         sb.append(filename);
422         sb.append(endline);
423         sb.append("Content-Type: ");
424         sb.append(contentType);
425         sb.append(endline);
426         sb.append(endline);
427         sb.append(content);
428 
429         return sb.toString();
430     }

432     private MultiPartRequestWrapper createMultipartRequest(HttpServletRequest req, int maxsize) throws IOException {
433         JakartaMultiPartRequest jak = new JakartaMultiPartRequest();
434         jak.setMaxSize(String.valueOf(maxsize));
435         return new MultiPartRequestWrapper(jak, req, tempDir.getAbsolutePath(), new DefaultLocaleProvider());
436     }
