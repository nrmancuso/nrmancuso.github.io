38  public class DateRangeValidatorTest extends XWorkTestCase {
39  
40      /**
41       * Tests whether the date range validation is working. Should produce an validation error,
42       * because the action config sets date to 12/20/2002 while expected range is Dec 22-25.
43       */
44      public void testRangeValidation() throws Exception {
45          Calendar date = Calendar.getInstance();
46          date.set(2002, Calendar.NOVEMBER, 20);
47          Map<String, Object> context = new HashMap<>();
48          HashMap<String, Object> params = new HashMap<>();
49          params.put("date", date.getTime());
50          context.put(ActionContext.PARAMETERS, HttpParameters.create(params).build());
51          context.put(ActionContext.LOCALE, Locale.US);  // Force US Locale for date conversion tests on JDK9+
52  
53          ActionProxy proxy = actionProxyFactory.createActionProxy("", MockConfigurationProvider.VALIDATION_ACTION_NAME, null, context);
54          proxy.execute();
55          assertTrue(((ValidationAware) proxy.getAction()).hasFieldErrors());
56  
57          Map<String, List<String>> errors = ((ValidationAware) proxy.getAction()).getFieldErrors();
58  
59          List<String> errorMessages = errors.get("date");
60          assertNotNull("Expected date range validation error message.", errorMessages);
61          assertEquals(1, errorMessages.size());
62  
63          String errorMessage = errorMessages.get(0);
64          assertEquals("The date must be between 12-22-2002 and 12-25-2002.", errorMessage);
65      }
66  
67      public void testGetSetMinMax() throws Exception {
68          DateRangeFieldValidator val = new DateRangeFieldValidator();
69          Date max = new Date();
70          val.setMax(max);
71          assertEquals(max, val.getMax());
72  
73          Date min = new Date();
74          val.setMin(min);
75          assertEquals(min, val.getMin());
76      }
