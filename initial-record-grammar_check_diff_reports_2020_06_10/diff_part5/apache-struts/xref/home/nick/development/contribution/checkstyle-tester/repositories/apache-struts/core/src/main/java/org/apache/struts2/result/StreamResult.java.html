19  package org.apache.struts2.result;
20  
21  import com.opensymphony.xwork2.ActionInvocation;
22  import org.apache.logging.log4j.LogManager;
23  import org.apache.logging.log4j.Logger;
24  
25  import javax.servlet.http.HttpServletResponse;
26  import java.io.InputStream;
27  import java.io.OutputStream;
28  
29  /**
30   * A custom Result type for sending raw data (via an InputStream) directly to the
31   * HttpServletResponse. Very useful for allowing users to download content.
32   *
33   * <b>This result type takes the following parameters:</b>
34   *
35   * <ul>
36   * <li><b>contentType</b> - the stream mime-type as sent to the web browser
37   * (default = <code>text/plain</code>).</li>
38   * <li><b>contentLength</b> - the stream length in bytes (the browser displays a
39   * progress bar).</li>
40   * <li><b>contentDisposition</b> - the content disposition header value for
41   * specifing the file name (default = <code>inline</code>, values are typically
42   * <i>attachment;filename="document.pdf"</i>.</li>
43   * <li><b>inputName</b> - the name of the InputStream property from the chained
44   * action (default = <code>inputStream</code>).</li>
45   * <li><b>bufferSize</b> - the size of the buffer to copy from input to output
46   * (default = <code>1024</code>).</li>
47   * <li><b>allowCaching</b> if set to 'false' it will set the headers 'Pragma' and 'Cache-Control'
48   * to 'no-cahce', and prevent client from caching the content. (default = <code>true</code>)
49   * <li><b>contentCharSet</b> if set to a string, ';charset=value' will be added to the
50   * content-type header, where value is the string set. If set to an expression, the result
51   * of evaluating the expression will be used. If not set, then no charset will be set on
52   * the header</li>
53   * </ul>
54   *
55   * <p>
56   * These parameters can also be set by exposing a similarly named getter method on your Action.  For example, you can
57   * provide <code>getContentType()</code> to override that parameter for the current action.
58   * </p>
59   *
60   * <b>Example:</b>
61   *
62   * <pre>
63   * &lt;result name="success" type="stream"&gt;
64   *   &lt;param name="contentType"&gt;image/jpeg&lt;/param&gt;
65   *   &lt;param name="inputName"&gt;imageStream&lt;/param&gt;
66   *   &lt;param name="contentDisposition"&gt;attachment;filename="document.pdf"&lt;/param&gt;
67   *   &lt;param name="bufferSize"&gt;1024&lt;/param&gt;
68   * &lt;/result&gt;
69   * </pre>
70   */
71  public class StreamResult extends StrutsResultSupport {
72  
73      private static final long serialVersionUID = -1468409635999059850L;
74  
75      protected static final Logger LOG = LogManager.getLogger(StreamResult.class);
76  
77      public static final String DEFAULT_PARAM = "inputName";
78  
79      protected String contentType = "text/plain";
80      protected String contentLength;
81      protected String contentDisposition = "inline";
82      protected String contentCharSet;
83      protected String inputName = "inputStream";
84      protected InputStream inputStream;
85      protected int bufferSize = 1024;
86      protected boolean allowCaching = true;
87  
88      public StreamResult() {
89          super();
90      }
91  
92      public StreamResult(InputStream in) {
93          this.inputStream = in;
94      }
95  
96      /**
97       * @return Returns the whether or not the client should be requested to allow caching of the data stream.
98       */
99      public boolean getAllowCaching() {
100         return allowCaching;
101     }
102 
103     /**
104      * Set allowCaching to <tt>false</tt> to indicate that the client should be requested not to cache the data stream.
105      * This is set to <tt>false</tt> by default
106      *
107      * @param allowCaching Enable caching.
108      */
109     public void setAllowCaching(boolean allowCaching) {
110         this.allowCaching = allowCaching;
111     }
112 
113 
114     /**
115      * @return Returns the bufferSize.
116      */
117     public int getBufferSize() {
118         return (bufferSize);
119     }
120 
121     /**
122      * @param bufferSize The bufferSize to set.
123      */
124     public void setBufferSize(int bufferSize) {
125         this.bufferSize = bufferSize;
126     }
127 
128     /**
129      * @return Returns the contentType.
130      */
131     public String getContentType() {
132         return (contentType);
133     }
134 
135     /**
136      * @param contentType The contentType to set.
137      */
138     public void setContentType(String contentType) {
139         this.contentType = contentType;
140     }
141 
142     /**
143      * @return Returns the contentLength.
144      */
145     public String getContentLength() {
146         return contentLength;
147     }
148 
149     /**
150      * @param contentLength The contentLength to set.
151      */
152     public void setContentLength(String contentLength) {
153         this.contentLength = contentLength;
154     }
155 
156     /**
157      * @return Returns the Content-disposition header value.
158      */
159     public String getContentDisposition() {
160         return contentDisposition;
161     }
162 
163     /**
164      * @param contentDisposition the Content-disposition header value to use.
165      */
166     public void setContentDisposition(String contentDisposition) {
167         this.contentDisposition = contentDisposition;
168     }
169 
170     /**
171      * @return Returns the charset specified by the user
172      */
173     public String getContentCharSet() {
174         return contentCharSet;
175     }
176 
177     /**
178      * @param contentCharSet the charset to use on the header when sending the stream
179      */
180     public void setContentCharSet(String contentCharSet) {
181         this.contentCharSet = contentCharSet;
182     }
183 
184     /**
185      * @return Returns the inputName.
186      */
187     public String getInputName() {
188         return (inputName);
189     }
190 
191     /**
192      * @param inputName The inputName to set.
193      */
194     public void setInputName(String inputName) {
195         this.inputName = inputName;
196     }
197 
198     /**
199      * @see StrutsResultSupport#doExecute(java.lang.String, com.opensymphony.xwork2.ActionInvocation)
200      */
201     protected void doExecute(String finalLocation, ActionInvocation invocation) throws Exception {
202         LOG.debug("Find the Response in context");
203 
204         OutputStream oOutput = null;
205 
206         try {
207             if (inputStream == null) {
208                 LOG.debug("Find the inputstream from the invocation variable stack");
209                 inputStream = (InputStream) invocation.getStack().findValue(conditionalParse(inputName, invocation));
210             }
211 
212             if (inputStream == null) {
213                 String msg = ("Can not find a java.io.InputStream with the name [" + inputName + "] in the invocation stack. " +
214                     "Check the <param name=\"inputName\"> tag specified for this action.");
215                 LOG.error(msg);
216                 throw new IllegalArgumentException(msg);
217             }
218 
219 
220             HttpServletResponse oResponse = invocation.getInvocationContext().getServletResponse();
221 
222             LOG.debug("Set the content type: {};charset{}", contentType, contentCharSet);
223             if (contentCharSet != null && !contentCharSet.equals("")) {
224                 oResponse.setContentType(conditionalParse(contentType, invocation) + ";charset=" + conditionalParse(contentCharSet, invocation));
225             } else {
226                 oResponse.setContentType(conditionalParse(contentType, invocation));
227             }
228 
229             LOG.debug("Set the content length: {}", contentLength);
230             if (contentLength != null) {
231                 String _contentLength = conditionalParse(contentLength, invocation);
232                 int _contentLengthAsInt;
233                 try {
234                     _contentLengthAsInt = Integer.parseInt(_contentLength);
235                     if (_contentLengthAsInt >= 0) {
236                         oResponse.setContentLength(_contentLengthAsInt);
237                     }
238                 } catch (NumberFormatException e) {
239                     LOG.warn("failed to recognize {} as a number, contentLength header will not be set", _contentLength, e);
240                 }
241             }
242 
243             LOG.debug("Set the content-disposition: {}", contentDisposition);
244             if (contentDisposition != null) {
245                 oResponse.addHeader("Content-Disposition", conditionalParse(contentDisposition, invocation));
246             }
247 
248             LOG.debug("Set the cache control headers if necessary: {}", allowCaching);
249             if (!allowCaching) {
250                 oResponse.addHeader("Pragma", "no-cache");
251                 oResponse.addHeader("Cache-Control", "no-cache");
252             }
253 
254             oOutput = oResponse.getOutputStream();
255 
256             LOG.debug("Streaming result [{}] type=[{}] length=[{}] content-disposition=[{}] charset=[{}]",
257                 inputName, contentType, contentLength, contentDisposition, contentCharSet);
258 
259             LOG.debug("Streaming to output buffer +++ START +++");
260             byte[] oBuff = new byte[bufferSize];
261             int iSize;
262             while (-1 != (iSize = inputStream.read(oBuff))) {
263                 LOG.debug("Sending stream ... {}", iSize);
264                 oOutput.write(oBuff, 0, iSize);
265             }
266             LOG.debug("Streaming to output buffer +++ END +++");
267 
268             // Flush
269             oOutput.flush();
270         } finally {
271             if (inputStream != null) {
272                 inputStream.close();
273             }
274             if (oOutput != null) {
275                 oOutput.close();
276             }
277         }
278     }
