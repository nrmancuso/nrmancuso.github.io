34      public void testNulls() throws IntrospectionException, InvocationTargetException, NoSuchMethodException,
35              JSONException, InstantiationException, IllegalAccessException {
36          JSONPopulator populator = new JSONPopulator();
37          OtherBean bean = new OtherBean();
38          Map jsonMap = new HashMap();
39  
40          jsonMap.put("intField", null);
41          jsonMap.put("booleanField", null);
42          jsonMap.put("charField", null);
43          jsonMap.put("longField", null);
44          jsonMap.put("floatField", null);
45          jsonMap.put("doubleField", null);
46          jsonMap.put("byteField", null);
47  
48          populator.populateObject(bean, jsonMap);
49          assertNull(bean.getIntField());
50          assertNull(bean.isBooleanField());
51          assertNull(bean.getCharField());
52          assertNull(bean.getLongField());
53          assertNull(bean.getDoubleField());
54          assertNull(bean.getByteField());
55      }

57      public void testPrimitiveBean() throws Exception {
58          StringReader stringReader = new StringReader(TestUtils.readContent(JSONInterceptorTest.class
59                  .getResource("json-7.txt")));
60          Object json = JSONUtil.deserialize(stringReader);
61          assertNotNull(json);
62          assertTrue(json instanceof Map);
63          Map jsonMap = (Map) json;
64          JSONPopulator populator = new JSONPopulator();
65          Bean bean = new Bean();
66          populator.populateObject(bean, jsonMap);
67          assertTrue(bean.isBooleanField());
68          assertEquals("test\u000E\u000f", bean.getStringField());
69          assertEquals(10, bean.getIntField());
70          assertEquals('s', bean.getCharField());
71          assertEquals(10.1d, bean.getDoubleField(), 0d);
72          assertEquals(3, bean.getByteField());
73          assertEquals(new BigDecimal(111111.5d), bean.getBigDecimal());
74          assertEquals(new BigInteger("111111"), bean.getBigInteger());
75      }

77      public void testObjectBean() throws Exception {
78          String text = TestUtils.readContent(JSONInterceptorTest.class.getResource("json-7.txt"));
79          Object json = JSONUtil.deserialize(text);
80          assertNotNull(json);
81          assertTrue(json instanceof Map);
82          Map jsonMap = (Map) json;
83          JSONPopulator populator = new JSONPopulator();
84          WrapperClassBean bean = new WrapperClassBean();
85          populator.populateObject(bean, jsonMap);
86          assertEquals(Boolean.TRUE, bean.getBooleanField());
87          assertEquals(true, bean.isPrimitiveBooleanField1());
88          assertEquals(false, bean.isPrimitiveBooleanField2());
89          assertEquals(false, bean.isPrimitiveBooleanField3());
90          assertEquals("test\u000E\u000f", bean.getStringField());
91          assertEquals(new Integer(10), bean.getIntField());
92          assertEquals(0, bean.getNullIntField());
93          assertEquals(new Character('s'), bean.getCharField());
94          assertEquals(10.1d, bean.getDoubleField());
95          assertEquals(new Byte((byte) 3), bean.getByteField());
96  
97          assertEquals(2, bean.getListField().size());
98          assertEquals("1", bean.getListField().get(0).getValue());
99          assertEquals("2", bean.getListField().get(1).getValue());
100 
101         assertEquals(1, bean.getListMapField().size());
102         assertEquals(2, bean.getListMapField().get(0).size());
103         assertEquals(new Long(2073501), bean.getListMapField().get(0).get("id1"));
104         assertEquals(new Long(3), bean.getListMapField().get(0).get("id2"));
105 
106         assertEquals(2, bean.getMapListField().size());
107         assertEquals(3, bean.getMapListField().get("id1").size());
108         assertEquals(new Long(2), bean.getMapListField().get("id1").get(1));
109         assertEquals(4, bean.getMapListField().get("id2").size());
110         assertEquals(new Long(3), bean.getMapListField().get("id2").get(1));
111 
112         assertEquals(1, bean.getArrayMapField().length);
113         assertEquals(2, bean.getArrayMapField()[0].size());
114         assertEquals(new Long(2073501), bean.getArrayMapField()[0].get("id1"));
115         assertEquals(new Long(3), bean.getArrayMapField()[0].get("id2"));
116 
117         assertEquals(3, bean.getSetField().size());
118         assertEquals(true, bean.getSetField().contains("A"));
119         assertEquals(true, bean.getSetField().contains("B"));
120         assertEquals(true, bean.getSetField().contains("C"));
121 
122         assertEquals(3, bean.getSortedSetField().size());
123         assertEquals("A", bean.getSortedSetField().first());
124         assertEquals(true, bean.getSortedSetField().contains("B"));
125         assertEquals("C", bean.getSortedSetField().last());
126 
127         assertEquals(3, bean.getNavigableSetField().size());
128         assertEquals("A", bean.getNavigableSetField().first());
129         assertEquals(true, bean.getNavigableSetField().contains("B"));
130         assertEquals("C", bean.getNavigableSetField().last());
131 
132         assertEquals(3, bean.getQueueField().size());
133         assertEquals("A", bean.getQueueField().poll());
134         assertEquals("B", bean.getQueueField().poll());
135         assertEquals("C", bean.getQueueField().poll());
136 
137         assertEquals(3, bean.getDequeField().size());
138         assertEquals("A", bean.getDequeField().pollFirst());
139         assertEquals("B", bean.getDequeField().pollFirst());
140         assertEquals("C", bean.getDequeField().pollFirst());
141     }

160         assertEquals(null, bean.getListField());
161         assertEquals(null, bean.getListMapField());
