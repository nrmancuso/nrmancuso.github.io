36  /*
37   * Changes:  This is a copy of org.apache.struts2.StrutsPortletTestCase from the Struts 2 portlet-plugin, moved
38   *           into the junit-plugin (same package org.apache.struts2).
39   *           The import order above was changed to alphabetical.
40   *
41   * Note:     The assumption is that anyone utilizing StrutsPortletTestCase currently from the portlet-plugin will almost
42   *           certainly be using the junit-plugin.  Under that assumption, the refactored-move of StrutsPortletTestCase
43   *           should not cause issues for pre-existing usage of StrutsPortletTestCase.
44   */
45  
46  /**
47   * Base class used to test action in portlet environment
48   */
49  public abstract class StrutsPortletTestCase extends StrutsTestCase {
50  
51      private static final Logger LOG = LogManager.getLogger(StrutsPortletTestCase.class);
52  
53      protected MockPortletSession portletSession;
54      protected MockPortletRequest portletRequest;
55      protected MockPortletResponse portletResponse;
56      protected MockContext portletContext;
57  
58      @Override
59      protected void initActionContext(ActionContext actionContext) {
60          super.initActionContext(actionContext);
61          initPortletContext(actionContext);
62      }
63  
64      protected void initPortletContext(ActionContext actionContext) {
65          LOG.debug("Initializing mock portlet environment");
66          portletContext = new MockContext();
67          portletContext.setMajorVersion(getMajorVersion());
68          actionContext.put(StrutsStatics.STRUTS_PORTLET_CONTEXT, portletContext);
69  
70          portletRequest = new MockPortletRequest(portletContext);
71          portletResponse = new MockStateAwareResponse();
72          portletSession = new MockPortletSession();
73          portletRequest.setSession(portletSession);
74          actionContext.withSession(createSession());
75          actionContext.put(PortletConstants.REQUEST, portletRequest);
76          actionContext.put(PortletConstants.RESPONSE, portletResponse);
77          actionContext.put(PortletConstants.MODE_NAMESPACE_MAP, new HashMap<PortletMode, String>());
78          actionContext.put(PortletConstants.PHASE, PortletPhase.EVENT_PHASE);
79      }
80  
81      /**
82       * Override to define version of your portlet environment
83       *
84       * @return portlet version
85       */
86      protected int getMajorVersion() {
87          return 2;
88      }
89  
90      /**
91       * Override to create your own session
92       *
93       * @return Map with session parameters
94       */
95      private Map<String, Object> createSession() {
96          return new HashMap<String, Object>(portletRequest.getPortletSession().getAttributeMap());
97      }
98  
99  }
100 
101 /**
102  * Simple workaround to define Portlet version
103  */
104 class MockContext extends MockPortletContext {
105 
106     private int majorVersion;
107 
108     @Override
109     public int getMajorVersion() {
110         return majorVersion;
111     }
112 
113     public void setMajorVersion(int majorVersion) {
114         this.majorVersion = majorVersion;
115     }
116 
117 }
