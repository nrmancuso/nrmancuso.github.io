123     private static boolean isMacosLionOrBetter() {
124         // split the "10.x.y" version number
125         String osname = getProperty("os.name");
126         if (!osname.contains("OS X")) {
127             return false;
128         }
129 
130         String osVersion = getProperty("os.version");
131         String[] fragments = osVersion.split("\\.");
132 
133         // sanity check the "10." part of the version
134         if (!fragments[0].equals("10")) return false;
135         if (fragments.length < 2) return false;
136 
137         // check if Mac OS X 10.7(.y)
138         try {
139             int minorVers = Integer.parseInt(fragments[1]);
140             if (minorVers >= 7) return true;
141         } catch (NumberFormatException e) {
142             // was not an integer
143         }
144 
145         return false;
146     }

148     /**
149      * Private constructor - can not be instantiated externally.
150      */
151     private Config() throws KrbException {
152         /*
153          * If either one system property is specified, we throw exception.
154          */
155         String tmp = getProperty("java.security.krb5.kdc");
156         if (tmp != null) {
157             // The user can specify a list of kdc hosts separated by ":"
158             defaultKDC = tmp.replace(':', ' ');
159         } else {
160             defaultKDC = null;
161         }
162         defaultRealm = getProperty("java.security.krb5.realm");
163         if ((defaultKDC == null && defaultRealm != null) ||
164             (defaultRealm == null && defaultKDC != null)) {
165             throw new KrbException
166                 ("System property java.security.krb5.kdc and " +
167                  "java.security.krb5.realm both must be set or " +
168                  "neither must be set.");
169         }
170 
171         // Always read the Kerberos configuration file
172         try {
173             List<String> configFile;
174             String fileName = getJavaFileName();
175             if (fileName != null) {
176                 configFile = loadConfigFile(fileName);
177                 stanzaTable = parseStanzaTable(configFile);
178                 if (DEBUG) {
179                     System.out.println("Loaded from Java config");
180                 }
181             } else {
182                 boolean found = false;
183                 if (isMacosLionOrBetter()) {
184                     try {
185                         stanzaTable = SCDynamicStoreConfig.getConfig();
186                         if (DEBUG) {
187                             System.out.println("Loaded from SCDynamicStoreConfig");
188                         }
189                         found = true;
190                     } catch (IOException ioe) {
191                         // OK. Will go on with file
192                     }
193                 }
194                 if (!found) {
195                     fileName = getNativeFileName();
196                     configFile = loadConfigFile(fileName);
197                     stanzaTable = parseStanzaTable(configFile);
198                     if (DEBUG) {
199                         System.out.println("Loaded from native config");
200                     }
201                 }
202             }
203         } catch (IOException ioe) {
204             if (DEBUG) {
205                 System.out.println("Exception thrown in loading config:");
206                 ioe.printStackTrace(System.out);
207             }
208             throw new KrbException("krb5.conf loading failed");
209         }
210     }

212     /**
213      * Gets the last-defined string value for the specified keys.
214      * @param keys the keys, as an array from section name, sub-section names
215      * (if any), to value name.
216      * @return the value. When there are multiple values for the same key,
217      * returns the first one. {@code null} is returned if not all the keys are
218      * defined. For example, {@code get("libdefaults", "forwardable")} will
219      * return null if "forwardable" is not defined in [libdefaults], and
220      * {@code get("realms", "R", "kdc")} will return null if "R" is not
221      * defined in [realms] or "kdc" is not defined for "R".
222      * @throws IllegalArgumentException if any of the keys is illegal, either
223      * because a key not the last one is not a (sub)section name or the last
224      * key is still a section name. For example, {@code get("libdefaults")}
225      * throws this exception because [libdefaults] is a section name instead of
226      * a value name, and {@code get("libdefaults", "forwardable", "tail")}
227      * also throws this exception because "forwardable" is already a value name
228      * and has no sub-key at all (given "forwardable" is defined, otherwise,
229      * this method has no knowledge if it's a value name or a section name),
230      */
231     public String get(String... keys) {
232         Vector<String> v = getString0(keys);
233         if (v == null) return null;
234         return v.firstElement();
235     }

237     /**
238      * Gets the boolean value for the specified keys. Returns TRUE if the
239      * string value is "yes", or "true", FALSE if "no", or "false", or null
240      * if otherwise or not defined. The comparision is case-insensitive.
241      *
242      * @param keys the keys, see {@link #get(String...)}
243      * @return the boolean value, or null if there is no value defined or the
244      * value does not look like a boolean value.
245      * @throws IllegalArgumentException see {@link #get(String...)}
246      */
247     public Boolean getBooleanObject(String... keys) {
248         String s = get(keys);
249         if (s == null) {
250             return null;
251         }
252         switch (s.toLowerCase(Locale.US)) {
253             case "yes": case "true":
254                 return Boolean.TRUE;
255             case "no": case "false":
256                 return Boolean.FALSE;
257             default:
258                 return null;
259         }
260     }

262     /**
263      * Gets all values (at least one) for the specified keys separated by
264      * a whitespace, or null if there is no such keys.
265      * The values can either be provided on a single line, or on multiple lines
266      * using the same key. When provided on a single line, the value can be
267      * comma or space separated.
268      * @throws IllegalArgumentException if any of the keys is illegal
269      *         (See {@link #get})
270      */
271     public String getAll(String... keys) {
272         Vector<String> v = getString0(keys);
273         if (v == null) return null;
274         StringBuilder sb = new StringBuilder();
275         boolean first = true;
276         for (String s: v) {
277             s = s.replaceAll("[\\s,]+", " ");
278             if (first) {
279                 sb.append(s);
280                 first = false;
281             } else {
282                 sb.append(' ').append(s);
283             }
284         }
285         return sb.toString();
286     }

307     // Internal method. Returns the value for keys, which can be a sub-section
308     // (as a Hashtable) or final string value(s) (as a Vector). This is the
309     // only method (except for toString) that reads stanzaTable directly.
310     @SuppressWarnings("unchecked")
311     private Object get0(String... keys) {
312         Object current = stanzaTable;
313         try {
314             for (String key: keys) {
315                 current = ((Hashtable<String,Object>)current).get(key);
316                 if (current == null) return null;
317             }
318             return current;
319         } catch (ClassCastException cce) {
320             throw new IllegalArgumentException(cce);
321         }
322     }

324     /**
325      * Translates a duration value into seconds.
326      *
327      * The format can be one of "h:m[:s]", "NdNhNmNs", and "N". See
328      * http://web.mit.edu/kerberos/krb5-devel/doc/basic/date_format.html#duration
329      * for definitions.
330      *
331      * @param s the string duration
332      * @return time in seconds
333      * @throws KrbException if format is illegal
334      */
335     public static int duration(String s) throws KrbException {
336 
337         if (s.isEmpty()) {
338             throw new KrbException("Duration cannot be empty");
339         }
340 
341         // N
342         if (s.matches("\\d+")) {
343             return Integer.parseInt(s);
344         }
345 
346         // h:m[:s]
347         Matcher m = Pattern.compile("(\\d+):(\\d+)(:(\\d+))?").matcher(s);
348         if (m.matches()) {
349             int hr = Integer.parseInt(m.group(1));
350             int min = Integer.parseInt(m.group(2));
351             if (min >= 60) {
352                 throw new KrbException("Illegal duration format " + s);
353             }
354             int result = hr * 3600 + min * 60;
355             if (m.group(4) != null) {
356                 int sec = Integer.parseInt(m.group(4));
357                 if (sec >= 60) {
358                     throw new KrbException("Illegal duration format " + s);
359                 }
360                 result += sec;
361             }
362             return result;
363         }
364 
365         // NdNhNmNs
366         // 120m allowed. Maybe 1h120m is not good, but still allowed
367         m = Pattern.compile(
368                     "((\\d+)d)?\\s*((\\d+)h)?\\s*((\\d+)m)?\\s*((\\d+)s)?",
369                 Pattern.CASE_INSENSITIVE).matcher(s);
370         if (m.matches()) {
371             int result = 0;
372             if (m.group(2) != null) {
373                 result += 86400 * Integer.parseInt(m.group(2));
374             }
375             if (m.group(4) != null) {
376                 result += 3600 * Integer.parseInt(m.group(4));
377             }
378             if (m.group(6) != null) {
379                 result += 60 * Integer.parseInt(m.group(6));
380             }
381             if (m.group(8) != null) {
382                 result += Integer.parseInt(m.group(8));
383             }
384             return result;
385         }
386 
387         throw new KrbException("Illegal duration format " + s);
388     }

391      * Gets the int value for the specified keys.
392      * @param keys the keys
393      * @return the int value, Integer.MIN_VALUE is returned if it cannot be
394      * found or the value is not a legal integer.
395      * @throws IllegalArgumentException if any of the keys is illegal
396      * @see #get(java.lang.String[])

398     public int getIntValue(String... keys) {
399         String result = get(keys);
400         int value = Integer.MIN_VALUE;
401         if (result != null) {
402             try {
403                 value = parseIntValue(result);
404             } catch (NumberFormatException e) {
405                 if (DEBUG) {
406                     System.out.println("Exception in getting value of " +
407                                        Arrays.toString(keys) + " " +
408                                        e.getMessage());
409                     System.out.println("Setting " + Arrays.toString(keys) +
410                                        " to minimum value");
411                 }
412                 value = Integer.MIN_VALUE;
413             }
414         }
415         return value;
416     }

418     /**
419      * Parses a string to an integer. The convertible strings include the
420      * string representations of positive integers, negative integers, and
421      * hex decimal integers.  Valid inputs are, e.g., -1234, +1234,
422      * 0x40000.
423      *
424      * @param input the String to be converted to an Integer.
425      * @return an numeric value represented by the string
426      * @exception NumberFormatException if the String does not contain a
427      * parsable integer.
428      */
429     private int parseIntValue(String input) throws NumberFormatException {
430         int value = 0;
431         if (input.startsWith("+")) {
432             String temp = input.substring(1);
433             return Integer.parseInt(temp);
434         } else if (input.startsWith("0x")) {
435             String temp = input.substring(2);
436             char[] chars = temp.toCharArray();
437             if (chars.length > 8) {
438                 throw new NumberFormatException();
439             } else {
440                 for (int i = 0; i < chars.length; i++) {
441                     int index = chars.length - i - 1;
442                     switch (chars[i]) {
443                     case '0':
444                         value += 0;
445                         break;
446                     case '1':
447                         value += 1 * getBase(index);
448                         break;
449                     case '2':
450                         value += 2 * getBase(index);
451                         break;
452                     case '3':
453                         value += 3 * getBase(index);
454                         break;
455                     case '4':
456                         value += 4 * getBase(index);
457                         break;
458                     case '5':
459                         value += 5 * getBase(index);
460                         break;
461                     case '6':
462                         value += 6 * getBase(index);
463                         break;
464                     case '7':
465                         value += 7 * getBase(index);
466                         break;
467                     case '8':
468                         value += 8 * getBase(index);
469                         break;
470                     case '9':
471                         value += 9 * getBase(index);
472                         break;
473                     case 'a':
474                     case 'A':
475                         value += 10 * getBase(index);
476                         break;
477                     case 'b':
478                     case 'B':
479                         value += 11 * getBase(index);
480                         break;
481                     case 'c':
482                     case 'C':
483                         value += 12 * getBase(index);
484                         break;
485                     case 'd':
486                     case 'D':
487                         value += 13 * getBase(index);
488                         break;
489                     case 'e':
490                     case 'E':
491                         value += 14 * getBase(index);
492                         break;
493                     case 'f':
494                     case 'F':
495                         value += 15 * getBase(index);
496                         break;
497                     default:
498                         throw new NumberFormatException("Invalid numerical format");
499                     }
500                 }
501             }
502             if (value < 0) {
503                 throw new NumberFormatException("Data overflow.");
504             }
505         } else {
506             value = Integer.parseInt(input);
507         }
508         return value;
509     }

511     private int getBase(int i) {
512         int result = 16;
513         switch (i) {
514         case 0:
515             result = BASE16_0;
516             break;
517         case 1:
518             result = BASE16_1;
519             break;
520         case 2:
521             result = BASE16_2;
522             break;
523         case 3:
524             result = BASE16_3;
525             break;
526         default:
527             for (int j = 1; j < i; j++) {
528                 result *= 16;
529             }
530         }
531         return result;
532     }

534     /**
535      * Reads the lines of the configuration file. All include and includedir
536      * directives are resolved by calling this method recursively.
537      *
538      * @param file the krb5.conf file, must be absolute
539      * @param content the lines. Comment and empty lines are removed,
540      *                all lines trimmed, include and includedir
541      *                directives resolved, unknown directives ignored
542      * @param dups a set of Paths to check for possible infinite loop
543      * @throws IOException if there is an I/O error
544      */
545     private static Void readConfigFileLines(
546             Path file, List<String> content, Set<Path> dups)
547             throws IOException {
548 
549         if (DEBUG) {
550             System.out.println("Loading krb5 profile at " + file);
551         }
552         if (!file.isAbsolute()) {
553             throw new IOException("Profile path not absolute");
554         }
555 
556         if (!dups.add(file)) {
557             throw new IOException("Profile path included more than once");
558         }
559 
560         List<String> lines = Files.readAllLines(file);
561 
562         boolean inDirectives = true;
563         for (String line: lines) {
564             line = line.trim();
565             if (line.isEmpty() || line.startsWith("#") || line.startsWith(";")) {
566                 continue;
567             }
568             if (inDirectives) {
569                 if (line.charAt(0) == '[') {
570                     inDirectives = false;
571                     content.add(line);
572                 } else if (line.startsWith("includedir ")) {
573                     Path dir = Paths.get(
574                             line.substring("includedir ".length()).trim());
575                     try (DirectoryStream<Path> files =
576                                  Files.newDirectoryStream(dir)) {
577                         for (Path p: files) {
578                             if (Files.isDirectory(p)) continue;
579                             String name = p.getFileName().toString();
580                             if (name.matches("[a-zA-Z0-9_-]+")) {
581                                 // if dir is absolute, so is p
582                                 readConfigFileLines(p, content, dups);
583                             }
584                         }
585                     }
586                 } else if (line.startsWith("include ")) {
587                     readConfigFileLines(
588                             Paths.get(line.substring("include ".length()).trim()),
589                             content, dups);
590                 } else {
591                     // Unsupported directives
592                     if (DEBUG) {
593                         System.out.println("Unknown directive: " + line);
594                     }
595                 }
596             } else {
597                 content.add(line);
598             }
599         }
600         return null;
601     }

603     /**
604      * Reads the configuration file and return normalized lines.
605      * If the original file is:
606      *
607      *     [realms]
608      *     EXAMPLE.COM =
609      *     {
610      *         kdc = kerberos.example.com
611      *         ...
612      *     }
613      *     ...
614      *
615      * The result will be (no indentations):
616      *
617      *     {
618      *         realms = {
619      *             EXAMPLE.COM = {
620      *                 kdc = kerberos.example.com
621      *                 ...
622      *             }
623      *         }
624      *         ...
625      *     }
626      *
627      * @param fileName the configuration file
628      * @return normalized lines
629      */
630     private List<String> loadConfigFile(final String fileName)
631             throws IOException, KrbException {
632 
633         List<String> result = new ArrayList<>();
634         List<String> raw = new ArrayList<>();
635         Set<Path> dupsCheck = new HashSet<>();
636 
637         try {
638             Path fullp = AccessController.doPrivileged((PrivilegedAction<Path>)
639                         () -> Paths.get(fileName).toAbsolutePath(),
640                     null,
641                     new PropertyPermission("user.dir", "read"));
642             AccessController.doPrivileged(
643                     new PrivilegedExceptionAction<Void>() {
644                         @Override
645                         public Void run() throws IOException {
646                             Path path = Paths.get(fileName);
647                             if (!Files.exists(path)) {
648                                 // This is OK. There are other ways to get
649                                 // Kerberos 5 settings
650                                 return null;
651                             } else {
652                                 return readConfigFileLines(
653                                         fullp, raw, dupsCheck);
654                             }
655                         }
656                     },
657                     null,
658                     // include/includedir can go anywhere
659                     new FilePermission("<<ALL FILES>>", "read"));
660         } catch (java.security.PrivilegedActionException pe) {
661             throw (IOException)pe.getException();
662         }
663         String previous = null;
664         for (String line: raw) {
665             if (line.startsWith("[")) {
666                 if (!line.endsWith("]")) {
667                     throw new KrbException("Illegal config content:"
668                             + line);
669                 }
670                 if (previous != null) {
671                     result.add(previous);
672                     result.add("}");
673                 }
674                 String title = line.substring(
675                         1, line.length()-1).trim();
676                 if (title.isEmpty()) {
677                     throw new KrbException("Illegal config content:"
678                             + line);
679                 }
680                 previous = title + " = {";
681             } else if (line.startsWith("{")) {
682                 if (previous == null) {
683                     throw new KrbException(
684                         "Config file should not start with \"{\"");
685                 }
686                 previous += " {";
687                 if (line.length() > 1) {
688                     // { and content on the same line
689                     result.add(previous);
690                     previous = line.substring(1).trim();
691                 }
692             } else {
693                 if (previous == null) {
694                     // This won't happen, because before a section
695                     // all directives have been resolved
696                     throw new KrbException(
697                         "Config file must starts with a section");
698                 }
699                 result.add(previous);
700                 previous = line;
701             }
702         }
703         if (previous != null) {
704             result.add(previous);
705             result.add("}");
706         }
707         return result;
708     }

710     /**
711      * Parses the input lines to a hashtable. The key would be section names
712      * (libdefaults, realms, domain_realms, etc), and the value would be
713      * another hashtable which contains the key-value pairs inside the section.
714      * The value of this sub-hashtable can be another hashtable containing
715      * another sub-sub-section or a non-empty vector of strings for final values
716      * (even if there is only one value defined).
717      * <p>
718      * For top-level sections with duplicates names, their contents are merged.
719      * For sub-sections the former overwrites the latter. For final values,
720      * they are stored in a vector in their appearing order. Please note these
721      * values must appear in the same sub-section. Otherwise, the sub-section
722      * appears first should have already overridden the others.
723      * <p>
724      * As a corner case, if the same name is used as both a section name and a
725      * value name, the first appearance decides the type. That is to say, if the
726      * first one is for a section, all latter appearances are ignored. If it's
727      * a value, latter appearances as sections are ignored, but those as values
728      * are added to the vector.
729      * <p>
730      * The behavior described above is compatible to other krb5 implementations
731      * but it's not decumented publicly anywhere. the best practice is not to
732      * assume any kind of override functionality and only specify values for
733      * a particular key in one place.
734      *
735      * @param v the normalized input as return by loadConfigFile
736      * @throws KrbException if there is a file format error
737      */
738     @SuppressWarnings("unchecked")
739     private Hashtable<String,Object> parseStanzaTable(List<String> v)
740             throws KrbException {
741         Hashtable<String,Object> current = stanzaTable;
742         for (String line: v) {
743             // There are only 3 kinds of lines
744             // 1. a = b
745             // 2. a = {
746             // 3. }
747             if (line.equals("}")) {
748                 // Go back to parent, see below
749                 current = (Hashtable<String,Object>)current.remove(" PARENT ");
750                 if (current == null) {
751                     throw new KrbException("Unmatched close brace");
752                 }
753             } else {
754                 int pos = line.indexOf('=');
755                 if (pos < 0) {
756                     throw new KrbException("Illegal config content:" + line);
757                 }
758                 String key = line.substring(0, pos).trim();
759                 String value = unquote(line.substring(pos + 1));
760                 if (value.equals("{")) {
761                     Hashtable<String,Object> subTable;
762                     if (current == stanzaTable) {
763                         key = key.toLowerCase(Locale.US);
764                     }
765                     // When there are dup names for sections
766                     if (current.containsKey(key)) {
767                         if (current == stanzaTable) {   // top-level, merge
768                             // The value at top-level must be another Hashtable
769                             subTable = (Hashtable<String,Object>)current.get(key);
770                         } else {                        // otherwise, ignored
771                             // read and ignore it (do not put into current)
772                             subTable = new Hashtable<>();
773                         }
774                     } else {
775                         subTable = new Hashtable<>();
776                         current.put(key, subTable);
777                     }
778                     // A special entry for its parent. Put whitespaces around,
779                     // so will never be confused with a normal key
780                     subTable.put(" PARENT ", current);
781                     current = subTable;
782                 } else {
783                     Vector<String> values;
784                     if (current.containsKey(key)) {
785                         Object obj = current.get(key);
786                         if (obj instanceof Vector) {
787                             // String values are merged
788                             values = (Vector<String>)obj;
789                             values.add(value);
790                         } else {
791                             // If a key shows as section first and then a value,
792                             // ignore the value.
793                         }
794                     } else {
795                         values = new Vector<String>();
796                         values.add(value);
797                         current.put(key, values);
798                     }
799                 }
800             }
801         }
802         if (current != stanzaTable) {
803             throw new KrbException("Not closed");
804         }
805         return current;
806     }

835      * Gets the default native configuration file name.
836      *
837      * Depending on the OS type, the method returns the default native
838      * kerberos config file name, which is at windows directory with
839      * the name of "krb5.ini" for Windows, /etc/krb5/krb5.conf for Solaris,

844      * C:\Windows), and a user-private one (say, C:\Users\Me\Windows).
845      * We will first look for krb5.ini in the user-private one. If not

847      *
848      * This method will always return a non-null non-empty file name,
849      * even if that file does not exist.
850      */
851     private String getNativeFileName() {
852         String name = null;
853         String osname = getProperty("os.name");
854         if (osname.startsWith("Windows")) {
855             try {
856                 Credentials.ensureLoaded();
857             } catch (Exception e) {
858                 // ignore exceptions
859             }
860             if (Credentials.alreadyLoaded) {
861                 String path = getWindowsDirectory(false);
862                 if (path != null) {
863                     if (path.endsWith("\\")) {
864                         path = path + "krb5.ini";
865                     } else {
866                         path = path + "\\krb5.ini";
867                     }
868                     if (fileExists(path)) {
869                         name = path;
870                     }
871                 }
872                 if (name == null) {
873                     path = getWindowsDirectory(true);
874                     if (path != null) {
875                         if (path.endsWith("\\")) {
876                             path = path + "krb5.ini";
877                         } else {
878                             path = path + "\\krb5.ini";
879                         }
880                         name = path;
881                     }
882                 }
883             }
884             if (name == null) {

888             name =  "/etc/krb5/krb5.conf";
889         } else if (osname.contains("OS X")) {
890             name = findMacosConfigFile();
891         } else {
892             name =  "/etc/krb5.conf";
893         }
894         if (DEBUG) {
895             System.out.println("Native config name: " + name);
896         }

922         s = s.trim();
923         if (s.length() >= 2 &&
924                 ((s.charAt(0) == '"' && s.charAt(s.length()-1) == '"') ||
925                  (s.charAt(0) == '\'' && s.charAt(s.length()-1) == '\''))) {

939     /**
940      * Returns all etypes specified in krb5.conf for the given configName,
941      * or all the builtin defaults. This result is always non-empty.
942      * If no etypes are found, an exception is thrown.
943      */
944     public int[] defaultEtype(String configName) throws KrbException {
945         String default_enctypes;
946         default_enctypes = get("libdefaults", configName);
947         int[] etype;
948         if (default_enctypes == null) {
949             if (DEBUG) {
950                 System.out.println("Using builtin default etypes for " +
951                     configName);
952             }
953             etype = EType.getBuiltInDefaults();
954         } else {
955             String delim = " ";
956             StringTokenizer st;
957             for (int j = 0; j < default_enctypes.length(); j++) {
958                 if (default_enctypes.substring(j, j + 1).equals(",")) {
959                     // only two delimiters are allowed to use
960                     // according to Kerberos DCE doc.
961                     delim = ",";
962                     break;
963                 }
964             }
965             st = new StringTokenizer(default_enctypes, delim);
966             int len = st.countTokens();
967             ArrayList<Integer> ls = new ArrayList<>(len);
968             int type;
969             for (int i = 0; i < len; i++) {
970                 type = Config.getType(st.nextToken());
971                 if (type != -1 && EType.isSupported(type)) {
972                     ls.add(type);
973                 }
974             }
975             if (ls.isEmpty()) {
976                 throw new KrbException("no supported default etypes for "
977                         + configName);
978             } else {
979                 etype = new int[ls.size()];
980                 for (int i = 0; i < etype.length; i++) {
981                     etype[i] = ls.get(i);
982                 }
983             }
984         }
985 
986         if (DEBUG) {
987             System.out.print("default etypes for " + configName + ":");
988             for (int i = 0; i < etype.length; i++) {
989                 System.out.print(" " + etype[i]);
990             }
991             System.out.println(".");
992         }
993         return etype;
994     }

997     /**
998      * Get the etype and checksum value for the specified encryption and
999      * checksum type.
1000      *
1001      */
1002     /*
1003      * This method converts the string representation of encryption type and
1004      * checksum type to int value that can be later used by EType and
1005      * Checksum classes.
1006      */
1007     public static int getType(String input) {
1008         int result = -1;
1009         if (input == null) {
1010             return result;
1011         }
1012         if (input.startsWith("d") || (input.startsWith("D"))) {
1013             if (input.equalsIgnoreCase("des-cbc-crc")) {
1014                 result = EncryptedData.ETYPE_DES_CBC_CRC;
1015             } else if (input.equalsIgnoreCase("des-cbc-md5")) {
1016                 result = EncryptedData.ETYPE_DES_CBC_MD5;
1017             } else if (input.equalsIgnoreCase("des-mac")) {
1018                 result = Checksum.CKSUMTYPE_DES_MAC;
1019             } else if (input.equalsIgnoreCase("des-mac-k")) {
1020                 result = Checksum.CKSUMTYPE_DES_MAC_K;
1021             } else if (input.equalsIgnoreCase("des-cbc-md4")) {
1022                 result = EncryptedData.ETYPE_DES_CBC_MD4;
1023             } else if (input.equalsIgnoreCase("des3-cbc-sha1") ||
1024                 input.equalsIgnoreCase("des3-hmac-sha1") ||
1025                 input.equalsIgnoreCase("des3-cbc-sha1-kd") ||
1026                 input.equalsIgnoreCase("des3-cbc-hmac-sha1-kd")) {
1027                 result = EncryptedData.ETYPE_DES3_CBC_HMAC_SHA1_KD;
1028             }
1029         } else if (input.startsWith("a") || (input.startsWith("A"))) {
1030             // AES
1031             if (input.equalsIgnoreCase("aes128-cts") ||
1032                 input.equalsIgnoreCase("aes128-cts-hmac-sha1-96")) {
1033                 result = EncryptedData.ETYPE_AES128_CTS_HMAC_SHA1_96;
1034             } else if (input.equalsIgnoreCase("aes256-cts") ||
1035                 input.equalsIgnoreCase("aes256-cts-hmac-sha1-96")) {
1036                 result = EncryptedData.ETYPE_AES256_CTS_HMAC_SHA1_96;
1037             // ARCFOUR-HMAC
1038             } else if (input.equalsIgnoreCase("arcfour-hmac") ||
1039                    input.equalsIgnoreCase("arcfour-hmac-md5")) {
1040                 result = EncryptedData.ETYPE_ARCFOUR_HMAC;
1041             }
1042         // RC4-HMAC
1043         } else if (input.equalsIgnoreCase("rc4-hmac")) {
1044             result = EncryptedData.ETYPE_ARCFOUR_HMAC;
1045         } else if (input.equalsIgnoreCase("CRC32")) {
1046             result = Checksum.CKSUMTYPE_CRC32;
1047         } else if (input.startsWith("r") || (input.startsWith("R"))) {
1048             if (input.equalsIgnoreCase("rsa-md5")) {
1049                 result = Checksum.CKSUMTYPE_RSA_MD5;
1050             } else if (input.equalsIgnoreCase("rsa-md5-des")) {
1051                 result = Checksum.CKSUMTYPE_RSA_MD5_DES;
1052             }
1053         } else if (input.equalsIgnoreCase("hmac-sha1-des3-kd")) {
1054             result = Checksum.CKSUMTYPE_HMAC_SHA1_DES3_KD;
1055         } else if (input.equalsIgnoreCase("hmac-sha1-96-aes128")) {
1056             result = Checksum.CKSUMTYPE_HMAC_SHA1_96_AES128;
1057         } else if (input.equalsIgnoreCase("hmac-sha1-96-aes256")) {
1058             result = Checksum.CKSUMTYPE_HMAC_SHA1_96_AES256;
1059         } else if (input.equalsIgnoreCase("hmac-md5-rc4") ||
1060                 input.equalsIgnoreCase("hmac-md5-arcfour") ||
1061                 input.equalsIgnoreCase("hmac-md5-enc")) {
1062             result = Checksum.CKSUMTYPE_HMAC_MD5_ARCFOUR;
1063         } else if (input.equalsIgnoreCase("NULL")) {
1064             result = EncryptedData.ETYPE_NULL;
1065         }
1066 
1067         return result;
1068     }

1218     /**
1219      * Locate Kerberos realm using DNS
1220      *
1221      * @return the Kerberos realm
1222      */
1223     private String getRealmFromDNS() throws KrbException {
1224         // use DNS to locate Kerberos realm
1225         String realm = null;
1226         String hostName = null;
1227         try {
1228             hostName = InetAddress.getLocalHost().getCanonicalHostName();
1229         } catch (UnknownHostException e) {
1230             KrbException ke = new KrbException(Krb5.KRB_ERR_GENERIC,
1231                 "Unable to locate Kerberos realm: " + e.getMessage());
1232             ke.initCause(e);
1233             throw (ke);
1234         }
1235         // get the domain realm mapping from the configuration
1236         String mapRealm = PrincipalName.mapHostToRealm(hostName);
1237         if (mapRealm == null) {
1238             // No match. Try search and/or domain in /etc/resolv.conf
1239             List<String> srchlist = ResolverConfiguration.open().searchlist();
1240             for (String domain: srchlist) {
1241                 realm = checkRealm(domain);
1242                 if (realm != null) {
1243                     break;
1244                 }
1245             }
1246         } else {
1247             realm = checkRealm(mapRealm);
1248         }
1249         if (realm == null) {
1250             throw new KrbException(Krb5.KRB_ERR_GENERIC,
1251                                 "Unable to locate Kerberos realm");
1252         }
1253         return realm;
1254     }

1256     /**
1257      * Check if the provided realm is the correct realm
1258      * @return the realm if correct, or null otherwise
1259      */
1260     private static String checkRealm(String mapRealm) {
1261         if (DEBUG) {
1262             System.out.println("getRealmFromDNS: trying " + mapRealm);
1263         }
1264         String[] records = null;
1265         String newRealm = mapRealm;
1266         while ((records == null) && (newRealm != null)) {
1267             // locate DNS TXT record
1268             records = KrbServiceLocator.getKerberosService(newRealm);
1269             newRealm = Realm.parseRealmComponent(newRealm);
1270             // if no DNS TXT records found, try again using sub-realm
1271         }
1272         if (records != null) {
1273             for (int i = 0; i < records.length; i++) {
1274                 if (records[i].equalsIgnoreCase(mapRealm)) {
1275                     return records[i];
1276                 }
1277             }
1278         }
1279         return null;
1280     }

1282     /**
1283      * Locate KDC using DNS
1284      *
1285      * @param realm the realm for which the master KDC is desired
1286      * @return the KDC
1287      */
1288     private String getKDCFromDNS(String realm) throws KrbException {
1289         // use DNS to locate KDC
1290         String kdcs = "";
1291         String[] srvs = null;
1292         // locate DNS SRV record using UDP
1293         if (DEBUG) {
1294             System.out.println("getKDCFromDNS using UDP");
1295         }
1296         srvs = KrbServiceLocator.getKerberosService(realm, "_udp");
1297         if (srvs == null) {
1298             // locate DNS SRV record using TCP
1299             if (DEBUG) {
1300                 System.out.println("getKDCFromDNS using TCP");
1301             }
1302             srvs = KrbServiceLocator.getKerberosService(realm, "_tcp");
1303         }
1304         if (srvs == null) {
1305             // no DNS SRV records
1306             throw new KrbException(Krb5.KRB_ERR_GENERIC,
1307                 "Unable to locate KDC for realm " + realm);
1308         }
1309         if (srvs.length == 0) {
1310             return null;
1311         }
1312         for (int i = 0; i < srvs.length; i++) {
1313             kdcs += srvs[i].trim() + " ";
1314         }
1315         kdcs = kdcs.trim();
1316         if (kdcs.equals("")) {
1317             return null;
1318         }
1319         return kdcs;
1320     }

1360     private static void toStringInternal(String prefix, Object obj,
1361             StringBuffer sb) {
1362         if (obj instanceof String) {
1363             // A string value, just print it
1364             sb.append(obj).append('\n');
1365         } else if (obj instanceof Hashtable) {
1366             // A table, start a new sub-section...
1367             Hashtable<?, ?> tab = (Hashtable<?, ?>)obj;
1368             sb.append("{\n");
1369             for (Object o: tab.keySet()) {
1370                 // ...indent, print "key = ", and
1371                 sb.append(prefix).append("    ").append(o).append(" = ");
