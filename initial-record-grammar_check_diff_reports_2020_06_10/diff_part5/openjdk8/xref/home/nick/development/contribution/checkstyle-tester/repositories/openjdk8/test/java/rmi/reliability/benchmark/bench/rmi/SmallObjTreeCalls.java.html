28  package bench.rmi;
29  
30  import bench.Benchmark;
31  import java.io.Serializable;
32  import java.rmi.Remote;
33  import java.rmi.RemoteException;
34  import java.rmi.server.UnicastRemoteObject;
35  
36  /**
37   * Benchmark for testing speed of calls with small object tree parameters and
38   * return values.
39   */
40  public class SmallObjTreeCalls implements Benchmark {
41  
42      static class Node implements Serializable {
43          Object parent, left, right;
44  
45          Node(Object parent, int depth) {
46              this.parent = parent;
47              if (depth > 0) {
48                  left = new Node(this, depth - 1);
49                  right = new Node(this, depth - 1);
50              }
51          }
52      }
53  
54      interface Server extends Remote {
55          public Node call(Node val) throws RemoteException;
56      }
57  
58      static class ServerImpl extends UnicastRemoteObject implements Server {
59          public ServerImpl() throws RemoteException {
60          }
61  
62          public Node call(Node val) throws RemoteException {
63              return val;
64          }
65      }
66  
67      static class ServerFactory implements BenchServer.RemoteObjectFactory {
68          public Remote create() throws RemoteException {
69              return new ServerImpl();
70          }
71      }
72  
73      /**
74       * Issue calls using trees of small objects as parameters/return values.
75       * Arguments: <tree depth> <# calls>
76       */
77      public long run(String[] args) throws Exception {
78          int depth = Integer.parseInt(args[0]);
79          int reps = Integer.parseInt(args[1]);
80          BenchServer bsrv = Main.getBenchServer();
81          Server stub = (Server) bsrv.create(new ServerFactory());
82          Node node = new Node(null, depth);
83  
84          long start = System.currentTimeMillis();
85          for (int i = 0; i < reps; i++)
86              stub.call(node);
87          long time = System.currentTimeMillis() - start;
88  
89          bsrv.unexport(stub, true);
90          return time;
91      }
92  }
