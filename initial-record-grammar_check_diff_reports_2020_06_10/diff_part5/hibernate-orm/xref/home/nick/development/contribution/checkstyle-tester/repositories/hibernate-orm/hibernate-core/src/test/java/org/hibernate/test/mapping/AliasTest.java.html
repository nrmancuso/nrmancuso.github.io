37  /**
38   * Column aliases utilize {@link Table#getUniqueInteger()} for naming.  The
39   * unique integer used to be statically generated by the Table class, meaning
40   * it was dependent on mapping order.  HHH-2448 made the alias names
41   * deterministic by having Configuration determine the unique integers on its
42   * second pass over the Tables tree map.  AliasTest and
43   * {@link MappingReorderedAliasTest} ensure that the unique integers are the
44   * same, regardless of mapping ordering.
45   * 
46   * @author Brett Meyer
47   */
48  public class AliasTest extends BaseCoreFunctionalTestCase {
49  	
50  	/**
51  	 * Column aliases utilize {@link Table#getUniqueInteger()} for naming.  The unique integer used to be statically
52  	 * generated by the Table class, meaning it was dependent on mapping order.  HHH-2448 made the alias names
53  	 * deterministic by having Configuration determine the unique integers on its second pass over the Tables tree map.
54  	 * AliasTest and {@link MappingReorderedAliasTest} ensure that the unique integers are the same, regardless of
55  	 * mapping ordering.
56  	 */
57  	@Test
58  	@TestForIssue( jiraKey = "HHH-2448" )
59  	public void testAliasOrdering() {
60  		Iterator<Table> tables = configuration().getTableMappings();
61  		Table table1 = null;
62  		Table table2 = null;
63  		while ( tables.hasNext() ) {
64  			Table table = tables.next();
65  			if ( table.getName().equals( "Table1" ) ) {
66  				table1 = table;
67  			}
68  			else if ( table.getName().equals( "Table2" ) ) {
69  				table2 = table;
70  			}
71  		}
72  		
73  		assertTrue( table1.getUniqueInteger() < table2.getUniqueInteger() );
74  	}
75  	
76  	@Test
77  	@TestForIssue( jiraKey = "HHH-8371" )
78  	public final void testUnderscoreInColumnName() throws Throwable {
79  		final Session s = openSession();
80  		s.getTransaction().begin();
81  		
82  		UserEntity user = new UserEntity();
83  		user.setName( "foo" );
84  		s.persist(user);
85  		final ConfEntity conf =  new ConfEntity();
86  		conf.setConfKey("counter");
87  		conf.setConfValue("3");
88  		final UserConfEntity uc = new UserConfEntity();
89  		uc.setUser(user);
90  		uc.setConf(conf);
91  		conf.getUserConf().add(uc);
92  		s.persist(conf);
93  
94  		s.getTransaction().commit();
95  		s.clear();
96  		
97  		s.getTransaction().begin();
98  		user = (UserEntity) s.get(UserEntity.class, user.getId());
99  
100 		try {
101 			s.flush();
102 		}
103 		catch ( HibernateException e ) {
104 			// original issue from HHH-8371
105 			fail( "The explicit column name's underscore(s) were not considered during alias creation." );
106 		}
107 		
108 		assertNotNull( user );
109 		assertEquals( user.getName(), "foo" );
110 
111 		s.getTransaction().commit();
112 		s.close();
113 	}
