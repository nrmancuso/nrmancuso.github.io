141 		StringBuilder buf = new StringBuilder();
142 		FromElement fromElement = getFromElement();
143 		buf.append( "{propertyName=" ).append( propertyName );
144 		buf.append( ",dereferenceType=" ).append( getWalker().getASTPrinter().getTokenTypeName( dereferenceType ) );

165     public void resolveFirstChild() throws SemanticException {
166 		FromReferenceNode lhs = ( FromReferenceNode ) getFirstChild();
167 		SqlNode property = ( SqlNode ) lhs.getNextSibling();
168 
169 		// Set the attributes of the property reference expression.
170 		String propName = property.getText();
171 		propertyName = propName;
172 		// If the uresolved property path isn't set yet, just use the property name.
173 		if ( propertyPath == null ) {
174 			propertyPath = propName;
175 		}
176 		// Resolve the LHS fully, generate implicit joins.  Pass in the property name so that the resolver can
177 		// discover foreign key (id) properties.
178 		lhs.resolve( true, true, null, this );
179 		setFromElement( lhs.getFromElement() );			// The 'from element' that the property is in.
180 
181 		checkSubclassOrSuperclassPropertyReference( lhs, propName );
182 	}

184 	@Override
185     public void resolveInFunctionCall(boolean generateJoin, boolean implicitJoin) throws SemanticException {
186 		if ( isResolved() ) {
187 			return;
188 		}
189 		Type propertyType = prepareLhs();			// Prepare the left hand side and get the data type.
190 		if ( propertyType!=null && propertyType.isCollectionType() ) {
191 			resolveIndex(null);
192 		}
193 		else {
194 			resolveFirstChild();
195 			super.resolve(generateJoin, implicitJoin);
196 		}
197 	}

208 	public void resolve(boolean generateJoin, boolean implicitJoin, String classAlias, AST parent)
209 	throws SemanticException {
210 		// If this dot has already been resolved, stop now.
211 		if ( isResolved() ) {
212 			return;
213 		}
214 		Type propertyType = prepareLhs(); // Prepare the left hand side and get the data type.
215 
216 		// If there is no data type for this node, and we're at the end of the path (top most dot node), then
217 		// this might be a Java constant.
218 		if ( propertyType == null ) {
219 			if ( parent == null ) {
220 				getWalker().getLiteralProcessor().lookupConstant( this );
221 			}
222 			// If the propertyType is null and there isn't a parent, just
223 			// stop now... there was a problem resolving the node anyway.
224 			return;
225 		}
226 
227 		if ( propertyType.isComponentType() ) {
228 			// The property is a component...
229 			checkLhsIsNotCollection();
230 			dereferenceComponent( parent );
231 			initText();
232 		}
233 		else if ( propertyType.isEntityType() ) {
234 			// The property is another class..
235 			checkLhsIsNotCollection();
236 			dereferenceEntity( ( EntityType ) propertyType, implicitJoin, classAlias, generateJoin, parent );
237 			initText();
238 		}
239 		else if ( propertyType.isCollectionType() ) {
240 			// The property is a collection...
241 			checkLhsIsNotCollection();
242 			dereferenceCollection( ( CollectionType ) propertyType, implicitJoin, false, classAlias, parent );
243 		}
244 		else {
245 			// Otherwise, this is a primitive type.
246 			if ( ! CollectionProperties.isAnyCollectionProperty( propertyName ) ) {
247 				checkLhsIsNotCollection();
248 			}
249 			dereferenceType = DEREF_PRIMITIVE;
250 			initText();
251 		}
252 		setResolved();
253 	}

255 	private void initText() {
256 		String[] cols = getColumns();
257 		String text = StringHelper.join( ", ", cols );
258 		boolean countDistinct = getWalker().isInCountDistinct()
259 				&& getWalker().getSessionFactoryHelper().getFactory().getDialect().requiresParensForTupleDistinctCounts();
260 		if ( cols.length > 1 &&
261 				( getWalker().isComparativeExpressionClause() || countDistinct ) ) {
262 			text = "(" + text + ")";
263 		}
264 		setText( text );
265 	}

270 		return getDataType();

273 	private void dereferenceCollection(CollectionType collectionType, boolean implicitJoin, boolean indexed, String classAlias, AST parent)
274 	throws SemanticException {
275 
276 		dereferenceType = DEREF_COLLECTION;
277 		String role = collectionType.getRole();
278 
279 		//foo.bars.size (also handles deprecated stuff like foo.bars.maxelement for backwardness)
280 		boolean isSizeProperty = getNextSibling()!=null &&
281 			CollectionProperties.isAnyCollectionProperty( getNextSibling().getText() );
282 
283 		if ( isSizeProperty ) indexed = true; //yuck!
284 
285 		QueryableCollection queryableCollection = getSessionFactoryHelper().requireQueryableCollection( role );
286 		String propName = getPath();
287 		FromClause currentFromClause = getWalker().getCurrentFromClause();
288 
289 		// If the lhs of the join is a "component join", we need to go back to the
290 		// first non-component-join as the origin to properly link aliases and
291 		// join columns
292 		FromElement lhsFromElement = getLhs().getFromElement();
293 		while ( lhsFromElement != null && ComponentJoin.class.isInstance( lhsFromElement ) ) {
294 			lhsFromElement = lhsFromElement.getOrigin();
295 		}
296 		if ( lhsFromElement == null ) {
297 			throw new QueryException( "Unable to locate appropriate lhs" );
298 		}
299 
300 		// determine whether we should use the table name or table alias to qualify the column names...
301 		// we need to use the table-name when:
302 		//		1) the top-level statement is not a SELECT
303 		//		2) the LHS FromElement is *the* FromElement from the top-level statement
304 		//
305 		// there is a caveat here.. if the update/delete statement are "multi-table" we should continue to use
306 		// the alias also, even if the FromElement is the root one...
307 		//
308 		// in all other cases, we should use the table alias
309 		if ( getWalker().getStatementType() != SqlTokenTypes.SELECT ) {
310 			if ( isFromElementUpdateOrDeleteRoot( lhsFromElement ) ) {
311 				// at this point we know we have the 2 conditions above,
312 				// lets see if we have the mentioned "multi-table" caveat...
313 				boolean useAlias = false;
314 				if ( getWalker().getStatementType() != SqlTokenTypes.INSERT ) {
315 					final Queryable persister = lhsFromElement.getQueryable();
316 					if ( persister.isMultiTable() ) {
317 						useAlias = true;
318 					}
319 				}
320 				if ( ! useAlias ) {
321 					final String lhsTableName = lhsFromElement.getQueryable().getTableName();
322 					columns = getFromElement().toColumns( lhsTableName, propertyPath, false, true );
323 				}
324 			}
325 		}
326 
327 		// We do not look for an existing join on the same path, because
328 		// it makes sense to join twice on the same collection role
329 		FromElementFactory factory = new FromElementFactory(
330 		        currentFromClause,
331 				lhsFromElement,
332 				propName,
333 				classAlias,
334 		        getColumns(),
335 		        implicitJoin
336 		);
337 		FromElement elem = factory.createCollection( queryableCollection, role, joinType, fetch, indexed );
338 
339 		LOG.debugf( "dereferenceCollection() : Created new FROM element for %s : %s", propName, elem );
340 
341 		setImpliedJoin( elem );
342 		setFromElement( elem );	// This 'dot' expression now refers to the resulting from element.
343 
344 		if ( isSizeProperty ) {
345 			elem.setText("");
346 			elem.setUseWhereFragment(false);
347 		}
348 
349 		if ( !implicitJoin ) {
350 			EntityPersister entityPersister = elem.getEntityPersister();
351 			if ( entityPersister != null ) {
352 				getWalker().addQuerySpaces( entityPersister.getQuerySpaces() );
353 			}
354 		}
355 		getWalker().addQuerySpaces( queryableCollection.getCollectionSpaces() );	// Always add the collection's query spaces.
356 	}

358 	private void dereferenceEntity(EntityType entityType, boolean implicitJoin, String classAlias, boolean generateJoin, AST parent) throws SemanticException {
359 		checkForCorrelatedSubquery( "dereferenceEntity" );
360 		// three general cases we check here as to whether to render a physical SQL join:
361 		// 1) is our parent a DotNode as well?  If so, our property reference is
362 		// 		being further de-referenced...
363 		// 2) is this a DML statement
364 		// 3) we were asked to generate any needed joins (generateJoins==true) *OR*
365 		//		we are currently processing a select or from clause
366 		// (an additional check is the REGRESSION_STYLE_JOIN_SUPPRESSION check solely intended for the test suite)
367 		//
368 		// The REGRESSION_STYLE_JOIN_SUPPRESSION is an additional check
369 		// intended solely for use within the test suite.  This forces the
370 		// implicit join resolution to behave more like the classic parser.
371 		// The underlying issue is that classic translator is simply wrong
372 		// about its decisions on whether or not to render an implicit join
373 		// into a physical SQL join in a lot of cases.  The piece it generally
374 		// tends to miss is that INNER joins effect the results by further
375 		// restricting the data set!  A particular manifestation of this is
376 		// the fact that the classic translator will skip the physical join
377 		// for ToOne implicit joins *if the query is shallow*; the result
378 		// being that Query.list() and Query.iterate() could return
379 		// different number of results!
380 		DotNode parentAsDotNode = null;
381 		String property = propertyName;
382 		final boolean joinIsNeeded;
383 
384 		if ( isDotNode( parent ) ) {
385 			// our parent is another dot node, meaning we are being further dereferenced.
386 			// thus we need to generate a join unless the parent refers to the associated
387 			// entity's PK (because 'our' table would know the FK).
388 			parentAsDotNode = ( DotNode ) parent;
389 			property = parentAsDotNode.propertyName;
390 			joinIsNeeded = generateJoin && !isReferenceToPrimaryKey( parentAsDotNode.propertyName, entityType );
391 		}
392 		else if ( ! getWalker().isSelectStatement() ) {
393 			// in non-select queries, the only time we should need to join is if we are in a subquery from clause
394 			joinIsNeeded = getWalker().getCurrentStatementType() == SqlTokenTypes.SELECT && getWalker().isInFrom();
395 		}
396 		else if ( REGRESSION_STYLE_JOIN_SUPPRESSION ) {
397 			// this is the regression style determination which matches the logic of the classic translator
398 			joinIsNeeded = generateJoin && ( !getWalker().isInSelect() || !getWalker().isShallowQuery() );
399 		}
400 		else {
401 			joinIsNeeded = generateJoin || ( getWalker().isInSelect() || getWalker().isInFrom() );
402 		}
403 
404 		if ( joinIsNeeded ) {
405 			dereferenceEntityJoin( classAlias, entityType, implicitJoin, parent );
406 		}
407 		else {
408 			dereferenceEntityIdentifier( property, parentAsDotNode );
409 		}
410 
411 	}

417 	private void dereferenceEntityJoin(String classAlias, EntityType propertyType, boolean impliedJoin, AST parent)
418 	throws SemanticException {
419 		dereferenceType = DEREF_ENTITY;
420         if (LOG.isDebugEnabled()) LOG.debugf("dereferenceEntityJoin() : generating join for %s in %s (%s) parent = %s",
421                                              propertyName,
422                                              getFromElement().getClassName(),
423                                              classAlias == null ? "<no alias>" : classAlias,
424                                              ASTUtil.getDebugString(parent));
425 		// Create a new FROM node for the referenced class.
426 		String associatedEntityName = propertyType.getAssociatedEntityName();
427 		String tableAlias = getAliasGenerator().createName( associatedEntityName );
428 
429 		String[] joinColumns = getColumns();
430 		String joinPath = getPath();
431 
432 		if ( impliedJoin && getWalker().isInFrom() ) {
433 			joinType = getWalker().getImpliedJoinType();
434 		}
435 
436 		FromClause currentFromClause = getWalker().getCurrentFromClause();
437 		FromElement elem = currentFromClause.findJoinByPath( joinPath );
438 
439 ///////////////////////////////////////////////////////////////////////////////
440 //
441 // This is the piece which recognizes the condition where an implicit join path
442 // resolved earlier in a correlated subquery is now being referenced in the
443 // outer query.  For 3.0final, we just let this generate a second join (which
444 // is exactly how the old parser handles this).  Eventually we need to add this
445 // logic back in and complete the logic in FromClause.promoteJoin; however,
446 // FromClause.promoteJoin has its own difficulties (see the comments in
447 // FromClause.promoteJoin).
448 //
449 //		if ( elem == null ) {
450 //			// see if this joinPath has been used in a "child" FromClause, and if so
451 //			// promote that element to the outer query
452 //			FromClause currentNodeOwner = getFromElement().getFromClause();
453 //			FromClause currentJoinOwner = currentNodeOwner.locateChildFromClauseWithJoinByPath( joinPath );
454 //			if ( currentJoinOwner != null && currentNodeOwner != currentJoinOwner ) {
455 //				elem = currentJoinOwner.findJoinByPathLocal( joinPath );
456 //				if ( elem != null ) {
457 //					currentFromClause.promoteJoin( elem );
458 //					// EARLY EXIT!!!
459 //					return;
460 //				}
461 //			}
462 //		}
463 //
464 ///////////////////////////////////////////////////////////////////////////////
465 
466 		boolean found = elem != null;
467 		// even though we might find a pre-existing element by join path, we may not be able to reuse it...
468 		boolean useFoundFromElement = found && canReuse( elem );
469 
470 		if ( ! useFoundFromElement ) {
471 			// If this is an implied join in a from element, then use the impled join type which is part of the
472 			// tree parser's state (set by the gramamar actions).
473 			JoinSequence joinSequence = getSessionFactoryHelper()
474 				.createJoinSequence( impliedJoin, propertyType, tableAlias, joinType, joinColumns );
475 
476 			// If the lhs of the join is a "component join", we need to go back to the
477 			// first non-component-join as the origin to properly link aliases and
478 			// join columns
479 			FromElement lhsFromElement = getLhs().getFromElement();
480 			while ( lhsFromElement != null &&  ComponentJoin.class.isInstance( lhsFromElement ) ) {
481 				lhsFromElement = lhsFromElement.getOrigin();
482 			}
483 			if ( lhsFromElement == null ) {
484 				throw new QueryException( "Unable to locate appropriate lhs" );
485 			}
486 
487 			FromElementFactory factory = new FromElementFactory(
488 			        currentFromClause,
489 					lhsFromElement,
490 					joinPath,
491 					classAlias,
492 					joinColumns,
493 					impliedJoin
494 			);
495 			elem = factory.createEntityJoin(
496 					associatedEntityName,
497 					tableAlias,
498 					joinSequence,
499 					fetch,
500 					getWalker().isInFrom(),
501 					propertyType
502 			);
503 		}
504 		else {
505 			// NOTE : addDuplicateAlias() already performs nullness checks on the alias.
506 			currentFromClause.addDuplicateAlias( classAlias, elem );
507 		}
508 		setImpliedJoin( elem );
509 		getWalker().addQuerySpaces( elem.getEntityPersister().getQuerySpaces() );
510 		setFromElement( elem );	// This 'dot' expression now refers to the resulting from element.
511 	}

513 	private boolean canReuse(FromElement fromElement) {
514 		// if the from-clauses are the same, we can be a little more aggressive in terms of what we reuse
515 		if ( fromElement.getFromClause() == getWalker().getCurrentFromClause() ) {
516 			return true;
517 		}
518 
519 		// otherwise (subquery case) dont reuse the fromElement if we are processing the from-clause of the subquery
520 		return getWalker().getCurrentClauseType() != SqlTokenTypes.FROM;
521 	}

527 			if ( dotLhs.getImpliedJoin() != null ) {
528 				this.impliedJoin = dotLhs.getImpliedJoin();
529 			}
530 		}

538 	/**
539 	 * Is the given property name a reference to the primary key of the associated
540 	 * entity construed by the given entity type?
541 	 * <p/>
542 	 * For example, consider a fragment like order.customer.id
543 	 * (where order is a from-element alias).  Here, we'd have:
544 	 * propertyName = "id" AND
545 	 * owningType = ManyToOneType(Customer)
546 	 * and are being asked to determine whether "customer.id" is a reference
547 	 * to customer's PK...
548 	 *
549 	 * @param propertyName The name of the property to check.
550 	 * @param owningType The type represeting the entity "owning" the property
551 	 * @return True if propertyName references the entity's (owningType->associatedEntity)
552 	 * primary key; false otherwise.
553 	 */
554 	private boolean isReferenceToPrimaryKey(String propertyName, EntityType owningType) {
555 		EntityPersister persister = getSessionFactoryHelper()
556 				.getFactory()
557 				.getEntityPersister( owningType.getAssociatedEntityName() );
558 		if ( persister.getEntityMetamodel().hasNonIdentifierPropertyNamedId() ) {
559 			// only the identifier property field name can be a reference to the associated entity's PK...
560 			return propertyName.equals( persister.getIdentifierPropertyName() ) && owningType.isReferenceToPrimaryKey();
561 		}
562         // here, we have two possibilities:
563         // 1) the property-name matches the explicitly identifier property name
564         // 2) the property-name matches the implicit 'id' property name
565         // the referenced node text is the special 'id'
566         if (EntityPersister.ENTITY_ID.equals(propertyName)) return owningType.isReferenceToPrimaryKey();
567         String keyPropertyName = getSessionFactoryHelper().getIdentifierOrUniqueKeyPropertyName(owningType);
568         return keyPropertyName != null && keyPropertyName.equals(propertyName) && owningType.isReferenceToPrimaryKey();
569 	}

583 		if ( getLhs().getDataType() != null && getLhs().getDataType().isCollectionType() ) {

592 	private void dereferenceEntityIdentifier(String propertyName, DotNode dotParent) {
593 		// special shortcut for id properties, skip the join!
594 		// this must only occur at the _end_ of a path expression
595 		if ( LOG.isDebugEnabled() ) {
596 			LOG.debugf( "dereferenceShortcut() : property %s in %s does not require a join.",
597 					propertyName,
598 					getFromElement().getClassName() );
599 		}
600 
601 		initText();
602 		setPropertyNameAndPath( dotParent ); // Set the unresolved path in this node and the parent.
603 		// Set the text for the parent.
604 		if ( dotParent != null ) {
605 			dotParent.dereferenceType = DEREF_IDENTIFIER;
606 			dotParent.setText( getText() );
607 			dotParent.columns = getColumns();
608 		}
609 	}

611 	private void setPropertyNameAndPath(AST parent) {
612 		if ( isDotNode( parent ) ) {
613 			DotNode dotNode = ( DotNode ) parent;
614 			AST lhs = dotNode.getFirstChild();
615 			AST rhs = lhs.getNextSibling();
616 			propertyName = rhs.getText();
617 			propertyPath = propertyPath + "." + propertyName; // Append the new property name onto the unresolved path.
618 			dotNode.propertyPath = propertyPath;
619 			LOG.debugf( "Unresolved property path is now '%s'", dotNode.propertyPath );
620 		}
621 		else {
622 			LOG.debugf( "Terminal propertyPath = [%s]", propertyPath );
623 		}
624 	}

662 		if ( path == null ) {

664 			if ( lhs == null ) {
665 				path = getText();
666 			}
667 			else {

671 		}
672 		return path;

684 	/**
685 	 * Special method to resolve expressions in the SELECT list.
686 	 *
687 	 * @throws SemanticException if this cannot be resolved.
688 	 */
689 	public void resolveSelectExpression() throws SemanticException {
690 		if ( getWalker().isShallowQuery() || getWalker().getCurrentFromClause().isSubQuery() ) {
691 			resolve(false, true);
692 		}
693 		else {
694 			resolve(true, false);
695 			Type type = getDataType();
696 			if ( type.isEntityType() ) {
697 				FromElement fromElement = getFromElement();
698 				fromElement.setIncludeSubclasses( true ); // Tell the destination fromElement to 'includeSubclasses'.
699 				if ( useThetaStyleImplicitJoins ) {
700 					fromElement.getJoinSequence().setUseThetaStyle( true );	// Use theta style (for regression)
701 					// Move the node up, after the origin node.
702 					FromElement origin = fromElement.getOrigin();
703 					if ( origin != null ) {
704 						ASTUtil.makeSiblingOfParent( origin, fromElement );
705 					}
706 				}
707 			}
708 		}
709 
710 		FromReferenceNode lhs = getLhs();
711 		while ( lhs != null ) {
712 			checkSubclassOrSuperclassPropertyReference( lhs, lhs.getNextSibling().getText() );
713 			lhs = ( FromReferenceNode ) lhs.getFirstChild();
714 		}
715 	}

723 	private boolean checkSubclassOrSuperclassPropertyReference(FromReferenceNode lhs, String propertyName) {
724 		if ( lhs != null && !( lhs instanceof IndexNode ) ) {
725 			final FromElement source = lhs.getFromElement();
726 			if ( source != null ) {
727 				source.handlePropertyBeingDereferenced( lhs.getDataType(), propertyName );
728 			}
729 		}
730 		return false;
