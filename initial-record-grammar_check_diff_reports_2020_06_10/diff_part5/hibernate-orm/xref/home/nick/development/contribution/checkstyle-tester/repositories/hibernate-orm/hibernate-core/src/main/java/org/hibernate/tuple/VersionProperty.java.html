25  package org.hibernate.tuple;
26  import org.hibernate.engine.spi.CascadeStyle;
27  import org.hibernate.engine.spi.VersionValue;
28  import org.hibernate.type.Type;
29  
30  /**
31   * Represents a version property within the Hibernate runtime-metamodel.
32   *
33   * @author Steve Ebersole
34   */
35  public class VersionProperty extends StandardProperty {
36  
37      private final VersionValue unsavedValue;
38  
39      /**
40       * Constructs VersionProperty instances.
41       *
42       * @param name The name by which the property can be referenced within
43       * its owner.
44       * @param node The node name to use for XML-based representation of this
45       * property.
46       * @param type The Hibernate Type of this property.
47       * @param lazy Should this property be handled lazily?
48       * @param insertable Is this property an insertable value?
49       * @param updateable Is this property an updateable value?
50       * @param insertGenerated Is this property generated in the database on insert?
51       * @param updateGenerated Is this property generated in the database on update?
52       * @param nullable Is this property a nullable value?
53       * @param checkable Is this property a checkable value?
54       * @param versionable Is this property a versionable value?
55       * @param cascadeStyle The cascade style for this property's value.
56       * @param unsavedValue The value which, if found as the value of
57       * this (i.e., the version) property, represents new (i.e., un-saved)
58       * instances of the owning entity.
59       */
60      public VersionProperty(
61              String name,
62              String node,
63              Type type,
64              boolean lazy,
65              boolean insertable,
66              boolean updateable,
67              boolean insertGenerated,
68              boolean updateGenerated,
69              boolean nullable,
70              boolean checkable,
71              boolean versionable,
72              CascadeStyle cascadeStyle,
73              VersionValue unsavedValue) {
74          super( name, node, type, lazy, insertable, updateable, insertGenerated, updateGenerated, nullable, checkable, versionable, cascadeStyle, null );
75          this.unsavedValue = unsavedValue;
76      }
77  
78      public VersionValue getUnsavedValue() {
79          return unsavedValue;
80      }
81  }
