24  package org.hibernate.envers.entities.mapper.relation.lazy.initializor;
25  
26  import java.lang.reflect.InvocationTargetException;
27  import java.util.List;
28  import java.util.Map;
29  
30  import org.hibernate.envers.configuration.AuditConfiguration;
31  import org.hibernate.envers.entities.mapper.relation.MiddleComponentData;
32  import org.hibernate.envers.entities.mapper.relation.query.RelationQueryGenerator;
33  import org.hibernate.envers.exception.AuditException;
34  import org.hibernate.envers.reader.AuditReaderImplementor;
35  import org.hibernate.internal.util.ReflectHelper;
36  
37  /**
38   * Initializes a map.
39   * @author Adam Warski (adam at warski dot org)
40   */
41  public class MapCollectionInitializor<T extends Map> extends AbstractCollectionInitializor<T> {
42      protected final Class<? extends T> collectionClass;
43      private final MiddleComponentData elementComponentData;
44      private final MiddleComponentData indexComponentData;
45  
46      public MapCollectionInitializor(AuditConfiguration verCfg,
47                                      AuditReaderImplementor versionsReader,
48                                      RelationQueryGenerator queryGenerator,
49                                      Object primaryKey, Number revision, boolean removed,
50                                      Class<? extends T> collectionClass,
51                                      MiddleComponentData elementComponentData,
52                                      MiddleComponentData indexComponentData) {
53          super(verCfg, versionsReader, queryGenerator, primaryKey, revision, removed);
54  
55          this.collectionClass = collectionClass;
56          this.elementComponentData = elementComponentData;
57          this.indexComponentData = indexComponentData;
58      }
59  
60      protected T initializeCollection(int size) {
61          try {
62              return (T) ReflectHelper.getDefaultConstructor(collectionClass).newInstance();
63          } catch (InstantiationException e) {
64              throw new AuditException(e);
65          } catch (IllegalAccessException e) {
66              throw new AuditException(e);
67          } catch (InvocationTargetException e) {
68              throw new AuditException(e);
69          }
70      }
71  
72      @SuppressWarnings({"unchecked"})
73      protected void addToCollection(T collection, Object collectionRow) {
74          // collectionRow will be the actual object if retrieved from audit relation or middle table
75          // otherwise it will be a List
76          Object elementData = collectionRow;
77  	Object indexData = collectionRow;
78  	if (collectionRow instanceof java.util.List) {
79              elementData = ((List) collectionRow).get(elementComponentData.getComponentIndex());
80              indexData = ((List) collectionRow).get(indexComponentData.getComponentIndex());
81  	}
82          Object element = elementComponentData.getComponentMapper().mapToObjectFromFullMap(entityInstantiator,
83                  (Map<String, Object>) elementData, null, revision);
84  
85          Object index = indexComponentData.getComponentMapper().mapToObjectFromFullMap(entityInstantiator,
86                  (Map<String, Object>) indexData, element, revision);
87  
88          collection.put(index, element);
89      }
90  }
