61  	/**
62  	 * Retrieves a jandex annotation element value. If the value is {@code null}, the default value specified in the
63  	 * annotation class is retrieved instead.
64  	 * <p>
65  	 * There are two special cases. {@code Class} parameters should be retrieved as strings (and then can later be
66  	 * loaded) and enumerated values should be retrieved via {@link #getEnumValue(AnnotationInstance, String, Class)}.
67  	 * </p>
68  	 *
69  	 * @param annotation the annotation containing the element with the supplied name
70  	 * @param element the name of the element value to be retrieve
71  	 * @param type the type of element to retrieve. The following types are supported:
72  	 * <ul>
73  	 * <li>Byte</li>
74  	 * <li>Short</li>
75  	 * <li>Integer</li>
76  	 * <li>Character</li>
77  	 * <li>Float</li>
78  	 * <li>Double</li>
79  	 * <li>Long</li>
80  	 * <li>Boolean</li>
81  	 * <li>String</li>
82  	 * <li>AnnotationInstance</li>
83  	 *
84  	 * @return the value if not {@code null}, else the default value if not
85  	 *         {@code null}, else {@code null}.
86  	 *
87  	 * @throws AssertionFailure in case the specified {@code type} is a class instance or the specified type causes a {@code ClassCastException}
88  	 * when retrieving the value.
89  	 */
90  	@SuppressWarnings("unchecked")
91  	public static <T> T getValue(AnnotationInstance annotation, String element, Class<T> type) throws AssertionFailure {
92  		if ( Class.class.equals( type ) ) {
93  			throw new AssertionFailure(
94  					"Annotation parameters of type Class should be retrieved as strings (fully qualified class names)"
95  			);
96  		}
97  
98  		// try getting the untyped value from Jandex
99  		AnnotationValue annotationValue = annotation.value( element );
100 
101 		try {
102 			if ( annotationValue != null ) {
103 				return explicitAnnotationParameter( annotationValue, type );
104 			}
105 			else {
106 				return defaultAnnotationParameter( getDefaultValue( annotation, element ), type );
107 			}
108 		}
109 		catch ( ClassCastException e ) {
110 			throw new AssertionFailure(
111 					String.format(
112 							"the annotation property %s of annotation %s is not of type %s",
113 							element,
114 							annotation.name(),
115 							type.getName()
116 					)
117 			);
118 		}
119 	}

121 	/**
122 	 * Retrieves a jandex annotation element value, converting it to the supplied enumerated type.  If the value is
123 	 * <code>null</code>, the default value specified in the annotation class is retrieved instead.
124 	 *
125 	 * @param <T> an enumerated type
126 	 * @param annotation the annotation containing the enumerated element with the supplied name
127 	 * @param element the name of the enumerated element value to be retrieve
128 	 * @param type the type to which to convert the value before being returned
129 	 *
130 	 * @return the value converted to the supplied enumerated type if the value is not <code>null</code>, else the default value if
131 	 *         not <code>null</code>, else <code>null</code>.
132 	 *
133 	 * @see #getValue(AnnotationInstance, String, Class)
134 	 */
135 	@SuppressWarnings("unchecked")
136 	public static <T extends Enum<T>> T getEnumValue(AnnotationInstance annotation, String element, Class<T> type) {
137 		AnnotationValue val = annotation.value( element );
138 		if ( val == null ) {
139 			return (T) getDefaultValue( annotation, element );
140 		}
141 		return Enum.valueOf( type, val.asEnum() );
142 	}

144 	/**
145 	 * Expects a method or field annotation target and returns the property name for this target
146 	 *
147 	 * @param target the annotation target
148 	 *
149 	 * @return the property name of the target. For a field it is the field name and for a method name it is
150 	 *         the method name stripped of 'is', 'has' or 'get'
151 	 */
152 	public static String getPropertyName(AnnotationTarget target) {
153 		if ( !( target instanceof MethodInfo || target instanceof FieldInfo ) ) {
154 			throw new AssertionFailure( "Unexpected annotation target " + target.toString() );
155 		}
156 
157 		if ( target instanceof FieldInfo ) {
158 			return ( (FieldInfo) target ).name();
159 		}
160 		else {
161 			final String methodName = ( (MethodInfo) target ).name();
162 			String propertyName;
163 			if ( methodName.startsWith( "is" ) ) {
164 				propertyName = Introspector.decapitalize( methodName.substring( 2 ) );
165 			}
166 			else if ( methodName.startsWith( "has" ) ) {
167 				propertyName = Introspector.decapitalize( methodName.substring( 3 ) );
168 			}
169 			else if ( methodName.startsWith( "get" ) ) {
170 				propertyName = Introspector.decapitalize( methodName.substring( 3 ) );
171 			}
172 			else {
173 				throw new AssertionFailure( "Expected a method following the Java Bean notation" );
174 			}
175 			return propertyName;
176 		}
177 	}

179 	/**
180 	 * @param classInfo the class info from which to retrieve the annotation instance
181 	 * @param annotationName the annotation to retrieve from the class info
182 	 *
183 	 * @return the single annotation defined on the class or {@code null} in case the annotation is not specified at all
184 	 *
185 	 * @throws org.hibernate.AssertionFailure in case there is there is more than one annotation of this type.
186 	 */
187 	public static AnnotationInstance getSingleAnnotation(ClassInfo classInfo, DotName annotationName)
188 			throws AssertionFailure {
189 		return getSingleAnnotation( classInfo.annotations(), annotationName );
190 	}

192 	/**
193 	 * @param annotations List of annotation instances keyed against their dot name.
194 	 * @param annotationName the annotation to retrieve from map
195 	 *
196 	 * @return the single annotation of the specified dot name or {@code null} in case the annotation is not specified at all
197 	 *
198 	 * @throws org.hibernate.AssertionFailure in case there is there is more than one annotation of this type.
199 	 */
200 	public static AnnotationInstance getSingleAnnotation(Map<DotName, List<AnnotationInstance>> annotations, DotName annotationName)
201 			throws AssertionFailure {
202 		List<AnnotationInstance> annotationList = annotations.get( annotationName );
203 		if ( annotationList == null ) {
204 			return null;
205 		}
206 		else if ( annotationList.size() == 1 ) {
207 			return annotationList.get( 0 );
208 		}
209 		else {
210 			throw new AssertionFailure(
211 					"Found more than one instance of the annotation "
212 							+ annotationList.get( 0 ).name().toString()
213 							+ ". Expected was one."
214 			);
215 		}
216 	}

218 	/**
219 	 * @param annotations List of annotation instances keyed against their dot name.
220 	 * @param annotationName the annotation to check
221 	 *
222 	 * @return returns {@code true} if the map contains only a single instance of specified annotation or {@code false} otherwise.
223 	 *
224 	 * @throws org.hibernate.AssertionFailure in case there is there is more than one annotation of this type.
225 	 */
226 	public static boolean containsSingleAnnotations(Map<DotName, List<AnnotationInstance>> annotations, DotName annotationName)
227 			throws AssertionFailure {
228 		return getSingleAnnotation( annotations, annotationName ) != null;
229 	}

231 	/**
232 	 * Creates a jandex index for the specified classes
233 	 *
234 	 * @param classLoaderService class loader service
235 	 * @param classes the classes to index
236 	 *
237 	 * @return an annotation repository w/ all the annotation discovered in the specified classes
238 	 */
239 	public static Index indexForClass(ClassLoaderService classLoaderService, Class<?>... classes) {
240 		Indexer indexer = new Indexer();
241 		for ( Class<?> clazz : classes ) {
242 			InputStream stream = classLoaderService.locateResourceStream(
243 					clazz.getName().replace( '.', '/' ) + ".class"
244 			);
245 			try {
246 				indexer.index( stream );
247 			}
248 			catch ( IOException e ) {
249 				StringBuilder builder = new StringBuilder();
250 				builder.append( "[" );
251 				int count = 0;
252 				for ( Class<?> c : classes ) {
253 					builder.append( c.getName() );
254 					if ( count < classes.length - 1 ) {
255 						builder.append( "," );
256 					}
257 					count++;
258 				}
259 				builder.append( "]" );
260 				throw new HibernateException( "Unable to create annotation index for " + builder.toString() );
261 			}
262 		}
263 		return indexer.complete();
264 	}

266 	public static Map<DotName, List<AnnotationInstance>> getMemberAnnotations(ClassInfo classInfo, String name) {
267 		if ( classInfo == null ) {
268 			throw new IllegalArgumentException( "classInfo cannot be null" );
269 		}
270 
271 		if ( name == null ) {
272 			throw new IllegalArgumentException( "name cannot be null" );
273 		}
274 
275 		Map<DotName, List<AnnotationInstance>> annotations = new HashMap<DotName, List<AnnotationInstance>>();
276 		for ( List<AnnotationInstance> annotationList : classInfo.annotations().values() ) {
277 			for ( AnnotationInstance instance : annotationList ) {
278 				String targetName = null;
279 				if ( instance.target() instanceof FieldInfo ) {
280 					targetName = ( (FieldInfo) instance.target() ).name();
281 				}
282 				else if ( instance.target() instanceof MethodInfo ) {
283 					targetName = ( (MethodInfo) instance.target() ).name();
284 				}
285 				if ( targetName != null && name.equals( targetName ) ) {
286 					addAnnotationToMap( instance, annotations );
287 				}
288 			}
289 		}
290 		return annotations;
291 	}
292 
293 	private static void addAnnotationToMap(AnnotationInstance instance, Map<DotName, List<AnnotationInstance>> annotations) {
294 		DotName dotName = instance.name();
295 		List<AnnotationInstance> list;
296 		if ( annotations.containsKey( dotName ) ) {
297 			list = annotations.get( dotName );
298 		}
299 		else {
300 			list = new ArrayList<AnnotationInstance>();
301 			annotations.put( dotName, list );
302 		}
303 		list.add( instance );
304 	}

306 	private static Object getDefaultValue(AnnotationInstance annotation, String element) {
307 		String name = annotation.name().toString();
308 		String fqElement = name + '.' + element;
309 		Object val = DEFAULT_VALUES_BY_ELEMENT.get( fqElement );
310 		if ( val != null ) {
311 			return val;
312 		}
313 		try {
314 			val = Index.class.getClassLoader().loadClass( name ).getMethod( element ).getDefaultValue();
315 			DEFAULT_VALUES_BY_ELEMENT.put( fqElement, val );
316 			return val == null ? null : val;
317 		}
318 		catch ( RuntimeException error ) {
319 			throw error;
320 		}
321 		catch ( Exception error ) {
322 			throw new AssertionFailure(
323 					String.format( "The annotation %s does not define a parameter '%s'", name, element ),
324 					error
325 			);
326 		}
327 	}

329 	private static <T> T defaultAnnotationParameter(Object defaultValue, Class<T> type) {
330 		Object returnValue = defaultValue;
331 
332 		// resolve some mismatches between what's stored in jandex and what the defaults are for annotations
333 		// in case of nested annotation arrays, jandex returns arrays of AnnotationInstances, hence we return
334 		// an empty array of this type here
335 		if ( defaultValue.getClass().isArray() && defaultValue.getClass().getComponentType().isAnnotation() ) {
336 			returnValue = new AnnotationInstance[0];
337 		}
338 		return type.cast( returnValue );
339 	}
340 
341 	private static <T> T explicitAnnotationParameter(AnnotationValue annotationValue, Class<T> type) {
342 		Object returnValue = annotationValue.value();
343 
344 		// if the jandex return type is Type we actually try to retrieve a class parameter
345 		// for our purposes we just return the fqcn of the class
346 		if ( returnValue instanceof Type ) {
347 			returnValue = ( (Type) returnValue ).name().toString();
348 		}
349 
350 		// arrays we have to handle explicitly
351 		if ( type.isArray() ) {
352 			AnnotationValue[] values = (AnnotationValue[]) returnValue;
353 			Class<?> componentType = type.getComponentType();
354 			Object[] arr = (Object[]) Array.newInstance( componentType, values.length );
355 			for ( int i = 0; i < values.length; i++ ) {
356 				arr[i] = componentType.cast( values[i].value() );
357 			}
358 			returnValue = arr;
359 		}
360 
361 		return type.cast( returnValue );
362 	}
