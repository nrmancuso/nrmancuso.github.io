24  package org.hibernate.type;
25  
26  import java.io.Serializable;
27  
28  import org.hibernate.type.descriptor.java.SerializableTypeDescriptor;
29  import org.hibernate.type.descriptor.sql.VarbinaryTypeDescriptor;
30  
31  /**
32   * A type that maps between a {@link java.sql.Types#VARBINARY VARBINARY} and {@link Serializable} classes.
33   * <p/>
34   * Notice specifically the 2 forms:<ul>
35   * <li>{@link #INSTANCE} indicates a mapping using the {@link Serializable} interface itself.</li>
36   * <li>{@link #SerializableType(Class)} indicates a mapping using the specific class</li>
37   * </ul>
38   * The important distinction has to do with locating the appropriate {@link ClassLoader} to use during deserialization.
39   * In the fist form we are always using the {@link ClassLoader} of the JVM (Hibernate will always fallback to trying
40   * its classloader as well).  The second form is better at targeting the needed {@link ClassLoader} actually needed.
41   *
42   * @author Gavin King
43   * @author Steve Ebersole
44   */
45  public class SerializableType<T extends Serializable> extends AbstractSingleColumnStandardBasicType<T> {
46  	public static final SerializableType<Serializable> INSTANCE = new SerializableType<Serializable>( Serializable.class );
47  
48  	private final Class<T> serializableClass;
49  
50  	public SerializableType(Class<T> serializableClass) {
51  		super( VarbinaryTypeDescriptor.INSTANCE, new SerializableTypeDescriptor<T>( serializableClass )  );
52  		this.serializableClass = serializableClass;
53  	}
54  
55  	public String getName() {
56  		return (serializableClass==Serializable.class) ? "serializable" : serializableClass.getName();
57  	}
58  }
