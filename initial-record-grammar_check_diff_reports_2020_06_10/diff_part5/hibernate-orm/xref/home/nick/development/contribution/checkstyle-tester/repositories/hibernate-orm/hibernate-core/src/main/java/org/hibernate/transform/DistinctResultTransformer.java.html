25  package org.hibernate.transform;
26  import java.util.ArrayList;
27  import java.util.HashSet;
28  import java.util.List;
29  import java.util.Set;
30  
31  import org.jboss.logging.Logger;
32  
33  import org.hibernate.internal.CoreMessageLogger;
34  
35  /**
36   * Distinctions the result tuples in the final result based on the defined
37   * equality of the tuples.
38   * <p/>
39   * Since this transformer is stateless, all instances would be considered equal.
40   * So for optimization purposes we limit it to a single, singleton {@link #INSTANCE instance}.
41   *
42   * @author Steve Ebersole
43   */
44  public class DistinctResultTransformer extends BasicTransformerAdapter {
45  
46  	public static final DistinctResultTransformer INSTANCE = new DistinctResultTransformer();
47  
48      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class,
49                                                                         DistinctResultTransformer.class.getName());
50  
51  	/**
52  	 * Helper class to handle distincting
53  	 */
54  	private static final class Identity {
55  		final Object entity;
56  
57  		private Identity(Object entity) {
58  			this.entity = entity;
59  		}
60  
61  		/**
62  		 * {@inheritDoc}
63  		 */
64  		@Override
65          public boolean equals(Object other) {
66  			return Identity.class.isInstance( other )
67  					&& this.entity == ( ( Identity ) other ).entity;
68  		}
69  
70  		/**
71  		 * {@inheritDoc}
72  		 */
73  		@Override
74          public int hashCode() {
75  			return System.identityHashCode( entity );
76  		}
77  	}
78  
79  	/**
80  	 * Disallow instantiation of DistinctResultTransformer.
81  	 */
82  	private DistinctResultTransformer() {
83  	}
84  
85  	/**
86  	 * Uniquely distinct each tuple row here.
87  	 */
88  	@Override
89      public List transformList(List list) {
90  		List result = new ArrayList( list.size() );
91  		Set distinct = new HashSet();
92  		for ( int i = 0; i < list.size(); i++ ) {
93  			Object entity = list.get( i );
94  			if ( distinct.add( new Identity( entity ) ) ) {
95  				result.add( entity );
96  			}
97  		}
98  		LOG.debugf( "Transformed: %s rows to: %s distinct results", list.size(), result.size() );
99  		return result;
100 	}
101 
102 	/**
103 	 * Serialization hook for ensuring singleton uniqueing.
104 	 *
105 	 * @return The singleton instance : {@link #INSTANCE}
106 	 */
107 	private Object readResolve() {
108 		return INSTANCE;
109 	}
110 }
