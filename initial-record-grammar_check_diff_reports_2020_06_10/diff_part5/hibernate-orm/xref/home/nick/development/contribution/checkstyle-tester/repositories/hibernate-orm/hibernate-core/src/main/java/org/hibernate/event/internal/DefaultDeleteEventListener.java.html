55  /**
56   * Defines the default delete event listener used by hibernate for deleting entities
57   * from the datastore in response to generated delete events.
58   *
59   * @author Steve Ebersole
60   */
61  public class DefaultDeleteEventListener implements DeleteEventListener {
62  
63      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class,
64                                                                         DefaultDeleteEventListener.class.getName());
65  
66  	/**
67  	 * Handle the given delete event.
68  	 *
69  	 * @param event The delete event to be handled.
70  	 *
71  	 * @throws HibernateException
72  	 */
73  	public void onDelete(DeleteEvent event) throws HibernateException {
74  		onDelete( event, new IdentitySet() );
75  	}
76  
77  	/**
78  	 * Handle the given delete event.  This is the cascaded form.
79  	 *
80  	 * @param event The delete event.
81  	 * @param transientEntities The cache of entities already deleted
82  	 *
83  	 * @throws HibernateException
84  	 */
85  	public void onDelete(DeleteEvent event, Set transientEntities) throws HibernateException {
86  
87  		final EventSource source = event.getSession();
88  
89  		final PersistenceContext persistenceContext = source.getPersistenceContext();
90  		Object entity = persistenceContext.unproxyAndReassociate( event.getObject() );
91  
92  		EntityEntry entityEntry = persistenceContext.getEntry( entity );
93  		final EntityPersister persister;
94  		final Serializable id;
95  		final Object version;
96  
97  		if ( entityEntry == null ) {
98  			LOG.trace( "Entity was not persistent in delete processing" );
99  
100 			persister = source.getEntityPersister( event.getEntityName(), entity );
101 
102 			if ( ForeignKeys.isTransient( persister.getEntityName(), entity, null, source ) ) {
103 				deleteTransientEntity( source, entity, event.isCascadeDeleteEnabled(), persister, transientEntities );
104 				// EARLY EXIT!!!
105 				return;
106 			}
107 			performDetachedEntityDeletionCheck( event );
108 
109 			id = persister.getIdentifier( entity, source );
110 
111 			if ( id == null ) {
112 				throw new TransientObjectException(
113 						"the detached instance passed to delete() had a null identifier"
114 				);
115 			}
116 
117 			final EntityKey key = source.generateEntityKey( id, persister );
118 
119 			persistenceContext.checkUniqueness( key, entity );
120 
121 			new OnUpdateVisitor( source, id, entity ).process( entity, persister );
122 
123 			version = persister.getVersion( entity );
124 
125 			entityEntry = persistenceContext.addEntity(
126 					entity,
127 					( persister.isMutable() ? Status.MANAGED : Status.READ_ONLY ),
128 					persister.getPropertyValues( entity ),
129 					key,
130 					version,
131 					LockMode.NONE,
132 					true,
133 					persister,
134 					false,
135 					false
136 			);
137 		}
138 		else {
139 			LOG.trace( "Deleting a persistent instance" );
140 
141 			if ( entityEntry.getStatus() == Status.DELETED || entityEntry.getStatus() == Status.GONE ) {
142 				LOG.trace( "Object was already deleted" );
143 				return;
144 			}
145 			persister = entityEntry.getPersister();
146 			id = entityEntry.getId();
147 			version = entityEntry.getVersion();
148 		}
149 
150 		/*if ( !persister.isMutable() ) {
151 			throw new HibernateException(
152 					"attempted to delete an object of immutable class: " +
153 					MessageHelper.infoString(persister)
154 				);
155 		}*/
156 
157 		if ( invokeDeleteLifecycle( source, entity, persister ) ) {
158 			return;
159 		}
160 
161 		deleteEntity( source, entity, entityEntry, event.isCascadeDeleteEnabled(),
162 				event.isOrphanRemovalBeforeUpdates(), persister, transientEntities );
163 
164 		if ( source.getFactory().getSettings().isIdentifierRollbackEnabled() ) {
165 			persister.resetIdentifier( entity, id, version, source );
166 		}
167 	}

169 	/**
170 	 * Called when we have recognized an attempt to delete a detached entity.
171 	 * <p/>
172 	 * This is perfectly valid in Hibernate usage; JPA, however, forbids this.
173 	 * Thus, this is a hook for HEM to affect this behavior.
174 	 *
175 	 * @param event The event.
176 	 */
177 	protected void performDetachedEntityDeletionCheck(DeleteEvent event) {
178 		// ok in normal Hibernate usage to delete a detached entity; JPA however
179 		// forbids it, thus this is a hook for HEM to affect this behavior
180 	}

182 	/**
183 	 * We encountered a delete request on a transient instance.
184 	 * <p/>
185 	 * This is a deviation from historical Hibernate (pre-3.2) behavior to
186 	 * align with the JPA spec, which states that transient entities can be
187 	 * passed to remove operation in which case cascades still need to be
188 	 * performed.
189 	 *
190 	 * @param session The session which is the source of the event
191 	 * @param entity The entity being delete processed
192 	 * @param cascadeDeleteEnabled Is cascading of deletes enabled
193 	 * @param persister The entity persister
194 	 * @param transientEntities A cache of already visited transient entities
195 	 * (to avoid infinite recursion).
196 	 */
197 	protected void deleteTransientEntity(
198 			EventSource session,
199 			Object entity,
200 			boolean cascadeDeleteEnabled,
201 			EntityPersister persister,
202 			Set transientEntities) {
203 		LOG.handlingTransientEntity();
204 		if ( transientEntities.contains( entity ) ) {
205 			LOG.trace( "Already handled transient entity; skipping" );
206 			return;
207 		}
208 		transientEntities.add( entity );
209 		cascadeBeforeDelete( session, persister, entity, null, transientEntities );
210 		cascadeAfterDelete( session, persister, entity, transientEntities );
211 	}
212 
213 	/**
214 	 * Perform the entity deletion.  Well, as with most operations, does not
215 	 * really perform it; just schedules an action/execution with the
216 	 * {@link org.hibernate.engine.spi.ActionQueue} for execution during flush.
217 	 *
218 	 * @param session The originating session
219 	 * @param entity The entity to delete
220 	 * @param entityEntry The entity's entry in the {@link PersistenceContext}
221 	 * @param isCascadeDeleteEnabled Is delete cascading enabled?
222 	 * @param persister The entity persister.
223 	 * @param transientEntities A cache of already deleted entities.
224 	 */
225 	protected final void deleteEntity(
226 			final EventSource session,
227 			final Object entity,
228 			final EntityEntry entityEntry,
229 			final boolean isCascadeDeleteEnabled,
230 			final boolean isOrphanRemovalBeforeUpdates,
231 			final EntityPersister persister,
232 			final Set transientEntities) {
233 
234 		if ( LOG.isTraceEnabled() ) {
235 			LOG.tracev( "Deleting {0}", MessageHelper.infoString( persister, entityEntry.getId(), session.getFactory() ) );
236 		}
237 
238 		final PersistenceContext persistenceContext = session.getPersistenceContext();
239 		final Type[] propTypes = persister.getPropertyTypes();
240 		final Object version = entityEntry.getVersion();
241 
242 		final Object[] currentState;
243 		if ( entityEntry.getLoadedState() == null ) { //ie. the entity came in from update()
244 			currentState = persister.getPropertyValues( entity );
245 		}
246 		else {
247 			currentState = entityEntry.getLoadedState();
248 		}
249 
250 		final Object[] deletedState = createDeletedState( persister, currentState, session );
251 		entityEntry.setDeletedState( deletedState );
252 
253 		session.getInterceptor().onDelete(
254 				entity,
255 				entityEntry.getId(),
256 				deletedState,
257 				persister.getPropertyNames(),
258 				propTypes
259 		);
260 
261 		// before any callbacks, etc, so subdeletions see that this deletion happened first
262 		persistenceContext.setEntryStatus( entityEntry, Status.DELETED );
263 		final EntityKey key = session.generateEntityKey( entityEntry.getId(), persister );
264 
265 		cascadeBeforeDelete( session, persister, entity, entityEntry, transientEntities );
266 
267 		new ForeignKeys.Nullifier( entity, true, false, session )
268 				.nullifyTransientReferences( entityEntry.getDeletedState(), propTypes );
269 		new Nullability( session ).checkNullability( entityEntry.getDeletedState(), persister, true );
270 		persistenceContext.getNullifiableEntityKeys().add( key );
271 
272 		if (isOrphanRemovalBeforeUpdates) {
273 			// TODO: The removeOrphan concept is a temporary "hack" for HHH-6484.  This should be removed once action/task
274 			// ordering is improved.
275 			session.getActionQueue().addAction(
276 					new OrphanRemovalAction(
277 							entityEntry.getId(),
278 							deletedState,
279 							version,
280 							entity,
281 							persister,
282 							isCascadeDeleteEnabled,
283 							session
284 					)
285 			);
286 		}
287 		else {
288 			// Ensures that containing deletions happen before sub-deletions
289 			session.getActionQueue().addAction(
290 					new EntityDeleteAction(
291 							entityEntry.getId(),
292 							deletedState,
293 							version,
294 							entity,
295 							persister,
296 							isCascadeDeleteEnabled,
297 							session
298 					)
299 			);
300 		}
301 
302 		cascadeAfterDelete( session, persister, entity, transientEntities );
303 
304 		// the entry will be removed after the flush, and will no longer
305 		// override the stale snapshot
306 		// This is now handled by removeEntity() in EntityDeleteAction
307 		//persistenceContext.removeDatabaseSnapshot(key);
308 	}
309 
310 	private Object[] createDeletedState(EntityPersister persister, Object[] currentState, EventSource session) {
311 		Type[] propTypes = persister.getPropertyTypes();
312 		final Object[] deletedState = new Object[propTypes.length];
313 //		TypeFactory.deepCopy( currentState, propTypes, persister.getPropertyUpdateability(), deletedState, session );
314 		boolean[] copyability = new boolean[propTypes.length];
315 		java.util.Arrays.fill( copyability, true );
316 		TypeHelper.deepCopy( currentState, propTypes, copyability, deletedState, session );
317 		return deletedState;
318 	}
319 
320 	protected boolean invokeDeleteLifecycle(EventSource session, Object entity, EntityPersister persister) {
321 		if ( persister.implementsLifecycle() ) {
322 			LOG.debug( "Calling onDelete()" );
323 			if ( ( ( Lifecycle ) entity ).onDelete( session ) ) {
324 				LOG.debug( "Deletion vetoed by onDelete()" );
325 				return true;
326 			}
327 		}
328 		return false;
329 	}

331 	protected void cascadeBeforeDelete(
332 			EventSource session,
333 			EntityPersister persister,
334 			Object entity,
335 			EntityEntry entityEntry,
336 			Set transientEntities) throws HibernateException {
337 
338 		CacheMode cacheMode = session.getCacheMode();
339 		session.setCacheMode( CacheMode.GET );
340 		session.getPersistenceContext().incrementCascadeLevel();
341 		try {
342 			// cascade-delete to collections BEFORE the collection owner is deleted
343 			new Cascade( CascadingAction.DELETE, Cascade.AFTER_INSERT_BEFORE_DELETE, session )
344 					.cascade( persister, entity, transientEntities );
345 		}
346 		finally {
347 			session.getPersistenceContext().decrementCascadeLevel();
348 			session.setCacheMode( cacheMode );
349 		}
350 	}
351 
352 	protected void cascadeAfterDelete(
353 			EventSource session,
354 			EntityPersister persister,
355 			Object entity,
356 			Set transientEntities) throws HibernateException {
357 
358 		CacheMode cacheMode = session.getCacheMode();
359 		session.setCacheMode( CacheMode.GET );
360 		session.getPersistenceContext().incrementCascadeLevel();
361 		try {
362 			// cascade-delete to many-to-one AFTER the parent was deleted
363 			new Cascade( CascadingAction.DELETE, Cascade.BEFORE_INSERT_AFTER_DELETE, session )
364 					.cascade( persister, entity, transientEntities );
365 		}
366 		finally {
367 			session.getPersistenceContext().decrementCascadeLevel();
368 			session.setCacheMode( cacheMode );
369 		}
370 	}
