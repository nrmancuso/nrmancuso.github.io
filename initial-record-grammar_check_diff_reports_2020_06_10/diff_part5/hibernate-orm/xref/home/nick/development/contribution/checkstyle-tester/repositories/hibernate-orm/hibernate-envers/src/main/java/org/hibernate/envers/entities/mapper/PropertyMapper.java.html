24  package org.hibernate.envers.entities.mapper;
25  import java.io.Serializable;
26  import java.util.List;
27  import java.util.Map;
28  
29  import org.hibernate.collection.spi.PersistentCollection;
30  import org.hibernate.engine.spi.SessionImplementor;
31  import org.hibernate.envers.configuration.AuditConfiguration;
32  import org.hibernate.envers.reader.AuditReaderImplementor;
33  
34  /**
35   * @author Adam Warski (adam at warski dot org)
36   * @author Michal Skowronek (mskowr at o2 dot pl)
37   */
38  public interface PropertyMapper {
39      /**
40       * Maps properties to the given map, basing on differences between properties of new and old objects.
41       * @param session The current session.
42  	 * @param data Data to map to.
43  	 * @param newObj New state of the entity.
44  	 * @param oldObj Old state of the entity.
45  	 * @return True if there are any differences between the states represented by newObj and oldObj.
46       */
47      boolean mapToMapFromEntity(SessionImplementor session, Map<String, Object> data, Object newObj, Object oldObj);
48  
49      /**
50       * Maps properties from the given map to the given object.
51       * @param verCfg Versions configuration.
52       * @param obj Object to map to.
53       * @param data Data to map from.
54       * @param primaryKey Primary key of the object to which we map (for relations)
55       * @param versionsReader VersionsReader for reading relations
56       * @param revision Revision at which the object is read, for reading relations
57       */
58      void mapToEntityFromMap(AuditConfiguration verCfg, Object obj, Map data, Object primaryKey,
59                              AuditReaderImplementor versionsReader, Number revision);
60  
61      /**
62       * Maps collection changes.
63  	 * @param session The current session.
64       * @param referencingPropertyName Name of the field, which holds the collection in the entity.
65       * @param newColl New collection, after updates.
66       * @param oldColl Old collection, before updates.
67       * @param id Id of the object owning the collection.
68       * @return List of changes that need to be performed on the persistent store.
69       */
70      List<PersistentCollectionChangeData> mapCollectionChanges(SessionImplementor session, String referencingPropertyName,
71                                                                PersistentCollection newColl,
72                                                                Serializable oldColl, Serializable id);
73  
74  	void mapModifiedFlagsToMapFromEntity(SessionImplementor session, Map<String, Object> data, Object newObj, Object oldObj);
75  	void mapModifiedFlagsToMapForCollectionChange(String collectionPropertyName, Map<String, Object> data);
76  }
