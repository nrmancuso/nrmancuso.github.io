12  import java.util.Arrays;

15   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)

17  @TestForIssue(jiraKey = "HHH-7106")
18  public class SchemaExportTest extends BaseEnversFunctionalTestCase  {
19      private Integer id = null;
20  
21      @Override
22      protected Class<?>[] getAnnotatedClasses() {
23          return new Class[] {StrTestEntity.class};
24      }
25  
26  	protected boolean createSchema() {
27  		// Disable schema auto generation.
28  		return false;
29  	}
30      @Test
31      @Priority(10)
32      public void testSchemaCreation() {
33          // Generate complete schema.
34          new EnversSchemaGenerator(configuration()).export().create( true, true );
35  
36          // Populate database with test data.
37          Session session = getSession();
38          session.getTransaction().begin();
39          StrTestEntity entity = new StrTestEntity("data");
40          session.save(entity);
41          session.getTransaction().commit();
42  
43          id = entity.getId();
44      }
45  
46  	@Test
47      @Priority(9)
48      public void testAuditDataRetrieval() {
49          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(StrTestEntity.class, id));
50          Assert.assertEquals(new StrTestEntity("data", id), getAuditReader().find(StrTestEntity.class, id, 1));
51      }
52  
53      @Test
54      @Priority(8)
55      public void testSchemaDrop() {
56          new EnversSchemaGenerator(configuration()).export().drop( true, true );
57      }
58  }
