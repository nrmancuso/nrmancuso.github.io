59  	@Test
60  	public void testTernary() {
61  		Session s = openSession();
62  		Transaction t = s.beginTransaction();
63  		Employee bob = new Employee("Bob");
64  		Employee tom = new Employee("Tom");
65  		Employee jim = new Employee("Jim");
66  		Employee tim = new Employee("Tim");
67  		Site melb = new Site("Melbourne");
68  		Site geel = new Site("Geelong");
69  		s.persist(bob);
70  		s.persist(tom);
71  		s.persist(jim);
72  		s.persist(tim);
73  		s.persist(melb);
74  		s.persist(geel);
75  		bob.getManagerBySite().put(melb, tom);
76  		bob.getManagerBySite().put(geel, jim);
77  		tim.getManagerBySite().put(melb, tom);
78  		t.commit();
79  		s.close();
80  		
81  		s = openSession();
82  		t = s.beginTransaction();
83  		tom = (Employee) s.get(Employee.class, "Tom");
84  		assertFalse( Hibernate.isInitialized(tom.getUnderlings()) );
85  		assertEquals( tom.getUnderlings().size(), 2 );
86  		bob = (Employee) s.get(Employee.class, "Bob");
87  		assertFalse( Hibernate.isInitialized(bob.getManagerBySite()) );
88  		assertTrue( tom.getUnderlings().contains(bob) );
89  		melb = (Site) s.get(Site.class, "Melbourne");
90  		assertSame( bob.getManagerBySite().get(melb), tom );
91  		assertTrue( melb.getEmployees().contains(bob) );
92  		assertTrue( melb.getManagers().contains(tom) );
93  		t.commit();
94  		s.close();		
95  
96  		s = openSession();
97  		t = s.beginTransaction();
98  		List l = s.createQuery("from Employee e join e.managerBySite m where m.name='Bob'").list();
99  		assertEquals( l.size(), 0 );
100 		l = s.createQuery("from Employee e join e.managerBySite m where m.name='Tom'").list();
101 		assertEquals( l.size(), 2 );
102 		t.commit();
103 		s.close();
104 
105 		s = openSession();
106 		t = s.beginTransaction();
107 		l = s.createQuery("from Employee e left join fetch e.managerBySite").list();
108 		assertEquals( l.size(), 5 );
109 		Set set = new HashSet(l);
110 		assertEquals( set.size(), 4 );
111 		Iterator iter = set.iterator();
112 		int total=0;
113 		while ( iter.hasNext() ) {
114 			Map map = ( (Employee) iter.next() ).getManagerBySite();
115 			assertTrue( Hibernate.isInitialized(map) );
116 			total += map.size();
117 		}
118 		assertTrue(total==3);
119 
120 		l = s.createQuery("from Employee e left join e.managerBySite m left join m.managerBySite m2").list();
121 
122 		// clean up...
123 		l = s.createQuery("from Employee e left join fetch e.managerBySite").list();
124 		Iterator itr = l.iterator();
125 		while ( itr.hasNext() ) {
126 			Employee emp = ( Employee ) itr.next();
127 			emp.setManagerBySite( new HashMap() );
128 			s.delete( emp );
129 		}
130 		for ( Object entity : s.createQuery( "from Site" ).list() ) {
131 			s.delete( entity );
132 		}
133 		t.commit();
134 		s.close();
135 	}

137 	@Test
138 	public void testIndexRelatedFunctions() {
139 		Session session = openSession();
140 		session.beginTransaction();
141 		session.createQuery( "from Employee e join e.managerBySite as m where index(m) is not null" )
142 				.list();
143 		session.createQuery( "from Employee e join e.managerBySite as m where minIndex(m) is not null" )
144 				.list();
145 		session.createQuery( "from Employee e join e.managerBySite as m where maxIndex(m) is not null" )
146 				.list();
147 		session.getTransaction().commit();
148 		session.close();
149 	}
