49   * SQL where clause.

157 	// Handles things like:
158 	// a and b or c
159 	// a and ( b or c )
160 	// not a and not b
161 	// not ( a and b )
162 	// x between y and z            (overloaded "and")
163 	// x in ( a, b, c )             (overloaded brackets)
164 	// not not a
165 	// a is not null                (overloaded "not")
166 	// etc......
167 	// and expressions like
168 	// foo = bar                    (maps to: foo.id = bar.id)
169 	// foo.Bar = 'foo'              (maps to: foo.bar = 'foo')
170 	// foo.Bar.Baz = 1.0            (maps to: foo.bar = bar.id and bar.baz = 1.0)
171 	// 1.0 = foo.Bar.Baz            (maps to: bar.baz = 1.0 and foo.Bar = bar.id)
172 	// foo.Bar.Baz = a.B.C          (maps to: bar.Baz = b.C and foo.Bar = bar.id and a.B = b.id)
173 	// foo.Bar.Baz + a.B.C          (maps to: bar.Baz + b.C and foo.Bar = bar.id and a.B = b.id)
174 	// ( foo.Bar.Baz + 1.0 ) < 2.0  (maps to: ( bar.Baz + 1.0 ) < 2.0 and foo.Bar = bar.id)

184 	private int expectingIndex = 0;

186 	// The following variables are stacks that keep information about each subexpression
187 	// in the list of nested subexpressions we are currently processing.
188 
189 	private LinkedList<Boolean> nots = new LinkedList<Boolean>();           //were an odd or even number of NOTs encountered
190 	private LinkedList joins = new LinkedList();          //the join string built up by compound paths inside this expression
191 	private LinkedList<Boolean> booleanTests = new LinkedList<Boolean>();   //a flag indicating if the subexpression is known to be boolean

193 	private String getElementName(PathExpressionParser.CollectionElement element, QueryTranslatorImpl q) throws QueryException {
194 		String name;
195 		if ( element.isOneToMany ) {
196 			name = element.alias;
197 		}
198 		else {
199 			Type type = element.elementType;
200 			if ( type.isEntityType() ) { //ie. a many-to-many
201 				String entityName = ( ( EntityType ) type ).getAssociatedEntityName();
202 				name = pathExpressionParser.continueFromManyToMany( entityName, element.elementColumns, q );
203 			}
204 			else {
205 				throw new QueryException( "illegally dereferenced collection element" );
206 			}
207 		}
208 		return name;
209 	}

211 	public void token(String token, QueryTranslatorImpl q) throws QueryException {
212 
213 		String lcToken = token.toLowerCase();
214 
215 		//Cope with [,]
216 		if ( token.equals( "[" ) && !expectingPathContinuation ) {
217 			expectingPathContinuation = false;
218 			if ( expectingIndex == 0 ) throw new QueryException( "unexpected [" );
219 			return;
220 		}
221 		else if ( token.equals( "]" ) ) {
222 			expectingIndex--;
223 			expectingPathContinuation = true;
224 			return;
225 		}
226 
227 		//Cope with a continued path expression (ie. ].baz)
228 		if ( expectingPathContinuation ) {
229 			boolean pathExpressionContinuesFurther = continuePathExpression( token, q );
230 			if ( pathExpressionContinuesFurther ) return; //NOTE: early return
231 		}
232 
233 		//Cope with a subselect
234 		if ( !inSubselect && ( lcToken.equals( "select" ) || lcToken.equals( "from" ) ) ) {
235 			inSubselect = true;
236 			subselect = new StringBuilder( 20 );
237 		}
238 		if ( inSubselect && token.equals( ")" ) ) {
239 			bracketsSinceSelect--;
240 
241 			if ( bracketsSinceSelect == -1 ) {
242 				QueryTranslatorImpl subq = new QueryTranslatorImpl(
243 				        subselect.toString(),
244 						q.getEnabledFilters(),
245 						q.getFactory()
246 				);
247 				try {
248 					subq.compile( q );
249 				}
250 				catch ( MappingException me ) {
251 					throw new QueryException( "MappingException occurred compiling subquery", me );
252 				}
253 				appendToken( q, subq.getSQLString() );
254 				inSubselect = false;
255 				bracketsSinceSelect = 0;
256 			}
257 		}
258 		if ( inSubselect ) {
259 			if ( token.equals( "(" ) ) bracketsSinceSelect++;
260 			subselect.append( token ).append( ' ' );
261 			return;
262 		}
263 
264 		//Cope with special cases of AND, NOT, ()
265 		specialCasesBefore( lcToken );
266 
267 		//Close extra brackets we opened
268 		if ( !betweenSpecialCase && EXPRESSION_TERMINATORS.contains( lcToken ) ) {
269 			closeExpression( q, lcToken );
270 		}
271 
272 		//take note when this is a boolean expression
273 		if ( BOOLEAN_OPERATORS.contains( lcToken ) ) {
274 			booleanTests.removeLast();
275 			booleanTests.addLast( Boolean.TRUE );
276 		}
277 
278 		if ( lcToken.equals( "not" ) ) {
279 			nots.addLast(  !(  nots.removeLast() ) );
280 			negated = !negated;
281 			return; //NOTE: early return
282 		}
283 
284 		//process a token, mapping OO path expressions to SQL expressions
285 		doToken( token, q );
286 
287 		//Open any extra brackets we might need.
288 		if ( !betweenSpecialCase && EXPRESSION_OPENERS.contains( lcToken ) ) {
289 			openExpression( q, lcToken );
290 		}
291 
292 		//Cope with special cases of AND, NOT, )
293 		specialCasesAfter( lcToken );
294 
295 	}

297 	public void start(QueryTranslatorImpl q) throws QueryException {
298 		token( "(", q );
299 	}
300 
301 	public void end(QueryTranslatorImpl q) throws QueryException {
302 		if ( expectingPathContinuation ) {
303 			expectingPathContinuation = false;
304 			PathExpressionParser.CollectionElement element = pathExpressionParser.lastCollectionElement();
305 			if ( element.elementColumns.length != 1 ) throw new QueryException( "path expression ended in composite collection element" );
306 			appendToken( q, element.elementColumns[0] );
307 			addToCurrentJoin( element );
308 		}
309 		token( ")", q );
310 	}

312 	private void closeExpression(QueryTranslatorImpl q, String lcToken) {
313 		if ( booleanTests.removeLast() ) { //it was a boolean expression
314 
315 			if ( booleanTests.size() > 0 ) {
316 				// the next one up must also be
317 				booleanTests.removeLast();
318 				booleanTests.addLast( Boolean.TRUE );
319 			}
320 
321 			// Add any joins
322 			appendToken( q, ( joins.removeLast() ).toString() );
323 
324 		}
325 		else {
326 			StringBuilder join = ( StringBuilder ) joins.removeLast();
327 			( ( StringBuilder ) joins.getLast() ).append( join.toString() );
328 		}
329 
330 		if ( nots.removeLast() ) negated = !negated;
331 
332 		if ( !")".equals( lcToken ) ) appendToken( q, ")" );
333 	}
334 
335 	private void openExpression(QueryTranslatorImpl q, String lcToken) {
336 		nots.addLast( Boolean.FALSE );
337 		booleanTests.addLast( Boolean.FALSE );
338 		joins.addLast( new StringBuilder() );
339 		if ( !"(".equals( lcToken ) ) appendToken( q, "(" );
340 	}

342 	private void preprocess(String token, QueryTranslatorImpl q) throws QueryException {
343 		// ugly hack for cases like "elements(foo.bar.collection)"
344 		// (multi-part path expression ending in elements or indices)
345 		String[] tokens = StringHelper.split( ".", token, true );
346 		if (
347 				tokens.length > 5 &&
348 				( CollectionPropertyNames.COLLECTION_ELEMENTS.equals( tokens[tokens.length - 1] )
349 				|| CollectionPropertyNames.COLLECTION_INDICES.equals( tokens[tokens.length - 1] ) )
350 		) {
351 			pathExpressionParser.start( q );
352 			for ( int i = 0; i < tokens.length - 3; i++ ) {
353 				pathExpressionParser.token( tokens[i], q );
354 			}
355 			pathExpressionParser.token( null, q );
356 			pathExpressionParser.end( q );
357 			addJoin( pathExpressionParser.getWhereJoin(), q );
358 			pathExpressionParser.ignoreInitialJoin();
359 		}
360 	}

362 	private void doPathExpression(String token, QueryTranslatorImpl q) throws QueryException {
363 
364 		preprocess( token, q );
365 
366 		StringTokenizer tokens = new StringTokenizer( token, ".", true );
367 		pathExpressionParser.start( q );
368 		while ( tokens.hasMoreTokens() ) {
369 			pathExpressionParser.token( tokens.nextToken(), q );
370 		}
371 		pathExpressionParser.end( q );
372 		if ( pathExpressionParser.isCollectionValued() ) {
373 			openExpression( q, "" );
374 			appendToken( q, pathExpressionParser.getCollectionSubquery( q.getEnabledFilters() ) );
375 			closeExpression( q, "" );
376 			// this is ugly here, but needed because its a subquery
377 			q.addQuerySpaces( q.getCollectionPersister( pathExpressionParser.getCollectionRole() ).getCollectionSpaces() );
378 		}
379 		else {
380 			if ( pathExpressionParser.isExpectingCollectionIndex() ) {
381 				expectingIndex++;
382 			}
383 			else {
384 				addJoin( pathExpressionParser.getWhereJoin(), q );
385 				appendToken( q, pathExpressionParser.getWhereColumn() );
386 			}
387 		}
388 	}

390 	private void addJoin(JoinSequence joinSequence, QueryTranslatorImpl q) throws QueryException {
391 		//JoinFragment fromClause = q.createJoinFragment(true);
392 		//fromClause.addJoins( join.toJoinFragment().toFromFragmentString(), StringHelper.EMPTY_STRING );
393 		q.addFromJoinOnly( pathExpressionParser.getName(), joinSequence );
394 		try {
395 			addToCurrentJoin( joinSequence.toJoinFragment( q.getEnabledFilters(), true ).toWhereFragmentString() );
396 		}
397 		catch ( MappingException me ) {
398 			throw new QueryException( me );
399 		}
400 	}

402 	private void doToken(String token, QueryTranslatorImpl q) throws QueryException {
403 		if ( q.isName( StringHelper.root( token ) ) ) { //path expression
404 			doPathExpression( q.unalias( token ), q );
405 		}
406 		else if ( token.startsWith( ParserHelper.HQL_VARIABLE_PREFIX ) ) { //named query parameter
407 			q.addNamedParameter( token.substring( 1 ) );
408 			appendToken( q, "?" );
409 		}
410 		else {
411 			Queryable persister = q.getEntityPersisterUsingImports( token );
412 			if ( persister != null ) { // the name of a class
413 				final String discrim = persister.getDiscriminatorSQLValue();
414 				if ( InFragment.NULL.equals(discrim) || InFragment.NOT_NULL.equals(discrim) ) {
415 					throw new QueryException( "subclass test not allowed for null or not null discriminator" );
416 				}
417 				else {
418 					appendToken( q, discrim );
419 				}
420 			}
421 			else {
422 				Object constant;
423 				if (
424 						token.indexOf( '.' ) > -1 &&
425 						( constant = ReflectHelper.getConstantValue( token ) ) != null
426 				) {
427 					Type type;
428 					try {
429 						type = q.getFactory().getTypeResolver().heuristicType( constant.getClass().getName() );
430 					}
431 					catch ( MappingException me ) {
432 						throw new QueryException( me );
433 					}
434 					if ( type == null ) throw new QueryException( QueryTranslator.ERROR_CANNOT_DETERMINE_TYPE + token );
435 					try {
436 						appendToken( q, ( ( LiteralType ) type ).objectToSQLString( constant, q.getFactory().getDialect() ) );
437 					}
438 					catch ( Exception e ) {
439 						throw new QueryException( QueryTranslator.ERROR_CANNOT_FORMAT_LITERAL + token, e );
440 					}
441 				}
442 				else { //anything else
443 
444 					String negatedToken = negated ? ( String ) NEGATIONS.get( token.toLowerCase() ) : null;
445 					if ( negatedToken != null && ( !betweenSpecialCase || !"or".equals( negatedToken ) ) ) {
446 						appendToken( q, negatedToken );
447 					}
448 					else {
449 						appendToken( q, token );
450 					}
451 				}
452 			}
453 		}
454 	}

456 	private void addToCurrentJoin(String sql) {
457 		( ( StringBuilder ) joins.getLast() ).append( sql );
458 	}
459 
460 	private void addToCurrentJoin(PathExpressionParser.CollectionElement ce)
461 			throws QueryException {
462 		try {
463 			addToCurrentJoin( ce.joinSequence.toJoinFragment().toWhereFragmentString() + ce.indexValue.toString() );
464 		}
465 		catch ( MappingException me ) {
466 			throw new QueryException( me );
467 		}
468 	}

470 	private void specialCasesBefore(String lcToken) {
471 		if ( lcToken.equals( "between" ) || lcToken.equals( "not between" ) ) {
472 			betweenSpecialCase = true;
473 		}
474 	}
475 
476 	private void specialCasesAfter(String lcToken) {
477 		if ( betweenSpecialCase && lcToken.equals( "and" ) ) {
478 			betweenSpecialCase = false;
479 		}
480 	}
481 
482 	void appendToken(QueryTranslatorImpl q, String token) {
483 		if ( expectingIndex > 0 ) {
484 			pathExpressionParser.setLastCollectionElementIndexValue( token );
485 		}
486 		else {
487 			q.appendWhereToken( token );
488 		}
489 	}
490 
491 	private boolean continuePathExpression(String token, QueryTranslatorImpl q) throws QueryException {
492 
493 		expectingPathContinuation = false;
494 
495 		PathExpressionParser.CollectionElement element = pathExpressionParser.lastCollectionElement();
496 
497 		if ( token.startsWith( "." ) ) { // the path expression continues after a ]
498 
499 			doPathExpression( getElementName( element, q ) + token, q ); // careful with this!
500 
501 			addToCurrentJoin( element );
502 			return true; //NOTE: EARLY EXIT!
503 
504 		}
505 
506 		else { // the path expression ends at the ]
507 			if ( element.elementColumns.length != 1 ) {
508 				throw new QueryException( "path expression ended in composite collection element" );
509 			}
510 			appendToken( q, element.elementColumns[0] );
511 			addToCurrentJoin( element );
512 			return false;
513 		}
514 	}
