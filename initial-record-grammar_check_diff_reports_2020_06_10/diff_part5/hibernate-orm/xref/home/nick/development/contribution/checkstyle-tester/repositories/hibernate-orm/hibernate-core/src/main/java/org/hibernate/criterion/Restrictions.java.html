35  /**
36   * The <tt>criterion</tt> package may be used by applications as a framework for building
37   * new kinds of <tt>Criterion</tt>. However, it is intended that most applications will
38   * simply use the built-in criterion types via the static factory methods of this class.
39   *
40   * @see org.hibernate.Criteria
41   * @see Projections factory methods for <tt>Projection</tt> instances
42   * @author Gavin King
43   */

50  	/**
51  	 * Apply an "equal" constraint to the identifier property
52  	 * @param value
53  	 * @return Criterion
54  	 */
55  	public static Criterion idEq(Object value) {
56  		return new IdentifierEqExpression(value);
57  	}
58  	/**
59  	 * Apply an "equal" constraint to the named property
60  	 * @param propertyName
61  	 * @param value
62  	 * @return SimpleExpression
63  	 */
64  	public static SimpleExpression eq(String propertyName, Object value) {
65  		return new SimpleExpression(propertyName, value, "=");
66  	}
67  	/**
68  	 * Apply an "equal" constraint to the named property.  If the value
69  	 * is null, instead apply "is null".
70  	 * @param propertyName
71  	 * @param value
72  	 * @return Criterion
73  	 */
74  	public static Criterion eqOrIsNull(String propertyName, Object value) {
75  		if (value == null) {
76  			return isNull(propertyName);
77  		}
78  		return new SimpleExpression(propertyName, value, "=");
79  	}
80  	/**
81  	 * Apply a "not equal" constraint to the named property
82  	 * @param propertyName
83  	 * @param value
84  	 * @return SimpleExpression
85  	 */
86  	public static SimpleExpression ne(String propertyName, Object value) {
87  		return new SimpleExpression(propertyName, value, "<>");
88  	}
89  	/**
90  	 * Apply a "not equal" constraint to the named property.  If the value
91  	 * is null, instead apply "is not null".
92  	 * @param propertyName
93  	 * @param value
94  	 * @return Criterion
95  	 */
96  	public static Criterion neOrIsNotNull(String propertyName, Object value) {
97  		if (value == null) {
98  			return isNotNull(propertyName);
99  		}
100 		return new SimpleExpression(propertyName, value, "<>");
101 	}

132 	/**
133 	 * A case-insensitive "like", similar to Postgres <tt>ilike</tt>
134 	 * operator
135 	 *
136 	 * @param propertyName
137 	 * @param value
138 	 * @return Criterion
139 	 */
140 	public static Criterion ilike(String propertyName, Object value) {
141 		if ( value == null ) {
142 			throw new IllegalArgumentException( "Comparison value passed to ilike cannot be null" );
143 		}
144 		return ilike( propertyName, value.toString(), MatchMode.EXACT );

271 	/**
272 	 * Return the conjuction of multiple expressions
273 	 *
274 	 * @param predicates The predicates making up the initial junction
275 	 *
276 	 * @return The conjunction
277 	 */
278 	public static Conjunction and(Criterion... predicates) {
279 		Conjunction conjunction = conjunction();
280 		if ( predicates != null ) {
281 			for ( Criterion predicate : predicates ) {
282 				conjunction.add( predicate );
283 			}
284 		}
285 		return conjunction;
286 	}

377 	/**
378 	 * Apply an "equals" constraint to each property in the
379 	 * key set of a <tt>Map</tt>
380 	 *
381 	 * @param propertyNameValues a map from property names to values
382 	 * @return Criterion
383 	 */
384 	public static Criterion allEq(Map propertyNameValues) {
385 		Conjunction conj = conjunction();
386 		Iterator iter = propertyNameValues.entrySet().iterator();
387 		while ( iter.hasNext() ) {
388 			Map.Entry me = (Map.Entry) iter.next();
389 			conj.add( eq( (String) me.getKey(), me.getValue() ) );
390 		}
391 		return conj;
392 	}
