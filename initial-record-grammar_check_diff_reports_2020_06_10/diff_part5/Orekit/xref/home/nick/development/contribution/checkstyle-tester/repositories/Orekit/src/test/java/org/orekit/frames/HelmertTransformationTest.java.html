34      @Test
35      public void testHelmert20052008() throws OrekitException {
36          Frame itrf2008 = FramesFactory.getITRF(IERSConventions.IERS_2010, true);
37          Frame itrf2005 =
38                  HelmertTransformation.Predefined.ITRF_2008_TO_ITRF_2005.createTransformedITRF(itrf2008, "2005");
39          Vector3D pos2005 = new Vector3D(1234567.8, 2345678.9, 3456789.0);
40  
41          // check the Helmert transformation as per http://itrf.ign.fr/ITRF_solutions/2008/tp_08-05.php
42          AbsoluteDate date = new AbsoluteDate(2005, 1, 1, 12, 0, 0, TimeScalesFactory.getTT());
43          Vector3D pos2008 = itrf2005.getTransformTo(itrf2008, date).transformPosition(pos2005);
44          Vector3D generalOffset = pos2005.subtract(pos2008);
45          Vector3D linearOffset  = computeOffsetLinearly(-0.5, -0.9, -4.7, 0.000, 0.000, 0.000,
46                                                          0.3,  0.0,  0.0, 0.000, 0.000, 0.000,
47                                                         pos2005, 0.0);
48          Vector3D error         = generalOffset.subtract(linearOffset);
49          Assert.assertEquals(0.0, error.getNorm(), 2.0e-13 * pos2005.getNorm());
50  
51          date = date.shiftedBy(Constants.JULIAN_YEAR);
52          pos2008 = itrf2005.getTransformTo(itrf2008, date).transformPosition(pos2005);
53          generalOffset = pos2005.subtract(pos2008);
54          linearOffset  = computeOffsetLinearly(-0.5, -0.9, -4.7, 0.000, 0.000, 0.000,
55                                                 0.3,  0.0,  0.0, 0.000, 0.000, 0.000,
56                                                pos2005, 1.0);
57          error         = generalOffset.subtract(linearOffset);
58          Assert.assertEquals(0.0, error.getNorm(), 2.0e-13 * pos2005.getNorm());
59  
60      }

152     private Vector3D computeOffsetLinearly(final double t1,    final double t2,    final double t3,
153                                            final double r1,    final double r2,    final double r3,
154                                            final double t1Dot, final double t2Dot, final double t3Dot,
155                                            final double r1Dot, final double r2Dot, final double r3Dot,
156                                            final Vector3D p,   final double dt) {
157         double t1U = (t1 + dt * t1Dot) * 1.0e-3;
158         double t2U = (t2 + dt * t2Dot) * 1.0e-3;
159         double t3U = (t3 + dt * t3Dot) * 1.0e-3;
160         double r1U = FastMath.toRadians((r1 + dt * r1Dot) / 3.6e6);
161         double r2U = FastMath.toRadians((r2 + dt * r2Dot) / 3.6e6);
162         double r3U = FastMath.toRadians((r3 + dt * r3Dot) / 3.6e6);
163         return new Vector3D(t1U - r3U * p.getY() + r2U * p.getZ(),
164                             t2U + r3U * p.getX() - r1U * p.getZ(),
165                             t3U - r2U * p.getX() + r1U * p.getY());
166     }
