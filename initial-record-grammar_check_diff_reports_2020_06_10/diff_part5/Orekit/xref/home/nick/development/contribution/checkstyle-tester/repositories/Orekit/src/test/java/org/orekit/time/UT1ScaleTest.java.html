43      @Test
44      public void testLeap2006() throws OrekitException {
45          TimeScale utc = TimeScalesFactory.getUTC();
46          AbsoluteDate dateA = new AbsoluteDate(2005, 12, 30, 23, 59,  0.0, utc);
47          AbsoluteDate dateB = new AbsoluteDate(2006,  1,  2,  0,  1,  0.0, utc);
48          double deltaAUT1 = ut1.offsetFromTAI(dateA);
49          double deltaAUTC = utc.offsetFromTAI(dateA);
50          double deltaBUT1 = ut1.offsetFromTAI(dateB);
51          double deltaBUTC = utc.offsetFromTAI(dateB);
52  
53          // there is a leap second between the two dates
54          Assert.assertEquals(deltaAUTC - 1.0, deltaBUTC, 1.0e-15);
55  
56          // the leap second induces UTC goes from above UT1 to below UT1
57          Assert.assertTrue(deltaAUTC > deltaAUT1);
58          Assert.assertTrue(deltaBUTC < deltaBUT1);
59  
60          // UT1 is continuous, so change should be very small in two days
61          Assert.assertEquals(deltaAUT1, deltaBUT1, 3.0e-4);
62  
63      }

65      @Test
66      public void testSymmetry() throws OrekitException {
67          for (double dt = -10000; dt < 10000; dt += 123.456789) {
68              AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt * Constants.JULIAN_DAY);
69              double dt1 = ut1.offsetFromTAI(date);
70              DateTimeComponents components = date.getComponents(ut1);
71              double dt2 = ut1.offsetToTAI(components.getDate(), components.getTime());
72              Assert.assertEquals( 0.0, dt1 + dt2, 1.0e-10);
73          }
74      }

76      @Test
77      public void testConcurrent() throws InterruptedException, ExecutionException {
78          // set up
79          final AbsoluteDate date = AbsoluteDate.J2000_EPOCH;
80          final int threads = 10;
81          final int timesPerThread = 100;
82          final double dt = 123.456789 * Constants.JULIAN_DAY;
83  
84          // calculate expected values in single-thread mode
85          final double[] expected = new double[timesPerThread];
86          for (int i = 0; i < timesPerThread; i++) {
87              expected[i] = ut1.offsetFromTAI(date.shiftedBy(i * dt));
88          }
89  
90          // build jobs for concurrent execution
91          final List<Callable<Boolean>> jobs = new ArrayList<Callable<Boolean>>();
92          for (int i = 0; i < threads; i++) {
93              jobs.add(new Callable<Boolean>() {
94                  public Boolean call() throws Exception {
95                      for (int j = 0; j < timesPerThread; j++) {
96                          final double actual = ut1.offsetFromTAI(date.shiftedBy(j * dt));
97                          Assert.assertEquals(expected[j], actual, 0);
98                      }
99                      return true;
100                 }
101             });
102         }
103 
104         // action
105         final List<Future<Boolean>> futures = Executors.newFixedThreadPool(threads).invokeAll(jobs);
106 
107         // verify - necessary to throw AssertionErrors from the Callable
108         for (Future<Boolean> future : futures) {
109             Assert.assertEquals(true, future.get());
110         }
111     }

143     @Test
144     public void testSerialization() throws OrekitException, IOException, ClassNotFoundException {
145         UT1Scale ut1 = TimeScalesFactory.getUT1(IERSConventions.IERS_2010, true);
146         
147         ByteArrayOutputStream bos = new ByteArrayOutputStream();
148         ObjectOutputStream    oos = new ObjectOutputStream(bos);
149         oos.writeObject(ut1);
150 
151         Assert.assertTrue(bos.size() > 145000);
152         Assert.assertTrue(bos.size() < 155000);
153 
154         ByteArrayInputStream  bis = new ByteArrayInputStream(bos.toByteArray());
155         ObjectInputStream     ois = new ObjectInputStream(bis);
156         UT1Scale deserialized  = (UT1Scale) ois.readObject();
157         for (double dt = 0; dt < 7 * Constants.JULIAN_DAY; dt += 3600) {
158             AbsoluteDate date = ut1.getEOPHistory().getStartDate().shiftedBy(dt);
159             Assert.assertEquals(ut1.offsetFromTAI(date), deserialized.offsetFromTAI(date), 1.0e-15);
160         }
161 
162     }

164     @Test
165     public void testDuringLeap() throws OrekitException {
166         final TimeScale utc   = TimeScalesFactory.getUTC();
167         final TimeScale scale = TimeScalesFactory.getUT1(IERSConventions.IERS_2010, true);
168         final AbsoluteDate before = new AbsoluteDate(new DateComponents(1983, 06, 30),
169                                                      new TimeComponents(23, 59, 59),
170                                                      utc);
171         final AbsoluteDate during = before.shiftedBy(1.25);
172         Assert.assertEquals(61, utc.minuteDuration(during));
173         Assert.assertEquals(1.0, utc.getLeap(during), 1.0e-10);
174         Assert.assertEquals(60, scale.minuteDuration(during));
175         Assert.assertEquals(0.0, scale.getLeap(during), 1.0e-10);
176     }
