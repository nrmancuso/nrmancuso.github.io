47      @Test
48      public void testEuler1976() throws OrekitException {
49  
50          TransformProvider eulerBasedProvider = new TransformProvider() {
51              private static final long serialVersionUID = 1L;
52              private final PolynomialNutation<DerivativeStructure> zetaA =
53                      new PolynomialNutation<DerivativeStructure>(0.0,
54                              2306.2181 * Constants.ARC_SECONDS_TO_RADIANS,
55                              0.30188   * Constants.ARC_SECONDS_TO_RADIANS,
56                              0.017998  * Constants.ARC_SECONDS_TO_RADIANS);
57              private final PolynomialNutation<DerivativeStructure> thetaA =
58                      new PolynomialNutation<DerivativeStructure>(0.0,
59                              2004.3109 * Constants.ARC_SECONDS_TO_RADIANS,
60                              -0.42665  * Constants.ARC_SECONDS_TO_RADIANS,
61                              -0.041833 * Constants.ARC_SECONDS_TO_RADIANS);
62              private final PolynomialNutation<DerivativeStructure> zA =
63                      new PolynomialNutation<DerivativeStructure>(0.0,
64                              2306.2181 * Constants.ARC_SECONDS_TO_RADIANS,
65                              1.09468   * Constants.ARC_SECONDS_TO_RADIANS,
66                              0.018203  * Constants.ARC_SECONDS_TO_RADIANS);
67  
68              public Transform getTransform(AbsoluteDate date) {
69                  final double tc = IERSConventions.IERS_1996.evaluateTC(date);
70                  final Rotation r1 = new Rotation(Vector3D.PLUS_K,  zA.value(tc), RotationConvention.VECTOR_OPERATOR);
71                  final Rotation r2 = new Rotation(Vector3D.PLUS_J, -thetaA.value(tc), RotationConvention.VECTOR_OPERATOR);
72                  final Rotation r3 = new Rotation(Vector3D.PLUS_K,  zetaA.value(tc), RotationConvention.VECTOR_OPERATOR);
73                  return new Transform(date, r1.compose(r2.compose(r3,
74                                                                   RotationConvention.VECTOR_OPERATOR),
75                                                        RotationConvention.VECTOR_OPERATOR));
76              }
77          };
78  
79          MODProvider modProvider = new MODProvider(IERSConventions.IERS_1996);
80  
81          for (double dt = -5 * Constants.JULIAN_YEAR; dt < 5 * Constants.JULIAN_YEAR; dt += 10 * Constants.JULIAN_DAY) {
82              AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt);
83              Transform t = new Transform(date,
84                                          modProvider.getTransform(date).getInverse(),
85                                          eulerBasedProvider.getTransform(date));
86              Assert.assertEquals(0, t.getRotation().getAngle(), 1.01e-11);
87          }
88  
89      }

94          // this alternate representation of the transform

104                                              -0.0000002 * Constants.ARC_SECONDS_TO_RADIANS);
105             private final PolynomialNutation<DerivativeStructure> thetaA =
106                     new PolynomialNutation<DerivativeStructure>(0.0,
107                                            2004.1917476 * Constants.ARC_SECONDS_TO_RADIANS,
108                                              -0.4269353 * Constants.ARC_SECONDS_TO_RADIANS,
109                                              -0.0418251 * Constants.ARC_SECONDS_TO_RADIANS,
110                                              -0.0000601 * Constants.ARC_SECONDS_TO_RADIANS,
111                                              -0.0000001 * Constants.ARC_SECONDS_TO_RADIANS);
112             private final PolynomialNutation<DerivativeStructure> zA =
113                     new PolynomialNutation<DerivativeStructure>(  -2.5976176 * Constants.ARC_SECONDS_TO_RADIANS,
114                                            2306.0803226 * Constants.ARC_SECONDS_TO_RADIANS,
115                                               1.0947790 * Constants.ARC_SECONDS_TO_RADIANS,
116                                               0.0182273 * Constants.ARC_SECONDS_TO_RADIANS,
117                                               0.0000470 * Constants.ARC_SECONDS_TO_RADIANS,
118                                              -0.0000003 * Constants.ARC_SECONDS_TO_RADIANS);
119 
120             public Transform getTransform(AbsoluteDate date) {
121                 final double tc = IERSConventions.IERS_2003.evaluateTC(date);
122                 final Rotation r1 = new Rotation(Vector3D.PLUS_K,  zA.value(tc), RotationConvention.VECTOR_OPERATOR);
123                 final Rotation r2 = new Rotation(Vector3D.PLUS_J, -thetaA.value(tc), RotationConvention.VECTOR_OPERATOR);
124                 final Rotation r3 = new Rotation(Vector3D.PLUS_K,  zetaA.value(tc), RotationConvention.VECTOR_OPERATOR);

128             }

130 
131         MODProvider modProvider = new MODProvider(IERSConventions.IERS_2003);
132 
133         for (double dt = -Constants.JULIAN_CENTURY; dt < Constants.JULIAN_CENTURY; dt += 50 * Constants.JULIAN_DAY) {
134             AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt);
135             Transform t = new Transform(date,
136                                         modProvider.getTransform(date).getInverse(),
137                                         eulerBasedProvider.getTransform(date));
138             Assert.assertEquals(0, t.getRotation().getAngle(), 6.6e-13);
139         }

218     @Test
219     public void testSofaPmat76() throws OrekitException {
220 
221         // the reference value has been computed using the March 2012 version of the SOFA library
222         // http://www.iausofa.org/2012_0301_C.html, with the following code
223         //
224         //        double utc1, utc2, tai1, tai2, tt1, tt2, rmatp[3][3];
225         //        
226         //        // 2004-02-14:00:00:00Z, MJD = 53049, UT1-UTC = -0.4093509
227         //        utc1  = DJM0 + 53049.0;
228         //        utc2  = 0.0;
229         //        iauUtctai(utc1, utc2, &tai1, &tai2);
230         //        iauTaitt(tai1, tai2, &tt1, &tt2);
231         //
232         //        iauPmat76(tt1, tt2, rmatp);
233         //
234         //        printf("iauPmat76(%.20g, %.20g,rmatp)\n"
235         //               "  --> %.20g %.20g %.20g\n"
236         //               "      %.20g %.20g %.20g\n"
237         //               "      %.20g %.20g %.20g\n",
238         //               tt1, tt2,
239         //               rmatp[0][0], rmatp[0][1], rmatp[0][2],
240         //               rmatp[1][0], rmatp[1][1], rmatp[1][2],
241         //               rmatp[2][0], rmatp[2][1], rmatp[2][2]);
242         //
243         // the output of this test reads:
244         //      iauPmat76(2453049.5, 0.00074287037037037029902,rmatp)
245         //        --> 0.9999994956729996165 -0.00092111268696996330928 -0.00040025637336518803469
246         //            0.00092111268696944094067 0.99999957577560194544 -1.843419633938077413e-07
247         //            0.00040025637336639019806 -1.8433935312187383064e-07 0.99999991989739756004
248 
249         // the SOFA routine iauPmat76 uses the Euler angles zetaA, thetaA and zA, whereas
250         // Orekit uses the "canonical 4-rotation method" with angles epsilon0, psiA, omegaA
251         // and chiA. As seen in the thresholds of the test testEuler1976 above, the two methods
252         // drift with respect to each other at a rate about 2e-12 radians/year (i.e. about 0.42
253         // microarcseconds per year) on the time range around J2000.0). We see the same models
254         // difference in this test as in the testEuler1976 above.
255         // Note that the difference is much smaller for IERS 2003 (see another test below)
256         // because the Euler models for these conventions match much better the 4-rotation angles
257 
258         AbsoluteDate date = new AbsoluteDate(2004, 2, 14, TimeScalesFactory.getUTC());
259         Frame mod  = FramesFactory.getFrame(Predefined.MOD_CONVENTIONS_1996);
260         Frame gcrf = FramesFactory.getFrame(Predefined.GCRF);
261         checkRotation(new double[][] {
262             { 0.9999994956729996165,     -0.00092111268696996330928, -0.00040025637336518803469 },
263             { 0.00092111268696944094067,  0.99999957577560194544,    -1.843419633938077413e-07  },
264             { 0.00040025637336639019806, -1.8433935312187383064e-07,  0.99999991989739756004    }
265 
266         }, gcrf.getTransformTo(mod, date), 9e-12);
267 
268     }

380     @Test
381     public void testMOD1976vs2006() throws OrekitException {
382 
383         final Frame mod1976 = FramesFactory.getMOD(IERSConventions.IERS_1996);
384         final Frame mod2006 = FramesFactory.getMOD(IERSConventions.IERS_2010);
385         for (double dt = 0; dt < 10 * Constants.JULIAN_YEAR; dt += 10 * Constants.JULIAN_DAY) {
386             AbsoluteDate date = new AbsoluteDate(AbsoluteDate.J2000_EPOCH, dt);
387             double delta = mod1976.getTransformTo(mod2006, date).getRotation().getAngle();
388             // MOD2006 and MOD2000 are similar to about 33 milli-arcseconds between 2000 and 2010
389             Assert.assertEquals(0.0, delta, 2.0e-7);
390         }

415         Assert.assertTrue(bos.size() < 250);
416

418         ObjectInputStream     ois = new ObjectInputStream(bis);
419         MODProvider deserialized  = (MODProvider) ois.readObject();
420         for (double dt = 0; dt < Constants.JULIAN_DAY; dt += 3600) {
421             AbsoluteDate date = AbsoluteDate.J2000_EPOCH.shiftedBy(dt);
422             Transform expectedIdentity = new Transform(date,
423                                                        provider.getTransform(date).getInverse(),
424                                                        deserialized.getTransform(date));
425             Assert.assertEquals(0.0, expectedIdentity.getTranslation().getNorm(), 1.0e-15);
426             Assert.assertEquals(0.0, expectedIdentity.getRotation().getAngle(),   1.0e-15);
