6   package spoon.reflect.meta.impl;
7   
8   import java.util.AbstractMap;
9   import java.util.Collection;
10  import java.util.LinkedHashMap;
11  import java.util.Map;
12  import java.util.Set;
13  
14  import spoon.reflect.meta.ContainerKind;
15  import spoon.reflect.meta.RoleHandler;
16  import spoon.reflect.path.CtRole;
17  
18  /**
19   * implementation of {@link RoleHandler}, which handles attributes of type Map&lt;String, V&gt;
20   * @param <T> the type of node whose attribute has to be manipulated
21   * @param <V> the type of item value of the attribute
22   */
23  abstract class MapHandler<T, V> extends AbstractRoleHandler<T, Map<String, V>, V> {
24  
25  	protected MapHandler(CtRole role, Class<T> targetType, Class<?> valueClass) {
26  		super(role, targetType, valueClass);
27  	}
28  
29  	@Override
30  	public ContainerKind getContainerKind() {
31  		return ContainerKind.MAP;
32  	}
33  
34  	@Override
35  	protected Map<String, V> castValue(Object value) {
36  		Map<String, V> map = super.castValue(value);
37  		//check that each item has expected class
38  		checkItemsClass(map.values());
39  		return map;
40  	}
41  
42  	@Override
43  	@SuppressWarnings({ "unchecked", "rawtypes" })
44  	public <W, X> java.util.Collection<X> asCollection(W element) {
45  		return (Collection) asMap(element).values();
46  	}
47  
48  	@Override
49  	public <W, X> Map<String, X> asMap(W e) {
50  		// TODO Auto-generated method stub
51  		return new AbstractMap<String, X>() {
52  			T element = castTarget(e);
53  
54  			@SuppressWarnings({ "unchecked", "rawtypes" })
55  			@Override
56  			public Set<Map.Entry<String, X>> entrySet() {
57  				return (Set) MapHandler.this.entrySet(element);
58  			}
59  
60  			@SuppressWarnings("unchecked")
61  			@Override
62  			public X get(Object key) {
63  				return (X) MapHandler.this.get(element, key);
64  			}
65  
66  			@SuppressWarnings("unchecked")
67  			@Override
68  			public X put(String key, X value) {
69  				return (X) MapHandler.this.put(element, key, castItemValue(value));
70  			}
71  		};
72  	}
73  
74  	protected V get(T element, Object key) {
75  		return this.<T, Map<String, V>>getValue(element).get(key);
76  	}
77  
78  	protected V put(T element, String key, V value) {
79  		Map<String, V> values = new LinkedHashMap<>(this.<T, Map<String, V>>getValue(element));
80  		V ret = values.put(key, value);
81  		setValue(element, values);
82  		return ret;
83  	}
84  
85  	protected Set<Map.Entry<String, V>> entrySet(T element) {
86  		return this.<T, Map<String, V>>getValue(element).entrySet();
87  	}
88  }
