6   package spoon.support.sniper.internal;
7   
8   /**
9    * A {@link SourceFragmentContext}, which prints the element using standard pretty printing
10   */
11  public class SourceFragmentContextPrettyPrint implements SourceFragmentContext {
12  	/**
13  	 * This context is used to force normal pretty printing of element
14  	 */
15  	public static final SourceFragmentContextPrettyPrint INSTANCE = new SourceFragmentContextPrettyPrint();
16  
17  	private SourceFragmentContextPrettyPrint() {
18  	}
19  
20  	@Override
21  	public void onPrintEvent(PrinterEvent event) {
22  		event.print(null);
23  	}
24  
25  	@Override
26  	public void onFinished() {
27  	}
28  
29  	@Override
30  	public boolean matchesPrinterEvent(PrinterEvent event) {
31  		return true;
32  	}
33  }
