17  package spoon.support.compiler.jdt;
18  
19  import org.junit.Test;
20  import spoon.compiler.builder.AdvancedOptions;
21  import spoon.compiler.builder.AnnotationProcessingOptions;
22  import spoon.compiler.builder.ClasspathOptions;
23  import spoon.compiler.builder.ComplianceOptions;
24  import spoon.compiler.builder.JDTBuilderImpl;
25  import spoon.compiler.builder.SourceOptions;
26  
27  import java.io.File;
28  
29  import static org.junit.Assert.assertEquals;
30  
31  public class JDTBuilderTest {
32  	private static final String TEST_CLASSPATH = "./src/test/java/spoon/test/";
33  
34  	@Test
35  	public void testJdtBuilder() {
36  		final String[] builder = new JDTBuilderImpl() //
37  				.classpathOptions(new ClasspathOptions().classpath(TEST_CLASSPATH).bootclasspath(TEST_CLASSPATH).binaries(".").encoding("UTF-8")) //
38  				.complianceOptions(new ComplianceOptions().compliance(8)) //
39  				.annotationProcessingOptions(new AnnotationProcessingOptions().compileProcessors()) //
40  				.advancedOptions(new AdvancedOptions().continueExecution().enableJavadoc().preserveUnusedVars()) //
41  				.sources(new SourceOptions().sources(".")) //
42  				.build();
43  
44  		assertEquals("-cp", builder[0]);
45  		assertEquals(TEST_CLASSPATH, builder[1]);
46  		assertEquals("-bootclasspath", builder[2]);
47  		assertEquals(TEST_CLASSPATH, builder[3]);
48  		assertEquals("-d", builder[4]);
49  		assertEquals(new File(".").getAbsolutePath(), builder[5]);
50  		assertEquals("-encoding", builder[6]);
51  		assertEquals("UTF-8", builder[7]);
52  		assertEquals("-1.8", builder[8]);
53  		assertEquals("-proc:none", builder[9]);
54  		assertEquals("-noExit", builder[10]);
55  		assertEquals("-enableJavadoc", builder[11]);
56  		assertEquals("-preserveAllLocals", builder[12]);
57  		assertEquals(".", builder[13]);
58  	}
59  }
