1   package edu.umd.cs.findbugs;
2   
3   public class FieldWarningSuppressor extends ClassWarningSuppressor {
4   
5       FieldAnnotation field;
6   
7       public FieldWarningSuppressor(String bugPattern, ClassAnnotation clazz, FieldAnnotation field) {
8           super(bugPattern, clazz);
9           this.field = field;
10      }
11  
12  
13      @Override
14      public String toString() {
15          return String.format("Suppress %s in %s.%s", bugPattern, clazz, field);
16      }
17      @Override
18      public boolean match(BugInstance bugInstance) {
19  
20          if (!super.match(bugInstance)) {
21              return false;
22          }
23  
24          FieldAnnotation bugField = bugInstance.getPrimaryField();
25          if (bugField == null || !field.equals(bugField)) {
26              return false;
27          }
28          if (DEBUG) {
29              System.out.println("Suppressing " + bugInstance);
30          }
31          return true;
32      }
33  }
