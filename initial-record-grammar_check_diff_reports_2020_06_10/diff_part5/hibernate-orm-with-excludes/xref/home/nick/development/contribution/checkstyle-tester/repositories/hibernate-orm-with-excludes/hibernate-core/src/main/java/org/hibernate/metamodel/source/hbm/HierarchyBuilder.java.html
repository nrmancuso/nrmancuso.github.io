24  package org.hibernate.metamodel.source.hbm;
25  
26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.Iterator;
29  import java.util.List;
30  import java.util.Map;
31  
32  import org.hibernate.MappingException;
33  import org.hibernate.internal.jaxb.mapping.hbm.EntityElement;
34  import org.hibernate.internal.jaxb.mapping.hbm.JaxbHibernateMapping;
35  import org.hibernate.internal.jaxb.mapping.hbm.JaxbJoinedSubclassElement;
36  import org.hibernate.internal.jaxb.mapping.hbm.JaxbSubclassElement;
37  import org.hibernate.internal.jaxb.mapping.hbm.JaxbUnionSubclassElement;
38  import org.hibernate.internal.jaxb.mapping.hbm.SubEntityElement;
39  import org.hibernate.metamodel.source.binder.SubclassEntityContainer;
40  import org.hibernate.metamodel.source.binder.SubclassEntitySource;
41  
42  /**
43   * @author Steve Ebersole
44   */
45  public class HierarchyBuilder {
46  	private final List<EntityHierarchyImpl> entityHierarchies = new ArrayList<EntityHierarchyImpl>();
47  
48  	// process state
49  	private final Map<String,SubclassEntityContainer> subEntityContainerMap = new HashMap<String, SubclassEntityContainer>();
50  	private final List<ExtendsQueueEntry> extendsQueue = new ArrayList<ExtendsQueueEntry>();
51  
52  	// mapping file specific state
53  	private MappingDocument currentMappingDocument;
54  
55  	public void processMappingDocument(MappingDocument mappingDocument) {
56  		this.currentMappingDocument = mappingDocument;
57  		try {
58  			processCurrentMappingDocument();
59  		}
60  		finally {
61  			this.currentMappingDocument = null;
62  		}
63  	}
64  
65  	private void processCurrentMappingDocument() {
66  		for ( Object entityElementO : currentMappingDocument.getMappingRoot().getClazzOrSubclassOrJoinedSubclass() ) {
67  			final EntityElement entityElement = (EntityElement) entityElementO;
68  			if ( JaxbHibernateMapping.JaxbClass.class.isInstance( entityElement ) ) {
69  				// we can immediately handle <class/> elements in terms of creating the hierarchy entry
70  				final JaxbHibernateMapping.JaxbClass jaxbClass = (JaxbHibernateMapping.JaxbClass) entityElement;
71  				final RootEntitySourceImpl rootEntitySource = new RootEntitySourceImpl( currentMappingDocument,
72  																						jaxbClass
73  				);
74  				final EntityHierarchyImpl hierarchy = new EntityHierarchyImpl( rootEntitySource );
75  
76  				entityHierarchies.add( hierarchy );
77  				subEntityContainerMap.put( rootEntitySource.getEntityName(), rootEntitySource );
78  
79  				processSubElements( entityElement, rootEntitySource );
80  			}
81  			else {
82  				// we have to see if this things super-type has been found yet, and if not add it to the
83  				// extends queue
84  				final SubclassEntitySourceImpl subClassEntitySource = new SubclassEntitySourceImpl( currentMappingDocument, entityElement );
85  				final String entityName = subClassEntitySource.getEntityName();
86  				subEntityContainerMap.put( entityName, subClassEntitySource );
87  				final String entityItExtends = currentMappingDocument.getMappingLocalBindingContext().qualifyClassName(
88  						((SubEntityElement) entityElement).getExtends()
89  				);
90  				processSubElements( entityElement, subClassEntitySource );
91  				final SubclassEntityContainer container = subEntityContainerMap.get( entityItExtends );
92  				if ( container != null ) {
93  					// we already have this entity's super, attach it and continue
94  					container.add( subClassEntitySource );
95  				}
96  				else {
97  					// we do not yet have the super and have to wait, so add it fto the extends queue
98  					extendsQueue.add( new ExtendsQueueEntry( subClassEntitySource, entityItExtends ) );
99  				}
100 			}
101 		}
102 	}
103 
104 	public List<EntityHierarchyImpl> groupEntityHierarchies() {
105 		while ( ! extendsQueue.isEmpty() ) {
106 			// set up a pass over the queue
107 			int numberOfMappingsProcessed = 0;
108 			Iterator<ExtendsQueueEntry> iterator = extendsQueue.iterator();
109 			while ( iterator.hasNext() ) {
110 				final ExtendsQueueEntry entry = iterator.next();
111 				final SubclassEntityContainer container = subEntityContainerMap.get( entry.entityItExtends );
112 				if ( container != null ) {
113 					// we now have this entity's super, attach it and remove entry from extends queue
114 					container.add( entry.subClassEntitySource );
115 					iterator.remove();
116 					numberOfMappingsProcessed++;
117 				}
118 			}
119 
120 			if ( numberOfMappingsProcessed == 0 ) {
121 				// todo : we could log the waiting dependencies...
122 				throw new MappingException( "Unable to process extends dependencies in hbm files" );
123 			}
124 		}
125 
126 		return entityHierarchies;
127 	}
128 
129 	private void processSubElements(EntityElement entityElement, SubclassEntityContainer container) {
130 		if ( JaxbHibernateMapping.JaxbClass.class.isInstance( entityElement ) ) {
131 			final JaxbHibernateMapping.JaxbClass jaxbClass = (JaxbHibernateMapping.JaxbClass) entityElement;
132 			processElements( jaxbClass.getJoinedSubclass(), container );
133 			processElements( jaxbClass.getSubclass(), container );
134 			processElements( jaxbClass.getUnionSubclass(), container );
135 		}
136 		else if ( JaxbSubclassElement.class.isInstance( entityElement ) ) {
137 			final JaxbSubclassElement jaxbSubclass = (JaxbSubclassElement) entityElement;
138 			processElements( jaxbSubclass.getSubclass(), container );
139 		}
140 		else if ( JaxbJoinedSubclassElement.class.isInstance( entityElement ) ) {
141 			final JaxbJoinedSubclassElement jaxbJoinedSubclass = (JaxbJoinedSubclassElement) entityElement;
142 			processElements( jaxbJoinedSubclass.getJoinedSubclass(), container );
143 		}
144 		else if ( JaxbUnionSubclassElement.class.isInstance( entityElement ) ) {
145 			final JaxbUnionSubclassElement jaxbUnionSubclass = (JaxbUnionSubclassElement) entityElement;
146 			processElements( jaxbUnionSubclass.getUnionSubclass(), container );
147 		}
148 	}
149 
150 	private void processElements(List subElements, SubclassEntityContainer container) {
151 		for ( Object subElementO : subElements ) {
152 			final SubEntityElement subElement = (SubEntityElement) subElementO;
153 			final SubclassEntitySourceImpl subclassEntitySource = new SubclassEntitySourceImpl( currentMappingDocument, subElement );
154 			container.add( subclassEntitySource );
155 			final String subEntityName = subclassEntitySource.getEntityName();
156 			subEntityContainerMap.put( subEntityName, subclassEntitySource );
157 		}
158 	}
159 
160 	private static class ExtendsQueueEntry {
161 		private final SubclassEntitySource subClassEntitySource;
162 		private final String entityItExtends;
163 
164 		private ExtendsQueueEntry(SubclassEntitySource subClassEntitySource, String entityItExtends) {
165 			this.subClassEntitySource = subClassEntitySource;
166 			this.entityItExtends = entityItExtends;
167 		}
168 	}
169 }
