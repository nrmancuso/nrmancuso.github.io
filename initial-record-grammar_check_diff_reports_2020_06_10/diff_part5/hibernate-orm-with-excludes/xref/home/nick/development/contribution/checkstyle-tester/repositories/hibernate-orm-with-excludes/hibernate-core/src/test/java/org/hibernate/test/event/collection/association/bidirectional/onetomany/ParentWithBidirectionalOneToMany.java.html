22  package org.hibernate.test.event.collection.association.bidirectional.onetomany;
23  import java.util.Collection;
24  import java.util.Iterator;
25  
26  import org.hibernate.test.event.collection.AbstractParentWithCollection;
27  import org.hibernate.test.event.collection.Child;
28  
29  /**
30   *
31   * @author Gail Badner
32   */
33  public class ParentWithBidirectionalOneToMany extends AbstractParentWithCollection {
34  	public ParentWithBidirectionalOneToMany() {
35  	}
36  
37  	public ParentWithBidirectionalOneToMany(String name) {
38  		super( name );
39  	}
40  
41  	public Child createChild( String name ) {
42  		return new ChildWithManyToOne( name );
43  	}
44  
45  	public Child addChild(String name) {
46  		Child child = createChild( name );
47  		addChild( child );
48  		return child;
49  	}
50  
51  	public void addChild(Child child) {
52  		super.addChild( child );
53  		( ( ChildWithManyToOne ) child ).setParent( this );
54  	}
55  
56  	public void newChildren(Collection children) {
57  		if ( children == getChildren() ) {
58  			return;
59  		}
60  		if ( getChildren() != null ) {
61  			for ( Iterator it = getChildren().iterator(); it.hasNext(); ) {
62  				ChildWithManyToOne child =  ( ChildWithManyToOne ) it.next();
63  				child.setParent( null );
64  			}
65  		}
66  		if ( children != null ) {
67  			for ( Iterator it = children.iterator(); it.hasNext(); ) {
68  				ChildWithManyToOne child = ( ChildWithManyToOne ) it.next();
69  				child.setParent( this );
70  			}
71  		}
72  		super.newChildren( children );
73  	}
74  
75  	public void removeChild(Child child) {
76  		// Note: there can be more than one child in the collection
77  		super.removeChild( child );
78  		// only set the parent to null if child is no longer in the bag
79  		if ( ! getChildren().contains( child ) ) {
80  			( ( ChildWithManyToOne ) child ).setParent( null );
81  		}
82  	}
83  
84  }
