24  package org.hibernate.bytecode.internal.javassist;
25  
26  /**
27   * The interface defining how interception of a field should be handled.
28   *
29   * @author Muga Nishizawa
30   */
31  public interface FieldHandler {
32  
33  	/**
34  	 * Called to handle writing an int value to a given field.
35  	 *
36  	 * @param obj ?
37  	 * @param name The name of the field being written
38  	 * @param oldValue The old field value
39  	 * @param newValue The new field value.
40  	 * @return ?
41  	 */
42  	int writeInt(Object obj, String name, int oldValue, int newValue);
43  
44  	char writeChar(Object obj, String name, char oldValue, char newValue);
45  
46  	byte writeByte(Object obj, String name, byte oldValue, byte newValue);
47  
48  	boolean writeBoolean(Object obj, String name, boolean oldValue,
49  			boolean newValue);
50  
51  	short writeShort(Object obj, String name, short oldValue, short newValue);
52  
53  	float writeFloat(Object obj, String name, float oldValue, float newValue);
54  
55  	double writeDouble(Object obj, String name, double oldValue, double newValue);
56  
57  	long writeLong(Object obj, String name, long oldValue, long newValue);
58  
59  	Object writeObject(Object obj, String name, Object oldValue, Object newValue);
60  
61  	int readInt(Object obj, String name, int oldValue);
62  
63  	char readChar(Object obj, String name, char oldValue);
64  
65  	byte readByte(Object obj, String name, byte oldValue);
66  
67  	boolean readBoolean(Object obj, String name, boolean oldValue);
68  
69  	short readShort(Object obj, String name, short oldValue);
70  
71  	float readFloat(Object obj, String name, float oldValue);
72  
73  	double readDouble(Object obj, String name, double oldValue);
74  
75  	long readLong(Object obj, String name, long oldValue);
76  
77  	Object readObject(Object obj, String name, Object oldValue);
78  
79  }
