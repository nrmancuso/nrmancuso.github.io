22  package org.hibernate.ejb.util;
23  import java.net.URL;
24  import java.util.Enumeration;
25  import java.util.List;
26  import java.util.Properties;
27  import javax.persistence.spi.PersistenceUnitInfo;
28  
29  /**
30   * @author Emmanuel Bernard
31   */
32  public final class LogHelper {
33  	private LogHelper() {
34  	}
35  
36  	public static String logPersistenceUnitInfo(PersistenceUnitInfo unitInfo) {
37  		StringBuilder sb = new StringBuilder();
38  		sb.append( "PersistenceUnitInfo [\n\t" )
39  				.append( "name: " )
40  				.append( unitInfo.getPersistenceUnitName() )
41  				.append( "\n\t" )
42  				.append( "persistence provider classname: " )
43  				.append( unitInfo.getPersistenceProviderClassName() )
44  				.append( "\n\t" )
45  				.append( "classloader: " )
46  				.append( unitInfo.getClassLoader() )
47  				.append( "\n\t" )
48  				.append( "Temporary classloader: " )
49  				.append( unitInfo.getNewTempClassLoader() )
50  				.append( "\n\t" )
51  				.append( "excludeUnlistedClasses: " )
52  				.append( unitInfo.excludeUnlistedClasses() )
53  				.append( "\n\t" )
54  				.append( "JTA datasource: " )
55  				.append( unitInfo.getJtaDataSource() )
56  				.append( "\n\t" )
57  				.append( "Non JTA datasource: " )
58  				.append( unitInfo.getNonJtaDataSource() )
59  				.append( "\n\t" )
60  				.append( "Transaction type: " )
61  				.append( unitInfo.getTransactionType() )
62  				.append( "\n\t" )
63  				.append( "PU root URL: " )
64  				.append( unitInfo.getPersistenceUnitRootUrl() )
65  				.append( "\n\t" )
66  				.append( "Shared Cache Mode: " )
67  				.append( unitInfo.getSharedCacheMode() )
68  				.append( "\n\t" )
69  				.append( "Validation Mode: " )
70  				.append( unitInfo.getValidationMode() )
71  				.append( "\n\t" );
72  		sb.append( "Jar files URLs [" );
73  		List<URL> jarFileUrls = unitInfo.getJarFileUrls();
74  		if ( jarFileUrls != null ) {
75  			for ( URL url : jarFileUrls ) {
76  				sb.append( "\n\t\t" ).append( url );
77  			}
78  		}
79  		sb.append( "]\n\t" );
80  		sb.append( "Managed classes names [" );
81  		List<String> classNames = unitInfo.getManagedClassNames();
82  		if ( classNames != null ) {
83  			for ( String className : classNames ) {
84  				sb.append( "\n\t\t" ).append( className );
85  			}
86  		}
87  		sb.append( "]\n\t" );
88  		sb.append( "Mapping files names [" );
89  		List<String> mappingFiles = unitInfo.getMappingFileNames();
90  		if ( mappingFiles != null ) {
91  			for ( String file : mappingFiles ) {
92  				sb.append( "\n\t\t" ).append( file );
93  			}
94  		}
95  		sb.append( "]\n\t" );
96  		sb.append( "Properties [" );
97  		Properties properties = unitInfo.getProperties();
98  		if (properties != null) {
99  			Enumeration names = properties.propertyNames();
100 			while ( names.hasMoreElements() ) {
101 				String name = (String) names.nextElement();
102 				sb.append( "\n\t\t" ).append( name ).append( ": " ).append( properties.getProperty( name ) );
103 			}
104 		}
105 		sb.append( "]" );
106 		return sb.toString();
107 	}
108 
109 }
