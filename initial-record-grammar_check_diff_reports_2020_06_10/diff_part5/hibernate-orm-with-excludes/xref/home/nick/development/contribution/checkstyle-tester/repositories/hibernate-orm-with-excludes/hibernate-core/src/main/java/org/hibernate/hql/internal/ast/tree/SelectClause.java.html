122 	/**
123 	 * Prepares an explicitly defined select clause.
124 	 *
125 	 * @param fromClause The from clause linked to this select clause.
126 	 * @throws SemanticException indicates a semntic issue with the explicit select clause.
127 	 */
128 	public void initializeExplicitSelectClause(FromClause fromClause) throws SemanticException {
129 		if ( prepared ) {
130 			throw new IllegalStateException( "SelectClause was already prepared!" );
131 		}
132 
133 		//explicit = true;	// This is an explict Select.
134 		//ArrayList sqlResultTypeList = new ArrayList();
135 		ArrayList queryReturnTypeList = new ArrayList();
136 
137 		// First, collect all of the select expressions.
138 		// NOTE: This must be done *before* invoking setScalarColumnText() because setScalarColumnText()
139 		// changes the AST!!!
140 		SelectExpression[] selectExpressions = collectSelectExpressions();
141 
142 		for ( int i = 0; i < selectExpressions.length; i++ ) {
143 			SelectExpression selectExpression = selectExpressions[i];
144 
145 			if ( AggregatedSelectExpression.class.isInstance( selectExpression ) ) {
146 				aggregatedSelectExpression = (AggregatedSelectExpression) selectExpression;
147 				queryReturnTypeList.addAll( aggregatedSelectExpression.getAggregatedSelectionTypeList() );
148 				scalarSelect = true;
149 			}
150 			else {
151 				Type type = selectExpression.getDataType();
152 				if ( type == null ) {
153 					throw new IllegalStateException( "No data type for node: " + selectExpression.getClass().getName() + " "
154 							+ new ASTPrinter( SqlTokenTypes.class ).showAsString( ( AST ) selectExpression, "" ) );
155 				}
156 				//sqlResultTypeList.add( type );
157 
158 				// If the data type is not an association type, it could not have been in the FROM clause.
159 				if ( selectExpression.isScalar() ) {
160 					scalarSelect = true;
161 				}
162 
163 				if ( isReturnableEntity( selectExpression ) ) {
164 					fromElementsForLoad.add( selectExpression.getFromElement() );
165 				}
166 
167 				// Always add the type to the return type list.
168 				queryReturnTypeList.add( type );
169 			}
170 		}
171 
172 		//init the aliases, after initing the constructornode
173 		initAliases(selectExpressions);
174 
175 		if ( !getWalker().isShallowQuery() ) {
176 			// add the fetched entities
177 			List fromElements = fromClause.getProjectionList();
178 
179 			ASTAppender appender = new ASTAppender( getASTFactory(), this );	// Get ready to start adding nodes.
180 			int size = fromElements.size();
181 	
182 			Iterator iterator = fromElements.iterator();
183 			for ( int k = 0; iterator.hasNext(); k++ ) {
184 				FromElement fromElement = ( FromElement ) iterator.next();
185 
186 				if ( fromElement.isFetch() ) {
187 					FromElement origin = null;
188 					if ( fromElement.getRealOrigin() == null ) {
189 						// work around that crazy issue where the tree contains
190 						// "empty" FromElements (no text); afaict, this is caused
191 						// by FromElementFactory.createCollectionJoin()
192 						if ( fromElement.getOrigin() == null ) {
193 							throw new QueryException( "Unable to determine origin of join fetch [" + fromElement.getDisplayText() + "]" );
194 						}
195 						else {
196 							origin = fromElement.getOrigin();
197 						}
198 					}
199 					else {
200 						origin = fromElement.getRealOrigin();
201 					}
202 					if ( !fromElementsForLoad.contains( origin ) ) {
203 						throw new QueryException(
204 								"query specified join fetching, but the owner " +
205 								"of the fetched association was not present in the select list " +
206 								"[" + fromElement.getDisplayText() + "]"
207 						);
208 					}
209 					Type type = fromElement.getSelectType();
210 					addCollectionFromElement( fromElement );
211 					if ( type != null ) {
212 						boolean collectionOfElements = fromElement.isCollectionOfValuesOrComponents();
213 						if ( !collectionOfElements ) {
214 							// Add the type to the list of returned sqlResultTypes.
215 							fromElement.setIncludeSubclasses( true );
216 							fromElementsForLoad.add( fromElement );
217 							//sqlResultTypeList.add( type );
218 							// Generate the select expression.
219 							String text = fromElement.renderIdentifierSelect( size, k );
220 							SelectExpressionImpl generatedExpr = ( SelectExpressionImpl ) appender.append( SqlTokenTypes.SELECT_EXPR, text, false );
221 							if ( generatedExpr != null ) {
222 								generatedExpr.setFromElement( fromElement );
223 							}
224 						}
225 					}
226 				}
227 			}
228 
229 			// generate id select fragment and then property select fragment for
230 			// each expression, just like generateSelectFragments().
231 			renderNonScalarSelects( collectSelectExpressions(), fromClause );
232 		}
233 
234 		if ( scalarSelect || getWalker().isShallowQuery() ) {
235 			// If there are any scalars (non-entities) selected, render the select column aliases.
236 			renderScalarSelects( selectExpressions, fromClause );
237 		}
238 
239 		finishInitialization( /*sqlResultTypeList,*/ queryReturnTypeList );
240 	}

242 	private void finishInitialization(ArrayList queryReturnTypeList) {
243 		queryReturnTypes = ( Type[] ) queryReturnTypeList.toArray( new Type[queryReturnTypeList.size()] );
244 		initializeColumnNames();
245 		prepared = true;
246 	}

248 	private void initializeColumnNames() {
249 		// Generate an 2d array of column names, the first dimension is parallel with the
250 		// return types array.  The second dimension is the list of column names for each
251 		// type.
252 
253 		// todo: we should really just collect these from the various SelectExpressions, rather than regenerating here
254 		columnNames = getSessionFactoryHelper().generateColumnNames( queryReturnTypes );
255 		columnNamesStartPositions = new int[ columnNames.length ];
256 		int startPosition = 1;
257 		for ( int i = 0 ; i < columnNames.length ; i ++ ) {
258 			columnNamesStartPositions[ i ] = startPosition;
259 			startPosition += columnNames[ i ].length;
260 		}
261 	}

263 	public int getColumnNamesStartPosition(int i) {
264 		return columnNamesStartPositions[ i ];
265 	}

267 	/**
268 	 * Prepares a derived (i.e., not explicitly defined in the query) select clause.
269 	 *
270 	 * @param fromClause The from clause to which this select clause is linked.
271 	 */
272 	public void initializeDerivedSelectClause(FromClause fromClause) throws SemanticException {
273 		if ( prepared ) {
274 			throw new IllegalStateException( "SelectClause was already prepared!" );
275 		}
276 		//Used to be tested by the TCK but the test is no longer here
277 //		if ( getSessionFactoryHelper().isStrictJPAQLComplianceEnabled() && !getWalker().isSubQuery() ) {
278 //			// NOTE : the isSubQuery() bit is a temporary hack...
279 //			throw new QuerySyntaxException( "JPA-QL compliance requires select clause" );
280 //		}
281 		List fromElements = fromClause.getProjectionList();
282 
283 		ASTAppender appender = new ASTAppender( getASTFactory(), this );	// Get ready to start adding nodes.
284 		int size = fromElements.size();
285 		ArrayList queryReturnTypeList = new ArrayList( size );
286 
287 		Iterator iterator = fromElements.iterator();
288 		for ( int k = 0; iterator.hasNext(); k++ ) {
289 			FromElement fromElement = ( FromElement ) iterator.next();
290 			Type type = fromElement.getSelectType();
291 
292 			addCollectionFromElement( fromElement );
293 
294 			if ( type != null ) {
295 				boolean collectionOfElements = fromElement.isCollectionOfValuesOrComponents();
296 				if ( !collectionOfElements ) {
297 					if ( !fromElement.isFetch() ) {
298 						// Add the type to the list of returned sqlResultTypes.
299 						queryReturnTypeList.add( type );
300 					}
301 					fromElementsForLoad.add( fromElement );
302 					// Generate the select expression.
303 					String text = fromElement.renderIdentifierSelect( size, k );
304 					SelectExpressionImpl generatedExpr = ( SelectExpressionImpl ) appender.append( SqlTokenTypes.SELECT_EXPR, text, false );
305 					if ( generatedExpr != null ) {
306 						generatedExpr.setFromElement( fromElement );
307 					}
308 				}
309 			}
310 		}
311 
312 		// Get all the select expressions (that we just generated) and render the select.
313 		SelectExpression[] selectExpressions = collectSelectExpressions();
314 
315 		if ( getWalker().isShallowQuery() ) {
316 			renderScalarSelects( selectExpressions, fromClause );
317 		}
318 		else {
319 			renderNonScalarSelects( selectExpressions, fromClause );
320 		}
321 		finishInitialization( queryReturnTypeList );
322 	}

324 	public static boolean VERSION2_SQL = false;
325 
326 	private void addCollectionFromElement(FromElement fromElement) {
327 		if ( fromElement.isFetch() ) {
328 			if ( fromElement.getQueryableCollection() != null ) {
329 				String suffix;
330 				if (collectionFromElements==null) {
331 					collectionFromElements = new ArrayList();
332 					suffix = VERSION2_SQL ? "__" : "0__";
333 				}
334 				else {
335 					suffix = Integer.toString( collectionFromElements.size() ) + "__";
336 				}
337 				collectionFromElements.add( fromElement );
338 				fromElement.setCollectionSuffix( suffix );
339 			}
340 		}
341 	}

343 	protected AST getFirstSelectExpression() {
344 		AST n = getFirstChild();
345 		// Skip 'DISTINCT' and 'ALL', so we return the first expression node.
346 		while ( n != null && ( n.getType() == SqlTokenTypes.DISTINCT || n.getType() == SqlTokenTypes.ALL ) ) {
347 			n = n.getNextSibling();
348 		}
349 		return n;
350 	}

352 	private boolean isReturnableEntity(SelectExpression selectExpression) throws SemanticException {
353 		FromElement fromElement = selectExpression.getFromElement();
354 		boolean isFetchOrValueCollection = fromElement != null && 
355 				( fromElement.isFetch() || fromElement.isCollectionOfValuesOrComponents() );
356 		if ( isFetchOrValueCollection ) {
357 			return false;
358 		}
359 		else {
360 			return selectExpression.isReturnableEntity();
361 		}
362 	}

364 	private void renderScalarSelects(SelectExpression[] se, FromClause currentFromClause) throws SemanticException {
365 		if ( !currentFromClause.isSubQuery() ) {
366 			for ( int i = 0; i < se.length; i++ ) {
367 				SelectExpression expr = se[i];
368 				expr.setScalarColumn( i );	// Create SQL_TOKEN nodes for the columns.
369 			}
370 		}
371 	}

373 	private void initAliases(SelectExpression[] selectExpressions) {
374 		if ( aggregatedSelectExpression == null ) {
375 			aliases = new String[selectExpressions.length];
376 			for ( int i=0; i<selectExpressions.length; i++ ) {
377 				String alias = selectExpressions[i].getAlias();
378 				aliases[i] = alias==null ? Integer.toString(i) : alias;
379 			}
380 		}
381 		else {
382 			aliases = aggregatedSelectExpression.getAggregatedAliases();
383 		}
384 	}

386 	private void renderNonScalarSelects(SelectExpression[] selectExpressions, FromClause currentFromClause) 
387 	throws SemanticException {
388 		ASTAppender appender = new ASTAppender( getASTFactory(), this );
389 		final int size = selectExpressions.length;
390 		int nonscalarSize = 0;
391 		for ( int i = 0; i < size; i++ ) {
392 			if ( !selectExpressions[i].isScalar() ) nonscalarSize++;
393 		}
394 
395 		int j = 0;
396 		for ( int i = 0; i < size; i++ ) {
397 			if ( !selectExpressions[i].isScalar() ) {
398 				SelectExpression expr = selectExpressions[i];
399 				FromElement fromElement = expr.getFromElement();
400 				if ( fromElement != null ) {
401 					renderNonScalarIdentifiers( fromElement, nonscalarSize, j, expr, appender );
402 					j++;
403 				}
404 			}
405 		}
406 
407 		if ( !currentFromClause.isSubQuery() ) {
408 			// Generate the property select tokens.
409 			int k = 0;
410 			for ( int i = 0; i < size; i++ ) {
411 				if ( !selectExpressions[i].isScalar() ) {
412 					FromElement fromElement = selectExpressions[i].getFromElement();
413 					if ( fromElement != null ) {
414 						renderNonScalarProperties( appender, fromElement, nonscalarSize, k );
415 						k++;
416 					}
417 				}
418 			}
419 		}
420 	}
421 
422 	private void renderNonScalarIdentifiers(FromElement fromElement, int nonscalarSize, int j, SelectExpression expr, ASTAppender appender) {
423 		String text = fromElement.renderIdentifierSelect( nonscalarSize, j );
424 		if ( !fromElement.getFromClause().isSubQuery() ) {
425 			if ( !scalarSelect && !getWalker().isShallowQuery() ) {
426 				//TODO: is this a bit ugly?
427 				expr.setText( text );
428 			}
429 			else {
430 				appender.append( SqlTokenTypes.SQL_TOKEN, text, false );
431 			}
432 		}
433 	}
434 
435 	private void renderNonScalarProperties(ASTAppender appender, FromElement fromElement, int nonscalarSize, int k) {
436 		String text = fromElement.renderPropertySelect( nonscalarSize, k );
437 		appender.append( SqlTokenTypes.SQL_TOKEN, text, false );
438 		if ( fromElement.getQueryableCollection() != null && fromElement.isFetch() ) {
439 			text = fromElement.renderCollectionSelectFragment( nonscalarSize, k );
440 			appender.append( SqlTokenTypes.SQL_TOKEN, text, false );
441 		}
442 		// Look through the FromElement's children to find any collections of values that should be fetched...
443 		ASTIterator iter = new ASTIterator( fromElement );
444 		while ( iter.hasNext() ) {
445 			FromElement child = ( FromElement ) iter.next();
446 			if ( child.isCollectionOfValuesOrComponents() && child.isFetch() ) {
447 				// Need a better way to define the suffixes here...
448 				text = child.renderValueCollectionSelectFragment( nonscalarSize, nonscalarSize + k );
449 				appender.append( SqlTokenTypes.SQL_TOKEN, text, false );
450 			}
451 		}
452 	}
