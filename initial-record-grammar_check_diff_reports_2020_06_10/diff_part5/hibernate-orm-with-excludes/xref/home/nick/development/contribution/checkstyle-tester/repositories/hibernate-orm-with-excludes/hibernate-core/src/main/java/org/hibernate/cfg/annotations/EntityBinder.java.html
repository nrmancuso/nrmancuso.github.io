168 	@SuppressWarnings("SimplifiableConditionalExpression")
169 	private void bindHibernateAnnotation(org.hibernate.annotations.Entity hibAnn) {
170 		{
171 			final DynamicInsert dynamicInsertAnn = annotatedClass.getAnnotation( DynamicInsert.class );
172 			this.dynamicInsert = dynamicInsertAnn == null
173 					? ( hibAnn == null ? false : hibAnn.dynamicInsert() )
174 					: dynamicInsertAnn.value();
175 		}
176 
177 		{
178 			final DynamicUpdate dynamicUpdateAnn = annotatedClass.getAnnotation( DynamicUpdate.class );
179 			this.dynamicUpdate = dynamicUpdateAnn == null
180 					? ( hibAnn == null ? false : hibAnn.dynamicUpdate() )
181 					: dynamicUpdateAnn.value();
182 		}
183 
184 		{
185 			final SelectBeforeUpdate selectBeforeUpdateAnn = annotatedClass.getAnnotation( SelectBeforeUpdate.class );
186 			this.selectBeforeUpdate = selectBeforeUpdateAnn == null
187 					? ( hibAnn == null ? false : hibAnn.selectBeforeUpdate() )
188 					: selectBeforeUpdateAnn.value();
189 		}
190 
191 		{
192 			final OptimisticLocking optimisticLockingAnn = annotatedClass.getAnnotation( OptimisticLocking.class );
193 			this.optimisticLockType = optimisticLockingAnn == null
194 					? ( hibAnn == null ? OptimisticLockType.VERSION : hibAnn.optimisticLock() )
195 					: optimisticLockingAnn.type();
196 		}
197 
198 		{
199 			final Polymorphism polymorphismAnn = annotatedClass.getAnnotation( Polymorphism.class );
200 			this.polymorphismType = polymorphismAnn == null
201 					? ( hibAnn == null ? PolymorphismType.IMPLICIT : hibAnn.polymorphism() )
202 					: polymorphismAnn.type();
203 		}
204 
205 		if ( hibAnn != null ) {
206 			// used later in bind for logging
207 			explicitHibernateEntityAnnotation = true;
208 			//persister handled in bind
209 		}
210 	}

239 	public void bindEntity() {
240 		persistentClass.setAbstract( annotatedClass.isAbstract() );
241 		persistentClass.setClassName( annotatedClass.getName() );
242 		persistentClass.setNodeName( name );
243 		persistentClass.setJpaEntityName(name);
244 		//persistentClass.setDynamic(false); //no longer needed with the Entity name refactoring?
245 		persistentClass.setEntityName( annotatedClass.getName() );
246 		bindDiscriminatorValue();
247 
248 		persistentClass.setLazy( lazy );
249 		if ( proxyClass != null ) {
250 			persistentClass.setProxyInterfaceName( proxyClass.getName() );
251 		}
252 		persistentClass.setDynamicInsert( dynamicInsert );
253 		persistentClass.setDynamicUpdate( dynamicUpdate );
254 
255 		if ( persistentClass instanceof RootClass ) {
256 			RootClass rootClass = (RootClass) persistentClass;
257 			boolean mutable = true;
258 			//priority on @Immutable, then @Entity.mutable()
259 			if ( annotatedClass.isAnnotationPresent( Immutable.class ) ) {
260 				mutable = false;
261 			}
262 			else {
263 				org.hibernate.annotations.Entity entityAnn =
264 						annotatedClass.getAnnotation( org.hibernate.annotations.Entity.class );
265 				if ( entityAnn != null ) {
266 					mutable = entityAnn.mutable();
267 				}
268 			}
269 			rootClass.setMutable( mutable );
270 			rootClass.setExplicitPolymorphism( isExplicitPolymorphism( polymorphismType ) );
271 			if ( StringHelper.isNotEmpty( where ) ) rootClass.setWhere( where );
272 			if ( cacheConcurrentStrategy != null ) {
273 				rootClass.setCacheConcurrencyStrategy( cacheConcurrentStrategy );
274 				rootClass.setCacheRegionName( cacheRegion );
275 				rootClass.setLazyPropertiesCacheable( cacheLazyProperty );
276 			}
277 			rootClass.setNaturalIdCacheRegionName( naturalIdCacheRegion );
278 			boolean forceDiscriminatorInSelects = forceDiscriminator == null
279 					? mappings.forceDiscriminatorInSelectsByDefault()
280 					: forceDiscriminator;
281 			rootClass.setForceDiscriminator( forceDiscriminatorInSelects );
282 			if( insertableDiscriminator != null) {
283 				rootClass.setDiscriminatorInsertable( insertableDiscriminator );
284 			}
285 		}
286 		else {
287             if (explicitHibernateEntityAnnotation) {
288 				LOG.entityAnnotationOnNonRoot(annotatedClass.getName());
289 			}
290             if (annotatedClass.isAnnotationPresent(Immutable.class)) {
291 				LOG.immutableAnnotationOnNonRoot(annotatedClass.getName());
292 			}
293 		}
294 		persistentClass.setOptimisticLockMode( getVersioning( optimisticLockType ) );
295 		persistentClass.setSelectBeforeUpdate( selectBeforeUpdate );
296 
297 		//set persister if needed
298 		Persister persisterAnn = annotatedClass.getAnnotation( Persister.class );
299 		Class persister = null;
300 		if ( persisterAnn != null ) {
301 			persister = persisterAnn.impl();
302 		}
303 		else {
304 			org.hibernate.annotations.Entity entityAnn = annotatedClass.getAnnotation( org.hibernate.annotations.Entity.class );
305 			if ( entityAnn != null && !BinderHelper.isEmptyAnnotationValue( entityAnn.persister() ) ) {
306 				try {
307 					persister = ReflectHelper.classForName( entityAnn.persister() );
308 				}
309 				catch (ClassNotFoundException cnfe) {
310 					throw new AnnotationException( "Could not find persister class: " + persister );
311 				}
312 			}
313 		}
314 		if ( persister != null ) {
315 			persistentClass.setEntityPersisterClass( persister );
316 		}
317 
318 		persistentClass.setBatchSize( batchSize );
319 
320 		//SQL overriding
321 		SQLInsert sqlInsert = annotatedClass.getAnnotation( SQLInsert.class );
322 		SQLUpdate sqlUpdate = annotatedClass.getAnnotation( SQLUpdate.class );
323 		SQLDelete sqlDelete = annotatedClass.getAnnotation( SQLDelete.class );
324 		SQLDeleteAll sqlDeleteAll = annotatedClass.getAnnotation( SQLDeleteAll.class );
325 		Loader loader = annotatedClass.getAnnotation( Loader.class );
326 
327 		if ( sqlInsert != null ) {
328 			persistentClass.setCustomSQLInsert( sqlInsert.sql().trim(), sqlInsert.callable(),
329 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlInsert.check().toString().toLowerCase() )
330 			);
331 
332 		}
333 		if ( sqlUpdate != null ) {
334 			persistentClass.setCustomSQLUpdate( sqlUpdate.sql(), sqlUpdate.callable(),
335 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlUpdate.check().toString().toLowerCase() )
336 			);
337 		}
338 		if ( sqlDelete != null ) {
339 			persistentClass.setCustomSQLDelete( sqlDelete.sql(), sqlDelete.callable(),
340 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlDelete.check().toString().toLowerCase() )
341 			);
342 		}
343 		if ( sqlDeleteAll != null ) {
344 			persistentClass.setCustomSQLDelete( sqlDeleteAll.sql(), sqlDeleteAll.callable(),
345 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlDeleteAll.check().toString().toLowerCase() )
346 			);
347 		}
348 		if ( loader != null ) {
349 			persistentClass.setLoaderName( loader.namedQuery() );
350 		}
351 
352 		if ( annotatedClass.isAnnotationPresent( Synchronize.class )) {
353 			Synchronize synchronizedWith = annotatedClass.getAnnotation(Synchronize.class);
354 
355 			String [] tables = synchronizedWith.value();
356 			for (String table : tables) {
357 				persistentClass.addSynchronizedTable(table);
358 			}
359 		}
360 
361 		if ( annotatedClass.isAnnotationPresent(Subselect.class )) {
362 			Subselect subselect = annotatedClass.getAnnotation(Subselect.class);
363 			this.subselect = subselect.value();
364 		}
365 
366 		//tuplizers
367 		if ( annotatedClass.isAnnotationPresent( Tuplizers.class ) ) {
368 			for (Tuplizer tuplizer : annotatedClass.getAnnotation( Tuplizers.class ).value()) {
369 				EntityMode mode = EntityMode.parse( tuplizer.entityMode() );
370 				//todo tuplizer.entityModeType
371 				persistentClass.addTuplizer( mode, tuplizer.impl().getName() );
372 			}
373 		}
374 		if ( annotatedClass.isAnnotationPresent( Tuplizer.class ) ) {
375 			Tuplizer tuplizer = annotatedClass.getAnnotation( Tuplizer.class );
376 			EntityMode mode = EntityMode.parse( tuplizer.entityMode() );
377 			//todo tuplizer.entityModeType
378 			persistentClass.addTuplizer( mode, tuplizer.impl().getName() );
379 		}
380 
381 		for ( Filter filter : filters ) {
382 			String filterName = filter.name();
383 			String cond = filter.condition();
384 			if ( BinderHelper.isEmptyAnnotationValue( cond ) ) {
385 				FilterDefinition definition = mappings.getFilterDefinition( filterName );
386 				cond = definition == null ? null : definition.getDefaultFilterCondition();
387 				if ( StringHelper.isEmpty( cond ) ) {
388 					throw new AnnotationException(
389 							"no filter condition found for filter " + filterName + " in " + this.name
390 					);
391 				}
392 			}
393 			persistentClass.addFilter(filterName, cond, filter.deduceAliasInjectionPoints(), 
394 					toAliasTableMap(filter.aliases()), toAliasEntityMap(filter.aliases()));
395 		}
396 		LOG.debugf( "Import with entity name %s", name );
397 		try {
398 			mappings.addImport( persistentClass.getEntityName(), name );
399 			String entityName = persistentClass.getEntityName();
400 			if ( !entityName.equals( name ) ) {
401 				mappings.addImport( entityName, entityName );
402 			}
403 		}
404 		catch (MappingException me) {
405 			throw new AnnotationException( "Use of the same entity name twice: " + name, me );
406 		}
407 	}

409 	public void bindDiscriminatorValue() {
410 		if ( StringHelper.isEmpty( discriminatorValue ) ) {
411 			Value discriminator = persistentClass.getDiscriminator();
412 			if ( discriminator == null ) {
413 				persistentClass.setDiscriminatorValue( name );
414 			}
415 			else if ( "character".equals( discriminator.getType().getName() ) ) {
416 				throw new AnnotationException(
417 						"Using default @DiscriminatorValue for a discriminator of type CHAR is not safe"
418 				);
419 			}
420 			else if ( "integer".equals( discriminator.getType().getName() ) ) {
421 				persistentClass.setDiscriminatorValue( String.valueOf( name.hashCode() ) );
422 			}
423 			else {
424 				persistentClass.setDiscriminatorValue( name ); //Spec compliant
425 			}
426 		}
427 		else {
428 			//persistentClass.getDiscriminator()
429 			persistentClass.setDiscriminatorValue( discriminatorValue );
430 		}
431 	}

433 	int getVersioning(OptimisticLockType type) {
434 		switch ( type ) {
435 			case VERSION:
436 				return Versioning.OPTIMISTIC_LOCK_VERSION;
437 			case NONE:
438 				return Versioning.OPTIMISTIC_LOCK_NONE;
439 			case DIRTY:
440 				return Versioning.OPTIMISTIC_LOCK_DIRTY;
441 			case ALL:
442 				return Versioning.OPTIMISTIC_LOCK_ALL;
443 			default:
444 				throw new AssertionFailure( "optimistic locking not supported: " + type );
445 		}
446 	}

469 	public void setProxy(Proxy proxy) {
470 		if ( proxy != null ) {
471 			lazy = proxy.lazy();
472 			if ( !lazy ) {
473 				proxyClass = null;
474 			}
475 			else {
476 				if ( AnnotationBinder.isDefault(
477 						mappings.getReflectionManager().toXClass( proxy.proxyClass() ), mappings
478 				) ) {
479 					proxyClass = annotatedClass;
480 				}
481 				else {
482 					proxyClass = mappings.getReflectionManager().toXClass( proxy.proxyClass() );
483 				}
484 			}
485 		}
486 		else {
487 			lazy = true; //needed to allow association lazy loading.
488 			proxyClass = annotatedClass;
489 		}
490 	}

560 	public void bindTable(
561 			String schema,
562 			String catalog,
563 			String tableName,
564 			List<UniqueConstraintHolder> uniqueConstraints,
565 			String constraints,
566 			Table denormalizedSuperclassTable) {
567 		EntityTableObjectNameSource tableNameContext = new EntityTableObjectNameSource( tableName, name );
568 		EntityTableNamingStrategyHelper namingStrategyHelper = new EntityTableNamingStrategyHelper(
569 				persistentClass.getEntityName(),
570 				name
571 		);
572 		final Table table = TableBinder.buildAndFillTable(
573 				schema,
574 				catalog,
575 				tableNameContext,
576 				namingStrategyHelper,
577 				persistentClass.isAbstract(),
578 				uniqueConstraints,
579 				constraints,
580 				denormalizedSuperclassTable,
581 				mappings,
582 				this.subselect
583 		);
584 		final RowId rowId = annotatedClass.getAnnotation( RowId.class );
585 		if ( rowId != null ) {
586 			table.setRowId( rowId.value() );
587 		}
588 
589 		if ( persistentClass instanceof TableOwner ) {
590 			LOG.debugf( "Bind entity %s on table %s", persistentClass.getEntityName(), table.getName() );
591 			( (TableOwner) persistentClass ).setTable( table );
592 		}
593 		else {
594 			throw new AssertionFailure( "binding a table for a subclass" );
595 		}
596 	}

598 	public void finalSecondaryTableBinding(PropertyHolder propertyHolder) {
599 		/*
600 		 * Those operations has to be done after the id definition of the persistence class.
601 		 * ie after the properties parsing
602 		 */
603 		Iterator joins = secondaryTables.values().iterator();
604 		Iterator joinColumns = secondaryTableJoins.values().iterator();
605 
606 		while ( joins.hasNext() ) {
607 			Object uncastedColumn = joinColumns.next();
608 			Join join = (Join) joins.next();
609 			createPrimaryColumnsToSecondaryTable( uncastedColumn, propertyHolder, join );
610 		}
611 		mappings.addJoins( persistentClass, secondaryTables );
612 	}
613 
614 	private void createPrimaryColumnsToSecondaryTable(Object uncastedColumn, PropertyHolder propertyHolder, Join join) {
615 		Ejb3JoinColumn[] ejb3JoinColumns;
616 		PrimaryKeyJoinColumn[] pkColumnsAnn = null;
617 		JoinColumn[] joinColumnsAnn = null;
618 		if ( uncastedColumn instanceof PrimaryKeyJoinColumn[] ) {
619 			pkColumnsAnn = (PrimaryKeyJoinColumn[]) uncastedColumn;
620 		}
621 		if ( uncastedColumn instanceof JoinColumn[] ) {
622 			joinColumnsAnn = (JoinColumn[]) uncastedColumn;
623 		}
624 		if ( pkColumnsAnn == null && joinColumnsAnn == null ) {
625 			ejb3JoinColumns = new Ejb3JoinColumn[1];
626 			ejb3JoinColumns[0] = Ejb3JoinColumn.buildJoinColumn(
627 					null,
628 					null,
629 					persistentClass.getIdentifier(),
630 					secondaryTables,
631 					propertyHolder, mappings
632 			);
633 		}
634 		else {
635 			int nbrOfJoinColumns = pkColumnsAnn != null ?
636 					pkColumnsAnn.length :
637 					joinColumnsAnn.length;
638 			if ( nbrOfJoinColumns == 0 ) {
639 				ejb3JoinColumns = new Ejb3JoinColumn[1];
640 				ejb3JoinColumns[0] = Ejb3JoinColumn.buildJoinColumn(
641 						null,
642 						null,
643 						persistentClass.getIdentifier(),
644 						secondaryTables,
645 						propertyHolder, mappings
646 				);
647 			}
648 			else {
649 				ejb3JoinColumns = new Ejb3JoinColumn[nbrOfJoinColumns];
650 				if ( pkColumnsAnn != null ) {
651 					for (int colIndex = 0; colIndex < nbrOfJoinColumns; colIndex++) {
652 						ejb3JoinColumns[colIndex] = Ejb3JoinColumn.buildJoinColumn(
653 								pkColumnsAnn[colIndex],
654 								null,
655 								persistentClass.getIdentifier(),
656 								secondaryTables,
657 								propertyHolder, mappings
658 						);
659 					}
660 				}
661 				else {
662 					for (int colIndex = 0; colIndex < nbrOfJoinColumns; colIndex++) {
663 						ejb3JoinColumns[colIndex] = Ejb3JoinColumn.buildJoinColumn(
664 								null,
665 								joinColumnsAnn[colIndex],
666 								persistentClass.getIdentifier(),
667 								secondaryTables,
668 								propertyHolder, mappings
669 						);
670 					}
671 				}
672 			}
673 		}
674 
675 		for (Ejb3JoinColumn joinColumn : ejb3JoinColumns) {
676 			joinColumn.forceNotNull();
677 		}
678 		bindJoinToPersistentClass( join, ejb3JoinColumns, mappings );
679 	}

681 	private void bindJoinToPersistentClass(Join join, Ejb3JoinColumn[] ejb3JoinColumns, Mappings mappings) {
682 		SimpleValue key = new DependantValue( mappings, join.getTable(), persistentClass.getIdentifier() );
683 		join.setKey( key );
684 		setFKNameIfDefined( join );
685 		key.setCascadeDeleteEnabled( false );
686 		TableBinder.bindFk( persistentClass, null, ejb3JoinColumns, key, false, mappings );
687 		join.createPrimaryKey();
688 		join.createForeignKey();
689 		persistentClass.addJoin( join );
690 	}

699 	private org.hibernate.annotations.Table findMatchingComplimentTableAnnotation(Join join) {
700 		String tableName = join.getTable().getQuotedName();
701 		org.hibernate.annotations.Table table = annotatedClass.getAnnotation( org.hibernate.annotations.Table.class );
702 		org.hibernate.annotations.Table matchingTable = null;
703 		if ( table != null && tableName.equals( table.appliesTo() ) ) {
704 			matchingTable = table;
705 		}
706 		else {
707 			Tables tables = annotatedClass.getAnnotation( Tables.class );
708 			if ( tables != null ) {
709 				for (org.hibernate.annotations.Table current : tables.value()) {
710 					if ( tableName.equals( current.appliesTo() ) ) {
711 						matchingTable = current;
712 						break;
713 					}
714 				}
715 			}
716 		}
717 		return matchingTable;
718 	}

780 	private Join addJoin(
781 			SecondaryTable secondaryTable,
782 			JoinTable joinTable,
783 			PropertyHolder propertyHolder,
784 			boolean noDelayInPkColumnCreation) {
785 		// A non null propertyHolder means than we process the Pk creation without delay
786 		Join join = new Join();
787 		join.setPersistentClass( persistentClass );
788 
789 		final String schema;
790 		final String catalog;
791 		final SecondaryTableNameSource secondaryTableNameContext;
792 		final Object joinColumns;
793 		final List<UniqueConstraintHolder> uniqueConstraintHolders;
794 
795 		if ( secondaryTable != null ) {
796 			schema = secondaryTable.schema();
797 			catalog = secondaryTable.catalog();
798 			secondaryTableNameContext = new SecondaryTableNameSource( secondaryTable.name() );
799 			joinColumns = secondaryTable.pkJoinColumns();
800 			uniqueConstraintHolders = TableBinder.buildUniqueConstraintHolders( secondaryTable.uniqueConstraints() );
801 		}
802 		else if ( joinTable != null ) {
803 			schema = joinTable.schema();
804 			catalog = joinTable.catalog();
805 			secondaryTableNameContext = new SecondaryTableNameSource( joinTable.name() );
806 			joinColumns = joinTable.joinColumns();
807 			uniqueConstraintHolders = TableBinder.buildUniqueConstraintHolders( joinTable.uniqueConstraints() );
808 		}
809 		else {
810 			throw new AssertionFailure( "Both JoinTable and SecondaryTable are null" );
811 		}
812 
813 		final Table table = TableBinder.buildAndFillTable(
814 				schema,
815 				catalog,
816 				secondaryTableNameContext,
817 				SEC_TBL_NS_HELPER,
818 				false,
819 				uniqueConstraintHolders,
820 				null,
821 				null,
822 				mappings,
823 				null
824 		);
825 
826 		//no check constraints available on joins
827 		join.setTable( table );
828 
829 		//somehow keep joins() for later.
830 		//Has to do the work later because it needs persistentClass id!
831 		LOG.debugf( "Adding secondary table to entity %s -> %s", persistentClass.getEntityName(), join.getTable().getName() );
832 		org.hibernate.annotations.Table matchingTable = findMatchingComplimentTableAnnotation( join );
833 		if ( matchingTable != null ) {
834 			join.setSequentialSelect( FetchMode.JOIN != matchingTable.fetch() );
835 			join.setInverse( matchingTable.inverse() );
836 			join.setOptional( matchingTable.optional() );
837 			if ( !BinderHelper.isEmptyAnnotationValue( matchingTable.sqlInsert().sql() ) ) {
838 				join.setCustomSQLInsert( matchingTable.sqlInsert().sql().trim(),
839 						matchingTable.sqlInsert().callable(),
840 						ExecuteUpdateResultCheckStyle.fromExternalName(
841 								matchingTable.sqlInsert().check().toString().toLowerCase()
842 						)
843 				);
844 			}
845 			if ( !BinderHelper.isEmptyAnnotationValue( matchingTable.sqlUpdate().sql() ) ) {
846 				join.setCustomSQLUpdate( matchingTable.sqlUpdate().sql().trim(),
847 						matchingTable.sqlUpdate().callable(),
848 						ExecuteUpdateResultCheckStyle.fromExternalName(
849 								matchingTable.sqlUpdate().check().toString().toLowerCase()
850 						)
851 				);
852 			}
853 			if ( !BinderHelper.isEmptyAnnotationValue( matchingTable.sqlDelete().sql() ) ) {
854 				join.setCustomSQLDelete( matchingTable.sqlDelete().sql().trim(),
855 						matchingTable.sqlDelete().callable(),
856 						ExecuteUpdateResultCheckStyle.fromExternalName(
857 								matchingTable.sqlDelete().check().toString().toLowerCase()
858 						)
859 				);
860 			}
861 		}
862 		else {
863 			//default
864 			join.setSequentialSelect( false );
865 			join.setInverse( false );
866 			join.setOptional( true ); //perhaps not quite per-spec, but a Good Thing anyway
867 		}
868 
869 		if ( noDelayInPkColumnCreation ) {
870 			createPrimaryColumnsToSecondaryTable( joinColumns, propertyHolder, join );
871 		}
872 		else {
873 			secondaryTables.put( table.getQuotedName(), join );
874 			secondaryTableJoins.put( table.getQuotedName(), joinColumns );
875 		}
876 		return join;
877 	}

946 	public void processComplementaryTableDefinitions(org.hibernate.annotations.Table table) {
947 		//comment and index are processed here
948 		if ( table == null ) return;
949 		String appliedTable = table.appliesTo();
950 		Iterator tables = persistentClass.getTableClosureIterator();
951 		Table hibTable = null;
952 		while ( tables.hasNext() ) {
953 			Table pcTable = (Table) tables.next();
954 			if ( pcTable.getQuotedName().equals( appliedTable ) ) {
955 				//we are in the correct table to find columns
956 				hibTable = pcTable;
957 				break;
958 			}
959 			hibTable = null;
960 		}
961 		if ( hibTable == null ) {
962 			//maybe a join/secondary table
963 			for ( Join join : secondaryTables.values() ) {
964 				if ( join.getTable().getQuotedName().equals( appliedTable ) ) {
965 					hibTable = join.getTable();
966 					break;
967 				}
968 			}
969 		}
970 		if ( hibTable == null ) {
971 			throw new AnnotationException(
972 					"@org.hibernate.annotations.Table references an unknown table: " + appliedTable
973 			);
974 		}
975 		if ( !BinderHelper.isEmptyAnnotationValue( table.comment() ) ) hibTable.setComment( table.comment() );
976 		TableBinder.addIndexes( hibTable, table.indexes(), mappings );
977 	}
978 
979 	public void processComplementaryTableDefinitions(Tables tables) {
980 		if ( tables == null ) return;
981 		for (org.hibernate.annotations.Table table : tables.value()) {
982 			processComplementaryTableDefinitions( table );
983 		}
984 	}

1006 	public AccessType getExplicitAccessType(XAnnotatedElement element) {
1007 		AccessType accessType = null;
1008 
1009 		AccessType hibernateAccessType = null;
1010 		AccessType jpaAccessType = null;
1011 
1012 		org.hibernate.annotations.AccessType accessTypeAnnotation = element.getAnnotation( org.hibernate.annotations.AccessType.class );
1013 		if ( accessTypeAnnotation != null ) {
1014 			hibernateAccessType = AccessType.getAccessStrategy( accessTypeAnnotation.value() );
1015 		}
1016 
1017 		Access access = element.getAnnotation( Access.class );
1018 		if ( access != null ) {
1019 			jpaAccessType = AccessType.getAccessStrategy( access.value() );
1020 		}
1021 
1022 		if ( hibernateAccessType != null && jpaAccessType != null && hibernateAccessType != jpaAccessType ) {
1023 			throw new MappingException(
1024 					"Found @Access and @AccessType with conflicting values on a property in class " + annotatedClass.toString()
1025 			);
1026 		}
1027 
1028 		if ( hibernateAccessType != null ) {
1029 			accessType = hibernateAccessType;
1030 		}
1031 		else if ( jpaAccessType != null ) {
1032 			accessType = jpaAccessType;
1033 		}
1034 
1035 		return accessType;
1036 	}
