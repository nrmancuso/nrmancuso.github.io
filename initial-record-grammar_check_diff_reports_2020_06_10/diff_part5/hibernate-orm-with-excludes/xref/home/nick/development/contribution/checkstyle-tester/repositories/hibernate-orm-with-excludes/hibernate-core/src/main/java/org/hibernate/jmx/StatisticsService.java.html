21  /**
22   * JMX service for Hibernate statistics<br>
23   * <br>
24   * Register this MBean in your JMX server for a specific session factory
25   * <pre>
26   * //build the ObjectName you want
27   * Hashtable tb = new Hashtable();
28   * tb.put("type", "statistics");
29   * tb.put("sessionFactory", "myFinancialApp");
30   * ObjectName on = new ObjectName("hibernate", tb);
31   * StatisticsService stats = new StatisticsService();
32   * stats.setSessionFactory(sessionFactory);
33   * server.registerMBean(stats, on);
34   * </pre>
35   * And call the MBean the way you want<br>
36   * <br>
37   * Register this MBean in your JMX server with no specific session factory
38   * <pre>
39   * //build the ObjectName you want
40   * Hashtable tb = new Hashtable();
41   * tb.put("type", "statistics");
42   * tb.put("sessionFactory", "myFinancialApp");
43   * ObjectName on = new ObjectName("hibernate", tb);
44   * StatisticsService stats = new StatisticsService();
45   * server.registerMBean(stats, on);
46   * </pre>
47   * And call the MBean by providing the <code>SessionFactoryJNDIName</code> first.
48   * Then the session factory will be retrieved from JNDI and the statistics
49   * loaded.
50   *
51   * @author Emmanuel Bernard
52   * @deprecated See <a href="http://opensource.atlassian.com/projects/hibernate/browse/HHH-6190">HHH-6190</a> for details
53   */

64  	/**
65  	 * @see StatisticsServiceMBean#setSessionFactoryJNDIName(java.lang.String)
66  	 */
67  	public void setSessionFactoryJNDIName(String sfJNDIName) {
68  		this.sfJNDIName = sfJNDIName;
69  		try {
70  			final SessionFactory sessionFactory;
71  			final Object jndiValue = new InitialContext().lookup( sfJNDIName );
72  			if ( jndiValue instanceof Reference ) {
73  				final String uuid = (String) ( (Reference) jndiValue ).get( 0 ).getContent();
74  				sessionFactory = SessionFactoryRegistry.INSTANCE.getSessionFactory( uuid );
75  			}
76  			else {
77  				sessionFactory = (SessionFactory) jndiValue;
78  			}
79  			setSessionFactory( sessionFactory );
80  		}
81  		catch (NameNotFoundException e) {
82  			LOG.noSessionFactoryWithJndiName( sfJNDIName, e );
83  			setSessionFactory(null);
84  		}
85  		catch (NamingException e) {
86  			LOG.unableToAccessSessionFactory( sfJNDIName, e );
87  			setSessionFactory(null);
88  		}
89  		catch (ClassCastException e) {
90  			LOG.jndiNameDoesNotHandleSessionFactoryReference( sfJNDIName, e );
91  			setSessionFactory(null);
92  		}
93  	}
