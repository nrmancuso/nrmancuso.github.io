24  package org.hibernate.envers.tools.graph;
25  import java.util.ArrayList;
26  import java.util.List;
27  
28  /**
29   * A graph vertex - stores its representation, neighbours, start and end time in (D|B)FS.
30   * @author Adam Warski (adam at warski dot org)
31   */
32  public class Vertex<R> {
33      private final R representation;
34      private final List<Vertex<R>> neighbours;
35      
36      private int startTime;
37      private int endTime;
38  
39      public Vertex(R representation) {
40          this.representation = representation;
41          this.neighbours = new ArrayList<Vertex<R>>();
42          this.startTime = 0;
43          this.endTime = 0;
44      }
45  
46      public R getRepresentation() {
47          return representation;
48      }
49  
50      public List<Vertex<R>> getNeighbours() {
51          return neighbours;
52      }
53  
54      public void addNeighbour(Vertex<R> n) {
55          neighbours.add(n);
56      }
57  
58      public int getStartTime() {
59          return startTime;
60      }
61  
62      public void setStartTime(int startTime) {
63          this.startTime = startTime;
64      }
65  
66      public int getEndTime() {
67          return endTime;
68      }
69  
70      public void setEndTime(int endTime) {
71          this.endTime = endTime;
72      }
73  }
