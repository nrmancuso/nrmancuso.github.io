97  	public boolean hasFormula() {
98  		Iterator iter = getColumnIterator();
99  		while ( iter.hasNext() ) {
100 			Object o = iter.next();
101 			if (o instanceof Formula) return true;
102 		}
103 		return false;
104 	}

127 	public void createForeignKeyOfEntity(String entityName) {
128 		if ( !hasFormula() && !"none".equals(getForeignKeyName())) {

134 	public IdentifierGenerator createIdentifierGenerator(
135 			IdentifierGeneratorFactory identifierGeneratorFactory,
136 			Dialect dialect, 
137 			String defaultCatalog, 
138 			String defaultSchema, 
139 			RootClass rootClass) throws MappingException {
140 		
141 		Properties params = new Properties();
142 		
143 		//if the hibernate-mapping did not specify a schema/catalog, use the defaults
144 		//specified by properties - but note that if the schema/catalog were specified
145 		//in hibernate-mapping, or as params, they will already be initialized and
146 		//will override the values set here (they are in identifierGeneratorProperties)
147 		if ( defaultSchema!=null ) {
148 			params.setProperty(PersistentIdentifierGenerator.SCHEMA, defaultSchema);
149 		}
150 		if ( defaultCatalog!=null ) {
151 			params.setProperty(PersistentIdentifierGenerator.CATALOG, defaultCatalog);
152 		}
153 		
154 		//pass the entity-name, if not a collection-id
155 		if (rootClass!=null) {
156 			params.setProperty( IdentifierGenerator.ENTITY_NAME, rootClass.getEntityName() );
157 			params.setProperty( IdentifierGenerator.JPA_ENTITY_NAME, rootClass.getJpaEntityName() );
158 		}
159 		
160 		//init the table here instead of earlier, so that we can get a quoted table name
161 		//TODO: would it be better to simply pass the qualified table name, instead of
162 		//      splitting it up into schema/catalog/table names
163 		String tableName = getTable().getQuotedName(dialect);
164 		params.setProperty( PersistentIdentifierGenerator.TABLE, tableName );
165 		
166 		//pass the column name (a generated id almost always has a single column)
167 		String columnName = ( (Column) getColumnIterator().next() ).getQuotedName(dialect);
168 		params.setProperty( PersistentIdentifierGenerator.PK, columnName );
169 		
170 		if (rootClass!=null) {
171 			StringBuilder tables = new StringBuilder();
172 			Iterator iter = rootClass.getIdentityTables().iterator();
173 			while ( iter.hasNext() ) {
174 				Table table= (Table) iter.next();
175 				tables.append( table.getQuotedName(dialect) );
176 				if ( iter.hasNext() ) tables.append(", ");
177 			}
178 			params.setProperty( PersistentIdentifierGenerator.TABLES, tables.toString() );
179 		}
180 		else {
181 			params.setProperty( PersistentIdentifierGenerator.TABLES, tableName );
182 		}
183 
184 		if (identifierGeneratorProperties!=null) {
185 			params.putAll(identifierGeneratorProperties);
186 		}
187 
188 		// TODO : we should pass along all settings once "config lifecycle" is hashed out...
189 		params.put(
190 				Environment.PREFER_POOLED_VALUES_LO,
191 				mappings.getConfigurationProperties().getProperty( Environment.PREFER_POOLED_VALUES_LO, "false" )
192 		);
193 
194 		identifierGeneratorFactory.setDialect( dialect );
195 		return identifierGeneratorFactory.createIdentifierGenerator( identifierGeneratorStrategy, getType(), params );
196 		
197 	}

274 	public boolean isNullable() {
275 		if ( hasFormula() ) return true;
276 		boolean nullable = true;
277 		Iterator iter = getColumnIterator();
278 		while ( iter.hasNext() ) {
279 			if ( !( (Column) iter.next() ).isNullable() ) {
280 				nullable = false;
281 				return nullable; //shortcut
282 			}
283 		}
284 		return nullable;
285 	}

295 	public Type getType() throws MappingException {
296 		if ( typeName == null ) {
297 			throw new MappingException( "No type name" );
298 		}
299 		if ( typeParameters != null
300 				&& Boolean.valueOf( typeParameters.getProperty( DynamicParameterizedType.IS_DYNAMIC ) )
301 				&& typeParameters.get( DynamicParameterizedType.PARAMETER_TYPE ) == null ) {
302 			createParameterImpl();
303 		}
304 
305 		Type result = mappings.getTypeResolver().heuristicType( typeName, typeParameters );
306 		if ( result == null ) {
307 			String msg = "Could not determine type for: " + typeName;
308 			if ( table != null ) {
309 				msg += ", at table: " + table.getName();
310 			}
311 			if ( columns != null && columns.size() > 0 ) {
312 				msg += ", for columns: " + columns;
313 			}
314 			throw new MappingException( msg );
315 		}
316 		return result;
317 	}

320 		if (typeName==null) {
321 			if (className==null) {
322 				throw new MappingException("you must specify types for a dynamic entity: " + propertyName);

324 			typeName = ReflectHelper.reflectedPropertyClass(className, propertyName).getName();

349 	public boolean[] getColumnInsertability() {
350 		boolean[] result = new boolean[ getColumnSpan() ];
351 		int i = 0;
352 		Iterator iter = getColumnIterator();
353 		while ( iter.hasNext() ) {
354 			Selectable s = (Selectable) iter.next();
355 			result[i++] = !s.isFormula();
356 		}
357 		return result;
358 	}

360 	public boolean[] getColumnUpdateability() {
361 		return getColumnInsertability();
362 	}

364 	private void createParameterImpl() {
365 		try {
366 			String[] columnsNames = new String[columns.size()];
367 			for ( int i = 0; i < columns.size(); i++ ) {
368 				Object column = columns.get(i);
369 				if (column instanceof Column){
370 					columnsNames[i] = ((Column) column).getName();
371 				}
372 			}
373 
374 			final XProperty xProperty = (XProperty) typeParameters.get( DynamicParameterizedType.XPROPERTY );
375 			// todo : not sure this works for handling @MapKeyEnumerated
376 			final Annotation[] annotations = xProperty == null
377 					? null
378 					: xProperty.getAnnotations();
379 
380 			typeParameters.put(
381 					DynamicParameterizedType.PARAMETER_TYPE,
382 					new ParameterTypeImpl(
383 							ReflectHelper.classForName(
384 									typeParameters.getProperty( DynamicParameterizedType.RETURNED_CLASS )
385 							),
386 							annotations,
387 							table.getCatalog(),
388 							table.getSchema(),
389 							table.getName(),
390 							Boolean.valueOf( typeParameters.getProperty( DynamicParameterizedType.IS_PRIMARY_KEY ) ),
391 							columnsNames
392 					)
393 			);
394 		}
395 		catch ( ClassNotFoundException cnfe ) {
396 			throw new MappingException( "Could not create DynamicParameterizedType for type: " + typeName, cnfe );
397 		}
398 	}

400 	private final class ParameterTypeImpl implements DynamicParameterizedType.ParameterType {
401 
402 		private final Class returnedClass;
403 		private final Annotation[] annotationsMethod;
404 		private final String catalog;
405 		private final String schema;
406 		private final String table;
407 		private final boolean primaryKey;
408 		private final String[] columns;
409 
410 		private ParameterTypeImpl(Class returnedClass, Annotation[] annotationsMethod, String catalog, String schema,
411 				String table, boolean primaryKey, String[] columns) {
412 			this.returnedClass = returnedClass;
413 			this.annotationsMethod = annotationsMethod;
414 			this.catalog = catalog;
415 			this.schema = schema;
416 			this.table = table;
417 			this.primaryKey = primaryKey;
418 			this.columns = columns;
419 		}

424 		}
