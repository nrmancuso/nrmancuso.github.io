209     @Override
210     public void start(BundleContext context) throws Exception {
211         super.start(context);
212 
213         Version.registerApplication("SpotBugs-Eclipse", Version.VERSION_STRING);
214 
215          // configure debugging
216         configurePluginDebugOptions();
217 
218         // initialize resource strings
219         try {
220             resourceBundle = ResourceBundle.getBundle("de.tobject.findbugs.messages"); //this is correct //$NON-NLS-1$
221         } catch (MissingResourceException x) {
222             resourceBundle = null;
223         }
224 
225         if (System.getProperty("findbugs.home") == null) {
226             // TODO workaround for findbugs home property
227             // - see de.tobject.findbugs.builder.FindBugsWorker.work() too
228             String findBugsHome = getFindBugsEnginePluginLocation();
229             if (DEBUG) {
230                 logInfo("SpotBugs home is: " + findBugsHome);
231             }
232             System.setProperty("findbugs.home", findBugsHome);
233         }
234         if (System.getProperty("findbugs.cloud.default") == null) {
235             // TODO workaround for findbugs default cloud property
236             // - see edu.umd.cs.findbugs.cloud.CloudFactory and messages.xml
237             String defCloud = DEFAULT_CLOUD_ID;
238             if (DEBUG) {
239                 logInfo("Using default cloud: " + defCloud);
240             }
241             System.setProperty("findbugs.cloud.default", defCloud);
242         }
243 
244         /** Don't load main classes */
245         FindBugs.setNoMains();
246 
247         // enable source searching for "fall" (ignore case) in switch statements
248         SystemProperties.setProperty("findbugs.sf.comment", "true");
249 
250         // Register our save participant
251         FindbugsSaveParticipant saveParticipant = new FindbugsSaveParticipant();
252         ResourcesPlugin.getWorkspace().addSaveParticipant(PLUGIN_ID, saveParticipant);
253     }

255     public static void dumpClassLoader(Class<?> c) {
256         System.out.printf("Class loaders for %s:%n", c.getName());
257         ClassLoader loader = c.getClassLoader();
258         while (loader != null) {
259             System.out.printf("  %s %s%n", loader.toString(),  loader.getClass().getSimpleName());
260             loader = loader.getParent();
261         }

265     /**
266      * The complexity of the code below is partly caused by the fact that we
267      * might have multiple ways to install and/or enable custom plugins. There
268      * are plugins discovered by FB itself, plugins contributed to Eclipse and
269      * plugins added by user manually via properties. Plugins can be disabled
270      * via code or properties. The code below is still work in progress, see
271      * also {@link DetectorProvider#getPluginElements(UserPreferences)}.
272      *
273      * @param force
274      *            true if we MUST set plugins even if the given list is empty
275      */
276     public static synchronized void applyCustomDetectors(boolean force) {
277         if(customDetectorsInitialized && !force) {
278             return;
279         }
280         customDetectorsInitialized = true;
281         DetectorValidator validator = new DetectorValidator();
282         final SortedSet<String> detectorPaths = new TreeSet<>();
283         SortedMap<String, String> contributedDetectors = DetectorsExtensionHelper.getContributedDetectors();
284         UserPreferences corePreferences = getCorePreferences(null, force);
285         detectorPaths.addAll(corePreferences.getCustomPlugins(true));
286         if(DEBUG) {
287             dumpClassLoader(FindbugsPlugin.class);
288             dumpClassLoader(Plugin.class);
289             System.out.println("applyCustomDetectors - going to add " + detectorPaths.size() + " plugin urls...");
290             for (String url : detectorPaths) {
291                 System.out.println("\t" + url);
292             }
293         }
294 
295         // disable custom plugins configured via properties, if they are already loaded
296         Set<String> disabledPlugins = corePreferences.getCustomPlugins(false);
297         Map<URI, Plugin> allPlugins = Plugin.getAllPluginsMap();
298         for (Entry<URI, Plugin> entry : allPlugins.entrySet()) {
299             Plugin fbPlugin = entry.getValue();
300             String pluginId = fbPlugin.getPluginId();
301             // ignore all custom plugins with the same plugin id as already loaded
302             if(contributedDetectors.containsKey(pluginId)) {
303                 contributedDetectors.remove(pluginId);
304                 detectorPaths.remove(pluginId);
305             }
306 
307             if (fbPlugin.isCorePlugin() || fbPlugin.isInitialPlugin()) {
308                 continue;
309             }
310             if (disabledPlugins.contains(entry.getKey().getPath())
311                     || disabledPlugins.contains(pluginId)) {
312                 fbPlugin.setGloballyEnabled(false);
313                 Plugin.removeCustomPlugin(fbPlugin);
314                 if (DEBUG) {
315                     System.out.println("Removed plugin: " + fbPlugin + " loaded from " + entry.getKey());
316                 }
317             }
318         }
319 
320         HashSet<Plugin> enabled = new HashSet<>();
321 
322         // adding FindBugs *Eclipse* plugins, key plugin id, value is path
323         for (Entry<String, String> entry : contributedDetectors.entrySet()) {
324             String pluginId = entry.getKey();
325             String pluginPath = entry.getValue();
326             URI uri = new File(pluginPath).toURI();
327             if (disabledPlugins.contains(pluginId)
328                     || disabledPlugins.contains(pluginPath)
329                     || allPlugins.containsKey(uri)) {
330                 continue;
331             }
332             addCustomPlugin(enabled, uri);
333         }
334 
335         // adding custom plugins configured via properties, but only if they are not loaded yet
336         for (String path : detectorPaths) {
337             // this is plugin id, so we can't use it as URL
338             if(new Path(path).segmentCount() == 1) {
339                 continue;
340             }
341             path = FindBugsWorker.getFilterPath(path, null).toOSString();
342             URI uri = new File(path).toURI();
343             if(allPlugins.containsKey(uri)) {
344                 continue;
345             }
346             ValidationStatus status = validator.validate(path);
347             if (status.isOK()) {
348                 addCustomPlugin(enabled, uri);
349             } else {
350                 getDefault().getLog().log(status);
351             }
352         }
353 
354         if (DEBUG) {
355             System.out.println("applyCustomDetectors - there was " + detectorPaths.size()
356                     + " extra FB plugin urls with " + enabled.size()
357                     + " valid FB plugins and " + allPlugins.size()
358                     + " total plugins registered by FB.");
359             for (Entry<URI, Plugin> entry : allPlugins.entrySet()) {
360                 Plugin fbPlugin = entry.getValue();
361                 if (fbPlugin.isGloballyEnabled()) {
362                     System.out.println("IS  enabled:\t" + fbPlugin.getPluginId());
363                 } else {
364                     System.out.println("NOT enabled:\t" + fbPlugin.getPluginId());
365                 }
366             }
367         }
368     }

370     protected static void addCustomPlugin(HashSet<Plugin> enabled, URI uri) {
371         try {
372             // bug 3117769 - we must provide our own classloader
373             // to allow third-party plugins extend the classpath via
374             // "Buddy" classloading
375             // see also: Eclipse-BuddyPolicy attribute in MANIFEST.MF
376             Plugin fbPlugin = Plugin.addCustomPlugin(uri, FindbugsPlugin.class.getClassLoader());
377             if(fbPlugin != null) {
378                 // TODO line below required to enable this *optional* plugin
379                 // but it should be taken by FB core from the findbugs.xml,
380                 // which currently only works for *core* plugins only
381                 fbPlugin.setGloballyEnabled(true);
382                 enabled.add(fbPlugin);
383             }
384         } catch (PluginException e) {
385             getDefault().logException(e, "Failed to load plugin for custom detector: " + uri);
386         } catch (DuplicatePluginIdException e) {
387             getDefault().logException(e, e.getPluginId() + " already loaded from " + e.getPreviouslyLoadedFrom()
388                     + ", ignoring: " + uri);
389         }
390     }

472     public void configurePluginDebugOptions() {
473         if (isDebugging()) {
474             // debugging for the plugin itself
475             String option = Platform.getDebugOption(PLUGIN_DEBUG);
476             FindbugsPlugin.DEBUG = Boolean.valueOf(option).booleanValue();
477 
478             // debugging for the builder and friends
479             option = Platform.getDebugOption(BUILDER_DEBUG);
480             FindBugsBuilder.DEBUG = Boolean.valueOf(option).booleanValue();
481             FindBugsWorker.DEBUG = FindBugsBuilder.DEBUG;
482 
483             // debugging for the nature
484             option = Platform.getDebugOption(NATURE_DEBUG);
485             FindBugsNature.DEBUG = Boolean.valueOf(option).booleanValue();
486 
487             // debugging for the reporter
488             option = Platform.getDebugOption(REPORTER_DEBUG);
489             Reporter.DEBUG = Boolean.valueOf(option).booleanValue();
490 
491             // debugging for the content provider
492             option = Platform.getDebugOption(CONTENT_DEBUG);
493             BugContentProvider.DEBUG = Boolean.valueOf(option).booleanValue();
494 
495             option = Platform.getDebugOption(PROFILER_DEBUG);
496             if (Boolean.valueOf(option).booleanValue()) {
497                 System.setProperty("profiler.report", "true");
498             }
499         }
500     }

502     /**
503      * Find the filesystem path of the FindBugs plugin directory.
504      *
505      * @return the filesystem path of the FindBugs plugin directory, or null if
506      *         the FindBugs plugin directory cannot be found
507      */
508     public static String getFindBugsEnginePluginLocation() {
509         // findbugs.home should be set to the directory the plugin is
510         // installed in.
511         URL u = plugin.getBundle().getEntry("/");
512         try {
513             URL bundleRoot = FileLocator.resolve(u);
514             String path = bundleRoot.getPath();
515             if (FindBugsBuilder.DEBUG) {
516                 System.out.println("Pluginpath: " + path); //$NON-NLS-1$
517             }
518             if (path.endsWith("/eclipsePlugin/")) {
519                 File f = new File(path);
520                 f = f.getParentFile();
521                 f = new File(f, "findbugs");
522                 path = f.getPath() + "/";
523             }
524 
525             return path;
526         } catch (IOException e) {
527             FindbugsPlugin.getDefault().logException(e, "IO Exception locating engine plugin");
528         }
529         return null;
530     }

582     public void logMessage(int severity, String message, Throwable e) {
583         if (DEBUG) {
584             String what = (severity == IStatus.ERROR) ? (e != null ? "Exception" : "Error") : "Warning";
585             System.out.println(what + " in SpotBugs plugin: " + message);
586             if (e != null) {
587                 e.printStackTrace();
588             }
589         }
590         IStatus status = createStatus(severity, message, e);
591         getLog().log(status);
592     }

814     public static boolean isProjectSettingsEnabled(IProject project) {
815         // fast path: read from session, if available
816         Boolean enabled;
817         try {
818             enabled = (Boolean) project.getSessionProperty(SESSION_PROPERTY_SETTINGS_ON);
819         } catch (CoreException e) {
820             enabled = null;
821         }
822         if (enabled != null) {
823             return enabled.booleanValue();
824         }
825 
826         // legacy support: before 1.3.8, there was ONLY project preferences in
827         // .fbprefs
828         // so check if the file is there...
829         IFile file = getUserPreferencesFile(project);
830         boolean projectPropsEnabled = file.isAccessible();
831         if (projectPropsEnabled) {
832             ScopedPreferenceStore store = new ScopedPreferenceStore(new ProjectScope(project), PLUGIN_ID);
833             // so if the file is there, we can check if after 1.3.8 the flag is
834             // set
835             // to use workspace properties instead
836             projectPropsEnabled = !store.contains(FindBugsConstants.PROJECT_PROPS_DISABLED)
837                     || !store.getBoolean(FindBugsConstants.PROJECT_PROPS_DISABLED);
838         }
839         // remember in the session to speedup access, don't touch the store
840         setProjectSettingsEnabled(project, null, projectPropsEnabled);
841         return projectPropsEnabled;
842     }

953     /**
954      * Save current UserPreferences for given project or workspace.
955      *
956      * @param project
957      *            the project or null for workspace
958      * @throws CoreException
959      */
960     public static void saveUserPreferences(IProject project, final UserPreferences userPrefs) throws CoreException {
961 
962         FileOutput userPrefsOutput = new FileOutput() {
963             @Override
964             public void writeFile(OutputStream os) throws IOException {
965                 userPrefs.write(os);
966             }
967 
968             @Override
969             public String getTaskDescription() {
970                 return "writing user preferences";
971             }
972         };
973 
974         if (project != null) {
975             // Make the new user preferences current for the project
976             project.setSessionProperty(SESSION_PROPERTY_USERPREFS, userPrefs);
977             IFile userPrefsFile = getUserPreferencesFile(project);
978             ensureReadWrite(userPrefsFile);
979             IO.writeFile(userPrefsFile, userPrefsOutput, null);
980             if (project.getFile(DEPRECATED_PREFS_PATH).equals(userPrefsFile)) {
981                 String message = "Found old style FindBugs preferences for project '" + project.getName()
982                         + "'. This preferences are not at the default location: '" + DEFAULT_PREFS_PATH + "'." + " Please move '"
983                         + DEPRECATED_PREFS_PATH + "' to '" + DEFAULT_PREFS_PATH + "'.";
984                 getDefault().logWarning(message);
985             }
986         } else {
987             // write the workspace preferences to the eclipse preference store
988             ByteArrayOutputStream bos = new ByteArrayOutputStream(10000);
989             try {
990                 userPrefs.write(bos);
991             } catch (IOException e) {
992                 getDefault().logException(e, "Failed to write user preferences");
993                 return;
994             }
995             Properties props = new Properties();
996             try {
997                 props.load(new ByteArrayInputStream(bos.toByteArray()));
998             } catch (IOException e) {
999                 getDefault().logException(e, "Failed to save user preferences");
1000                 return;
1001             }
1002             IPreferenceStore store = getDefault().getPreferenceStore();
1003             // Reset any existing custom group entries
1004             resetStore(store, UserPreferences.KEY_PLUGIN);
1005             resetStore(store, UserPreferences.KEY_EXCLUDE_BUGS);
1006             resetStore(store, UserPreferences.KEY_EXCLUDE_FILTER);
1007             resetStore(store, UserPreferences.KEY_INCLUDE_FILTER);
1008             for (Entry<Object, Object> entry : props.entrySet()) {
1009                 store.putValue((String) entry.getKey(), (String) entry.getValue());
1010             }
1011             if(store instanceof IPersistentPreferenceStore){
1012                 IPersistentPreferenceStore store2 = (IPersistentPreferenceStore) store;
1013                 try {
1014                     store2.save();
1015                 } catch (IOException e) {
1016                     getDefault().logException(e, "Failed to save user preferences");
1017                 }
1018             }
1019         }
1020     }

1022     /**
1023      * Removes all consequent enumerated keys from given store staring with given prefix
1024      */
1025     private static void resetStore(IPreferenceStore store, String prefix) {
1026         int start = 0;
1027         // 99 is paranoia.
1028         while(start < 99){
1029             String name = prefix + start;
1030             if(store.contains(name)){
1031                 store.setToDefault(name);
1032             } else {
1033                 break;
1034             }
1035             start ++;
1036         }
1037     }

1060      * Read UserPreferences for project from the file in the project directory.
1061      * Returns null if the preferences have not been saved to a file, or if
1062      * there is an error reading the preferences file.
1063      *
1064      * @param project
1065      *            the project to get the UserPreferences for
1066      * @return the UserPreferences, or null if the UserPreferences file could
1067      *         not be read
1068      * @throws CoreException
1069      */
1070     private static UserPreferences readUserPreferences(IProject project) throws CoreException {
1071         IFile userPrefsFile = getUserPreferencesFile(project);
1072         if (!userPrefsFile.exists()) {
1073             return null;
1074         }
1075         try {
1076             // force is preventing us for out-of-sync exception if file was
1077             // changed externally
1078             InputStream in = userPrefsFile.getContents(true);
1079             UserPreferences userPrefs = FindBugsPreferenceInitializer.createDefaultUserPreferences();
1080             userPrefs.read(in);
1081             return userPrefs;
1082         } catch (IOException e) {
1083             FindbugsPlugin.getDefault().logException(e, "Could not read user preferences for project");
1084             return null;
1085         }

1088     public static void showMarker(IMarker marker, String viewId, IWorkbenchPart source) {
1089         IWorkbenchPage page = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage();
1090         IViewPart view = page.findView(viewId);
1091         if (!page.isPartVisible(view)) {
1092             try {
1093                 view = page.showView(viewId);
1094             } catch (PartInitException e) {
1095                 FindbugsPlugin.getDefault().logException(e, "Could not open view: " + viewId);
1096                 return;
1097             }
1098         }
1099         if (view instanceof IMarkerSelectionHandler) {
1100             IMarkerSelectionHandler handler = (IMarkerSelectionHandler) view;
1101             handler.markerSelected(source, marker);
1102         } else if (DETAILS_VIEW_ID.equals(viewId) && view instanceof ISelectionListener) {
1103             ISelectionListener listener = (ISelectionListener) view;
1104             listener.selectionChanged(source, new StructuredSelection(marker));
1105         }
