54      private static final String EQUALS_NAME = "equals";
55  
56      private static final String EQUALS_SIGNATURE = "(Ljava/lang/Object;)Z";
57  
58      private static final String STATIC_EQUALS_SIGNATURE = "(Ljava/lang/Object;Ljava/lang/Object;)Z";

74      public OverridingEqualsNotSymmetrical(BugReporter bugReporter) {
75          this.bugReporter = bugReporter;
76          this.bugAccumulator = new BugAccumulator(bugReporter);
77          equalsKindSummary = AnalysisContext.currentAnalysisContext().getEqualsKindSummary();
78      }

80      @Override
81      public void visit(Code obj) {
82          if (EQUALS_NAME.equals(getMethodName()) && !getMethod().isStatic() && getMethod().isPublic()
83                  && EQUALS_SIGNATURE.equals(getMethodSig())) {
84              sawCheckedCast = sawSuperEquals = sawInstanceOf = sawGetClass = sawReturnSuper = sawCompare = sawReturnNonSuper = prevWasSuperEquals = sawGoodEqualsClass = sawBadEqualsClass = dangerDanger = sawInstanceOfSupertype = alwaysTrue = alwaysFalse = sawStaticDelegate = sawEqualsBuilder = false;
85              sawInitialIdentityCheck = obj.getCode().length == 11 || obj.getCode().length == 9;
86              equalsCalls = 0;
87              super.visit(obj);
88              EqualsKindSummary.KindOfEquals kind = EqualsKindSummary.KindOfEquals.UNKNOWN;
89              if (alwaysTrue) {
90                  kind = EqualsKindSummary.KindOfEquals.ALWAYS_TRUE;
91              } else if (alwaysFalse) {
92                  kind = EqualsKindSummary.KindOfEquals.ALWAYS_FALSE;
93              } else if (sawReturnSuper && !sawReturnNonSuper) {
94                  kind = EqualsKindSummary.KindOfEquals.RETURNS_SUPER;
95              } else if (sawSuperEquals) {
96                  kind = EqualsKindSummary.KindOfEquals.INVOKES_SUPER;
97              } else if (sawInstanceOfSupertype) {
98                  kind = EqualsKindSummary.KindOfEquals.INSTANCE_OF_SUPERCLASS_EQUALS;
99              } else if (sawInstanceOf) {
100                 kind = getThisClass().isAbstract() ? EqualsKindSummary.KindOfEquals.ABSTRACT_INSTANCE_OF
101                         : EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS;
102             } else if (sawGetClass && sawGoodEqualsClass) {
103                 kind = getThisClass().isAbstract() ? EqualsKindSummary.KindOfEquals.ABSTRACT_GETCLASS_GOOD_EQUALS
104                         : EqualsKindSummary.KindOfEquals.GETCLASS_GOOD_EQUALS;
105             } else if (sawGetClass && sawBadEqualsClass) {
106                 kind = EqualsKindSummary.KindOfEquals.GETCLASS_BAD_EQUALS;
107             } else if (equalsCalls == 1 || sawStaticDelegate || sawEqualsBuilder) {
108                 kind = EqualsKindSummary.KindOfEquals.DELEGATE_EQUALS;
109             } else if (sawInitialIdentityCheck) {
110                 kind = EqualsKindSummary.KindOfEquals.TRIVIAL_EQUALS;
111             } else if (sawCheckedCast) {
112                 kind = EqualsKindSummary.KindOfEquals.CHECKED_CAST_EQUALS;
113             } else if (sawCompare) {
114                 kind = EqualsKindSummary.KindOfEquals.COMPARE_EQUALS;
115             } else {
116                 if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
117                     bugReporter
118                     .reportBug(new BugInstance(this, "EQ_UNUSUAL", Priorities.NORMAL_PRIORITY).addClassAndMethod(this));
119                 }
120             }
121             ClassAnnotation classAnnotation = new ClassAnnotation(getDottedClassName());
122             equalsKindSummary.put(classAnnotation, kind);
123 
124             count(kind);
125             if (kind == EqualsKindSummary.KindOfEquals.GETCLASS_GOOD_EQUALS
126                     || kind == EqualsKindSummary.KindOfEquals.ABSTRACT_GETCLASS_GOOD_EQUALS
127                     || kind == EqualsKindSummary.KindOfEquals.GETCLASS_BAD_EQUALS) {
128 
129                 ClassDescriptor classDescriptor = getClassDescriptor();
130                 try {
131                     Set<ClassDescriptor> subtypes = AnalysisContext.currentAnalysisContext().getSubtypes2()
132                             .getSubtypes(classDescriptor);
133                     if (subtypes.size() > 1) {
134                         classesWithGetClassBasedEquals.put(classDescriptor, subtypes);
135                     }
136                 } catch (ClassNotFoundException e) {
137                     assert true;
138                 }
139 
140             }
141             if (kind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS
142                     || kind == EqualsKindSummary.KindOfEquals.ABSTRACT_INSTANCE_OF) {
143 
144                 ClassDescriptor classDescriptor = getClassDescriptor();
145                 try {
146                     Set<ClassDescriptor> subtypes = AnalysisContext.currentAnalysisContext().getSubtypes2()
147                             .getSubtypes(classDescriptor);
148                     if (subtypes.size() > 1) {
149                         classesWithInstanceOfBasedEquals.put(classDescriptor, subtypes);
150                     }
151                 } catch (ClassNotFoundException e) {
152                     assert true;
153                 }
154 
155             }
156 
157             String superClassName = getSuperclassName().replace('/', '.');
158             if (!"java.lang.Object".equals(superClassName)) {
159                 parentMap.put(classAnnotation, new ClassAnnotation(superClassName));
160             }
161             equalsMethod.put(classAnnotation, getMethodDescriptor());
162 
163         }
164         bugAccumulator.reportAccumulatedBugs();
165     }

195     private final EnumMap<EqualsKindSummary.KindOfEquals, Integer> count = new EnumMap<>(
196             EqualsKindSummary.KindOfEquals.class);
197 
198     private void count(EqualsKindSummary.KindOfEquals k) {
199         Integer v = count.get(k);
200         if (v == null) {
201             count.put(k, 1);
202         } else {
203             count.put(k, v + 1);
204         }
205     }

207     @Override
208     public void sawOpcode(int seen) {
209         if (getPC() == 2 && seen != Const.IF_ACMPEQ && seen != Const.IF_ACMPNE) {
210             // System.out.println(OPCODE_NAMES[seen]);
211             sawInitialIdentityCheck = false;
212         }
213         if (getPC() == 2
214                 && seen == Const.INVOKESTATIC
215                 && getCode().getCode().length == 6
216                 && (getPrevOpcode(1) == Const.ALOAD_0 && getPrevOpcode(2) == Const.ALOAD_1 || getPrevOpcode(1) == Const.ALOAD_1
217                 && getPrevOpcode(2) == Const.ALOAD_0)) {
218             sawStaticDelegate = true;
219         }
220 
221         if ((seen == Const.INVOKESTATIC || seen == Const.INVOKESPECIAL || seen == Const.INVOKEVIRTUAL)
222                 && ("org/apache/commons/lang/builder/EqualsBuilder".equals(getClassConstantOperand())
223                         || "org/apache/commons/lang3/builder/EqualsBuilder".equals(getClassConstantOperand()))) {
224             sawEqualsBuilder = true;
225         }
226 
227         if (seen == Const.IRETURN && getPC() == 1 && getPrevOpcode(1) == Const.ICONST_0) {
228             alwaysFalse = true;
229             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
230                 bugReporter.reportBug(new BugInstance(this, "EQ_ALWAYS_FALSE", Priorities.HIGH_PRIORITY).addClassAndMethod(this)
231                         .addSourceLine(this));
232             }
233 
234         }
235         if (seen == Const.IRETURN && getPC() == 1 && getPrevOpcode(1) == Const.ICONST_1) {
236             alwaysTrue = true;
237             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
238                 bugReporter.reportBug(new BugInstance(this, "EQ_ALWAYS_TRUE", Priorities.HIGH_PRIORITY).addClassAndMethod(this)
239                         .addSourceLine(this));
240             }
241 
242         }
243         if (seen == Const.IF_ACMPEQ || seen == Const.IF_ACMPNE) {
244             checkForComparingClasses();
245         }
246         if (callToInvoke(seen)) {
247             equalsCalls++;
248             checkForComparingClasses();
249             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass()) && dangerDanger) {
250                 bugReporter.reportBug(new BugInstance(this, "EQ_COMPARING_CLASS_NAMES", Priorities.NORMAL_PRIORITY)
251                 .addClassAndMethod(this).addSourceLine(this));
252             }
253         }
254 
255         if ((seen == Const.INVOKEINTERFACE || seen == Const.INVOKEVIRTUAL) && "compare".equals(getNameConstantOperand())
256                 && stack.getStackDepth() >= 2) {
257             Item left = stack.getStackItem(1);
258             Item right = stack.getStackItem(0);
259             if (left.getRegisterNumber() + right.getRegisterNumber() == 1) {
260                 sawCompare = true;
261             }
262         }
263         dangerDanger = false;
264 
265         if (seen == Const.INVOKEVIRTUAL && "java/lang/Class".equals(getClassConstantOperand())
266                 && "getName".equals(getNameConstantOperand()) && "()Ljava/lang/String;".equals(getSigConstantOperand())
267                 && stack.getStackDepth() >= 2) {
268             Item left = stack.getStackItem(1);
269             XMethod leftM = left.getReturnValueOf();
270             Item right = stack.getStackItem(0);
271             XMethod rightM = right.getReturnValueOf();
272             if (leftM != null && rightM != null && "getName".equals(leftM.getName()) && "getClass".equals(rightM.getName())) {
273                 dangerDanger = true;
274             }
275 
276         }
277         if (seen == Const.INVOKESPECIAL && EQUALS_NAME.equals(getNameConstantOperand())
278                 && EQUALS_SIGNATURE.equals(getSigConstantOperand())) {
279             sawSuperEquals = prevWasSuperEquals = true;
280         } else {
281             if (seen == Const.IRETURN) {
282                 if (prevWasSuperEquals) {
283                     sawReturnSuper = true;
284                 } else {
285                     sawReturnNonSuper = true;
286                 }
287             }
288             prevWasSuperEquals = false;
289         }
290 
291         if (seen == Const.INSTANCEOF && stack.getStackDepth() > 0 && stack.getStackItem(0).getRegisterNumber() == 1) {
292             ClassDescriptor instanceOfCheck = getClassDescriptorOperand();
293             if (instanceOfCheck.equals(getClassDescriptor())) {
294                 sawInstanceOf = true;
295             } else {
296                 try {
297                     if (AnalysisContext.currentAnalysisContext().getSubtypes2().isSubtype(getClassDescriptor(), instanceOfCheck)) {
298                         sawInstanceOfSupertype = true;
299                     }
300                 } catch (ClassNotFoundException e) {
301                     sawInstanceOfSupertype = true;
302                 }
303             }
304         }
305 
306         if (seen == Const.CHECKCAST && stack.getStackDepth() > 0 && stack.getStackItem(0).getRegisterNumber() == 1) {
307             ClassDescriptor castTo = getClassDescriptorOperand();
308             if (castTo.equals(getClassDescriptor())) {
309                 sawCheckedCast = true;
310             }
311             try {
312                 if (AnalysisContext.currentAnalysisContext().getSubtypes2().isSubtype(getClassDescriptor(), castTo)) {
313                     sawCheckedCast = true;
314                 }
315             } catch (ClassNotFoundException e) {
316                 sawCheckedCast = true;
317             }
318         }
319         if (seen == Const.INVOKEVIRTUAL && "getClass".equals(getNameConstantOperand())
320                 && "()Ljava/lang/Class;".equals(getSigConstantOperand())) {
321             sawGetClass = true;
322         }
323 
324     }

326     private boolean callToInvoke(int seen) {
327         if (seen == Const.INVOKEVIRTUAL || seen == Const.INVOKEINTERFACE || seen == Const.INVOKESPECIAL) {
328             return invokesMethodWithEqualLikeName() && EQUALS_SIGNATURE.equals(getSigConstantOperand());
329         }
330         if (seen == Const.INVOKESTATIC) {
331             String sig = getSigConstantOperand();
332             return invokesMethodWithEqualLikeName() && sig.endsWith("Ljava/lang/Object;)Z");
333         }
334 
335         return false;
336 
337     }
338 
339     public boolean invokesMethodWithEqualLikeName() {
340         return getNameConstantOperand().toLowerCase().indexOf(EQUALS_NAME) >= 0;
341     }

346     private void checkForComparingClasses() {
347         if (stack.getStackDepth() >= 2) {
348             Item left = stack.getStackItem(1);
349             XMethod leftM = left.getReturnValueOf();
350             Item right = stack.getStackItem(0);
351             XMethod rightM = right.getReturnValueOf();
352             if ("Ljava/lang/Class;".equals(left.getSignature()) && "Ljava/lang/Class;".equals(right.getSignature())) {
353                 boolean leftMatch = leftM != null && "getClass".equals(leftM.getName());
354                 boolean rightMatch = rightM != null && "getClass".equals(rightM.getName());
355                 if (leftMatch && rightMatch) {
356                     sawGoodEqualsClass = true;
357                 } else {
358                     if (getClassName().equals(left.getConstant()) && rightMatch || leftMatch
359                             && getClassName().equals(right.getConstant())) {
360                         if (getThisClass().isFinal()) {
361                             sawGoodEqualsClass = true;
362                         } else {
363                             sawBadEqualsClass = true;
364                             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
365 
366                                 int priority = Priorities.NORMAL_PRIORITY;
367 
368                                 BugInstance bug = new BugInstance(this, "EQ_GETCLASS_AND_CLASS_CONSTANT", priority)
369                                 .addClassAndMethod(this);
370 
371                                 try {
372 
373                                     Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
374                                     Set<ClassDescriptor> subtypes = subtypes2.getDirectSubtypes(getClassDescriptor());
375                                     for (ClassDescriptor c : subtypes) {
376                                         try {
377                                             Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
378                                         } catch (CheckedAnalysisException e) {
379                                             continue;
380                                         }
381                                         XMethod m = Hierarchy2.findMethod(c, "equals", "(Ljava/lang/Object;)Z", false);
382                                         if (m == null) {
383                                             bug.addClass(c).describe(ClassAnnotation.SUBCLASS_ROLE);
384                                             priority--;
385                                             bug.setPriority(priority);
386                                         }
387                                     }
388 
389                                 } catch (ClassNotFoundException e) {
390                                     bugReporter.reportMissingClass(e);
391                                 }
392                                 bugAccumulator.accumulateBug(bug, this);
393                             }
394                         }
395                     }
396                 }
397             }
398 
399         }
400     }

402     @Override
403     public void report() {
404 
405         if (false) {
406             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
407             for (Map.Entry<ClassDescriptor, Set<ClassDescriptor>> e : classesWithGetClassBasedEquals.entrySet()) {
408                 ClassAnnotation parentClass = ClassAnnotation.fromClassDescriptor(e.getKey());
409                 XClass xParent = AnalysisContext.currentXFactory().getXClass(e.getKey());
410                 if (xParent == null) {
411                     continue;
412                 }
413                 EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
414                 for (ClassDescriptor child : e.getValue()) {
415                     if (child.equals(e.getKey())) {
416                         continue;
417                     }
418                     XClass xChild = AnalysisContext.currentXFactory().getXClass(child);
419                     if (xChild == null) {
420                         continue;
421                     }
422                     ClassAnnotation childClass = ClassAnnotation.fromClassDescriptor(child);
423                     EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
424                     int fieldsOfInterest = 0;
425                     for (XField f : xChild.getXFields()) {
426                         if (!f.isStatic() && !f.isSynthetic()) {
427                             fieldsOfInterest++;
428                         }
429                     }
430                     int grandchildren = -1;
431                     try {
432 
433                         grandchildren = subtypes2.getSubtypes(child).size();
434                     } catch (ClassNotFoundException e1) {
435                         assert true;
436                     }
437                     System.out.println(parentKind + " " + childKind + " " + parentClass + " " + childClass + " "
438                             + fieldsOfInterest + " " + grandchildren);
439                     try {
440                         if (grandchildren >= 2) {
441                             for (ClassDescriptor g : subtypes2.getSubtypes(child)) {
442                                 if (!g.equals(child)) {
443                                     System.out.println("  " + g);
444                                 }
445                             }
446                         }
447                     } catch (ClassNotFoundException e1) {
448                         assert true;
449                     }
450 
451                 }
452 
453             }
454             int overridden = 0, total = 0;
455             for (Map.Entry<ClassDescriptor, Set<ClassDescriptor>> e : classesWithInstanceOfBasedEquals.entrySet()) {
456                 ClassAnnotation parentClass = ClassAnnotation.fromClassDescriptor(e.getKey());
457                 XClass xParent = AnalysisContext.currentXFactory().getXClass(e.getKey());
458                 if (xParent == null) {
459                     continue;
460                 }
461                 EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
462                 boolean isOverridden = false;
463                 for (ClassDescriptor child : e.getValue()) {
464                     if (child.equals(e.getKey())) {
465                         continue;
466                     }
467                     XClass xChild = AnalysisContext.currentXFactory().getXClass(child);
468                     if (xChild == null) {
469                         continue;
470                     }
471                     ClassAnnotation childClass = ClassAnnotation.fromClassDescriptor(child);
472                     EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
473                     if (childKind != null) {
474                         isOverridden = true;
475                     }
476                 }
477                 total++;
478                 if (isOverridden) {
479                     overridden++;
480                 }
481                 System.out.println("IS_OVERRIDDEN: " + e.getKey().getClassName());
482             }
483             System.out.println("Instance of equals: " + total + " subclassed, " + overridden + " overrridden");
484             for (Map.Entry<EqualsKindSummary.KindOfEquals, Integer> e : count.entrySet()) {
485                 System.out.println(e);
486             }
487 
488         }
489 
490         for (Map.Entry<ClassAnnotation, ClassAnnotation> e : parentMap.entrySet()) {
491             ClassAnnotation childClass = e.getKey();
492             EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
493             ClassAnnotation parentClass = e.getValue();
494             EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
495 
496             if (childKind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS
497                     && parentKind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS) {
498                 bugReporter.reportBug(new BugInstance(this, "EQ_OVERRIDING_EQUALS_NOT_SYMMETRIC", NORMAL_PRIORITY)
499                 .add(childClass).addMethod(equalsMethod.get(childClass)).addMethod(equalsMethod.get(parentClass))
500                 .describe(MethodAnnotation.METHOD_OVERRIDDEN));
501             }
502 
503         }
