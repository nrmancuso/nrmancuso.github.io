56      public static interface Filter {
57          public boolean accepts(Profile p);
58      }
59  
60      public static class FilterByTime implements Filter {
61          private final long minNanoSeconds;
62  
63          public FilterByTime(long minNanoSeconds) {
64              this.minNanoSeconds = minNanoSeconds;
65          }
66  
67          @Override
68          public boolean accepts(Profile p) {
69              long time = p.totalTime.get();
70              if (time < minNanoSeconds) {
71                  return false;
72              }
73              return true;
74          }
75      }
76  
77      public static class FilterByTimePerCall implements Filter {
78          private final long minNanoSeconds;
79  
80          public FilterByTimePerCall(long minNanoSeconds) {
81              this.minNanoSeconds = minNanoSeconds;
82          }
83  
84          @Override
85          public boolean accepts(Profile p) {
86              int totalCalls = p.totalCalls.get();
87              long time = p.totalTime.get();
88              if (time / totalCalls < minNanoSeconds) {
89                  return false;
90              }
91              return true;
92          }
93      }

95      public static class FilterByCalls implements Filter {
96          private final int minCalls;
97  
98          public FilterByCalls(int minCalls) {
99              this.minCalls = minCalls;
100         }
101 
102         @Override
103         public boolean accepts(Profile p) {

105             if (totalCalls < minCalls) {
106                 return false;
107             }
108             return true;
109         }

112     public static class Profile implements XMLWriteable {
113         /** time in nanoseconds */
114         final AtomicLong totalTime = new AtomicLong();
115 
116         final AtomicInteger totalCalls = new AtomicInteger();
117 
118         /** time in nanoseconds */
119         final AtomicLong maxTime = new AtomicLong();
120 
121         final AtomicLong totalSquareMicroseconds = new AtomicLong();
122 
123         private final String className;
124 
125         Object maxContext;
126 
127         /**
128          * @param className
129          *            non null full qualified class name
130          */
131         public Profile(String className) {
132             this.className = className;
133         }
134 
135         public void handleCall(long nanoTime, Object context) {
136             totalCalls.incrementAndGet();
137             totalTime.addAndGet(nanoTime);
138             long oldMax = maxTime.get();
139             if (nanoTime > oldMax) {
140                 maxTime.compareAndSet(oldMax, nanoTime);
141                 if (MAX_CONTEXT) {
142                     maxContext = context;
143                 }
144             }
145             long microseconds = TimeUnit.MICROSECONDS.convert(nanoTime, TimeUnit.NANOSECONDS);
146             totalSquareMicroseconds.addAndGet(microseconds * microseconds);
147         }
148 
149         public long getTotalTime() {
150             return totalTime.get();
151         }
152 
153         /**
154          * @param xmlOutput
155          * @throws IOException
156          */
157 
158         @Override
159         public void writeXML(XMLOutput xmlOutput) throws IOException {
160             long time = totalTime.get();
161             int callCount = totalCalls.get();
162             long maxTimeMicros = TimeUnit.MICROSECONDS.convert(maxTime.get(), TimeUnit.NANOSECONDS);
163             long timeMillis = TimeUnit.MILLISECONDS.convert(time, TimeUnit.NANOSECONDS);
164             long timeMicros = TimeUnit.MICROSECONDS.convert(time, TimeUnit.NANOSECONDS);
165 
166             long averageTimeMicros = timeMicros / callCount;
167             long totalSquareMicros = totalSquareMicroseconds.get();
168             long averageSquareMicros = totalSquareMicros / callCount;
169             long timeVariance = averageSquareMicros - averageTimeMicros * averageTimeMicros;
170             long timeStandardDeviation = (long) Math.sqrt(timeVariance);
171             if (timeMillis > 10) {
172                 xmlOutput.startTag("ClassProfile");
173 
174                 xmlOutput.addAttribute("name", className);
175                 xmlOutput.addAttribute("totalMilliseconds", String.valueOf(timeMillis));
176                 xmlOutput.addAttribute("invocations", String.valueOf(callCount));
177                 xmlOutput.addAttribute("avgMicrosecondsPerInvocation", String.valueOf(averageTimeMicros));
178                 xmlOutput.addAttribute("maxMicrosecondsPerInvocation", String.valueOf(maxTimeMicros));
179                 if (maxContext != null) {
180                     xmlOutput.addAttribute("maxContext", String.valueOf(maxContext));
181                 }
182                 xmlOutput.addAttribute("standardDeviationMicrosecondsPerInvocation", String.valueOf(timeStandardDeviation));
183                 xmlOutput.stopTag(true);
184             }
185         }
186     }

188     static class Clock {
189         final Class<?> clazz;
190 
191         long startTimeNanos;
192 
193         long accumulatedTime;
194 
195         Clock(Class<?> clazz, long currentNanoTime) {
196             this.clazz = clazz;
197             startTimeNanos = currentNanoTime;
198         }
199 
200         void accumulateTime(long currentNanoTime) {
201             accumulatedTime += currentNanoTime - startTimeNanos;
202         }
203 
204         void restartClock(long currentNanoTime) {
205             startTimeNanos = currentNanoTime;
206         }
207 
208     }
209 
210     final Stack<Clock> startTimes;
211 
212     final ConcurrentMap<Class<?>, Profile> profile;
213 
214     final Stack<Object> context = new Stack<>();
215 
216     public void startContext(Object context) {
217         this.context.push(context);
218     }
219 
220     public void endContext(Object context) {
221         Object o = this.context.pop();
222         assert o == context;
223     }
224 
225     private Object getContext() {
226         if (context.size() == 0) {
227             return "";
228         }
229         try {
230             return context.peek();
231         } catch (EmptyStackException e) {
232             return "";
233         }
234     }
235     public void start(Class<?> c) {
236         long currentNanoTime = System.nanoTime();
237 
238         Stack<Clock> stack = startTimes;
239         if (!stack.isEmpty()) {
240             stack.peek().accumulateTime(currentNanoTime);
241         }
242         stack.push(new Clock(c, currentNanoTime));
243         // System.err.println("push " + c.getSimpleName());
244 
245     }
246 
247     public void end(Class<?> c) {
248         // System.err.println("pop " + c.getSimpleName());
249         long currentNanoTime = System.nanoTime();
250 
251         Stack<Clock> stack = startTimes;
252         Clock ending = stack.pop();
253         if (ending.clazz != c) {
254             throw new AssertionError("Asked to end timing for " + c + " but top of stack is " + ending.clazz
255                     + ", remaining stack is " + stack);
256         }
257         ending.accumulateTime(currentNanoTime);
258         if (!stack.isEmpty()) {
259             Clock restarting = stack.peek();
260             restarting.restartClock(currentNanoTime);
261         }
262         long accumulatedTime = ending.accumulatedTime;
263         if (accumulatedTime == 0) {
264             return;
265         }
266         Profile counter = profile.get(c);
267         if (counter == null) {
268             counter = new Profile(c.getName());
269             Profile counter2 = profile.putIfAbsent(c, counter);
270             if (counter2 != null) {
271                 counter = counter2;
272             }
273         }
274         counter.handleCall(accumulatedTime, getContext());
275 
276     }

278     public static class ClassNameComparator implements Comparator<Class<?>> {
279         final protected Profiler profiler;
280 
281         public ClassNameComparator(Profiler p) {
282             this.profiler = p;
283         }
284 
285         @Override
286         public int compare(Class<?> c1, Class<?> c2) {
287             try {
288                 return c1.getSimpleName().compareTo(c2.getSimpleName());
289             } catch (RuntimeException e) {
290                 AnalysisContext.logError("Error comparing " + c1 + " and " + c2, e);
291                 int i1 = System.identityHashCode(c1);
292                 int i2 = System.identityHashCode(c2);
293                 if (i1 < i2) {
294                     return -1;
295                 }
296                 if (i1 > i2) {
297                     return 1;
298                 }
299                 return 0;
300             }
301         }
302     }
303 
304     public static class TotalTimeComparator extends ClassNameComparator {
305 
306         public TotalTimeComparator(Profiler p) {
307             super(p);
308         }
309 
310         @Override
311         public int compare(Class<?> c1, Class<?> c2) {
312             long v1 = profiler.getProfile(c1).totalTime.get();
313             long v2 = profiler.getProfile(c2).totalTime.get();
314             if (v1 < v2) {
315                 return 1;
316             }
317             if (v1 > v2) {
318                 return -1;
319             }
320             return super.compare(c1, c2);
321         }
322     }

324     public static class TimePerCallComparator extends ClassNameComparator {
325         public TimePerCallComparator(Profiler p) {
326             super(p);
327         }
328 
329         @Override
330         public int compare(Class<?> c1, Class<?> c2) {
331             Profile profile1 = profiler.getProfile(c1);
332             Profile profile2 = profiler.getProfile(c2);
333             long time1 = profile1.totalTime.get() / profile1.totalCalls.get();
334             long time2 = profile2.totalTime.get() / profile2.totalCalls.get();
335             if (time1 < time2) {
336                 return 1;
337             }
338             if (time1 > time2) {
339                 return -1;
340             }
341             return super.compare(c1, c2);
342         }
343     }

345     public static class TotalCallsComparator extends ClassNameComparator {
346         public TotalCallsComparator(Profiler p) {
347             super(p);
348         }
349 
350         @Override
351         public int compare(Class<?> c1, Class<?> c2) {
352             Profile profile1 = profiler.getProfile(c1);
353             Profile profile2 = profiler.getProfile(c2);
354             int calls1 = profile1.totalCalls.get();
355             int calls2 = profile2.totalCalls.get();
356             if (calls1 < calls2) {
357                 return 1;
358             }
359             if (calls1 > calls2) {
360                 return -1;
361             }
362             return super.compare(c1, c2);
363         }
364     }

366     /**
367      * Default implementation uses {@link TotalTimeComparator} and prints out
368      * class statistics based on total time spent fot a class
369      */
370     public void report() {
371         if (!REPORT) {
372             return;
373         }
374         report(new TotalTimeComparator(this), new FilterByTime(10000000), System.err);
375     }
376 
377     /**
378      * @param reportComparator
379      *            non null comparator instance which will be used to sort the
380      *            report statistics
381      */
382     public void report(Comparator<Class<?>> reportComparator, Filter filter, PrintStream stream) {
383         stream.println("PROFILE REPORT");
384         try {
385 
386             TreeSet<Class<?>> treeSet = new TreeSet<>(reportComparator);
387             treeSet.addAll(profile.keySet());
388 
389             stream.printf("%8s  %8s %9s %s%n", "msecs", "#calls", "usecs/call", "Class");
390 
391             for (Class<?> c : treeSet) {
392                 Profile p = getProfile(c);
393                 long time = p.totalTime.get();
394                 int callCount = p.totalCalls.get();
395                 if (filter.accepts(p)) {
396                     stream.printf("%8d  %8d  %8d %s%n", Long.valueOf(TimeUnit.MILLISECONDS.convert(time, TimeUnit.NANOSECONDS)),
397                             Integer.valueOf(callCount),
398                             Long.valueOf(TimeUnit.MICROSECONDS.convert(time / callCount, TimeUnit.NANOSECONDS)),
399                             c.getSimpleName());
400                 }
401 
402             }
403             stream.flush();
404         } catch (RuntimeException e) {
405             System.err.println(e);
406         }
407     }

409     /**
410      * Clears the previously accumulated data. This method is public because it
411      * can be accessed explicitely from clients (like Eclipse).
412      * <p>
413      * There is no need to clear profiler data after each run, because a new
414      * profiler instance is used for each analysis run (see
415      * {@link FindBugs2#execute()}).
416      */
417     public void clear() {
418         profile.clear();
419         startTimes.clear();
420     }
421 
422     public Profile getProfile(Class<?> c) {
423         Profile result = profile.get(c);
424         if (result == null) {
425             AnalysisContext.logError("Unexpected null profile for " + c.getName(), new NullPointerException());
426             result = new Profile(c.getName());
427             Profile tmp = profile.putIfAbsent(c, result);
428             if (tmp != null) {
429                 return tmp;
430             }
431         }
432         return result;
433     }

435     /*
436      * (non-Javadoc)
437      *
438      * @see
439      * edu.umd.cs.findbugs.xml.XMLWriteable#writeXML(edu.umd.cs.findbugs.xml
440      * .XMLOutput)
441      */
442     @Override
443     public void writeXML(XMLOutput xmlOutput) throws IOException {
444         xmlOutput.startTag("FindBugsProfile");
445         xmlOutput.stopTag(false);
446         TreeSet<Class<?>> treeSet = new TreeSet<>(new TotalTimeComparator(this));
447         treeSet.addAll(profile.keySet());
448         long totalTime = 0;
449         for (Profile p : profile.values()) {
450             totalTime += p.totalTime.get();
451         }
452 
453         long accumulatedTime = 0;
454 
455         for (Class<?> c : treeSet) {
456             Profile p = getProfile(c);
457             if (p == null) {
458                 continue;
459             }
460             p.writeXML(xmlOutput);
461             accumulatedTime += p.totalTime.get();
462             if (accumulatedTime > 3 * totalTime / 4) {
463                 break;
464             }
465         }
466         xmlOutput.closeTag("FindBugsProfile");
467     }
