387    * Tests AssignmentManager balance function.  Runs a balance moving a region
388    * from one server to another mocking regionserver responding over zk.
389    * @throws IOException
390    * @throws KeeperException
391    * @throws DeserializationException

393   @Test (timeout=180000)
394   public void testBalance() throws IOException, KeeperException, DeserializationException,
395       InterruptedException, CoordinatedStateException {
396     // Create and startup an executor.  This is used by AssignmentManager
397     // handling zk callbacks.
398     ExecutorService executor = startupMasterExecutor("testBalanceExecutor");
399 
400     // We need a mocked catalog tracker.
401     LoadBalancer balancer = LoadBalancerFactory.getLoadBalancer(server
402         .getConfiguration());
403     // Create an AM.
404     AssignmentManager am = new AssignmentManager(this.server,
405       this.serverManager, balancer, executor, null, master.getTableLockManager());
406     am.failoverCleanupDone.set(true);
407     try {
408       // Make sure our new AM gets callbacks; once registered, can't unregister.
409       // Thats ok because we make a new zk watcher for each test.
410       this.watcher.registerListenerFirst(am);
411       // Call the balance function but fake the region being online first at
412       // SERVERNAME_A.  Create a balance plan.
413       am.regionOnline(REGIONINFO, SERVERNAME_A);
414       // Balance region from A to B.
415       RegionPlan plan = new RegionPlan(REGIONINFO, SERVERNAME_A, SERVERNAME_B);
416       am.balance(plan);
417 
418       RegionStates regionStates = am.getRegionStates();
419       // Must be failed to close since the server is fake
420       assertTrue(regionStates.isRegionInTransition(REGIONINFO)
421         && regionStates.isRegionInState(REGIONINFO, State.FAILED_CLOSE));
422       // Move it back to pending_close
423       regionStates.updateRegionState(REGIONINFO, State.PENDING_CLOSE);
424 
425       // Now fake the region closing successfully over on the regionserver; the
426       // regionserver will have set the region in CLOSED state.  This will
427       // trigger callback into AM. The below zk close call is from the RS close
428       // region handler duplicated here because its down deep in a private
429       // method hard to expose.
430       int versionid =
431         ZKAssign.transitionNodeClosed(this.watcher, REGIONINFO, SERVERNAME_A, -1);
432       assertNotSame(versionid, -1);
433       // AM is going to notice above CLOSED and queue up a new assign.  The
434       // assign will go to open the region in the new location set by the
435       // balancer.  The zk node will be OFFLINE waiting for regionserver to
436       // transition it through OPENING, OPENED.  Wait till we see the OFFLINE
437       // zk node before we proceed.
438       Mocking.waitForRegionPendingOpenInRIT(am, REGIONINFO.getEncodedName());
439 
440       // Get current versionid else will fail on transition from OFFLINE to OPENING below
441       versionid = ZKAssign.getVersion(this.watcher, REGIONINFO);
442       assertNotSame(-1, versionid);
443       // This uglyness below is what the openregionhandler on RS side does.
444       versionid = ZKAssign.transitionNode(server.getZooKeeper(), REGIONINFO,
445         SERVERNAME_B, EventType.M_ZK_REGION_OFFLINE,
446         EventType.RS_ZK_REGION_OPENING, versionid);
447       assertNotSame(-1, versionid);
448       // Move znode from OPENING to OPENED as RS does on successful open.
449       versionid =
450         ZKAssign.transitionNodeOpened(this.watcher, REGIONINFO, SERVERNAME_B, versionid);
451       assertNotSame(-1, versionid);
452       // Wait on the handler removing the OPENED znode.
453       while(regionStates.isRegionInTransition(REGIONINFO)) Threads.sleep(1);
454     } finally {
455       executor.shutdown();
456       am.shutdown();
457       // Clean up all znodes
458       ZKAssign.deleteAllNodes(this.watcher);
459     }
