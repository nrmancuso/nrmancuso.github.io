19  package org.jsecurity.util;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  import org.jsecurity.mgt.SecurityManager;
24  import org.jsecurity.subject.Subject;
25  
26  import java.net.InetAddress;
27  import java.util.HashMap;
28  import java.util.Map;
29  
30  /**
31   * A ThreadContext provides a means of binding and unbinding objects to the
32   * current thread based on key/value pairs.
33   *
34   * <p>An internal {@link java.util.HashMap} is used to maintain the key/value pairs
35   * for each thread.</p>
36   *
37   * <p>If the desired behavior is to ensure that bound data is not shared across
38   * threads in a pooled or reusable threaded environment, the application (or more likely a framework) must
39   * bind and remove any necessary values at the beginning and end of stack
40   * execution, respectively (i.e. individually explicitly or all via the <tt>clear</tt> method).</p>
41   *
42   * @author Les Hazlewood
43   * @see #clear()
44   * @since 0.1
45   */
46  @SuppressWarnings(value = {"unchecked", "unsafe"})
47  public abstract class ThreadContext {
48  
49      private static final Log log = LogFactory.getLog(ThreadContext.class);
50  
51      public static final String SECURITY_MANAGER_KEY = ThreadContext.class.getName() + "_SECURITY_MANAGER_KEY";
52      public static final String SUBJECT_KEY = ThreadContext.class.getName() + "_SUBJECT_KEY";
53      public static final String INET_ADDRESS_KEY = ThreadContext.class.getName() + "_INET_ADDRESS_KEY";
54  
55      protected static ThreadLocal<Map<Object, Object>> resources =
56              new InheritableThreadLocal<Map<Object, Object>>() {
57                  protected Map<Object, Object> initialValue() {
58                      return new HashMap<Object, Object>();
59                  }
60              };
61  
62      protected ThreadContext() {
63      }
64  
65      /**
66       * Returns the ThreadLocal Map. This Map is used internally to bind objects
67       * to the current thread by storing each object under a unique key.
68       *
69       * @return the map of bound resources
70       */
71      protected static Map<Object, Object> getResources() {
72          return resources.get();
73      }
74  
75      /**
76       * Returns the object for the specified <code>key</code> that is bound to
77       * the current thread.
78       *
79       * @param key the key that identifies the value to return
80       * @return the object keyed by <code>key</code> or <code>null</code> if
81       *         no value exists for the specified <code>key</code>
82       */
83      public static Object get(Object key) {
84          if (log.isTraceEnabled()) {
85              String msg = "get() - in thread [" + Thread.currentThread().getName() + "]";
86              log.trace(msg);
87          }
88          Object value = getResources().get(key);
89          if ((value != null) && log.isTraceEnabled()) {
90              String msg = "Retrieved value of type [" + value.getClass().getName() + "] for key [" +
91                      key + "] " + "bound to thread [" + Thread.currentThread().getName() + "]";
92              log.trace(msg);
93          }
94          return value;
95      }
96  
97      /**
98       * Binds <tt>value</tt> for the given <code>key</code> to the current thread.
99       *
100      * <p>A <tt>null</tt> <tt>value</tt> has the same effect as if <tt>remove</tt> was called for the given
101      * <tt>key</tt>, i.e.:
102      *
103      * <pre>
104      * if ( value == null ) {
105      *     remove( key );
106      * }</pre>
107      *
108      * @param key   The key with which to identify the <code>value</code>.
109      * @param value The value to bind to the thread.
110      * @throws IllegalArgumentException if the <code>key</code> argument is <tt>null</tt>.
111      */
112     public static void put(Object key, Object value) {
113         if (key == null) {
114             throw new IllegalArgumentException("key cannot be null");
115         }
116 
117         if (value == null) {
118             remove(key);
119             return;
120         }
121 
122         getResources().put(key, value);
123 
124         if (log.isTraceEnabled()) {
125             String msg = "Bound value of type [" + value.getClass().getName() + "] for key [" +
126                     key + "] to thread " + "[" + Thread.currentThread().getName() + "]";
127             log.trace(msg);
128         }
129     }
130 
131     /**
132      * Unbinds the value for the given <code>key</code> from the current
133      * thread.
134      *
135      * @param key The key identifying the value bound to the current thread.
136      * @return the object unbound or <tt>null</tt> if there was nothing bound
137      *         under the specified <tt>key</tt> name.
138      */
139     public static Object remove(Object key) {
140         Object value = getResources().remove(key);
141 
142         if ((value != null) && log.isTraceEnabled()) {
143             String msg = "Removed value of type [" + value.getClass().getName() + "] for key [" +
144                     key + "]" + "from thread [" + Thread.currentThread().getName() + "]";
145             log.trace(msg);
146         }
147 
148         return value;
149     }
150 
151     /**
152      * Returns true if a value for the <code>key</code> is bound to the current thread, false otherwise.
153      *
154      * @param key the key that may identify a value bound to the current thread.
155      * @return true if a value for the key is bound to the current thread, false
156      *         otherwise.
157      */
158     public static boolean containsKey(Object key) {
159         return getResources().containsKey(key);
160     }
161 
162     /**
163      * Removes <em>all</em> values bound to this ThreadContext, which includes any Subject, Session, or InetAddress
164      * that may be bound by these respective objects' conveninece methods, as well as all values bound by your
165      * application code.
166      *
167      * <p>This operation is meant as a clean-up operation that may be called at the end of
168      * thread execution to prevent data corruption in a pooled thread environment.
169      */
170     public static void clear() {
171         getResources().clear();
172         if (log.isTraceEnabled()) {
173             log.trace("Removed all ThreadContext values from thread [" + Thread.currentThread().getName() + "]");
174         }
175     }
176 
177     /**
178      * Convenience method that simplifies retrieval of the application's SecurityManager instance from the current
179      * thread. If there is no SecurityManager bound to the thread (probably because framework code did not bind it
180      * to the thread), this method returns <tt>null</tt>.
181      * <p/>
182      * It is merely a convenient wrapper for the following:
183      * <p/>
184      * <code>return (SecurityManager)get( SECURITY_MANAGER_KEY );</code>
185      * <p/>
186      * This method only returns the bound value if it exists - it does not remove it
187      * from the thread.  To remove it, one must call {@link #unbindSecurityManager() unbindSecurityManager()} instead.
188      *
189      * @return the Subject object bound to the thread, or <tt>null</tt> if there isn't one bound.
190      * @since 0.9
191      */
192     public static SecurityManager getSecurityManager() {
193         return (SecurityManager) get(SECURITY_MANAGER_KEY);
194     }
195 
196 
197     /**
198      * Convenience method that simplifies binding the application's SecurityManager instance to the ThreadContext.
199      *
200      * <p>The method's existence is to help reduce casting in code and to simplify remembering of
201      * ThreadContext key names.  The implementation is simple in that, if the SecurityManager is not <tt>null</tt>,
202      * it binds it to the thread, i.e.:
203      *
204      * <pre>
205      * if (securityManager != null) {
206      *     put( SECURITY_MANAGER_KEY, securityManager);
207      * }</pre>
208      *
209      * @param securityManager the application's SecurityManager instance to bind to the thread.  If the argument is
210      *                        null, nothing will be done.
211      * @since 0.9
212      */
213     public static void bind(SecurityManager securityManager) {
214         if (securityManager != null) {
215             put(SECURITY_MANAGER_KEY, securityManager);
216         }
217     }
218 
219     /**
220      * Convenience method that simplifies removal of the application's SecurityManager instance from the thread.
221      * <p/>
222      * The implementation just helps reduce casting and remembering of the ThreadContext key name, i.e it is
223      * merely a conveient wrapper for the following:
224      * <p/>
225      * <code>return (SecurityManager)remove( SECURITY_MANAGER_KEY );</code>
226      * <p/>
227      * If you wish to just retrieve the object from the thread without removing it (so it can be retrieved later
228      * during thread execution), use the {@link #getSecurityManager() getSecurityManager()} method instead.
229      *
230      * @return the application's SecurityManager instance previously bound to the thread, or <tt>null</tt> if there
231      *         was none bound.
232      * @since 0.9
233      */
234     public static SecurityManager unbindSecurityManager() {
235         return (SecurityManager) remove(SECURITY_MANAGER_KEY);
236     }
237 
238     /**
239      * Convenience method that simplifies retrieval of a thread-bound Subject.  If there is no
240      * Subject bound to the thread, this method returns <tt>null</tt>.  It is merely a convenient wrapper
241      * for the following:
242      * <p/>
243      * <code>return (Subject)get( SUBJECT_KEY );</code>
244      * <p/>
245      * This method only returns the bound value if it exists - it does not remove it
246      * from the thread.  To remove it, one must call {@link #unbindSubject() unbindSubject()} instead.
247      *
248      * @return the Subject object bound to the thread, or <tt>null</tt> if there isn't one bound.
249      * @since 0.2
250      */
251     public static Subject getSubject() {
252         return (Subject) get(SUBJECT_KEY);
253     }
254 
255 
256     /**
257      * Convenience method that simplifies binding a Subject to the ThreadContext.
258      *
259      * <p>The method's existence is to help reduce casting in your own code and to simplify remembering of
260      * ThreadContext key names.  The implementation is simple in that, if the Subject is not <tt>null</tt>,
261      * it binds it to the thread, i.e.:
262      *
263      * <pre>
264      * if (subject != null) {
265      *     put( SUBJECT_KEY, subject );
266      * }</pre>
267      *
268      * @param subject the Subject object to bind to the thread.  If the argument is null, nothing will be done.
269      * @since 0.2
270      */
271     public static void bind(Subject subject) {
272         if (subject != null) {
273             put(SUBJECT_KEY, subject);
274         }
275     }
276 
277     /**
278      * Convenience method that simplifies removal of a thread-local Subject from the thread.
279      * <p/>
280      * The implementation just helps reduce casting and remembering of the ThreadContext key name, i.e it is
281      * merely a conveient wrapper for the following:
282      * <p/>
283      * <code>return (Subject)remove( SUBJECT_KEY );</code>
284      * <p/>
285      * If you wish to just retrieve the object from the thread without removing it (so it can be retrieved later during
286      * thread execution), you should use the {@link #getSubject() getSubject()} method for that purpose.
287      *
288      * @return the Subject object previously bound to the thread, or <tt>null</tt> if there was none bound.
289      * @since 0.2
290      */
291     public static Subject unbindSubject() {
292         return (Subject) remove(SUBJECT_KEY);
293     }
294 
295     /**
296      * Convenience method that simplifies retrieval of a thread-bound InetAddress.  If there is no
297      * InetAddress bound to the thread, this method returns <tt>null</tt>.  It is merely a convenient wrapper
298      * for the following:
299      * <p/>
300      * <code>return (InetAddress)get( INET_ADDRESS_KEY );</code>
301      * <p/>
302      * This method only returns the bound value if it exists - it does not remove it
303      * from the thread.  To remove it, one must call {@link #unbindInetAddress() unbindInetAddress} instead.
304      *
305      * @return the InetAddress object bound to the thread, or <tt>null</tt> if there isn't one bound.
306      * @since 0.2
307      */
308     public static InetAddress getInetAddress() {
309         return (InetAddress) get(INET_ADDRESS_KEY);
310     }
311 
312     /**
313      * Convenience method that simplifies binding an InetAddress to the ThreadContext.
314      *
315      * <p>The method's existence is to help reduce casting in your own code and to simplify remembering of
316      * ThreadContext key names.  The implementation is simple in that, if the inetAddress is not <tt>null</tt>,
317      * it binds it to the thread, i.e.:
318      *
319      * <pre>
320      * if (inetAddress != null) {
321      *     put( INET_ADDRESS_KEY, inetAddress );
322      * }</pre>
323      *
324      * @param inetAddress the InetAddress to bind to the thread.  If the argument is null, nothing will be done.
325      * @since 0.2
326      */
327     public static void bind(InetAddress inetAddress) {
328         if (inetAddress != null) {
329             put(INET_ADDRESS_KEY, inetAddress);
330         }
331     }
332 
333     /**
334      * Convenience method that simplifies removal of a thread-local InetAddress from the thread.
335      * <p/>
336      * The implementation just helps reduce casting and remembering of the ThreadContext key name, i.e it is
337      * merely a conveient wrapper for the following:
338      * <p/>
339      * <code>return (InetAddress)remove( INET_ADDRESS_KEY );</code>
340      * <p/>
341      * If you wish to just retrieve the object from the thread without removing it (so it can be retrieved later during
342      * thread execution), you should use the {@link #getInetAddress() getInetAddress()} method for that purpose.
343      *
344      * @return the InetAddress object previously bound to the thread, or <tt>null</tt> if there was none bound.

349     }
