40  /**
41   * Web-application capable <tt>SessionManager</tt> implementation.

50      /**
51       * Property specifying if, after a session object is acquired from the request, if that session should be
52       * validated to ensure the starting origin of the session is the same as the incoming request.
53       */
54      private boolean validateRequestOrigin = false; //default
55  
56      protected CookieAttribute<Serializable> sessionIdCookieAttribute = null;
57      protected RequestParamAttribute<Serializable> sessionIdRequestParamAttribute = null;
58  
59      public DefaultWebSessionManager() {
60          ensureCookieSessionIdStore();
61          ensureRequestParamSessionIdStore();
62      }

80      /**
81       * If set to <tt>true</tt>, this implementation will ensure that any
82       * <tt>HttpRequest</tt> attempting
83       * to join a session (i.e. via {@link #getSession getSession} must have the same
84       * IP Address of the <tt>HttpRequest</tt> that started the session.
85       *
86       * <p> If set to <tt>false</tt>, any <tt>HttpRequest</tt> with a reference to a valid
87       * session id may acquire that <tt>Session</tt>.
88       *
89       * <p>Although convenient, this should only be enabled in environments where the
90       * system can <em>guarantee</em> that each IP address represents one and only one
91       * machine accessing the system.
92       *
93       * <p>Public websites are not good candidates for enabling this
94       * feature since many browser clients often sit behind NAT routers (in
95       * which case many machines are viewed to come from the same IP, thereby making this
96       * validation check useless).  Also, some internet service providers (e.g. AOL) may change a
97       * client's IP in mid-session, making subsequent requests appear to come from a different
98       * location.  Again, this feature should only be enabled where IP Addresses can be guaranteed a
99       * 1-to-1 relationship with a user's session.
100      *
101      * <p>For the reasons specified above, this property is <tt>false</tt> by default.
102      *
103      * @return true if this factory will verify each HttpRequest joining a session
104      */
105     public boolean isValidateRequestOrigin() {
106         return validateRequestOrigin;
107     }
108 
109     /**
110      * Sets whether or not a request's origin will be validated when accessing a session.  See
111      * the {@link #isValidateRequestOrigin} JavaDoc for an in-depth explanation of this property.
112      *
113      * @param validateRequestOrigin whether or not to validate the request's origin when accessing
114      *                              a session.
115      * @see #isValidateRequestOrigin
116      */
117     public void setValidateRequestOrigin(boolean validateRequestOrigin) {
118         this.validateRequestOrigin = validateRequestOrigin;
119     }

137     protected void ensureCookieSessionIdStore() {
138         CookieAttribute<Serializable> cookieStore = getSessionIdCookieAttribute();
139         if (cookieStore == null) {
140             cookieStore = new CookieAttribute<Serializable>(JSecurityHttpSession.DEFAULT_SESSION_ID_NAME);
141             cookieStore.setCheckRequestParams(false);
142             setSessionIdCookieAttribute(cookieStore);
143         }
144     }
145 
146     protected void ensureRequestParamSessionIdStore() {
147         RequestParamAttribute<Serializable> reqParamStore = getSessionIdRequestParamAttribute();
148         if (reqParamStore == null) {
149             reqParamStore = new RequestParamAttribute<Serializable>(JSecurityHttpSession.DEFAULT_SESSION_ID_NAME);
150             setSessionIdRequestParamAttribute(reqParamStore);
151         }
152     }
153 
154     protected void validateSessionOrigin(ServletRequest request, Session session)
155             throws HostUnauthorizedException {
156         InetAddress requestIp = WebUtils.getInetAddress(request);
157         InetAddress originIp = session.getHostAddress();
158         Serializable sessionId = session.getId();
159 
160         if (originIp == null) {
161             if (requestIp != null) {
162                 String msg = "No IP Address was specified when creating session with id [" +
163                         sessionId + "].  Attempting to access session from " +
164                         "IP [" + requestIp + "].  Origin IP and request IP must match.";
165                 throw new HostUnauthorizedException(msg);
166             }
167         } else {
168             if (requestIp != null) {
169                 if (!requestIp.equals(originIp)) {
170                     String msg = "Session with id [" + sessionId + "] originated from [" +
171                             originIp + "], but the current HttpServletRequest originated " +
172                             "from [" + requestIp + "].  Disallowing session access: " +
173                             "session origin and request origin must match to allow access.";
174                     throw new HostUnauthorizedException(msg);
175                 }
176 
177             } else {
178                 String msg = "No IP Address associated with the current HttpServletRequest.  " +
179                         "Session with id [" + sessionId + "] originated from " +
180                         "[" + originIp + "].  Request IP must match the session's origin " +
181                         "IP in order to gain access to that session.";
182                 throw new HostUnauthorizedException(msg);
183             }
184         }
185     }
186 
187     protected void storeSessionId(Serializable currentId, ServletRequest request, ServletResponse response) {
188         if (currentId == null) {
189             String msg = "sessionId cannot be null when persisting for subsequent requests.";
190             throw new IllegalArgumentException(msg);
191         }
192         //ensure that the id has been set in the idStore, or if it already has, that it is not different than the
193         //'real' session value:
194         Serializable existingId = retrieveSessionId(request, response);
195         if (existingId == null || !currentId.equals(existingId)) {
196             getSessionIdCookieAttribute().storeValue(currentId, request, response);
197         }
198     }
199 
200     protected Serializable retrieveSessionId(ServletRequest request, ServletResponse response) {
201         WebAttribute<Serializable> cookieSessionIdAttribute = getSessionIdCookieAttribute();
202         Serializable id = cookieSessionIdAttribute.retrieveValue(request, response);
203         if (id != null) {
204             request.setAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_ID_SOURCE,
205                     JSecurityHttpServletRequest.COOKIE_SESSION_ID_SOURCE);
206         } else {
207             id = getSessionIdRequestParamAttribute().retrieveValue(request, response);
208             if (id != null) {
209                 request.setAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_ID_SOURCE,
210                         JSecurityHttpServletRequest.URL_SESSION_ID_SOURCE);
211             }
212         }
213         return id;
214     }
215 
216     public Serializable start(InetAddress hostAddress) throws HostUnauthorizedException, IllegalArgumentException {
217         ServletRequest request = WebUtils.getServletRequest();
218         ServletResponse response = WebUtils.getServletResponse();
219         return start(request, response, hostAddress);
220     }
221 
222     protected Serializable start(ServletRequest request, ServletResponse response, InetAddress inetAddress) {
223         Serializable sessionId = super.start(inetAddress);
224         storeSessionId(sessionId, request, response);
225         request.removeAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_ID_SOURCE);
226         request.setAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_IS_NEW, Boolean.TRUE);
227         return sessionId;
228     }
229 
230     public Session retrieveSession(Serializable sessionId) throws InvalidSessionException, AuthorizationException {
231         if (sessionId != null) {
232             return super.retrieveSession(sessionId);
233         } else {
234             ServletRequest request = WebUtils.getServletRequest();
235             ServletResponse response = WebUtils.getServletResponse();
236             return getSession(request, response);
237         }
238     }
239 
240     /**
241      * Returns the Session associated with the specified request if it is valid or <tt>null</tt> if a Session doesn't
242      * exist or it was invalid.
243      *
244      * @param request  incoming servlet request
245      * @param response outgoing servlet response
246      * @return the Session associated with the incoming request or <tt>null</tt> if one does not exist.
247      * @throws org.jsecurity.session.InvalidSessionException
248      *          if the associated Session has expired prior to invoking this method.
249      * @throws org.jsecurity.authz.AuthorizationException
250      *          if the caller is not authorized to access the session associated with the request.
251      */
252     public final Session getSession(ServletRequest request, ServletResponse response)
253             throws InvalidSessionException, AuthorizationException {
254 
255         Session session;
256         try {
257             session = doGetSession(request, response);
258         } catch (InvalidSessionException ise) {
259             if (log.isTraceEnabled()) {
260                 log.trace("Request Session is invalid, message: [" + ise.getMessage() + "].  Removing any " +
261                         "associated session cookie...");
262             }
263             getSessionIdCookieAttribute().removeValue(request, response);
264 
265             //give subclass a chance to do something additional if necessary.  Otherwise returning null is just fine:
266             session = handleInvalidSession(request, response, ise);
267         }
268 
269         return session;
270     }
271 
272     protected Session doGetSession(ServletRequest request, ServletResponse response) {
273 
274         Session session = null;
275         Serializable sessionId = retrieveSessionId(request, response);
276 
277         if (sessionId != null) {
278             request.setAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_ID, sessionId);
279             session = super.retrieveSession(sessionId);
280             if (isValidateRequestOrigin()) {
281                 if (log.isDebugEnabled()) {
282                     log.debug("Validating request origin against session origin");
283                 }
284                 validateSessionOrigin(request, session);
285             }
286             if (session != null) {
287                 request.setAttribute(JSecurityHttpServletRequest.REFERENCED_SESSION_ID_IS_VALID, Boolean.TRUE);
288             }
289         } else {
290             if (log.isTraceEnabled()) {
291                 log.trace("No JSecurity session id associated with the given " +
292                         "HttpServletRequest.  A Session will not be returned.");
293             }
294         }
295 
296         return session;
297     }
298 
299     protected Session handleInvalidSession(ServletRequest request,
300                                            ServletResponse response,
301                                            InvalidSessionException ise) {
302         if (log.isTraceEnabled()) {
303             log.trace("Sesssion associated with the current request is nonexistent or invalid.  Returning null.");
304         }
305         return null;
306     }
307 
308     protected void onStop(Session session) {
309         super.onStop(session);
310         ServletRequest request = WebUtils.getServletRequest();
311         ServletResponse response = WebUtils.getServletResponse();
312         getSessionIdCookieAttribute().removeValue(request, response);
313     }
