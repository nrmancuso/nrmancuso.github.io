19  package org.jsecurity.session.mgt;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  import org.jsecurity.authz.HostUnauthorizedException;
24  import org.jsecurity.session.ExpiredSessionException;
25  import org.jsecurity.session.InvalidSessionException;
26  import org.jsecurity.session.Session;
27  import org.jsecurity.util.Destroyable;
28  import org.jsecurity.util.LifecycleUtils;
29  
30  import java.io.Serializable;
31  import java.net.InetAddress;
32  import java.util.Collection;
33  
34  /**
35   * Default business-tier implementation of the {@link ValidatingSessionManager} interface.
36   *
37   * @author Les Hazlewood
38   * @author Jeremy Haile
39   * @since 0.1
40   */
41  public abstract class AbstractValidatingSessionManager extends AbstractSessionManager
42          implements ValidatingSessionManager, Destroyable {
43  
44      private static final Log log = LogFactory.getLog(AbstractValidatingSessionManager.class);
45  
46      protected static final long MILLIS_PER_SECOND = 1000;
47      protected static final long MILLIS_PER_MINUTE = 60 * MILLIS_PER_SECOND;
48      protected static final long MILLIS_PER_HOUR = 60 * MILLIS_PER_MINUTE;
49  
50      /**
51       * Default main session timeout value (30 * 60 * 1000 milliseconds = 30 minutes).
52       */
53      public static final long DEFAULT_GLOBAL_SESSION_TIMEOUT = 30 * MILLIS_PER_MINUTE;
54  
55      /**
56       * The default interval at which sessions will be validated (1 hour);
57       * This can be overridden by calling {@link #setSessionValidationInterval(long)}
58       */
59      public static final long DEFAULT_SESSION_VALIDATION_INTERVAL = MILLIS_PER_HOUR;
60  
61      protected boolean sessionValidationSchedulerEnabled = true; //default
62      /**
63       * Scheduler used to validate sessions on a regular basis.
64       */
65      protected SessionValidationScheduler sessionValidationScheduler = null;
66  
67      protected long sessionValidationInterval = DEFAULT_SESSION_VALIDATION_INTERVAL;
68      protected long globalSessionTimeout = DEFAULT_GLOBAL_SESSION_TIMEOUT;
69  
70      public AbstractValidatingSessionManager() {
71      }
72  
73      public boolean isSessionValidationSchedulerEnabled() {
74          return sessionValidationSchedulerEnabled;
75      }
76  
77      public void setSessionValidationSchedulerEnabled(boolean sessionValidationSchedulerEnabled) {
78          this.sessionValidationSchedulerEnabled = sessionValidationSchedulerEnabled;
79      }
80  
81      public void setSessionValidationScheduler(SessionValidationScheduler sessionValidationScheduler) {
82          this.sessionValidationScheduler = sessionValidationScheduler;
83      }
84  
85      public SessionValidationScheduler getSessionValidationScheduler() {
86          return sessionValidationScheduler;
87      }
88  
89      public void enableSessionValidationIfNecessary() {
90          SessionValidationScheduler scheduler = getSessionValidationScheduler();
91          if (isSessionValidationSchedulerEnabled() && (scheduler == null || !scheduler.isEnabled())) {
92              enableSessionValidation();
93          }
94      }
95  
96      /**
97       * Returns the time in milliseconds that any session may remain idle before expiring.  This
98       * value is just a main default for all sessions and may be overridden by subclasses on a
99       * <em>per-session</em> basis by overriding the {@link #getTimeout(Session)} method if
100      * so desired.
101      *
102      * <ul>
103      * <li>A negative return value means sessions never expire.</li>
104      * <li>A non-negative return value (0 or greater) means session timeout will occur as expected.</li>
105      * </ul>
106      *
107      * <p>Unless overridden via the {@link #setGlobalSessionTimeout} method, the default value is
108      * {@link #DEFAULT_GLOBAL_SESSION_TIMEOUT}.
109      *
110      * @return the time in milliseconds that any session may remain idle before expiring.
111      */
112     public long getGlobalSessionTimeout() {
113         return globalSessionTimeout;
114     }
115 
116     /**
117      * Sets the time in milliseconds that any session may remain idle before expiring.  This
118      * value is just a main default for all sessions.  Subclasses may override the
119      * {@link #getTimeout} method to determine time-out values on a <em>per-session</em> basis.
120      *
121      * @param globalSessionTimeout the time in milliseconds any session may remain idle before
122      *                             expiring.
123      */
124     public void setGlobalSessionTimeout(int globalSessionTimeout) {
125         this.globalSessionTimeout = globalSessionTimeout;
126     }
127 
128     /**
129      * If using the underlying default <tt>SessionValidationScheduler</tt> (that is, the
130      * {@link #setSessionValidationScheduler(SessionValidationScheduler) setSessionValidationScheduler} method is
131      * never called) , this method allows one to specify how
132      * frequently session should be validated (to check for orphans).  The default value is
133      * {@link #DEFAULT_SESSION_VALIDATION_INTERVAL}.
134      *
135      * <p>If you override the default scheduler, it is assumed that overriding instance 'knows' how often to
136      * validate sessions, and this attribute will be ignored.
137      *
138      * <p>Unless this method is called, the default value is {@link #DEFAULT_SESSION_VALIDATION_INTERVAL}.
139      *
140      * @param sessionValidationInterval the time in milliseconds between checking for valid sessions to reap orphans.
141      */
142     public void setSessionValidationInterval(long sessionValidationInterval) {
143         this.sessionValidationInterval = sessionValidationInterval;
144     }
145 
146     public long getSessionValidationInterval() {
147         return sessionValidationInterval;
148     }
149 
150     protected final Session doGetSession(Serializable sessionId) throws InvalidSessionException {
151         enableSessionValidationIfNecessary();
152         return retrieveSession(sessionId);
153     }
154 
155     protected abstract Session retrieveSession(Serializable sessionId) throws InvalidSessionException;
156 
157     protected final Session createSession(InetAddress originatingHost) throws HostUnauthorizedException, IllegalArgumentException {
158         enableSessionValidationIfNecessary();
159         return doCreateSession(originatingHost);
160     }
161 
162     protected abstract Session doCreateSession(InetAddress originatingHost) throws HostUnauthorizedException, IllegalArgumentException;
163 
164     protected void validate(Session session) throws InvalidSessionException {
165         if (session instanceof ValidatingSession) {
166             try {
167                 ((ValidatingSession) session).validate();
168             } catch (ExpiredSessionException ese) {
169                 notifyExpiration(session);
170                 onExpiration(session);
171                 //propagate to caller:
172                 throw ese;
173             }
174         } else {
175             String msg = "The " + getClass().getName() + " implementation only supports validating " +
176                     "Session implementations of the " + ValidatingSession.class.getName() + " interface.  " +
177                     "Please either implement this interface in your session implementation or override the " +
178                     getClass().getName() + ".validate(Session) method to perform validation.";
179             throw new IllegalStateException(msg);
180         }
181     }
182 
183     /**
184      * Subclass template hook in case per-session timeout is not based on
185      * {@link org.jsecurity.session.Session#getTimeout()}.
186      *
187      * <p>This implementation merely returns {@link org.jsecurity.session.Session#getTimeout()}</p>
188      *
189      * @param session the session for which to determine session timeout.
190      * @return the time in milliseconds the specified session may remain idle before expiring.
191      */
192     protected long getTimeout(Session session) {
193         return session.getTimeout();
194     }
195 
196     protected SessionValidationScheduler createSessionValidationScheduler() {
197         SessionValidationScheduler scheduler;
198 
199         if (log.isDebugEnabled()) {
200             log.debug("No sessionValidationScheduler set.  Attempting to create default instance.");
201         }
202         scheduler = new ExecutorServiceSessionValidationScheduler(this);
203         ((ExecutorServiceSessionValidationScheduler) scheduler).setInterval(getSessionValidationInterval());
204         if (log.isTraceEnabled()) {
205             log.trace("Created default SessionValidationScheduler instance of type [" + scheduler.getClass().getName() + "].");
206         }
207         return scheduler;
208     }
209 
210     protected void enableSessionValidation() {
211         SessionValidationScheduler scheduler = getSessionValidationScheduler();
212         if (scheduler == null) {
213             scheduler = createSessionValidationScheduler();
214             setSessionValidationScheduler(scheduler);
215         }
216         if (log.isInfoEnabled()) {
217             log.info("Enabling session validation scheduler...");
218         }
219         scheduler.enableSessionValidation();
220         afterSessionValidationEnabled();
221     }
222 
223     protected void afterSessionValidationEnabled() {
224     }
225 
226     protected void disableSessionValidation() {
227         beforeSessionValidationDisabled();
228         SessionValidationScheduler scheduler = getSessionValidationScheduler();
229         if (scheduler != null) {
230             try {
231                 scheduler.disableSessionValidation();
232                 if (log.isInfoEnabled()) {
233                     log.info("Disabled session validation scheduler.");
234                 }
235             } catch (Exception e) {
236                 if (log.isDebugEnabled()) {
237                     String msg = "Unable to disable SessionValidationScheduler.  Ignoring (shutting down)...";
238                     log.debug(msg, e);
239                 }
240             }
241             LifecycleUtils.destroy(scheduler);
242             setSessionValidationScheduler(null);
243         }
244     }
245 
246     protected void beforeSessionValidationDisabled() {
247     }
248 
249     public void destroy() {
250         disableSessionValidation();
251     }
252 
253     /**
254      * @see ValidatingSessionManager#validateSessions()
255      */
256     public void validateSessions() {
257         if (log.isInfoEnabled()) {
258             log.info("Validating all active sessions...");
259         }
260 
261         int invalidCount = 0;
262 
263         Collection<Session> activeSessions = getActiveSessions();
264 
265         if (activeSessions != null && !activeSessions.isEmpty()) {
266             for (Session s : activeSessions) {
267                 try {
268                     validate(s);
269                 } catch (InvalidSessionException e) {
270                     if (log.isDebugEnabled()) {
271                         boolean expired = (e instanceof ExpiredSessionException);
272                         String msg = "Invalidated session with id [" + s.getId() + "]" +
273                                 (expired ? " (expired)" : " (stopped)");
274                         log.debug(msg);
275                     }
276                     invalidCount++;
277                 }
278             }
279         }
280 
281         if (log.isInfoEnabled()) {
282             String msg = "Finished session validation.";
283             if (invalidCount > 0) {
284                 msg += "  [" + invalidCount + "] sessions were stopped.";
285             } else {
286                 msg += "  No sessions were stopped.";
287             }
288             log.info(msg);
289         }
290     }
291 
292     protected abstract Collection<Session> getActiveSessions();
293 
294     public void validateSession(Serializable sessionId) {
295         //standard getSession call will validate, so just call the method:
296         getSession(sessionId);
297     }
