154     static FileFilter createFilter(final String extension) {
155         return new FileFilter() {
156             @Override
157             public boolean accept(File pathname) {
158                 String name = pathname.getName();
159                 if (name.endsWith(extension)) {
160                     return true;
161                 }
162                 return false;
163             }
164         };
165     }

198         int cut = name.length() - extension.length();

200                 ? name.substring(0, cut)
201                 : name;

206      * Suppose a path is provided which consists of a full path
207      * this method returns the sub path for a full path ex: /foo/bar/baz/foobar.z
208      * and the base path is /foo/bar it will will return baz/foobar.z.

210     private static String getEntryPath(String basePath, String fullPath) {
211         if (!fullPath.startsWith(basePath)) {

221     public static void recursiveCopy(File src, File dest) throws IOException {
222         if (!src.exists() || !src.canRead()) {
223             throw new IOException("file not found or readable: " + src);
224         }
225         if (dest.exists() && !dest.isDirectory() && !dest.canWrite()) {
226             throw new IOException("file not found or writeable: " + dest);
227         }
228         if (!dest.exists()) {
229             dest.mkdirs();
230         }
231         List<File> a = directoryList(src);
232         for (File f : a) {
233             copyFile(f, new File(dest, getEntryPath(src, f)));
234         }
235     }

237     static List<File> directoryList(File dirname) {
238         List<File>  dirList = new ArrayList<File>();
239         return directoryList(dirname, dirList, null);
240     }

242     private static List<File> directoryList(File dirname, List<File> dirList,
243             File[] dirs) {
244         dirList.addAll(Arrays.asList(dirname.listFiles(FILE_FILTER)));
245         dirs = dirname.listFiles(DIR_FILTER);
246         for (File f : dirs) {
247             if (f.isDirectory() && !f.equals(dirname)) {
248                 dirList.add(f);
249                 directoryList(f, dirList, dirs);
250             }
251         }
252         return dirList;
253     }

255     static void recursiveDelete(File dir) throws IOException {
256         if (dir.isFile()) {
257             dir.delete();
258         } else if (dir.isDirectory()) {
259             File[] entries = dir.listFiles();
260             for (int i = 0; i < entries.length; i++) {
261                 if (entries[i].isDirectory()) {
262                     recursiveDelete(entries[i]);
263                 }
264                 entries[i].delete();
265             }
266             dir.delete();
267         }
268     }

277      * finds files in the start directory using the the filter, appends
278      * the files to the dirList.

280     private static void findFiles0(File startDir, List<File> list,
281                                     FileFilter filter) throws IOException {
282         File[] foundFiles = startDir.listFiles(filter);
283         list.addAll(Arrays.asList(foundFiles));
284         File[] dirs = startDir.listFiles(DIR_FILTER);
285         for (File dir : dirs) {
286             findFiles0(dir, list, filter);
287         }

314     // given a jar file foo.jar will write to foo.pack
315     static void pack(JarFile jarFile, File packFile) throws IOException {
316         Pack200.Packer packer = Pack200.newPacker();
317         Map<String, String> p = packer.properties();
318         // Take the time optimization vs. space
319         p.put(packer.EFFORT, "1");  // CAUTION: do not use 0.
320         // Make the memory consumption as effective as possible
321         p.put(packer.SEGMENT_LIMIT, "10000");
322         // ignore all JAR deflation requests to save time
323         p.put(packer.DEFLATE_HINT, packer.FALSE);
324         // save the file ordering of the original JAR
325         p.put(packer.KEEP_FILE_ORDER, packer.TRUE);
326         FileOutputStream fos = null;
327         try {
328             // Write out to a jtreg scratch area
329             fos = new FileOutputStream(packFile);
330             // Call the packer
331             packer.pack(jarFile, fos);
332         } finally {
333             close(fos);
334         }
335     }

350     // given a packed file, create the jar file in the current directory.
351     private static void unpack0(File inFile, JarOutputStream jarStream,
352             boolean useJavaUnpack) throws IOException {
353         // Unpack the files
354         Pack200.Unpacker unpacker = Pack200.newUnpacker();
355         Map<String, String> props = unpacker.properties();
356         if (useJavaUnpack) {
357             props.put("com.sun.java.util.jar.pack.disable.native", "true");
358         }
359         // Call the unpacker
360         unpacker.unpack(inFile, jarStream);
361     }

363     static byte[] getBuffer(ZipFile zf, ZipEntry ze) throws IOException {
364         ByteArrayOutputStream baos = new ByteArrayOutputStream();
365         byte buf[] = new byte[8192];
366         InputStream is = null;
367         try {
368             is = zf.getInputStream(ze);
369             int n = is.read(buf);
370             while (n > 0) {
371                 baos.write(buf, 0, n);
372                 n = is.read(buf);
373             }
374             return baos.toByteArray();
375         } finally {
376             close(is);
377         }
378     }

390     static List<String> runExec(List<String> cmdsList, Map<String, String> penv) {
391         ArrayList<String> alist = new ArrayList<String>();
392         ProcessBuilder pb =
393                 new ProcessBuilder(cmdsList);
394         Map<String, String> env = pb.environment();
395         if (penv != null && !penv.isEmpty()) {
396             env.putAll(penv);
397         }
398         pb.directory(new File("."));
399         dirlist(new File("."));
400         for (String x : cmdsList) {
401             System.out.print(x + " ");
402         }
403         System.out.println("");
404         int retval = 0;
405         Process p = null;
406         InputStreamReader ir = null;
407         BufferedReader rd = null;
408         InputStream is = null;
409         try {
410             pb.redirectErrorStream(true);
411             p = pb.start();
412             is = p.getInputStream();
413             ir = new InputStreamReader(is);
414             rd = new BufferedReader(ir, 8192);
415 
416             String in = rd.readLine();
417             while (in != null) {
418                 alist.add(in);
419                 System.out.println(in);
420                 in = rd.readLine();
421             }
422             retval = p.waitFor();
423             if (retval != 0) {
424                 throw new RuntimeException("process failed with non-zero exit");
425             }
426         } catch (Exception ex) {
427             throw new RuntimeException(ex.getMessage());
428         } finally {
429             close(rd);
430             close(ir);
431             close(is);
432             if (p != null) {
433                 p.destroy();
434             }
435         }
436         return alist;
437     }

465     private static List<File> locaterCache = null;
466     // search the source dir and jdk dir for requested file and returns
467     // the first location it finds.
468     static File locateJar(String name) {
469         try {
470             if (locaterCache == null) {
471                 locaterCache = new ArrayList<File>();
472                 locaterCache.addAll(findFiles(TEST_SRC_DIR, createFilter(JAR_FILE_EXT)));
473                 locaterCache.addAll(findFiles(JavaSDK, createFilter(JAR_FILE_EXT)));
474             }
475             for (File f : locaterCache) {
476                 if (f.getName().equals(name)) {
477                     return f;
478                 }
479             }
480             throw new IOException("file not found: " + name);
481         } catch (IOException e) {
482             throw new RuntimeException(e);
483         }
484     }
