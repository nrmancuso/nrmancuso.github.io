23  /*
24   * @test
25   * @bug 5102289
26   * @summary Stress test for ResourceBundle.getBundle with ResourceBundle.Control.
27   * @run main/othervm -esa StressTest 2 15
28   */
29  
30  import java.util.*;
31  import java.util.concurrent.atomic.*;
32  
33  // Usage: java StressTest [threadsFactor [duration]]
34  public class StressTest {
35      static final Locale ROOT_LOCALE = new Locale("");
36      static final Random rand = new Random();
37      static final Locale[] locales = {
38          Locale.US,
39          Locale.CHINA,
40          ROOT_LOCALE,
41          Locale.JAPAN,
42          Locale.CANADA,
43          Locale.KOREA
44      };
45      static final String[] expected = {
46          "U.S.A.",
47          "China",
48          "U.S.A.",
49          "Japan",
50          "U.S.A.", // StressOut_en_CA.properties is empty.
51          "Korea"
52      };
53      final static long startTime = System.currentTimeMillis();
54  
55      // increment each element when one getBundle call is done.
56      static AtomicIntegerArray counters;
57      static int[] prevCounters;
58      static int intervalForCounterCheck;
59      static AtomicInteger clearCounter = new AtomicInteger();
60  
61      static volatile boolean runrun = true;
62  
63      public static void main(String[] args) {
64          int threadsFactor = 2;
65          if (args.length > 0) {
66              threadsFactor = Math.max(2, Integer.parseInt(args[0]));
67          }
68          int duration = 180;
69          if (args.length > 1) {
70              duration = Math.max(5, Integer.parseInt(args[1]));
71          }
72          Locale.setDefault(Locale.US);
73          Thread[] tasks = new Thread[locales.length * threadsFactor];
74          counters = new AtomicIntegerArray(tasks.length);
75  
76          for (int i = 0; i < tasks.length; i++) {
77              tasks[i] = new Thread(new Worker(i));
78          }
79          for (int i = 0; i < tasks.length; i++) {
80              tasks[i].start();
81          }
82  
83          int nProcessors = Runtime.getRuntime().availableProcessors();
84          intervalForCounterCheck = Math.max(tasks.length / nProcessors, 1);
85          System.out.printf("%d processors, intervalForCounterCheck = %d [sec]%n",
86                            nProcessors, intervalForCounterCheck);
87          try {
88              for (int i = 0; runrun && i < duration; i++) {
89                  Thread.sleep(1000); // 1 second
90                  if ((i % intervalForCounterCheck) == 0) {
91                      checkCounters();
92                  }
93              }
94              runrun = false;
95              for (int i = 0; i < tasks.length; i++) {
96                  tasks[i].join();
97              }
98          } catch (InterruptedException e) {
99          }
100 
101         printCounters();
102     }
103 
104     static void checkCounters() {
105         int length = counters.length();
106         int[] snapshot = new int[length];
107         for (int i = 0; i < length; i++) {
108             snapshot[i] = counters.get(i);
109         }
110 
111         if (prevCounters == null) {
112             prevCounters = snapshot;
113             return;
114         }
115 
116         for (int i = 0; i < length; i++) {
117             if (snapshot[i] > prevCounters[i]) {
118                 continue;
119             }
120             System.out.printf(
121                 "Warning: Thread #%d hasn't updated its counter for the last %d second(s).%n",
122                 i, intervalForCounterCheck);
123         }
124         prevCounters = snapshot;
125     }
126 
127     static void printCounters() {
128         long total = 0;
129         int min = Integer.MAX_VALUE;
130         int max = Integer.MIN_VALUE;
131         for (int i = 0; i < counters.length(); i++) {
132             int counter = counters.get(i);
133             total += counter;
134             min = Math.min(min, counter);
135             max = Math.max(max, counter);
136         }
137         System.out.printf("Total: %d calls, min=%d, max=%d, cache cleared %d times%n",
138                           total, min, max, clearCounter.get());
139     }
140 
141     static class Worker implements Runnable {
142         final int id;
143         final int index;
144         final Locale locale;
145         final String str;
146         final int max;
147         final boolean cleaner;
148         ResourceBundle.Control control;
149 
150         Worker(int i) {
151             id = i;
152             index = i % locales.length;
153             locale = locales[index];
154             cleaner = locale.equals(ROOT_LOCALE);
155             str = expected[index];
156             max = rand.nextInt((index + 1) * 500) + 1000;
157             control = new TestControl(max);
158             System.out.println("Worker" + i + ": locale="+locale+", expected="+str+
159                                ", max="+max);
160         }
161 
162         public void run() {
163             while (runrun) {
164                 ResourceBundle rb = ResourceBundle.getBundle("StressOut", locale, control);
165                 counters.incrementAndGet(id);
166                 String s = rb.getString("data");
167                 if (!s.equals(str)) {
168                     runrun = false;
169                     throw new RuntimeException(locale + ": rb.locale=" + rb.getLocale() +
170                                                ", got " + s + ", expected " + str);
171                 }
172                 try {
173                     Thread.sleep(rand.nextInt(max/500));
174                 } catch (InterruptedException e) {
175                 }
176                 if (cleaner && (rand.nextInt(10000) == 0)) {
177                     //System.out.println("Clearing cache!");
178                     ResourceBundle.clearCache();
179                     clearCounter.incrementAndGet();
180                 }
181             }
182         }
183 
184         static class TestControl extends ResourceBundle.Control {
185             int max;
186 
187             public List<Locale> getCandidateLocales(String baseName, Locale locale) {
188                 List<Locale> list = super.getCandidateLocales(baseName, locale);
189                 //System.out.println("Candidate locales=" + list);
190                 return list;
191             }
192             public TestControl(int max) {
193                 this.max = max;
194             }
195             public long getTimeToLive(String baseName, Locale locale) {
196                 // This will set TTL to a random value for each bundle.
197                 long ttl = rand.nextInt(max);
198                 //System.out.println("TTL: " + baseName + "_" + locale + " for " + ttl);
199                 return ttl;
200             }
201             public boolean needsReload(String baseName, Locale locale,
202                                        String format, ClassLoader loader,
203                                        ResourceBundle bundle, long loadTime) {
204                 //System.out.println("Expired: " + baseName + "_" + locale + "." + format +
205                 //                 " at " + (loadTime-startTime) + ", bundle=" + bundle);
206                 return rand.nextBoolean();
207             }
208         }
209     }
