32  package j2dbench.tests;
33  
34  import java.awt.Dimension;
35  import java.awt.Graphics;
36  
37  import j2dbench.Group;
38  import j2dbench.Option;
39  import j2dbench.TestEnvironment;
40  
41  public abstract class MiscTests extends GraphicsTests {
42      static Group miscroot;
43      static Group copytestroot;
44  
45      public MiscTests(Group parent, String nodeName, String description) {
46          super(parent, nodeName, description);
47      }
48  
49      public static void init() {
50          miscroot = new Group(graphicsroot, "misc",
51                               "Misc Benchmarks");
52          copytestroot = new Group(miscroot, "copytests",
53                                   "copyArea() Tests");
54  
55          new CopyArea("copyAreaVert", "Vertical copyArea()", 0, 1);
56          new CopyArea("copyAreaHoriz", "Horizontal copyArea()", 1, 0);
57          new CopyArea("copyAreaDiag", "Diagonal copyArea()", 1, 1);
58      }
59  
60      private static class CopyArea extends MiscTests {
61          private int dx, dy;
62  
63          CopyArea(String nodeName, String desc, int dx, int dy) {
64              super(copytestroot, nodeName, desc);
65              this.dx = dx;
66              this.dy = dy;
67          }
68  
69          public Dimension getOutputSize(int w, int h) {
70              // we add one to each dimension to avoid copying outside the
71              // bounds of the destination when "bounce" is enabled
72              return new Dimension(w+1, h+1);
73          }
74  
75          public void runTest(Object ctx, int numReps) {
76              GraphicsTests.Context gctx = (GraphicsTests.Context)ctx;
77              int size = gctx.size;
78              int x = gctx.initX;
79              int y = gctx.initY;
80              Graphics g = gctx.graphics;
81              g.translate(gctx.orgX, gctx.orgY);
82              if (gctx.animate) {
83                  do {
84                      g.copyArea(x, y, size, size, dx, dy);
85                      if ((x -= 3) < 0) x += gctx.maxX;
86                      if ((y -= 1) < 0) y += gctx.maxY;
87                  } while (--numReps > 0);
88              } else {
89                  do {
90                      g.copyArea(x, y, size, size, dx, dy);
91                  } while (--numReps > 0);
92              }
93              g.translate(-gctx.orgX, -gctx.orgY);
94          }
95      }
96  }
