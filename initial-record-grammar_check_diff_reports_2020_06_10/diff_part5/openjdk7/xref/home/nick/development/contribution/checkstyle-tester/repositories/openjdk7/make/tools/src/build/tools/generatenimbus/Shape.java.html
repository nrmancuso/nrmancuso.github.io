26  package build.tools.generatenimbus;

28  import java.util.ArrayList;
29  import java.util.List;
30  import javax.xml.bind.annotation.XmlAttribute;
31  import javax.xml.bind.annotation.XmlElement;
32  import javax.xml.bind.annotation.XmlElementWrapper;
33  import javax.xml.bind.annotation.XmlElements;

35  
36  public abstract class Shape {
37      @XmlElement
38      private PaintPoints paintPoints;
39      public double getPaintX1() { return paintPoints.x1; }
40      public double getPaintX2() { return paintPoints.x2; }
41      public double getPaintY1() { return paintPoints.y1; }
42      public double getPaintY2() { return paintPoints.y2; }
43  
44      @XmlElements({
45          @XmlElement(name = "matte", type = Matte.class),
46          @XmlElement(name = "gradient", type = Gradient.class),
47          @XmlElement(name = "radialGradient", type = RadialGradient.class)
48      })
49      private Paint paint;
50      public Paint getPaint() { return paint; }
51  
52      static class PaintPoints {
53          @XmlAttribute double x1;
54          @XmlAttribute double y1;
55          @XmlAttribute double x2;
56          @XmlAttribute double y2;
57      }
58  }
59  
60  class Point {
61      @XmlAttribute private double x;
62      public double getX() { return x; }
63  
64      @XmlAttribute private double y;
65      public double getY() { return y; }
66  
67      @XmlAttribute(name="cp1x") private double cp1x;
68      public double getCp1X() { return cp1x; }
69  
70      @XmlAttribute(name="cp1y") private double cp1y;
71      public double getCp1Y() { return cp1y; }
72  
73      @XmlAttribute(name="cp2x") private double cp2x;
74      public double getCp2X() { return cp2x; }
75  
76      @XmlAttribute(name="cp2y") private double cp2y;
77      public double getCp2Y() { return cp2y; }
78  
79      public boolean isP1Sharp() {
80          return cp1x == x && cp1y == y;
81      }
82  
83      public boolean isP2Sharp() {
84          return cp2x == x && cp2y == y;
85      }
86  }
87  
88  class Path extends Shape {
89      @XmlElement(name="point")
90      @XmlElementWrapper(name="points")
91      private List<Point> controlPoints = new ArrayList<Point>();
92      public List<Point> getControlPoints() { return controlPoints; }
93  }

95  class Rectangle extends Shape {
96      @XmlAttribute private double x1;
97      public double getX1() { return x1; }
98  
99      @XmlAttribute private double x2;
100     public double getX2() { return x2; }
101 
102     @XmlAttribute private double y1;
103     public double getY1() { return y1; }
104 
105     @XmlAttribute private double y2;
106     public double getY2() { return y2; }
107 
108     @XmlAttribute
109     public double getRounding() {
110         double rounding = Math.abs(roundingX - x1) * 2;
111         return rounding > 2 ? rounding : 0;
112     }
113 
114     public void setRounding(double rounding) {
115         if (rounding > 0 && rounding < 2) {
116             rounding = 0;
117         }
118         roundingX = rounding / 2d + x1;
119     }
120     private double roundingX;
121 
122     public boolean isRounded() {
123         return getRounding() > 0;
124     }
125 
126 }
