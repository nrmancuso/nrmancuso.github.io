54  public class GenericArrayTypeTest {
55      // A version of java.lang.management.MonitorInfo so we can run this test
56      // on JDK 5, where that class didn't exist.
57      public static class MonitorInfo {
58          private final String className;
59          private final int identityHashCode;
60          private final int lockedStackDepth;
61          private final StackTraceElement lockedStackFrame;
62  
63          public MonitorInfo(
64                  String className, int identityHashCode,
65                  int lockedStackDepth, StackTraceElement lockedStackFrame) {
66              this.className = className;
67              this.identityHashCode = identityHashCode;
68              this.lockedStackDepth = lockedStackDepth;
69              this.lockedStackFrame = lockedStackFrame;
70          }
71  
72          public static MonitorInfo from(CompositeData cd) {
73              try {
74                  CompositeData stecd = (CompositeData) cd.get("lockedStackFrame");
75                  StackTraceElement ste = new StackTraceElement(
76                          (String) stecd.get("className"),
77                          (String) stecd.get("methodName"),
78                          (String) stecd.get("fileName"),
79                          (Integer) stecd.get("lineNumber"));
80                  return new MonitorInfo(
81                          (String) cd.get("className"),
82                          (Integer) cd.get("identityHashCode"),
83                          (Integer) cd.get("lockedStackDepth"),
84                          ste);
85              } catch (Exception e) {
86                  throw new RuntimeException(e);
87              }
88          }
89  
90          public String getClassName() {
91              return className;
92          }
93  
94          public int getIdentityHashCode() {
95              return identityHashCode;
96          }
97  
98          public int getLockedStackDepth() {
99              return lockedStackDepth;
100         }
101 
102         public StackTraceElement getLockedStackFrame() {
103             return lockedStackFrame;
104         }
105     }

246     public static void main(String[] args) throws Exception {
247 
248         int error = 0;
249         JMXConnector cc = null;
250         JMXConnectorServer cs = null;
251 
252         try {
253             // Instantiate the MBean server
254             //
255             echo("\n>>> Create the MBean server");
256             MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
257 
258             // Get default domain
259             //
260             echo("\n>>> Get the MBean server's default domain");
261             String domain = mbs.getDefaultDomain();
262             echo("\tDefault Domain = " + domain);
263 
264             // Register TestMXBean
265             //
266             echo("\n>>> Register TestMXBean");
267             ObjectName name =
268                 ObjectName.getInstance(domain + ":type=" + TestMXBean.class);
269             mbs.createMBean(Test.class.getName(), name);
270 
271             // Create an RMI connector server
272             //
273             echo("\n>>> Create an RMI connector server");
274             JMXServiceURL url = new JMXServiceURL("service:jmx:rmi://");
275             cs =
276                 JMXConnectorServerFactory.newJMXConnectorServer(url, null, mbs);
277 
278             // Start the RMI connector server
279             //
280             echo("\n>>> Start the RMI connector server");
281             cs.start();
282 
283             // Create an RMI connector client
284             //
285             echo("\n>>> Create an RMI connector client");
286             cc = JMXConnectorFactory.connect(cs.getAddress(), null);
287             MBeanServerConnection mbsc = cc.getMBeanServerConnection();
288 
289             // Create TestMXBean proxy
290             //
291             echo("\n>>> Create the TestMXBean proxy");
292             TestMXBean test = JMX.newMXBeanProxy(mbsc, name, TestMXBean.class);
293 
294             // Play with proxy getters and setters
295             //
296             echo("\n>>> Play with proxy getters and setters");
297             String[] t1 = new String[] { "t1" };
298             MonitorInfo[] t2 = new MonitorInfo[] {
299                 new MonitorInfo("dummy", 0, 0,
300                     new StackTraceElement("dummy", "dummy", "dummy", 0)) };
301             Map<String,String[]> t3 = new HashMap<String,String[]>();
302             t3.put("key", t1);
303             Map<String,MonitorInfo[]> t4 = new HashMap<String,MonitorInfo[]>();
304             t4.put("key", t2);
305             Set<String[]> t5 = new HashSet<String[]>();
306             t5.add(t1);
307             Set<MonitorInfo[]> t6 = new HashSet<MonitorInfo[]>();
308             t6.add(t2);
309             List<String[]> t7 = new ArrayList<String[]>();
310             t7.add(t1);
311             List<MonitorInfo[]> t8 = new ArrayList<MonitorInfo[]>();
312             t8.add(t2);
313             Set<List<String[]>> t9 = new HashSet<List<String[]>>();
314             t9.add(t7);
315             Set<List<MonitorInfo[]>> t10 = new HashSet<List<MonitorInfo[]>>();
316             t10.add(t8);
317             Map<String,Set<List<String[]>>> t11 = new HashMap<String,Set<List<String[]>>>();
318             t11.put("key", t9);
319             Map<String,Set<List<MonitorInfo[]>>> t12 = new HashMap<String,Set<List<MonitorInfo[]>>>();
320             t12.put("key", t10);
321 
322             test.setT1(t1);
323             test.setT2(t2);
324             test.setT3(t3);
325             test.setT4(t4);
326             test.setT5(t5);
327             test.setT6(t6);
328             test.setT7(t7);
329             test.setT8(t8);
330             test.setT9(t9);
331             test.setT10(t10);
332             test.setT11(t11);
333             test.setT12(t12);
334 
335             String r;
336             String e1 = "t1";
337             String e2 = "dummy";
338             echo("\tT1 = " + test.getT1());
339             r = ((String[])test.getT1())[0];
340             echo("\tR1 = " + r);
341             if (!e1.equals(r)) error++;
342             echo("\tT2 = " + test.getT2());
343             r = ((MonitorInfo[])test.getT2())[0].getClassName();
344             echo("\tR2 = " + r);
345             if (!e2.equals(r)) error++;
346             echo("\tT3 = " + test.getT3());
347             r = ((String[])((Map<String,String[]>)test.getT3()).get("key"))[0];
348             echo("\tR3 = " + r);
349             if (!e1.equals(r)) error++;
350             echo("\tT4 = " + test.getT4());
351             r = ((MonitorInfo[])((Map<String,MonitorInfo[]>)test.getT4()).get("key"))[0].getClassName();
352             echo("\tR4 = " + r);
353             if (!e2.equals(r)) error++;
354             echo("\tT5 = " + test.getT5());
355             r = ((String[])((Set<String[]>)test.getT5()).iterator().next())[0];
356             echo("\tR5 = " + r);
357             if (!e1.equals(r)) error++;
358             echo("\tT6 = " + test.getT6());
359             r = ((MonitorInfo[])((Set<MonitorInfo[]>)test.getT6()).iterator().next())[0].getClassName();
360             echo("\tR6 = " + r);
361             if (!e2.equals(r)) error++;
362             echo("\tT7 = " + test.getT7());
363             r = ((String[])((List<String[]>)test.getT7()).get(0))[0];
364             echo("\tR7 = " + r);
365             if (!e1.equals(r)) error++;
366             echo("\tT8 = " + test.getT8());
367             r = ((MonitorInfo[])((List<MonitorInfo[]>)test.getT8()).get(0))[0].getClassName();
368             echo("\tR8 = " + r);
369             if (!e2.equals(r)) error++;
370             echo("\tT9 = " + test.getT9());
371             r = ((String[])((List<String[]>)((Set<List<String[]>>)test.getT9()).iterator().next()).get(0))[0];
372             echo("\tR9 = " + r);
373             if (!e1.equals(r)) error++;
374             echo("\tT10 = " + test.getT10());
375             r = ((MonitorInfo[])((List<MonitorInfo[]>)((Set<List<MonitorInfo[]>>)test.getT10()).iterator().next()).get(0))[0].getClassName();
376             echo("\tR10 = " + r);
377             if (!e2.equals(r)) error++;
378             echo("\tT11 = " + test.getT11());
379             r = ((String[])((List<String[]>)((Set<List<String[]>>)((Map<String,Set<List<String[]>>>)test.getT11()).get("key")).iterator().next()).get(0))[0];
380             echo("\tR11 = " + r);
381             if (!e1.equals(r)) error++;
382             echo("\tT12 = " + test.getT12());
383             r = ((MonitorInfo[])((List<MonitorInfo[]>)((Set<List<MonitorInfo[]>>)((Map<String,Set<List<MonitorInfo[]>>>)test.getT12()).get("key")).iterator().next()).get(0))[0].getClassName();
384             echo("\tR12 = " + r);
385             if (!e2.equals(r)) error++;
386         } catch (Exception e) {
387             e.printStackTrace(System.out);
388             error++;
389         } finally {
390             // Close client
391             //
392             echo("\n>>> Close the RMI connector client");
393             if (cc != null)
394                 cc.close();
395 
396             // Stop server
397             //
398             echo("\n>>> Stop the RMI connector server");
399             if (cs != null)
400                 cs.stop();
401 
402             echo("\n>>> Bye! Bye!");
403         }
404 
405         if (error > 0) {
406             echo("\nTest failed! " + error + " errors.\n");
407             throw new IllegalArgumentException("Test failed");
408         } else {
409             echo("\nTest passed!\n");
410         }
411     }
