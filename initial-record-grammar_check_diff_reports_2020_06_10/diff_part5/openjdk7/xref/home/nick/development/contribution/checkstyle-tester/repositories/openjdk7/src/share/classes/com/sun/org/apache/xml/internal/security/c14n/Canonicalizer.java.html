102     /**
103      * Constructor Canonicalizer
104      *
105      * @param algorithmURI
106      * @throws InvalidCanonicalizerException
107      */
108     private Canonicalizer(String algorithmURI)
109            throws InvalidCanonicalizerException {
110 
111         try {
112             Class implementingClass = getImplementingClass(algorithmURI);
113 
114             this.canonicalizerSpi =
115                 (CanonicalizerSpi) implementingClass.newInstance();
116             this.canonicalizerSpi.reset=true;
117         } catch (Exception e) {
118             Object exArgs[] = { algorithmURI };
119 
120             throw new InvalidCanonicalizerException(
121                "signature.Canonicalizer.UnknownCanonicalizer", exArgs);
122         }
123     }
124

140     /**
141      * Method register
142      *
143      * @param algorithmURI
144      * @param implementingClass
145      * @throws AlgorithmAlreadyRegisteredException
146      */
147     public static void register(String algorithmURI, String implementingClass)
148            throws AlgorithmAlreadyRegisteredException {
149 
150         // check whether URI is already registered
151         Class registeredClass = getImplementingClass(algorithmURI);
152 
153         if (registeredClass != null)  {
154             Object exArgs[] = { algorithmURI, registeredClass };
155 
156             throw new AlgorithmAlreadyRegisteredException(
157                 "algorithm.alreadyRegistered", exArgs);
158         }
159 
160         try {
161             _canonicalizerHash.put(algorithmURI, Class.forName(implementingClass));
162         } catch (ClassNotFoundException e) {
163             throw new RuntimeException("c14n class not found");
164         }
165     }

185     /**
186      * This method tries to canonicalize the given bytes. It's possible to even
187      * canonicalize non-wellformed sequences if they are well-formed after being
188      * wrapped with a <CODE>&gt;a&lt;...&gt;/a&lt;</CODE>.
189      *
190      * @param inputBytes
191      * @return the result of the conicalization.
192      * @throws CanonicalizationException
193      * @throws java.io.IOException
194      * @throws javax.xml.parsers.ParserConfigurationException
195      * @throws org.xml.sax.SAXException
196      */
197     public byte[] canonicalize(byte[] inputBytes)
198            throws javax.xml.parsers.ParserConfigurationException,
199                   java.io.IOException, org.xml.sax.SAXException,
200                   CanonicalizationException {
201 
202         ByteArrayInputStream bais = new ByteArrayInputStream(inputBytes);
203         InputSource in = new InputSource(bais);
204         DocumentBuilderFactory dfactory = DocumentBuilderFactory.newInstance();
205 
206         dfactory.setNamespaceAware(true);
207 
208         // needs to validate for ID attribute nomalization
209         dfactory.setValidating(true);
210 
211         DocumentBuilder db = dfactory.newDocumentBuilder();
212 
213         /*
214          * for some of the test vectors from the specification,
215          * there has to be a validatin parser for ID attributes, default
216          * attribute values, NMTOKENS, etc.
217          * Unfortunaltely, the test vectors do use different DTDs or
218          * even no DTD. So Xerces 1.3.1 fires many warnings about using
219          * ErrorHandlers.
220          *
221          * Text from the spec:
222          *
223          * The input octet stream MUST contain a well-formed XML document,
224          * but the input need not be validated. However, the attribute
225          * value normalization and entity reference resolution MUST be
226          * performed in accordance with the behaviors of a validating
227          * XML processor. As well, nodes for default attributes (declared
228          * in the ATTLIST with an AttValue but not specified) are created
229          * in each element. Thus, the declarations in the document type
230          * declaration are used to help create the canonical form, even
231          * though the document type declaration is not retained in the
232          * canonical form.
233          *
234          */
235         db.setErrorHandler(new com.sun.org.apache.xml.internal.security.utils
236             .IgnoreAllErrorHandler());
237 
238         Document document = db.parse(in);
239         byte result[] = this.canonicalizeSubtree(document);
240 
241         return result;
242     }
