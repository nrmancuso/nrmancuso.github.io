198     /// convenience fcn to create AffineTransform of appropriate type
199     private AffineTransform getAffineTransform( int transform ) {
200             /// ABP
201             AffineTransform at = new AffineTransform();
202             switch ( transform )
203             {
204             case SCALE:
205               at.setToScale( 1.5f, 1.5f ); break;
206             case ROTATE:
207               at.setToRotation( Math.PI / 6 ); break;
208             case SHEAR:
209               at.setToShear( 0.4f, 0 ); break;
210             case NONE:
211               break;
212             default:
213               //System.err.println( "Illegal G2 Transform Arg: " + transform);
214               break;
215             }
216 
217             return at;
218     }

225     public void setFontParams(String name, float size,
226                               int style, int transform) {
227         boolean fontModified = false;
228         if ( !name.equals( fontName ) || style != fontStyle )
229           fontModified = true;
230 
231         fontName = name;
232         fontSize = size;
233         fontStyle = style;
234         fontTransform = transform;
235 
236         /// Recreate the font as specified
237         testFont = new Font(fontName, fontStyle, (int)fontSize);
238         if ((float)((int)fontSize) != fontSize) {
239             testFont = testFont.deriveFont(fontSize);
240         }
241 
242         if ( fontTransform != NONE ) {
243             AffineTransform at = getAffineTransform( fontTransform );
244             testFont = testFont.deriveFont( at );
245         }
246         updateBackBuffer = true;
247         updateFontMetrics = true;
248         fc.repaint();
249         if ( fontModified ) {
250             /// Tell main panel to update the font info
251             updateFontInfo();
252             f2dt.fireUpdateFontInfo();
253         }
254     }

307         if ( printer == null ) {
308             printer = PrinterJob.getPrinterJob();
309             page = printer.defaultPage();

311         printMode = i;

314         if ( printer.printDialog() ) {
315             try {
316                 printer.print();

333     /// Obtains the information about selected font
334     private void updateFontInfo() {
335         int numGlyphs = 0, numCharsInRange = drawRange[1] - drawRange[0] + 1;
336         fontInfos[0] = "Font Face Name: " + testFont.getFontName();
337         fontInfos[1] = "Glyphs in This Range: ";
338 
339         if ( textToUse == RANGE_TEXT ) {
340             for ( int i = drawRange[0]; i < drawRange[1]; i++ )
341               if ( testFont.canDisplay( i ))
342                 numGlyphs++;
343             fontInfos[1] = fontInfos[1] + numGlyphs + " / " + numCharsInRange;
344         }
345         else
346           fontInfos[1] = null;
347     }

507         /// Sets the font, hints, according to the set parameters
508         private void setParams( Graphics2D g2 ) {
509             g2.setFont( testFont );
510             g2.setRenderingHint(KEY_TEXT_ANTIALIASING, antiAliasType);
511             g2.setRenderingHint(KEY_FRACTIONALMETRICS, fractionalMetricsType);
512             g2.setRenderingHint(KEY_TEXT_LCD_CONTRAST, lcdContrast);
513             /* I am preserving a somewhat dubious behaviour of this program.
514              * Outline text would be drawn anti-aliased by setting the
515              * graphics anti-aliasing hint if the text anti-aliasing hint
516              * was set. The dubious element here is that people simply
517              * using this program may think this is built-in behaviour
518              * but its not - at least not when the app explictly draws
519              * outline text.
520              * This becomes more dubious in cases such as "GASP" where the
521              * size at which text is AA'ed is not something you can easily
522              * calculate, so mimicing that behaviour isn't going to be easy.
523              * So I precisely preserve the behaviour : this is done only
524              * if the AA value is "ON". Its not applied in the other cases.
525              */
526             if (antiAliasType == VALUE_TEXT_ANTIALIAS_ON &&
527                 (drawMethod == TL_OUTLINE || drawMethod == GV_OUTLINE)) {
528                 g2.setRenderingHint(KEY_ANTIALIASING, VALUE_ANTIALIAS_ON);
529             } else {
530                 g2.setRenderingHint(KEY_ANTIALIASING, VALUE_ANTIALIAS_OFF);
531             }
532         }

534         /// Draws the grid (Used for unicode/glyph range drawing)
535         private void drawGrid( Graphics2D g2 ) {
536             int totalGridWidth = numCharAcross * gridWidth;
537             int totalGridHeight = numCharDown * gridHeight;
538 
539             g2.setColor( Color.black );
540             for ( int i = 0; i < numCharDown + 1; i++ )
541               g2.drawLine( canvasInset_X, i * gridHeight + canvasInset_Y,
542                            canvasInset_X + totalGridWidth, i * gridHeight + canvasInset_Y );
543             for ( int i = 0; i < numCharAcross + 1; i++ )
544               g2.drawLine( i * gridWidth + canvasInset_X, canvasInset_Y,
545                            i * gridWidth + canvasInset_X, canvasInset_Y + totalGridHeight );
546         }

548         /// Draws one character at time onto the canvas according to
549         /// the method requested (Used for RANGE_TEXT and ALL_GLYPHS)
550         public void modeSpecificDrawChar( Graphics2D g2, int charCode,
551                                           int baseX, int baseY ) {
552             GlyphVector gv;
553             int oneGlyph[] = { charCode };
554             char charArray[] = Character.toChars( charCode );
555 
556             FontRenderContext frc = g2.getFontRenderContext();
557             AffineTransform oldTX = g2.getTransform();
558 
559             /// Create GlyphVector to measure the exact visual advance
560             /// Using that number, adjust the position of the character drawn
561             if ( textToUse == ALL_GLYPHS )
562               gv = testFont.createGlyphVector( frc, oneGlyph );
563             else
564               gv = testFont.createGlyphVector( frc, charArray );
565             Rectangle2D r2d2 = gv.getPixelBounds(frc, 0, 0);
566             int shiftedX = baseX;
567             // getPixelBounds returns a result in device space.
568             // we need to convert back to user space to be able to
569             // calculate the shift as baseX is in user space.
570             try {
571                  double pt[] = new double[4];
572                  pt[0] = r2d2.getX();
573                  pt[1] = r2d2.getY();
574                  pt[2] = r2d2.getX()+r2d2.getWidth();
575                  pt[3] = r2d2.getY()+r2d2.getHeight();
576                  oldTX.inverseTransform(pt,0,pt,0,2);
577                  shiftedX = baseX - (int) ( pt[2] / 2 + pt[0] );
578             } catch (NoninvertibleTransformException e) {
579             }
580 
581             /// ABP - keep track of old tform, restore it later
582 
583             g2.translate( shiftedX, baseY );
584             g2.transform( getAffineTransform( g2Transform ) );
585 
586             if ( textToUse == ALL_GLYPHS )
587               g2.drawGlyphVector( gv, 0f, 0f );
588             else {
589                 if ( testFont.canDisplay( charCode ))
590                   g2.setColor( Color.black );
591                 else {
592                   g2.setColor( Color.lightGray );
593                 }
594 
595                 switch ( drawMethod ) {
596                   case DRAW_STRING:
597                     g2.drawString( new String( charArray ), 0, 0 );
598                     break;
599                   case DRAW_CHARS:
600                     g2.drawChars( charArray, 0, 1, 0, 0 );
601                     break;
602                   case DRAW_BYTES:
603                     if ( charCode > 0xff )
604                       throw new CannotDrawException( DRAW_BYTES_ERROR );
605                     byte oneByte[] = { (byte) charCode };
606                     g2.drawBytes( oneByte, 0, 1, 0, 0 );
607                     break;
608                   case DRAW_GLYPHV:
609                     g2.drawGlyphVector( gv, 0f, 0f );
610                     break;
611                   case TL_DRAW:
612                     TextLayout tl = new TextLayout( new String( charArray ), testFont, frc );
613                     tl.draw( g2, 0f, 0f );
614                     break;
615                   case GV_OUTLINE:
616                     r2d2 = gv.getVisualBounds();
617                     shiftedX = baseX - (int) ( r2d2.getWidth() / 2 + r2d2.getX() );
618                     g2.draw( gv.getOutline( 0f, 0f ));
619                     break;
620                   case TL_OUTLINE:
621                     r2d2 = gv.getVisualBounds();
622                     shiftedX = baseX - (int) ( r2d2.getWidth() / 2 + r2d2.getX() );
623                     TextLayout tlo =
624                       new TextLayout( new String( charArray ), testFont,
625                                       g2.getFontRenderContext() );
626                     g2.draw( tlo.getOutline( null ));
627                 }
628             }
629 
630             /// ABP - restore old tform
631             g2.setTransform ( oldTX );
632         }

634         /// Draws one line of text at given position
635         private void modeSpecificDrawLine( Graphics2D g2, String line,
636                                            int baseX, int baseY ) {
637             /// ABP - keep track of old tform, restore it later
638             AffineTransform oldTx = null;
639             oldTx = g2.getTransform();
640             g2.translate( baseX, baseY );
641             g2.transform( getAffineTransform( g2Transform ) );
642 
643             switch ( drawMethod ) {
644               case DRAW_STRING:
645                 g2.drawString( line, 0, 0 );
646                 break;
647               case DRAW_CHARS:
648                 g2.drawChars( line.toCharArray(), 0, line.length(), 0, 0 );
649                 break;
650               case DRAW_BYTES:
651                 try {
652                     byte lineBytes[] = line.getBytes( "ISO-8859-1" );
653                     g2.drawBytes( lineBytes, 0, lineBytes.length, 0, 0 );
654                 }
655                 catch ( Exception e ) {
656                     e.printStackTrace();
657                 }
658                 break;
659               case DRAW_GLYPHV:
660                 GlyphVector gv =
661                   testFont.createGlyphVector( g2.getFontRenderContext(), line );
662                 g2.drawGlyphVector( gv, (float) 0, (float) 0 );
663                 break;
664               case TL_DRAW:
665                 TextLayout tl = new TextLayout( line, testFont,

671                   testFont.createGlyphVector( g2.getFontRenderContext(), line );

673                 break;
674               case TL_OUTLINE:

676                   new TextLayout( line, testFont,
677                                   g2.getFontRenderContext() );
678                 AffineTransform at = new AffineTransform();
679                 g2.draw( tlo.getOutline( at ));
680             }
681 
682             /// ABP - restore old tform
683             g2.setTransform ( oldTx );

718         /// Resets the scrollbar to display correct range of text currently on screen
719         /// (This scrollbar is not part of a "ScrollPane". It merely simulates its effect by
720         ///  indicating the necessary area to be drawn within the panel.
721         ///  By doing this, it prevents creating gigantic panel when large text range,

755         /// Calculates the font's metrics that will be used for draw
756         private void calcFontMetrics( Graphics2D g2d, int w, int h ) {
757             FontMetrics fm;
758             Graphics2D g2 = (Graphics2D)g2d.create();
759 
760             /// ABP
761             if ( g2Transform != NONE && textToUse != FILE_TEXT ) {
762                 g2.setFont( g2.getFont().deriveFont( getAffineTransform( g2Transform )) );
763                 fm = g2.getFontMetrics();
764             }
765             else {
766                 fm = g2.getFontMetrics();
767             }
768 
769             maxAscent = fm.getMaxAscent();
770             maxDescent = fm.getMaxDescent();
771             if (maxAscent == 0) maxAscent = 10;
772             if (maxDescent == 0) maxDescent = 5;
773             if ( textToUse == RANGE_TEXT || textToUse == ALL_GLYPHS ) {
774                 /// Give slight extra room for each character
775                 maxAscent += 3;
776                 maxDescent += 3;
777                 gridWidth = fm.getMaxAdvance() + 6;
778                 gridHeight = maxAscent + maxDescent;
779                 if ( force16Cols )
780                   numCharAcross = 16;
781                 else
782                   numCharAcross = ( w - 10 ) / gridWidth;
783                 numCharDown = ( h - 10 ) / gridHeight;
784 
785                 canvasInset_X = ( w - numCharAcross * gridWidth ) / 2;
786                 canvasInset_Y = ( h - numCharDown * gridHeight ) / 2;
787                 if ( numCharDown == 0 || numCharAcross == 0 )
788                   throw new CannotDrawException( isPrinting ? CANT_FIT_PRINT : CANT_FIT_DRAW );
789 
790                 if ( !isPrinting )
791                   resetScrollbar( verticalBar.getValue() * numCharAcross );
792             }
793             else {
794                 maxDescent += fm.getLeading();
795                 canvasInset_X = 5;
796                 canvasInset_Y = 5;
797                 /// gridWidth and numCharAcross will not be used in this mode...
798                 gridHeight = maxAscent + maxDescent;
799                 numCharDown = ( h - canvasInset_Y * 2 ) / gridHeight;
800 
801                 if ( numCharDown == 0 )
802                   throw new CannotDrawException( isPrinting ? CANT_FIT_PRINT : CANT_FIT_DRAW );
803                 /// If this is text loaded from file, prepares the LineBreak'ed
804                 /// text layout at this point
805                 if ( textToUse == FILE_TEXT ) {
806                     if ( !isPrinting )
807                       f2dt.fireChangeStatus( "LineBreaking Text... Please Wait", false );
808                     lineBreakTLs = new Vector();
809                     for ( int i = 0; i < fileText.length; i++ ) {
810                         AttributedString as =
811                           new AttributedString( fileText[i], g2.getFont().getAttributes() );
812 
813                         LineBreakMeasurer lbm =
814                           new LineBreakMeasurer( as.getIterator(), g2.getFontRenderContext() );
815 
816                         while ( lbm.getPosition() < fileText[i].length() )
817                           lineBreakTLs.add( lbm.nextLayout( (float) w ));
818 
819                     }
820                 }
821                 if ( !isPrinting )
822                   resetScrollbar( verticalBar.getValue() );
823             }
824         }

826         /// Calculates the amount of text that will be displayed on screen
827         private void calcTextRange() {
828             String displaying = null;
829 
830             if ( textToUse == RANGE_TEXT || textToUse == ALL_GLYPHS ) {
831                 if ( isPrinting )
832                   if ( printMode == ONE_PAGE )
833                     drawStart = currentlyShownChar;
834                   else /// printMode == CUR_RANGE
835                     drawStart = numCharAcross * numCharDown * printPageNumber;
836                 else
837                   drawStart = verticalBar.getValue() * numCharAcross;
838                 if ( textToUse == RANGE_TEXT ) {
839                     drawStart += drawRange[0];
840                     drawLimit = drawRange[1];
841                 }
842                 else
843                   drawLimit = testFont.getNumGlyphs();
844                 drawEnd = drawStart + numCharAcross * numCharDown - 1;
845 
846                 if ( drawEnd >= drawLimit )
847                   drawEnd = drawLimit;
848             }
849             else {
850                 if ( isPrinting )
851                   if ( printMode == ONE_PAGE )
852                     drawStart = currentlyShownChar;
853                   else /// printMode == ALL_TEXT
854                     drawStart = numCharDown * printPageNumber;
855                 else {
856                     drawStart = verticalBar.getValue();
857                 }
858 
859                 drawEnd = drawStart + numCharDown - 1;
860 
861                 if ( textToUse == USER_TEXT )
862                   drawLimit = userText.length - 1;
863                 else
864                   drawLimit = lineBreakTLs.size() - 1;
865 
866                 if ( drawEnd >= drawLimit )
867                   drawEnd = drawLimit;
868             }
869 
870             // ABP
871             if ( drawStart > drawEnd ) {
872               drawStart = 0;
873               verticalBar.setValue(drawStart);
874             }
875 
876 
877             /// Change the status bar if not printing...
878             if ( !isPrinting ) {
879                 backupStatusString = ( "Displaying" + MS_OPENING[textToUse] +
880                                        modeSpecificNumStr( drawStart ) + " to " +
881                                        modeSpecificNumStr( drawEnd ) +
882                                        MS_CLOSING[textToUse] );
883                 f2dt.fireChangeStatus( backupStatusString, false );
884             }
885         }

887         /// Draws text according to the parameters set by Font2DTest GUI
888         private void drawText( Graphics g, int w, int h ) {
889             Graphics2D g2;
890 
891             /// Create back buffer when not printing, and its Graphics2D
892             /// Then set drawing parameters for that Graphics2D object
893             if ( isPrinting )
894               g2 = (Graphics2D) g;
895             else  {
896                 backBuffer = (BufferedImage) this.createImage( w, h );
897                 g2 = backBuffer.createGraphics();
898                 g2.setColor(Color.white);
899                 g2.fillRect(0, 0, w, h);
900                 g2.setColor(Color.black);
901             }
902 
903             /// sets font, RenderingHints.
904             setParams( g2 );
905 
906             /// If flag is set, recalculate fontMetrics and reset the scrollbar
907             if ( updateFontMetrics || isPrinting ) {
908                 /// NOTE: re-calculates in case G2 transform
909                 /// is something other than NONE
910                 calcFontMetrics( g2, w, h );
911                 updateFontMetrics = false;
912             }
913             /// Calculate the amount of text that can be drawn...
914             calcTextRange();
915 
916             /// Draw according to the set "Text to Use" mode
917             if ( textToUse == RANGE_TEXT || textToUse == ALL_GLYPHS ) {
918                 int charToDraw = drawStart;
919                 if ( showGrid )
920                   drawGrid( g2 );
921                 if ( !isPrinting )
922                   g.drawImage( backBuffer, 0, 0, this );
923 
924                 for ( int i = 0; i < numCharDown && charToDraw <= drawEnd; i++ ) {
925                   for ( int j = 0; j < numCharAcross && charToDraw <= drawEnd; j++, charToDraw++ ) {
926                       int gridLocX = j * gridWidth + canvasInset_X;
927                       int gridLocY = i * gridHeight + canvasInset_Y;
928 
929                       modeSpecificDrawChar( g2, charToDraw,
930                                             gridLocX + gridWidth / 2,
931                                             gridLocY + maxAscent );
932                       //if ( !isPrinting ) {
933                       //    g.setClip( gridLocX, gridLocY, gridWidth + 1, gridHeight + 1 );
934                       //    g.drawImage( backBuffer, 0, 0, this );
935                             //}
936 
937                   }
938                 }
939             }
940             else if ( textToUse == USER_TEXT ) {
941                 g2.drawRect( 0, 0, w - 1, h - 1 );
942                 if ( !isPrinting )
943                   g.drawImage( backBuffer, 0, 0, this );
944 
945                 for ( int i = drawStart; i <= drawEnd; i++ ) {
946                     int lineStartX = canvasInset_Y;
947                     int lineStartY = ( i - drawStart ) * gridHeight + maxAscent;
948                     modeSpecificDrawLine( g2, userText[i], lineStartX, lineStartY );
949                 }
950             }
951             else {
952                 float xPos, yPos = (float) canvasInset_Y;
953                 g2.drawRect( 0, 0, w - 1, h - 1 );
954                 if ( !isPrinting )
955                   g.drawImage( backBuffer, 0, 0, this );
956 
957                 for ( int i = drawStart; i <= drawEnd; i++ ) {
958                     TextLayout oneLine = (TextLayout) lineBreakTLs.elementAt( i );
959                     xPos =
960                       oneLine.isLeftToRight() ?
961                       canvasInset_X : ( (float) w - oneLine.getAdvance() - canvasInset_X );
962 
963                     float fmData[] = {0, oneLine.getAscent(), 0, oneLine.getDescent(), 0, oneLine.getLeading()};
964                     if (g2Transform != NONE) {
965                         AffineTransform at = getAffineTransform(g2Transform);
966                         at.transform( fmData, 0, fmData, 0, 3);
967                     }
968                     //yPos += oneLine.getAscent();
969                     yPos += fmData[1]; // ascent
970                     //oneLine.draw( g2, xPos, yPos );
971                     tlDrawLine( g2, oneLine, xPos, yPos );
972                     //yPos += oneLine.getDescent() + oneLine.getLeading();
973                     yPos += fmData[3] + fmData[5]; // descent + leading
974                 }
975             }
976                 if ( !isPrinting )
977                 g.drawImage( backBuffer, 0, 0, this );
978             g2.dispose();
979         }

1005         /// Printable interface function
1006         /// Component print function...
1007         public int print( Graphics g, PageFormat pf, int pageIndex ) {
1008             if ( pageIndex == 0 ) {
1009                 /// Reset the last page index to max...
1010                 lastPage = Integer.MAX_VALUE;
1011                 currentlyShownChar = verticalBar.getValue() * numCharAcross;
1012             }
1013 
1014             if ( printMode == ONE_PAGE ) {

1016                   return NO_SUCH_PAGE;
1017             }
1018             else {
1019                 if ( pageIndex > lastPage )
1020                   return NO_SUCH_PAGE;
1021             }
1022 
1023             int pageWidth = (int) pf.getImageableWidth();
1024             int pageHeight = (int) pf.getImageableHeight();
1025             /// Back up metrics and other drawing info before printing modifies it
1026             int backupDrawStart = drawStart, backupDrawEnd = drawEnd;
1027             int backupNumCharAcross = numCharAcross, backupNumCharDown = numCharDown;
1028             Vector backupLineBreakTLs = null;
1029             if ( textToUse == FILE_TEXT )
1030               backupLineBreakTLs = (Vector) lineBreakTLs.clone();

1034             /// Push the actual draw area 60 down to allow info to be printed
1035             g.translate( (int) pf.getImageableX(), (int) pf.getImageableY() + 60 );
1036             try {
1037                 drawText( g, pageWidth, pageHeight - 60 );
1038             }
1039             catch ( CannotDrawException e ) {
1040                 f2dt.fireChangeStatus( ERRORS[ e.id ], true );
1041                 return NO_SUCH_PAGE;
1042             }
1043

1045             String hints = ( " with antialias " + antiAliasType + "and" +
1046                              " fractional metrics " + fractionalMetricsType +
1047                              " and lcd contrast = " + lcdContrast);
1048             String infoLine1 = ( "Printing" + MS_OPENING[textToUse] +
1049                                  modeSpecificNumStr( drawStart ) + " to " +
1050                                  modeSpecificNumStr( drawEnd ) + MS_CLOSING[textToUse] );
1051             String infoLine2 = ( "With " + fontName + " " + STYLES[fontStyle] + " at " +
1052                                  fontSize + " point size " + TRANSFORMS[fontTransform] );
1053             String infoLine3 = "Using " + METHODS[drawMethod] + hints;
1054             String infoLine4 = "Page: " + ( pageIndex + 1 );
1055             g.setFont( new Font( "dialog", Font.PLAIN, 12 ));
1056             g.setColor( Color.black );
1057             g.translate( 0, -60 );
1058             g.drawString( infoLine1, 15, 10 );
1059             g.drawString( infoLine2, 15, 22 );
1060             g.drawString( infoLine3, 15, 34 );
1061             g.drawString( infoLine4, 15, 46 );
1062 
1063             if ( drawEnd == drawLimit )
1064               /// This indicates that the draw will be completed with this page
1065               lastPage = pageIndex;
1066 
1067             /// Restore the changed values back...
1068             /// This is important for JScrollBar settings and LineBreak'ed TLs
1069             drawStart = backupDrawStart;
1070             drawEnd = backupDrawEnd;
1071             numCharAcross = backupNumCharAcross;
1072             numCharDown = backupNumCharDown;
1073             if ( textToUse == FILE_TEXT )
1074               lineBreakTLs = backupLineBreakTLs;
1075             return PAGE_EXISTS;
1076         }

1089         /// Figures out whether a character at the pointer location is valid
1090         /// And if so, updates mouse location informations, as well as
1091         /// the information on the status bar
1092         private boolean checkMouseLoc( MouseEvent e ) {
1093             if ( gridWidth != 0 && gridHeight != 0 )
1094               if ( textToUse == RANGE_TEXT || textToUse == ALL_GLYPHS ) {
1095                   int charLocX = ( e.getX() - canvasInset_X ) / gridWidth;
1096                   int charLocY = ( e.getY() - canvasInset_Y ) / gridHeight;
1097 
1098                   /// Check to make sure the mouse click location is within drawn area
1099                   if ( charLocX >= 0 && charLocY >= 0 &&
1100                        charLocX < numCharAcross && charLocY < numCharDown ) {
1101                       int mouseOverChar =
1102                         charLocX + ( verticalBar.getValue() + charLocY ) * numCharAcross;
1103                       if ( textToUse == RANGE_TEXT )
1104                         mouseOverChar += drawRange[0];
1105                       if ( mouseOverChar > drawEnd )
1106                         return false;
1107 
1108                       mouseOverCharX = charLocX;
1109                       mouseOverCharY = charLocY;
1110                       currMouseOverChar = mouseOverChar;
1111                       /// Update status bar
1112                       f2dt.fireChangeStatus( "Pointing to" + MS_OPENING[textToUse] +
1113                                              modeSpecificNumStr( mouseOverChar ), false );
1114                       return true;
1115                   }
1116               }
1117             return false;
1118         }

1120         /// Shows (updates) the character zoom window
1121         public void showZoomed() {
1122             GlyphVector gv;
1123             Font backup = testFont;
1124             Point canvasLoc = this.getLocationOnScreen();
1125 
1126             /// Calculate the zoom area's location and size...
1127             int dialogOffsetX = (int) ( gridWidth * ( ZOOM - 1 ) / 2 );
1128             int dialogOffsetY = (int) ( gridHeight * ( ZOOM - 1 ) / 2 );
1129             int zoomAreaX =
1130               mouseOverCharX * gridWidth + canvasInset_X - dialogOffsetX;
1131             int zoomAreaY =
1132               mouseOverCharY * gridHeight + canvasInset_Y - dialogOffsetY;
1133             int zoomAreaWidth = (int) ( gridWidth * ZOOM );
1134             int zoomAreaHeight = (int) ( gridHeight * ZOOM );

1137             zoomWindow.setLocation( canvasLoc.x + zoomAreaX, canvasLoc.y + zoomAreaY );
1138             if ( !nowZooming ) {

1140                   /// If this is not opened as a "secure" window,
1141                   /// it has a banner below the zoom dialog which makes it look really BAD
1142                   /// So enlarge it by a bit
1143                   zoomWindow.setSize( zoomAreaWidth + 1, zoomAreaHeight + 20 );
1144                 else
1145                   zoomWindow.setSize( zoomAreaWidth + 1, zoomAreaHeight + 1 );
1146             }
1147 
1148             /// Prepare zoomed image
1149             zoomImage =
1150               (BufferedImage) zoomWindow.createImage( zoomAreaWidth + 1,
1151                                                       zoomAreaHeight + 1 );

1154             setParams( g2 );
1155             g2.setColor( Color.white );
1156             g2.fillRect( 0, 0, zoomAreaWidth, zoomAreaHeight );
1157             g2.setColor( Color.black );
1158             g2.drawRect( 0, 0, zoomAreaWidth, zoomAreaHeight );
1159             modeSpecificDrawChar( g2, currMouseOverChar,
1160                                   zoomAreaWidth / 2, (int) ( maxAscent * ZOOM ));
1161             g2.dispose();
1162             if ( !nowZooming )
1163               zoomWindow.show();
1164             /// This is sort of redundant... since there is a paint function
1165             /// inside zoomWindow definition that does the drawImage.
1166             /// (I should be able to call just repaint() here)
1167             /// However, for some reason, that paint function fails to respond
1168             /// from second time and on; So I have to force the paint here...
1169             zoomWindow.getGraphics().drawImage( zoomImage, 0, 0, this );
1170 
1171             nowZooming = true;
1172             prevZoomChar = currMouseOverChar;
1173             testFont = backup;
1174 
1175             // Windows does not repaint correctly, after
1176             // a zoom. Thus, we need to force the canvas
1177             // to repaint, but only once. After the first repaint,
1178             // everything stabilizes. [ABP]
1179             if ( firstTime() ) {
1180                 refresh();
1181             }
1182         }
