4     @summary verifies that Robot is accepting extra mouse buttons
5     @author Andrei Dmitriev : area=awt.mouse

10  
11  //if we do robot.mousePress(InputEvent.BUTTON1_DOWN_MASK) the test must
12  // 1) accept it (i.e. don't throw an IllegalArgumentException
13  // 2) actually post a MouseEvent
14  // Also, Robot should still accept InputEvent.BUTTONx_MASKs
15  
16  import java.awt.*;
17  import java.awt.event.*;
18  import sun.awt.SunToolkit;
19  import test.java.awt.regtesthelpers.Util;
20  
21  public class AcceptExtraMouseButtons extends Frame {
22      static String tk = Toolkit.getDefaultToolkit().getClass().getName();
23      static Robot robot;
24      static int [] standardButtonMasks = {InputEvent.BUTTON1_MASK,
25                                           InputEvent.BUTTON2_MASK,
26                                           InputEvent.BUTTON3_MASK};
27      static int [] buttonsPressed;
28      static int [] buttonsReleased;
29      static int [] buttonsClicked;
30  
31      static int buttonsNum = MouseInfo.getNumberOfButtons();
32  
33      public static void main(String []s){
34  
35          //MouseInfo.getNumberOfButtons() reports two more buttons on XToolkit
36          //as they reserved for wheel (both directions).
37          if (tk.equals("sun.awt.X11.XToolkit") || tk.equals("sun.awt.motif.MToolkit")) {
38              buttonsNum = buttonsNum - 2;
39          }
40          System.out.println("Number Of Buttons = "+ buttonsNum);
41          if (buttonsNum < 3) {
42              System.out.println("Linux and Windows systems should emulate three buttons if even there are only 1 or 2 are phsically available. Setting number of buttons to 3.");
43              buttonsNum = 3;
44          }
45  
46          buttonsPressed = new int [buttonsNum];
47          buttonsReleased = new int [buttonsNum];
48          buttonsClicked = new int [buttonsNum];
49  
50          AcceptExtraMouseButtons frame = new AcceptExtraMouseButtons();
51  
52          MouseAdapter ma1 = new MouseAdapter() {
53                  public void mousePressed(MouseEvent e) {
54                      buttonsPressed[e.getButton() - 1] += 1;
55                      System.out.println("PRESSED "+e);
56                  }
57                  public void mouseReleased(MouseEvent e) {
58                      buttonsReleased[e.getButton() - 1] += 1;
59                      System.out.println("RELEASED "+e);
60                  }
61                  public void mouseClicked(MouseEvent e) {
62                      buttonsClicked[e.getButton() - 1] += 1;
63                      System.out.println("CLICKED "+e);
64                  }
65              };
66          frame.addMouseListener(ma1);
67  
68          frame.setSize(300, 300);
69          frame.setVisible(true);
70  
71          Util.waitForIdle(robot);  //a time to show a Frame
72  
73          try {
74              robot = new Robot();
75              robot.delay(1000);
76              robot.mouseMove(frame.getLocationOnScreen().x + frame.getWidth()/2,
77                              frame.getLocationOnScreen().y + frame.getHeight()/2);
78  
79              //TestCase 1: verify that all BUTTONx_DOWN_MASKs are accepted by the Robot.
80  
81              for (int i = 0; i < buttonsNum; i++){
82                  int buttonMask = InputEvent.getMaskForButton(i+1);
83                  System.out.println("button to press = " +(i+1) + " : value passed to robot = " +buttonMask);
84                  robot.mousePress(buttonMask);
85                  robot.delay(30);
86                  robot.mouseRelease(buttonMask);
87                  Util.waitForIdle(robot);
88              }
89              for (int i = 0; i < buttonsNum; i++){
90                  if (buttonsPressed[i] != 1 || buttonsReleased[i] != 1 || buttonsClicked[i] !=1 ) {
91                      throw new RuntimeException("TESTCASE 1 FAILED : button " + (i+1) + " wasn't single pressed|released|clicked : "+ buttonsPressed[i] +" : "+ buttonsReleased[i] +" : "+ buttonsClicked[i]);
92                  }
93              }
94  
95              java.util.Arrays.fill(buttonsPressed, 0);
96              java.util.Arrays.fill(buttonsReleased, 0);
97              java.util.Arrays.fill(buttonsClicked, 0);
98              //TestCase 2: verify that all BUTTONx_MASKs are accepted by the Robot.
99              for (int i = 0; i < standardButtonMasks.length; i++){
100                 int buttonMask = standardButtonMasks[i];
101                 System.out.println("button to press = " +(i+1) + " : value passed to robot = " +buttonMask);
102                 robot.mousePress(buttonMask);
103                 robot.delay(30);
104                 robot.mouseRelease(buttonMask);
105                 Util.waitForIdle(robot);
106             }
107             for (int i = 0; i < standardButtonMasks.length; i++){
108                 if (buttonsPressed[i] != 1 || buttonsReleased[i] != 1 || buttonsClicked[i] !=1 ) {
109                     throw new RuntimeException("TESTCASE 2 FAILED : button " + (i+1) + " wasn't single pressed|released|clicked : "+ buttonsPressed[i] +" : "+ buttonsReleased[i] +" : "+ buttonsClicked[i]);
110                 }
111             }
112 
113         } catch (Exception e){
114             e.printStackTrace();
115             throw new RuntimeException(e);
116         }
117     }
118 }
