84      @Test
85      public void shouldMemoize() throws Throwable {
86          final AtomicInteger integer = new AtomicInteger();
87          final CheckedFunction8<Integer, Integer, Integer, Integer, Integer, Integer, Integer, Integer, Integer> f = (i1, i2, i3, i4, i5, i6, i7, i8) -> i1 + i2 + i3 + i4 + i5 + i6 + i7 + i8 + integer.getAndIncrement();
88          final CheckedFunction8<Integer, Integer, Integer, Integer, Integer, Integer, Integer, Integer, Integer> memo = f.memoized();
89          // should apply f on first apply()
90          final int expected = memo.apply(1, 2, 3, 4, 5, 6, 7, 8);
91          // should return memoized value of second apply()
92          assertThat(memo.apply(1, 2, 3, 4, 5, 6, 7, 8)).isEqualTo(expected);
93          // should calculate new values when called subsequently with different parameters
94          assertThat(memo.apply(2 , 3 , 4 , 5 , 6 , 7 , 8 , 9 )).isEqualTo(2  + 3  + 4  + 5  + 6  + 7  + 8  + 9  + 1);
95          // should return memoized value of second apply() (for new value)
96          assertThat(memo.apply(2 , 3 , 4 , 5 , 6 , 7 , 8 , 9 )).isEqualTo(2  + 3  + 4  + 5  + 6  + 7  + 8  + 9  + 1);
97      }

136         final MessageDigest md5 = recover.apply("M", "D", "5", "", "", "", "", "");
137         assertThat(md5).isNotNull();

139         assertThat(md5.getDigestLength()).isEqualTo(16);
140         final Try<MessageDigest> unknown = Function8.liftTry(recover).apply("U", "n", "k", "n", "o", "w", "n", "");

143         assertThat(unknown.getCause()).isNotNull().isInstanceOf(NullPointerException.class);
144         assertThat(unknown.getCause().getMessage()).isNotEmpty().isEqualToIgnoringCase("recover return null for class java.security.NoSuchAlgorithmException: Unknown MessageDigest not available");
