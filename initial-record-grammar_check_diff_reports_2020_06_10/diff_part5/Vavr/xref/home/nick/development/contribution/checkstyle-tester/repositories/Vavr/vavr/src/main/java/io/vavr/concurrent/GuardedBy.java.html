7   package io.vavr.concurrent;
8   
9   import java.lang.annotation.Documented;
10  import java.lang.annotation.Retention;
11  import java.lang.annotation.Target;
12  
13  import static java.lang.annotation.ElementType.FIELD;
14  import static java.lang.annotation.ElementType.METHOD;
15  import static java.lang.annotation.RetentionPolicy.RUNTIME;
16  
17  /**
18   * <strong>INTERNAL API - Used for documentation purpose only.</strong>
19   * <p>
20   * An annotated field or method must only be accessed when holding the lock specified in {@code value()}.
21   * <p>
22   * This is an annotation known from <a href="http://jcip.net">Java Concurrency in Practice</a>.
23   * See also <a href="https://jcp.org/en/jsr/detail?id=305">JSR 305</a>
24   *
25   * @author Daniel Dietrich
26   */
27  @Documented
28  @Target(value = { FIELD, METHOD })
29  @Retention(RUNTIME)
30  @interface GuardedBy {
31  
32      /**
33       * Specifies the lock that guards the annotated field or method.
34       *
35       * @return a valid lock that guards the annotated field or method
36       */
37      String value();
38  }
