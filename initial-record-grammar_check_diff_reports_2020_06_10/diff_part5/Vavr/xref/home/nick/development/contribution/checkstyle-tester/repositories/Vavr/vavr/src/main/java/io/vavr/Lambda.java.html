7   package io.vavr;
8   
9   import java.io.Serializable;
10  import java.util.Map;
11  
12  /**
13   * <strong>INTERNAL API - This class is subject to change.</strong>
14   * <p>
15   * This is a general definition of a (checked/unchecked) function of unknown parameters and a return type R.
16   * <p>
17   * A checked function may throw an exception. The exception type cannot be expressed as a generic type parameter
18   * because Java cannot calculate type bounds on function composition.
19   *
20   * @param <R> Return type of the function.
21   * @author Daniel Dietrich
22   */
23  interface Lambda<R> extends Serializable {
24  
25      /**
26       * The <a href="https://docs.oracle.com/javase/8/docs/api/index.html">serial version uid</a>.
27       */
28      long serialVersionUID = 1L;
29  
30      /**
31       * @return the number of function arguments.
32       * @see <a href="http://en.wikipedia.org/wiki/Arity">Arity</a>
33       */
34      int arity();
35  
36      /**
37       * Returns a curried version of this function.
38       *
39       * @return a curried function equivalent to this.
40       */
41      Lambda<?> curried();
42  
43      /**
44       * Returns a tupled version of this function.
45       *
46       * @return a tupled function equivalent to this.
47       */
48      Lambda<R> tupled();
49  
50      /**
51       * Returns a reversed version of this function. This may be useful in a recursive context.
52       *
53       * @return a reversed function equivalent to this.
54       */
55      Lambda<R> reversed();
56  
57      /**
58       * Returns a memoizing version of this function, which computes the return value for given arguments only one time.
59       * On subsequent calls given the same arguments the memoized value is returned.
60       * <p>
61       * Please note that memoizing functions do not permit {@code null} as single argument or return value.
62       *
63       * @return a memoizing function equivalent to this.
64       */
65      Lambda<R> memoized();
66  
67      /**
68       * Checks if this function is memoizing (= caching) computed values.
69       *
70       * @return true, if this function is memoizing, false otherwise
71       */
72      default boolean isMemoized() {
73          return this instanceof Memoized;
74      }
75  
76      /**
77       * Zero Abstract Method (ZAM) interface for marking functions as memoized using intersection types.
78       */
79      interface Memoized {
80          static <T extends Tuple, R> R of(Map<T, R> cache, T key, Function1<T, R> tupled) {
81              synchronized (cache) {
82                  if (cache.containsKey(key)) {
83                      return cache.get(key);
84                  } else {
85                      final R value = tupled.apply(key);
86                      cache.put(key, value);
87                      return value;
88                  }
89              }
90          }
91      }
92  }
