7   package io.vavr.collection;

9   import org.junit.Test;
10  
11  import java.math.BigDecimal;
12  
13  public abstract class AbstractLinearSeqTest extends AbstractSeqTest {
14  
15      @Override
16      abstract protected <T> LinearSeq<T> of(T element);
17  
18      // -- static narrow
19  
20      @Test
21      public void shouldNarrowIndexedSeq() {
22          final LinearSeq<Double> doubles = of(1.0d);
23          final LinearSeq<Number> numbers = LinearSeq.narrow(doubles);
24          final int actual = numbers.append(new BigDecimal("2.0")).sum().intValue();
25          assertThat(actual).isEqualTo(3);
26      }
