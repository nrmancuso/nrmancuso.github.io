58  	@SuppressWarnings("unchecked")
59  	@Test
60  	public <F> void testRemoteDirWithEmptyString() throws Exception {
61  		SessionFactory<F> sf = mock(SessionFactory.class);
62  		Session<F> session = mock(Session.class);
63  
64  		when(sf.getSession()).thenReturn(session);
65  		doAnswer(invocation -> {
66  			String path =  invocation.getArgument(1);
67  			assertThat(path.startsWith("/")).isFalse();
68  			return null;
69  		}).when(session).rename(Mockito.anyString(), Mockito.anyString());
70  		ExpressionParser parser = new SpelExpressionParser();
71  		FileTransferringMessageHandler<F> handler = new FileTransferringMessageHandler<F>(sf);
72  		handler.setRemoteDirectoryExpression(parser.parseExpression("''"));
73  		handler.setBeanFactory(mock(BeanFactory.class));
74  		handler.afterPropertiesSet();
75  		handler.handleMessage(new GenericMessage<String>("hello"));
76  		verify(session, times(1)).write(Mockito.any(InputStream.class), Mockito.anyString());
77  	}

80  	@Test
81  	public <F> void testTemporaryRemoteDir() throws Exception {
82  		SessionFactory<F> sf = mock(SessionFactory.class);
83  		Session<F> session = mock(Session.class);
84  
85  		final AtomicReference<String> temporaryPath = new AtomicReference<String>();
86  		final AtomicReference<String> finalPath = new AtomicReference<String>();
87  		when(sf.getSession()).thenReturn(session);
88  		doAnswer(invocation -> {
89  			temporaryPath.set(invocation.getArgument(0));
90  			finalPath.set(invocation.getArgument(1));
91  			return null;
92  		}).when(session).rename(Mockito.anyString(), Mockito.anyString());
93  		FileTransferringMessageHandler<F> handler = new FileTransferringMessageHandler<F>(sf);
94  		handler.setRemoteDirectoryExpression(new LiteralExpression("foo"));
95  		handler.setTemporaryRemoteDirectoryExpression(new LiteralExpression("bar"));
96  		handler.setBeanFactory(mock(BeanFactory.class));
97  		handler.afterPropertiesSet();
98  		handler.handleMessage(new GenericMessage<String>("hello"));
99  		verify(session, times(1)).write(Mockito.any(InputStream.class), Mockito.anyString());
100 		assertThat(temporaryPath.get().substring(0, 3)).isEqualTo("bar");
101 		assertThat(finalPath.get().substring(0, 3)).isEqualTo("foo");
102 	}

159 	@SuppressWarnings("unchecked")
160 	@Test
161 	public <F> void testServerException() throws Exception {
162 		SessionFactory<F> sf = mock(SessionFactory.class);
163 		CachingSessionFactory<F> csf = new CachingSessionFactory<F>(sf, 2);
164 		FileTransferringMessageHandler<F> handler = new FileTransferringMessageHandler<F>(csf);
165 		handler.setBeanFactory(mock(BeanFactory.class));
166 		Session<F> session1 = newSession();
167 		Session<F> session2 = newSession();
168 		Session<F> session3 = newSession();
169 		when(sf.getSession()).thenReturn(session1, session2, session3);
170 		handler.setRemoteDirectoryExpressionString("'foo'");
171 		handler.afterPropertiesSet();
172 		for (int i = 0; i < 3; i++) {
173 			try {
174 				handler.handleMessage(new GenericMessage<String>("hello"));
175 			}
176 			catch (Exception e) {
177 				assertThat(e.getCause().getCause().getMessage()).isEqualTo("test");
178 			}
179 		}
180 		verify(session1, times(1)).write(Mockito.any(InputStream.class), Mockito.anyString());
181 		verify(session2, times(1)).write(Mockito.any(InputStream.class), Mockito.anyString());
182 		verify(session3, times(1)).write(Mockito.any(InputStream.class), Mockito.anyString());
183 		SimplePool<?> pool = TestUtils.getPropertyValue(csf, "pool", SimplePool.class);
184 		assertThat(pool.getAllocatedCount()).isEqualTo(1);
185 		assertThat(pool.getIdleCount()).isEqualTo(1);
186 		assertThat(TestUtils.getPropertyValue(pool, "allocated", Set.class).iterator().next()).isSameAs(session3);
187 	}

189 	private <F> Session<F> newSession() throws IOException {
190 		@SuppressWarnings("unchecked")
191 		Session<F> session = mock(Session.class);
192 		doThrow(new IOException("test")).when(session).write(any(InputStream.class), anyString());
193 		when(session.isOpen()).thenReturn(false);
194 		return session;
195 	}
