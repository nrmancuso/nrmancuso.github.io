53  	@Test
54  	public void verifyDifferentThread() throws Exception {
55  		SimpleAsyncTaskExecutor taskExecutor = new SimpleAsyncTaskExecutor();
56  		taskExecutor.setThreadNamePrefix("test-");
57  		ExecutorChannel channel = new ExecutorChannel(taskExecutor);
58  		CountDownLatch latch = new CountDownLatch(1);
59  		TestHandler handler = new TestHandler(latch);
60  		channel.subscribe(handler);
61  		channel.send(new GenericMessage<>("test"));
62  		latch.await(1000, TimeUnit.MILLISECONDS);
63  		assertThat(latch.getCount()).isEqualTo(0);
64  		assertThat(handler.thread).isNotNull();
65  		assertThat(Thread.currentThread().equals(handler.thread)).isFalse();
66  		assertThat(handler.thread.getName()).isEqualTo("test-1");
67  	}

69  	@Test
70  	public void roundRobinLoadBalancing() throws Exception {
71  		int numberOfMessages = 11;
72  		ScheduledExecutorService exec = Executors
73  				.newSingleThreadScheduledExecutor(new CustomizableThreadFactory("test-"));
74  		ConcurrentTaskExecutor taskExecutor = new ConcurrentTaskExecutor(exec);
75  		ExecutorChannel channel = new ExecutorChannel(
76  				taskExecutor, new RoundRobinLoadBalancingStrategy());
77  		CountDownLatch latch = new CountDownLatch(numberOfMessages);
78  		TestHandler handler1 = new TestHandler(latch);
79  		TestHandler handler2 = new TestHandler(latch);
80  		TestHandler handler3 = new TestHandler(latch);
81  		channel.subscribe(handler1);
82  		channel.subscribe(handler2);
83  		channel.subscribe(handler3);
84  		for (int i = 0; i < numberOfMessages; i++) {
85  			channel.send(new GenericMessage<>("test-" + i));
86  		}
87  		latch.await(3000, TimeUnit.MILLISECONDS);
88  		assertThat(latch.getCount()).isEqualTo(0);
89  		assertThat(handler1.thread).isNotNull();
90  		assertThat(Thread.currentThread().equals(handler1.thread)).isFalse();
91  		assertThat(handler1.thread.getName().startsWith("test-")).isTrue();
92  		assertThat(handler2.thread).isNotNull();
93  		assertThat(Thread.currentThread().equals(handler2.thread)).isFalse();
94  		assertThat(handler2.thread.getName().startsWith("test-")).isTrue();
95  		assertThat(handler3.thread).isNotNull();
96  		assertThat(Thread.currentThread().equals(handler3.thread)).isFalse();
97  		assertThat(handler3.thread.getName().startsWith("test-")).isTrue();
98  		assertThat(handler1.count.get()).isEqualTo(4);
99  		assertThat(handler2.count.get()).isEqualTo(4);
100 		assertThat(handler3.count.get()).isEqualTo(3);
101 		exec.shutdownNow();
102 	}

173 	@Test
174 	public void interceptorWithModifiedMessage() {
175 		ExecutorChannel channel = new ExecutorChannel(new SyncTaskExecutor());
176 		channel.setBeanFactory(mock(BeanFactory.class));
177 		channel.afterPropertiesSet();
178 
179 		MessageHandler handler = mock(MessageHandler.class);
180 		Message<?> expected = mock(Message.class);
181 		BeforeHandleInterceptor interceptor = new BeforeHandleInterceptor();
182 		interceptor.setMessageToReturn(expected);
183 		channel.addInterceptor(interceptor);
184 		channel.subscribe(handler);
185 		channel.send(new GenericMessage<Object>("foo"));
186 		verify(handler).handleMessage(expected);
187 		assertThat(interceptor.getCounter().get()).isEqualTo(1);
188 		assertThat(interceptor.wasAfterHandledInvoked()).isTrue();
189 	}

191 	@Test
192 	public void interceptorWithException() {
193 		ExecutorChannel channel = new ExecutorChannel(new SyncTaskExecutor());
194 		channel.setBeanFactory(mock(BeanFactory.class));
195 		channel.afterPropertiesSet();
196 
197 		Message<Object> message = new GenericMessage<>("foo");
198 
199 		MessageHandler handler = mock(MessageHandler.class);
200 		IllegalStateException expected = new IllegalStateException("Fake exception");
201 		willThrow(expected).given(handler).handleMessage(message);
202 		BeforeHandleInterceptor interceptor = new BeforeHandleInterceptor();
203 		channel.addInterceptor(interceptor);
204 		channel.subscribe(handler);
205 		try {
206 			channel.send(message);
207 		}
208 		catch (MessageDeliveryException actual) {
209 			assertThat(actual.getCause()).isSameAs(expected);
210 		}
211 		verify(handler).handleMessage(message);
212 		assertThat(interceptor.getCounter().get()).isEqualTo(1);
213 		assertThat(interceptor.wasAfterHandledInvoked()).isTrue();
214 	}

219 		try {

233 	private static class TestHandler implements MessageHandler {
234 
235 		private final CountDownLatch latch;
236 
237 		private final AtomicInteger count = new AtomicInteger();
238 
239 		private volatile Thread thread;
240 
241 		private volatile boolean shouldFail;
242 
243 		TestHandler(CountDownLatch latch) {
244 			this.latch = latch;
245 		}
246 
247 		@Override
248 		public void handleMessage(Message<?> message) {
249 			this.thread = Thread.currentThread();
250 			if (this.shouldFail) {
251 				throw new RuntimeException("intentional test failure");
252 			}
253 			this.count.incrementAndGet();
254 			this.latch.countDown();
255 		}
256 
257 	}

259 	private static class BeforeHandleInterceptor implements ExecutorChannelInterceptor {
260 
261 		private final AtomicInteger counter = new AtomicInteger();
262 
263 		private volatile boolean afterHandledInvoked;
264 
265 		private Message<?> messageToReturn;
266 
267 		BeforeHandleInterceptor() {
268 			super();
269 		}
270 
271 		public void setMessageToReturn(Message<?> messageToReturn) {
272 			this.messageToReturn = messageToReturn;
273 		}
274 
275 		public AtomicInteger getCounter() {
276 			return this.counter;
277 		}
278 
279 		public boolean wasAfterHandledInvoked() {
280 			return this.afterHandledInvoked;
281 		}
282 
283 		@Override
284 		public Message<?> beforeHandle(Message<?> message, MessageChannel channel, MessageHandler handler) {
285 			assertThat(message).isNotNull();
286 			this.counter.incrementAndGet();
287 			return (this.messageToReturn != null ? this.messageToReturn : message);
288 		}
289 
290 		@Override
291 		public void afterMessageHandled(Message<?> message, MessageChannel channel, MessageHandler handler,
292 				Exception ex) {
293 
294 			this.afterHandledInvoked = true;
295 		}
